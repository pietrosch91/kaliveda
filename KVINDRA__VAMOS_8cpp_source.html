21 //
22 // This class descibes the coupling of INDRA ( a KVINDRA object ) and
23 // VAMOS ( a KVVAMOS object ) in a same KVMultiDetArray object.
24 // These two multidetectors are built in the method Build. After the call
25 // of this method the following global pointers correspond to:
26 // gIndra --> INDRA
27 // gVamos --> VAMOS
28 // gMultiDetArray --> INDRA_VAMOS
29 //
30 // The method GetACQParams returns the list of ALL the acquisition parameters of
31 // INDRA and VAMOS. Idem for the list of detectors and the list of ID telescopes
32 // with the methods GetDetectors and GetListOfIDTelescopes respectively.
33 //
34 // To create a KVINDRA_VAMOS instance for a given dataset by using KVDataSet::BuildMultiDetector()
35 // method change the Plugin.KVMultiDetArray plugin in the $KVROOT/KVFiles/.kvrootrc
36 // configuration file:
37 // +Plugin.KVMultiDetArray: INDRA_e494s KVINDRA_VAMOS VAMOS "KVINDRA_VAMOS()"
38 //
39 // TARGET
40 //
41 // The target has to be set to the main multidetector pointed by gMultiDetArray as it is done in KVIVUpdater::SetTarget.
42 // The target is handled with gMultiDetArray then the following commands
43 // gIndra->GetTarget()
44 // gVamos->GetTarget()
45 //
46 // gIndra->GetTargetEnergyLoss(...)
47 // gVamos->GetTargetEnergyLoss(...)
48 //
49 // will return zero.

129 void KVINDRA_VAMOS::Build(Int_t run)
130 {
131  // Overrides KVMultiDetArray::Build in order to set the name of the
132  // couple INDRA+VAMOS. Set up the geometry of INDRA and VAMOS,
133  // associate the acquistion parameters with detectors, etc...
134 
135  SetName("INDRA_VAMOS");
136  SetTitle("INDRA+VAMOS experimental setup");
137  Info("Build", "Building INDRA+VAMOS ...");
138 
139  // Build INDRA multidetector
140  cout << endl << setw(20) << "" << "********************************" << endl;
141  cout << setw(20) << "" << "* INDRA *" << endl;
142  cout << setw(20) << "" << "********************************" << endl << endl;
143 
144  KVINDRA* indra = new KVINDRAe503;
145  ((KVINDRAe503*)indra)->SetDataSet(fDataSet);
146  indra->Build();
147  Add(indra);
148 
149  // Build VAMOS spectrometer
150  cout << endl << setw(20) << "" << "********************************" << endl;
151  cout << setw(20) << "" << "* VAMOS *" << endl;
152  cout << setw(20) << "" << "********************************" << endl << endl;
153 
154  KVVAMOS* vamos = KVVAMOS::MakeVAMOS(fDataSet.Data());
155  Add(vamos);
156 
157  // Add the lists of ACQ parameters, detectors and ID telescopes of
158  // INDRA and VAMOS to INDRA_VAMOS
159 
160  TIter next_i(indra->GetACQParams());
161  TIter next_v(vamos->GetACQParams());
162  TObject* obj;
163  while ((obj = next_i()) || (obj = next_v()))
164  AddACQParam((KVACQParam*)obj);
165 
166  fDetectors.AddAll(indra->GetDetectors());
167  fDetectors.AddAll(vamos->GetDetectors());
168 
169  fIDTelescopes->AddAll(indra->GetListOfIDTelescopes());
170  fIDTelescopes->AddAll(vamos->GetListOfIDTelescopes());
171 
172  // To be sure that gMultiDetArray points on this object.
173  gMultiDetArray = this;
174 
175  if (run > -1) SetParameters(run);
176 }

208 KVNameValueList* KVINDRA_VAMOS::DetectParticle(KVNucleus* part)
209 {
210  // Simulate detection of a charged particle by the array (INDRA+VAMOS).
211  // The actual method called depends on the value of fROOTGeometry:
212  // fROOTGeometry=kTRUE: calls DetectParticle_TGEO, particle propagation performed using
213  // TGeo description of array and algorithms from ROOT TGeo package.
214  // For the moment, this case is not implemented (TO BE DONE)
215  // fROOTGeometry=kFALSE: only calls DetectParticle_KV of gIndra (INDRA), uses simple KaliVeda geometry
216  // to simulate propagation of particle.
217  // No propagation of the particle into VAMOS is simulated.
218 
219  if (IsROOTGeometry()) {
220  return DetectParticle_TGEO(part);
221  } else {
222  gIndra->SetROOTGeometry(kFALSE);
223  return gIndra->DetectParticle(part);
224  }
225 };

232 void KVINDRA_VAMOS::DetectEvent(KVEvent* event, KVReconstructedEvent* rec_event, const Char_t* detection_frame)
233 {
234 
235  // the ROOT geometry can not be use for filtering
236  SetROOTGeometry(kFALSE);
237 
238  // iterate through list of particles
239  // and detect in VAMOS
240  KVNucleus* part, *_part;
241  KVNameValueList* nvl = NULL;
242  Bool_t isVAMOSevent = kFALSE;
243  static KVVAMOSReconEvent tmp_rec_vamos_event;
244  tmp_rec_vamos_event.Clear();
245 
246  while ((part = event->GetNextParticle())) { // loop over particles
247 
248 
249 #ifdef KV_DEBUG
250  cout << "DetectEvent(): looking at particle in VAMOS---->" << endl;
251  part->Print();
252 #endif
253  _part = (KVNucleus*)part->GetFrame(detection_frame, kFALSE);
254 
255  if (_part->GetTheta() > 14) continue;
256 
257  _part->SetE0();
258  Double_t eLostInTarget = 0;
259 
260  if (fTarget && (fFilterType != kFilterType_Geo)) {
261  fTarget->SetOutgoing(kTRUE);
262  //simulate passage through target material
263  Double_t ebef = _part->GetKE();
264  fTarget->DetectParticle(_part);
265  eLostInTarget = ebef - _part->GetKE();
266  if (_part->GetKE() < 1.e-3) {
267  part->GetParameters()->SetValue("UNDETECTED", "STOPPED IN TARGET");
268 
269  part->AddGroup("UNDETECTED");
270  part->AddGroup("STOPPED IN TARGET");
271  }
272  fTarget->SetOutgoing(kFALSE);
273 
274  }
275 
276 
277  if ((_part->GetKE() > 1.e-3) && (nvl = GetVAMOS()->DetectParticle(_part)) && (nvl->GetNpar() > 0)) {
278 
279  _part->SetMomentum(*_part->GetPInitial());
280 
281  isVAMOSevent = kTRUE;
282  KVVAMOSReconNuc* recon_nuc = (KVVAMOSReconNuc*)tmp_rec_vamos_event.AddParticle();
283 
284  // copy parameter list
285  part->GetParameters()->Copy(*(recon_nuc->GetParameters()));
286 
287 
288  recon_nuc->SetZandA(part->GetZ(), part->GetA());
289 
290  recon_nuc->SetMomentum(_part->GetMomentum());
291  recon_nuc->SetQ(_part->GetParameters()->GetIntValue("Q"));
292  recon_nuc->SetBrho(_part->GetParameters()->GetDoubleValue("Brho"));
293  recon_nuc->GetParameters()->SetValue("Delta", _part->GetParameters()->GetDoubleValue("Delta"));
294  recon_nuc->SetThetaVandPhiV(_part->GetParameters()->GetDoubleValue("ThetaV"),
295  _part->GetParameters()->GetDoubleValue("PhiV"));
296 
297  recon_nuc->GetParameters()->SetValue("DETECTED", "OK");
298  recon_nuc->SetTargetEnergyLoss(eLostInTarget);
299  recon_nuc->SetIsIdentified();
300  recon_nuc->SetIsCalibrated();
301  recon_nuc->SetIsOK();
302  recon_nuc->SetZMeasured();
303  recon_nuc->SetQMeasured();
304  recon_nuc->SetAMeasured();
305  //recon_nuc->SetIsBasicQandAidentified();
306 
307  // Clear this nucleus detected in VAMOS in order to inhibit the
308  // its detection in INDRA
309  part->Clear();
310 
311  } else _part->SetMomentum(*_part->GetPInitial());
312 
313  SafeDelete(nvl);
314  } //fin de loop over particles
315 
316  // if a nucleus is detected in VAMOS then continue filtering the event in INDRA
317  if (isVAMOSevent) {
318  // Info("DetectEvent","event %d is a VAMOS event",event->GetNumber());
319  GetINDRA()->DetectEvent(event, rec_event, detection_frame);
320 
321  // Set the reconstructed nucleus detected in VAMOS in the VAMOS event
322  KVVAMOSReconEvent* rec_vamos_event = ((KVIVReconEvent*)rec_event)->GetVAMOSEvent();
323  tmp_rec_vamos_event.Copy(*rec_vamos_event);
324 
325  // rec_event->Print();
326  }
327 }

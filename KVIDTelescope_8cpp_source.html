44 //KVIDTelescope
45 //
46 //Charged particle identification is handled by KVIDTelescope and derived classes.
47 //A KVIDTelescope is an association of one or more detectors which is capable of particle
48 //identification.
49 //Although initially conceived in terms of DE-E two-stage telescopes, the identification
50 //method can be quite different (see KVIDCsI for example).
51 //
52 //Visualising Identification (ID) Grids & Maps
53 //--------------------------------------------
54 //Identification in such 'telescopes' is often associated with a 'grid' or set of
55 //graphical cuts permitting to associate the informations measured in the 'telescope'
56 //with certain types of particles. For any class derived from KVIDTelescope, the
57 //identification grid (KVIDGraph), if one exists, can be obtained and visualised using :
58 //
59 // idtelescope->GetIDGrid()->Draw();
60 //
61 //Note that if no grid has been defined for the telescope, GetIDGrid() returns a null
62 //pointer, so you should only use the previous method if you are absolutely sure that
63 //the ID telescope in question has a grid.
64 //
65 //Visualising identifications based on fitted functionals
66 //--------------------------------------------------------
67 //The identification functionals devised by L. Tassan-Got and defined in KVTGIDFunctions
68 //can be visualised as identification grids for classes having KVTGIDManager as a base
69 //class. In this case a grid can be created and drawn using :
70 //
71 // KVIDGrid* g = idtelescope->GetTGIDGrid("name_of_functional", max_X_coord_of_lines);
72 // g->Draw();
73 //
74 //You should keep a pointer to the grid because it is the user's responsibility to delete it after
75 //use.
76 //
77 //Particle Identification and Calibration/Coherency
78 //-------------------------------------------------
79 //Particles reconstructed from data measured in the detectors of a multidetector array
80 //(KVReconstructedNucleus) are identified using the KVReconstructedNucleus::Identify()
81 //method. This takes the list of ID telescopes through whose detectors the particle passed,
82 //beginning with the telescope in which the particle stopped, and calls the Identify()
83 //method of each ID telescope in turn. This continues until identification is achieved.
84 //Thus the identification method, specific to each type of ID telescope, can be defined
85 //by overriding the Identify() method in child classes.
86 //
87 //Similarly, calibration of identified particles is achieved by KVReconstructedNucleus::
88 //Calibrate() which uses the Calibrate() method of the 'identifying telescope' (i.e. the
89 //KVIDTelescope in which identification was achieved). Again, this may be redefined in
90 //child classes (see KVIDCsI for example), but the default procedure defined here is to
91 //calculate the energy losses of the particle in all the detectors preceding the ID
92 //telescope and compare them to the measured energy losses in these detectors, in order
93 //to check for double-hits. The results of this procedure may then be used by some
94 //coherency algorithm.
95 //
96 //Setting identification parameters
97 //---------------------------------------------
98 //The method SetIdentificationParameters(const KVMultiDetArray*) will be used by the
99 //KVMultiDetArray to which this telescope belongs in order to set the parameters of
100 //ALL telescopes of this type in the array.
101 //By default, this method looks for the file with name given by the environment variable
102 //[dataset name].IdentificationParameterFile.[telescope label]: [filename]
103 //which is assumed to contain identification grids. The file will be read in by gIDGridManager
104 //and the grids added to its list.

226 void KVIDTelescope::Initialize(void)
227 {
228  // Default initialisation for ID telescopes.
229  // kReadyForID is set to kFALSE, unless the current dataset (if defined)
230  // has been declared to have no associated identification/calibration parameters,
231  // in which case kReadyForID is by default set to kTRUE (for filtering simulations).
232  //
233  // In order to enable mass identification for certain telescopes without a dedicated
234  // implementation (e.g. for simulating array response), put the following lines
235  // in your .kvrootrc:
236  //
237  // [dataset].[telescope label].MassID: yes
238  //
239  // If you want to limit mass identification to certain values of Z and/or A,
240  // add the following line:
241  //
242  // [dataset].[telescope label].MassID.Validity: [expression]
243  //
244  // where [expression] is some valid C++ boolean expression involving Z and/or A,
245  // for example
246  //
247  // [dataset].[telescope label].MassID.Validity: (Z>3)&&(A<20)
248  //
249  // To implement identification, make a class derived from KVIDTelescope
250  // and set kReadyForID to kTRUE in Initialize() method of derived class if
251  // necessary conditions for identification are met (has an ID grid etc.).
252  ResetBit(kReadyForID);
253  if (gDataSet && !gDataSet->HasCalibIdentInfos()) SetBit(kReadyForID);
254 
255  if (gDataSet) {
256  SetHasMassID(gDataSet->GetDataSetEnv(Form("%s.MassID", GetLabel()), kFALSE));
257  TString valid;
258  if ((valid = gDataSet->GetDataSetEnv(Form("%s.MassID.Validity", GetLabel()), "")) != "") {
259  valid.ReplaceAll("Z", "_NUC_->GetZ()");
260  valid.ReplaceAll("A", "_NUC_->GetA()");
261  fMassIDValidity = new KVParticleCondition(valid);
262  }
263  }

421 TGraph* KVIDTelescope::MakeIDLine(KVNucleus* nuc, Double_t Emin,
422  Double_t Emax, Double_t Estep)
423 {
424  //For a given nucleus, generate a TGraph representing the line in the deltaE-E
425  //plane of the telescope which can be associated with nuclei of this (A,Z) with total
426  //incident energies between the two limits.
427  //NOTE: if there are other absorbers/detectors placed before this telescope,
428  //the energy losses of the particle in these will be taken into account.
429  //If the step in energy is not given, it is taken equal to 100 equal steps between min and max.
430  //The TGraph should be deleted by the user after use.
431 
432 
433  if (!Estep)
434  Estep = (Emax - Emin) / 100.;
435 
436  Int_t nsteps = 1 + (Int_t)((Emax - Emin) / Estep);
437 
438  if (nsteps < 1)
439  return 0;
440 
441  Double_t* y = new Double_t[nsteps];
442  Double_t* x = new Double_t[nsteps];
443  Int_t step = 0;
444 
445  //get list of all detectors through which particle must pass in order to reach
446  //2nd member of ID Telescope
447  TList* detectors =
448  GetDetector(2)->GetAlignedDetectors(KVGroup::kForwards);
449  //detectors->ls();
450  TIter next_det(detectors);
451  //cout << "nsteps =" << nsteps << endl;
452 
453  for (Double_t E = Emin; E <= Emax; E += Estep) {
454  //Set energy of nucleus
455  nuc->SetEnergy(E);
456  //cout << "Einc=" << E << endl;
457 
458  //Calculate energy loss in each member and stock in arrays x & y
459  //first member
460  KVDetector* det = 0;
461  x[step] = y[step] = -1;
462  while ((det = (KVDetector*) next_det())) {
463  //det->Print();
464  Double_t eloss = det->GetELostByParticle(nuc);
465  if (det == GetDetector(1))
466  y[step] = eloss;
467  else if (det == GetDetector(2))
468  x[step] = eloss;
469  Double_t E1 = nuc->GetEnergy() - eloss;
470  nuc->SetEnergy(E1);
471  //cout << "Eloss=" << eloss << endl;
472  //cout << "Enuc=" << nuc->GetEnergy() << endl;
473  if (E1 < 1.e-3) break;
474  }
475 
476  //cout << "step = " << step << " x = " << x[step] << " y = " << y[step] << endl;
477 
478  //make sure that some energy is lost in each member
479  //otherwise miss a step and reduce number of points in graph
480  if (x[step] > 0 && y[step] > 0) {
481  step++;
482  }
483  else {
484  nsteps--;
485  }
486 
487  //cout << "nsteps =" << nsteps << endl;
488  //reset iterator ready for next loop on detectors
489  next_det.Reset();
490  }
491  TGraph* tmp = 0;
492  if (nsteps > 1)
493  tmp = new TGraph(nsteps, x, y);
494  delete[]x;
495  delete[]y;
496  return tmp;
497 }

697 KVIDTelescope* KVIDTelescope::MakeIDTelescope(const Char_t* uri)
698 {
699  //Static function which will create an instance of the KVIDTelescope-derived class
700  //corresponding to 'name'
701  //These are defined as 'Plugin' objects in the file $KVROOT/KVFiles/.kvrootrc :
702  // # The KVMultiDetArray::GetIDTelescopes(KVDetector*de, KVDetector*e) method uses these plugins to
703  // # create KVIDTelescope instances adapted to the specific array geometry and detector types.
704  // # For each pair of detectors we look for a plugin with one of the following names:
705  // # [name_of_dataset].de_detector_type[de detector thickness]-e_detector_type[de detector thickness]
706  // # Each characteristic in [] brackets may or may not be present in the name; first we test for names
707  // # with these characteristics, then all combinations where one or other of the characteristics is not present.
708  // # In addition, we first test all combinations which begin with [name_of_dataset].
709  // # The first plugin found in this order will be used.
710  // # In addition, if for one of the two detectors there is a plugin called
711  // # [name_of_dataset].de_detector_type[de detector thickness]
712  // # [name_of_dataset].e_detector_type[e detector thickness]
713  // # then we add also an instance of this 1-detector identification telescope.
714 
715  //check and load plugin library
716  TPluginHandler* ph;
717  if (!(ph = LoadPlugin("KVIDTelescope", uri)))
718  return 0;
719 
720  //execute constructor/macro for identification telescope
721  KVIDTelescope* mda = (KVIDTelescope*) ph->ExecPlugin(0);
722  if (mda) {
723  //set label of telescope with URI used to find plugin (minus dataset name)
724  mda->SetLabelFromURI(uri);
725  }
726 
727  return mda;
728 }

864 void KVIDTelescope::CalculateParticleEnergy(KVReconstructedNucleus* nuc)
865 {
866  // The energy of each particle is calculated as follows:
867  //
868  // E = dE_1 + dE_2 + ... + dE_N
869  //
870  // dE_1, dE_2, ... = energy losses measured in each detector through which
871  // the particle has passed (or stopped, in the case of dE_N).
872  // These energy losses are corrected for (Z,A)-dependent effects
873  // such as pulse-heigth defect in silicon detectors, losses in
874  // windows of gas detectors, etc.
875  //
876  // Whenever possible, the energy loss for fired detectors which are uncalibrated
877  // or not functioning is calculated. In this case the status returned by GetCalibStatus()
878  // will be KVIDTelescope::kCalibStatus_Calculated.
879  // If none of the detectors is calibrated, the particle's energy cannot be calculated &
880  // the status will be KVIDTelescope::kCalibStatus_NoCalibrations.
881  // Otherwise, the status code will be KVIDTelescope::kCalibStatus_OK.
882 
883  //status code
884  fCalibStatus = kCalibStatus_NoCalibrations;
885 
886  UInt_t z = nuc->GetZ();
887  //uncharged particles
888  if (z == 0) return;
889 
890  KVDetector* d1 = GetDetector(1);
891  KVDetector* d2 = (fDetectors->GetSize() > 1 ? GetDetector(2) : 0);
892  Bool_t d1_cal = d1->IsCalibrated();
893  Bool_t d2_cal = (d2 ? d2->IsCalibrated() : kFALSE);
894 
895  //no calibrations
896  if (!d1_cal && !d2)
897  return;

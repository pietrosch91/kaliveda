54 KVDBParameterSet::KVDBParameterSet(const Char_t* name,
55  const Char_t* title,
56  UShort_t pnum): KVDBRecord(name, title)
57 {
58  //Initialise a KVDBRecord for a set of "pnum" parameters.
59  //The names of the parameters are "par_1", "par_2", etc.
60  //The KVDBKey "Runs" is initialised to cross-reference the runs for which
61  //this set of parameters is valid.
62  fParamNumber = pnum;
63  for (UInt_t i = 0; i < pnum; i++) {
64  fParameters.SetValue(Form("par_%u", i + 1), 0.0);
65  }
66  AddKey("Runs", "List of Runs");
67  //Unique not used because it makes the setting up of the list horribly slow
68  // if(key) key->SetUniqueStatus(kTRUE);

86 void KVDBParameterSet::SetParameters(Double_t val, ...)
87 {
88  // Be cautious when using this method, if the number of arguments is lower than
89  // the expected number of parameters a segmantation fault will occur !!!!!
90  // exceeding argument won't be considered at all
91 
92  va_list ap;
93  va_start(ap, val);
94 
95  int arg_n = 0;
96  SetParameter(arg_n++, val);
97  while (arg_n < fParamNumber) {
98  SetParameter(arg_n, va_arg(ap, Double_t));
99  arg_n++;
100  }
101 
102  va_end(ap);
103 }

106 void KVINDRADB_e503::ReadVamosScalers()
107 {
108  ifstream fin;
109 
110  if (!OpenCalibFile("VamosScaler", fin)) {
111  Warning("ReadVamosScalers", "VAMOS scalers file not found : %s",
112  GetCalibFileName("VamosScaler"));
113  return;
114  }
115  Info("ReadVamosScalers", "Reading in Vamos Scaler file : %s",
116  GetCalibFileName("VamosScaler"));
117 
118  Float_t NormScaler = 0;
119  Float_t DT = 0;
120  Float_t NormVamos = 0;
121  TString sline;
122  Int_t irun = 0;
123  Int_t ScalerValue;
124  Char_t Crunnumber[40];
125  Char_t str[30];
126  while (fin.good()) { //reading the file
127  sline.ReadLine(fin);
128  if (!fin.eof()) { //fin du fichier
129  if (sline.Sizeof() > 1 && !sline.BeginsWith("#")) {
130  if (sline.BeginsWith("ScalersOfRunNumber")) {
131  sscanf(sline.Data(), "%30s %d ", Crunnumber, &irun);
132  KVINDRADBRun* idb = GetRun(irun);
133  if (idb) {
134  for (Int_t i = 0; i < 5; i++) { //retrouve dans les 4 prochaines lignes NormVamos,normSca etDT
135  sline.ReadLine(fin);
136 
137  if (sline.BeginsWith("NormVamos")) {
138  sscanf(sline.Data(), "%20s %f ", str, &NormVamos);
139  idb->Set(str, NormVamos);
140  }
141  if (sline.BeginsWith("NormScalers")) {
142  sscanf(sline.Data(), "%20s %f ", str, &NormScaler);
143  idb->Set(str, NormScaler);
144  }
145  if (sline.BeginsWith("DT")) {
146  sscanf(sline.Data(), "%20s %f ", str, &DT);
147  idb->Set(str, DT);
148  }
149 
150  }
151  for (Int_t i = 0; i < 40; i++) {
152  sline.ReadLine(fin);
153  sscanf(sline.Data(), "%s %d ", str, &ScalerValue);
154  idb->SetScaler(str, ScalerValue);
155  }
156 
157  }
158  }
159  }
160  }
161  }
162  fin.close();
163 }

217 void KVINDRADB_e503::ReadPedestalCorrection()
218 {
219  // Reading the pedestal correction (DeltaPed). DeltaPed depends on
220  // the run number but it is the same for all detectors connected to the
221  // same QDC. The corrected pedestal is given by the reference
222  // pedestal + DeltaPed.
223  ifstream fin;
224 
225  if (!OpenCalibFile("PedestalCorrections", fin)) {
226  Warning("ReadPedestalCorrection", "Pedestal correction file not found : %s",
227  GetCalibFileName("PedestalCorrections"));
228  return;
229  }
230  Info("ReadPedestalCorrection", "Reading in pedestal correction file : %s",
231  GetCalibFileName("PedestalCorrections"));
232 
233 
234  TString sline;
235  while (fin.good()) { //reading the file
236  sline.ReadLine(fin);
237  // Skip comment line
238  if (sline.BeginsWith("#")) continue;
239 
240  if (sline.BeginsWith("+DeltaPed")) ReadDeltaPedestal(fin);
241  }
242  fin.close();
243 }

256 void KVINDRADB_e503::ReadDeltaPedestal(ifstream& ifile)
257 {
258  // Reading the pedestal correction (DeltaPed). Method called by
259  // KVINDRADB_e503::ReadPedestalCorrection(). DeltaPed is equal to
260  // DeltaNoise for high gain coder data. For anyother coder data,
261  // DetaPed is given by DeltaGene if DeltaGene is less than 20 channels,
262  // otherwise it is given by DeltaNoise. DeltaGene (DeltaNoise) is the
263  // difference between pedestal positions from the generator (noise)
264  // of the current run and of the reference run.
265 
266  KVString sline, signal, parname;
267  KVDBParameterSet* parset = NULL;
268  Int_t numQDC = -1;
269  KVNumberList runs;
270  Double_t Dped = -1., Dgene = -1., Dbruit = -1.;
271 
272  while (ifile.good()) { //reading the file
273  sline.ReadLine(ifile);
274  // Skip comment line
275  if (sline.BeginsWith("#")) continue;
276  // End character
277  if (sline.BeginsWith("!")) return;
278 
279  // QDC number
280  if (sline.BeginsWith("QDC=")) {
281  sline.Remove(0, 4);
282  numQDC = sline.Atoi();
283  continue;
284  }
285  // DeltaPed for each signal
286  Ssiz_t idx = sline.Index(":");
287  runs.SetList(TString(sline).Remove(idx));
288  sline.Remove(0, idx + 1);
289  sline.Begin(" ");
290  while (!sline.End()) {
291  signal = sline.Next(kTRUE);
292  Dgene = sline.Next(kTRUE).Atof();
293  Dbruit = sline.Next(kTRUE).Atof();
294 
295  // Marie-France's selection
296  Dped = ((Dgene > 20) || !strcmp(signal.Data(), "GG") ? Dbruit : Dgene);
297  if (Dped > 20) Warning("ReadDeltaPedestal", "DeltaPed>20 for runs %s, signal %s", runs.GetList(), signal.Data());
298 
299  parname.Form("QDC%.2d", numQDC);
300  parset = new KVDBParameterSet(parname.Data(), signal.Data(), 1);
301  parset->SetParameter(Dped);
302  fDeltaPed->AddRecord(parset);
303  LinkRecordToRunRange(parset, runs);
304  }
305  }
306 }

315 void KVINDRADB_e503::ReadPedestalList()
316 {
317 //Read the names of pedestal files to use for each run range, found
318 //in file with name defined by the environment variable:
319 // [dataset name].INDRADB.Pedestals: ...
320 
321  ifstream fin;
322  if (!OpenCalibFile("Pedestals", fin)) {
323  Error("ReadPedestalList()", "Could not open file %s",
324  GetCalibFileName("Pedestals"));
325  return;
326  }
327  Info("ReadPedestalList()", "Reading pedestal file list...");
328 
329  KVString line;
330  Char_t filename_chio[80], filename_csi[80], filename_etalons[80];
331  UInt_t runlist[1][2];
332 
333  while (fin.good()) { //lecture du fichier
334 
335 // lecture des informations
336  line.ReadLine(fin);
337 
338 //recherche une balise '+'
339  if (line.BeginsWith('+')) { //balise trouvee
340 
341  line.Remove(0, 1);
342 
343  if (sscanf
344  (line.Data(), "Run Range : %u %u", &runlist[0][0],
345  &runlist[0][1]) != 2) {
346  Warning("ReadPedestalList()", "Format problem in line \n%s",
347  line.Data());
348  }
349 
350  line.ReadLine(fin);
351  sscanf(line.Data(), "%s", filename_chio);
352 
353  line.ReadLine(fin);
354  sscanf(line.Data(), "%s", filename_csi);
355 
356  line.ReadLine(fin);
357  sscanf(line.Data(), "%s", filename_etalons);
358 
359  TList RRList;
360  KVDBRecord* dummy = 0;
361  dummy = new KVDBRecord(filename_chio, "ChIo/Si pedestals");
362  dummy->AddKey("Runs", "Runs for which to use this pedestal file");
363  fPedestals->AddRecord(dummy);
364  RRList.Add(dummy);
365  dummy = new KVDBRecord(filename_csi, "CsI pedestals");
366  dummy->AddKey("Runs", "Runs for which to use this pedestal file");
367  fPedestals->AddRecord(dummy);
368  RRList.Add(dummy);
369  dummy = new KVDBRecord(filename_etalons, "Si75/SiLi pedestals");
370  dummy->AddKey("Runs", "Runs for which to use this pedestal file");
371  fPedestals->AddRecord(dummy);
372  RRList.Add(dummy);
373  LinkListToRunRanges(&RRList, 1, runlist);
374  } // balise trouvee
375  } // lecture du fichier
376  fin.close();
377  cout << "Pedestals Read" << endl;
378 }

158 void KVRotatedTensor::FillN(KVEvent* e)
159 {
160  // Calculation of contribution to N-body global variable of particles in event e.
161 
162  for (int i = 0; i < fNROT; ++i) {
163 
164  // random phi rotation
165  fRot.RotateZ(gRandom->Uniform(TMath::TwoPi()));
166 
167  // turn event
168  e->SetFrame("KVRotatedTensor_rotatedFrame", "CM", fRot);
169 
170  e->ResetGetNextParticle();
171  KVNucleus* nuc;
172  while ((nuc = e->GetNextParticle("ok"))) {
173  FillWithCondition(nuc);
174  }
175 
176  }
177 }

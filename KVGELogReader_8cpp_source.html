56 void KVGELogReader::ReadLine(TString& line, Bool_t& ok)
57 {
58  //analyse contents of line read from log file
59 
60  KVLogReader::ReadLine(line, ok);
61  //if we find a seg fault after having copied a file to SRB (raw->recon, recon->ident, ident->root tasks)
62  //then we ignore it: the job is good!
63  if (!ok && fileCopiedtoSRB) {
64  ok = kTRUE;
65  fOK = kTRUE;
66  }
67  if (!ok) return;
68 
69  if (line.Contains("LOCAL:") && line.Contains("->SRB:")) {
70  fileCopiedtoSRB = kTRUE;
71  } else if (line.Contains("Cputime limit exceeded")) {
72  ok = kFALSE;
73  fStatus = line;
74  fOK = kFALSE;
75  return;
76  } else if (line.Contains("Filesize limit exceeded")) {
77  ok = kFALSE;
78  fStatus = line;
79  fOK = kFALSE;
80  return;
81  } else if (line.Contains("File size limit exceeded")) {
82  ok = kFALSE;
83  fStatus = line;
84  fOK = kFALSE;
85  return;
86  } else if (line.Contains("XCPU")) {
87  ok = kFALSE;
88  fStatus = "KILLED";
89  fOK = kFALSE;
90  return;
91  } else if (line.Contains("cpu time:"))
92  ReadCPULimit(line);
93  else if (line.Contains("CpuUser ="))
94  ReadKVCPU(line);
95  else if (line.Contains(" vmem:"))
96  ReadMemUsed(line);
97  else if (line.Contains("Exit status:"))
98  ReadStatus(line);
99  else if (line.Contains("Jobname:"))
100  ReadJobname(line);
101  else if (line.Contains("DISK_REQ") || line.Contains("MEM_REQ"))
102  ReadStorageReq(line);
103 }

111 void KVGELogReader::ReadKVCPU(TString& line)
112 {
113  // update infos on CPU time, memoire & disk from lines such as
114  // CpuUser = 137.8598 s. VirtMem = 250.32647 MB DiskUsed = 356M
115 
116  TObjArray* toks = line.Tokenize("= ");
117  Int_t ntoks = toks->GetEntries();
118  for (int i = 0; i < ntoks; i++) {
119  KVString token = ((TObjString*)toks->At(i))->String();
120  if (token == "CpuUser") {
121  fCPUused = ((TObjString*)toks->At(i + 1))->String().Atof();
122  } else if (token == "VirtMem") {
123  fMemKB = ((TObjString*)toks->At(i + 1))->String().Atof();
124  fMemKB *= GetByteMultiplier(((TObjString*)toks->At(i + 2))->String());
125  } else if (token == "DiskUsed") {
126  token = ((TObjString*)toks->At(i + 1))->String();
127  fScratchKB = ReadStorage(token);
128  }
129  }
130 }

138 void KVGELogReader::ReadCPULimit(TString& line)
139 {
140  //read line of type "* cpu time: 693 / 15000 *"
141  //which contains total normalized used CPU time and the time limit from the job request
142 
143  TObjArray* toks = line.Tokenize("*:/ ");
144  //get used CPU time (in seconds)
145  KV__TOBJSTRING_TO_INT(toks, 2, sec)
146  if (sec) fCPUused = sec; //if there is a GE problem, could be "0 / 3000"
147  //in this case we use values printed at end of job by KV
148  //get requested CPU time (in seconds)
149  KV__TOBJSTRING_TO_INT(toks, 3, sec_)
150  fCPUreq = sec_;
151  delete toks;
152 }

173 void KVGELogReader::ReadMemUsed(TString& line)
174 {
175  //read line of type "* vmem: 406.879M *"
176  //corresponding to memory used by job
177 
178  TObjArray* toks = line.Tokenize("*: ");
179  if (toks->GetEntries() < 2) {
180  //in case Grid Engine goes west & doesn't write vmem in log
181  delete toks;
182  return;
183  }
184  //value read is converted to KB, depending on units
185  TString vmem = ((TObjString*)toks->At(1))->String();
186  TString units = vmem[vmem.Length() - 1];
187  vmem.Remove(vmem.Length() - 1);
188  fMemKB = vmem.Atof() * GetByteMultiplier(units);
189  delete toks;
190 }

198 Int_t KVGELogReader::GetByteMultiplier(TString& unit)
199 {
200  //unit = "K", "M" or "G"
201  //value returned is 1, 2**10 or 2**20, respectively
202  static Int_t KB = 1;
203  static Int_t MB = 2 << 9;
204  static Int_t GB = 2 << 19;
205  return (unit.BeginsWith("K") ? KB : (unit.BeginsWith("M") ? MB : GB));
206 }

237 Double_t KVGELogReader::ReadStorage(KVString& stor)
238 {
239  //'stor' is a string such as "200M", "3G" etc.
240  //value returned is corresponding storage space in KB
241  //if no units symbol found, we assume MB (as output of 'du -h')
242 
243  static const Char_t* units[] = { "K", "M", "G" };
244  Int_t i = 0, index = -1;
245  while ((index = stor.Index(units[i])) < 0 && i < 2)
246  i++;
247  TString u;
248  if (index >= 0) {
249  stor.Remove(index);
250  u = units[i];
251  } else {
252  u = "M";
253  }
254  return (stor.Atof() * GetByteMultiplier(u));
255 }

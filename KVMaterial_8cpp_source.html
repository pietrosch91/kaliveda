184 void KVMaterial::SetMaterial(const Char_t* mat_type)
185 {
186  // Intialise material of a given type.
187  // The material must exist in the currently used range tables (fIonRangeTable).
188  // For materials which are elements of the periodic table you can specify
189  // the isotope such as "64Ni", "13C", "natSn", etc. etc.
190 
191  init();
192  //are we dealing with an isotope ?
193  Char_t type[10];
194  Int_t iso_mass = 0;
195  if (sscanf(mat_type, "nat%s", type) != 1) {
196  if (sscanf(mat_type, "%d%s", &iso_mass, type) != 2) {
197  strcpy(type, mat_type);
198  }
199  }
200  if (iso_mass) SetMass(iso_mass);
201  SetType(type);
202  if (!fIonRangeTable->IsMaterialKnown(type))
203  Warning("SetMaterial",
204  "Called for material %s which is unknown in current range table %s. Energy loss & range calculations impossible.",
205  type, fIonRangeTable->GetName());
206  else {
207  SetType(fIonRangeTable->GetMaterialName(type));
208  SetName(type);
209  }
210 }

526 Double_t KVMaterial::GetEffectiveThickness(TVector3& norm,
527  TVector3& direction)
528 {
529  // Calculate effective linear thickness of absorber (in cm) as 'seen' in 'direction', taking into
530  // account the arbitrary orientation of the 'norm' normal to the material's surface
531 
532  TVector3 n = norm.Unit();
533  TVector3 d = direction.Unit();
534  //absolute value of scalar product, in case direction is opposite to normal
535  Double_t prod = TMath::Abs(n * d);
536  return GetThickness() / TMath::Max(prod, 1.e-03);
537 }

590 Double_t KVMaterial::GetELostByParticle(KVNucleus* kvn, TVector3* norm)
591 {
592  // Method to simulate passage of an ion through a given thickness of material.
593  // The energy loss of the particle (in MeV) in the material is returned by the method.
594  //
595  // If the optional argument 'norm' is given, it is supposed to be a vector
596  // normal to the material, oriented from the origin towards the material.
597  // In this case the effective thickness of the material 'seen' by the particle
598  // depending on its direction of motion is used for the calculation.
599 
600  Double_t thickness;
601  if (norm) {
602  TVector3 p = kvn->GetMomentum();
603  thickness = GetEffectiveThickness((*norm), p);
604  } else
605  thickness = GetThickness();
606  Double_t E_loss =
607  fIonRangeTable->GetLinearDeltaEOfIon(GetType(), kvn->GetZ(), kvn->GetA(), kvn->GetKE(),
608  thickness, fAmasr, fTemp, fPressure);
609  return E_loss;
610 }

741 Double_t KVMaterial::GetEResFromDeltaE(Int_t Z, Int_t A, Double_t dE, enum SolType type)
742 {
743  // Calculate residual kinetic energy Eres (MeV) after the absorber from
744  // energy loss in absorber for nucleus (Z,A). If dE is not given, the energy
745  // loss in this absorber is used.
746  // For a KVDetector, dE = energy loss in the ACTIVE layer, not the total
747  // energy lost in crossing the detector (i.e. the incident energy of the
748  // particle is NOT given by dE + Eres: due to losses in the inactive layers
749  // of the detector, Einc > dE + Eres).
750  //
751  //By default the solution corresponding to the highest incident energy is returned
752  //This is the solution found for Einc greater than the maximum of the dE(Einc) curve.
753  //If you want the low energy solution (i.e. below Bragg peak for gas detector)
754  //set SolType = KVMaterial::kEmin.
755  //
756  //If the given energy loss in the absorber is greater than the maximum theoretical dE
757  //(dE > GetBraggDE) then we return the residual energy corresponding to the
758  //maximum dE.
759 
760  Double_t EINC = GetIncidentEnergy(Z, A, dE, type);
761  return GetERes(Z, A, EINC);
762 }

779 Double_t KVMaterial::GetIncidentEnergy(Int_t Z, Int_t A, Double_t delta_e, enum SolType type)
780 {
781  //Calculate incident energy of nucleus (Z,A) corresponding to the energy loss
782  //in this absorber. If delta_e is not given, the energy loss in this absorber is used.
783  //
784  //By default the solution corresponding to the highest incident energy is returned
785  //This is the solution found for Einc greater than the maximum of the dE(Einc) curve.
786  //If you want the low energy solution set SolType = KVIonRangeTable::kEmin.
787  //
788  //If the given energy loss in the absorber is greater than the maximum theoretical dE
789  //(dE > GetMaxDeltaE(Z,A)) then we return the incident energy corresponding to the maximum,
790  //GetEIncOfMaxDeltaE(Z,A)
791 
792  if (Z < 1) return 0.;
793 
794  Double_t DE = (delta_e > 0 ? delta_e : GetEnergyLoss());
795 
796  return fIonRangeTable->GetLinearEIncFromDeltaEOfIon(GetType(), Z, A, DE, GetThickness(),
797  (enum KVIonRangeTable::SolType)type, fAmasr, fTemp, fPressure);

835 void KVMaterial::DetectParticle(KVNucleus* kvp, TVector3* norm)
836 {
837  //The energy loss of a charged particle traversing the absorber is calculated,
838  //and the particle is slowed down.
839  //
840  //If the optional argument 'norm' is given, it is supposed to be a unit vector
841  //normal to the material, oriented from the origin towards the material.
842  //In this case the effective thickness of the material 'seen' by the particle
843  //depending on its direction of motion is used for the calculation.
844 
845  //set flag to say that particle has been slowed down
846  kvp->SetIsDetected();
847  //If this is the first absorber that the particle crosses, we set a "reminder" of its
848  //initial energy
849  if (!kvp->GetPInitial())
850  kvp->SetE0();
851 
852 #ifdef DBG_TRGT
853  cout << "detectparticle in material " << GetType() << " of thickness "
854  << GetThickness() << endl;
855 #endif
856  Double_t el = GetELostByParticle(kvp, norm);
857  // set particle residual energy
858  Double_t Eres = kvp->GetKE() - el;
859  kvp->SetKE(Eres);
860  // add to total of energy losses in absorber
861  fELoss += el;
862 }

962 TGeoMedium* KVMaterial::GetGeoMedium(const Char_t* med_name)
963 {
964  // By default, return pointer to TGeoMedium corresponding to this KVMaterial.
965  // If argument "med_name" is given and corresponds to the name of an already existing
966  // medium, we return a pointer to this medium, or 0x0 if it does not exist.
967  // med_name = "Vacuum" is a special case: if the "Vacuum" does not exist, we create it.
968  //
969  // Instance of geometry manager class TGeoManager must be created before calling this
970  // method, otherwise 0x0 will be returned.
971  // If the required TGeoMedium is not already available in the TGeoManager, we create
972  // a new TGeoMedium corresponding to the properties of this KVMaterial.
973  // The name of the TGeoMedium (and associated TGeoMaterial) is the name of the KVMaterial.
974 
975  if (!gGeoManager) return NULL;
976 
977  if (strcmp(med_name, "")) {
978  TGeoMedium* gmed = gGeoManager->GetMedium(med_name);
979  if (gmed) return gmed;
980  else if (!strcmp(med_name, "Vacuum")) {
981  // create material
982  TGeoMaterial* gmat = new TGeoMaterial("Vacuum", 0, 0, 0);
983  gmat->SetTitle("Vacuum");
984  gmed = new TGeoMedium("Vacuum", 0, gmat);
985  gmed->SetTitle("Vacuum");
986  return gmed;
987  }
988  return NULL;
989  }
990 
991  // if object is a KVDetector, we return medium corresponding to the active layer
992  if (GetActiveLayer()) return GetActiveLayer()->GetGeoMedium();
993 
994  // for gaseous materials, the TGeoMedium/Material name is of the form
995  // gasname_pressure
996  // e.g. C3F8_37.5 for C3F8 gas at 37.5 torr
997  // each gas with different pressure has to have a separate TGeoMaterial/Medium
998  TString medName;
999  if (IsGas()) medName.Form("%s_%f", GetName(), GetPressure());
1000  else medName = GetName();
1001 
1002  TGeoMedium* gmed = gGeoManager->GetMedium(medName);
1003 
1004  if (gmed) return gmed;
1005 
1006  TGeoMaterial* gmat = gGeoManager->GetMaterial(medName);
1007 
1008  if (!gmat) {
1009  // create material
1010  gmat = GetRangeTable()->GetTGeoMaterial(GetName());
1011  gmat->SetPressure(GetPressure());
1012  gmat->SetTemperature(GetTemperature());
1013  gmat->SetTransparency(0);
1014  gmat->SetName(medName);
1015  gmat->SetTitle(GetName());
1016  }
1017 
1018  // create medium
1019  static Int_t numed = 1; // static counter variable used to number media
1020  gmed = new TGeoMedium(medName, numed, gmat);
1021  numed += 1;
1022 
1023  return gmed;
1024 }

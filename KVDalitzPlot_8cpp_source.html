136 Int_t KVDalitzPlot::FillAsDalitz(Double_t a1, Double_t a2, Double_t a3)
137 {
138  //Place the 3 observable in an equilateral triangle
139  //after normalized the sum to 1
140  //Info : the permutation is not done
141 
142  if (!fOrdered) return FillMe(a1, a2, a3);
143 
144  Int_t result = 0;
145  result += FillMe(a1, a2, a3);
146  result += FillMe(a2, a3, a1);
147  result += FillMe(a3, a1, a2);
148  result += FillMe(a1, a3, a2);
149  result += FillMe(a2, a1, a3);
150  result += FillMe(a3, a2, a1);
151  return result;
152 }

158 Int_t KVDalitzPlot::FillMe(Double_t a1, Double_t a2, Double_t a3)
159 {
160  Double_t sum = a1 + a2 + a3;
161  if (sum > 0) {
162  Double_t a1n = a1 / sum;
163  Double_t a2n = a2 / sum;
164 
165  Double_t xI = 1. / TMath::Sqrt(3) * (a1n + 2 * a2n);
166  Double_t yI = a1n;
167 
168  return TH2F::Fill(xI, yI);
169  }
170  return -1;
171 }

289 TH1* KVDalitzPlot::GetDistanceFromCenter()
290 {
291  //Linearization of the Dalitz representation
292  //histogram is filled according to the distance from each cell
293  //to the center
294  //
295  TH1F* h1 = new TH1F(Form("distance_%s", GetName()), Form("distance_%s", GetName()), 150, 0, 1.5);
296  for (Int_t nx = 1; nx <= GetNbinsX(); nx += 1) {
297  Double_t xx = GetXaxis()->GetBinCenter(nx);
298  for (Int_t ny = 1; ny <= GetNbinsY(); ny += 1) {
299  Double_t yy = GetYaxis()->GetBinCenter(ny);
300  Double_t dist = TMath::Sqrt(TMath::Power(xx - 0.5, 2.) + TMath::Power(yy - 1. / 3., 2.));
301  h1->Fill(dist, GetBinContent(nx, ny));
302  }
303  }
304 
305  return h1;
306 
307 }

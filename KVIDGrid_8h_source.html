40  virtual KVIDLine* FindNearestIDLineFast(Double_t x, Double_t y, const Char_t* position,
41  Int_t& idx, Int_t& idx_min, Int_t& idx_max, Double_t& dist, Double_t& dist_min, Double_t& dist_max) const
42  {
43  //Fast algorithm for finding the ID line the closest to a given point (x,y) by dichotomy.
44  //
45  // WARNING: will not always give right result if the distance from the point to each line
46  // is not a smooth function of the line index with a single minimum. In this case
47  // you should use FindNearestIDLine.
48  // IF the grid's lines all start and end at the same abscissa, this will work fine.
49  // IF NOT, (i.e. lines start and end at different 'x' values), it will sometimes give the wrong answer.
50  // -----------------------------------------------------------------
51  //The returned pointer is the closest line, idx is its index in the list of identifiers,
52  //whereas idx_min and idx_max are the indices of the two closest lines between which (x,y) lies.
53  //dist, dist_min, and dist_max are the distances between the point and these lines.
54  //
55  //The order of the indices follows the order of the sorting of the ID line list which is in turn
56  //determined by the Compare() function of the class of which the ID lines are members.
57  //For the algorithm to work, the ID lines must be sorted according to the Compare() function defined for the
58  //KVIDLine-based class of which they are members. The "position" string then corresponds to the
59  //way lines are sorted in the map (vertically or horizontally, and in which direction). For this reason
60  //it is essential to call the Initialise() method of the grid before using it (this will sort the lines).
61  //
62  //For example:
63  // Suppose we have lines of Z identification (KVIDZALine) in a map with x=E and y=dE.
64  //The KVIDZALine::Compare() function sorts lines in order of increasing Z, i.e. in terms of increasing
65  //dE, or from bottom to top when looking at the (E,dE) map. Thus the "position" string to use is
66  //"above" : if the point (E,dE) is "above" the line currently tested, the algorithm will choose a line
67  //which is "higher up" i.e. has a larger Z, or in fact, has a larger index in the list of ID lines. The indices
68  //idx_min and idx_max would correspond to the line immediately below and above the point, respectively.
69  // This is for a case of more-or-less horizontal lines sorted from bottom to top.
70  //In a map of more-or-less vertical ID lines, with the sorting going from left to right,
71  //we would use position="right". (see KVIDLine::WhereAmI for an explanation of "above", "below", etc.).
72 
73  Int_t nlines = GetNumberOfIdentifiers();
74  idx_min = 0; //minimum index
75  idx_max = nlines - 1; // maximum index
76  idx = idx_max / 2; //current index i.e. we begin in the middle
77  dist = dist_min = dist_max = -1.;
78 
79  while (idx_max > idx_min + 1) {
80 
81  KVIDLine* line = (KVIDLine*)GetIdentifierAt(idx);
82  Bool_t point_above_line = line->WhereAmI(x, y, position);
83 
84  if (point_above_line) {
85  //increase index
86  idx_min = idx;
87  idx += (Int_t)((idx_max - idx) / 2 + 0.5);
88  } else {
89  //decrease index
90  idx_max = idx;
91  idx -= (Int_t)((idx - idx_min) / 2 + 0.5);
92  }
93  }
94  //calculate distance of point to the two lines above and below
95  //and return pointer of the closest one
96  KVIDLine* upper = (KVIDLine*)GetIdentifierAt(idx_max);
97  KVIDLine* lower = (KVIDLine*)GetIdentifierAt(idx_min);
98  Int_t dummy = 0;
99  dist_max = TMath::Abs(upper->DistanceToLine(x, y, dummy));
100  dist_min = TMath::Abs(lower->DistanceToLine(x, y, dummy));
101  if (dist_max < dist_min) {
102  dist = dist_max;
103  idx = idx_max;
104  return upper;
105  }
106  dist = dist_min;
107  idx = idx_min;
108  return lower;
109  };

111  virtual KVIDLine* FindNearestEmbracingIDLine(Double_t x, Double_t y, const Char_t* position, const Char_t* axis,
112  Int_t& idx, Int_t& idx_min, Int_t& idx_max, Double_t& dist, Double_t& dist_min, Double_t& dist_max) const
113  {
114  // This is the same as FindNearestIDLineFast except that only lines for which
115  // KVIDLine::IsBetweenEndPoints(x,y,axis) returns kTRUE are considered.
116  // As we only consider lines between whose endpoints our point lies, this method
117  // always gives the correct answer.
118 
119  static TList emL;
120 
121  Int_t nlines = GetIDLinesEmbracingPoint(axis, x, y, emL);
122  if (!nlines) return 0; // no lines
123  idx_min = 0; //minimum index
124  idx_max = nlines - 1; // maximum index
125  idx = idx_max / 2; //current index i.e. we begin in the middle
126  dist = dist_min = dist_max = -1.;
127 
128  while (idx_max > idx_min + 1) {
129 
130  KVIDLine* line = (KVIDLine*)emL.At(idx);
131  Bool_t point_above_line = line->WhereAmI(x, y, position);
132 
133  if (point_above_line) {
134  //increase index
135  idx_min = idx;
136  idx += (Int_t)((idx_max - idx) / 2 + 0.5);
137  } else {
138  //decrease index
139  idx_max = idx;
140  idx -= (Int_t)((idx - idx_min) / 2 + 0.5);
141  }
142  }
143  //calculate distance of point to the two lines above and below
144  KVIDLine* upper = (KVIDLine*)emL.At(idx_max);
145  KVIDLine* lower = (KVIDLine*)emL.At(idx_min);
146  Int_t dummy = 0;
147  //if idx_max = nlines-1, the point may be above the last line
148  //in which case we put idx_max = -1 (no line above point)
149  if (idx_max == nlines - 1) {
150  if (upper->WhereAmI(x, y, position)) {
151  // above last line
152  idx = idx_min = fIdentifiers->IndexOf(upper); // index of last line
153  idx_max = -1;
154  dist = dist_min = TMath::Abs(upper->DistanceToLine(x, y, dummy));
155  return upper;
156  }
157  }
158  //if idx_min = 0, the point may be below the first line
159  //in which case we put idx_min = -1 (no line below point)
160  if (idx_min == 0) {
161  if (!lower->WhereAmI(x, y, position)) {
162  // below first line
163  idx_min = -1;
164  idx = idx_max = fIdentifiers->IndexOf(lower); // index of first line
165  dist = dist_max = TMath::Abs(lower->DistanceToLine(x, y, dummy));
166  return lower;
167  }
168  }
169 
170  dist_max = TMath::Abs(upper->DistanceToLine(x, y, dummy));
171  dist_min = TMath::Abs(lower->DistanceToLine(x, y, dummy));
172  // convert indices back to index in main list
173  idx_min = fIdentifiers->IndexOf(lower);
174  idx_max = fIdentifiers->IndexOf(upper);
175  if (dist_max < dist_min) {
176  dist = dist_max;
177  idx = idx_max;
178  return upper;
179  }
180  dist = dist_min;
181  idx = idx_min;
182  return lower;
183  };

185  KVIDLine* FindNextEmbracingLine(Int_t& index, Int_t inc_index, Double_t x, Double_t y, const Char_t* axis) const
186  {
187  // Starting from the line with given 'index', we search for the next line in the list of identifiers for which
188  // KVIDLine::IsBetweenEndPoints(x,y,axis) returns kTRUE.
189  // 'inc_index' is the step used to scan the list, i.e. inc_index=1 will scan index+1, index+2, ...
190  // inc_index=-1 will scan index-1, index-2, ...
191  // Returns pointer to line (or 0x0 if not found) and 'index' contains index of this line (or -1 if no line found)
192 
193  Int_t ii = index + inc_index;
194  Int_t nlines = GetNumberOfIdentifiers();
195  KVIDLine* l = 0;
196  while ((ii > -1 && ii < nlines)) {
197  l = (KVIDLine*)GetIdentifierAt(ii);
198  if (l->IsBetweenEndPoints(x, y, axis)) break;
199  ii += inc_index;
200  }
201  if (ii < 0 || ii >= nlines) {
202  // no line found
203  index = -1;
204  return 0;
205  }
206  index = ii;
207  return l;
208  };

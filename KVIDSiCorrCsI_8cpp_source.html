24 //KVIDSiCorrCsI
26 //
27 // Identification in SiCorrelated:CsI(total light) matrices for INDRA_E503.
28 // Each fit has an associated Zmin & ZMax (entered when generating the fit)
29 // Z identification is considered possible between max(Zmin-0.5, 0.5) and min(Zmax+0.5, 100.5)
30 //
31 // All of the grids were fitted using the LTG fit functional "fede" (10 fit parameters, extended functional, total light)
32 
33 // ID SUB-CODES: (Standard KVTGIDManager Codes)
34 // -------------
35 // 0 KVTGIDManager::kStatus_OK "ok",
36 // 1 KVTGIDManager::kStatus_noTGID "no KVTGID for requested identification",
37 // 2 KVTGIDManager::kStatus_OutOfIDRange "point to identify outside of identification range of function",
38 // 3 KVTGIDManager::kStatus_ZtooSmall "IdentA called with Z<1",
39 // 4 KVTGIDManager::kStatus_ZtooLarge "IdentA called with Z larger than max Z defined for KVTGIDZA isotopic identification object"

71 void KVIDSiCorrCsI::Initialize()
72 {
73  // Initialisation of telescope before identification.
74  // This method MUST be called once before any identification is attempted.
75  // IsReadyForID() will return kTRUE if KVTGID objects are associated
76  // to this telescope for the current run.
77  //
78  // This method deducts also X and Y thresholds from VarX and VarY (see GetThresholdFromVar( varX/varY ).
79 
80  fSi = (KVSilicon*)GetDetector(1);
81  fCsI = (KVCsI*)GetDetector(2);
82 
83  Bool_t ok = fSi && fCsI && GetListOfIDFunctions().GetEntries();
84 
85  if (ok) {
86  KVTGID* idf = (KVTGID*)GetListOfIDFunctions().First();
87  fThresholdX = GetThesholdFromVar(idf->GetVarX());
88  fThresholdY = GetThesholdFromVar(idf->GetVarY());
89 // Info("Initialize","Thresholds for %s: X= %f, Y= %f", GetName(), fThresholdX, fThresholdY);
90  }
91 
92  SetBit(kReadyForID, ok);
93 }

123 Double_t KVIDSiCorrCsI::GetIDMapY(Option_t* opt)
124 {
125  // This method gives the Y-coordinate in a 2D identification map
126  // associated with the Si-CsI identification telescope.
127  // The Y-coordinate is the silicon current petit gain coder data minus the petit gain pedestal. If the grand gain coder
128  // data is less than 3900 then the petit gain value is calculated
129  // from the current grand gain coder data (see KVINDRADetector::GetPGFromGG())
130 
131  IGNORE_UNUSED(opt);
132 
133  if (fSi->GetGG() < 3900.) return fSi->GetPGfromGG() - fSi->GetPedestal("PG");
134  return fSi->GetPG() - fSi->GetPedestal("PG");
135 }

153 Bool_t KVIDSiCorrCsI::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
154 {
155  //Identification of particles using SiCorrelated-CsI matrices for E503/E494s
156  //First of all, Z identification is attempted with KVIDSiCorrCsI::IdentZ.
157  //If successful, if this telescope has mass identification capabilities
158  //(HasMassID()=kTRUE), then if the identified Z is not too large for the
159  //GG grid Z&A identification, we identify the mass of the particle.
160  //In this case, a further check is made that the resulting nucleus is not
161  //too bizarre, i.e. it must be in the KVNucleus mass table (KVNucleus::IsKnown = kTRUE).
162  //If not, or if mass identification is not possible, the mass of the nucleus
163  //is deduced from Z using the default mass formula of class KVNucleus.
164  //
165  // Note that optional arguments (x,y) for testing identification are not used.
166 
167  Double_t X = (x < 0. ? GetIDMapX() : x);
168  Double_t Y = (y < 0. ? GetIDMapY() : y);
169 
170  Double_t funLTG_Z = -1;
171  Double_t funLTG_A = -1;
172  Double_t mass = -1;
173  Int_t ia = -1;
174  Int_t iz = -1;
175 
176  IDR->SetIDType(GetType());
177  IDR->IDattempted = kTRUE;
178  IDR->IDquality = 15;
179 
180  // set general ID code
181  IDR->IDcode = fIDCode;
182 
183  Double_t Z = -1.;
184 
185  const Bool_t inRange = (fThresholdX < X) && (fThresholdY < Y) && (Y < 4090.);
186 
187  if (inRange) Z = IdentZ(GetName(), X, Y, funLTG_Z, ""); //IdentZ(this, funLTG_Z, "", "");
188  else return kFALSE;
189 
190  //use KVTGIDManager::GetStatus value for IdentZ as identification subcode
191  IDR->IDquality = GetStatus();
192 
193  if (GetStatus() == KVTGID::kStatus_NotBetween_IDMin_IDMax) return kFALSE;
194 
195  if (GetStatus() != KVTGIDManager::kStatus_OK) return kFALSE; // no ID
196 
197  iz = TMath::Nint(Z);
198 
199  if (iz < 1) {
200  IDR->IDquality = KVTGIDManager::kStatus_ZtooSmall;
201  return kFALSE;
202  }
203 
204  IDR->IDOK = kTRUE;
205 
206  //is mass identification a possibility ?
207  if (iz < 9) {
208 
209  mass = IdentA(GetName(), X, Y, funLTG_A, "", iz); //IdentA(this, funLTG_A, "", "", iz);
210 
211  if (GetStatus() != KVTGIDManager::kStatus_OK) { //mass ID not good ?
212 
213  //only Z identified
214  IDR->Z = iz;
215  IDR->PID = Z;
216  IDR->Zident = kTRUE;
217  //subcode says "Z ok but A failed because..."
218  IDR->IDquality = GetStatus();
219  } else { //both Z and A successful ?
220 
221  ia = TMath::Nint(mass);
222  // fix 16B
223  if (iz == 5 && ia == 16) ia = (mass > 16.0 ? 17 : 15);
224  // fix 9B
225  if (iz == 5 && ia == 9) ia = (mass > 9.0 ? 10 : 8);
226  // fix 8Be
227  if (iz == 4 && ia == 8) ia = (mass > 8.0 ? 9 : 7);
228  // fix 13Be
229  if (iz == 4 && ia == 13) ia = (mass > 13.0 ? 14 : 12);
230  // fix 10Li
231  if (iz == 3 && ia == 10) ia = (mass > 10.0 ? 11 : 9);
232  // fix 5He
233  if (iz == 2 && ia == 5) ia = (mass > 5.0 ? 6 : 4);
234  // fix 7He
235  if (iz == 2 && ia == 7) ia = (mass > 7.0 ? 8 : 6);
236  // check that mass is not too bizarre
237  KVNucleus tmp(iz, ia);
238  if (!tmp.IsKnown()) {
239  // if ia seems too light, we try with iz-1
240  // if ia seems too heavy, we try with iz+1
241  Info("Identify", "%s : initial ID Z=%d A=%d funLTG=%g", GetName(), iz, ia, funLTG_A);
242  Int_t iz2 = (ia < 2 * iz ? iz - 1 : iz + 1);
243  if (iz2 > 0) {
244  Double_t old_funLTG_A = funLTG_A;
245  Double_t new_mass = IdentA(GetName(), X, Y, funLTG_A, "", iz2); //IdentA(this, funLTG_A, "", "", iz2);
246  // is this a better solution ?
247  if (GetStatus() == KVTGIDManager::kStatus_OK) {
248  Int_t new_ia = TMath::Nint(new_mass);
249  tmp.SetZ(iz2);
250  tmp.SetA(new_ia);
251  Info("Identify", "%s : new ID Z=%d A=%d funLTG=%g", GetName(), iz2, new_ia, funLTG_A);
252  if (tmp.IsKnown() && (TMath::Abs(funLTG_A) < TMath::Abs(old_funLTG_A))) {
253  // new nucleus is known and point is closer to line
254  iz = iz2;
255  ia = new_ia;
256  mass = new_mass;
257  Info("Identify", "%s : accepted new ID", GetName());
258  } else {
259  Info("Identify", "%s : rejected new ID", GetName());
260  }
261  } else {
262  Info("Identify", "%s : failed to obtain new ID with Z=%d", GetName(), iz2);
263  }
264  }
265  }
266  IDR->Z = iz;
267  IDR->A = ia;
268  IDR->PID = mass;
269  IDR->Zident = kTRUE;
270  IDR->Aident = kTRUE;
271  }
272  } else {
273  //only Z identified
274  //ID subcode remains 'Zstatus'
275  IDR->Z = iz;
276  IDR->PID = Z;
277  IDR->Zident = kTRUE;
278 
279  }
280 
281  return kTRUE;
282 }

294 Bool_t KVIDSiCorrCsI::SetIdentificationParameters(const KVMultiDetArray*)
295 {
296  //Initialise the identification parameters (grids, etc.) of ALL identification telescopes of this
297  //kind (label) in the multidetector array. Therefore this method need only be called once, and not
298  //called for each telescope. The kind/label (returned by GetLabel) of the telescope corresponds
299  //to the URI used to find the plugin class in $KVROOT/KVFiles/.kvrootrc.
300  //Parameters are read from the file with name given by the environment variable
301  //INDRA_camp5.IdentificationParameterFile.SI-CSI: [filename]
302 
303  TString filename = gDataSet->GetDataSetEnv(Form("IdentificationParameterFile.%s", GetLabel()));
304  if (filename == "") {
305  Warning("SetIdentificationParameters",
306  "No filename defined. Should be given by %s.IdentificationParameterFile.%s",
307  gDataSet->GetName(), GetLabel());
308  return kFALSE;
309  }
310  TString path;
311  if (!SearchKVFile(filename.Data(), path, gDataSet->GetName())) {
312  Error("SetIdentificationParameters",
313  "File %s not found. Should be in $KVROOT/KVFiles/%s",
314  filename.Data(), gDataSet->GetName());
315  return kFALSE;
316  }
317  //read grids from file
318  Info("SetIdentificationParameters", "Using file %s", path.Data());
319  return ReadAsciiFile(path.Data());
320 }

369 Double_t KVIDSiCorrCsI::GetThesholdFromVar(const Char_t* var)
370 {
371  // returns the threshold deduced from VarX or VarY.
372  // If the string 'var' contains the character '>', we consider
373  // that the threshold is given in the string after this character.
374 
375  TString thresh = var;
376  Int_t idx = thresh.Index(">");
377  if (idx > -1) {
378  thresh.Remove(0, idx + 1);
379  return (thresh.IsFloat() ? thresh.Atof() : 0.);
380  }
381  return 0.;
382 }

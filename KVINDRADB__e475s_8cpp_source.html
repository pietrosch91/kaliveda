213 void KVINDRADB_e475s::ReadCalibFile(ifstream& fin, TString dettype, TString detgain)
214 {
215 //------------------------------
216  Info("ReadCalibFile()",
217  "Reading calibration parameters of file %s%s", dettype.Data(), detgain.Data());
218 
219  KVDBParameterSet* parset = nullptr;
220  TList list_record;
221 
222  TString detname;
223  TString sline;
224  TString calib_type, calib_signal, calib_runs, calib_ring, calib_mods;
225  unique_ptr<TObjArray> toks;
226  unique_ptr<TObjArray> tokspara;
227  while (fin.good()) { //reading the file
228  sline.ReadLine(fin);
229  while (sline.BeginsWith("+KVCalibrator")) {
230  sline.ReadLine(fin);
231  while (sline.BeginsWith("<LIST> Type=")) {
232  sline.ReplaceAll("<LIST> Type=", "");
233  if (sline.Contains("(")) {
234  toks.reset(sline.Tokenize("("));
235  sline = ((TObjString*)toks->At(0))->GetString();
236  }
237  calib_type = sline;
238  sline.ReadLine(fin);
239  while (sline.BeginsWith("<LIST> Signal=")) {
240  sline.ReplaceAll("<LIST> Signal=", "");
241  calib_signal = sline;
242  cout << calib_signal << endl;
243  if (calib_signal != dettype + detgain) Error("ReadCalibFile()", "Erreur de synchro ...");
244  sline.ReadLine(fin);
245  while (sline.BeginsWith("++<LIST> Runs=")) {
246  sline.ReplaceAll("++<LIST> Runs=", "");
247  calib_runs = sline;
248  cout << calib_runs << endl;
249  sline.ReadLine(fin);
250  list_record.RemoveAll();
251  while (sline.BeginsWith("<LIST> Rings=")) {
252  sline.ReplaceAll("<LIST> Rings=", "");
253  calib_ring = sline;
254  cout << calib_ring << endl;
255  sline.ReadLine(fin);
256  while (sline.BeginsWith("<LIST> Modules=") || sline.BeginsWith("<PARAMETER> Function=")) {
257  if (sline.BeginsWith("<LIST> Modules=")) {
258  sline.ReplaceAll("<LIST> Modules=", "");
259  calib_mods = sline;
260  cout << calib_mods << endl;
261  //sprintf(detname,"%s_%02d%02d_%s",dettype.Data(),calib_ring.Atoi(),calib_mods.Atoi(),detgain.Data());
262  detname.Form("%s_%02d%02d", dettype.Data(), calib_ring.Atoi(), calib_mods.Atoi());
263  } else {
264  sline.ReplaceAll("<PARAMETER> Function=", "");
265  cout << sline << endl;
266  toks.reset(sline.Tokenize(":"));
267  TF1 ff(Form("f%s", detname.Data()), ((TObjString*)toks->At(0))->GetString().Data());
268 
269  TString sparaline = ((TObjString*)toks->At(1))->GetString();
270 
271  tokspara.reset(sparaline.Tokenize(" "));
272  if (tokspara->GetEntries() != ff.GetNpar()) cout << tokspara->GetEntries() << " " << ff.GetNpar() << endl;
273  for (Int_t pp = 0; pp < ff.GetNpar(); pp += 1) {
274  ff.SetParameter(pp, ((TObjString*)tokspara->At(pp))->GetString().Atof());
275  printf("%d %g\n", pp, ff.GetParameter(pp));
276  }
277 
278  parset = new KVDBParameterSet(detname, Form("%s(%s)", calib_type.Data(), detgain.Data()), ff.GetNpar() + 3);
279  parset->SetParamName(0, ff.GetExpFormula().Data());
280  for (Int_t pp = 0; pp < ff.GetNpar(); pp += 1) {
281  parset->SetParameter(pp + 1, ff.GetParameter(pp));
282  parset->SetParamName(pp + 1, ff.GetParName(pp));
283  }
284  parset->SetParameter(ff.GetNpar() + 1, 0.);
285  parset->SetParamName(ff.GetNpar() + 1, "xmin");
286  parset->SetParameter(ff.GetNpar() + 2, 4096.);
287  parset->SetParamName(ff.GetNpar() + 2, "xmax");
288 
289  fCalibrations->AddRecord(parset);
290  list_record.Add(parset);
291 
292  }
293  sline.ReadLine(fin);
294  }
295  }
296  KVNumberList nl(calib_runs);
297  nl.Begin();
298  while (!nl.End()) {
299  Int_t run = nl.Next();
300  for (Int_t ll = 0; ll < list_record.GetEntries(); ll += 1)
301  if (!(((KVDBRecord*)list_record.At(ll))->AddLink("Runs", GetRun(run))))
302  Error("ReadCalibFile()", "Pb de chargements ...");
303  }
304  }
305  }
306  }
307  }
308  }
309 
310  fin.close();
311 }

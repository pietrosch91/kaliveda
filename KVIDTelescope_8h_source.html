226  virtual Bool_t CanIdentify(Int_t Z, Int_t /*A*/)
227  {
228  // Used for filtering simulations
229  // Returns kTRUE if this telescope is theoretically capable of identifying a given nucleus,
230  // without considering thresholds etc.
231  // By default this method returns true for Z>0, i.e. all KVIDTelescopes are in principle
232  // dE-E telescopes used to identify charged ions.
233  return (Z > 0);
234  }
235  virtual void SetIdentificationStatus(KVReconstructedNucleus*);
236  Bool_t IsIndependent() const
237  {
238  // Returns kTRUE is this identification can be made independently of any
239  // other telescopes/detectors in the group/array etc.
240  // This is the case if
241  // - the identification is made with a single detector e.g. fast-slow id in Cesium Iodide
242  // - for a dE-E telescope, if only one trajectory passes through the dE detector
243 
244  return (GetSize() == 1 || (GetSize() == 2 && GetDetector(1)->GetNode()->GetNTraj() == 1));
245  }

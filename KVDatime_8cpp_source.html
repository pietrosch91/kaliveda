252 void KVDatime::SetGanacq2010Date(const Char_t* GanacqDateString)
253 {
254  // Decodes dates in new (2010) format of GANIL acquisition
255  // run files, e.g. run_0058.dat.27Nov10_02h07m40s

257  Char_t Month[5];
258  TString tmp(GanacqDateString);
259  tmp.ToUpper();
260  tmp.ReplaceAll("_", " ");
261  if (sscanf(tmp.Data(), "%02d%3s%02d %02dH%02dM%02dS", &D, Month, &Y, &H, &M, &S) == 6) {
262  TObject* mm = fmonths->FindObject(Month);

302 void KVDatime::SetGanacqDate(const Char_t* GanacqDateString)
303 {
304  //Decodes GANIL acquisition (INDRA) run-sheet format date into a TDatime
305  //Format of date string is:
306  // 29-SEP-2005 09:42:17.00
307  // or 29-SEP-2005 09:42:17
308  // or 29-SEP-05 09:42:17.00
309  // or 29-SEP-05 09:42:17
310  //
311  //If format is not respected, we set to current time & date
312 
313  KVString tmp(GanacqDateString);
314  TObjArray* toks = tmp.Tokenize("-:. ");
315  if (toks->GetEntries() < 6 || toks->GetEntries() > 7) {
316  // if format is correct, there should be 6 or 7 elements in toks
317  delete toks;
318  Error("SetGanacqDate", "Format is incorrect: %s (should be like \"29-SEP-2005 09:42:17.00\" or \"29-SEP-05 09:42:17\")",
319  GanacqDateString);
320  Set();
321  return;
322  }
323  KV__TOBJSTRING_TO_INT(toks, 0, day)
324  TObject* mm =
325  fmonths->FindObject(((TObjString*) toks->At(1))->String().Data());
326  Int_t month = 0;
327  if (mm)
328  month = fmonths->IndexOf(mm) + 1;
329  KV__TOBJSTRING_TO_INT(toks, 2, year)
330  // year may be written in shortened form: 97 instead of 1997
331  if (year < 100) {
332 // Warning("SetGanacqDate",
333 // "Ambiguous value for year: %d. Assuming this means: %d",
334 // year, (year < 82 ? year + 2000 : year + 1900));
335  (year < 82 ? year += 2000 : year += 1900);
336  }
337  KV__TOBJSTRING_TO_INT(toks, 3, hour)
338  KV__TOBJSTRING_TO_INT(toks, 4, min)
339  KV__TOBJSTRING_TO_INT(toks, 5, sec)
340  delete toks;
341  Set(year, month, day, hour, min, sec);
342 }

533 Double_t KVDatime::GetNumberOfSeconds(Int_t ref_year)
534 {
535 
536  Double_t total_ins = 0;
537 
538  Double_t year_ins = GetYear();
539  year_ins -= ref_year;
540 
541  year_ins *= 365.25; //1y=365.25 d
542  year_ins *= 24; //1d=24 h
543  year_ins *= 3660; //1h=60x60=3660 s
544 
545  total_ins += year_ins;
546 
547  //nombre de jours depuis le debut de l annee
548  Double_t month_ins = GetMonth() - 1;
549  month_ins *= 365.25 / 12; //1month = 30.437d (365.25/12) approximatif
550  month_ins += GetDay(); //-> nombre de jours
551  month_ins *= 24; //1d=24 h
552  month_ins *= 3660; //1h=60x60=3660 s
553 
554  total_ins += month_ins; //nombre de jours totals en second
555 
556  total_ins += GetHour() * 3600;
557  total_ins += GetMinute() * 60;
558  total_ins += GetSecond();
559 
560  return total_ins;

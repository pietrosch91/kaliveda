86 void KVBQSLogReader::ReadCPULimit(TString& line)
87 {
88  //read line of type "* total normalized: 68:57:52 (time limit: 100:00:00) *"
89  //which contains total normalized used CPU time and the time limit from the job request
90  TObjArray* toks = line.Tokenize("(:) ");
91 
92  //get used CPU time (in seconds)
93  KV__TOBJSTRING_TO_INT(toks, 3, hrs)
94  KV__TOBJSTRING_TO_INT(toks, 4, mins)
95  KV__TOBJSTRING_TO_INT(toks, 5, sec)
96  fCPUused = 3600 * hrs + 60 * mins + sec;
97  //get requested CPU time (in seconds)
98  KV__TOBJSTRING_TO_INT(toks, 8, hrs_)
99  KV__TOBJSTRING_TO_INT(toks, 9, mins_)
100  KV__TOBJSTRING_TO_INT(toks, 10, sec_)
101  fCPUreq = 3600 * hrs_ + 60 * mins_ + sec_;
102  delete toks;
103 }

149 Int_t KVBQSLogReader::GetByteMultiplier(TString& unit)
150 {

187 Double_t KVBQSLogReader::ReadStorage(KVString& stor)
188 {
189  //'stor' is a string such as "200MB", "3GB" etc.
190  //value returned is corresponding storage space in KB
191  static const Char_t* units[] = { "KB", "MB", "GB" };
192  Int_t i = 0, index = -1;
193  while ((index = stor.Index(units[i])) < 0 && i < 2)
194  i++;
195  if (index < 0)
196  return 0;
197  stor.Remove(index);
198  TString u(units[i]);
199  return (stor.Atof() * GetByteMultiplier(u));
200 }

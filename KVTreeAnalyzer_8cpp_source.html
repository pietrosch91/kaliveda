263  // The expression and selection should be valid TTreeFormula strings

268  //
269  // "expr1[:expr2]"
270  // "expr1[:expr2] {selection}"
271  // "expr1[:expr2] {active selection}"
272  // "expr1[:expr2] {(active selection) && (selection)}"

277 
278

281  if (fChain->GetEntryList()) _elist = fChain->GetEntryList()->GetTitle();
282  if (strcmp(weight, "")) {
283  if (_selection != "" && _elist != "")
284  title.Form("%s [%s] {(%s) && (%s)}", expr, weight, _elist.Data(), selection);
285  else if (_selection != "")
286  title.Form("%s [%s] {%s}", expr, weight, selection);
287  else if (_elist != "")
288  title.Form("%s [%s] {%s}", expr, weight, _elist.Data());
289  else
290  title.Form("%s [%s]", expr, weight);
291  } else {
292  if (_selection != "" && _elist != "")
293  title.Form("%s {(%s) && (%s)}", expr, _elist.Data(), selection);
294  else if (_selection != "")
295  title.Form("%s {%s}", expr, selection);

322 TH1* KVTreeAnalyzer::MakeHisto(const Char_t* expr, const Char_t* selection, Int_t nX, Int_t nY, const Char_t* weight)
323 {
324  // Create and fill a new histogram with the desired expression (expr="expr1[:expr2]" etc.)
325  // with the given selection (selection="" if no selection required).
326  // Any currently active selection (TEntryList set on TTree) will also be applied.
327  // The new histogram is not drawn but added to the internal list of histograms
328  // (see method AddHisto).
329  //
330  // Histograms are automatically named 'h1', 'h2', etc. in order of creation.
331  // Histogram title is generated with method GenerateHistoTitle.
332  // Number of bins on X (and Y for a 2-D spectrum) are given. Axis limits are
333  // automatically adjusted to data.
334  //
335  // For 2-D spectra the initial drawing option is set to "COL"
336  //
337  // If normalisation of spectra is required (fNormHisto = kTRUE) the histogram
338  // bin contents are divided by the integral (sum of weights).
339 
340  TString name;
341  name.Form("h%d", fHistoNumber);
342  TString drawexp(expr), histo, histotitle;
343  if (strcmp(weight, "")) GenerateHistoTitle(histotitle, expr, selection, weight);
344  else GenerateHistoTitle(histotitle, expr, selection);
345  if ((!nY) && (fUserBinning)) {
346  ResetMethodCalled();
347  Bool_t ok = KVBase::OpenContextMenu("DefineUserBinning1F", this, "DefineUserBinning");
348  if (!ok) return 0;
349  // cancel was pressed ?
350  if (MethodNotCalled()) return 0;
351  }
352 
353  TString Selection;
354  if (strcmp(weight, "")) {
355  if (strcmp(selection, "")) Selection.Form("(%s)&&(%s)", selection, weight);
356  else Selection = weight;
357  } else
358  Selection = selection;
359  if (nY) histo.Form(">>%s(%d,0.,0.,%d,0.,0.)", name.Data(), nX, nY);
360  else histo.Form(">>%s(%d,%lf,%lf)", name.Data(), (fUserBinning ? fNxF : nX), (fUserBinning ? fXminF : 0.), (fUserBinning ? fXmaxF : 0));
361 
362  if (!fProfileHisto) drawexp += histo;
363  Long64_t drawResult;
364  if (fProfileHisto) drawResult = fTree->Draw(Form("%s>>%s", drawexp.Data(), name.Data()), Selection, "prof,goff");
365  else drawResult = fTree->Draw(drawexp, Selection, "goff");
366  if (drawResult < 0) {
367  // Error with Draw: probably a bad expression
368  new TGMsgBox(gClient->GetRoot(), fMain_histolist, "Error", "Problem drawing histogram: check the expressions?", kMBIconExclamation, kMBDismiss);
369  return nullptr;
370  }
371  TH1* h;
372  if (IsPROOFEnabled()) h = (TH1*)gProof->GetOutputList()->FindObject(name);
373  else h = (TH1*)gDirectory->Get(name);
374  h->SetTitle(histotitle);
375  if (h->InheritsFrom("TH2")) h->SetOption(fDrawOption);
376  h->SetDirectory(0);
377  AddHisto(h);
378  fHistoNumber++;
379  if (!fProfileHisto) {
380  if (fNormHisto || fNormHistoEvents) {
381  h->Sumw2();
382  if (fNormHisto) {
383  h->Scale(1. / h->Integral());
384  } else {
385  h->Scale(1. / GetEntriesInCurrentSelection());
386  }
387  }
388  }
389  return h;
390 }

488 Bool_t KVTreeAnalyzer::MakeSelection(const Char_t* selection)
489 {
490  // Generate a new user-selection (TEntryList) of events in the TTree
491  // according to the given selection expression (valid TTreeFormula expression
492  // using TTree leaf and/or alias names).
493  // The new selection is not applied immediately but added to the internal
494  // list of selections (see method AddSelection).
495  //
496  // If there is already an active selection (TEntryList set on TTree)
497  // this will generate the composite selection, {(active selection) && (selection)}.
498  // The selection's title will then be in the following format:
499  //
500  // "[(active selection) && ](selection)"
501  //
502  // TEntryList objects are automatically named 'el1', 'el2', etc. in order of creation.
503 
504  TObject* tmpObj = gROOT->FindObject(selection);
505  if (tmpObj) {
506  if (tmpObj->InheritsFrom("TCutG")) {
507  TCutG* cut = (TCutG*) tmpObj;
508  cut->SetTitle(cut->GetName());
509  AddCut(cut);
510  }
511  }
512 
513  TString name;
514  name.Form("el%d", fSelectionNumber);
515  TString drawexp(name.Data());
516  drawexp.Prepend(">>");
517  fChain->SetProof(kFALSE);
518  if (IsPROOFEnabledForSelections()) fChain->SetProof(kTRUE);
519  if (fChain->Draw(drawexp, selection, "entrylist") < 0) {
520  new TGMsgBox(gClient->GetRoot(), 0, "Warning", "Mistake in your new selection!", kMBIconExclamation, kMBClose);
521  return kFALSE;
522  }
523  if (!IsPROOFEnabledForSelections() && IsPROOFEnabled()) fChain->SetProof(kTRUE);
524  TEntryList* el;
525  if (IsPROOFEnabledForSelections())
526  el = (TEntryList*)gProof->GetOutputList()->FindObject(name);
527  else
528  el = (TEntryList*)gDirectory->Get(name);
529  el->SetTitle(selection);//needed with PROOF
530  if (fChain->GetEntryList()) {
531  TString _elist = fChain->GetEntryList()->GetTitle();
532  TString title;
533  title.Form("(%s) && (%s)", _elist.Data(), selection);
534  el->SetTitle(title);
535  }
536  fSelectionNumber++;
537  AddSelection(el);
538  SelectionChanged();
539  return kTRUE;
540 }

608 void KVTreeAnalyzer::FillLeafList()
609 {
610  // Fills the GUI list with the names of all leaves in the TTree
611  // and any friend TTrees and all aliases defined by the user
612 
613  TList stuff;
614  if (fTree) {
615  // clone list of leaves
616  fLeafList.Clear();
617  TCollection* clones = (TCollection*)fTree->GetListOfLeaves()->Clone();
618  // when using a split object to fill the tree
619  // there is a redundant leaf/branch corresponding to the object itself
620  // more precisely there will be a TLeafElement in the list of leaves
621  // and a TBranchElement in the list of branches with the same name
622  TIter next(clones);
623  TObject* o;
624  while ((o = next())) {
625  if (o->InheritsFrom("TLeafElement")
626  && fTree->GetListOfBranches()->FindObject(o->GetName())
627  && fTree->GetListOfBranches()->FindObject(o->GetName())->InheritsFrom("TBranchElement"))
628  continue;
629  fLeafList.Add(o);
630  }
631  delete clones;
632 
633  stuff.AddAll(&fLeafList);
634  stuff.AddAll(fTree->GetListOfAliases());
635  if (fTree->GetListOfFriends()) {
636  TIter it(fTree->GetListOfFriends());
637  TFriendElement* fel;
638  while ((fel = (TFriendElement*)it())) {
639  stuff.AddAll(fel->GetTree()->GetListOfLeaves());
640  stuff.AddAll(fel->GetTree()->GetListOfAliases());
641  }
642  }
643  }
644  stuff.AddAll(&fAliasList);
645  G_leaflist->Display(&stuff);
646 }

299 Double_t KVSpIdGUI::GetFactor()
300 {
301  Double_t fact = TMath::Sqrt((fXm - fPdx) * (fXm - fPdx) * (fSfx * fSfx) + (fYm - fPdy) * (fYm - fPdy) * (fSfy * fSfy)) / (20.*fZp * 12.);
302  return fact;

309 void KVSpIdGUI::SpiderIdentification()
310 {
311  if ((!fHisto) || (!fGrid)) return;
312 
313  TVirtualPad* pad = fGrid->GetPad();
314  fGrid->UnDraw();
315 
316  fZp = fZpEntry->GetIntNumber();
317  if (!fUserParameter) fSpFactor = GetFactor();
318  else fSpFactor = fSpiderFactorEntry->GetNumber();
319 
320  fAnglesUp = fAngleUpEntry->GetIntNumber();
321  fAnglesDown = fAngleDownEntry->GetIntNumber();
322  fAlpha = fApertureUpEntry->GetNumber();
323  fPiedType = fPiedChoice->GetSelected();
324  fMatrixType = fTypeChoice->GetSelected();
325  Int_t type = fMatrixType;
326 
327  TH2* tmpHisto = fHisto;
328  TList* tmpCut = 0;
329  if (fUseCut) {
330  tmpHisto = (TH2*)fHisto->Clone(Form("%s_cut", fHisto->GetName()));
331  tmpHisto->Reset();
332  for (int i = 1; i <= fHisto->GetNbinsX(); i++) {
333  for (int j = 1; j <= fHisto->GetNbinsY(); j++) {
334  Stat_t ww = fHisto->GetBinContent(i, j);
335  Axis_t x0 = fHisto->GetXaxis()->GetBinCenter(i);
336  Axis_t y0 = fHisto->GetYaxis()->GetBinCenter(j);
337  if (fGrid->IsIdentifiable(x0, y0)) tmpHisto->Fill(x0, y0, ww);
338  }
339  }
340  tmpCut = (TList*)fGrid->GetCuts()->Clone("tmpCuts");
341  }
342 
343  fGrid->Clear();
344 
345  if (fScaledHisto) delete fScaledHisto;
346  KVHistoManipulator hm;
347 
348  TF1 RtLt("RtLt", Form("x*%lf", fSfx), 0, tmpHisto->GetXaxis()->GetXmax());
349  TF1 RtLty("RtLty", Form("x*%lf", fSfy), 0, tmpHisto->GetXaxis()->GetXmax());
350  fScaledHisto = (TH2F*)hm.ScaleHisto(tmpHisto, &RtLt, &RtLty);
351 
352  if (fIdentificator) delete fIdentificator;
353  fIdentificator = new KVSpiderIdentificator(fScaledHisto, fXm * fSfx, fYm * fSfy, -1, -1, fMatrixType);
354 
355  switch (fPiedType) {
356  case kUser:
357  fIdentificator->SetX0(fPdx * fSfx);
358  fIdentificator->SetY0(fPdy * fSfy);
359  break;
360  case kAuto:
361  break;
362  case kNone:
363  fIdentificator->SetX0(0.);
364  fIdentificator->SetY0(0.);
365  }
366 
367  fIdentificator->SetParameters(fSpFactor);
368  fIdentificator->SetNangles(fAnglesUp, fAnglesDown);
369  fIdentificator->SetAlpha(fAlpha);
370 
371  fProgressBar->SetRange(0, fAnglesUp + fAnglesDown + 1);
372  fProgressBar->Reset();
373  fIdentificator->Connect("Increment(Float_t)", "TGHProgressBar", fProgressBar, "SetPosition(Float_t)");
374 
375  fTestButton->SetEnabled(kFALSE);
376  fCloseButton->SetEnabled(kFALSE);
377  fIdentificator->ProcessIdentification();
378  fTestButton->SetEnabled(kTRUE);
379  fCloseButton->SetEnabled(kTRUE);
380 
381  fIdentificator->Disconnect("Increment(Float_t)", fProgressBar, "SetPosition(Float_t)");
382  fProgressBar->Reset();
383 
384  if (fDebug) fIdentificator->Draw(fOption.Data());
385 
386  TList* ll = (TList*)fIdentificator->GetListOfLines();
387 
388  KVIDZALine* TheLine = 0;
389  int zmax = 0;
390 
391  KVSpiderLine* spline = 0;
392  TIter next_line(ll);
393  while ((spline = (KVSpiderLine*)next_line())) {
394  if ((spline->GetN() > 10)) { //&&(spline->GetX(0)<=fIdentificator->GetX0()+200.))
395  TF1* ff1 = 0;
396 
397 // printf("#%s# z=%d\n",spline->GetName(),spline->GetZ());
398 
399  if (type == KVSpiderIdentificator::kSiCsI) ff1 = spline->GetFunction(fPdx * fSfx, TMath::Max(fScaledHisto->GetXaxis()->GetXmax() * 0.9, spline->GetX(spline->GetN() - 1)));
400  else if (type == KVSpiderIdentificator::kSiSi) ff1 = spline->GetFunction(fPdx * fSfx, TMath::Min(fScaledHisto->GetXaxis()->GetXmax() * 0.9, spline->GetX(spline->GetN() - 1) * 1.5));
401  else if (type == KVSpiderIdentificator::kChIoSi) ff1 = spline->GetFunction(fPdx * fSfx, TMath::Min(fScaledHisto->GetXaxis()->GetXmax() * 0.9, spline->GetX(spline->GetN() - 1) * 1.5));
402  else ff1 = spline->GetFunction();
403 
404  /*
405  if ((type == kSiCsI) && (ff1->GetParameter(1) >= 3000. || (ff1->GetParameter(2) <= 0.35) || (ff1->GetParameter(2) >= 1.))) {
406  Info("SpiderIdentification", "Z = %d has been rejected (fit parameters)", spline->GetZ());
407  continue;
408  }
409  */
410  TheLine = (KVIDZALine*)((KVIDZAGrid*)fGrid)->NewLine("ID");
411  TheLine->SetZ(spline->GetZ());
412  double min, max;
413  ff1->GetRange(min, max);
414  double step = TMath::Min((max - min) * 0.05, 20.); //20.;
415  double stepmax = (max - min) * 0.2; //800.;
416  double x = 0.;
417  for (x = min + 1; x < max + step; x += step) {
418  if (step <= stepmax) step *= 1.3;
419  if (ff1->Eval(x) < 4000) TheLine->SetPoint(TheLine->GetN(), x, ff1->Eval(x));
420  }
421  if (max > x) TheLine->SetPoint(TheLine->GetN(), max, ff1->Eval(max));
422 
423  fGrid->Add("ID", TheLine);
424  if (spline->GetZ() >= zmax) zmax = spline->GetZ();
425  } else {
426  Info("SpiderIdentification", "Z = %d has been rejected (too few points)", spline->GetZ());
427  }
428  }
429 
430  TF1 fx("fx12", Form("x/%lf", fSfx), 0., fScaledHisto->GetNbinsX() * 1.);
431  TF1 fy("fy12", Form("x/%lf", fSfy), 0., fScaledHisto->GetNbinsY() * 1.);
432  fGrid->Scale(&fx, &fy);
433 
434  if (fUseCut) delete tmpHisto;
435 
436  if (tmpCut) fGrid->GetCuts()->AddAll(tmpCut);
437  pad->cd();
438  fGrid->Draw();
439  pad->Modified();
440  pad->Update();
441 
442  DoClose();
443 }

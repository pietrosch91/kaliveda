293 Bool_t KVZGOUBIReconstruction::ReconstructFPtoLab(Double_t x_f, Double_t y_f, Double_t theta_f, Double_t phi_f, Double_t& brho, Double_t& path, Double_t& theta_v, Double_t& phi_v)
294 {
295  // x_f and y_f in cm.
296  // phi_f, theta_f, phi_l and theta_l in degree.
297 
298  static KVVAMOSReconTrajectory traj;
299 
300  traj.pointFP[0] = x_f;
301  traj.pointFP[1] = y_f;
302  Double_t x = TMath::Sin(theta_f * TMath::DegToRad()) * TMath::Cos(phi_f * TMath::DegToRad());
303  Double_t y = TMath::Sin(phi_f * TMath::DegToRad());
304  Double_t z = TMath::Cos(theta_f * TMath::DegToRad()) * TMath::Cos(phi_f * TMath::DegToRad());
305  traj.dirFP.SetXYZ(x, y, z);
306  traj.SetFPparamsReady();
307 
308  Bool_t status = ReconstructFPtoLab(&traj);
309 
310  path = traj.GetPath();
311  brho = traj.GetBrho();
312  theta_v = traj.GetThetaV();
313  phi_v = traj.GetPhiV();
314 
315  return status;
316 }

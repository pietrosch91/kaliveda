98 NewRooMinuit::NewRooMinuit(RooAbsReal& function)
99 {
100  // Construct MINUIT interface to given function. Function can be anything,
101  // but is typically a -log(likelihood) implemented by RooNLLVar or a chi^2
102  // (implemented by RooChi2Var). Other frequent use cases are a RooAddition
103  // of a RooNLLVar plus a penalty or constraint term. This class propagates
104  // all RooFit information (floating parameters, their values and errors)
105  // to MINUIT before each MINUIT call and propagates all MINUIT information
106  // back to the RooFit object at the end of each call (updated parameter
107  // values, their (asymmetric errors) etc. The default MINUIT error level
108  // for HESSE and MINOS error analysis is taken from the defaultErrorLevel()
109  // value of the input function.
110 
111  RooSentinel::activate() ;
112 
113  // Store function reference
114  _evalCounter = 0 ;
115  _extV = 0 ;
116  _func = &function ;
117  _logfile = 0 ;
118  _optConst = kFALSE ;
119  _verbose = kFALSE ;
120  _profile = kFALSE ;
121  _handleLocalErrors = kTRUE ;
122  _printLevel = 1 ;
123  _printEvalErrors = 10 ;
124  _warnLevel = -999 ;
125  _maxEvalMult = 500 ;
126  _doEvalErrorWall = kTRUE ;
127 
128  // Examine parameter list
129  RooArgSet* paramSet = function.getParameters(RooArgSet()) ;
130  RooArgList paramList(*paramSet) ;
131  delete paramSet ;
132 
133  _floatParamList = (RooArgList*) paramList.selectByAttrib("Constant", kFALSE) ;
134  if (_floatParamList->getSize() > 1) {
135  _floatParamList->sort() ;
136  }
137  _floatParamList->setName("floatParamList") ;
138 
139  _constParamList = (RooArgList*) paramList.selectByAttrib("Constant", kTRUE) ;
140  if (_constParamList->getSize() > 1) {
141  _constParamList->sort() ;
142  }
143  _constParamList->setName("constParamList") ;
144 
145  // Remove all non-RooRealVar parameters from list (MINUIT cannot handle them)
146  TIterator* pIter = _floatParamList->createIterator() ;
147  RooAbsArg* arg ;
148  while ((arg = (RooAbsArg*)pIter->Next())) {
149  if (!arg->IsA()->InheritsFrom(RooAbsRealLValue::Class())) {
150  coutW(Minimization) << "NewRooMinuit::NewRooMinuit: removing parameter " << arg->GetName()
151  << " from list because it is not of type RooRealVar" << endl ;
152  _floatParamList->remove(*arg) ;
153  }
154  }
155  _nPar = _floatParamList->getSize() ;
156  delete pIter ;
157 
158  updateFloatVec() ;
159 
160  // Save snapshot of initial lists
161  _initFloatParamList = (RooArgList*) _floatParamList->snapshot(kFALSE) ;
162  _initConstParamList = (RooArgList*) _constParamList->snapshot(kFALSE) ;
163 
164  // Initialize MINUIT
165  Int_t nPar = _floatParamList->getSize() + _constParamList->getSize() ;
166  if (_theFitter) delete _theFitter ;
167  _theFitter = TVirtualFitter::Fitter(this, nPar * 2 + 1);
168 
169  //_theFitter = new TFitter(nPar*2+1) ; //WVE Kludge, nPar*2 works around TMinuit memory allocation bug
170  //_theFitter->SetObjectFit(this) ;
171 
172  // Shut up for now
173  setPrintLevel(-1) ;
174  _theFitter->Clear();
175 
176  // Tell MINUIT to use our global glue function
177  _theFitter->SetFCN(RooMinuitGlue);
178 
179  // Use +0.5 for 1-sigma errors
180  setErrorLevel(function.defaultErrorLevel()) ;
181 
182  // Declare our parameters to MINUIT
183  synchronize(kFALSE) ;
184 
185  // Reset the *largest* negative log-likelihood value we have seen so far
186  _maxFCN = -1e30 ;
187  _numBadNLL = 0 ;
188 
189  // Now set default verbosity
190  if (RooMsgService::instance().silentMode()) {
191  setWarnLevel(-1) ;
192  setPrintLevel(-1) ;
193  } else {
194  setWarnLevel(1) ;
195  setPrintLevel(1) ;
196  }
197 }

312 NewRooFitResult* NewRooMinuit::fit(const char* options)
313 {
314  // Parse traditional RooAbsPdf::fitTo driver options
315  //
316  // s - Run Hesse first to estimate initial step size
317  // m - Run Migrad only
318  // h - Run Hesse to estimate errors
319  // v - Verbose mode
320  // l - Log parameters after each Minuit steps to file
321  // t - Activate profile timer
322  // r - Save fit result
323  // 0 - Run Migrad with strategy 0
324 
325  if (_floatParamList->getSize() == 0) {
326  return 0 ;
327  }
328 
329  _theFitter->SetObjectFit(this) ;
330 
331  TString opts(options) ;
332  opts.ToLower() ;
333 
334  // Initial configuration
335  if (opts.Contains("v")) setVerbose(1) ;
336  if (opts.Contains("t")) setProfile(1) ;
337  if (opts.Contains("l")) setLogFile(Form("%s.log", _func->GetName())) ;
338  if (opts.Contains("c")) optimizeConst(1) ;
339 
340  // Fitting steps
341  if (opts.Contains("s")) hesse() ;
342  if (opts.Contains("0")) setStrategy(0) ;
343  migrad() ;
344  if (opts.Contains("0")) setStrategy(1) ;
345  if (opts.Contains("h") || !opts.Contains("m")) hesse() ;
346  if (!opts.Contains("m")) minos() ;
347 
348  return (opts.Contains("r")) ? save() : 0 ;
349 }

489 Int_t NewRooMinuit::minos(const RooArgSet& minosParamList)
490 {
491  // Execute MINOS for given list of parameters. Changes in parameter values
492  // and calculated errors are automatically
493  // propagated back the RooRealVars representing
494  // the floating parameters in the MINUIT operation
495 
496  if (_floatParamList->getSize() == 0) {
497  return -1 ;
498  }
499 
500  _theFitter->SetObjectFit(this) ;
501 
502  Int_t nMinosPar(0) ;
503  Double_t* arglist = new Double_t[_nPar + 1];
504 
505  if (minosParamList.getSize() > 0) {
506  TIterator* aIter = minosParamList.createIterator() ;
507  RooAbsArg* arg ;
508  while ((arg = (RooAbsArg*)aIter->Next())) {
509  RooAbsArg* par = _floatParamList->find(arg->GetName());
510  if (par && !par->isConstant()) {
511  Int_t index = _floatParamList->index(par);
512  nMinosPar++;

519 
520  synchronize(_verbose) ;
521  profileStart() ;
522  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::CollectErrors) ;
523  RooAbsReal::clearEvalErrorLog() ;
524  _status = _theFitter->ExecuteCommand("MINOS", arglist, 1 + nMinosPar);
525  // check also the status of Minos looking at fCstatu
526  if (_status == 0 && gMinuit->fCstatu != "SUCCESSFUL") {
527  if (gMinuit->fCstatu == "FAILURE" ||
528  gMinuit->fCstatu == "PROBLEMS") _status = 5;
529  _status = 6;
530  }
531  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::PrintErrors) ;
532  profileStop() ;
533  backProp() ;
534 
535  delete[] arglist ;

537  saveStatus("MINOS", _status) ;
538 
539  return _status ;

733 Bool_t NewRooMinuit::synchronize(Bool_t verbose)
734 {
735  // Internal function to synchronize TMinuit with current
736  // information in RooAbsReal function parameters
737 
738  Int_t oldPrint = setPrintLevel(-1) ;
739  gMinuit->fNwrmes[0] = 0; // to clear buffer
740  Int_t oldWarn = setWarnLevel(-1) ;
741 
742  Bool_t constValChange(kFALSE) ;
743  Bool_t constStatChange(kFALSE) ;
744 
745  Int_t index(0) ;
746 
747  // Handle eventual migrations from constParamList -> floatParamList
748  for (index = 0; index < _constParamList->getSize() ; index++) {
749  RooRealVar* par = dynamic_cast<RooRealVar*>(_constParamList->at(index)) ;
750  if (!par) continue ;
751 
752  RooRealVar* oldpar = dynamic_cast<RooRealVar*>(_initConstParamList->at(index)) ;
753  if (!oldpar) continue ;
754 
755  // Test if constness changed
756  if (!par->isConstant()) {
757 
758  // Remove from constList, add to floatList
759  _constParamList->remove(*par) ;
760  _floatParamList->add(*par) ;
761  _initFloatParamList->addClone(*oldpar) ;
762  _initConstParamList->remove(*oldpar) ;
763  constStatChange = kTRUE ;
764  _nPar++ ;
765 
766  if (verbose) {
767  coutI(Minimization) << "NewRooMinuit::synchronize: parameter " << par->GetName() << " is now floating." << endl ;
768  }
769  }
770 
771  // Test if value changed
772  if (par->getVal() != oldpar->getVal()) {
773  constValChange = kTRUE ;
774  if (verbose) {
775  coutI(Minimization) << "NewRooMinuit::synchronize: value of constant parameter " << par->GetName()
776  << " changed from " << oldpar->getVal() << " to " << par->getVal() << endl ;
777  }
778  }
779 
780  }
781 
782  // Update reference list
783  *_initConstParamList = *_constParamList ;
784 
785 
786  // Synchronize MINUIT with function state
787  for (index = 0; index < _nPar; index++) {
788  RooRealVar* par = dynamic_cast<RooRealVar*>(_floatParamList->at(index)) ;
789  if (!par) continue ;
790 
791  Double_t pstep(0) ;
792  Double_t pmin(0) ;
793  Double_t pmax(0) ;
794 
795  if (!par->isConstant()) {
796 
797  // Verify that floating parameter is indeed of type RooRealVar
798  if (!par->IsA()->InheritsFrom(RooRealVar::Class())) {
799  coutW(Minimization) << "NewRooMinuit::fit: Error, non-constant parameter " << par->GetName()
800  << " is not of type RooRealVar, skipping" << endl ;
801  continue ;
802  }
803 
804  // Set the limits, if not infinite
805  if (par->hasMin() && par->hasMax()) {
806  pmin = par->getMin();
807  pmax = par->getMax();
808  }
809 
810  // Calculate step size
811  pstep = par->getError();
812  if (pstep <= 0) {
813  // Floating parameter without error estitimate
814  if (par->hasMin() && par->hasMax()) {
815  pstep = 0.1 * (pmax - pmin);
816 
817  // Trim default choice of error if within 2 sigma of limit
818  if (pmax - par->getVal() < 2 * pstep) {
819  pstep = (pmax - par->getVal()) / 2 ;
820  } else if (par->getVal() - pmin < 2 * pstep) {
821  pstep = (par->getVal() - pmin) / 2 ;
822  }
823 
824  // If trimming results in zero error, restore default
825  if (pstep == 0) {
826  pstep = 0.1 * (pmax - pmin);
827  }
828 
829  } else {
830  pstep = 1 ;
831  }
832  if (_verbose) {
833  coutW(Minimization) << "NewRooMinuit::synchronize: WARNING: no initial error estimate available for "
834  << par->GetName() << ": using " << pstep << endl;
835  }
836  }
837  } else {
838  pmin = par->getVal() ;
839  pmax = par->getVal() ;
840  }
841 
842  // Extract previous information
843  Double_t oldVar, oldVerr, oldVlo, oldVhi ;
844  char oldParname[100] ;
845  Int_t ierr = _theFitter->GetParameter(index, oldParname, oldVar, oldVerr, oldVlo, oldVhi) ;
846 
847  // Determine if parameters is currently fixed in MINUIT
848 
849  Int_t ix ;
850  Bool_t oldFixed(kFALSE) ;
851  if (ierr >= 0) {
852  for (ix = 1; ix <= gMinuit->fNpfix; ++ix) {
853  if (gMinuit->fIpfix[ix - 1] == index + 1) oldFixed = kTRUE ;
854  }
855  }
856 
857  if (par->isConstant() && !oldFixed) {
858

60 void KVINDRAUpDater_e613::SetGains(KVDBRun* kvrun)
61 {
62  //Set gains used during this run
63  //Read gains in database just print detectors for which gains have changed
64  //
65 
66  KVRList* gain_list = kvrun->GetLinks("Gains");
67  if (!gain_list) {
68  return;
69  Warning("SetGains", "No gains defined for this run in database");
70  }
71  Int_t ndets = gain_list->GetSize();
72  Info("SetGains", "Loop on %d detectors : ...", ndets);
73 
74  Int_t nchange = 0;
75  KVDetector* kvd = 0;
76  Double_t oldgain;
77  TString list;
78  for (Int_t i = 0; i < ndets; i++) {
79  KVDBParameterSet* dbps = (KVDBParameterSet*) gain_list->At(i);
80  kvd = gIndra->GetDetector(dbps->GetName());
81  if (!kvd) {
82  //Error("SetGains",
83  // "Detector %s is unknown or does not exist at the current time...???",

85  continue;
86  } else {
87  oldgain = kvd->GetGain();
88  if (oldgain != dbps->GetParameter(0)) {
89  kvd->SetGain(dbps->GetParameter(0));
90  //cout << " " << kvd->GetName() << " set gain from " << oldgain << " to G=" << kvd->GetGain() << endl;
91  list += kvd->GetName();
92  list += ",";
93  nchange += 1;
94  }
95  }
96  }
97  if (nchange == 0)
98  Info("SetGains", "Gains of the %d detectors are the same than the run before ", ndets);

169

182  if (!kvc)

184  "Calibrator %s %s not found !", kvps->GetName(),
185  kvps->GetTitle());

187  //Prise en compte du gain du detecteur quand la rampe gene a ete faite

189  Double_t gain_ref = kvps->GetParameter(kvc->GetNumberParams());
190  ((KVChannelVolt*)kvc)->SetGainRef(gain_ref);
191  for (Int_t i = 0; i < kvc->GetNumberParams(); i++) {
192  kvc->SetParameter(i, kvps->GetParameter(i));
193  }
194  kvc->SetStatus((gain_ref != 0)); // calibrator ready
195  } //calibrator found

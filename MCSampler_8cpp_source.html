156  void MCSampler::CalculateWeights(Double_t excitation_energy)
157  {
158  // calculate weights of all partitions for the given excitation energy
159  // (in MeV) of the initial compound nucleus
160 
161  //Info("CalculateWeights","Calculating channel weights for E*=%f",excitation_energy);
162 
163  if (!fWeightList) initialiseWeightList();
164 
165  fSumWeights = 0;
166 
167  for (int i = 0; i < fPartitions; i++) {
168  StatWeight* w = (StatWeight*)fWeightList->ConstructedAt(i);
169  GetPartition(i);
170  if (fModifyMasses) UpdateMasses();
171  w->SetWeight(fPartition, excitation_energy + fPartition->GetChannelQValue());
172  w->SetIndex(i);
173  fSumWeights += w->GetWeight();

176  }
177

179  fWeightList->Sort();

192  Long64_t MCSampler::PickRandomChannel()
193  {
194  // Return the TTree index of a random channel according to the
195  // previously calculated weights
196  // The corresponding weight for the chosen channel can be retrieved
197  // by calling GetRandomChannelWeight()
198  // If no channel is open (i.e. all weights = 0, E* < Q value of first channel),
199  // we return -1.
200 
201  Double_t x = gRandom->Uniform(fSumWeights);
202 
203  int i;
204  for (i = 0; i < fPartitions; i++) {
205  Double_t w = ((StatWeight*)(*fWeightList)[i])->GetWeight();
206  if (x < w) break;
207  x -= w;
208  }
209  if (i == fPartitions) {
210  fLastPicked = nullptr;
211  return -1;
212  }
213  fLastPicked = (StatWeight*)(*fWeightList)[i];
214  return fLastPicked->GetIndex();
215  }

274  void MCSampler::GenerateEvents(TTree* theTree, KVEvent* event, Double_t Exx, Long64_t npartitions, Long64_t nev_part)
275  {
276  // Generate (npartitions*nev_part) events for a given value of excitation energy 'Exx' [MeV].
277  // For efficiency, for each chosen partition we generate nev_part events.
278  // This means :
279  // - calculating the statistical weight of all available decay channels/partitions
280  // - picking a channel at random
281  // - generating momenta of all nuclei in chosen channel
282  // - filling the TTree with the new event
283 
284  Info("GenerateEvents", "Generating events for E*=%f", Exx);
285 
286  if (!SetExcitationEnergy(Exx)) {
287  Error("GenerateEvents", "Excitation energy is too low, no channels are open");
288  return;
289  }
290 
291  // generate events
292  while (npartitions--) {
293 
294  SetDecayChannel();
295 
296 
297  for (Long64_t iev = 0; iev < nev_part; iev++) {
298 
299  // generate nev_part events for chosen partition
300 
301  GenerateEvent(theTree, event);
302 
303  }
304 
305  }

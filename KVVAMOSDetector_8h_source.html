146 Bool_t KVVAMOSDetector::Fired(Option_t* opt)
147 {
148  //Returns kTRUE if detector was hit (fired) in an event
149  //
150  //The actual meaning of hit/fired depends on the context and the option string opt.
151  //
152  //opt="any" (default):
153  //Returns true if at least one working acquisition parameter
154  //associated with the detector was fired in an event, for ANY of the types
155  //in the list*.
156  //
157  //opt="all" :
158  //Returns true if at least one working acquisition parameter
159  //associated with the detector was fired in an event, for ALL of the types
160  //in the list*.
161  //
162  //opt="Pany" :
163  //Returns true if at least one working acquisition parameter
164  //associated with the detector was fired in an event and have a value
165  //greater than their pedestal value, for ANY of the types in the list*.
166  //
167  //opt="Pall" :
168  //Returns true if at least one working acquisition parameter
169  //associated with the detector was fired in an event and have a value
170  //greater than their pedestal value, for ALL of the types in the list*.
171  //
172  // *the actual parameters taken into account can be fine tuned using environment variables such as
173  // KVVAMOSDetector.Fired.ACQParameterList.[type]: Q,E,T,T_HF,X,Y
174  // See KVAMOSDetector::SetFiredBitmask() for more details.
175 
176  if (opt[0] == 'P') return FiredP(opt + 1);
177  return Fired(opt, "");
178 }

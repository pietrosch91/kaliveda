150 void KVGeoStrucElement::Add(KVBase* element)
151 {
152  // Add a detector or a daughter structure to this structure.
153  // Note that daughter structures are by defeault owned by their mother and will be
154  // deleted by her destructor. Call SetOwnsDaughters(kFALSE) to change.
155  // Detectors are not owned by default - call SetOwnsDetectors(kTRUE) to change.
156 
157  if (element->InheritsFrom(KVDetector::Class())) {
158  fDetectors.Add(element);
159  dynamic_cast<KVDetector*>(element)->AddParentStructure(this);
160  } else if (element->InheritsFrom(KVGeoStrucElement::Class())) {
161  fStructures.Add(element);
162  dynamic_cast<KVGeoStrucElement*>(element)->AddParentStructure(this);
163  } else {
164  Error("Add", "Cannot add elements of class %s", element->ClassName());
165  }
166 }

288 KVDetector* KVGeoStrucElement::GetDetectorAny(const Char_t* name)
289 {
290  // Return detector in structure with given name.
291  // If not found in this structure, we look in all daughter structures
292  // for a detector with the name.
293 
294  KVDetector* det = (KVDetector*)GetDetectors()->FindObject(name);
295  if (!det) {
296  TIter next(GetStructures());
297  KVGeoStrucElement* elem;
298  while ((elem = (KVGeoStrucElement*)next())) {
299  det = elem->GetDetectorAny(name);
300  if (det) return det;
301  }
302  }
303  return (KVDetector*)nullptr;
304

313 Bool_t KVGeoStrucElement::Fired(Option_t* opt) const
314 {
315  // Returns kTRUE if any detector or structure element in this structure

317

324  return kFALSE;

333 KVGeoStrucElement* KVGeoStrucElement::GetParentStructure(const Char_t* type, const Char_t* name) const
334 {
335  // Get parent geometry structure element of given type.
336  // Give unique name of structure if more than one element of same type is possible.
337  KVGeoStrucElement* el = 0;
338  if (strcmp(name, "")) {
339  KVSeqCollection* strucs = fParentStrucList.GetSubListWithType(type);
340  el = (KVGeoStrucElement*)strucs->FindObject(name);
341  delete strucs;
342  } else
343  el = (KVGeoStrucElement*)fParentStrucList.FindObjectByType(type);
344  return el;

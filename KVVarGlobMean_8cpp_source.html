16 // ***** Base class for Global variables with ******
17 // a mean value and a standard deviation
18 //
19 // This class is a base abstact class for the management of global variable with a mean value and a standard
20 //deviation. A global variable class deriving from KVVarGlobmean has to provide the following methods:
21 // virtual void Init(void) initiates the internal fields
22 // virtual void Reset(void) resets internal fields before the treatment
23 // virtual void Fill(KVNucleus *c) tells how to compute the global variable using a KVNucleus
24 //
25 // A call to the GetValue() method returns the mean value of the variable. One can also obtain the array which
26 // contains the values of the internal fields by a call to GetValuePtr(). Individual values can also be
27 // obtained by a call to GetValue(Int_t index) or by a call to GetValue(Chat_t *name). Parenthesis can also be
28 // used. The correspondance between indexes ans names is as follows:
29 //
30 // Name Index Meaning
31 //----------------------------------------------------------------
32 // Mean 0 (default) Mean value of var = SumVar/SumOfWeights
33 // RMS 1 standard deviation of var = sqrt(SumVarSquared/SumOfWeights-Mean*Mean)
34 // SumVar 2 Sum of weighted var values
35 // SumVarSquared 3 Sum of weighted squared var values
36 // SumOfWeights 4 Sum of weights
37 // Min 5 Minimum value
38 // Max 6 Maximum value
39 //
40 //
41 // Here are examples on how obtaining values.
42 //
43 // KVZmean *pzmean=new KVZmean();
44 // KVZmean zmean;
45 // ...
46 // Double_t mean=pzmean->GetValue(); |
47 // mean=pzmean->GetValue("Mean"); |---> Mean value
48 // mean=zmean(); |
49 // mean=zmean("Mean"); |
50 // Double_t ect=zmean("RMS");---------------> Standard Deviation
51 // Double_t sw=pzmean->GetValue(4) ---------> Sum of weights
52 // Double_t *pvalues=zmean.GetValuePtr();---> Array of values
53 //
54 // The internal fields can be set by a call to the protected methods FillVar(Double_t v, Double_t w) or
55 // FillVar(Double_t v). In the latter case, the weight is set to 1.
56 //
57 // Method Action
58 //----------------------------------------------------------------
59 // FillVar(Double_t v, Double_t w) SumVar+=w*v;
60 // SumVarSquared+=w*v*v;
61 // SumOfWeights+=w;
62 //
63 // FillVar(Double_t v) SumVar+=v;
64 // SumVarSquared+=v*v;
65 // SumOfWeights+=1;
66 //
67 // Mean value and standard deviation are computed by the protected CalcVar() method.
68 // Have a look to the KVZmean class for an example of a global variable deriving from KVVarGlobMean.
69 // Look at KVVarGlob class to have an example of use.
70 //

170 void KVVarGlobMean::FillVar(Double_t v, Double_t w)
171 {
172 // Routine privee de remplissage
173 //
174  svar += w * v;
175  svar2 += w * v * v;
176  sw += w;

188 void KVVarGlobMean::CalcVar(void)
189 {
190 // Routine de calcul des valeurs moyennes
191 //
192  if (!calc) {
193  if (sw != 0) {
194  var = svar / sw;
195  if (svar2 / sw >= var * var) {
196  ect = TMath::Sqrt(svar2 / sw - var * var);
197  } else {
198  ect = -1.;
199  }
200  } else {
201  var = 0.;
202  ect = -1.;
203  }
204  calc = 1;
205  }
206 }

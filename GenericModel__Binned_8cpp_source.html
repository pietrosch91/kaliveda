411  void GenericModel_Binned::ImportModelData(Int_t parameter_num, RooArgList* plist)
412  {
413  //Initial RooWorksapce already given by the user for the fit, no save or importation
414  if (IsWorkspaceImported() == kTRUE) {
415  Info("ImportModelData", "RooWorspace for fit already provided, no importation needed !!!");
416  return;
417  }
418 
419  // Import all model datasets corresponding to the defined parameters, ranges & binnings
420  RooArgList PLIST;
421  if (plist) PLIST.addClone(*plist);
422  RooRealVar* par = GetParameter(parameter_num);
423  RooAbsBinning& bins = par->getBinning();
424  Int_t N = bins.numBins();
425  RooRealVar* par_in_list = (RooRealVar*)PLIST.find(par->GetName());
426 
427  if (!par_in_list) {
428  PLIST.addClone(RooRealVar(par->GetName(), par->GetTitle(), 0.));
429  par_in_list = (RooRealVar*)PLIST.find(par->GetName());
430  }
431 
432  for (int i = 0; i < N; i++) {
433  par_in_list->setMax(bins.binHigh(i));
434  par_in_list->setMin(bins.binLow(i));
435  par_in_list->setVal(bins.binCenter(i));
436 
437  if ((parameter_num + 1) < GetNumberOfParameters()) ImportModelData(parameter_num + 1, &PLIST);
438 
439  else AddModelData(PLIST, GetModelDataHist(PLIST));
440  }

565  void GenericModel_Binned::ConstructPseudoPDF(Bool_t debug)
566  {
567  //Build a parameterised pseudo-PDF from all the imported model datasets.
568  //Each dataset will be transformed into a RooHistPdf.
569  //
570  //The user can decide to initialize the parameters values using a vector<const Double_t>*
571  //For an extended fit the user need to declare the experimental DataSet entries
572  //If debug=kTRUE, each dataset and kernel PDF will be displayed in a canvas.
573  //If numint=kTRUE the integrals are force numerical
574 
575 
576  //-------------Control if the user gave the experimental RooDataHist to fit------------
577  if (IsExpDataHistSet() == kFALSE) {
578  Error("ConstructPseudoPDF", "... !!! No Experimental RooDataHist for the fit !!! Import experimental RooDataHist to fit with SetExperimentalDataHist() first...");
579  return;
580  }
581 
582  //-------------Control if DataSets list is given------------
583  if (!GetNumberOfDataSets() || GetNumberOfDataSets() == 0) {
584  if (IsWorkspaceImported() == kFALSE) Error("ConstructPseudoPDF", "... !!! No DataSets !!! Import model datasets with ImportModelData() first...");
585  else Error("ConstructPseudoPDF", "... !!! No DataSets !!! Verify provided worspace...");
586 
587  return;
588  }
589 
590  //-------------For the weights initialization--------------
591  if (fInitWeights == 0) Error("ConstructPseudoPDF", "Initial guess of parameters not given... use SetInitWeights() or SetUniformWeights() first...");
592 
593 
594  if (fModelPseudoPDF) {
595  delete fModelPseudoPDF;
596  fModelPseudoPDF = 0;
597  //fHistPdfs.Delete();
598  fFractions.removeAll();
599  }
600 
601  //-----------generate kernels------------
602  RooArgList kernels;
603 
604  //number of the last set added in kernels
605  //used for not extended pdf to remove last coef
606  Int_t last = -1;
607 
608  for (int i = 0; i < GetNumberOfDataSets(); i++) {
609  const RooDataHist* set = GetDataHist(i);
610 
611  //If stat in the DataSet
612  //Else error : RooFit can't create the RooHistPdf without statistic
613  if (set->sumEntries() > 0) {
614  ++fNDataSets; //For modifying counting if no stat
615  last = i; //For extended fit
616 
617  RooHistPdf* pdf = new RooHistPdf(Form("HistPdf%d", i), Form("RooHistPdf from datahist#%d", i), GetObservables(), *set, fSmoothing);
618  pdf->forceNumInt(IsNumInt());
619 
620  fHistPdfs.Add(pdf);
621  kernels.add(*pdf);
622 
623  if (debug) Info("ConstructPseudoPDF", "...datahist#%d... added into kernel list (entries=%d)...", i, (int) set->sumEntries());
624 
625  //Extended pdf
626  if (IsExtended() == kTRUE) {
627  RooRealVar pp(Form("W%d", i), Form("fractional weight of kernel-PDF #%d for extended fit", i), (*fInitWeights)[i], 0., 2.*fexpdatahist_counts);
628  fFractions.addClone(pp);
629  }
630 
631  //Not-Extended pdf
632  else {
633  RooRealVar pp(Form("W%d", i), Form("fractional weight of kernel-PDF #%d for not-extended fit", i), (*fInitWeights)[i], 0., 1.);
634  fFractions.addClone(pp);
635  }
636  }
637 
638 
639  else {
640  //Remove this parameters set from _datasetparams used to save the results
641  GetDataSetParametersList()->RemoveAt(i);
642  Info("ConstructPseudoPDF", "...datahist#%d... refused (nentries=%d)...", i, (int) set->sumEntries());
643  Info("ConstructPseudoPDF", "...RooArgList%d... removed from _datasetparams for the results...", i);
644  }
645  }
646 
647 
648  //-------------Remove last coefficient for not-extended fit-----------
649  if (IsExtended() == kFALSE) {
650  RooRealVar* p = (RooRealVar*) fFractions.find(Form("W%d", last));
651 
652  if (p) {
653  fFractions.remove(*p);
654  Info("ConstructPseudoPdf", "...last dataset (dataset#%d) removed for not-extended fit...", last);
655  }
656 
657  else Error("ConstructPseudoPdf", "...error in finding the last dataset to remove for not-extended fit...");
658  }
659 
660  //Extended
661  if (IsExtended() == kTRUE) {
662  fModelPseudoPDF = new NewRooAddPdf("Model", "Pseudo-PDF constructed from kernels for model datahists", kernels, fFractions, kFALSE);
663  fModelPseudoPDF->forceNumInt(IsNumInt());
664  }
665 
666  else {
667  //fFractions->remove();
668  fModelPseudoPDF = new NewRooAddPdf("Model", "Pseudo-PDF constructed from kernels for model datahists", kernels, fFractions, kTRUE);
669  fModelPseudoPDF->forceNumInt(IsNumInt());
670  }
671 
672 // if(save==kTRUE)
673 // {
674 // SavePseudoPDF(GetWorkspaceFileName());
675 // }
676  }

684  RooFitResult* GenericModel_Binned::fitTo(RooDataHist& data, const RooCmdArg& arg1, const RooCmdArg& arg2, const RooCmdArg& arg3, const RooCmdArg& arg4,
685  const RooCmdArg& arg5, const RooCmdArg& arg6, const RooCmdArg& arg7, const RooCmdArg& arg8,
686  const RooCmdArg& arg9, const RooCmdArg& arg10, const RooCmdArg& arg11, const RooCmdArg& arg12)
687  {
688  //Print informations
689  Info("fiTo...", "performing fit with %d parameters and %d observables", GetNumberOfParameters(), GetNumberOfObservables());
690  for (Int_t ii = 0; ii < GetNumberOfParameters(); ii++) {
691  RooRealVar* p = dynamic_cast<RooRealVar*>(GetParameters().at(ii));
692  RooAbsBinning& bins = p->getBinning();
693  Int_t N = bins.numBins();
694  Info("fiTo...", "parameter%d: name=%s, min=%e, max=%e, nbin=%d", ii, p->GetName(), p->getMin(), p->getMax(), N);
695  }
696  for (Int_t ii = 0; ii < GetNumberOfObservables(); ii++) {
697  RooRealVar* p = dynamic_cast<RooRealVar*>(GetObservables().at(ii));
698  RooAbsBinning& bins = p->getBinning();
699  Int_t N = bins.numBins();
700  Info("fiTo...", "observable%d: name=%s, min=%e, max=%e, nbin=%d", ii, p->GetName(), p->getMin(), p->getMax(), N);
701  }
702 
703  //Fit
704  fLastFit = fModelPseudoPDF->improvedFitTo(data, arg1, arg2, arg3, arg4, arg5, arg6, arg7, arg8, arg9, arg10, arg11, arg12);
705 
706  //Save Coefs
707  SafeDelete(fParamDataHist);
708  fParamDataHist = new RooDataHist("params", "params", GetParameters());
709 
710  fWeights.removeAll();
711  const RooArgList& coefs = fModelPseudoPDF->coefList();
712  for (int i = 0; i < fNDataSets; i++) {
713  RooAbsReal* coef = (RooAbsReal*)coefs.at(i);
714  RooRealVar w(Form("w%d", i), Form("Fitted weight of kernel#%d", i), coef->getVal());
715  printf("coef%d=%e\n", i, coef->getVal());
716 
717  if (coef->InheritsFrom(RooRealVar::Class())) {
718  w.setError(((RooRealVar*)coef)->getError());
719  } else {
720  w.setError(coef->getPropagatedError(*fLastFit));
721  }
722 
723  fWeights.addClone(w);
724  fParamDataHist->set(*GetParametersForDataset(i), w.getVal(), w.getError());
725  }
726 
727  SafeDelete(fParameterPDF);
728  fParameterPDF = new RooHistPdf("paramPDF", "paramPDF", GetParameters(), *fParamDataHist);
729 
730  return fLastFit;
731  }

24 //KVTGID
25 //
26 //Abstract base class for particle identfication using functionals developed by
27 //L. Tassan-Got (IPN Orsay). These functionals are defined in the KVTGIDFunctions
28 //namespace, and the functional used by each KVTGID object is defined by
29 //giving its name to the constructor. These objects are persistent, i.e. can be
30 //retrieved from a ROOT file and used for identification.
31 //
32 //GetIdentification() method performs an identification. Status code can be retrieved
33 //afterwards using GetStatus();
34 //
35 //A KVIDGrid identification grid can be generated from the functional in order
36 //to visualise the corresponding identification lines. (MakeIDGrid)
37 //
38 //The following methods must be defined by child-classes:
39 //
40 // SetIdent() -- determine how to set the identification for each line in the grid
41 // AddLine() -- define the type of IDLine added to the grid
42 // NewGrid() -- define the type of IDGrid to generate
43 //

234 Double_t KVTGID::GetIdentification(Double_t ID_min, Double_t ID_max,
235  Double_t& ID_quality, Double_t* par)
236 {
237  //Use the functional with the current parameter values in order to perform
238  //an identification.
239  //The value returned is the estimated identification value.
240  //If the identification is not possible, -1 is returned
241  //(ID_quality=-1 also).
242  //
243  //For status code use GetStatus().
244  //
245  //ID_min and ID_max are the lower & upper limits for the identification.
246  //The functional must change sign between these two limits in order for
247  //the identification to be possible.
248  //ID_quality is the (absolute) value of the functional corresponding to
249  //the estimated identification: it is the distance from the identified
250  //point to the nearest identification line. Ideally it should be zero or
251  //as small as possible.
252  //
253  //Optional argument "par" allows to replace the current parameters.
254 
255  if (par)
256  SetParameters(par);
257 
258  Double_t ID;
259  ID = ID_quality = -1.;
260  fStatus = kStatus_NotBetween_IDMin_IDMax;
261 
262  if (Eval(ID_min) * Eval(ID_max) < 0.) {
263  //if the sign of the identification function changes between ID_min
264  //and ID_max i.e. if the ID_min line is below the point and ID_max above
265  //it this means that the point to identify is in between these two lines
266  ID = GetX(0., ID_min, ID_max);
267  //what is actual value of Tassan-Got formula for this ID ?
268  ID_quality = TMath::Abs(Eval(ID));
269  fStatus = kStatus_OK;
270  }
271 
272  return ID;
273 }

287 void KVTGID::AddLineToGrid(KVIDGrid* g, Int_t ID, Int_t npoints,
288  Double_t xmin, Double_t xmax, Bool_t log_scale)
289 {
290  // Add a line to the grid 'g' for identification label 'ID' with 'npoints' points calculated between
291  // X-coordinates xmin and xmax. Points are omitted if the resulting value of the functional is
292  // not a number (TMath::IsNan = kTRUE).
293  //
294  // If log_scale=kTRUE (default is kFALSE), more points are used at the beginning
295  // of the line than at the end, with a logarithmic dependence.
296 
297  //add new line to grid
298  KVIDLine* new_line = AddLine(g);
299 
300  //set identification label for line
301  SetIdent(new_line, ID);
302  //set mass formula for line
303  new_line->SetMassFormula(fMassFormula);
304 
305  //loop over points of line
306  Int_t p_index = 0;
307  Double_t X, dX;
308  Double_t Y = 0.;
309  Double_t logXmin = TMath::Log(TMath::Max(xmin, 1.0));
310  if (log_scale)
311  dX = (TMath::Log(xmax) - logXmin) / (npoints - 1.);
312  else
313  dX = (xmax - xmin) / (Double_t)(npoints - 1);
314 
315  for (Int_t i = 0; i < npoints; i++) {
316 
317  //set x coordinate of this point
318  if (log_scale)
319  X = TMath::Exp(logXmin + i * dX);
320  else
321  X = xmin + dX * ((Double_t) i);
322 
323  //leave value Y as it is. The value of GetIDFunc()->Eval(ID)
324  //is the vertical distance delta_Y from point (X,Y) to the line; therefore the
325  //Y coordinate of the point on the line is Y + delta_Y, whatever the value Y.
326 
327  //set values of parameters which correspond to X and Y coordinates in grid
328  SetParameter("X", X);
329  SetParameter("Y", Y);
330 
331  Y += Eval((Double_t) ID);
332 
333  if (!TMath::IsNaN(Y))
334  new_line->SetPoint(p_index++, X, Y);
335  else
336  Y = 0.; //reset Y to 0 if it ever becomes NaN
337  }
338 }

474 KVTGID* KVTGID::MakeTGID(const Char_t* name, Int_t type, Int_t light, Int_t ZorA, Int_t mass)
475 {
476  // Static function used to create TGID objects
477  //
478  // type -----> type of functional
479  // * type : =0->basic functional <>0->extended functional
480  // * * For the basic formula :
481  // * yy = ((g*E)**(mu+1)+lambda**(mu+1)*Z**2*A**mu)**(1/(mu+1))-g*E + pdy
482  // * * For the extended formula :
483  // * yy = ((g*E)**(mu+nu+1)+(lambda*Z**alpha*A**beta)**(mu+nu+1)+
484  // * xi*A**mu*(g*E)**nu)**(1/(mu+mu+1))-g*E + pdy
485  //
486  // light -----> treatment of CsI total light output
487  // * light : =0->no non-linear light response <>0->non-linear light response included
488  // * * If ih=0 no non-linear light response : E=xx-pdx
489  // * * If ih<>0 non-linear light response included :
490  // * E = sqrt(h**2+2*rho*h*(1+log(1+h/rho)))
491  // * rho=eta*Z**2*A and h=xx-pdx
492  //
493  // ZorA -----> functional used to find Z or A
494  // * ZorA : =0->A <>0->Z
495 
496  Int_t npar = 6 + GetNumberOfLTGParameters(type, light);
497  KVTGID* _tgid_ = 0;
498  if (ZorA) {
499  _tgid_ = new KVTGIDZ(name, npar, type, light, mass);
500  } else {
501  _tgid_ = new KVTGIDZA(name, npar, type, light);
502  }
503  npar = 6;
504  _tgid_->fLambda = npar++;
505  if (type) {
506  _tgid_->fAlpha = npar++;
507  _tgid_->fBeta = npar++;
508  _tgid_->fMu = npar++;
509  _tgid_->fNu = npar++;
510  _tgid_->fXi = npar++;
511  } else {
512  _tgid_->fMu = npar++;
513  }
514  _tgid_->fG = npar++;
515  _tgid_->fPdx = npar++;
516  _tgid_->fPdy = npar++;
517  if (light) _tgid_->fEta = npar++;
518  return _tgid_;
519 }

743 Int_t KVTGID::GetNumberOfLTGParameters(Int_t type, Int_t light)
744 {
745  // Static function returning number of parameters used by Tassan-Got functional.
746  // Depends on type of functional (type) and whether non-linear light response
747  // is calculated or not (light).
748  // According to the (type, light) combination the numbers and orders
749  // of parameters are :
750  // type=0 light=0 5 parameters: lambda, mu, g, pdx, pdy
751  // type=0 light<>0 6 parameters: lambda, mu, g, pdx, pdy, eta
752  // type<>0 light=0 9 parameters: lambda, alpha, beta, mu, nu,
753  // xi, g, pdx, pdy
754  // type<>0 light<>0 10 parameters: lambda, alpha, beta, mu, nu,
755  // xi, g, pdx, pdy, eta}
756 
757  return type ? (light ? 10 : 9) : (light ? 6 : 5);
758 }

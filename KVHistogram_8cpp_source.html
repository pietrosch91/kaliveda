122 void KVHistogram::ParseHistoTitle(const Char_t* title, KVString& exp, KVString& sel, KVString& weight)
123 {
124  // Take histo title "VAREXP [WEIGHT] { SELECTION }"
125  // and separate the three components
126 
127  exp = "";
128  sel = "";
129  weight = "1";
130  TString tmp(title);
131  Int_t ss = tmp.Index("{");
132  if (ss > 0) {
133  Int_t se = tmp.Index("}");
134  sel = tmp(ss + 1, se - ss - 1);
135  sel.Remove(TString::kBoth, ' ');
136  exp = tmp(0, ss);
137  exp.Remove(TString::kBoth, ' ');
138  } else {
139  exp = tmp;
140  exp.Remove(TString::kBoth, ' ');
141  }
142  // get weight from expression string if any
143  KVString tmp1(exp);
144  tmp1.Begin(" ");
145  exp = tmp1.Next();
146  if (!tmp1.End()) {
147  weight = tmp1.Next();
148  weight.Remove(TString::kBoth, '[');
149  weight.Remove(TString::kBoth, ']');
150  }
151 }

158 void KVHistogram::ParseExpressionString(const Char_t* exp, KVString& varX, KVString& varY,
159  KVString& varZ)
160 {
161  // Parse expression strings "VARZ:VARY:VARX" or "VARY:VARX" or "VARX"
162 
163  KVString tmp(exp);
164  tmp.Remove(TString::kBoth, ' ');
165  // hide any "::" scope resolution operators (e.g. TMath::DegToRad)
166  tmp.ReplaceAll("::", "_scope_");
167  Int_t nvar = tmp.CountChar(':');
168  tmp.Begin(":");
169  varX = varY = varZ = "";
170  if (nvar == 2) {
171  varZ = tmp.Next();
172  varZ.ReplaceAll("_scope_", "::");
173  }
174  if (nvar >= 1) {
175  varY = tmp.Next();
176  varY.ReplaceAll("_scope_", "::");
177  }
178  varX = tmp.Next();
179  varX.ReplaceAll("_scope_", "::");
180 }

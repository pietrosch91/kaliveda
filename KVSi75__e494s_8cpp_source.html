70 Double_t KVSi75_e494s::GetVoltsFromCanalPG(Double_t chan)
71 {
72  //Return calibrated detector signal in Volts calculated from PG channel number.
73  //If "chan" is not given, the value of the "PG" acquisition parameter read from
74  //data for this detector is used to calculate the signal.
75  //If the PG parameter is not present (=-1) or no calib we return 0.
76  //Change in the coder pedestal between the current run and the reference pedestal
77  //is taken into account, as well as the shift between the pedestal of the
78  //calibration run (533) and the reference run (293).
79  // To get correct value for Cf alphas, we must add 7 channels, not 3-4
80 
81  if (!fChVoltPG || !fChVoltPG->GetStatus())
82  return 0;
83 
84  if (!chan) {
85  chan = GetPG();
86  }
87  if (chan < -0.5)
88  return 0.; //PG parameter absent
89  //correct for pedestal drift
90  chan = chan - (Float_t) GetACQParam("PG")->GetDeltaPedestal() + 7.0;
91  return (fChVoltPG->Compute(chan));
92 }

103 Int_t KVSi75_e494s::GetCanalPGFromVolts(Float_t volts)
104 {
105  //Return raw PG channel number corresponding to a given detector signal in volts
106  //
107  //Any change in the coder pedestal between the current run and the effective pedestal
108  //of the channel-volt calibration (GetCanal(V=0)) is automatically corrected for.
109  //
110  //Returns -1 if PG <-> Volts calibration is not available
111 
112 
113  if (!fChVoltPG || !fChVoltPG->GetStatus())
114  return -1;
115  Int_t chan = TMath::Nint(fChVoltPG->Invert(volts) +
116  (Float_t) GetACQParam("PG")->GetDeltaPedestal() - 3.5);
117  return chan;

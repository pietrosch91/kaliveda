246 TH1* KVHistoManipulator::ScaleHisto(TH1* hh, TF1* fx, TF1* fy, Int_t nx, Int_t ny, Double_t xmin, Double_t xmax, Double_t ymin, Double_t ymax, Option_t* norm)
247 {
248 //-------------------------------------------------
249  // Applique les transformations correspondantes aux fonctions (TF1 *) donnees en argument
250  // et donne en sortie l histogramme transforme celui ci a pour nom "nom_de_histo_input"_scaled
251  //
252  // IMPORTANT l'histo d'entree n'est pas modifie
253  //
254  // pour TH1 et TProfile n'est pris en compte que TF1 *fx;
255  // les parametres de la fonction doivent etre initialises avant.
256  // Si la fonction est un pointeur NULL, aucune transformation n est appliquee et l axe reste tel quel.
257  // L'intervalle de validite des fonctions TF1::SetRange est determine a partir des limites de l histo apres transformations
258  // nx xmin et xmax sont les parametres de binnage des histo
259  // si nx=-1, ceux ci sont recalcules automatiquement nx = nx_initial, xmin = fx(xmin), xmax = fx(xmax)
260  // si nx!=-1, xmin et xmax doivent etre aussi specifies
261  // il en est de meme pour l'axe y
262  //
263  // OPTION: norm
264  // norm = "" (default) : no adjustment is made for the change in bin width due to the transformation
265  // norm = "width" : bin contents are adjusted for width change, so that the integral of the histogram
266  // contents taking into account the bin width (i.e. TH1::Integral("width")) is the same.
267  //
268  // NOTE for 1D histos:
269  // If the binning of the scaled histogram is imposed by the user (nx!=-1), then in order
270  // to achieve a continuous scaled distribution we have to randomize X within each bin.
271  // This will only work if the bin contents of 'hh' are integer numbers, i.e. unnormalised raw data histogram.
272 
273  TRandom3* alea = new TRandom3(); //generateur de nbre aleatoire
274  Bool_t width = !strcmp(norm, "width");
275  TH1* gg = NULL;
276  Double_t abs;
277  Bool_t fixed_bins = (nx != -1);
278  if (fx) {
279  if (nx == -1) {
280  nx = hh->GetNbinsX();
281  abs = hh->GetXaxis()->GetBinLowEdge(1);
282  xmin = fx->Eval(abs);
283  abs = hh->GetXaxis()->GetBinUpEdge(hh->GetNbinsX());
284  xmax = fx->Eval(abs);
285  }
286  fx->SetRange(hh->GetXaxis()->GetBinLowEdge(1), hh->GetXaxis()->GetBinUpEdge(nx));
287  fx->SetNpx(nx + 1);
288  } else {
289  nx = hh->GetNbinsX();
290  xmin = hh->GetXaxis()->GetBinLowEdge(1);
291  xmax = hh->GetXaxis()->GetBinUpEdge(hh->GetNbinsX());
292  }
293 
294  if (hh->InheritsFrom("TH2")) {
295  if (fy) {
296  if (ny == -1) {
297  ny = hh->GetNbinsY();
298  abs = hh->GetYaxis()->GetBinLowEdge(1);
299  ymin = fy->Eval(abs);
300  abs = hh->GetYaxis()->GetBinUpEdge(hh->GetNbinsY());
301  ymax = fy->Eval(abs);
302  }
303  fy->SetRange(hh->GetYaxis()->GetBinLowEdge(1), hh->GetYaxis()->GetBinUpEdge(ny));
304  fy->SetNpx(ny + 1);
305  } else {
306  ny = hh->GetNbinsY();
307  ymin = hh->GetYaxis()->GetBinLowEdge(1);
308  ymax = hh->GetYaxis()->GetBinUpEdge(hh->GetNbinsY());
309  }
310  }
311 
312  TClass* clas = TClass::GetClass(hh->ClassName());
313  gg = (TH1*) clas->New();
314  if (!gg) return NULL;
315  TString hname;
316  hname.Form("%s_scaled", hh->GetName());
317  gg->SetNameTitle(hname.Data(), hh->GetTitle());
318 
319  if (hh->InheritsFrom("TH2")) gg->SetBins(nx, xmin, xmax, ny, ymin, ymax);
320  else gg->SetBins(nx, xmin, xmax);
321 
322  // if option norm="width', take account of change of X bin width between original and scaled histograms
323  Double_t Xbin_width_corr = 1.0, Ybin_width_corr = 1.0;
324  if (width) {
325  Double_t orig_Xbin_width = (hh->GetXaxis()->GetXmax() - hh->GetXaxis()->GetXmin()) / hh->GetNbinsX();
326  Double_t new_Xbin_width = (gg->GetXaxis()->GetXmax() - gg->GetXaxis()->GetXmin()) / gg->GetNbinsX();
327  Xbin_width_corr = orig_Xbin_width / new_Xbin_width;
328  if (hh->InheritsFrom("TH2")) {
329  // Take account of change of X bin width between original and scaled histograms
330  Double_t orig_Ybin_width = (hh->GetYaxis()->GetXmax() - hh->GetYaxis()->GetXmin()) / hh->GetNbinsY();
331  Double_t new_Ybin_width = (gg->GetYaxis()->GetXmax() - gg->GetYaxis()->GetXmin()) / gg->GetNbinsY();
332  Ybin_width_corr = orig_Ybin_width / new_Ybin_width;
333  }
334  }
335 
336  for (Int_t xx = 1; xx <= hh->GetNbinsX(); xx += 1) {
337  Double_t bmin = hh->GetXaxis()->GetBinLowEdge(xx);
338  Double_t bmax = hh->GetXaxis()->GetBinUpEdge(xx);
339  abs = alea->Uniform(bmin, bmax);
340  if (abs == bmax) abs = bmin;
341  Double_t resx = abs;
342  if (fx) resx = fx->Eval(abs);
343  if (hh->InheritsFrom("TH2")) {
344  for (Int_t yy = 1; yy <= hh->GetNbinsY(); yy += 1) {
345  if (hh->GetBinContent(xx, yy) > 0) {
346  bmin = hh->GetYaxis()->GetBinLowEdge(yy);
347  bmax = hh->GetYaxis()->GetBinUpEdge(yy);
348  abs = alea->Uniform(bmin, bmax);
349  if (abs == bmax) abs = bmin;
350  Double_t resy = abs;
351  if (fy) resy = fy->Eval(abs);
352  gg->SetBinContent(gg->GetXaxis()->FindBin(resx),
353  gg->GetYaxis()->FindBin(resy),
354  hh->GetBinContent(xx, yy)*Xbin_width_corr * Ybin_width_corr);
355  //gg->SetBinError(gg->GetXaxis()->FindBin(resx),gg->GetYaxis()->FindBin(resy),hh->GetBinError(xx,yy));
356  }
357  }
358  } else {
359  // 1-D histogram
360  if (fixed_bins) {
361  // histogram binning imposed by user. we need to randomise inside bins of original histogram
362  // otherwise scaled histogram will be discontinuously filled.
363  Int_t nmax = (Int_t)hh->GetBinContent(xx);
364  for (int i = 0; i < nmax; i++) {
365  abs = alea->Uniform(bmin, bmax);
366  Double_t resx = fx->Eval(abs);
367  gg->Fill(resx, Xbin_width_corr);
368  //cout << resx << " " << Xbin_width_corr << endl;
369  }
370  } else {
371  // range and binning calculated from function.
372  Double_t resy = hh->GetBinContent(xx);
373  if (fy) resy = fy->Eval(resy);
374  gg->SetBinContent(gg->GetXaxis()->FindBin(resx), resy * Xbin_width_corr);
375  }
376  }
377  }
378 
379  delete alea;
380  return gg;
381 
382 }

401  // Applique les transformations correspondantes aux fonctions (TF1 *) donnees en argument
402  // et donne en sortie le graph transforme celui ci a pour nom "nom_de_histo_input"_scaled
403  //
404  // IMPORTANT le graph d'entree n'est pas modifie
405  //
406  // les parametres de la fonction doivent etre initialises avant.
407  // Si la fonction est un pointeur NULL, aucune transformation n est appliquee et l axe reste tel quel.

411  gg = (TGraph*) clas->New();
412  if (!gg) return NULL;
413  TString hname;
414  hname.Form("%s_scaled", hh->GetName());
415  gg->SetNameTitle(hname.Data(), hh->GetTitle());
416 
417  Int_t np = hh->GetN();
418  for (Int_t nn = 0; nn < np; nn += 1) {
419  Double_t xx1, yy1;
420  hh->GetPoint(nn, xx1, yy1);
421  Double_t xx2 = xx1;
422  if (fx) xx2 = fx->Eval(xx1);
423  Double_t yy2 = yy1;
424  if (fy) yy2 = fy->Eval(yy1);
425  gg->SetPoint(nn, xx2, yy2);
426  if (gg->InheritsFrom("TGraphErrors")) {
427  // transformation of errors
428  // if f = f(x) the error on f, e_f, for a given error on x, e_x, is

430  Double_t e_x = ((TGraphErrors*)hh)->GetErrorX(nn);
431  Double_t e_y = ((TGraphErrors*)hh)->GetErrorY(nn);
432  if (fx) e_x = TMath::Abs(fx->Derivative(xx1)) * e_x;
433  if (fy) e_y = TMath::Abs(fy->Derivative(yy1)) * e_y;
434  ((TGraphErrors*)gg)->SetPointError(nn, e_x, e_y);
435  }
436  }
437 
438  return gg;

564 TH2* KVHistoManipulator::RenormaliseHisto(TH2* hh, Int_t bmin, Int_t bmax, TString axis, Double_t valref)
565 {
566 //-------------------------------------------------
567  // Renormalisation de l histogramme 2D sous la contrainte d'une ditribution plate suivant X ou Y
568  // (signal de bimodalite par exemple)
569  // et donne en sortie l histogramme transforme celui ci a pour nom "nom_de_histo_input"_normalised
570  //
571  // IMPORTANT l'histo d'entree n'est pas modifie
572  //
573  // bmin, bmax : intervalle de bins ou l'on effectue la renormalisation (par defaut -1,-1 --> toute la plage)
574  // les contenus hors de l intervalle [bmin,bmax] sont remis a zero
575  // Si on veut une distribution plate en X ils indiquent l'intervalle sur cet axe x
576  // valref : valeur en stat de la distribution plate (par defaut 1)
577  //
578  // ATTENTION la propagation des erreurs n est pas (pour l instant) implementee
579 
580  if (!hh) {
581  cout << "pointeur histogramme nul" << endl;
582  return NULL;
583  }
584 
585  if (bmin == -1) bmin = 1;
586  TString hname;
587  hname.Form("%s_normalised", hh->GetName());
588  TH2* clone = 0;
589  if ((clone = (TH2*)gDirectory->Get(hname.Data()))) delete clone;
590  clone = (TH2*)hh->Clone(hname.Data());
591  clone->Reset();
592 
593  if (axis == "X") {
594  if (bmax == -1) {
595  bmax = hh->GetNbinsX();
596  }
597  for (Int_t nx = bmin; nx <= bmax; nx += 1) {
598  Double_t integ = 0;
599  for (Int_t ny = 1; ny <= hh->GetNbinsY(); ny += 1) integ += hh->GetBinContent(nx, ny);
600  if (integ > 0) {
601  for (Int_t ny = 1; ny <= hh->GetNbinsY(); ny += 1) {
602  clone->SetBinContent(nx, ny, hh->GetBinContent(nx, ny)*valref / integ);
603  if (hh->GetBinContent(nx, ny) > 0) {
604  Double_t erreur = clone->GetBinContent(nx, ny) * TMath::Sqrt(1. / hh->GetBinContent(nx, ny) + 1. / integ);
605  clone->SetBinError(nx, ny, erreur);
606  }
607  }
608  }
609  }
610  } else if (axis == "Y") {
611  if (bmax == -1) {
612  bmax = hh->GetNbinsY();
613  }
614  for (Int_t ny = bmin; ny <= bmax; ny += 1) {
615  Double_t integ = 0;
616  for (Int_t nx = 1; nx <= hh->GetNbinsX(); nx += 1) integ += hh->GetBinContent(nx, ny);
617  if (integ > 0) {
618  for (Int_t nx = 1; nx <= hh->GetNbinsX(); nx += 1) {
619  clone->SetBinContent(nx, ny, hh->GetBinContent(nx, ny)*valref / integ);
620  Double_t erreur = clone->GetBinContent(nx, ny) * TMath::Sqrt(1. / hh->GetBinContent(nx, ny) + 1. / integ);
621  clone->SetBinError(nx, ny, erreur);
622  }
623  }
624  }
625  } else {
626  cout << "l option TString axis doit etre X ou Y" << endl;
627  }
628  return clone;
629 }

682 TH1* KVHistoManipulator::CumulatedHisto(TH1* hh, TString direction, Int_t bmin, Int_t bmax, Option_t* norm)
683 {
684 //-------------------------------------------------
685  // Cumule le contenu de l histo hh entre bin bmin et bmax et retourne l histo correspondant
686  // Si direction="C" (default):
687  // SetBinContent(n) = GetBinContent(bmin)+GetBinContent(bmin+1)+ ... + GetBinContent(n)
688  // Si direction="D":
689  // SetBinContent(n) = GetBinContent(bmax)+GetBinContent(bmax-1)+ ... + GetBinContent(n)
690  //
691  // Donne en sortie l histogramme transforme celui ci a pour nom "nom_de_histo_input"_cumulated
692  //
693  // IMPORTANT l'histo d'entree n'est pas modifie
694  //
695  // si bmin=-1, bmin=1
696  // si bmax=-1, bmax=GetNbinsX()
697  //
698  // Avec norm = "surf" (default) l'integral de l histo cumule est egale a 1
699  // Avec norm = "max" le contenu de l'histogram est renormalise de facon a ce que le maximum soit 1
700 
701  if (!hh) {
702  cout << "pointeur histogramme nul" << endl;
703  return NULL;
704  }
705  direction.ToUpper();
706  if (direction != "C" && direction != "D") {
707  cout << "l option TString direction doit etre C ou D" << endl;
708  return NULL;
709  }
710  if (hh->GetDimension() == 1) {
711  if (bmin < 1) bmin = 1;
712  if (bmax < 1) bmax = hh->GetNbinsX();
713  TString hname;
714  hname.Form("%s_cumulated", hh->GetName());
715  TH1* clone = (TH1*)hh->Clone(hname.Data());
716  clone->Reset();
717  // get array of cumulated bins
718  Double_t* integral = hh->GetIntegral();
719  if (direction == "C") {
720  /* "standard" cumulative histogram, from bin 'bmin' to bin 'bmax' */
721  Double_t offset = integral[bmin - 1], rescale = 1.0;
722  if (bmax < hh->GetNbinsX() && integral[bmax] > offset) rescale = 1. / (integral[bmax] - offset);
723  for (Int_t nx = 1; nx <= hh->GetNbinsX(); nx += 1) {
724  if (nx >= bmin && nx <= bmax) clone->SetBinContent(nx, rescale * (integral[nx] - offset));
725  else if (nx > bmax) clone->SetBinContent(nx, 1.0);
726  else if (nx < bmin) clone->SetBinContent(nx, 0.0);
727  }
728  } else {
729  /* "reverse" cumulative histogram, from bin 'bmax' to bin 'bmin' */
730  Double_t offset = integral[bmax], rescale = 1.0;
731  if (integral[bmin - 1] < offset) rescale = 1. / (offset - integral[bmin - 1]);
732  for (Int_t nx = 1; nx <= hh->GetNbinsX(); nx += 1) {
733  if (nx >= bmin && nx <= bmax) clone->SetBinContent(nx, rescale * (offset - integral[nx - 1]));
734  else if (nx > bmax) clone->SetBinContent(nx, 0.0);
735  else if (nx < bmin) clone->SetBinContent(nx, 1.0);
736  }
737  }
738  // normalisation
739  if (!strcmp(norm, "surf")) {
740  Double_t sw = clone->GetSumOfWeights();
741  if (sw > 0) clone->Scale(1. / sw);
742  } else if (!strcmp(norm, "max")) {
743  Double_t max = clone->GetMaximum();
744  if (max > 0) clone->Scale(1. / max);
745  }
746  return clone;
747  } else {
748  cout << "cette methode n est pas prevue pour les TH2, TH3" << endl;
749  return NULL;
750  }

771 TH1* KVHistoManipulator::GetDerivative(TH1* hh, Int_t order)
772 {
773 //-------------------------------------------------
774  // retourne la derivee d ordre 0, 1 ou 2 d'un histogramme
775  // 0 -> correspond a un lissage (smooth) de la distribution
776  // 1 et 2 correspondent aux derivees premieres et deuxiemes
777  //
778  // 0 -> derivee zero Yi = 1/35*(-3yi-2 + 12yi-1 +17yi +12yi1 -3yi2)
779  // 1 -> derivee premiere Y'i = 1/12h*(yi-2 - 8yi-1 +8yi1 -1yi2)
780  // 2 -> derivee seconde Y''i = 1/7h/h*(2yi-2 -1 yi-1 -2yi -1yi1 +2yi2)
781  // les derivees pour les bins 1,2 et n-1,n ne sont pas calculees
782  //
783  // IMPORTANT l'histo d'entree n'est pas modifie
784 
785  if (!hh) {
786  cout << "pointeur histogramme nul" << endl;
787  return NULL;
788  }
789  if (!(0 <= order && order <= 2)) {
790  cout << "ordre " << order << "n est pas implemente" << endl;
791  return NULL;
792  }
793  if (hh->GetDimension() == 1) {
794 
795  TString hname;
796  hname.Form("%s_derivated_%d", hh->GetName(), order);
797  TH1* clone = 0;
798 
799  if (hh->InheritsFrom("TProfile")) clone = new TH1F(hname.Data(), hh->GetTitle(), hh->GetNbinsX(), hh->GetBinLowEdge(1), hh->GetBinLowEdge(hh->GetNbinsX() + 1));
800  else clone = (TH1*)hh->Clone(hname.Data());
801 
802  clone->Reset();
803 
804  for (Int_t nx = 3; nx <= hh->GetNbinsX() - 2; nx += 1) {
805  Double_t dev = 0;
806  if (order == 0) {
807  dev = 1. / 35.*(
808  -3 * hh->GetBinContent(nx - 2)

810  + 17 * hh->GetBinContent(nx)

813  );
814  } else if (order == 1) {

817  1 * hh->GetBinContent(nx - 2)
818  - 8 * hh->GetBinContent(nx - 1)
819  + 0 * hh->GetBinContent(nx)
820  + 8 * hh->GetBinContent(nx + 1)
821  - 1 * hh->GetBinContent(nx + 2)
822  );
823  } else {

825  dev = 1 / 7. / h2 * (
826  2 * hh->GetBinContent(nx - 2)
827  - 1 * hh->GetBinContent(nx - 1)
828  - 2 * hh->GetBinContent(nx)
829  - 1 * hh->GetBinContent(nx + 1)
830  + 2 * hh->GetBinContent(nx + 2)
831  );
832  }
833  clone->SetBinContent(nx, dev);
834  }
835  return clone;
836  } else {
837  cout << "cette methode n est pas prevue pour les TH2, TH3" << endl;
838  return NULL;
839  }

1039 {
1040 //-------------------------------------------------
1041  // Calcul le coefficient de correlation entre les variables X et Y
1042  // d'un bidim... Equivalent a TH2F::GetCorrelationFactor() de ROOT
1043  Double_t sumxy = 0;
1044  Double_t sumx = 0, sumx2 = 0;
1045  Double_t sumy = 0, sumy2 = 0;
1046 
1047  Double_t compt = 0;
1048  for (Int_t nx = 1; nx <= hh->GetNbinsX(); nx += 1) {
1049  for (Int_t ny = 1; ny <= hh->GetNbinsY(); ny += 1) {
1050  compt += hh->GetBinContent(nx, ny);
1051  sumxy += hh->GetBinContent(nx, ny) * hh->GetXaxis()->GetBinCenter(nx) * hh->GetYaxis()->GetBinCenter(ny);
1052  sumx += hh->GetBinContent(nx, ny) * hh->GetXaxis()->GetBinCenter(nx);
1053  sumy += hh->GetBinContent(nx, ny) * hh->GetYaxis()->GetBinCenter(ny);
1054  sumx2 += hh->GetBinContent(nx, ny) * pow(hh->GetXaxis()->GetBinCenter(nx), 2.);
1055  sumy2 += hh->GetBinContent(nx, ny) * pow(hh->GetYaxis()->GetBinCenter(ny), 2.);
1056  }
1057  }
1058 
1059  Double_t meanxy = sumxy / compt;
1060  Double_t meanx = sumx / compt;
1061  Double_t meany = sumy / compt;
1062  Double_t meanx2 = sumx2 / compt;
1063  Double_t sigmax = sqrt(meanx2 - pow(meanx, 2.));
1064  Double_t meany2 = sumy2 / compt;
1065  Double_t sigmay = sqrt(meany2 - pow(meany, 2.));
1066 
1067  Double_t rho = (meanxy - meanx * meany) / (sigmax * sigmay);
1068  return rho;

1570 Double_t KVHistoManipulator::GetX(TH1* ob, Double_t val, Double_t eps, Int_t nmax, Double_t xmin, Double_t xmax)
1571 {
1572  // Return value of abscissa X for which the interpolated value
1573  // of the histogram contents is equal to the given value, val.
1574  // We use the false position method (which should always converge...)
1575  // eps is required precision, i.e. convergence condition is that no further change
1576  // in result greater than eps is found.
1577  // nmax = maximum number of iterations
1578  // A solution is searched for X between the limits Xmin and Xmax of the X axis of the histo
1579  // unless arguments (xmin,xmax) are given to bracket the search
1580 
1581  TSpline5* spline = new TSpline5(ob);
1582  Double_t Xmax;
1583  Double_t Xmin;
1584  if (xmin == xmax) {
1585  Xmax = ob->GetXaxis()->GetXmax();
1586  Xmin = ob->GetXaxis()->GetXmin();
1587  } else {
1588  Xmax = xmax;
1589  Xmin = xmin;
1590  }
1591 
1592  Int_t n, side = 0;
1593  Double_t r = 0, fr, fs, s, ft, t;
1594  s = Xmin;
1595  t = Xmax;
1596  fs = spline->Eval(s) - val;
1597  ft = spline->Eval(t) - val;
1598 
1599  for (n = 1; n <= nmax; n++) {
1600  r = (fs * t - ft * s) / (fs - ft);
1601  if (fabs(t - s) < eps * fabs(t + s)) break;
1602  fr = spline->Eval(r) - val;
1603 
1604  if (fr * ft > 0) {
1605  t = r;
1606  ft = fr;
1607  if (side == -1) fs /= 2;
1608  side = -1;
1609  } else if (fs * fr > 0) {
1610  s = r;
1611  fs = fr;
1612  if (side == +1) ft /= 2;
1613  side = +1;
1614  } else break;
1615  }
1616  delete spline;
1617  return r;
1618 }

1661 TF1* KVHistoManipulator::RescaleX(TH1* hist1, TH1* hist2, Int_t degree, Double_t* params,
1662  Int_t npoints, const Char_t* direction, Double_t xmin, Double_t xmax, Double_t qmin, Double_t qmax, Double_t eps)
1663 {
1664  // Find the polynomial transformation of the X-axis of 1-D histogram hist1
1665  // so that the distribution ressembles that of histogram hist2.
1666  // Returns a TF1 which can be used to rescale hist1 (hist1 and hist2 are not modified).
1667  // User's responsibility to delete the TF1.
1668  // After fitting, the array params is filled with:
1669  // params[0] = a0
1670  // params[1] = a1
1671  // ...
1672  // params[degree] = an
1673  // params[degree+1] = Chisquare/NDF
1674  // Therefore params MUST BE at least of dimension (degree+2)
1675  //
1676  // npoints : by default (npoints=-1), we use npoints=degree+2 values of comparison
1677  // of the two cumulative distributions (see below).
1678  // more comparison points can be used by setting npoints>=degree+2.
1679  // direction : by default ("C") we use the cumulative histogram summed from low x to high x.
1680  // if direction="D", we use the cumulative histogram summed from high x to low x.
1681  // xmin, xmax : range of values of abscissa used to build cumulative histograms. default
1682  // (xmin=xmax=-1) include all values.
1683  // qmin, qmax : minimum & maximum values of cumulative histograms used for the
1684  // comparison (see below). by default qmin=0.05, qmax=0.95.
1685  //
1686  // METHOD
1687  // ======
1688  // 'hist1' contains the distribution P1 of variable X1
1689  // 'hist2' contains the distribution P2 of variable X2
1690  // Supposing that we can write X2=f_n(X1), with f_n(x)=a0 + a1*x + a2*x^2 + ... + an*x^n,
1691  // what are the parameters of the polynomial for which P1(f_n(X1))=P2(X2) ?
1692  //
1693  // Consider the cumulative distributions, C1(X1) and C2(X2).
1694  // For npoints=2 we compare the 2 X1 & X2 values for which C1=C2=qmin, qmax
1695  // For npoints=3 we compare the 3 X1 & X2 values for which C1=C2=qmin, qmin+(qmax-qmin)/2, qmax
1696  // For npoints=4 we compare the 4 X1 & X2 values for which C1=C2=qmin, qmin+(qmax-qmin)/3, qmin+2*(qmax-qmin)/3, qmax
1697  // etc. etc.
1698  // In each case we fit the npoints couples (X1,X2) with f_n
1699  //
1700 
1701  int i;
1702  // calculate comparison points
1703  npoints = TMath::Max(npoints, degree + 2);
1704  TString func_name = Form("pol%d", degree);
1705  TF1* fonc = new TF1("f", func_name.Data());
1706  fonc->SetName(Form("RescaleX-%s", func_name.Data()));
1707  RescaleX(hist1, hist2, fonc, npoints, direction, xmin, xmax, qmin, qmax, eps);
1708  for (i = 0; i < degree + 1; i++) {
1709  params[i] = fonc->GetParameter(i);
1710  }
1711  Double_t chisquare = fonc->GetChisquare();
1712  if (fonc->GetNDF() > 0.0) chisquare /= fonc->GetNDF();
1713  params[degree + 1] = chisquare;
1714 
1715  return fonc;
1716 }

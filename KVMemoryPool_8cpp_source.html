50 void* KVMemoryPool::GetMemory(size_t bytes)
51 {
52  // return pointer to memory of size 'bytes'
53  void* p = fFirst->GetMemory(bytes);
54  fLastChunkUsed = fFirst;
55  if (!p) {
56  // search for first available chunk which can provide memory
57  do {
58  fLastChunkUsed = fLastChunkUsed->Next();
59  if (fLastChunkUsed) p = fLastChunkUsed->GetMemory(bytes);
60  } while (!p && fLastChunkUsed);
61  }
62  if (!p) {
63  // there are no chunks big enough to provide memory
64  // add a bigger chunk
65  size_t new_chunk = fChunkSize;
66  while (new_chunk < bytes) new_chunk *= 2;
67  fChunkSize = new_chunk;
68  KVMemoryChunk* chunk = new KVMemoryChunk(fChunkSize);
69  fLast->SetNext(chunk);
70  fLast = chunk;
71  fLastChunkUsed = chunk;
72  p = fLastChunkUsed->GetMemory(bytes);
73  }
74  return p;
75 }

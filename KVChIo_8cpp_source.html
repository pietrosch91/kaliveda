252 Double_t KVChIo::GetELossMylar(UInt_t z, UInt_t a, Double_t egas, Bool_t stopped)
253 {
254  //Based on energy loss in gas, calculates sum of energy losses in mylar windows
255  //from energy loss tables.
256  //If argument 'egas' not given, KVChIo::GetEnergy() is used.
257  //if stopped=kTRUE, we give the correction for a particle which stops in the detector
258  //(by default we assume the particle continues after the detector)
259  //
260  // WARNING: if stopped=kFALSE, and if the residual energy after the detector
261  // is known (i.e. measured in a detector placed after this one), you should
262  // first call
263  // SetEResAfterDetector(Eres);
264  // before calling this method. Otherwise, especially for heavy ions, the
265  // correction may be false for particles which are just above the punch-through energy.
266 
267  egas = ((egas < 0.) ? GetEnergy() : egas);
268  if (egas <= 0.)
269  return 0.0; //check measured (calibrated) energy in gas is reasonable (>0)
270 
271  KVNucleus tmp(z, a);
272  Double_t emylar = GetCorrectedEnergy(&tmp, egas, !stopped) - egas;
273  return emylar;
274 }

346 Double_t KVChIo::GetVoltsFromCanalPG(Double_t chan)
347 {
348  //Return calibrated detector signal in Volts calculated from PG channel number.
349  //If "chan" is not given, the value of the "PG" acquisition parameter read from
350  //data for this detector is used to calculate the signal.
351  //If the PG parameter is not present (=-1) or no calib we return 0.
352  //Any change in the coder pedestal between the current run and the effective pedestal
353  //of the channel-volt calibration (GetCanal(V=0)) is automatically corrected for.
354 
355  if (!fChVoltPG || !fChVoltPG->GetStatus())
356  return 0;
357 
358  if (!chan) {
359  chan = GetPG();
360  }
361  if (chan < -0.5)
362  return 0.; //PG parameter absent
363  //correct for pedestal drift
364  chan = chan - (Double_t) GetPedestal("PG") + fChVoltPG->Invert(0.);
365  return (fChVoltPG->Compute(chan));
366 }

411 Double_t KVChIo::GetVolts()
412 {
413  //Returns Volts for this detector calculated from current PG coder values.
414  //We only use PG, as the two channel-volt calibrations do not coincide and so the passage
415  //from GG to PG produces a discontinuity (unless only GG calibration is available, then
416  //we convert PG to GG and use the GG calibration)
417  //Returns 0 if no calibration available
418 
419  if (fChVoltPG && fChVoltPG->GetStatus()) {
420  return GetVoltsFromCanalPG();
421  } else if (fChVoltGG && fChVoltGG->GetStatus()) {
422  return GetVoltsFromCanalGG(GetGGfromPG());
423  }
424 
425  return 0;
426 }

441 Double_t KVChIo::GetEnergy()
442 {
443  //Redefinition of KVDetector::GetEnergy().
444  //If energy lost in active (gas) layer is already set (e.g. by calculation of energy loss
445  //of charged particles), return its value.
446  //If not, we calculate it and set it using the values read from coders (if fired)
447  //and the channel-volts/volts-energy calibrations, if present
448  //
449  //Returns 0 if (i) no calibration present (ii) calibration present but no data in acquisition parameters
450 
451  //fELoss already set, return its value
452  Double_t ELoss = KVDetector::GetEnergy();
453  if (IsSimMode()) return ELoss; // in simulation mode, return calculated energy loss in active layer
454  if (ELoss > 0) return ELoss;
455  ELoss = GetCalibratedEnergy();
456  if (ELoss < 0) ELoss = 0;
457  SetEnergy(ELoss);
458  return ELoss;
459 }

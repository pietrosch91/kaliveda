70 Double_t KVChannelEnergyBrho::Compute(Double_t chan) const
71 {
72  //Calculate the calibrated energy (in MeV) for a given GG channel number.
73  //
74  //If GG is saturated (>4000: we avoid the top end of the GG),
75  //an "extrapolated" GG is calculated from PG using experimental fits.
76 
77  if (fReady) {
78  Double_t chan_corr;
79  if (chan > 4090.) {
80  chan = ((KVSilicon*) GetDetector())->GetGGfromPG();
81  }
82  chan_corr = chan - GetDetector()->GetPedestal("GG");
83  return (GetParameter(0) + GetParameter(1) * chan_corr);
84  }
85  return 0.;
86 }

123 Double_t KVChannelEnergyBrho::Invert(Double_t energy)
124 {
125  //Given the calibrated (or simulated) energy in MeV,
126  //calculate the corresponding channel number according to the
127  //calibration parameters (useful for filtering simulations).
128 
129  KVDetector* det = GetDetector();
130  Int_t channel = 0;
131 
132  if (fReady) {
133  // linear transfer function
134  channel = (Int_t)(0.5 + (energy - GetParameter(0)) / GetParameter(1)
135  + det->GetPedestal("GG"));
136  } else {
137  Warning("Compute", "Parameters not correctly initialized");
138  }
139  return (Double_t) channel;
140 }

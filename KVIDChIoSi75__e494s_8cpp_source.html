61 Double_t KVIDChIoSi75_e494s::GetIDMapX(Option_t*)
62 {
63  // This method gives the X-coordinate in a 2D identification map
64  // associated with the ChIo-Si75 identification telescope.
65  // The X-coordinate is the Si75 current low gain coder data minus the
66  // low gain pedestal correction (see KVACQParam::GetDeltaPedestal()).
67  // If the high gain coder data is less than 3900 the the low gain value
68  // is calculated from the current high gain coder data minus the high
69  // gain pedestal correction (see KVINDRADetector::GetPGfromGG()).
70 
71  KVINDRADetector* si = (KVINDRADetector*)fsi;
72 
73  if ((-0.5 <= si->GetGG()) && (si->GetGG() <= 3900.5))
74  return si->GetPGfromGG(si->GetGG() - si->GetACQParam("GG")->GetDeltaPedestal());
75 
76  return si->GetPG() - si->GetACQParam("PG")->GetDeltaPedestal();
77 }

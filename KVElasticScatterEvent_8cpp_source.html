33 Definition de la voie d'entrée avec les methodes suivantes :
34  - SetSystem(KVDBSystem* sys);
35  - SetSystem(Int_t zp,Int_t ap,Double_t ekin,Int_t zt,Int_t at);
36  - SetTargNucleus(KVNucleus *nuc);
37  - SetProjNucleus(KVNucleus *nuc);
38  - SetTargetMaterial(KVTarget *targ,Bool_t IsRandomized=kTRUE);
39 La possibilité est donnée d'effectuer des diffusions sur un noyau différent des noyaux de la cible
40 Ex : SetTargetNucleus(new KVNucleus("181Ta"));
41  SetTargetMaterial(new KVTarget("40Ca",1.0)) //cible de Ca de 1 mg/cm2
42  Diffusion sur un noyau de Ta et propagation dans une cible de Ca
43 Si SetTargetMaterial est appelé et pas SetTargNucleus, le noyau cible est débuit du materiel choisi pour la cible
44 
45 Definition du domaine angulaire de diffusion et mode de tirage
46  - DefineAngularRange(TObject*) domaine angulaire deilimité par une structure geometrique (KVPosition etc ...)
47  - DefineAngularRange(Double_t tmin, Double_t tmax, Double_t pmin, Double_t pmax) intervalle en theta et phi (degree)
48  - SetDiffNucleus(KVString name) name="PROJ" ou "TARG" determine a quel noyau projectile ou cible
49  est associe le domaine angulaire choisi
50  - SetRandomOption(Option_t* opt); opt="isotropic" ou "random" permet soit un tirage en theta aleatoire ou isotropique
51 
52 Réalistion des diffusions
53  la méthode Process(Int_t ntimes) permet la réalisation de tous le processus :
54  - propagation du noyau projectile dans la cible jusqu'au point d'intéraction
55  - tirage d'un théta et phi pour la diffusion sur le noyau cible
56  - calcul de la cinématique pour cette direction choisie (réalisé par la classe KV2Body)
57  - si un KVMultiDetArray est défini et que la méthode SetDetectionOn(), détection des projectiles et cibles en voie de sortie
58  - enregistrement des evts diffuses dans un arbre sous forme de KVEvent
59 
60 Exemple :
61 ---------
62

64 //Defintion de la voie d'entree
65 es->SetSystem(54,129,8*129,28,58); 129Xe+58Ni@8MeV/A
66 
67 es->SetRandomOption("isotropic"); tirage en theta isotrope
68 es->DefineAngularRange(6,8.5,12,30); theta 1 a 15 et phi 0 a 360

710 void KVElasticScatterEvent::MakeDiffusion()
711 {

713  //Propagation dans la cible du projectile jusqu'au point d interaction
714  // PropagateInTargetLayer();
715  //Tirage aleatoire d'un couple theta phi pour la direction de diffusion du projectile
716  //Determination de la cinematique de la voie de sortie
717  // SetAnglesForDiffusion(the,phi);
718  //Filtre si un multidetecteur est defini
719  // Filter
720  //Traitement de l evt (remplissage d'arbre ou d'histo
721  // TreateEvent();
722 
723  KVNucleus* knuc = 0;
724  sim_evt->GetParameters()->Clear();
725  while ((knuc = sim_evt->GetNextParticle())) {
726  knuc->GetParameters()->Clear();
727  knuc->RemoveAllGroups();
728  }
729 
730  //-------------------------
731  if (IsTargMatSet()) {
732  NewInteractionPointInTargetLayer();
733  PropagateInTargetLayer();
734  }
735  //-------------------------
736 
737 
738  Double_t tmin = GetTheta("min");
739  if (tmin >= kb2->GetMaxAngleLab(kDiffNuc)) {
740  GetNucleus("PROJ")->SetMomentum(*GetNucleus("PROJ")->GetPInitial());
741  return;
742  }
743  Double_t tmax = GetTheta("max");
744 
745  if (tmax <= kb2->GetMinAngleLab(kDiffNuc)) {
746  GetNucleus("PROJ")->SetMomentum(*GetNucleus("PROJ")->GetPInitial());
747  return;
748  }
749 
750  if (tmin < kb2->GetMinAngleLab(kDiffNuc)) tmin = kb2->GetMinAngleLab(kDiffNuc);
751  if (tmax > kb2->GetMaxAngleLab(kDiffNuc)) tmax = kb2->GetMaxAngleLab(kDiffNuc);
752 
753  Double_t pmin = GetPhi("min");
754  Double_t pmax = GetPhi("max");
755 
756  kposalea.SetPolarMinMax(tmin, tmax);
757  kposalea.SetAzimuthalMinMax(pmin, pmax);
758 
759  Double_t th_deg, ph_deg;
760  kposalea.GetRandomAngles(th_deg, ph_deg, kRandomOpt);
761 
762  SetAnglesForDiffusion(th_deg, ph_deg);
763 
764  ((TH2F*)lhisto->FindObject("phi_theta"))->Fill(th_deg, ph_deg);
765  ((TH1F*)lhisto->FindObject("costheta"))->Fill(TMath::Cos(TMath::DegToRad()*th_deg));
766 
767 
768  if (IsDetectionOn()) {
769  Filter();
770  } else {
771  if (IsTargMatSet())
772  SortieDeCible();
773  }
774 
775  TreateEvent();
776

778 }

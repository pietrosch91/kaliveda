123 void KVFAZIAIDTelescope::SetIdentificationStatus(KVReconstructedNucleus* n)
124 {
125  // For filtering simulations
126  //
127  // Z-dependence of A identification:
128  // all ok if Z<=20, decreasing probability for 21<=Z<25
129  // no A identification for Z>=25
130 
131  n->SetZMeasured();
132  fMassIDProb->SetParameters(22.5, .4);
133  Bool_t okmass = (n->GetZ() <= 20) || (n->GetZ() < 25 && gRandom->Uniform() < fMassIDProb->Eval(n->GetZ()));
134  if (okmass) {
135  n->SetAMeasured();
136  } else
137  n->SetZ(n->GetZ());

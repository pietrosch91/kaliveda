83 Double_t KVINDRADBRun::GetEventCrossSection(Double_t Q_apres_cible,
84  Double_t Coul_par_top) const
85 {
86  //Calculation of cross-section (in millibarn) per event of this run from the measured integrated beam current and correcting for dead time.
87  //Based on sect_effic.f by M.F. Rivet
88  //
89  //For meaning of arguments, see GetNIncidentIons()
90 
91  KVTarget* target = GetTarget();
92  if (!target)
93  return 0; //can't calculate X-section without knowing target characteristics
94 
95  //calculate number of atoms/cm**2 in target
96  Double_t atom_cib = target->GetAtomsPerCM2();
97 
98  //number of projectile nuclei during run
99  Double_t n_proj = GetNIncidentIons(Q_apres_cible, Coul_par_top);
100 
101  Double_t temps_mort = GetTempsMort();
102  if (temps_mort >= 1)
103  return 0.;
104 
105  //X-section [mb] per event = 1e27 / (no. atoms in target * no. of projectile nuclei * (1 - temps mort))
106  Double_t sect_effic = 1.e27 / (atom_cib * n_proj * (1 - temps_mort));
107  return sect_effic;
108 }

133 Double_t KVINDRADBRun::GetTempsMort() const
134 {
135  //Calculate temps mort for run from values of Gene dir, Gene MRQ and Gene TM, according to:
136  // T.M. = Gene TM / (Gene dir - Gene MRQ)
137  //If Gene dir = Gene MRQ, temps mort = 0.
138  //This can be compared to the value of GetTMpercent() (if present) which is the percentage value filled in the runsheet
139  //during the experimental run.
140  if (GetScaler("Gene DIRECT") == GetScaler("Gene MARQ")) {
141  //if Gene MRQ == Gene Dir, temps mort 0
142  return 0;
143  }
144  return (GetScaler("Gene TM") * 1.0 / (GetScaler("Gene DIRECT") - GetScaler("Gene MARQ")));
145 }

159 Double_t KVINDRADBRun::GetNIncidentIons(Double_t Q_apres_cible,
160  Double_t Coul_par_top) const
161 {
162  //Calculate total number of incident beam particles during run, based on measured integrated beam current
163  //Arguments:
164  // Q_apres_cible = (average) charge state of projectile ions AFTER passage through target
165  // Coul_par_top = calibration of Faraday current integrator (default: 1.e-10 Coul/top)
166  // Note (from sect_effic.f):
167  //**** Etalonnage de l'integrateur en 1993: voir cahier 4 page 87.
168  //*** Si l'oscillateur interne du trigger est a 169 Hz, on a 10**(-10) Cb/top.
169  //*** si c'est 160 Hz, on a 1.0610*(10**(-10)).
170 
171  return GetScaler("Faraday 1") * Coul_par_top / (TMath::Qe() * Q_apres_cible);
172 }

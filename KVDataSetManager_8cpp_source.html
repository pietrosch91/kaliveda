28 //KVDataSetManager handles a collection of datasets contained in a data repository,

102 Bool_t KVDataSetManager::Init(KVDataRepository* dr)
103 {
104  //Initialisation of dataset manager for the repository 'dr'.
105  //If dr=0x0 (default) then all known datasets are 'available', otherwise
106  //we check availability of datasets based on data present in repository.
107  //Initialise all possible data analysis tasks,
108  //then set list of possible tasks for each available dataset.
109  //
110  //Returns kTRUE if all goes well.
111  //For remote data repositories we return kFALSE if no datasets are available
112  //(as the access to these is read-only)
113 
114  fRepository = dr;
115 
116  ReadUserGroups();
117 
118  if (!ReadDataSetList())
119  return kFALSE;
120 
121  //use caching for dataset availability ?
122  if (dr) {
123  fCacheAvailable = gEnv->GetValue(Form("%s.DataRepository.CacheAvailable", dr->GetName()),
124  kFALSE);
125  fMaxCacheTime = (UInt_t)gEnv->GetValue(Form("%s.DataRepository.MaxCacheTime", dr->GetName()),
126  0);
127  }
128  //name of cache file
129  if (dr) fCacheFileName.Form("%s.available.datasets", dr->GetName());
130 
131  //check which datasets are available
132  CheckAvailability();
133  if (!GetNavailable() && dr && dr->IsRemote()) return kFALSE;
134 
135  if (!ReadTaskList())
136  return kFALSE;
137 
138  //set available data analysis tasks for available datasets
139  if (GetNavailable()) {
140  for (Int_t i = 1; i <= GetNavailable(); i++)
141  GetAvailableDataSet(i)->SetAnalysisTasks(&fTasks);
142  }
143 
144  // stand-alone dataset manager: make it the default
145  if (!dr) gDataSetManager = this;
146 
147  return kTRUE;
148 }

157 void KVDataSetManager::ReadUserGroups()
158 {
159  //Sets up list of user groups defining restricted access to certain datasets.
160  //Definition of different user groups is given in $KVROOT/KVFiles/.kvrootrc
161 
162  //UserGroups env var contains whitespace-separated list of group names
163  TString groups = gEnv->GetValue("UserGroup", "");
164  if (groups == "") {
165  cout << "No value for UserGroup" << endl;
166  return;
167  }
168  //split into array of group names
169  TObjArray* toks = groups.Tokenize(' ');
170  TObjString* group_name;
171  TIter next_name(toks);
172  fUserGroups.Clear();
173  while ((group_name = (TObjString*) next_name())) {
174  //for each group_name, the env var 'group_name.Users' contains a whitespace-separated list of user names
175  //we store this string in the fUserGroups parameter list with parameter name 'group_name'
176  TString users =
177  gEnv->
178  GetValue(Form("%s.Users", group_name->String().Data()), "");
179  fUserGroups.SetValue(group_name->String().Data(), users);
180  }
181  delete toks;
182 }

315 void KVDataSetManager::CheckAvailability()
316 {
317  //Check availability of datasets in repository associated to this data set manager
318  //
319  //If caching is activated for the parent repository, i.e. if
320  //
321  // [repository name].DataRepository.CacheAvailable: yes
322  //
323  //then instead of directly checking the existence of the directories for each dataset,
324  //we use the cached information written in the file
325  //KVBase::WorkingRepository()/[repository name].available.datasets
326  //unless (1) it doesn't exist, or (2) the file is older than the maximum
327  //cache time (in seconds) defined by
328  //
329  // [repository name].DataRepository.MaxCacheSeconds:
330  //
331  //In either of these 2 cases, we check the existence of the directories and update/
332  //create the cache file.
333  //
334  //If the repository appears to be empty (perhaps because we are using a remote access
335  //protocol to check it, and the protocol has some problems...), then as a last resort we
336  //we will use the cache if it exists, whatever its age.
337 
338  if (fCacheAvailable) {
339  //caching of dataset availability is activated
340  if (CheckCacheStatus()) {
341  //cache file exists and is not out of date
342  if (ReadAvailableDatasetsFile()) return;
343  }
344  }
345 
346  // print (repository-dependent) warning/informational message
347  if (fRepository) fRepository->PrintAvailableDatasetsUpdateWarning();
348 
349  //open temporary file
350  ofstream tmp_file;
351  TString tmp_file_path = fCacheFileName;
352  KVBase::OpenTempFile(tmp_file_path, tmp_file);
353 
354  fNavailable = 0;
355  if (fDataSets.GetSize()) {
356  TIter next(&fDataSets);
357  KVDataSet* ds;
358  while ((ds = (KVDataSet*) next())) {
359  //The results of this check are written in $KVROOT/KVFiles/[repository name].available.datasets
360  //This file may be read by KVRemoteDataSetManager::CheckAvailability when this
361  //data repository is accessed as a remote data repository from a remote machine.
362  //In this case we do not want the identity of the user to influence the contents of the file.
363  //Therefore even for 'unavailable' datasets we write the available datatypes (if any)
364  //in the file.
365  tmp_file << ds->GetName() << " : ";
366  ds->CheckAvailable();
367  tmp_file << ds->GetAvailableDataTypes() << endl;
368  if (ds->IsAvailable()) {
369  fNavailable++;
370  }
371  }
372 
373  //close temp file
374  tmp_file.close();
375  //if datasets are found, then we copy the temporary file to KVFiles directory,
376  //overwriting any previous version. if no datasets were found, we try the cache
377  //file (if it exists)
378  if (fNavailable && fRepository) { //if no repository is associated, no need to keep file
379  TString runlist = KVBase::GetWORKDIRFilePath(fCacheFileName.Data());
380  gSystem->CopyFile(tmp_file_path, runlist, kTRUE);
381  //set access permissions to 664
382  gSystem->Chmod(runlist.Data(), CHMODE(6, 6, 4));
383  }
384 
385  //delete temp file
386  gSystem->Unlink(tmp_file_path);
387 
388  if (!fNavailable) {
389  //no datasets found when checking file system ?
390  //can we rely on the cache file ?
391  ReadAvailableDatasetsFile();
392  } else {
393  //now set up array of available datasets' indices
394  fIndex.clear();
395  next.Reset();
396  Int_t j(0);
397  while ((ds = (KVDataSet*) next())) {
398  if (ds->IsAvailable()) {
399  fIndex.push_back(j);
400  }
401  j++;
402  }
403  }
404  }
405 }

546 Bool_t KVDataSetManager::ReadAvailableDatasetsFile()
547 {
548  //Opens and reads file containing cached info on available datasets, and sets
549  //the availability of the concerned datasets.
550  //Returns kTRUE if all goes well.
551  //Returns kFALSE if no cache exists or if file cannot be opened.
552  if (OpenAvailableDatasetsFile()) {
553  Info("ReadAvailableDataSetsFile",
554  "Reading cached information in file %s", fCacheFileName.Data());
555  //read file
556  TString line;
557  line.ReadLine(fDatasets);
558  while (fDatasets.good()) {
559 
560  TObjArray* toks = line.Tokenize(": ,");
561 
562  //first entry is dataset name
563  TString datasetname = ((TObjString*) toks->At(0))->String();
564  KVDataSet* dataset = GetDataSet(datasetname.Data());
565 
566  if (dataset) { //check dataset is known to local version of KaliVeda
567  //in case of remote repository, there may be datasets in the remote repository which are not defined here
568  if (toks->GetEntries() > 1 && dataset->CheckUserCanAccess()) {
569  //AVAILABLE DATASET
570  dataset->SetAvailable();
571  fNavailable++;
572  for (int i = 1; i < toks->GetEntries(); i++) {
573  //each following entry is a subdirectory name
574  dataset->AddAvailableDataType(((TObjString*) toks->At(i))->String().
575  Data());
576  }
577  } else {
578  //UNAVAILABLE DATASET (no subdirs)
579  dataset->SetAvailable(kFALSE);
580  }
581  }
582 
583  delete toks;
584  line.ReadLine(fDatasets);
585  }
586 
587  //close file
588  fDatasets.close();
589  fDatasets.clear();
590 
591  if (fNavailable) {
592  TIter next(&fDataSets);
593  //now set up array of available datasets' indices
594  fIndex.clear();
595  Int_t j(0);
596  KVDataSet* ds;
597  while ((ds = (KVDataSet*) next())) {
598  if (ds->IsAvailable()) {
599  fIndex.push_back(j);
600  }
601  j++;
602  }
603  }
604  //all is OK
605  return kTRUE;
606  }
607  //we could not find/open the cache file
608  return kFALSE;
609 }

619 Bool_t KVDataSetManager::CheckCacheStatus()
620 {
621  //We check the status of the available datasets cache file.
622  //We return kTRUE if the file exists & was last modified
623  //less than fMaxCacheTime seconds ago.
624 
625  TString fullpath;
626  Info("KVDataSetManager::CheckCacheStatus", "Checking for available datasets cache file...");
627  if (KVBase::SearchKVFile(KVBase::GetWORKDIRFilePath(fCacheFileName), fullpath)) {
628 
629  // file exists - how old is it ?
630  FileStat_t file_info;
631  gSystem->GetPathInfo(fullpath.Data(), file_info);
632  TDatime file_date(file_info.fMtime);
633  TDatime now;
634  UInt_t file_age = now.Convert() - file_date.Convert();
635  Info("KVDataSetManager::CheckCacheStatus", "...file found. It is %u seconds old", file_age);
636  if (file_age < fMaxCacheTime) {
637  Info("KVDataSetManager::CheckCacheStatus", "Using cached file");
638  return kTRUE;
639  } else
640  Info("KVDataSetManager::CheckCacheStatus", "File is too old (max time=%u). Update will be performed.", fMaxCacheTime);
641  } else
642  Info("KVDataSetManager::CheckCacheStatus", "...no file found");
643  return kFALSE;
644 }

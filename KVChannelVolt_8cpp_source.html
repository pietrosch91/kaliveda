23 //___________________________________________________________________________
24 //Calibration object dedicated to handling the electronic calibration (conversion
25 //channel->volts) of Si and ChIo detectors. The transfer function is a quadratic
26 //function
27 //
28 // Signal(Volts) = a0 + a1*Channel + a2*Channel^2
29 //
30 //determined by injecting a pre-determined sequence of impulsions
31 //of different amplitudes into the "test" entry of the detector pre-amplifiers and
32 //recording the corresponding channels in the coders.
33 //
34 //The pedestal for the corresponding run is not subtracted before fitting,
35 //therefore the Inverse operation (GetChannel(V=0)) does not return 0 but rather
36 //an estimation of the pedestal for the run.
37 //When used for different runs where the pedestal has changed, the change in pedestal
38 //should be corrected (this is done in KVSilicon and KVChIo).
39 //
40 //Depending on the signal type ("PG" or "GG") the type of the calibrator will
41 //be "Channel-Volt PG" or "Channel-Volt GG".
42 //___________________________________________________________________________

77 Double_t KVChannelVolt::Compute(Double_t chan) const
78 {
79  Double_t gain = 1.;
80  KVDetector* det = GetDetector();
81  if (det)
82  gain = det->GetGain();
83  //Calculate the calibrated signal strength in volts for a given channel number.
84  if (fReady) {
85  return (fPar[0] + fPar[1] * chan + fPar[2] * chan * chan) * gain_ref / gain;
86  } else {
87  return 0.;
88  }
89 }

141 Double_t KVChannelVolt::Invert(Double_t volts)
142 {
143  //Given the calibrated (or simulated) signal amplitude in volts,
144  //calculate the corresponding channel number according to the
145  //calibration parameters (useful for filtering simulations).
146 
147  Double_t gain = 1.;
148  KVDetector* det = GetDetector();
149  if (det)
150  gain = det->GetGain();
151  Int_t channel = 0;
152 
153  if (fReady) {
154  if (fPar[2]) {
155  // quadratic transfer function
156  Double_t c;
157  c = fPar[1] * fPar[1] - 4. * fPar[2] * (fPar[0] - gain / gain_ref * volts);
158  if (c < 0.0)
159  return -1;
160  c = (-fPar[1] + TMath::Sqrt(c)) / (2.0 * fPar[2]);
161  if (c < 0.0
162  && ((-fPar[1] - TMath::Sqrt(c)) / (2.0 * fPar[2])) > 0.0) {
163  c = (-fPar[1] - TMath::Sqrt(c)) / (2.0 * fPar[2]);
164  }
165  channel = (Int_t)(c + 0.5);
166  } else {
167  // linear transfer function
168  channel = (Int_t)(0.5 + (gain / gain_ref * volts - fPar[0]) / fPar[1]);
169  }
170  } else {
171  Warning("Compute", "Parameters not correctly initialized");
172  }
173  return (Double_t) channel;
174 }

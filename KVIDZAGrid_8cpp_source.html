32 <h2>KVIDZAGrid</h2>
33 <h4>Identification grid with lines corresponding to different nuclear isotopes (KVIDZALine)</h4>
34 Such a grid can be used either to identify simultaneously both the mass and charge of detected
35 particles (if lines for several isotopes of each atomic number are drawn), or solely the charge
36 (if only one isotope per Z is drawn, and if SetOnlyZId(kTRUE) is called).
37 
38 <h3>Identification quality codes</h3>
39 After each identification attempt, the value returned by GetQualityCode() indicates whether the
40 identification was successful or not. The meaning of the different codes depends on the type
41 of identification.
42 
43 <h4>Z & A (mass & charge) isotopic identification grid</h4>
44 <ul>
45  <li> KVIDZAGrid::kICODE0, ok</li>
46  <li> KVIDZAGrid::kICODE1, Z ok, slight ambiguity of A, which could be larger</li>
47  <li> KVIDZAGrid::kICODE2, Z ok, slight ambiguity of A, which could be smaller</li>
48  <li> KVIDZAGrid::kICODE3, Z ok, slight ambiguity of A, which could be larger or smaller</li>
49  <li> KVIDZAGrid::kICODE4, point is in between two isotopes of different Z, too far from either to be considered well-identified</li>
50  <li> KVIDZAGrid::kICODE5, point is in between two isotopes of different Z, too far from either to be considered well-identified</li>
51  <li> KVIDZAGrid::kICODE6, (x,y) is below first line in grid</li>
52  <li> KVIDZAGrid::kICODE7, (x,y) is above last line in grid</li>
53  <li> KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid</li>
54 </ul>
55 
56 <h4>Z-only charge identification grid</h4>
57 <ul>
58  <li> KVIDZAGrid::kICODE0, ok</li>
59  <li> KVIDZAGrid::kICODE1, slight ambiguity of Z, which could be larger</li>
60  <li> KVIDZAGrid::kICODE2, slight ambiguity of Z, which could be smaller</li>
61  <li> KVIDZAGrid::kICODE3, slight ambiguity of Z, which could be larger or smaller</li>
62  <li> KVIDZAGrid::kICODE4, point is in between two lines of different Z, too far from either to be considered well-identified</li>
63  <li> KVIDZAGrid::kICODE5, point is in between two lines of different Z, too far from either to be considered well-identified</li>
64  <li> KVIDZAGrid::kICODE6, (x,y) is below first line in grid</li>
65  <li> KVIDZAGrid::kICODE7, (x,y) is above last line in grid</li>
66  <li> KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid</li>
67 </ul>
68 
69 In both cases, an acceptable identification is achieved if the quality code is kICODE0, kICODE1, kICODE2, or kICODE3.<br>
70 Points with codes kICODE4 or kICODE5 are normally considered as "noise" and should be rejected.<br>
71 Points which are (vertically) out of range for this grid have code kICODE6 (point too far below) or kICODE7 (point too far above).<br>
72 Points with code kICODE8 are totally out of range.

254 KVIDZALine* KVIDZAGrid::GetZLine(Int_t z, Int_t& index) const
255 {
256  //Returns ID line for which GetZ() returns 'z'. index=index of line found in
257  //fIDLines list (-1 if not found).
258  //To increase speed, this is done by dichotomy, rather than by looping over
259  //all the lines in the list.
260 
261  index = -1;
262  Int_t nlines = GetNumberOfIdentifiers();
263  UInt_t idx_min = 0; //minimum index
264  UInt_t idx_max = nlines - 1; // maximum index
265  //as a first guess, we suppose that the lines are numbered sequentially from
266  //Z=1 (index 0) to Z=zmax=nlines (index nlines-1) with no gaps
267  UInt_t idx = ((UInt_t) z - 1 > idx_max ? idx_max : (UInt_t) z - 1);
268 
269  while (idx_max > idx_min + 1) {
270 
271  KVIDZALine* line = (KVIDZALine*) GetIdentifierAt(idx);
272  Int_t zline = line->GetZ();
273  //Found it ?
274  if (zline == z) {
275  index = idx;
276  return line;
277  }
278 
279 
280  if (z > zline) {
281  //increase index
282  idx_min = idx;
283  idx += (Int_t)((idx_max - idx) / 2 + 0.5);
284  } else {
285  //decrease index
286  idx_max = idx;
287  idx -= (Int_t)((idx - idx_min) / 2 + 0.5);
288  }
289  }
290  //if one of these two lines has the right Z, return its pointer
291  KVIDZALine* line = (KVIDZALine*) GetIdentifierAt(idx_min);
292  if (line->GetZ() == z) {
293  index = idx_min;
294  return line;
295  }
296  line = (KVIDZALine*) GetIdentifierAt(idx_max);
297  if (line->GetZ() == z) {
298  index = idx_max;
299  return line;
300  }
301  //if not, there is no line in the grid with the right Z
302  return 0;
303 }

314 KVIDZALine* KVIDZAGrid::GetZALine(Int_t z, Int_t a, Int_t& index) const
315 {
316  //Returns ID line corresponding to nucleus (Z,A) and its index in fIDLines.
317  //First we use GetLine(z) to find a line with the right Z (in
318  //principal there are several in the grid), then search for the correct
319  //isotope among these.
320 
321  Int_t idx;
322  index = -1;
323  KVIDZALine* line = GetZLine(z, idx);
324  //no line with correct Z ?
325  if (!line)
326  return 0;
327  //got right mass?
328  if (line->GetA() == a) {
329  index = idx;
330  return line;
331  }
332  //increase/decreas index depending on if mass of line is greater than or less than a
333  if (a > line->GetA()) {
334  for (int i = idx; i < (int) GetNumberOfIdentifiers(); i++) {
335  line = (KVIDZALine*) GetIdentifierAt(i);
336  if (line->GetZ() != z)
337  return 0; //no longer correct Z
338  if (line->GetA() == a) {
339  index = i;
340  return line;
341  }
342  }
343  //gone to the end of the list and still not found it
344  return 0;
345  } else if (a < line->GetA()) {
346  for (int i = idx; i > 0; i--) {
347  line = (KVIDZALine*) GetIdentifierAt(i);
348  if (line->GetZ() != z)
349  return 0; //no longer correct Z
350  if (line->GetA() == a) {
351  index = i;
352  return line;
353  }
354  }
355  //got to start of list and still not found
356  return 0;
357  }
358  return 0;
359 }

388 void KVIDZAGrid::CalculateLineWidths()
389 {
390  //Calculate natural "width" of each line in the grid.
391  //The lines in the grid are first sorted so that they are in order of ascending 'Y'
392  //i.e. first line is 1H, last line is the heaviest isotope (highest line).
393  //
394  //Then, for a given line :
395  //
396  // **** if the grid is to be used for A & Z identification (default)**** :
397  // - if the next line (above) has the same Z, we use the separation between these
398  // two lines corresponding to different isotopes of the same element
399  // - if the next line (above) has a different Z, but the line before (below) has
400  // the same Z, we use the separation between the line below and this one
401  // - if neither adjacent line has the same Z, the width is set to 16000 (huge).
402  //
403  // **** if the grid is to be used for Z identification (fOnlyZid=kTRUE)**** :
404  // - we use the separation between each pair of lines
405  //
406  //In each case we find D_L (the separation between the two lines at their extreme left)
407  //and D_R (their separation at extreme right). The width of the line is then calculated
408  //from these two using the method KVIDZALine::SetAsymWidth (which may be overridden in child
409  //classes).
410 
411 
412 // Info("CalculateLineWidths",
413 // "For grid %s (%s vs. %s, runs %s).", GetName(), GetVarY(), GetVarX(), GetRunList());
414 
415  for (Int_t i = 0; i < (Int_t) GetNumberOfIdentifiers(); i++) {
416 
417  KVIDentifier* obj = GetIdentifierAt(i);
418  if (!obj->InheritsFrom("KVIDZALine")) continue;
419  KVIDZALine* _line = (KVIDZALine*) obj;
420 // KVIDZALine* _line = (KVIDZALine*) GetIdentifierAt(i);
421 
422  //Z of lines above and below this line - Zxx=-1 if there is no line above or below
423  Int_t Zhi =
424  (i <
425  (Int_t) GetNumberOfIdentifiers() -
426  1 ? ((KVIDZALine*) GetIdentifierAt(i + 1))->GetZ() : -1);
427  Int_t Zlo = (i > 0 ? ((KVIDZALine*) GetIdentifierAt(i - 1))->GetZ() : -1);
428  Int_t Z = _line->GetZ();
429 
430  // find line for comparison.
431  // if we are dealing with a grid with several isotopes for each Z, the priority
432  // is to calculate isotopic line widths using lines with same Z, different A.
433  // if the grid is 'OnlyZId' it should have only one line per Z, so we calculate
434  // widths by comparing neighbouring Z.
435  // if the grid is not 'OnlyZId', it may still have some lines with just one isotope
436  // per Z, in this case we calculate a width based on the separation from the
437  // neighbouring Z, as if it were 'OnlyZId'.
438  Int_t i_other;// index of line used to calculate width
439  if (OnlyZId()) { //grid is 'OnlyZId':calculate widths by comparing neighbouring Z.
440  i_other = (Zhi > -1 ? i + 1 : (Zlo > -1 ? i - 1 : -1));
441  } else {
442  //grid with several isotopes for each Z:calculate isotopic widths using lines with same Z, different A
443  i_other = (Zhi == Z ? i + 1 : (Zlo == Z ? i - 1 : -1));
444  }
445  if (!OnlyZId() && i_other < 0) { //grid is not 'OnlyZId' but only one isotope for this Z
446  i_other = (Zhi > -1 ? i + 1 : (Zlo > -1 ? i - 1 : -1));
447  }
448 
449  //default width of 16000 in case of "orphan" line
450  if (i_other < 0) {
451  //Info("CalculateLineWidths", "No line found for comparison with %s. Width set to 16000", _line->GetName());
452  _line->SetWidth(16000.);
453  continue; // skip to next line
454  }
455 
456  KVIDZALine* _otherline = (KVIDZALine*) GetIdentifierAt(i_other);
457 
458  //calculate asymptotic distances between lines at left and right.
459  //do this by finding which line's endpoint is between both endpoints of the other line
460  Double_t D_L, D_R;
461 
462  //*************** LEFT SIDE ********************************
463 
464  //get coordinates of first point of our line
465  Double_t x1, y1;
466  _line->GetStartPoint(x1, y1);
467  KVIDZALine* _aline, *_bline;
468 
469  if (_otherline->IsBetweenEndPoints(x1, y1, "x")) {
470 
471  //first point of our line is "inside" the X-coords of the other line:
472  //asymptotic distance LEFT is distance from our line's starting point (x1,Y1) to the other line
473  _aline = _otherline;
474  _bline = _line;
475 
476  } else {
477 
478  //first point of other line is "inside" the X-coords of the our line:
479  //asymptotic distance LEFT is distance from other line's starting point (x1,Y1) to our line
480  _aline = _line;
481  _bline = _otherline;
482  _otherline->GetStartPoint(x1, y1);
483 
484  }
485 
486  //calculate D_L
487  Int_t dummy = 0;
488  D_L = _aline->DistanceToLine(x1, y1, dummy);
489 
490  //make sure that D_L is positive : if not, we try to calculate other way (inverse line and starting point)
491  if (D_L < 0.) {
492 
493  Double_t oldD_L = D_L;
494 
495  _aline->GetStartPoint(x1, y1);
496  D_L = _bline->DistanceToLine(x1, y1, dummy);
497 
498  //still no good ? then we keep the smallest absolute value
499  if (D_L < 0.) {
500  D_L = TMath::Abs(TMath::Max(D_L, oldD_L));
501  }
502  }
503  //*************** RIGHT SIDE ********************************
504 
505  //get coordinates of last point of our line
506  _line->GetEndPoint(x1, y1);
507 
508  if (_otherline->IsBetweenEndPoints(x1, y1, "x")) {
509 
510  //last point of our line is "inside" the X-coords of the other line:
511  //asymptotic distance RIGHT is distance from our line's end point (x1,Y1) to the other line
512  _aline = _otherline;
513  _bline = _line;
514 
515  } else {
516 
517  //last point of other line is "inside" the X-coords of the our line:
518  //asymptotic distance RIGHT is distance from other line's end point (x1,Y1) to our line
519  _aline = _line;
520  _bline = _otherline;
521  _otherline->GetEndPoint(x1, y1);
522 
523  }
524 
525  //calculate D_R
526  D_R = _aline->DistanceToLine(x1, y1, dummy);
527 
528  //make sure that D_R is positive : if not, we try to calculate other way (inverse line and end point)
529  if (D_R < 0.) {
530 
531  Double_t oldD_R = D_R;
532 
533  _aline->GetEndPoint(x1, y1);
534  D_R = _bline->DistanceToLine(x1, y1, dummy);
535 
536  //still no good ? then we keep the smallest absolute value
537  if (D_R < 0.) {
538  D_R = TMath::Abs(TMath::Max(D_R, oldD_R));
539  }
540  }
541  //*************** SET NATURAL WIDTH OF LINE ********************************
542 
543  _line->SetAsymWidth(D_L, D_R);
544 
545 
546  //Info("CalculateLineWidths", "...width for line %s set to : %f (D_L=%f,D_R=%f)", _line->GetName(), _line->GetWidth(), D_L, D_R);
547  }
548 }

615 Bool_t KVIDZAGrid::FindFourEmbracingLines(Double_t x, Double_t y, const Char_t* position)
616 {
617  // This method will locate (at most) four lines close to the point (x,y), the point must
618  // lie within the endpoints (in X) of each line (the lines "embrace" the point).
619  // Returns kTRUE if at least one line is found. Identification can then be carried out with
620  // either IdentZA or IdentZ (see Identify).
621  // Returns kFALSE if no lines are found (not even a closest embracing line).
622  //
623  // We look for two lines above the point and two lines below the point, as in one of the following two cases:
624  //
625  // ------------------------ ksups ---------------------
626  //
627  // closest ---> ------------------------ ksup ---------------------
628  // X
629  //
630  // X
631  // ------------------------ kinf --------------------- <--- closest
632  //
633  // ------------------------ kinfi ---------------------
634  //
635  // First we find the closest embracing line to the point, using FindNearestEmbracingIDLine.
636  // Then we search above and below for the other 'embracing' lines. Note that no condition is
637  // applied regarding the distances to these lines: the lines must have been sorted in order of increasing
638  // ordinate before hand in Initialize(), we simply use the order of lines in the list of identifiers.
639  // The Z, A, width and distance to each of these lines are stored in the variables
640  // Zsups, Asups, wsups, dsups
641  // etc. etc. to be used by IdentZA or IdentZ.
642 
643  kinfi = kinf = ksup = ksups = -1;
644  dinf = dsup = dinfi = dsups = 0.;
645  winf = wsup = winfi = wsups = 16000.;
646  Zinfi = Zinf = Zsup = Zsups = Ainfi = Ainf = Asup = Asups = -1;
647  fDistanceClosest = -1.;
648  fClosest = fLsups = fLsup = fLinf = fLinfi = 0;
649  fIdxClosest = -1;
650 
651  fClosest = FindNearestEmbracingIDLine(x, y, position, "x", fIdxClosest, kinf, ksup, fDistanceClosest, dinf, dsup);
652 
653  if (!fClosest) return kFALSE; // no lines found
654 
655  Int_t dummy = 0;
656  if (kinf > -1 && kinf == fIdxClosest) {
657  //point is above closest line, closest line is "kinf"
658  //need to look for 2 lines above (ksup, ksups) and 1 line below (kinfi)
659  fLinf = fClosest;
660  if (fLinf->InheritsFrom("KVIDZALine")) {
661  winf = ((KVIDZALine*)fLinf)->GetWidth();
662  Zinf = fLinf->GetZ();
663  Ainf = fLinf->GetA();
664  }
665  if (ksup > -1) {
666  fLsup = (KVIDLine*)GetIdentifierAt(ksup);
667  if (fLsup->InheritsFrom("KVIDZALine")) {
668  wsup = ((KVIDZALine*)fLsup)->GetWidth();
669  Zsup = fLsup->GetZ();
670  Asup = fLsup->GetA();
671  }
672  }
673  } else if (ksup > -1 && ksup == fIdxClosest) {
674  //point is below closest line, closest line is "ksup"
675  //need to look for 1 line above (ksups) and 2 lines below (kinf, kinfi)
676  fLsup = fClosest;
677  if (fLsup->InheritsFrom("KVIDZALine")) {
678  wsup = ((KVIDZALine*)fLsup)->GetWidth();
679  Zsup = fLsup->GetZ();
680  Asup = fLsup->GetA();
681  }
682  if (kinf > -1) {
683  fLinf = (KVIDLine*)GetIdentifierAt(kinf);
684  if (fLinf->InheritsFrom("KVIDZALine")) {
685  winf = ((KVIDZALine*)fLinf)->GetWidth();
686  Zinf = fLinf->GetZ();
687  Ainf = fLinf->GetA();
688  }
689  }
690  } else {
691  Error("FindFourEmbracingLines",
692  "I do not understand the result of FindNearestEmbracingIDLine!!!");
693  return kFALSE;
694  }
695 
696 
697  if (kinf > -1) {
698  // look for kinfi line -> next line below 'inf' line
699  kinfi = kinf;
700  fLinfi = FindNextEmbracingLine(kinfi, -1, x, y, "x");
701  if (!fLinfi) kinfi = -1; // no 'infi' line found
702  else {
703  dinfi = TMath::Abs(fLinfi->DistanceToLine(x, y, dummy));
704  if (fLinfi->InheritsFrom("KVIDZALine")) {
705  winfi = ((KVIDZALine*)fLinfi)->GetWidth();
706  Zinfi = fLinfi->GetZ();
707  Ainfi = fLinfi->GetA();
708  }
709  }
710  }
711  if (ksup > -1) {
712  // look for ksups line -> next line above 'sup' line
713  ksups = ksup;
714  fLsups = FindNextEmbracingLine(ksups, 1, x, y, "x");

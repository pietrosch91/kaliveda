129 Int_t KVINDRATelescope::Compare(const TObject* obj) const
130 {
131  // telescopes are sorted
132  // (i) according to layer number if they are in different layers
133  // (ii) according to lower edge polar angle if they belong to the same layer
134  // (iii) according to telescope number if they belong to the same ring
135 
136  KVINDRATelescope* tobj = (KVINDRATelescope*)obj;
137 
138  Int_t this_layer_number = GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
139  Int_t obj_layer_number = tobj->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
140  Int_t this_ring_number = GetParentStructure("RING")->GetNumber();
141  Int_t obj_ring_number = tobj->GetParentStructure("RING")->GetNumber();
142 
143  if (this_layer_number != obj_layer_number) {
144 // not in the same layer - sort according to layer number, smallest first
145 //(closest to target)
146  if (this_layer_number < obj_layer_number)
147  return -1;
148  else if (this_layer_number > obj_layer_number)
149  return 1;
150  else
151  return 0;
152  }
153  if (this_ring_number != obj_ring_number) {
154 // not in the same ring - sort according to lower edge polar angle
155  if (GetThetaMin() < tobj->GetThetaMin())
156  return -1;
157  else if (GetThetaMin() > tobj->GetThetaMin())
158  return 1;
159  else
160  return 0;
161  }
162  if (this_ring_number == obj_ring_number) {
163 // same ring - sort according to telescope number
164  if (GetNumber() < tobj->GetNumber())
165  return -1;
166  else if (GetNumber() > tobj->GetNumber())
167  return 1;
168  else
169  return 0;
170  }
171  return 0;
172 }

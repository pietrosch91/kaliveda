70 void KVVAMOSReconEvent::Streamer(TBuffer& R__b)
71 {
72  //Stream an object of class KVVAMOSReconEvent.
73  //We loop over the newly-read nuclei in order to set their
74  //IsOK() status by comparison with the event's code mask
75 
76  if (R__b.IsReading()) {
77  R__b.ReadClassBuffer(KVVAMOSReconEvent::Class(), this);
78  KVVAMOSReconNuc* nuc;
79  while ((nuc = GetNextNucleus())) {
80  if (CheckCodes(nuc->GetCodes()))
81  nuc->SetIsOK();
82  else
83  nuc->SetIsOK(kFALSE);
84  }
85  } else {
86  R__b.WriteClassBuffer(KVVAMOSReconEvent::Class(), this);
87  }

109 {
110  //Define the calibration codes for VAMOS's detectors that you want to include
111  //in your analysis.
112  //Example: to keep only ecodes 1 and 2, use
113  // event.AcceptECodes( kECode2 | kECode3 );
114  //If the nucleus reconstructed in VAMOS have the correct E codes it will
115  //have IsOK() = kTRUE.
116  //If this nucleus does not have IsOK() = kTRUE then the method GetNextNucleus("ok")
117  //will return a null pointer.
118  //
119  //To remove any previously defined acceptable calibration codes, use AcceptECodes(0).
120  //
121  //N.B. : this method is preferable to using directly the KVAMOSCodes pointer
122  //of the nucleus as the 'IsOK' status of this nucleus of the current event
123  //is automatically updated when the code mask is changed using this method.

288 KVVAMOSReconNuc* KVVAMOSReconEvent::GetNextNucleus(Option_t* opt)
289 {
290  //Use this method to iterate over the list of nuclei in the event.
291  //
292  //If opt="ok" only the nuclei whose ID codes, FP codes, and E codes correspond to those set
293  //using AcceptIDCodes, AccepFPCodes and AcceptECodes will be returned, in the order in which
294  //they appear in the event.
295  //
296  //After the last nucleus in the event GetNextNucleus() returns a null pointer and
297  //resets itself ready for a new iteration over the nucleus list.
298  //Therefore, to loop over all nucleus in an event, use a structure like:
299  //
300  // KVVAMOSReconNuc* n;
301  // while( (n = GetNextNucleus() ){
302  // ... your code here ...
303  // }
304 
305  return (KVVAMOSReconNuc*) KVEvent::GetNextParticle(opt);

355 void KVVAMOSReconEvent::IdentAndCalibEvent()
356 {
357 
358  //If the nuclei measured in VAMOS have not been previously Z-identified
359  //it will be Z-identified.
360 
361  // In order to make sure that target energy loss corrections are correctly
362  // calculated, we first set the state of the target in the current multidetector
363 
364  KVTarget* t = gMultiDetArray->GetTarget();
365  if (t) {
366  t->SetIncoming(kFALSE);
367  t->SetOutgoing(kTRUE);
368  }
369 
370  KVVAMOSReconNuc* d;
371  while ((d = GetNextNucleus())) {
372 
373  //debug
374  if (fkverbose) {
375  printf("\n");
376  Info("IdentAndCalibEvent", "... new particle ...");
377  }
378 
379  //-----------------------
380  // Z-identification
381  //-----------------------
382  if (!d->IsZidentified()) {
383  if (d->GetStatus() == KVReconstructedNucleus::kStatusOK) {
384  // identifiable particles
385  d->IdentifyZ();
386  } else if (d->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
387  // particles stopped in first member of a telescope
388  // estimation of Z (minimum) from energy loss (if detector is calibrated)
389  UInt_t zmin = d->GetStoppingDetector()->FindZmin(-1., d->GetMassFormula());
390  if (zmin) {
391  d->SetZ(zmin);
392  d->SetIsZidentified();
393  // "Identifying" telescope is taken from list of ID telescopes
394  // to which stopping detector belongs
395  d->SetIdentifyingTelescope((KVIDTelescope*)d->GetStoppingDetector()->GetIDTelescopes()->At(0));
396  d->SetIDCode(kIDCode5); // Zmin
397  }
398  }
399  }
400 
401  if (fkverbose) {
402  Info("IdentAndCalibEvent", "... Z identification done, results follow ...");
403  if (d->GetIDCode() == 3) {
404  Info("IdentAndCalibEvent", "IDCode=%d \nZident=%d, Zmeasured=%d, Ameasured=%d \nRealZ=%lf, RealA=%lf \nZ=%d, A=%d, A_CsI=%d",
405  d->GetIDCode(), (int) d->IsZidentified(), (int) d->IsZMeasured(), (int) d->IsAMeasured(), d->GetRealZ(), d->GetRealA(), d->GetZ(), d->GetA(), d->GetAMinimizer());
406  } else {
407  Info("IdentAndCalibEvent", "IDCode=%d \nZident=%d, Zmeasured=%d, Ameasured=%d \nRealZ=%lf, RealA=%lf \nZ=%d, A=%d",
408  d->GetIDCode(), (int) d->IsZidentified(), (int) d->IsZMeasured(), (int) d->IsAMeasured(), d->GetRealZ(), d->GetRealA(), d->GetZ(), d->GetA());
409  }
410  }
411 
412  //-----------------------
413  // Calibration
414  //-----------------------
415  if (d->IsZidentified() && !d->IsCalibrated()) {
416  d->Calibrate();
417  if (fkverbose) Info("IdentAndCalibEvent", "... calibration done ...");
418  }
419 
420  //--------------------------------------------------------------------------
421  //All particles which have been previously Z-identified (IsZidentified=kTRUE)
422  //and calibrated (IsCalibrated=kTRUE) will be Q and A identified.
423  //--------------------------------------------------------------------------
424  if (d->IsZidentified() && d->IsCalibrated() && !d->IsBasicQandAidentified()) {
425  d->IdentifyQandA();
426 
427  if (fkverbose) {
428  Info("IdentAndCalibEvent", "... Basic Q and A identification done, results follow ...");
429  Info("IdentAndCalibEvent", "IDCode=%d \nZident=%d, Zmeasured=%d, Ameasured=%d, QandAidentified=%d \nToF=%lf, nHF=%d, Path=%lf, Beta=%lf \nrealAE=%lf, realAoQ=%lf \nrealQ=%lf, intQ=%d, realA=%lf, intA=%d",
430  d->GetIDCode(), (int) d->IsZidentified(), (int) d->IsZMeasured(), (int) d->IsAMeasured(), d->IsBasicQandAidentified(), d->GetBasicToF(), d->GetBasicNHF(),
431  d->GetBasicPath(), d->GetBasicBeta(), d->GetBasicRealAE(), d->GetBasicRealAoverQ(), d->GetBasicRealQ(), d->GetBasicQ(), d->GetBasicRealA(), d->GetBasicA());
432  }
433  }
434  }

499 void KVVAMOSReconEvent::ReconstructEvent(KVMultiDetArray* mda, KVDetectorEvent* kvde)
500 {
501  //
502  // Reconstruction of detected nuclei
503  //
504  // - loop over last stage of group telescopes
505  // - if any detector is hit, construct "nucleus" from energy losses in all detectors
506  // directly in front of it.
507  // - loop over next to last stage...if any detector hit NOT ALREADY IN A "NUCLEUS"
508  // construct "nucleus" etc. etc.
509  //
510  // Then the trajectory (theta, phi, Brho, path) of each reconstructed nucleus is reconstructed
511  // from positions (Xf, Yf, Theta_f, Phi_f) mesured at the focal plane.
512 
513  mda->ReconstructEvent(this, kvde);
514 
515  KVVAMOSReconNuc* nuc = NULL;
516 // UChar_t Nnuc = 0;
517  while ((nuc = GetNextNucleus())) {
518 // if( Nnuc > 1 ){
519 // Warning("ReconstrucEvent","Ambiguous trajectory reconstruction, more than one nucleus in event %d", GetNumber());
520 // break;
521 // }
522  nuc->ReconstructTrajectory();
523 // Nnuc++;
524  }
525 }

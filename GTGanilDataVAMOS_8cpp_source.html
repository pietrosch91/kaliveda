142 bool GTGanilDataVAMOS::EventUnravelling(CTRL_EVENT* pCtrlEvent)
143 {
144  // PRIVATE
145  // If mode is variable length event, we have to reconstruct the Data buffer
146  // from the given event.
147  // WARNING: temporary the default: we dont check that it's really the case
148  // Before reading event, all parameters have their value set to -1 (65535 - fDataArray is UShort_t)
149  // Parameters which are not fired in the event will have value -1 (65535 - cast back to Short_t for real value)
150  //
151  //VAMOS parameters are treated here:
152  // clear all parameters
153  // check no parameters occur twice in same event (if so, the VAMOS treatment is skipped)
154 
155  Short_t* brutData = &(pCtrlEvent->ct_par);
156  Int_t eventLength = pCtrlEvent->ct_len;
157 
158  for (Int_t i = 1; i <= fDataArraySize; i++) {
159  fDataArray[i] = (Short_t) - 1;
160  }
161  Par->Clear();
162 
163  bool fOK;
164 
165  fOK = true;
166 
167  //VAMOS double parameter check
168  for (Int_t k = 0; k < eventLength - 6 - 2; k += 2) {
169  for (Int_t l = k + 2; l < eventLength - 6; l += 2) {
170  if (brutData[k] == brutData[l]) {
171  fOK = false;
172  }
173  }
174  }
175 
176  for (Int_t i = 0; i < eventLength; i += 2) {
177  //normal GTGanilData/INDRA treatment
178  if (brutData[i] <= fDataArraySize && brutData[i] >= 1) {
179  fDataArray[brutData[i]] = brutData[i + 1];
180  }
181  //VAMOS treatment
182  if ((i < (eventLength - 6)) && fOK) {
183  if (brutData[i] > 0 && brutData[i + 1] > 0) {
184  Par->GetData(brutData + i);
185  }
186  }
187  }
188 
189  return (true);
190 }

142 Bool_t KVSimReader_HIPSE::ReadEvent()
143 {
144 
145  evt->Clear();
146  Int_t mult = 0, mtotal = 0;
147 
148  /*---------------------------------------------
149  mul_vrai = multiplicity of charged particles
150  mult = total multiplicity (i.e. including neutrons)
151  //---------------------------------------------
152  */
153  Int_t res = ReadLineAndCheck(2, " ");
154  switch (res) {
155  case 0:
156  Info("ReadEvent", "case 0 line est vide");
157  return kFALSE;
158  case 1:
159  evt->SetNumber(nevt);
160  mult = GetIntReadPar(0); //mul_vrai
161  mtotal = GetIntReadPar(1);
162  evt->GetParameters()->SetValue("mult", mtotal);
163 
164  break;
165  default:
166 
167  return kFALSE;
168  }
169 
170  /*---------------------------------------------
171  Esa = excitation per nucleon
172  vcm = center of mass energy
173  Bparstore = impact parameter
174  //---------------------------------------------
175  */
176  res = ReadLineAndCheck(3, " ");
177  switch (res) {
178  case 0:
179  return kFALSE;
180  case 1:
181  evt->GetParameters()->SetValue("Esa", GetDoubleReadPar(0));
182  evt->GetParameters()->SetValue("vcm", GetDoubleReadPar(1));
183  evt->GetParameters()->SetValue("Bparstore", GetDoubleReadPar(2));
184 
185  break;
186  default:
187 
188  return kFALSE;
189  }
190 
191  /*---------------------------------------------
192  energetic information
193  excitat : total excitation energy
194  xmassav : Q-value
195  ekinav : total kinetic energy at freeze-out
196  epotav : total potential energy at freeze-out
197  erotav : total rotational energy at freeze-out
198  //---------------------------------------------
199  */
200  res = ReadLineAndCheck(2, " ");
201  switch (res) {
202  case 0:
203  return kFALSE;
204  case 1:
205  evt->GetParameters()->SetValue("excitat", GetDoubleReadPar(0));
206  evt->GetParameters()->SetValue("xmassav", GetDoubleReadPar(1));
207 
208  break;
209  default:
210 
211  return kFALSE;
212  }
213 
214  res = ReadLineAndCheck(3, " ");
215  switch (res) {
216  case 0:
217  return kFALSE;
218  case 1:
219 
220  evt->GetParameters()->SetValue("ekinav", GetDoubleReadPar(0));
221  evt->GetParameters()->SetValue("epotav", GetDoubleReadPar(1));
222  evt->GetParameters()->SetValue("erotav", GetDoubleReadPar(2));
223  break;
224 
225  default:
226  return kFALSE;
227  }
228 
229  evt->SetNumber(nevt);
230  for (Int_t mm = 0; mm < mult; mm += 1) {
231  nuc = (KVSimNucleus*)evt->AddParticle();
232  if (!ReadNucleus()) return kFALSE;
233  }
234 
235  nevt += 1;
236  return kTRUE;

245 {
246

264

266  return kFALSE;
267  }

271  case 0:
272  Info("ReadNucleus", "case 0 line est vide");

274 
275  case 1:
276  //Axe "faisceau dans HIPSE x -> on effectue une rotation X,Y,Z -> Y,Z,X"

278  break;

289  */

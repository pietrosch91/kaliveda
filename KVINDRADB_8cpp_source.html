186 KVList* KVINDRADB::GetCalibrationPeaks(Int_t run, KVDetector* detector,
187  Int_t peak_type, Int_t signal_type,
188  Double_t peak_energy)
189 {
190  //Use this method to access the 'peaks' (see class KVDBPeak and derivatives)
191  //used for the calibration of detectors in a given run.
192  //
193  //Mandatory argument :
194  // run : number of the run for which calibration peaks are valid
195  //Optional arguments :
196  // detector : detector for which peaks are valid
197  // peak_type : type of peak
198  // peak_type = 1 Thoron alpha peak E=6.062 MeV
199  // peak_type = 2 Thoron alpha peak E=8.785 MeV
200  // peak_type = 3 Elastic scattering peak
201  // peak_type = 4 Thoron alpha peak E=6.062 MeV (no gas in ChIo)
202  // peak_type = 5 Thoron alpha peak E=8.785 MeV (no gas in ChIo)
203  // signal_type : one of the INDRA signal types, see KVINDRA::GetDetectorByType
204  // peak_energy : nominal energy corresponding to peak (incident energy of projectile
205  // in case of elastic scattering peak).
206  //
207  //The peaks are read as and when necessary in the peak database file.
208  //
209  //USER'S RESPONSIBILITY TO DELETE LIST AFTERWARDS.
210 
211  if (!gIndra) {
212  Error("GetCalibrationPeaks",
213  "No KVINDRA object found. You need to build INDRA before using this method.");
214  return 0;
215  }
216 
217  if (!OpenCalibrationPeakFile()) {
218  //Calibration peak file not defined or not found
219  Error("GetCalibrationPeaks",
220  "Calibration peak file not defined or not found");
221  return 0;
222  }
223  //types of peaks
224  enum {
225  kAlpha6MeV = 1,
226  kAlpha8MeV = 2,
227  kElastic = 3,
228  kAlpha6MeVSG = 4,
229  kAlpha8MeVSG = 5
230  };
231 
232  TString sline;
233  Int_t frun = 0, lrun = 0;
234  KVList* peak_list = new KVList();
235 
236  //set to true after reading a run range corresponding to 'run'
237  Bool_t ok_for_this_run = kFALSE;
238 
239  Int_t cour, modu, sign, typ2, first, last, entries;
240  Float_t mean, error, sigma, constante;
241 
242  while (GetPeakFileStream().good()) {
243 
244  sline.ReadLine(GetPeakFileStream());
245 
246  if (sline.BeginsWith("Run Range :")) { // Run Range found
247 
248  if (sscanf(sline.Data(), "Run Range : %d %d", &frun, &lrun) != 2) {
249  Warning("ReadPeakList()",
250  "Bad format in line :\n%s\nUnable to read run range values",
251  sline.Data());
252  }
253  else {
254  if (TMath::Range(frun, lrun, run) == run) { //frun <= run <= lrun
255  ok_for_this_run = kTRUE;
256  }
257  else
258  ok_for_this_run = kFALSE;
259  }
260  } //Run Range found
261  else if (!sline.BeginsWith("#") && ok_for_this_run) {
262 
263  //only analyse (non-commented) lines which follow a valid run range
264 
265  if (sscanf(sline.Data(), "%d %d %d %d %f %f %f %d %f %d %d",
266  &cour, &modu, &sign, &typ2, &mean, &error, &sigma,
267  &entries, &constante, &first, &last) != 11) {
268  Warning("GetCalibrationPeaks()",
269  "Bad format in line :\n%s\nUnable to read peak parameters",
270  sline.Data());
271  }
272  else { // parameters correctly read
273 
274  //find corresponding detector
275 
276  KVDetector* pic_det =
277  //the chio's in the file are written with the
278  //ring,module of the Si/CsI in coinc
279  (sign > ChIo_T) ? gIndra->GetDetectorByType(cour, modu, sign)
280  : gIndra->GetDetectorByType(cour, modu, CsI_R)->GetChIo();
281 
282  //is it the right detector ?
283  if (detector && detector != pic_det)
284  continue; //read next line
285 
286  //is it the right type of peak ?
287  if (peak_type > 0 && peak_type != typ2)
288  continue;
289 
290  //is it the right signal type ?
291  if (signal_type && signal_type != sign)
292  continue;
293 
294  KVDBPeak* peak = 0;
295  KVDBSystem* sys = 0;
296 
297  switch (typ2) { //what type of peak
298 
299  case kAlpha8MeVSG:
300 
301  peak = new KVDBAlphaPeak(pic_det->GetName());
302  peak->SetGas(kFALSE);
303  peak->SetEnergy(PEAK_THORON_2);
304  break;
305 
306  case kAlpha6MeVSG:
307 
308  peak = new KVDBAlphaPeak(pic_det->GetName());
309  peak->SetGas(kFALSE);
310  peak->SetEnergy(PEAK_THORON_1);
311  break;
312 
313  case kElastic:
314 
315  peak = new KVDBElasticPeak(pic_det->GetName());
316  sys = (KVDBSystem*) GetRun(first)->GetSystem();
317  if (sys) {
318  peak->SetEnergy(sys->GetEbeam() * sys->GetAbeam());
319  ((KVDBElasticPeak*) peak)->SetZproj(sys->GetZbeam());
320  ((KVDBElasticPeak*) peak)->SetAproj(sys->GetAbeam());
321  }
322  break;
323 
324  case kAlpha8MeV:
325 
326  peak = new KVDBAlphaPeak(pic_det->GetName());
327  peak->SetGas(kTRUE);
328  peak->SetEnergy(PEAK_THORON_2);
329  break;
330 
331  case kAlpha6MeV:
332 
333  peak = new KVDBAlphaPeak(pic_det->GetName());
334  peak->SetGas(kTRUE);
335  peak->SetEnergy(PEAK_THORON_1);
336  break;
337 
338  } //what type of peak
339 
340  //test energy of peak if necessary
341  if (peak && peak_energy > 0) {
342  if (typ2 == kElastic) {
343  if (!
344  ((TMath::
345  Abs(peak->GetEnergy() /
346  (((KVDBElasticPeak*) peak)->GetAproj()) -
347  peak_energy))
348  <= 0.1 * (peak->GetEnergy() /
349  ((KVDBElasticPeak*) peak)->GetAproj())))
350  continue; //read next line
351  }
352  else {
353  if (!
354  (TMath::Abs(peak->GetEnergy() - peak_energy) <=
355  0.1 * peak->GetEnergy()))
356  continue;
357  }
358  }
359 
360  if (peak) {
361  peak->SetSigType(KVINDRA::SignalTypes[sign]);
362  peak->SetRing(cour);
363  peak->SetModule(modu);
364  peak->SetParameters(mean, error, sigma, (Double_t) entries,
365  constante, (Double_t) first,
366  (Double_t) last);
367  peak->SetParamName(0, "Mean");
368  peak->SetParamName(1, "Error");
369  peak->SetParamName(2, "Sigma");
370  peak->SetParamName(3, "Peak integral (entries)");
371  peak->SetParamName(4, "Gaussian amplitude");
372  peak->SetParamName(5, "First run");
373  peak->SetParamName(6, "Last run");
374  peak_list->Add(peak);
375 
376  //Set gain associated with peak.
377  //This is the gain of the detector during the first run used to trace the peak.
378  KVDBRun* kvrun = (KVDBRun*) GetRun(first);
379  KVRList* param_list = kvrun->GetLinks("Gains");
380  if (!param_list) {
381  //no gains defined - everybody has gain=1
382  peak->SetGain(1.00);
383  }
384  else {
385  KVDBParameterSet* kvdbps =
386  (KVDBParameterSet*) param_list->
387  FindObjectByName(pic_det->GetName());
388  if (!kvdbps) {
389  //no gain defined for this detector for this run - gain=1
390  peak->SetGain(1.00);
391  }
392  else {
393  peak->SetGain(kvdbps->GetParameter(0));
394  }
395  }
396  } //if (peak)
397 
398  } //parameters correctly read
399 
400  } //Run range found
401 
402  } //while( .good() )
403 
404  CloseCalibrationPeakFile();
405  return peak_list;
406 }

472 void KVINDRADB::ReadGainList()
473 {
474 
475  // Read the file listing any detectors whose gain value changes during exeriment
476 
477  ifstream fin;
478  if (!OpenCalibFile("Gains", fin)) {
479  Error("ReadGainList()", "Could not open file %s",
480  GetCalibFileName("Gains"));
481  return;
482  }
483  Info("ReadGainList()", "Reading gains ...");
484 
485  //Add table for gains
486  fGains = AddTable("Gains", "Gains of detectors during runs");
487 
488  TString sline;
489 
490  UInt_t frun = 0, lrun = 0;
491  UInt_t run_ranges[MAX_NUM_RUN_RANGES][2];
492  UInt_t rr_number = 0;
493  Bool_t prev_rr = kFALSE; // was the last line a run range indication ?
494 
495  Float_t gain;
496 
497  Char_t det_name[80];
498  KVDBParameterSet* parset = 0;
499  TList* par_list = new TList();
500 
501  while (fin.good()) { //reading the file
502  sline.ReadLine(fin);
503  if (fin.eof()) { //fin du fichier
504  LinkListToRunRanges(par_list, rr_number, run_ranges);
505  par_list->Clear();
506  delete par_list;
507  fin.close();
508  return;
509  }
510  if (sline.BeginsWith("Run Range :")) { // Run Range found
511  if (!prev_rr) { // new run ranges set
512  if (par_list->GetSize() > 0)
513  LinkListToRunRanges(par_list, rr_number, run_ranges);
514  par_list->Clear();
515  rr_number = 0;
516  }
517  if (sscanf(sline.Data(), "Run Range : %u %u", &frun, &lrun) != 2) {
518  Warning("ReadGainList()",
519  "Bad format in line :\n%s\nUnable to read run range values",
520  sline.Data());
521  cout << "sscanf=" << sscanf(sline.Data(), "Run Range : %u %u",
522  &frun, &lrun) << endl;
523  }
524  else {
525  prev_rr = kTRUE;
526  run_ranges[rr_number][0] = frun;
527  run_ranges[rr_number][1] = lrun;
528  rr_number++;
529  if (rr_number == MAX_NUM_RUN_RANGES) {
530  Error("ReadGainList", "Too many run ranges (>%d)",
531  rr_number);
532  rr_number--;
533  }
534  }
535  } //Run Range found
536  else if (sline.Sizeof() > 1 && !sline.BeginsWith("#")) { //non void nor comment line
537  if (sscanf(sline.Data(), "%7s %f", det_name, &gain) != 2) {
538  Warning("ReadGainList()",
539  "Bad format in line :\n%s\nUnable to read",
540  sline.Data());
541  }
542  else { //parameters correctly read
543  parset = new KVDBParameterSet(det_name, "Gains", 1);
544  parset->SetParameters((Double_t) gain);
545  prev_rr = kFALSE;
546  fGains->AddRecord(parset);
547  par_list->Add(parset);
548  } //parameters correctly read
549  } //non void nor comment line
550  } //reading the file
551  delete par_list;
552  fin.close();
553 }

573 void KVINDRADB::ReadChIoPressures()
574 {
575  //Read ChIo pressures for different run ranges and enter into database.
576  //Format of file is:
577  //
578  //# some comments
579  //#which start with '#'
580  //Run Range : 6001 6018
581  //ChIos 2_3 50.0
582  //ChIos 4_5 50.0
583  //ChIos 6_7 50.0
584  //ChIos 8_12 30.0
585  //ChIos 13_17 30.0
586  //
587  //Pressures (of C3F8) are given in mbar).
588 
589  ifstream fin;
590  if (!OpenCalibFile("Pressures", fin)) {
591  Error("ReadChIoPressures()", "Could not open file %s",
592  GetCalibFileName("Pressures"));
593  return;
594  }
595  Info("ReadChIoPressures()", "Reading ChIo pressures parameters...");
596 
597  TString sline;
598  UInt_t frun = 0, lrun = 0;
599  UInt_t run_ranges[MAX_NUM_RUN_RANGES][2];
600  UInt_t rr_number = 0;
601  Bool_t prev_rr = kFALSE; //was the previous line a run range indication ?
602  Bool_t read_pressure = kFALSE; // have we read any pressures recently ?
603 
604  KVDBChIoPressures* parset = 0;
605  TList* par_list = new TList();
606 
607  //any ChIo not in list is assumed absent (pressure = 0)
608  Float_t pressure[5] = { 0, 0, 0, 0, 0 };
609 
610  while (fin.good()) { // parcours du fichier
611 
612  sline.ReadLine(fin);
613  if (sline.BeginsWith("Run Range :")) { // run range found
614  if (!prev_rr) { // New set of run ranges to read
615 
616  //have we just finished reading some pressures ?
617  if (read_pressure) {
618  parset = new KVDBChIoPressures(pressure);
619  GetTable("ChIo Pressures")->AddRecord(parset);
620  par_list->Add(parset);
621  LinkListToRunRanges(par_list, rr_number, run_ranges);
622  par_list->Clear();
623  for (int zz = 0; zz < 5; zz++) pressure[zz] = 0.;
624  read_pressure = kFALSE;
625  }
626  rr_number = 0;
627 
628  }
629  if (sscanf(sline.Data(), "Run Range : %u %u", &frun, &lrun) != 2) {
630  Warning("ReadChIoPressures()",
631  "Bad format in line :\n%s\nUnable to read run range values",
632  sline.Data());
633  }
634  else {
635  prev_rr = kTRUE;
636  run_ranges[rr_number][0] = frun;
637  run_ranges[rr_number][1] = lrun;
638  rr_number++;
639  if (rr_number == MAX_NUM_RUN_RANGES) {
640  Error("ReadChIoPressures", "Too many run ranges (>%d)",
641  rr_number);
642  rr_number--;
643  }
644  }
645  } // Run Range found
646  if (fin.eof()) { //fin du fichier
647  //have we just finished reading some pressures ?
648  if (read_pressure) {
649  parset = new KVDBChIoPressures(pressure);
650  GetTable("ChIo Pressures")->AddRecord(parset);
651  par_list->Add(parset);
652  LinkListToRunRanges(par_list, rr_number, run_ranges);
653  par_list->Clear();
654  for (int zz = 0; zz < 5; zz++) pressure[zz] = 0.;
655  read_pressure = kFALSE;
656  }
657  }
658  if (sline.BeginsWith("ChIos")) { //line with chio pressure data
659 
660  prev_rr = kFALSE;
661 
662  //take off 'ChIos' and any leading whitespace
663  sline.Remove(0, 5);
664  sline.Strip(TString::kLeading);
665  //split up ChIo ring numbers and pressure
666  TObjArray* toks = sline.Tokenize(' ');
667  TString chio = ((TObjString*)(*toks)[0])->String();
668  KVString press = ((TObjString*)(*toks)[1])->String();
669  delete toks;
670 
671  read_pressure = kTRUE;
672 
673  if (chio == "2_3") pressure[0] = press.Atof();
674  else if (chio == "4_5") pressure[1] = press.Atof();
675  else if (chio == "6_7") pressure[2] = press.Atof();
676  else if (chio == "8_12") pressure[3] = press.Atof();
677  else if (chio == "13_17") pressure[4] = press.Atof();
678  else read_pressure = kFALSE;
679 
680  } //line with ChIo pressure data
681  } //parcours du fichier
682  delete par_list;
683  fin.close();
684 }

729 Double_t KVINDRADB::GetEventCrossSection(Int_t run1, Int_t run2,
730  Double_t Q_apres_cible,
731  Double_t Coul_par_top) const
732 {
733  // Returns calculated average cross-section [mb] per event for the runs in question.
734  // It is assumed that all runs correspond to the same reaction,
735  // with the same beam & target characteristics and multiplicity trigger.
736  // The target thickness etc. are taken from the first run.
737 
738  KVTarget* targ = GetRun(run1)->GetTarget();
739  if (!targ) {
740  Error("GetEventCrossSection", "No target for run %d", run1);
741  return 0;
742  }
743  Double_t sum_xsec = 0;
744  for (int run = run1; run <= run2; run++) {
745 
746  if (!GetRun(run))
747  continue; //skip non-existent runs
748  sum_xsec +=
749  GetRun(run)->GetNIncidentIons(Q_apres_cible,
750  Coul_par_top) * (1. - GetRun(run)->GetTempsMort());
751  }
752  //average X-section [mb] per event = 1e27 / (no. atoms in target * SUM(no. of projectile nuclei * (1 - TM)) )
753  return (1.e27 / (targ->GetAtomsPerCM2() * sum_xsec));
754 }

764 Double_t KVINDRADB::GetTotalCrossSection(Int_t run1, Int_t run2,
765  Double_t Q_apres_cible,
766  Double_t Coul_par_top) const
767 {
768  //Returns calculated total measured cross-section [mb] for the runs in question.
769  //This is SUM (GetEventCrossSection(run1,run2) * SUM( events )

792 Double_t KVINDRADB::GetEventCrossSection(const Char_t* system_name,
793  Int_t mult_trig,
794  Double_t Q_apres_cible,
795  Double_t Coul_par_top) const
796 {
797  //Returns calculated average cross-section [mb] per event for all runs of the named system* with trigger multiplicity 'mul_trig'.
798  // *to see the list of all system names, use gIndraDB->GetSystems()->ls()
799  //See KVINDRADBRun::GetNIncidentIons() for meaning of other arguments
800 
801  KVDBSystem* system = GetSystem(system_name);
802  if (!system) {
803  Error("GetEventCrossSection",
804  "System %s unknown. Check list of systems (gIndraDB->GetSystems()->ls()).",
805  system_name);
806  return 0.;
807  }
808  KVTarget* targ = system->GetTarget();
809  if (!targ) {
810  Error("GetEventCrossSection", "No target defined for system %s",
811  system_name);
812  return 0;
813  }
814  //loop over all runs of system, only using those with correct trigger multiplicity
815  Double_t sum_xsec = 0;
816  TIter next_run(system->GetRuns());
817  KVINDRADBRun* run;
818  while ((run = (KVINDRADBRun*) next_run())) {
819 
820  if (run->GetTrigger() != mult_trig)
821  continue; //skip runs with bad trigger
822  sum_xsec +=
823  run->GetNIncidentIons(Q_apres_cible,
824  Coul_par_top) * (1. - run->GetTempsMort());
825 
826  }
827  //average X-section [mb] per event = 1e27 / (no. atoms in target * SUM(no. of projectile nuclei * (1 - temps mort)))
828  return (1.e27 / (targ->GetAtomsPerCM2() * sum_xsec));
829 }

1852  Double_t Q_apres_cible,
1853  Double_t Coul_par_top) const
1854 {
1855  // Returns calculated average cross-section [mb] per event for the runs in question.
1856  // It is assumed that all runs correspond to the same reaction,
1857  // with the same beam & target characteristics and multiplicity trigger.
1858  // The target thickness etc. are taken from the first run.
1859 
1860  runs.Begin();
1861  Int_t run1 = runs.Next();

1863  if (!targ) {
1864  Error("GetEventCrossSection", "No target for run %d", run1);
1865  return 0;
1866  }
1867  Double_t sum_xsec = 0;
1868  runs.Begin();
1869  while (!runs.End()) {
1870 
1871  int run = runs.Next();
1872  if (!GetRun(run))
1873  continue; //skip non-existent runs
1874  sum_xsec +=
1875  GetRun(run)->GetNIncidentIons(Q_apres_cible,
1876  Coul_par_top) * (1. - GetRun(run)->GetTempsMort());
1877  }

1880  return (1.e27 / (targ->GetAtomsPerCM2() * sum_xsec));
1881 }

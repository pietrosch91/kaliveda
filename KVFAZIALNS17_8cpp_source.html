73 void KVFAZIALNS17::RutherfordTelescope()
74 {
75  // Telescope for elastic scattering monitoring
76  // Two 5mm diameter silicon detectors of 525um thickness
77  // placed 2m20 from the target at theta=1.84deg phi=-90deg.
78  // distance between centres of detectors = 1mm
79 
80  KVMaterial silicon("Si");
81 
82  const double radius = 9 * KVUnits::mm / 2.;
83 
84  const double thick = 525 * KVUnits::um;
85  const double centre_dist = 1 * KVUnits::mm;
86  double total_thickness = thick + centre_dist;
87 
88  TGeoVolume* si_det = gGeoManager->MakeTube("DET_SI", silicon.GetGeoMedium(), 0., radius, thick / 2);
89 
90  TGeoVolumeAssembly* ruth_tel = gGeoManager->MakeVolumeAssembly("STRUCT_RUTH");
91 
92  ruth_tel->AddNode(si_det, 1, new TGeoTranslation(0, 0, -centre_dist / 2));
93  ruth_tel->AddNode(si_det, 2, new TGeoTranslation(0, 0, centre_dist / 2));
94 
95  // front entrance of first detector at 2 metres from target
96  const double distance = 2.27 * KVUnits::m + 0.5 * total_thickness;
97  const double theta = 0.99;
98  const double phi = -54.07;
99 
100  TGeoRotation rot1, rot2;
101  rot2.SetAngles(phi + 90, theta, 0);
102  rot1.SetAngles(-90, 0., 0.);
103  TGeoTranslation trans(0, 0, distance);
104  TGeoHMatrix h = rot2 * trans * rot1;
105  gGeoManager->GetTopVolume()->AddNode(ruth_tel, 1, new TGeoHMatrix(h));
106 }

117 void KVFAZIALNS17::BuildFAZIA()
118 {
119  //Build geometry of FAZIASYM
120  //All telescopes are : Si(300µm)-Si(500µm)-CsI(10cm)
121  //No attempt has been made to implement real thicknesses
122  //
123  Info("BuildFAZIA", "Compact geometry, %f cm from target + 2BLK around 20 degrees",
124  fFDist);
125 
126  TGeoVolume* top = gGeoManager->GetTopVolume();
127 
128  Double_t distance_block_cible = fFDist * KVUnits::cm;
129  Double_t thick_si1 = 300 * KVUnits::um;
130  TGeoTranslation trans;
131  trans.SetDz(distance_block_cible + thick_si1 / 2.);
132 
133  KVFAZIABlock* block = new KVFAZIABlock;
134 
135  TGeoRotation rot1, rot2;
136  TGeoHMatrix h;
137  TGeoHMatrix* ph = 0;
138  Double_t theta = 0;
139  Double_t phi = 0;
140 
141  Double_t theta_min = fFThetaMin;//smallest lab polar angle in degrees
142  Double_t centre_hole = 2.*tan(theta_min * TMath::DegToRad()) * distance_block_cible;
143  Double_t dx = (block->GetTotalSideWithBlindage()) / 2.;
144 
145  TVector3 centre;
146  for (Int_t bb = 0; bb < fNblocks; bb += 1) {
147 
148  if (bb == 1) centre.SetXYZ(-1 * (dx - centre_hole / 2), -dx - centre_hole / 2, distance_block_cible);
149  else if (bb == 2) centre.SetXYZ(-1 * (dx + centre_hole / 2), dx - centre_hole / 2, distance_block_cible);
150  else if (bb == 3) centre.SetXYZ(-1 * (-dx + centre_hole / 2), dx + centre_hole / 2, distance_block_cible);
151  else if (bb == 0) centre.SetXYZ(-1 * (-dx - centre_hole / 2), -dx + centre_hole / 2, distance_block_cible);
152  else if (bb == 4) centre.SetXYZ(0, -25, distance_block_cible);
153  else if (bb == 5) centre.SetXYZ(0, 25, distance_block_cible);
154  else {
155  Warning("BuildFAZIA", "Block position definition is done only for %d blocks", fNblocks);
156  }
157  theta = centre.Theta() * TMath::RadToDeg();
158  phi = centre.Phi() * TMath::RadToDeg();
159  printf("BLK #%d => theta=%1.2lf - phi=%1.2lf\n", bb, theta, phi);
160 
161  rot2.SetAngles(phi + 90., theta, 0.);
162  rot1.SetAngles(-1.*phi, 0., 0.);
163  h = rot2 * trans * rot1;
164  ph = new TGeoHMatrix(h);
165  top->AddNode(block, bb, ph);
166  }
167 
168  // add telescope for elastic scattering monitoring
169  RutherfordTelescope();
170  // Change default geometry import angular range for rutherford telescope
171  SetGeometryImportParameters(.25, 1., 0.5) ;
172 }

149  else if (nameat == "FastShaperRiseTime") SetFastShaperRiseTime(par->GetParameters()->GetDoubleValue(ii));
150  else if (nameat == "FastShaperFlatTop") SetFastShaperFlatTop(par->GetParameters()->GetDoubleValue(ii));
151  else if (nameat == "TauRC") SetTauRC(par->GetParameters()->GetDoubleValue(ii));
152  else if (nameat == "MinimumAmplitude") SetAmplitudeTriggerValue(par->GetParameters()->GetDoubleValue(ii));
153  else if (nameat == "InterpolatedChannelWidth") SetInterpolatedChannelWidth(par->GetParameters()->GetDoubleValue(ii));
154  else if (nameat == "Interpolation") SetInterpolation((par->GetParameters()->GetDoubleValue(ii) == 1));
155  else if (nameat == "PZCorrection") SetPoleZeroCorrection((par->GetParameters()->GetDoubleValue(ii) == 1));
156  else {
157  if (nameat == "Detector" || nameat == "Signal" || nameat == "RunRange") {

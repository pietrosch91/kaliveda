32 /*
33 <h2>KVEvent</h2>
34 <h4>Base class for all types of multiparticle event consisting of KVNucleus objects</h4>
35 
36 Particles are stored in a TClonesArray and KVEvent provides some basic
37 functionality for accessing and manipulating the list.
38 
39 Events can be built using any class derived from KVNucleus to represent particles.
40 These classes can allocate memory in their default ctor: when filling events in a loop,
41 the same 'particle' objects are re-used for each new event, the ctor of each object will
42 only be called once when the object is first created (e.g. in the first event).
43 The particle class Clear() method will be called before each new event.
44 Therefore the cycle of use of the particle objects in a loop over many events is:
45 
46 ~~~~~~~~~~~~~~~
47 <particle ctor>
48  Building 1st event
49  <particle Clear()>
50  Building 2nd event
51  <particle Clear()>
52  ...
53  Building last event
54 <particle dtor>
55 ~~~~~~~~~~~~~~~
56 
57 When writing events in a TTree, it is very important to call the TBranch::SetAutoDelete(kFALSE)
58 method of the branch which is used to store the event object.
59 
60 If not, when the events are read back, the KVEvent constructor and destructor will be called
61 every time an event is read from the TTree meading to very slow reading times (& probably
62 memory leaks).
63 
64 For this reason we provide the method:
65 
66 ~~~~~~~~~~~~~~~
67  void MakeEventBranch(TTree*, const TString&, const TString&, void*)
68 ~~~~~~~~~~~~~~~
69 
70 which should be used whenever it is required to stock KVEvent-derived objects in a TTree.
71 */

219 KVNucleus* KVEvent::AddParticle()
220 {
221  // Method used for building an event particle by particle.
222  // DO NOT USE FOR READING EVENTS - use GetParticle(Int_t npart)!!
223  //
224  // This method increases the multiplicity fMult by one and "creates" a new particle with index (fMult-1).
225  // In actual fact a new object is only created if needed i.e. if the new multiplicity is greater
226  // than previously. Particle objects in the array are reused from one event to another.
227  // The default constructor for the class corresponding to the "new" particle will only be called
228  // once during its lifetime (i.e. if N events are generated, the particle ctor will be called only once,
229  // not N times). Once created, in subsequent events we just call the particle's Clear() method
230  // in order to reset its internal variables ready for a new event.
231 
232  Int_t mult = GetMult();
233 #ifdef __WITHOUT_TCA_CONSTRUCTED_AT
234  KVNucleus* tmp = (KVNucleus*) ConstructedAt(mult, "C");
235 #else
236  KVNucleus* tmp = (KVNucleus*) fParticles->ConstructedAt(mult, "C");
237 #endif
238  if (!tmp) {
239  Error("AddParticle", "Allocation failure, Mult=%d", mult);
240  return 0;
241  }
242  return tmp;
243 }

339 Int_t KVEvent::GetMult(Option_t* opt)
340 {
341  //Returns multiplicity (number of particles) of event.
342  //If opt = "" (default), returns number of particles in TClonesArray* fParticles
343  // i.e. the value of fParticles->GetEntriesFast() (we assume there are no gaps
344  // in the list)
345  //If opt = "ok" only particles with IsOK()==kTRUE are included.
346  //If opt = "name" only particles with GetName()=="name" are included.
347  //
348  //IN THE LATTER TWO CASES, YOU MUST NOT USE THIS METHOD INSIDE A LOOP
349  //OVER THE EVENT USING GETNEXTPARTICLE() !!!
350 
351  TString Opt(opt);
352  Opt.ToUpper();
353  Int_t fMultOK = 0;
354  if (Opt == "") {
355  //get total multiplicity of event
356  return fParticles->GetEntriesFast();
357  } else {
358  KVNucleus* tmp = 0;
359  ResetGetNextParticle();
360  while ((tmp = GetNextParticle(opt)))
361  fMultOK++;
362  }
363  return fMultOK;
364 }

381 Double_t KVEvent::GetSum(const Char_t* KVNucleus_method, Option_t* opt)
382 {
383  //Returns sum over particles of the observable given by the indicated KVNucleus_method
384  //for example
385  //if the method is called this way GetSum("GetZ"), it returns the sum of the charge
386  //of particles in the current event
387  //
388  //If opt = "ok" only particles with IsOK()==kTRUE are considered.
389  //If opt = "name" only particles with GetName()=="name" are considered.
390  //
391  //IN ANY CASE, YOU MUST NOT USE THIS METHOD INSIDE A LOOP
392  //OVER THE EVENT USING GETNEXTPARTICLE() !!!
393 
394  TString Opt(opt);
395  Opt.ToUpper();
396  Double_t fSum = 0;
397  TMethodCall mt;
398  mt.InitWithPrototype(KVNucleus::Class(), KVNucleus_method, "");
399 
400  if (mt.IsValid()) {
401  ResetGetNextParticle();
402  if (mt.ReturnType() == TMethodCall::kLong) {
403  Long_t ret;
404  KVNucleus* tmp;
405  while ((tmp = GetNextParticle(Opt))) {
406  mt.Execute(tmp, "", ret);
407  fSum += ret;
408  }
409  } else if (mt.ReturnType() == TMethodCall::kDouble) {
410  Double_t ret;
411  KVNucleus* tmp;
412  while ((tmp = GetNextParticle(Opt))) {
413  mt.Execute(tmp, "", ret);
414  fSum += ret;
415  }
416  }
417  }
418 
419  return fSum;

605 void KVEvent::GetMultiplicities(Int_t mult[], const TString& species)
606 {
607  // Fill array mult[] with the number of each nuclear species in the
608  // comma-separated list in this event. Make sure that mult[] is
609  // large enough for the list.
610  //
611  // Example:
612  // Int_t mult[4];
613  // event.GetMultiplicities(mult, "1n,1H,2H,3H");
614  //
615  // N.B. the species name must correspond to that given by KVNucleus::GetSymbol
616 
617  TObjArray* spec = species.Tokenize(", ");// remove any spaces
618  Int_t nspec = spec->GetEntries();
619  memset(mult, 0, nspec * sizeof(Int_t)); // set multiplicities to zero
620  KVNucleus* nuc;
621  while ((nuc = GetNextParticle())) {
622  for (int i = 0; i < nspec; i++) {
623  if (((TObjString*)(*spec)[i])->String() == nuc->GetSymbol()) mult[i] += 1;
624  }
625  }
626  delete spec;

656 KVNucleus* KVEvent::GetNextParticle(Option_t* opt)
657 {
658  // Use this method to iterate over the list of particles in the event
659  // After the last particle GetNextParticle() returns a null pointer and
660  // resets itself ready for a new iteration over the particle list.
661  //
662  // If opt="" all particles are included in the iteration.
663  // If opt="ok" or "OK" only particles whose IsOK() method returns kTRUE are included.
664  //
665  // Any other value of opt is interpreted as a (case-insensitive) particle group name: only
666  // particles with BelongsToGroup(opt) returning kTRUE are included.
667  //
668  // If you want to start from the beginning again before getting to the end
669  // of the list, especially if you want to change the selection criteria,
670  // call method ResetGetNextParticle() before continuing.
671  // If you interrupt an iteration before the end, then start another iteration
672  // without calling ResetGetNextParticle(), even if you change the argument of
673  // the call to GetNextParticle(), you will repeat exactly the same iteration
674  // as the previous one.
675  //
676  // WARNING: Only one iteration at a time over the event can be performed
677  // using this method. If you want/need to perform several i.e. nested
678  // iterations, use the KVEvent::Iterator class
679 
680  TString Opt(opt);
681  Opt.ToUpper();
682 
683  // continue iteration
684  if (fIter.IsIterating()) return &(*(fIter++));
685 
686  // start new iteration
687  Bool_t ok_iter = (Opt == "OK");
688  Bool_t grp_iter = (!ok_iter && Opt.Length());
689 
690 #ifdef WITH_CPP11
691  if (ok_iter) fIter = Iterator(this, Iterator::Type::OK);
692  else if (grp_iter) fIter = Iterator(this, Iterator::Type::Group, Opt);
693  else fIter = Iterator(this, Iterator::Type::All);
694 #else
695  if (ok_iter) fIter = Iterator(this, Iterator::OK);
696  else if (grp_iter) fIter = Iterator(this, Iterator::Group, Opt);
697  else fIter = Iterator(this, Iterator::All);
698 #endif
699  return &(*(fIter++));
700 }

780 void KVEvent::ResetEnergies()
781 {
782  //Used for simulated events after "detection" by some multidetector array.
783  //
784  //The passage of the event's particles through the different absorbers modifies their
785  //kinetic energies, indeed all those which are correctly identified by the detector
786  //actually stop. Calling this method will reset all the particles' energies to their
787  //initial value i.e. before they entered the first absorber.
788  //Particles which have not encountered any absorbers/detectors are left as they are.
789  KVNucleus* part = 0;
790  ResetGetNextParticle();
791  while ((part = GetNextParticle())) {
792  part->ResetEnergy();
793  }

1321 Double_t KVEvent::GetChannelQValue() const
1322 {
1323  // Calculate the Q-value [MeV] for this event as if all nuclei were produced by
1324  // the decay of an initial compound nucleus containing the sum of all nuclei
1325  // in the event, i.e.
1326  // A -> a1 + a2 + a3 + ...
1327  // We take into account any excitation energy of the nuclei of the event
1328  // (see GetGSChannelQValue() for an alternative), i.e. we calculate
1329  // Q = M(A) - ( m(a1) + m(a2) + m(a3) + ... )
1330  // where
1331  // M(X) = ground state mass of X
1332  // m(X) = M(X) + E*(X)
1333  // If Q<0, the excitation energy of the initial compound nucleus, A,
1334  // would have to be at least equal to (-Q) in order for the decay to occur.
1335  // i.e. decay is possible if
1336  // E*(A) > -Q
1337 
1338  Double_t sumM = 0;
1339  KVNucleus CN;
1340  Int_t M = const_cast<KVEvent*>(this)->GetMult();
1341  for (int i = 1; i <= M; i++) {
1342  sumM += GetParticle(i)->GetMass();
1343  CN += *(GetParticle(i));
1344  }
1345  return CN.GetMassGS() - sumM;
1346 }

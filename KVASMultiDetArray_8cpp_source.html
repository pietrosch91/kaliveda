311 void KVASMultiDetArray::CalculateGroupsFromGeometry()
312 {
313  ClearStructures("GROUP"); // clear out (delete) old groups
314 
315  const KVSeqCollection* fLayers = GetStructures();
316  ((KVUniqueNameList*)fLayers)->Sort();
317 
318  TIter nxtlay1(fLayers);
319  KVLayer* l1;
320  if (fLayers->GetSize() > 1) {
321  fGr = 0; // for numbering groups
322  TIter nxtlay2(fLayers);
323  KVLayer* l2;
324 
325  while ((l1 = (KVLayer*) nxtlay1())) { // loop over layers
326  while ((l2 = (KVLayer*) nxtlay2())) { // loop over layers
327  if ((l1->GetNumber() < l2->GetNumber())
328  && ((UInt_t) l1->GetNumber() <
329  (UInt_t) fLayers->GetSize()))
330  SetGroups(l1, l2); // compare all telescopes in different layers
331  }
332  nxtlay2.Reset();
333  }
334  } //if(fLayers->GetSize()>1)
335 
336  // Finally, create groups for all orphan telescopes which are not in any existing
337  // group
338 
339  nxtlay1.Reset(); // reset loop over layers
340  while ((l1 = (KVLayer*) nxtlay1())) { // loop over layers
341  if (l1->GetStructures()) {
342  TIter nxtrng(l1->GetStructures());
343  KVRing* robj;
344  while ((robj = (KVRing*) nxtrng())) { // loop over rings
345  TIter nxtscp(robj->GetTelescopes());
346  KVTelescope* tobj;
347  while ((tobj = (KVTelescope*) nxtscp())) { // loop over telescopes
348  if (!tobj->GetParentStructure("GROUP")) { // orphan telescope
349 
350  KVASGroup* kvg = new KVASGroup();
351  kvg->SetNumber(++fGr);
352  kvg->Add(tobj);
353  //group dimensions determined by detector dimensions
354  kvg->SetPolarMinMax(tobj->GetThetaMin(),
355  tobj->GetThetaMax());
356  kvg->SetAzimuthalMinMax(tobj->GetPhiMin(),
357  tobj->GetPhiMax());
358  Add(kvg);
359  }
360  }
361  }
362  }
363  }

374 void KVASMultiDetArray::AddToGroups(KVTelescope* kt1, KVTelescope* kt2)
375 {
376 // The two telescopes are in angular overlap.
377 // a) if neither are in groups already, create a new group to which they both belong
378 // b) if one of them is in a group already, add the orphan telescope to it
379 // c) if both are in groups already, merge the two groups
380  KVASGroup* kvg;
381 
382  if (!kt1->GetParentStructure("GROUP") && !kt2->GetParentStructure("GROUP")) { // case a)
383 #ifdef KV_DEBUG
384  cout << "Making new Group from " << kt1->
385  GetName() << " and " << kt2->GetName() << endl;
386 #endif
387  kvg = new KVASGroup();
388  kvg->SetNumber(++fGr);
389  kvg->Add(kt1);
390  kvg->Add(kt2);
391  kvg->Sort(); // sort telescopes in list
392  kvg->SetDimensions(kt1, kt2); // set group dimensions from telescopes
393  //add to list
394  Add(kvg);
395  } else if ((kvg = (KVASGroup*)(kt1->GetParentStructure("GROUP")))
396  && !kt2->GetParentStructure("GROUP")) { // case b) - kt1 is already in a group
397 #ifdef KV_DEBUG
398  cout << "Adding " << kt2->GetName() << " to group " << kvg->
399  GetNumber() << endl;
400 #endif
401  //add kt2 to group
402  kvg->Add(kt2);
403  kvg->Sort(); // sort telescopes
404  kvg->SetDimensions(kvg, kt2); //adjust dimensions depending on kt2
405  } else if ((kvg = (KVASGroup*)(kt2->GetParentStructure("GROUP")))
406  && !kt1->GetParentStructure("GROUP")) { // case b) - kt2 is already in a group
407 #ifdef KV_DEBUG
408  cout << "Adding " << kt1->GetName() << " to group " << kvg->
409  GetNumber() << endl;
410 #endif
411  //add kt1 to group
412  kvg->Add(kt1);
413  kvg->Sort(); // sort telescopes
414  kvg->SetDimensions(kvg, kt1); //adjust dimensions depending on kt1
415  } else if (kt1->GetParentStructure("GROUP") != kt2->GetParentStructure("GROUP")) { //both telescopes already in different groups
416 #ifdef KV_DEBUG
417 // cout << "Merging " << kt1->GetGroup()->
418 // GetNumber() << " and " << kt2->GetGroup()->GetNumber() << endl;
419 // cout << "because of " << kt1->GetName() << " and " << kt2->
420 // GetName() << endl;
421 #endif
422  MergeGroups((KVASGroup*)kt1->GetParentStructure("GROUP"), (KVASGroup*)kt2->GetParentStructure("GROUP"));
423  }

494 KVNameValueList* KVASMultiDetArray::DetectParticle_KV(KVNucleus* part)
495 {
496 // Simulate detection of a single particle (nucleus) by multidetector array.
497 // We look for the group in the array that the particle will hit
498 //
499 // If a group is found the detection of this particle
500 // by the different members of the group is simulated.
501 // The detectors concerned have their fEloss members set to the energy lost by the
502 // particle when it crosses them.
503 //
504 // Returns a list (KVNameValueList pointer) of the crossed detectors with their name and energy loss
505 // if particle hits detector in array, 0 if not (i.e. particle
506 // in beam pipe or dead zone of the multidetector)
507 // INFO User has to delete the KVNameValueList after its use
508 //
509  //find group in array hit by particle
510  KVASGroup* grp_tch = (KVASGroup*)GetGroupWithAngles(part->GetTheta(), part->GetPhi());
511  if (grp_tch) {
512  //simulate detection of particle by this group
513  KVNameValueList* nvl = grp_tch->DetectParticle(part);
514  return nvl;
515  }
516  return 0;
517 }

691 {
692 
693  KVASGroup* ASgrp = dynamic_cast<KVASGroup*>(grp);
694  if (!ASgrp) {
695  Fatal("AnalyseGroupAndReconstructedEvent",
696  "All groups in KVASMultiDetArray must derive from KVASGroup! %s is NOT a KVASGroup", grp->GetName());
697  return;
698  }
699  UInt_t nLayers = ASgrp->GetNumberOfLayers();
700 
701  UInt_t initial_hits_in_group = grp->GetHits();
702 
703  if (nLayers > 1) {
704  //multilayer group
705  //Start with layer furthest from target and work inwards (but don't look at layer
706  //nearest to target)
707  for (UInt_t i = ASgrp->GetLayerFurthestTarget();
708  i > ASgrp->GetLayerNearestTarget(); i--) {
709  TList* kvtl = ASgrp->GetTelescopesInLayer(i);
710  if (kvtl) {
711  AnalyseTelescopes(event, kvtl);
712  delete kvtl;
713  }
714  }
715 
716 
717  //if nothing has been found, then check for particles stopping in layer nearest target
718  if (grp->GetHits() == initial_hits_in_group) {
719  TList* kvtl =
720  ASgrp->GetTelescopesInLayer(ASgrp->GetLayerNearestTarget());
721  if (kvtl) {
722  AnalyseTelescopes(event, kvtl);
723  delete kvtl;
724  }
725  }
726 
727  } else {
728  //single layer group
729 #ifdef KV_DEBUG
730  Info("AnalyseGroup", "Single layer group");
731 #endif
732  //for a single layer group we should have
733  //kvg->GetLayerNearestTarget() = kvg->GetLayerFurthestTarget()
734  //so we can use either one as argument for kvg->GetTelescopesInLayer
735  TList* kvtl =
736  ASgrp->GetTelescopesInLayer(ASgrp->GetLayerNearestTarget());
737  if (kvtl) {
738  AnalyseTelescopes(event, kvtl);
739  delete kvtl;
740  }
741  }
742 
743 #ifdef KV_DEBUG
744  Info("AnalyseGroup", "OK after analysis of hit groups");
745 #endif
746

748  KVReconstructedNucleus::AnalyseParticlesInGroup(grp);

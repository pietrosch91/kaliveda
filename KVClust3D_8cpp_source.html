298 void KVClust3D::Clusterize()
299 {
300  //Apply clusterization of the 3D histogram
301  //For each cell, with a content greater than the threshold defined before
302  // using the SetThreshold() method, test the content of the closest neighbors (8)
303  // to see if they are connected
304  // At the end, a clone histogram is available, using this method : GetClusterIndex().
305  // The content of its cells corresponds
306  // to the index of each found cluster
307  //
308  Int_t Ntemp = 1;
309  Int_t NCellsTot = 0;
310  Double_t PopTot = 0;
311 
312  ResetObjects();
313 
314 // if (!fCluster)
315 // fCluster = ProduceTH3D("cluster_index");
316 
317  //Remplissage de l histo 3D fCluster :
318  // Contenu de l objet KVClust3D au dessus du seuil -> contenu 1
319  //
320  for (Int_t nx = 1; nx <= GetNbinsX(); nx += 1) {
321  for (Int_t ny = 1; ny <= GetNbinsY(); ny += 1) {
322  for (Int_t nz = 1; nz <= GetNbinsZ(); nz += 1) {
323  if (GetBinContent(nx, ny, nz) >= fThreshold) {
324  fCluster->SetBinContent(nx, ny, nz, 1);
325  NCellsTot += 1;
326  PopTot += GetBinContent(nx, ny, nz);
327  } else {
328  fCluster->SetBinContent(nx, ny, nz, 0);
329  }
330  }
331  }
332  }
333 
334  for (Int_t nx = 2; nx <= GetNbinsX() - 1; nx += 1) {
335  for (Int_t ny = 2; ny <= GetNbinsY() - 1; ny += 1) {
336  for (Int_t nz = 2; nz <= GetNbinsZ() - 1; nz += 1) {
337  Int_t idx = fCluster->GetBinContent(nx, ny, nz);
338  if (idx >= 1) { //Cellule avec la bonne densite
339  if (idx > 1) { //cellule deja affectee
340  Int_t ncv = -1;
341  for (Int_t nxv = nx - fNvoisins; nxv <= nx + fNvoisins; nxv += 1) { //on boucle sur les cellules voisines
342  for (Int_t nyv = ny - fNvoisins; nyv <= ny + fNvoisins; nyv += 1) {
343  for (Int_t nzv = nz - fNvoisins; nzv <= nz + fNvoisins; nzv += 1) {
344  Int_t idxv = fCluster->GetBinContent(nxv, nyv, nzv);
345  //cellule a affilier
346  if (idxv >= 1) {
347  ncv += 1;
348  if (idxv == 1) { //cellule mise dans le cluster idx
349  fNcells->AddAt(1 + fNcells->At(idx), idx);
350  fPop->AddAt(GetBinContent(nxv, nyv, nzv) + fPop->At(idx), idx);
351  fCluster->SetBinContent(nxv, nyv, nzv, idx);
352  } else {
353  // clusters idx et idxv sont voisins
354  fVoisins->Fill(TMath::Min(idx, idxv), TMath::Max(idx, idxv));
355  }
356  } else {
357  //cellule vide
358  //? SetBinContent(nxv,nyv,nzv,0);
359  //fCluster->SetBinContent(nxv,nyv,nzv,0);
360  }
361  }
362  }
363  }
364  if (ncv == 0)
365  printf("gros pb de coherence ... cluster %1.0lf declare avec une cellule isolee\n", fCluster->GetBinContent(nx, ny, nz));
366  } else if (idx == 1) { //cellule pas encore affectee
367  Int_t ncv = -1;
368  KVNumberList nl;
369  for (Int_t nxv = nx - fNvoisins; nxv <= nx + fNvoisins; nxv += 1) { //on boucle sur les cellules voisines
370  for (Int_t nyv = ny - fNvoisins; nyv <= ny + fNvoisins; nyv += 1) {
371  for (Int_t nzv = nz - fNvoisins; nzv <= nz + fNvoisins; nzv += 1) {
372  Int_t idxv = fCluster->GetBinContent(nxv, nyv, nzv);
373  if (idxv >= 1) { //cellule voisine a affilier
374  if (idxv > 1) { // la cellule appartient a un cluster
375  //cellule voisine appartenant a un cluster deja identifie
376  nl.Add(idxv);
377  } else {
378 
379  }
380  ncv += 1;
381  }
382  }
383  }
384  }
385  //au moins une cellule voisine non nulle
386  //
387  if (ncv >= 1) {
388  //les cellules voisines sont associees a un ou plusieur cluster
389  if (nl.GetNValues() >= 1) {
390  //on prend le premier cluster qui vient
391  Int_t newid = nl.First();
392  for (Int_t nxv = nx - fNvoisins; nxv <= nx + fNvoisins; nxv += 1) { //on boucle sur les cellules voisines
393  for (Int_t nyv = ny - fNvoisins; nyv <= ny + fNvoisins; nyv += 1) {
394  for (Int_t nzv = nz - fNvoisins; nzv <= nz + fNvoisins; nzv += 1) {
395  Int_t idxv = fCluster->GetBinContent(nxv, nyv, nzv);
396  //on rajoute les cellules au cluster choisi
397  if (idxv == 1) {
398  fNcells->AddAt(1 + fNcells->At(newid), newid);
399  fPop->AddAt(GetBinContent(nxv, nyv, nzv) + fPop->At(newid), newid);
400  fCluster->SetBinContent(nxv, nyv, nzv, newid);
401  }
402  }
403  }
404  }
405  if (nl.GetNValues() > 1) {
406  //on met les implementes les clusters contigus
407  nl.Begin();
408  while (!nl.End()) {
409  Int_t compt = nl.Next();
410  if (compt != newid) {
411  fVoisins->Fill(TMath::Min(compt, newid), TMath::Max(compt, newid));
412  }
413  }
414  }
415  }
416  //creation d un nouveau cluster
417  else {
418 
419  Ntemp += 1;
420  if (Ntemp > 999)
421  Warning("Clusterize", "Ntemp>999 ... should be some problems in the treatment of neigbouring clusters...");
422  //Info("Clusterize","New cluster %d",Ntemp);
423 
424  if (Ntemp >= fNcells->GetSize()) {
425  fNcells->Set(Ntemp + 1);
426  fPop->Set(Ntemp + 1);
427  }
428  for (Int_t nxv = nx - fNvoisins; nxv <= nx + fNvoisins; nxv += 1) { //on boucle sur les cellules voisines
429  for (Int_t nyv = ny - fNvoisins; nyv <= ny + fNvoisins; nyv += 1) {
430  for (Int_t nzv = nz - fNvoisins; nzv <= nz + fNvoisins; nzv += 1) {
431  Int_t idxv = fCluster->GetBinContent(nxv, nyv, nzv);
432  if (idxv == 1) {
433  fNcells->AddAt(1 + fNcells->At(Ntemp), Ntemp);
434  fPop->AddAt(GetBinContent(nxv, nyv, nzv) + fPop->At(Ntemp), Ntemp);
435  fCluster->SetBinContent(nxv, nyv, nzv, Ntemp);
436  }
437  }
438  }
439  }
440  }
441  } else {
442  if (IsLonelyCellsAreKept()) {
443  Ntemp += 1;
444  if (Ntemp > 999)
445  Warning("Clusterize", "Ntemp>999 ... should be some problems in the treatment of neigbouring clusters...");
446  //Info("Clusterize","Lonely cell %d",Ntemp);
447  if (Ntemp >= fNcells->GetSize()) {
448  fNcells->Set(Ntemp + 1);
449  fPop->Set(Ntemp + 1);
450  }
451  fNcells->AddAt(1 + fNcells->At(Ntemp), Ntemp);
452  fPop->AddAt(GetBinContent(nx, ny, nz) + fPop->At(Ntemp), Ntemp);
453  fCluster->SetBinContent(nx, ny, nz, Ntemp);
454  } else {
455  fCluster->SetBinContent(nx, ny, nz, 0);
456  }
457  }
458  }
459  } else {
460 
461  }
462  }
463  }
464  }
465 
466  //nclus commence a 1
467  //le nombre actuel de cluster est nclus-1
468  //
469  //boucle sur les clusters voisins
470  //on rassemble les clusters avec des cellules contigues
471  //et on decrement en consequence le nombre de cluster
472  //
473  fNclusters = Ntemp - 1;
474 
475  for (Int_t ii = Ntemp; ii >= 2; ii -= 1) {
476  for (Int_t jj = ii + 1; jj <= Ntemp; jj += 1) {
477  Int_t contenu = fVoisins->GetBinContent(ii, jj);
478  if (contenu > 0) {
479  if (fNcells->At(jj) > 0) {
480 
481  fNcells->AddAt(fNcells->At(jj) + fNcells->At(ii), ii);
482  fPop->AddAt(fPop->At(jj) + fPop->At(ii), ii);
483  for (Int_t nx = 1; nx <= GetNbinsX(); nx += 1) {
484  for (Int_t ny = 1; ny <= GetNbinsY(); ny += 1) {
485  for (Int_t nz = 1; nz <= GetNbinsZ(); nz += 1) {

487  fCluster->SetBinContent(nx, ny, nz, ii);
488  }
489  }
490  }
491  fNcells->SetAt(0, jj);
492  fPop->SetAt(0, jj);
493

495  }
496  }
497  }
498  }
499 
500  //on passe les contenus en ngatifs si ils appartiennent a aucun cluster
501  //
502  for (Int_t nx = 1; nx <= GetNbinsX(); nx += 1) {

507  }

510  }

515  // <=0 la cellule appartient a aucun cluster

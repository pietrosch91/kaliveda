174 Bool_t KVFAZIAIDSiPSA::CheckTheoreticalIdentificationThreshold(KVNucleus* ION, Double_t)
175 {
176  // Return kTRUE if energy of ION is > minimum incident energy required for identification
177  // in charge (Z) given by figure 9 of "Comparison of charged particle identification
178  // using pulse shape discrimination and DeltaE-E methods between front and rear injection in
179  // silicon detector" NIM A 701, 145 (2013) (FAZIA collaboration).
180  // Thresholds used are those for rear-injection (red curve, lowest values).
181 
182  Double_t seuil = fZThreshold->Eval(ION->GetZ());
183  return (ION->GetEnergy() >= seuil);
184 }

199 void KVFAZIAIDSiPSA::SetIdentificationStatus(KVReconstructedNucleus* n)
200 {
201  // For filtering simulations
202  // If n->GetEnergy() is above threshold for mass identification, we set
203  // n->IsAMeasured(kTRUE) (and n->IsZMeasured(kTRUE)).
204  // Otherwise, we just set n->IsZMeasured(kTRUE) and use the A given by
205  // the mass formula for the particle
206  //
207  // Z-dependence of A identification:
208  // all ok above threshold if Z<=16, decreasing probability for 17<=Z<=21
209  // no A identification for Z>21
210 
211  n->SetZMeasured();
212  fMassIDProb->SetParameters(18.5, .4);
213  Bool_t okmass = (n->GetZ() < 17) || (n->GetZ() < 22 && gRandom->Uniform() < fMassIDProb->Eval(n->GetZ()));
214  okmass = okmass && (n->GetEnergy() >= fAThreshold->Eval(n->GetZ()));
215  if (okmass) {
216  n->SetAMeasured();
217  } else
218  n->SetZ(n->GetZ());
219 }

104 {

106  //Id should be given as 100*Z + A

153 void KVTGIDZA::AddLineToGrid(KVIDGrid* g, Int_t Z, Int_t npoints,
154  Double_t xmin, Double_t xmax, Bool_t log_scale)
155 {
156  //Add a line to the grid 'g' for identification label 'Z' with 'npoints' points calculated between
157  //X-coordinates xmin and xmax. Points are omitted if the resulting value of the functional is
158  //not a number (TMath::IsNan = kTRUE).
159  //
160  //For Z<=8 several KVIDZALines are added for common isotopes (p,d,t, 3He, etc.)
161  //For Z>8 a line for the current Z is drawn, with the A corresponding to Charity's EAL formula
162  //
163  //Before using this method the user must have set the name of
164  //the parameter in the identification functional which corresponds to the
165  //atomic number Z i.e. using
166  // SetParName(i,"Z")
167  //with i=the relevant parameter index.
168  //
169  // If log_scale=kTRUE (default is kFALSE), more points are used at the beginning
170  // of the line than at the end, with a logarithmic dependence.
171 
172  //A & Z identification
173  //on va faire: p,d,t,3He,4He,6He,6Li,7Li,8Li,7Be,9Be, etc.
174  static Int_t Ziso[] = {
175  1, 1, 1,
176  2, 2, 2,
177  3, 3, 3,
178  4, 4, 4,
179  5, 5, 5,
180  6, 6, 6,
181  7, 7, 7,
182  8, 8, 8
183  };
184  static Int_t Aiso[] = {
185  1, 2, 3,
186  3, 4, 6,
187  6, 7, 8,
188  7, 9, 10,
189  10, 11, 12,
190  12, 13, 14,
191  14, 15, 16,
192  15, 16, 17
193  };
194  static Int_t nisotopes = 24; //number of isotopes in arrays
195  static Int_t Zindex_start[] = { -1, 0, 3, 6, 9, 12, 15, 18, 21 }; //index of first isotope for each Z
196  static Int_t Zindex_stop[] = { -1, 2, 5, 8, 11, 14, 17, 20, 23 }; //index of last isotope for each Z
197 
198  if (Z <= Ziso[nisotopes - 1]) {
199  //Z is included in table of isotopes
200  //we add several lines corresponding to the tabulated isotopes
201  for (int iso = Zindex_start[Z]; iso <= Zindex_stop[Z]; iso++) { //loop over isotopes for this Z
202  //add new line to grid
203  KVIDLine* new_line = AddLine(g);
204 
205  //set identification label for line
206  Double_t idlab = 100. * Ziso[iso] + Aiso[iso];
207  SetIdent(new_line, idlab);
208 
209  //set Z of line in functional
210  SetParameter("Z", (Double_t) Ziso[iso]);
211  //loop over points of line
212  Int_t p_index = 0;
213  Double_t X, dX;
214  Double_t Y = 0.;
215  Double_t logXmin = TMath::Log(TMath::Max(xmin, 1.0));
216  if (log_scale)
217  dX = (TMath::Log(xmax) - logXmin) / (npoints - 1.);
218  else
219  dX = (xmax - xmin) / (Double_t)(npoints - 1);
220 
221  for (Int_t i = 0; i < npoints; i++) {
222 
223  //x coordinate of this point
224  if (log_scale)
225  X = TMath::Exp(logXmin + i * dX);
226  else
227  X = xmin + dX * ((Double_t) i);
228 
229  //leave value Y as it is. The value of GetIDFunc()->Eval(ID)
230  //is the vertical distance delta_Y from point (X,Y) to the line; therefore the
231  //Y coordinate of the point on the line is Y + delta_Y, whatever the value Y.
232 
233  //set values of parameters which correspond to X and Y coordinates in grid
234  SetParameter("X", X);
235  SetParameter("Y", Y);
236 
237  Y += Eval((Double_t) Aiso[iso]);
238 
239  if (!TMath::IsNaN(Y))
240  new_line->SetPoint(p_index++, X, Y);
241  else
242  Y = 0.; //reset Y to 0 if it ever becomes NaN
243  }
244  }
245  } else {
246  //Z is bigger than largest Z for which we have isotopes
247  //we add one line for this Z using mass formula
248 
249  //get mass for nucleus
250  Int_t A = KVNucleus::GetAFromZ(Z, fMassFormula);
251  //add new line to grid
252  KVIDLine* new_line = AddLine(g);
253  //set identification label for line
254  Double_t idlab = 100. * Z + A;
255  SetIdent(new_line, idlab);
256 
257  //set Z of line in functional
258  SetParameter("Z", (Double_t) Z);
259  //loop over points of line
260  Int_t p_index = 0;
261  Double_t Y = 0., X, dX = (xmax - xmin) / (Double_t)(npoints - 1);
262 
263  for (Int_t i = 0; i < npoints; i++) {
264 
265  //x coordinate of this point
266  X = xmin + dX * ((Double_t) i);
267 
268  //leave value Y as it is. The value of GetIDFunc()->Eval(ID)
269  //is the vertical distance delta_Y from point (X,Y) to the line; therefore the
270  //Y coordinate of the point on the line is Y + delta_Y, whatever the value Y.
271 
272  //set values of parameters which correspond to X and Y coordinates in grid
273  SetParameter("X", X);
274  SetParameter("Y", Y);
275 
276  Y += Eval((Double_t) A);
277 
278  if (!TMath::IsNaN(Y))
279  new_line->SetPoint(p_index++, X, Y);
280  else
281  Y = 0.; //reset Y to 0 if it ever becomes NaN
282  }
283  }
284 }

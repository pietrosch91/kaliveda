204 Double_t KVCsI::GetCorrectedLumiereTotale(Double_t rapide, Double_t lente)
205 {
206  // Returns total light output corrected for gain variations, if known (see KVINDRADB::ReadCsITotalLightGainCorrections
207  // for how corrections are read into database, format of correction files, etc.).
208  // Correction is calculated using
209  // rap_corr = (rapide-piedR) * fGainCorrection + piedR
210  // len_corr = (lente-piedL) * fGainCorrection + piedL
211  // lum_corr = GetLumiereTotale( rap_corr, len_corr )
212  // For more details on arguments, etc., see GetLumiereTotale.
213 
214  Double_t rap_corr;
215  Double_t len_corr;
216  rap_corr = (rapide < 0 ? GetR() : rapide);
217  len_corr = (lente < 0 ? GetL() : lente);
218  Double_t piedR = fACQ_R->GetPedestal();
219  Double_t piedL = fACQ_L->GetPedestal();
220  rap_corr = (rap_corr - piedR) * fGainCorrection + piedR;
221  len_corr = (len_corr - piedL) * fGainCorrection + piedL;
222  return GetLumiereTotale(rap_corr, len_corr);
223 }

257 Double_t KVCsI::Calculate(UShort_t mode, Double_t rapide, Double_t lente)
258 {
259  //Private method
260  //This function is called by GetLumiereTotale and GetTauZero, the return value depends
261  //on the value of 'mode' (kLumiere or kTau).
262  //The 'TauZero' in the code is the variable 'x'.
263  //In either case, if problems are encountered in the calculation, the value returned
264  //is zero and fLumTotStatus should give information on the problem (GetStatusLumiere).
265 
266  UInt_t ring = GetRingNumber();
267  UInt_t module = GetModuleNumber();
268 
269  //pedestal-corrected fast and slow components from raw data
270  Double_t rap_corr;
271  Double_t len_corr;
272  rap_corr = (rapide < 0 ? GetR() : rapide);
273  len_corr = (lente < 0 ? GetL() : lente);
274 
275  if (rap_corr <= 0. && len_corr <= 0.) {
276  //no R/L coder values set
277  fLumiereTotale = 0.0;
278  return fLumiereTotale;
279  }
280  rap_corr -= fACQ_R->GetPedestal();
281  len_corr -= fACQ_L->GetPedestal();
282 
283  if (rap_corr < 0 || len_corr < 0) {
284  fLumTotStatus = NEGATIVE_PEDESTAL_CORRECTED_VALUE;
285  return (fLumiereTotale = 0.0);
286  }
287  Double_t p0 = 400;
288  Double_t p1 = 900;
289  Double_t eps = 1.e-4;
290  /* Cette variable n'est pas utilisee, et ne figure pas dans e.g. le fortran de la 4eme campagne
291  Float_t pre=0.4318;
292  */
293  Double_t c1 = 1.5;
294  Double_t c2;
295  Double_t x;
296  Double_t x2;
297  Double_t x3;
298  Double_t bx;
299  /****************************************************************************
300  * Definition of PM rise-time constant tau
301  * Comment based on a note from Bernard Borderie:
302  * rings 11-16 use PM bases with a different component, thus tau is 60 nsec
303  * ring 16, module 5 is an exception, it has a tau of 20 nsec !!
304  * rung 5, module 11 is also an exception, it has a tau of 60 nsec !!
305  */
306  Double_t tau = 20.;
307  if (ring >= 11 && ring <= 16)
308  tau = 60.;
309  if (ring == 16 && module == 5)
310  tau = 20.;
311  if (ring == 5 && module == 11)
312  tau = 60.;
313  /****************************************************************************/
314  Double_t tau0 = 390.;
315  Double_t tau1 = 1590.;
316  Double_t tau2 = 3090.;
317 
318  if (ring > 3) {
319  c2 = 2.2;
320  } else {
321  c2 = 3.3;
322  }
323  Double_t dx;
324  Double_t rp = (len_corr / rap_corr) * (c1 / c2);
325  if (rp >= 1 && rap_corr <= 10) {
326  //calculation without correction for PM gain
327  x = 600;
328  x2 = TMath::Exp(-tau1 / x);
329  x3 = TMath::Exp(-tau2 / x);
330  bx = x2 - x3;
331  fLumTotStatus = NO_GAIN_CORRECTION;
332  } else {
333  //iterative calculation of total light output
334  fLumTotStatus = CALCULATED_WITH_GAIN_CORRECTION;
335  x = p0 + p1 * rp;
336  x = x - 100;
337  Double_t at = 1. - TMath::Exp(-tau0 / tau);
338  at = tau * at;
339  Int_t niter = 0;
340  do {
341  niter++;
342  Double_t xx = x * x;
343  Double_t x1 = TMath::Exp(-tau0 / x);
344  x2 = TMath::Exp(-tau1 / x);
345  x3 = TMath::Exp(-tau2 / x);
346  Double_t ax = 1. - x1;
347  ax = ax - at / x;
348  bx = x2 - x3;
349  Double_t fx = bx / ax - rp;
350  Double_t apx = -tau0 * x1;
351  apx = (apx + at) / xx;
352  Double_t bpx = tau1 * x2 - tau2 * x3;
353  bpx = bpx / xx;
354  Double_t fpx = (bpx * ax - apx * bx) / (ax * ax);
355  dx = -fx / fpx;
356  x = x + dx;
357  } while (TMath::Abs(dx / x) >= eps && niter <= 50);
358 
359  if (niter > 50 || x <= tau) {
360  fLumTotStatus = CALCULATION_NOT_CONVERGED;
361  //Warning("GetLumiereTotale","Iterative calculation did not converge for %s",
362  // GetName());
363  return (fLumiereTotale = 0.0);
364  }
365  }
366 
367  //GetLumiereTotale calling function
368  if (mode == kLumiere)
369  return (fLumiereTotale = len_corr * (x - tau) / (x * bx));
370 
371  //GetTauZero calling function
372  return x;
373 }

567 Double_t KVCsI::GetCorrectedEnergy(KVNucleus* nuc, Double_t lum, Bool_t)
568 {
569  // Calculate calibrated energy loss for a nucleus (Z,A) giving total light output "lum".
570  // If "lum" is not given, the total light of the detector
571  // calculated from the current values of the "R" and "L" acquisition
572  // parameters will be used (taking into account an eventual correction for gain variations,
573  // see GetCorrectedLumiereTotale()).
574  //
575  //Two KVLightEnergyCsI calibrators are used, one for Z=1, the other for Z>1
576  // Returns -1 in case of problems (no calibration available or light calculation not valid).
577  //
578  // NOTE: Simulations
579  // If detector is in 'SimMode', then we look for a KVSimNucleus in the list of particles
580  // which hit the detector in the event, and use the Z & A of this nucleus and the energy
581  // deposited in the CsI to calculate the light; then we use the Z & A of 'nuc' (not necessarily
582  // the same) to calculate the calibrated energy from the light.
583 
584  Int_t Z = nuc->GetZ();
585  Int_t A = nuc->GetA();
586 
587  KVLightEnergyCsI* calib = 0;
588 
589  if (Z == 1 && fCalZ1) calib = fCalZ1;
590  else calib = fCal;
591 
592  if (calib && calib->GetStatus()) {
593  if (IsSimMode()) {
594  lum = GetLumiereTotale();
595  if (lum < 0.) return -1.;
596  //force "OK" status for light
597  fLumTotStatus = NO_GAIN_CORRECTION;
598  } else if (lum < 0.) {
599  //light not given - calculate from R and L components
600  lum = GetCorrectedLumiereTotale(); // include gain correction
601  } else {
602  //light given as argument - force "OK" status for light
603  fLumTotStatus = NO_GAIN_CORRECTION;
604  }
605 
606  //check light calculation status
607  if (LightIsGood()) {
608  calib->SetZ(Z);
609  calib->SetA(A);
610  Double_t eloss = calib->Compute(lum);
611  SetEnergy(eloss);
612  return eloss;
613  }
614  }
615  return -1.;
616 }

683 void KVCsI::DeduceACQParameters(KVEvent* e, KVNumberList& index)
684 {
685  // Deduce fast & slow acquisition parameters for particles hitting CsI in given
686  // (simulated) event. The KVNumberList contains the indices of particles in the event.
687  GetACQParam("R")->SetData(-1);
688  GetACQParam("L")->SetData(-1);
689  GetACQParam("T")->SetData(-1);
690 
691  UShort_t Mt = 110;
692  Double_t Xlen_sum = 0;
693  Double_t Yrap_sum = 0;
694 
695  KVIDTelescope* idcsi = (KVIDTelescope*)GetIDTelescopes()->At(0);
696 
697  KVIDZAGrid* idgcsi = (KVIDZAGrid*)idcsi->GetIDGrid();
698  if (!idgcsi) {
699  //Warning("DeduceACQParameters","%s, No grid available",GetName());
700  return;
701  }
702 
703  index.Begin();
704  while (!index.End()) { // loop over all particles hitting CsI
705  KVNucleus* nunuc = e->GetParticle(index.Next());
706  Int_t zz = nunuc->GetZ();
707  Int_t aa = nunuc->GetA();
708  // calculate total light output corresponding to this particle and its individual energy loss
709  Double_t lumiere = GetLightFromEnergy(zz, aa, nunuc->GetParameters()->GetDoubleValue(GetName()));
710 
711 // Info("DAQ","Z=%d A=%d E=%f H=%f status=%d",zz,aa,nunuc->GetParameters()->GetDoubleValue(GetName()),lumiere,
712 // GetStatusLumiere());
713 
714  KVIDZALine* idline = (KVIDZALine*)idgcsi->GetIdentifier(zz, aa);
715  if (!idline && zz <= idgcsi->GetZmax()) {
716  // Z within limits of grid, but we don't have the isotope
717  // Look for line with closest mass
718  Int_t closest_a = 1000;
719  Int_t closest_index = -1;
720  Int_t nids = idgcsi->GetNumberOfIdentifiers();
721  for (Int_t iid = 0; iid < nids; iid++) {
722  KVIDZALine* ll = (KVIDZALine*)idgcsi->GetIdentifierAt(iid);
723  if (ll->GetZ() == zz) {
724  if (TMath::Abs(ll->GetA() - aa) < TMath::Abs(closest_a - aa)) {
725  closest_index = iid;
726  closest_a = ll->GetA();
727  }
728  }
729  }
730  idline = (KVIDZALine*)idgcsi->GetIdentifierAt(closest_index);
731  }
732  Double_t Xlen = 0;
733  Double_t Yrap = 0;
734 
735  if (idline) {
736  Double_t Yrap1, Yrap2, Xlen1, Xlen2;
737  idline->GetStartPoint(Xlen1, Yrap1);
738  idline->GetEndPoint(Xlen2, Yrap2);
739

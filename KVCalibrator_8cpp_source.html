232 void KVCalibrator::SetParameters(Double_t val, ...)
233 {
234  //Set the fParamNumber parameters of the calibrator.
235  //The number of arguments must be exactly fParamNumber
236  //(which must previously have been set)
237  //Once the parameters have been set with this method, the status of the
238  //calibrator becomes 'OK' or 'Ready'
239 
240  va_list ap;
241  va_start(ap, val);
242 
243  int arg_n = 0;
244  SetParameter(arg_n++, val);
245  while (arg_n < fParamNumber) {
246  SetParameter(arg_n, (Double_t) va_arg(ap, Double_t));
247  arg_n++;
248  }
249 
250  va_end(ap);
251  SetStatus(kTRUE);
252 }

316 TGraph* KVCalibrator::MakeGraph(Double_t xmin, Double_t xmax,
317  Int_t npoints) const
318 {
319  //Creates a TGraph with npoints points (default 50 points) showing the calibration
320  //formula between input values xmin and xmax (npoints equally-spaced points).
321  //User should delete the TGraph after use.
322  TGraph* tmp = 0;
323  if (GetStatus()) { // check calibrator is ready
324  if (GetNumberParams()) { // check calibrator is ready
325  if (npoints > 1) {
326  Double_t dx = (xmax - xmin) / ((Double_t) npoints - 1.0);
327  if (dx) {
328  Double_t* xval = new Double_t[npoints];
329  Double_t* yval = new Double_t[npoints];
330  for (int i = 0; i < npoints; i++) {
331  xval[i] = xmin + dx * ((Double_t) i);
332  yval[i] = Compute(xval[i]);
333  }
334  tmp = new TGraph(npoints, xval, yval);
335  delete[]xval;
336  delete[]yval;
337  }
338  }
339  }
340  }
341  return tmp;
342 }

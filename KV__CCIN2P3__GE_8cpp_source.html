332  // This method overrides and augments KVBatchSystem::ChangeDefJobOpt (which

386 KVList* KV_CCIN2P3_GE::GetListOfJobs()
387 {
388  // Create and fill list with KVBatchJob objects describing current jobs
389  // Delete list after use
390 
391  KVList* list_of_jobs = new KVList;
392 
393  // use qstat -r to get list of job ids and jobnames
394  TString reply = gSystem->GetFromPipe("qstat -r");
395 
396  TObjArray* lines = reply.Tokenize("\n");
397  Int_t nlines = lines->GetEntries();
398  for (Int_t line_number = 0; line_number < nlines; line_number++) {
399  TString thisLine = ((TObjString*)(*lines)[line_number])->String();
400  if (thisLine.Contains("Full jobname:")) {
401  // previous line contains job-id and status
402  TString lastLine = ((TObjString*)(*lines)[line_number - 1])->String();
403  TObjArray* bits = lastLine.Tokenize(" ");
404  Int_t jobid = ((TObjString*)(*bits)[0])->String().Atoi();
405  TString status = ((TObjString*)(*bits)[4])->String();
406  // date & time jobs started (running job) or submitted (queued job)
407  TString sdate = ((TObjString*)(*bits)[5])->String();// mm/dd/yyyy
408  TString stime = ((TObjString*)(*bits)[6])->String();// hh:mm:ss
409  Int_t dd, MM, yyyy, hh, mm, ss;
410  sscanf(sdate.Data(), "%d/%d/%d", &MM, &dd, &yyyy);
411  sscanf(stime.Data(), "%d:%d:%d", &hh, &mm, &ss);
412  KVDatime submitted(yyyy, MM, dd, hh, mm, ss);
413  delete bits;
414  bits = thisLine.Tokenize(": ");
415  TString jobname = ((TObjString*)(*bits)[2])->String();
416  delete bits;
417 
418  KVGEBatchJob* job = new KVGEBatchJob();
419  job->SetName(jobname);
420  job->SetJobID(jobid);
421  job->SetStatus(status);
422  job->SetSubmitted(submitted);
423  list_of_jobs->Add(job);
424  }
425  }
426  delete lines;
427 
428  if (!list_of_jobs->GetEntries()) return list_of_jobs;
429 
430  // use qstat -j [jobid] to get cpu and memory used and also the resource requests
431  TIter next_job(list_of_jobs);
432  KVGEBatchJob* job;
433  while ((job = (KVGEBatchJob*)next_job())) {
434 
435  // for running jobs, read in from [jobname].status file
436  // the number of events read/to read, disk used
437  if (!strcmp(job->GetStatus(), "r")) job->UpdateDiskUsedEventsRead();
438 
439  reply = gSystem->GetFromPipe(Form("qstat -j %d", job->GetJobID()));
440  lines = reply.Tokenize("\n");
441  nlines = lines->GetEntries();
442  for (Int_t line_number = 0; line_number < nlines; line_number++) {
443  TString thisLine = ((TObjString*)(*lines)[line_number])->String();
444  if (thisLine.BeginsWith("usage")) {
445  TObjArray* bits = thisLine.Tokenize("=,");
446  TString stime = ((TObjString*)(*bits)[1])->String();// hh:mm:ss or d:hh:mm:ss
447  Int_t dd, hh, mm, ss;
448  TObjArray* tmp = stime.Tokenize(":");
449  dd = 0;
450  if (tmp->GetEntries() == 4) sscanf(stime.Data(), "%d:%2d:%2d:%2d", &dd, &hh, &mm, &ss);
451  else sscanf(stime.Data(), "%2d:%2d:%2d", &hh, &mm, &ss);
452  delete tmp;
453  job->SetCPUusage((dd * 24 + hh) * 3600 + mm * 60 + ss);
454  TString smem = ((TObjString*)(*bits)[7])->String();// xxx.xxxxM
455  job->SetMemUsed(smem);
456  delete bits;
457  } else if (thisLine.BeginsWith("hard resource_list:")) {
458  TObjArray* bits = thisLine.Tokenize(": ");
459  TString res = ((TObjString*)(*bits)[2])->String();//os=sl5,xrootd=1,irods=1,s_vmem=1024M,s_fsize=50M,s_cpu=36000
460  res.ReplaceAll("s_vmem", "vmem");
461  res.ReplaceAll("s_fsize", "fsize");
462  res.ReplaceAll("s_cpu", "ct");
463  job->SetResources(res);
464  TObjArray* bbits = res.Tokenize(",");
465  TIter next_res(bbits);
466  TObjString* ss;
467  while ((ss = (TObjString*)next_res())) {
468  TString g = ss->String();
469  if (g.BeginsWith("ct=")) {
470  g.Remove(0, 3);
471  job->SetCPUmax(g.Atoi());
472  } else if (g.BeginsWith("vmem=")) {
473  g.Remove(0, 5);
474  job->SetMemMax(g);
475  } else if (g.BeginsWith("fsize=")) {
476  g.Remove(0, 6);
477  job->SetDiskMax(g);
478  }
479  }
480  delete bits;
481  delete bbits;
482  }
483  }
484  delete lines;
485  //}
486  }
487 
488  return list_of_jobs;
489 }

531 void KV_CCIN2P3_GE::Run()
532 {
533  //Processes the job requests for the batch system.
534  //In normal mode, this submits one job for the data analyser fAnalyser
535  //In multijobs mode, this submits one job for each run in the runlist associated to fAnalyser
536 
537  if (!CheckJobParameters()) return;
538 
539  if (MultiJobsMode()) {
540  if (fAnalyser->InheritsFrom("KVDataSetAnalyser")) {
541  //submit jobs for every GetRunsPerJob() runs in runlist
542  KVDataSetAnalyser* ana = dynamic_cast<KVDataSetAnalyser*>(fAnalyser);
543  KVNumberList runs = ana->GetRunList();
544  runs.Begin();
545  Int_t remaining_runs = runs.GetNValues();
546  fCurrJobRunList.Clear();
547  while (remaining_runs && !runs.End()) {
548  Int_t run = runs.Next();
549  remaining_runs--;
550  fCurrJobRunList.Add(run);
551  if ((fCurrJobRunList.GetNValues() == GetRunsPerJob()) || runs.End()) {
552  // submit job for GetRunsPerJob() runs (or less if we have reached end of runlist 'runs')
553  ana->SetRuns(fCurrJobRunList, kFALSE);
554  ana->SetFullRunList(runs);
555  SubmitJob();
556  fCurrJobRunList.Clear();
557  }
558  }
559  ana->SetRuns(runs, kFALSE);
560  } else if (fAnalyser->InheritsFrom("KVSimDirAnalyser")) {
561  // here we understand "run" to mean "file"
562  KVSimDirAnalyser* ana = dynamic_cast<KVSimDirAnalyser*>(fAnalyser);
563  TList* file_list = ana->GetFileList();
564  Int_t remaining_runs = ana->GetNumberOfFilesToAnalyse();
565  fCurrJobRunList.Clear();
566  TList cur_file_list;
567  TObject* of;
568  TIter it(file_list);
569  Int_t file_no = 1;
570  while ((of = it())) {
571  cur_file_list.Add(of);
572  fCurrJobRunList.Add(file_no);
573  remaining_runs--;
574  file_no++;
575  if ((fCurrJobRunList.GetNValues() == GetRunsPerJob()) || (remaining_runs == 0)) {
576  // submit job for GetRunsPerJob() files (or less if we have reached end of list)
577  ana->SetFileList(&cur_file_list);
578  SubmitJob();
579  fCurrJobRunList.Clear();
580  cur_file_list.Clear();
581  }
582  }
583  ana->SetFileList(file_list);
584  }
585  } else {
586  SubmitJob();
587  }
588 
589 }

605 const Char_t* KV_CCIN2P3_GE::GetJobName() const
606 {
607  //Returns name of batch job, either during submission of batch jobs or when an analysis
608  //task is running in batch mode (access through gBatchSystem global pointer).
609  //
610  //In multi-job mode, the job name is generated from the base name set by SetJobName()
611  //plus the extension "_Rxxxx-yyyy" with "xxxx" and "yyyy" the number of the first and last run
612  //which will be analysed by the current job.
613  //
614  // Depending on the batch system, some sanitization of the jobname may be required
615  // e.g. to remove "illegal" characters from the jobname. This is done by SanitizeJobName()
616  // before the jobname is returned.
617 
618  if (!fAnalyser) {
619  //stand-alone batch submission ?
620  fCurrJobName = fJobName;
621  } else {
622  //replace any special symbols with their current values
623  fCurrJobName = fAnalyser->ExpandAutoBatchName(fJobName.Data());
624  if (MultiJobsMode() && !fAnalyser->BatchMode()) {
625  KVString tmp;
626  if (fCurrJobRunList.GetNValues() > 1)
627  tmp.Form("_R%d-%d", fCurrJobRunList.First(), fCurrJobRunList.Last());
628  else
629  tmp.Form("_R%d", fCurrJobRunList.First());
630  fCurrJobName += tmp;
631  }
632  }
633  SanitizeJobName();
634  return fCurrJobName.Data();
635 }

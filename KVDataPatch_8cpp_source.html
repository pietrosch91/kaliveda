15 // Abstract base class for patches to be applied to reconstructed, identified,
16 // and/or calibrated data.
17 //
18 // To make a patch, derive a class from this one, and override the following
19 // methods:
20 //
21 // virtual Bool_t IsRequired(TString dataset, TString datatype, Int_t runnumber,
22 // TString dataseries, Int_t datareleasenumber, TList* streamerinfolist)
23 //
24 // - returns kTRUE if patch is to be applied to the data
25 // "dataseries" and "datareleasenumber" are the KaliVeda (major) series
26 // and (minor) release number previously used to generate the data
27 // "streamerinfolist" is a pointer to the list of TStreamerInfo objects
28 // containing information on the versions of classes used for the different
29 // objects in the file (obtained with TFile::GetStreamerInfoCache())
30 //
31 // virtual Bool_t IsEventPatch()
32 //
33 // - returns kTRUE if patch is to be applied at the event level
34 // (in this case ApplyToEvent(KVEvent*) will be called for each event)
35 //
36 // virtual Bool_t IsParticlePatch()
37 //
38 // - returns kTRUE if patch is to be applied at the particle level
39 // (in this case ApplyToNucleus(KVNucleus*) will be called for each particle)
40 //
41 // virtual void ApplyToEvent(KVEvent*)
42 //
43 // - event-level patch. if not required, just define an empty method.
44 //
45 // virtual void ApplyToParticle(KVNucleus*)
46 //
47 // - particle-level patch. if not required, just define an empty method.
48 //
49 // virtual void PrintPatchInfo() const
50 //
51 // - print information about the patch. Will be called by Print().
52 //

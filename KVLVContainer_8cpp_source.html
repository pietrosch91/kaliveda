167 {

251 Int_t KVLVColumnData::Compare(TObject* ob1, TObject* ob2)
252 {
253  // Compare (for sorting) the two objects based on the type of data stored
254  // in this column. The sorting depends on the return type of the data.
255  // for integers: return 1 if ob1->data < ob2->data etc.
256  // for floats: return 1 if ob1->data < ob2->data etc.
257  // for strings: alphabetical sorting using TString::CompareTo
258  // for dates: chronological order (earliest first)
259 
260  switch (fRetType) {
261 
262  case (Int_t)TMethodCall::kString :
263  return Compare_string(ob1, ob2);
264  break;
265 
266  case (Int_t)TMethodCall::kLong :
267  return Compare_long(ob1, ob2);
268  break;
269 
270  case (Int_t)TMethodCall::kDouble :
271  return Compare_double(ob1, ob2);
272  break;
273 
274  case kDatimeRef:
275  case kDatimeInt:
276  return Compare_date(ob1, ob2);
277  break;
278 
279  default:
280  std::cout << "Error in <KVLVColumnData::Compare> : this type is not supported "
281  << (Int_t)fMethCall->ReturnType() << std::endl;
282  break;
283  }
284  return 0;
285 }

444 void KVLVContainer::Display(const TCollection* list_of_objects)
445 {
446  // Display the list of objects in the container.
447  // If the list is very long (> KVListView.MaxListLength)
448  // we pop up a dialog box to ask if all items should be displayed
449  // which can take quite a long time
450 
451  /* remove all items from display, but keep list in memory */
452  fKeepUserItems = kTRUE;
453  RemoveAll();
454  fKeepUserItems = kFALSE;
455 
456  Int_t max_entries = gEnv->GetValue("KVListView.MaxListLength", 1000);
457  if (list_of_objects->GetEntries() > max_entries) {
458  Int_t ret;
459  new TGMsgBox(gClient->GetRoot(), GetMainFrame(), "Long list",
460  Form("The list contains more than %d items. Do you want to display all items ?",
461  max_entries),
462  kMBIconQuestion, kMBYes | kMBNo, &ret);
463  if (ret & kMBNo) {
464  // display only the first KVListView.MaxListLength entries
465  TList list2;
466  Int_t N = 0;
467  TIter next(list_of_objects);
468  TObject* o;
469  while ((o = next()) && (N++) < max_entries) list2.Add(o);
470  FillList(&list2);
471  } else
472  FillList(list_of_objects);
473  } else
474  FillList(list_of_objects);
475  if (!fIsResized) {
476  GetListView()->ResizeColumns();
477  fIsResized = kTRUE;
478  }
479  TGCanvas* canvas = (TGCanvas*) this->GetParent()->GetParent();
480  canvas->Layout();
481 
482  MapSubwindows();
483 }

524 {

800 void KVLVContainer::OpenContextMenu(TGFrame* f, Int_t but, Int_t x, Int_t y)
801 {
802  // Open context menu when user right-clicks an object in the list.
803  // Calling AllowContextMenu(kFALSE) will disable this.
804  // We also fill the list fPickOrderedObjects with the selected objects
805  // in the order of clicking
806  //
807  // if fUseObjLabelAsRealClass=kTRUE (and if objects inherit from KVBase)
808  // then the context menu opened will be that of the class given by
809  // KVBase::GetLabel. The object's KVBase::GetObject() method must
810  // return the real object to use.
811 
812  if (but == kButton1) {
813  TGLVEntry* el = (TGLVEntry*)f;
814  TObject* ob = (TObject*)el->GetUserData();
815  if (!fControlClick || !GetMultipleSelection()) fPickOrderedObjects->Clear();
816  if (ob) {
817  Bool_t in_list = fPickOrderedObjects->FindObject(ob);
818  if (in_list) fPickOrderedObjects->Remove(ob);
819  else fPickOrderedObjects->AddLast(ob);
820  }
821  return;
822  }
823 
824  // context menus globally disabled and no exceptions defined
825  if (!fAllowContextMenu && !fContextMenuClassExceptions) return;
826 
827  if (but == kButton3) {
828  // Error("OpenContextMenu","x=%d y=%d",x,y);
829  // fContextMenu->Popup(x,y,this); return;
830 
831  TGLVEntry* el = (TGLVEntry*)f;
832  TObject* ob = (TObject*)el->GetUserData();
833  if (ob) {
834 
835  TObject* CMob = ob;
836  TString CMobClass = ob->ClassName();
837 
838  if (fUseObjLabelAsRealClass && ob->TestBit(KVBase::kIsKaliVedaObject)) {
839  KVBase* bob = dynamic_cast<KVBase*>(ob);
840  CMobClass = bob->GetLabel();
841  CMob = bob->GetObject();
842  }

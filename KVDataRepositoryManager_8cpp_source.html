29 <h2>KVDataRepositoryManager</h2>
30 <h4>Handles list of available data repositories</h4>
31 <p>
32 The data repository manager is set up and initialised in the following way:
33 </p>
34 <pre>
35 new KVDataRepositoryManager;
36 gDataRepositoryManager->Init();
37 </pre>
38 <p>
39 The available data repositories are defined in the user's $HOME/.kvrootrc file
40 (see <a href="KVDataRepository.html">KVDataRepository</a> for details
41 of configuration).</p>
42 <h4>Default data repository</h4>
43 <p>If more than one repository is defined, which one will be "active" (gDataRepository)
44 after initialisation of the data repository manager ? You can define the default repository
45 by setting the variable
46 </p>
47 <pre>
48 DataRepository.Default: [name]
49 </pre>
50 <p>in your .kvrootrc file. If you don't, the default repository "by default" will be either:
51 the data repository named "default" if there is one; or the last one defined in your .kvrootrc.

126 void KVDataRepositoryManager::Init()
127 {
128  //Read .kvrootrc and set up all data repositories it defines.
129  //The default data repository (gDataRepository) is defined by the environment variable
130  //DataRepository.Default (default value = "default").
131  //For this repository, KVDataRepository::cd() will be called and
132  //gDataRepository and gDataSetManager will point, respectively, to the repository and
133  //to its data set manager.
134  //If the repository corresponding to DataRepository.Default is not found,
135  //the last repository defined in the .kvrootrc file will be made default.
136 
137  //make sure KaliVeda environment is initialised
138  KVBase::InitEnvironment();
139 
140  //delete any previously defined repositories
141  if (fRepositories.GetSize())
142  fRepositories.Delete();
143 
144  //get whitespace-separated list of data repository names
145  TString rep_list = gEnv->GetValue("DataRepository", "");
146  if (rep_list == "") {
147  cout <<
148  "<KVDataRepositoryManager::Init> : no repositories defined in .kvrootrc"
149  << endl;
150  // Initialise a repository-less dataset manager which will give access to
151  // all datasets regardless of data availability (which is unknowable)
152  gDataSetManager = new KVDataSetManager;
153  gDataSetManager->Init();
154  return;
155  }
156  //split list
157  TObjArray* toks = rep_list.Tokenize(' ');
158  KVDataRepository* new_rep = 0;
159  KVDataRepository* last_defined = 0;
160  for (int i_rep = 0; i_rep < toks->GetEntries(); i_rep++) {
161 
162  //loop over each defined repository
163 
164  TString rep_name = ((TObjString*)(*toks)[i_rep])->String(); //name of repository
165  //look for repository type
166  TString rep_type =
167  gEnv->GetValue(Form("%s.DataRepository.Type", rep_name.Data()),
168  "local");
169  rep_type.ToLower();
170 
171  //create new repository
172  new_rep = KVDataRepository::NewRepository(rep_type.Data());
173  new_rep->SetName(rep_name.Data());
174  //new_rep->SetType(rep_type.Data()); 'type' is set in default ctor of each repository class
175  if (new_rep->Init()) {
176  fRepositories.Add(new_rep);
177  last_defined = new_rep; //keep pointer to last defined repository
178  }
179  else {
180  //problem with initialisation of data repository.
181  //it is ignored.
182  delete new_rep;
183  new_rep = 0;
184  }
185 
186  }
187  delete toks;
188  //look for 'default' repository
189  new_rep = (KVDataRepository*) fRepositories.FindObject(
190  gEnv->GetValue("DataRepository.Default", "default"));
191  if (new_rep) new_rep->cd();
192  else if (last_defined) last_defined->cd();
193 }

225 {
226  //Print list of repositories
227  //opt = "all" : print full configuration information for each repository
228  KVDataRepository* rep;
229  TString _opt(opt);

231  Bool_t _all = (_opt == "ALL");

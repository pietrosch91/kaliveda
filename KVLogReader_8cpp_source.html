80 void KVLogReader::ReadLine(TString& line, Bool_t& ok)
81 {

125 void KVLogReader::ReadStorageReq(TString& line)
126 {
127  //Read lines "DISK_REQ: 3GB" and "MEM_REQ: 150MB"
128  //and fill corresponding variable with appropriate value
129  TObjArray* toks = line.Tokenize("*: ");
130  TString type = ((TObjString*) toks->At(0))->GetString();
131  KVString stor = ((TObjString*) toks->At(1))->GetString();
132  Int_t size = ReadStorage(stor);
133  if (type == "DISK_REQ")
134  fSCRreq = size;
135  else if (type == "MEM_REQ")
136  fMEMreq = size;
137  // both requests must be present in the logfile otherwise
138  // something very wrong has happened
139  if (fSCRreq && fMEMreq) fGotRequests = kTRUE;
140  delete toks;
141 }

182 Bool_t KVLogReader::Incomplete() const

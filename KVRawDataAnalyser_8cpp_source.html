68 void KVRawDataAnalyser::ProcessRun()
69 {
70  // Perform treatment of a given run
71  // Before processing each run, after opening the associated file, user's InitRun() method is called.
72  // After each run, user's EndRun() is called.
73  // For each event of each run, user's Analysis() method is called.
74  //
75  // For further customisation, the pre/post-methods are called just before and just after
76  // each of these methods (preInitRun(), postAnalysis(), etc. etc.)
77 
78  //Open data file
79  KVString raw_file = gDataSet->GetFullPathToRunfile(GetDataType().Data(), fRunNumber);
80  fRunFile = (KVRawDataReader*)gDataSet->OpenRunfile(GetDataType().Data(), fRunNumber);
81  if ((!fRunFile) || fRunFile->IsZombie()) {
82  //skip run if file cannot be opened
83  if (fRunFile) delete fRunFile;
84  return;
85  }
86  //warning! real number of run may be different from that deduced from file name
87  //we get the real run number from gMultiDetArray and use it to name any new files
88  if (!gMultiDetArray)
89  Error("ProcessRun", "invalid gMultiDetArray pointer ... Go to crash");
90  Int_t newrun = gMultiDetArray->GetCurrentRunNumber();
91  if (newrun != fRunNumber) {
92  cout << " *** WARNING *** run number read from file = " << newrun << endl;
93  fRunNumber = newrun;
94  }
95 
96  fEventNumber = 1; //event number
97 
98  Long64_t nevents = GetNbEventToRead();
99  if (nevents <= 0) {
100  nevents = 1000000000;
101  cout << endl << "Reading all events from file " << raw_file.Data() << endl;
102  } else {
103  cout << endl << "Reading " << nevents << " events from file " << raw_file.Data() << endl;
104  }
105 
106  cout << "Starting analysis of run " << fRunNumber << " on : ";
107  TDatime now;
108  cout << now.AsString() << endl << endl;
109 
110  preInitRun();
111  //call user's beginning of run
112  InitRun();
113  postInitRun();
114 
115  fDetEv = new KVDetectorEvent;
116 
117  //loop over events in file
118  while ((nevents-- ? fRunFile->GetNextEvent() : kFALSE) && !AbortProcessingLoop()) {
119 
120  //reconstruct hit groups
121  KVSeqCollection* fired = fRunFile->GetFiredDataParameters();
122  gMultiDetArray->GetDetectorEvent(fDetEv, fired);
123 
124  preAnalysis();
125  //call user's analysis. stop if returns kFALSE.
126  if (!Analysis()) break;
127  postAnalysis();
128 
129  fDetEv->Clear();
130 
131  if (CheckStatusUpdateInterval(fEventNumber)) DoStatusUpdate(fEventNumber);
132 
133  fEventNumber += 1;
134  }
135 
136  delete fDetEv;
137 
138  cout << "Ending analysis of run " << fRunNumber << " on : ";
139  TDatime now2;
140  cout << now2.AsString() << endl << endl;
141  cout << endl << "Finished reading " << fEventNumber - 1 << " events from file " << raw_file.Data() << endl << endl;
142 
143  preEndRun();
144  //call user's end of run function
145  EndRun();
146  postEndRun();
147 
148  delete fRunFile;
149 }

166 void KVRawDataAnalyser::SubmitTask()
167 {
168  // Perform analysis of chosen runs
169  // Before beginning the loop over the runs, the user's InitAnalysis() method is called.
170  // After completing the analysis of all runs, the user's EndAnalysis() method is called.
171  //
172  // Further customisation of the event loop is possible by overriding the methods
173  // preInitAnalysis()
174  // postInitAnalysis()
175  // preEndAnalysis()
176  // postEndAnalysis()
177  // which are executed respectively just before and just after the methods.
178 
179  if (gDataSet != GetDataSet()) GetDataSet()->cd();
180 
181  preInitAnalysis();
182  //call user's initialisation
183  InitAnalysis();
184  postInitAnalysis();
185 
186  CalculateTotalEventsToRead();
187 
188  //loop over runs
189  GetRunList().Begin();
190  while (!GetRunList().End() && !AbortProcessingLoop()) {
191  fRunNumber = GetRunList().Next();
192  ProcessRun();
193  }
194 
195  preEndAnalysis();
196  //call user's end of analysis
197  EndAnalysis();
198  postEndAnalysis();
199 }

255 void KVRawDataAnalyser::AddHisto(TH1* h, const Char_t* family)
256 {
257  // Add user histo to internal list of spectra.
258  // "family" can be used to arrange histograms into a directory structure,
259  // e.g.
260  // toto->AddHisto(h1, "CSI_R_L/Ring_10")
261  // will create (if they don't exist) sublists "CSI_R_L" and "Ring_10"
262  // and store the histogram referenced by pointer h1 in the second one.
263  //
264  // All spectra can be saved at any time by calling SaveSpectra method.
265 
266  KVString dir_struc(family);
267  if (dir_struc == "") {
268  fHistoList.Add(h);
269  return;
270  }
271  dir_struc.Begin("/");
272  int level = 0;
273  KVHashList* sublist = 0;
274  while (!dir_struc.End()) {
275  KVString dir = dir_struc.Next();
276  if (!level) {
277  // use top-level directory name as name of fHistoList
278  fHistoList.SetName(dir.Data());
279  sublist = &fHistoList;
280  } else {
281  KVHashList* sublist2 = (KVHashList*)sublist->FindObject(dir.Data());
282  if (!sublist2) {
283  sublist2 = new KVHashList;
284  sublist2->SetName(dir.Data());
285  sublist->Add(sublist2);
286  }
287  sublist = sublist2;
288  }
289  level++;
290  }
291  sublist->Add(h);
292 }

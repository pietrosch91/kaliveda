157 void KVPartition::Update()
158 {
159 //protected method, Methode dérivée de KVIntegerList,
160 //Appel de KVIntegerList::Update()
161 //les deux TArrayI ftab et ftab_diff sont mis à jour
162 //un test de dimension est fait pour etendre si besoin
163 
164  if (fMult > ftab->fN) ftab->Set(fMult);
165  if (fMult_diff > ftab_diff->fN) ftab_diff->Set(fMult_diff);
166  //Info("Update","fMult=%d fMult_diff=%d",fMult,fMult_diff);
167  Int_t mdiff = 0, mtot = 0;
168  KVString snom = "", stamp = "";
169  for (Int_t ii = fLimiteRegle; ii >= 0; ii -= 1) {
170  Int_t contenu = fRegle->At(ii);
171  if (contenu > 0) {
172  ftab_diff->AddAt(ii, mdiff++);
173  for (Int_t mm = 0; mm < contenu; mm += 1) {
174  ftab->AddAt(ii, mtot++);
175  }
176 
177  //Formattage du nom de la partition
178  stamp.Form("%d", ii);
179  snom += stamp;
180  if (contenu > 1) {
181  stamp.Form("(%d)", contenu);
182  snom += stamp;
183  }
184  snom += " ";
185  }
186  }
187  if (snom != "") snom.Remove(snom.Length() - 1);
188  SetName(snom.Data());
189  fLength = snom.Length();
190 
191  SetBit(kHastobeComputed, kFALSE);
192 
193 }

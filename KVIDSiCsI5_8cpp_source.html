31 //KVIDSiCsI5
32 //
33 //Identification in Si(GG/PG)-CsI(lumiere) matrices of INDRA 5th campaign.
34 //(Mathieu Pichon & Josquin Marie)
35 //
36 //For each telescope, at least 2 grids were used :
37 // GG identification using Si canal GG
38 // PG1 identification using Si canal PG
39 //For certain telescopes, a second PG grid (PG2) is available, used for
40 //large Z identification.
41 //Each of the three grids has an associated maximum and minimum Z (GetZmax, GetZmin)
42 //corresponding to the lines drawn by hand used for the fit.
43 //Z identification is considered possible between max(Zmin-0.5, 0.5) and min(Zmax+0.5, 100.5)
44 //(Zmax is taken equal to 100 when using the highest Z grid of the telescope (i.e. either
45 //PG1 or PG2).
46 //In addition certain telescopes have both Z and A identification for the SiGG grid.
47 //In this case KVIDSiCsI5::HasMassID=kTRUE.
48 //
49 //All three grids were fitted using the polynomials tassangot_Z and tassangot_A
50 //defined in KVTGIDFunctions.
51 //
52 //ID SUB-CODES:
53 //-------------
54 // 0 KVIDSiCsI5::k_OK_GG "ok in GG",
55 // 1 KVIDSiCsI5::k_noTGID_GG "no Z identification available - last grid tried GG",
56 // 2 KVIDSiCsI5::k_OutOfIDRange_GG "point outside of range of possible Z identification in GG",
57 // 3 KVIDSiCsI5::k_OK_PG1 "ok in PG1",
58 // 4 KVIDSiCsI5::k_noTGID_PG1 "no Z identification available - last grid tried PG1",
59 // 5 KVIDSiCsI5::k_OutOfIDRange_PG1 "point outside of range of possible Z identification in PG1",
60 // 6 KVIDSiCsI5::k_OK_PG2 "ok in PG2",
61 // 7 KVIDSiCsI5::k_noTGID_PG2 "no Z identification available - last grid tried PG2",
62 // 8 KVIDSiCsI5::k_OutOfIDRange_PG2 "point outside of range of possible Z identification in PG2",
63 // 9 KVIDSiCsI5::kZOK_AnoTGID "Z ok, A identification tried but not available",
64 //10 KVIDSiCsI5::kZOK_AOutOfIDRange "Z ok, A identification tried but point to identify outside of identification range of function",
65 //11 KVIDSiCsI5::kZOK_A_ZtooSmall "Z ok, A identification tried but IdentA called with Z<1",
66 //12 KVIDSiCsI5::kZOK_A_ZtooLarge "Z ok, A identification tried but IdentA called with Z larger than max Z defined for KVTGIDZA isotopic identification object"

118 Double_t KVIDSiCsI5::IdentifyZ(Double_t& funLTG)
119 {
120  //Identify Z using the following method:
121  // - first try GG matrix
122  // if this doesn't work, or if the Z found is greater than the maximum
123  // Z for GG....
124  // - try PG1 matrix
125  // if this doesn't work, or if the Z found is greater than the maximum
126  // Z for PG1....
127  // - try PG2 matrix (if it exists)
128 
129  Double_t Z = -1.;
130 
131  if (((KVSilicon*) GetDetector(1))->GetGG() < 4090) {
132  //try GG identification if SIGG is not saturated
133  //the condition is applied directly to the raw channel number instead of GetIDMapY()
134  //so that it is correct even if GetIDMapY rescales for the change in gain
135  Z = IdentZ(GetName(), GetIDMapX("GG"), GetIDMapY("GG"), funLTG, "GG");//IdentZ(this, funLTG, "GG", "GG");
136  fWhichGrid = k_GG;
137  }
138 
139  if (Z < 0.) {
140 
141  Z = IdentZ(GetName(), GetIDMapX("PG"), GetIDMapY("PG"), funLTG, "PG1"); //IdentZ(this, funLTG, "PG1", "PG");
142  fWhichGrid = k_PG1;
143 
144  } else {
145  //successful GG identification
146  return Z;
147  }
148 
149  if (!HasPG2()) {
150  //no 2nd PG fit - this is the end
151  return Z;
152  }
153 
154  if (Z < 0.) {
155 
156  Z = IdentZ(GetName(), GetIDMapX("PG"), GetIDMapY("PG"), funLTG, "PG2"); //IdentZ(this, funLTG, "PG2", "PG");
157  fWhichGrid = k_PG2;
158 
159  }
160 
161  return Z;
162 }

179 Bool_t KVIDSiCsI5::Identify(KVIdentificationResult* IDR, Double_t, Double_t)
180 {
181  //Identification of particles using Si(300)-CsI matrices for 5th campaign.
182  //First of all, Z identification is attempted with KVIDSiCsI5::IdentZ.
183  //If successful, if this telescope has mass identification capabilities
184  //(HasMassID()=kTRUE), then if the identified Z is not too large for the
185  //GG grid Z&A identification, we identify the mass of the particle.
186  //In this case, a further check is made that the resulting nucleus is not
187  //too bizarre, i.e. it must be in the KVNucleus mass table (KVNucleus::IsKnown = kTRUE).
188  //If not, or if mass identification is not possible, the mass of the nucleus
189  //is deduced from Z using the default mass formula of class KVNucleus.
190  //
191  // Note that optional arguments (x,y) for testing identification are not used.
192 
193  Double_t funLTG_Z = -1;
194  Double_t funLTG_A = -1;
195  Double_t mass = -1;
196  Int_t ia = -1;
197  Int_t iz = -1;
198 
199  IDR->SetIDType(GetType());
200  IDR->IDattempted = kTRUE;
201  Double_t Z = IdentifyZ(funLTG_Z);
202 
203  //use KVTGIDManager::GetStatus value for IdentZ as identification subcode
204  Int_t Zstatus = (GetStatus() + 3 * fWhichGrid);
205  IDR->IDquality = Zstatus;
206 
207  if (GetStatus() != KVTGIDManager::kStatus_OK)
208  return kFALSE; // no ID
209 
210  iz = TMath::Nint(Z);
211 
212  IDR->IDOK = kTRUE;
213 
214  //is mass identification a possibility ?
215  if (HasMassID() && Zstatus == k_OK_GG) { //only in GG...
216 
217  mass = IdentA(GetName(), GetIDMapX("GG"), GetIDMapY("GG"), funLTG_A, "GG", iz); //IdentA(this, funLTG_A, "GG", "GG", iz);
218 
219  if (GetStatus() != KVTGIDManager::kStatus_OK) { //mass ID not good ?
220  //only Z identified
221  IDR->Z = iz;
222  IDR->PID = Z;
223  IDR->Zident = kTRUE;
224  //subcode says "Z ok but A failed because..."
225  IDR->IDquality = k_OutOfIDRange_PG2 + GetStatus();
226  } else { //both Z and A successful ?
227  ia = TMath::Nint(mass);
228  // fix 16B
229  if (iz == 5 && ia == 16) ia = (mass > 16.0 ? 17 : 15);
230  // fix 9B
231  if (iz == 5 && ia == 9) ia = (mass > 9.0 ? 10 : 8);
232  // fix 8Be
233  if (iz == 4 && ia == 8) ia = (mass > 8.0 ? 9 : 7);
234  // fix 13Be
235  if (iz == 4 && ia == 13) ia = (mass > 13.0 ? 14 : 12);
236  // fix 10Li
237  if (iz == 3 && ia == 10) ia = (mass > 10.0 ? 11 : 9);
238  // fix 5He
239  if (iz == 2 && ia == 5) ia = (mass > 5.0 ? 6 : 4);
240  // fix 7He
241  if (iz == 2 && ia == 7) ia = (mass > 7.0 ? 8 : 6);
242  // check that mass is not too bizarre
243  KVNucleus tmp(iz, ia);
244  if (!tmp.IsKnown()) {
245  // if ia seems too light, we try with iz-1
246  // if ia seems too heavy, we try with iz+1
247  Info("Identify", "%s : initial ID Z=%d A=%d funLTG=%g", GetName(), iz, ia, funLTG_A);
248  Int_t iz2 = (ia < 2 * iz ? iz - 1 : iz + 1);
249  if (iz2 > 0) {
250  Double_t old_funLTG_A = funLTG_A;
251  Double_t new_mass = IdentA(GetName(), GetIDMapX("GG"), GetIDMapY("GG"), funLTG_A, "GG", iz2); //IdentA(this, funLTG_A, "GG", "GG", iz2);
252  // is this a better solution ?
253  if (GetStatus() == KVTGIDManager::kStatus_OK) {
254  Int_t new_ia = TMath::Nint(new_mass);
255  tmp.SetZ(iz2);
256  tmp.SetA(new_ia);
257  Info("Identify", "%s : new ID Z=%d A=%d funLTG=%g", GetName(), iz2, new_ia, funLTG_A);
258  if (tmp.IsKnown() && (TMath::Abs(funLTG_A) < TMath::Abs(old_funLTG_A))) {
259  // new nucleus is known and point is closer to line
260  iz = iz2;
261  ia = new_ia;
262  mass = new_mass;
263  Info("Identify", "%s : accepted new ID", GetName());
264  } else {
265  Info("Identify", "%s : rejected new ID", GetName());
266  }
267  } else {
268  Info("Identify", "%s : failed to obtain new ID with Z=%d", GetName(), iz2);
269  }
270  }
271  }
272  IDR->Z = iz;
273  IDR->A = ia;
274  IDR->PID = mass;
275  IDR->Zident = kTRUE;
276  IDR->Aident = kTRUE;
277  }
278  } else {
279  //only Z identified
280  //ID subcode remains 'Zstatus'
281  IDR->Z = iz;
282  IDR->PID = Z;
283  IDR->Zident = kTRUE;
284  }
285 
286  // set general ID code
287  IDR->IDcode = kIDCode3;
288  return kTRUE;
289 }

320 Double_t KVIDSiCsI5::GetIDMapX(Option_t*)
321 {
322  //Calculates current X coordinate for identification.
323  //It is the CsI detector's total light output calculated from current values of 'R' and 'L'
324  //raw data without pedestal correction (because identification maps were drawn without
325  //correcting).
326  //'opt' has no effect.
327 
328  Double_t rapide = (Double_t)fCsI->GetR() + fCsIRPedestal;
329  Double_t lente = (Double_t)fCsI->GetL() + fCsILPedestal;
330  Double_t h = (Double_t)fCsI->GetLumiereTotale(rapide, lente);
331  return h;

397 Double_t KVIDSiCsI5::GetIDMapY(Option_t* opt)
398 {
399  //Calculates current Y coordinate for identification.
400  //It is the silicon detector's current grand gain (by default, i.e. opt="", or if opt="GG")
401  //or petit gain (opt="PG") coder data, without pedestal correction.
402  //We include a "correction" for the gain of the Silicon amplifier:
403  //this was set to 1.0 during the runs for which the identification grids were drawn;
404  //when it increases to 1.41 we simply scale the data down by the same factor.
405 
406  Double_t si, si_ped;
407  if (!strcmp(opt, "PG")) {
408  si = (Double_t)fSi->GetPG();
409  si_ped = fSiPGPedestal;
410  } else {
411  si = (Double_t)fSi->GetGG();
412  si_ped = fSiGGPedestal;
413  }
414  //gain "correction"
415  if (fSiGain > 1.1) si = (si - si_ped) / fSiGain + si_ped;
416  return si;
417 }

455 Bool_t KVIDSiCsI5::SetIdentificationParameters(const KVMultiDetArray* MDA)
456 {
457  //Initialise the identification parameters (grids, etc.) of ALL identification telescopes of this
458  //kind (label) in the multidetector array. Therefore this method need only be called once, and not
459  //called for each telescope. The kind/label (returned by GetLabel) of the telescope corresponds
460  //to the URI used to find the plugin class in $KVROOT/KVFiles/.kvrootrc.
461 //Parameters are read from the file with name given by the environment variable
462 //INDRA_camp5.IdentificationParameterFile.SI-CSI: [filename]
463 
464  TString filename = gDataSet->GetDataSetEnv(Form("IdentificationParameterFile.%s", GetLabel()));
465  if (filename == "") {
466  Warning("SetIdentificationParameters",
467  "No filename defined. Should be given by %s.IdentificationParameterFile.%s",
468  gDataSet->GetName(), GetLabel());
469  return kFALSE;
470  }
471  TString path;
472  if (!SearchKVFile(filename.Data(), path, gDataSet->GetName())) {
473  Error("SetIdentificationParameters",
474  "File %s not found. Should be in $KVROOT/KVFiles/%s",
475  filename.Data(), gDataSet->GetName());
476  return kFALSE;
477  }
478  Info("SetIdentificationParameters", "Using file %s", path.Data());
479  ifstream datfile;
480  datfile.open(path.Data());
481  KVString aline;
482  int zmin, zmax;
483  char type[5], name[20];
484  KVIDSiCsI5* idt = 0;
485 
486  aline.ReadLine(datfile);
487  while (datfile.good()) {
488 
489  if (!aline.BeginsWith('#')) { //skip comments in header
490 
491  if (aline.BeginsWith("+")) {
492 
493  aline.Remove(0, 2);
494  sscanf(aline.Data(), "%s", name);
495  idt = (KVIDSiCsI5*) MDA->GetIDTelescope(name);
496 
497  aline.ReadLine(datfile);
498  Bool_t _Aident = kFALSE;
499  if (aline.BeginsWith("MASS_ID")) {
500  if (idt) idt->SetHasMassID();
501  _Aident = kTRUE;
502  aline.Remove(0, 8);
503  }
504  sscanf(aline.Data(), "%s", type);
505 
506  if (!strcmp(type, "PG2")) {
507  if (idt) idt->SetHasPG2();
508  }
509 
510  aline.ReadLine(datfile);
511  sscanf(aline.Data(), "ZMIN=%d ZMAX=%d", &zmin, &zmax);
512 
513  //create new Tassan-Got ID object for telescope
514  KVTGID* _tgid = 0, *_tgid2 = 0;
515  if (idt) {
516  if (_Aident) {
517  //When mass ID is possible, have to create two objects: one
518  //for Z ident and the other for A ident
519  //A-identification functional. Set name of parameter "Z"
520  _tgid =
521  new KVTGIDZA(idt->
522  GetTGIDName(idt->GetName(), "A", type),
523  "tassangot_A", 0.1, 100., 13, 11, 12);
524  _tgid->SetParName(10, "Z");
525  _tgid2 =
526  new KVTGIDZ(idt->GetTGIDName(idt->GetName(), "Z", type),
527  "pichon_Z", 0.1, 100., 12, 10, 11);
528  } else {
529  //Z-identification functional.
530  _tgid =
531  new KVTGIDZ(idt->GetTGIDName(idt->GetName(), "Z", type),
532  "pichon_Z", 0.1, 100., 12, 10, 11);
533  }
534 
535  _tgid->SetIDmin((Double_t) zmin);
536  _tgid->SetIDmax((Double_t) zmax);
537  if (_Aident) {
538  _tgid2->SetIDmin((Double_t) zmin);
539  _tgid2->SetIDmax((Double_t) zmax);
540  }
541  }
542  //read line with parameters on it from file
543  aline.ReadLine(datfile);
544  if (idt) {
545  unique_ptr<TObjArray> par_arr(aline.Tokenize(" ")); //split up into 1 string per parameter
546  for (int i = 0; i < par_arr->GetEntries(); i++) {
547  //read Double_t value in string
548  KVString kvs(((TObjString*)(*par_arr)[i])->String());
549  Double_t param = kvs.Atof();
550  //set parameter of ID functional
551  _tgid->SetParameter(i, param);
552  if (_Aident)
553  _tgid2->SetParameter(i, param);
554  }
555  //add identification object to telescope's ID manager
556  idt->AddTGID(_tgid);
557  if (_Aident)
558  idt->AddTGID(_tgid2);
559  }
560  }
561  }
562  aline.ReadLine(datfile);
563  } //while( datfile.good()
564  datfile.close();
565  return kTRUE;
566 }

596 Double_t KVIDSiCsI5::GetMeanDEFromID(Int_t& status, Int_t Z, Int_t A, Double_t Eres)
597 {
598  // Returns the Y-axis value in the 2D identification map containing isotope (Z,A)
599  // corresponding to either the given X-axis/Eres value or the current X-axis value given by GetIDMapX.
600  // If no mass information is available, just give Z.
601  //
602  // This method overrides KVIDTelescope::GetMeanDEFromID, as here we have
603  // to handle the LTG fits used for identification. This means scanning the TGID objects associated with
604  // this telescope until we find one with the right Z range, and then calculating
605  // the Y-coordinate for the current X-coordinate value.
606  //
607  // Status is same as for KVIDTelescope::GetMeanDEFromID.
608 
609  status = kMeanDE_OK;
610  if (!HasMassID()) {
611  // mass identification not possible for telescope - ignore A
612  A = -1;
613  }
614  Double_t x = (Eres < 0 ? GetIDMapX() : Eres);
615  // loop over TGID objects
616  TIter next(&GetListOfIDFunctions());
617  KVTGID* tgid;
618  while ((tgid = (KVTGID*)next())) {
619  if (Z >= tgid->GetIDmin() && Z <= tgid->GetIDmax()) {
620  if (A > 0 && tgid->GetZorA()) {
621  // this is a Z-only identification object & we require isotopic resolution
622  continue;
623  }
624  if (A > 0 && !tgid->GetZorA()) {
625  // get position on isotopic line (Z,A)
626  Double_t y = 0;
627  tgid->SetParameter("Z", Z);
628  y = tgid->GetDistanceToLine(x, y, A);
629  return y;
630  }
631  if (A == -1 && !tgid->GetZorA()) {
632  // this is an A-identifying object, but we only care about Z
633  continue;
634  }
635  if (A < 0 && tgid->GetZorA()) {
636  // get position on Z line
637  Double_t y = 0;
638  y = tgid->GetDistanceToLine(x, y, Z);
639  return y;
640  }
641  }
642  }
643  status = kMeanDE_NoIdentifier;
644  return -1.0;
645 }

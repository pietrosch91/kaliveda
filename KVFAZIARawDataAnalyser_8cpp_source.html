90 void KVFAZIARawDataAnalyser::SubmitTask()
91 {
92  // Perform analysis of chosen runs
93  // Before beginning the loop over the runs, the user's InitAnalysis() method is called.
94  // After completing the analysis of all runs, the user's EndAnalysis() method is called.
95  //
96  // Further customisation of the event loop is possible by overriding the methods
97  // preInitAnalysis()
98  // postInitAnalysis()
99  // preEndAnalysis()
100  // postEndAnalysis()
101  // which are executed respectively just before and just after the methods.
102 
103  if (gDataSet != GetDataSet()) GetDataSet()->cd();
104 
105  //loop over runs
106  GetRunList().Begin();
107  while (!GetRunList().End() && !AbortProcessingLoop()) {
108  fRunNumber = GetRunList().Next();
109  Info("SubmitTask", "treatment of run# %d", fRunNumber);
110  ProcessRun();
111  }

126 void KVFAZIARawDataAnalyser::ProcessRun()
127 {
128  // Perform treatment of a given run
129  // Before processing each run, after opening the associated file, user's InitRun() method is called.
130  // After each run, user's EndRun() is called.
131  // For each event of each run, user's Analysis() method is called.
132  //
133  // For further customisation, the pre/post-methods are called just before and just after
134  // each of these methods (preInitRun(), postAnalysis(), etc. etc.)
135  TString fullPathToRunfile = gDataSet->GetFullPathToRunfile(GetDataType(), fRunNumber);
136  TFile* f = (TFile*)gDataSet->OpenRunfile(GetDataType(), fRunNumber);
137  if (!(f && !f->IsZombie())) {
138  Error("ProcessRun", "file %s does not exist or is made zombie", fullPathToRunfile.Data());
139  return;
140  }
141 
142  theTree = (TTree*)f->Get("FAZIA");
143 
144  TString option("");
145  Info("SubmitTask", "Beginning TTree::Process...");
146  if (GetNbEventToRead()) {
147  theTree->Process(GetUserClass(), option.Data(), GetNbEventToRead());
148  } else {
149  theTree->Process(GetUserClass(), option.Data());
150  }
151 
152  f->Close();
153 
154 }

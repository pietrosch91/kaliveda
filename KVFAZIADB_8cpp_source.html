424 void KVFAZIADB::StartTransfer(TString filename, TString ccali_rep, TString option)
425 {
426 
427  Int_t run;
428  TString datadir = "";
429 
430 //-----------
431 // list of runs for which transfer failed during the last tentative
432 //-----------
433  KVNumberList lFAILED;
434  KVFileReader fr;
435  fr.OpenFileToRead(fFAILEDfile.Data());
436  while (fr.IsOK()) {
437  fr.ReadLine(0);
438  if (fr.GetCurrentLine() != "") {
439  run = fr.GetCurrentLine().Atoi();
440  lFAILED.Add(run);
441  }
442  }
443  fr.CloseFile();
444 
445 //-----------
446 // list of runs for which transfer succeeded since the beginning
447 //-----------
448  KVNumberList lDONE;
449  fr.Clear();
450  fr.OpenFileToRead(fDONEfile.Data());
451  while (fr.IsOK()) {
452  fr.ReadLine(0);
453  if (fr.GetCurrentLine() != "") {
454  run = fr.GetCurrentLine().Atoi();
455  lDONE.Add(run);
456  }
457  }
458  if (lDONE.GetNValues() > 0)
459  Info("StartTransfer", "%d runs already tranfered:\n%s", lDONE.GetNValues(), lDONE.AsString());
460  fr.CloseFile();
461 
462 //-----------
463 // all runs produced
464 //-----------
465 
466  KVNumberList lALL;
467  fr.Clear();
468  fr.OpenFileToRead(filename.Data());
469  while (fr.IsOK()) {
470 
471  fr.ReadLine("|");
472  if (fr.GetCurrentLine().BeginsWith("#")) {
473  KVString line = fr.GetCurrentLine();
474  line.ReplaceAll("#", "");
475  line = line.StripAllExtraWhiteSpace();
476  line.Begin("=");
477  KVString name = line.Next();
478  if (!line.End() && name == "--dir") {
479  datadir = line.Next();
480  Info("ReadDBFile", "acquisition file directory: %s", datadir.Data());
481  }
482  } else {
483  if (datadir == "") {
484  fr.CloseFile();
485  Warning("ReadDBFile", "no data directory has been found ...");
486  return;
487  }
488  if (fr.GetNparRead() > 0) {
489  for (Int_t ii = 0; ii < fr.GetNparRead(); ii += 1) {
490 
491  KVString couple = fr.GetReadPar(ii);
492  couple.Begin("=");
493  KVString name = couple.Next();
494  name = name.Strip(TString::kBoth);
495  KVString value = "";
496  if (!couple.End()) {
497  value = couple.Next();
498  value = value.Strip(TString::kBoth);
499  }
500  if (name == "run") {
501  run = value.Atoi();
502  if (!lDONE.Contains(run))
503  lALL.Add(run);
504  }
505  }
506  }
507  }
508  }
509  fr.CloseFile();
510 
511  if (lALL.GetNValues() > 0) {
512  Info("StartTransfer", "Start transfer for %d runs:\n%s", lALL.GetNValues(), lALL.AsString());
513  TransferRunListToCcali(lALL, datadir.Data(), ccali_rep, option);
514  } else
515  Info("StartTransfer", "no runs to tranfer ...");
516 
517 }

588 void KVFAZIADB::ReadExceptions()
589 {
590  KVFileReader* fr = GetKVFileReader("ReadExceptions()", "Exceptions");
591  if (!fr)
592  return;
593 
594  TList* ll = new TList();
595  KVNumberList lruns;
596  KVDBParameterList* dbp = 0;
597 
598  ll->SetOwner(kFALSE);
599  while (fr->IsOK()) {
600  fr->ReadLine(":");
601  if (fr->GetNparRead() == 2) {
602  if (fr->GetReadPar(0) == "RunRange") {
603  if (ll->GetEntries() > 0) {
604  LinkListToRunRange(ll, lruns);
605  ll->Clear();
606  }
607  lruns.SetList(fr->GetReadPar(1));
608  //printf("nouvelle plage : %s\n", lruns.AsString());
609  } else {
610  KVString name(fr->GetReadPar(0));
611  name.Begin(".");
612  KVString sdet = name.Next();
613 
614  KVString sig = name.Next();
615  if (sig == "QH1" || sig == "QL1" || sig == "I1") sdet.Prepend("SI1-");
616  else if (sig == "Q2" || sig == "I2") sdet.Prepend("SI2-");
617  else if (sig == "Q3") sdet.Prepend("CSI-");
618 
619  KVString par = name.Next();
620  if (!(dbp = (KVDBParameterList*)ll->FindObject(Form("%s.%s", sdet.Data(), sig.Data())))) {
621  dbp = new KVDBParameterList(Form("%s.%s", sdet.Data(), sig.Data()), sdet.Data());
622  dbp->AddKey("Runs", "List of Runs");
623  fExceptions->AddRecord(dbp);
624  ll->Add(dbp);
625  dbp->GetParameters()->SetValue("RunRange", lruns.AsString());
626  dbp->GetParameters()->SetValue("Detector", sdet.Data());
627  dbp->GetParameters()->SetValue("Signal", sig.Data());
628  }
629  dbp->GetParameters()->SetValue(par.Data(), fr->GetDoubleReadPar(1));
630  }
631  }
632  }
633 
634  if (ll->GetEntries() > 0) {
635  LinkListToRunRange(ll, lruns);
636  ll->Clear();
637  }
638 
639  delete ll;
640  delete fr;
641 
642 }

765 void KVFAZIADB::ReadCalibFile(const Char_t* filename)
766 {
767 
768  Bool_t find = kFALSE;
769  TString fullpath = "";
770  if (gSystem->Exec(Form("test -e %s/%s", GetDataSetDir(), filename)) == 0) {
771  //Info("ReadCalibFile", "%s/%s do not exist", GetDataSetDir(), filename);
772  //return;
773  find = kTRUE;

776  //Info("ReadCalibFile", "%s do not exist", filename);
777  //return;
778  find = kTRUE;
779  fullpath.Form("%s", filename);

786  Info("ReadCalibFile", "file : %s found", fullpath.Data());
787  TEnv env;
788  env.ReadFile(Form("%s", fullpath.Data()), kEnvAll);
789  TIter next(env.GetTable());
790  TEnvRec* rec = 0;
791  KVDBParameterSet* par = 0;
792  KVNumberList default_run_list;
793  default_run_list.SetMinMax(GetFirstRunNumber(), GetLastRunNumber());
794 
795  TString ssignal = "";
796  TString stype = "";
797 
798  while ((rec = (TEnvRec*)next())) {
799  TString sname(rec->GetName());
800  if (sname == "Signal") {
801  ssignal = rec->GetValue();
802  } else if (sname == "CalibType") {
803  stype = rec->GetValue();
804  } else {
805  KVString lval(rec->GetValue());
806  Int_t np = 0;
807  lval.Begin(",");
808  while (!lval.End()) {
809  KVString sval = lval.Next();
810  np += 1;
811  }
812  par = new KVDBParameterSet(sname.Data(), stype.Data(), np);
813 
814  np = 0;
815  lval.Begin(",");
816  while (!lval.End()) {
817  par->SetParameter(np++, lval.Next().Atof());
818  }
819  fCalibrations->AddRecord(par);
820  LinkRecordToRunRange(par, default_run_list);
821  }
822  }
823 
824  if (ssignal == "") Error("ReadCalibFile", "No signal defined");
825  if (stype == "") Error("ReadCalibFile", "No calibration type defined");
826 }

874 void KVFAZIADB::BuildQuickAndDirtyDataBase(TString acqfiles_dir)
875 //---------------------------------------
876 
877 
892 {
893  /*
894  Use this method in order to read runs written on disk
895  during the experiment
896  File is produced with main information for each run :
897  number of files
898  size
899  starting date
900  Information on runs has to completed after reading acquisition file
901  specially information on trigger and deatdime
902 
903  usage : acqfiles_dir is the repository where acquisition files are written
904 
905  2 output files are written
906  - runlist.dat containing all runs
907  - database.log giving a summarize of stored data
908  the two file names are set by default in .kvrootrc
909 
910  */
911  if (gSystem->Exec(Form("test -d %s", acqfiles_dir.Data())) != 0) {
912  Error("BuildQuickAndDirtyDataBase", "%s is not an existing directory", acqfiles_dir.Data());
913  return;
914  }
915 
916  TString sfout = GetCalibFileName("Runlist");
917  FILE* fout = fopen(sfout.Data(), "w");
918  fprintf(fout, "# DataSet %s\n", gDataSet->GetName());
919  fprintf(fout, "# Runlist generated by KVFAZIADB::BuildQuickAndDirtyDataBase from files located at:\n");
920  fprintf(fout, "# --dir=%s\n", acqfiles_dir.Data());
921 
922  TString sflog = GetCalibFileName("DBLog");
923  FILE* flog = fopen(sflog.Data(), "w");
924  fprintf(flog, "# DataSet %s\n", gDataSet->GetName());
925  fprintf(flog, "# Log generated by KVFAZIADB::BuildQuickAndDirtyDataBase from files located at:\n");
926  fprintf(flog, "# --dir=%s\n", acqfiles_dir.Data());
927 
928  Int_t run, nfiles;
929  Long64_t size = 0;
930  Long64_t totalsize = 0;
931  Int_t date, idx;
932  KVDatime kvdate;
933 
934  KVString ldir = gSystem->GetFromPipe(Form("du %s", acqfiles_dir.Data()));
935  ldir.Begin("\n");
936  Int_t numberofruns = 0;
937  KVNumberList lruns;
938 
939  while (!ldir.End()) {
940  KVString sdir = ldir.Next();
941  sdir.Begin("\t");
942  size = sdir.Next().Atoll(); // /TMath::Power(2,10)+1; //read values are in KB
943  sdir = gSystem->BaseName(sdir.Next());
944  if (sscanf(sdir.Data(), "run%d", &run) == 1) {
945  totalsize += size;
946  Int_t dmin = 0;
947  Int_t dmax = 0;
948  KVString lfile = gSystem->GetFromPipe(Form("ls %s/%s", acqfiles_dir.Data(), sdir.Data()));
949  lfile.Begin("\n");
950  nfiles = 0;
951  while (!lfile.End()) {
952  KVString sfile = lfile.Next();
953  //FzEventSet-1434434675-41784.pb
954  if (sscanf(sfile.Data(), "FzEventSet-%d-%d.pb", &date, &idx) == 2) {
955  if (dmin == 0) {
956  dmin = dmax = date;
957  } else {
958  if (date < dmin) dmin = date;
959  if (date > dmax) dmax = date;
960  }
961  nfiles += 1;
962  }
963  }
964  if (nfiles == 0 || size == 0) {
965  Warning("BuildQuickAndDirtyDataBase", "%d -> empty run", run);
966  }
967  kvdate.Set(dmin);
968  fprintf(fout, "run=%d | starting date=%s | nfiles=%d | size(GB)=%1.2lf\n", run, kvdate.AsString(), nfiles, size * TMath::Power(2., -20));
969  numberofruns += 1;
970  lruns.Add(run);
971  } else {
972  Double_t conv = TMath::Power(2., -30);
973  fprintf(flog, "total size %lld - %lld\n", size, totalsize);
974  fprintf(flog, "total size (TB) %1.2lf - %1.2lf\n", size * conv, totalsize * conv);
975  fprintf(flog, "number of runs %d\n", numberofruns);
976  fprintf(flog, "last run %d done at %s\n", lruns.Last(), kvdate.AsString());
977  }
978  }
979 
980  fclose(fout);
981  fclose(flog);
982 }

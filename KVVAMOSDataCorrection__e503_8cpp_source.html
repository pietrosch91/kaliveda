333 void KVVAMOSDataCorrection_e503::ReadHFCutFileList(std::ifstream& file, Int_t type)
334 {
335  //Protected method to read the HF cuts list file for the dataset e503
336  //Set 'type' to 0 for SiCsI telescopes and set 'type' to 1 for ICSi telescopes
337  //For each file listed in this list file we:
338  //- read the file TEnv and to access to the run range of the file
339  //- if 'fRunNumber' is in the run range we save all associated cuts in
340  // 'flist_HFcuts_type' list and stop the research.
341  //
342  //2 cases: i) Si-CsI telescopes
343  // ii) IC-Si telescopes
344  //i) If 'fRunNumber' is in the run range we also save the values of N=
345  // number of time we add/remove the beam pulse in 'fvec_nHF_type'
346  //
347  //ii) For IC-Si telescopes the situation is little bit more complicated
348  // because of punch-through particles and average ToF longer than the Si-CsI case.
349  // For the particles punching through, the corrected nHF
350  // (see KVVAMOSReconNuc::CalculateCorrectedT_HF() method) is not correct. We then
351  // separate particles above (IDCode==4) or below (IDCode==11) punch-through.
352  // -For IDCode==1 (punch-through) we use the ToF directly from calibration function
353  // so without corrections.
354  // -For IDCode==4 we use the ToF from KVVAMOSReconNuc::CalculateCorrectedT_HF() method.
355  //
356  // Another problem comes from the ToF longer than the beam frequency (which is the case
357  // for a lot of particles identified in IC-Si telescopes). The chosen solution is to
358  // use the "TINDRA_HF" in channel (=difference in time between the first detector hit
359  // in INDRA and the same HF than the "TSI_HF" signal): long-time particles in VAMOS are
360  // in general correlated to long-time particle in INDRA as they come from the same source.
361  // We experimentaly see "domains" of TINDRA_HF corresponding to domains of long/short ToF,
362  // we use these domains to de-correlate long and short ToF.
363  // Finally, for IC-Si telescopes we save in 'fvec_nHF_type' a list of vectors for each cut with:
364  // -the IDCode (to know if we use the calibration ToF (IDCode==11) or the already corrected
365  // ToF from KVVAMOSReconNuc::CalculateCorrectedT_HF() method (IDCode==4))
366  // -the domain of TINDRA_HF associated to the cut
367  // -the number of time we add/remove the beam pulse for the cut
368 
369  //Initialisation
370  Bool_t kDE_ToF_found = kFALSE; //flag set to kTRUE as soon as we have found the good cut file for DE-ToF
371  Bool_t kAE_AoQ_found = kFALSE; //flag set to kTRUE as soon as we have found the good cut file for AE-AoQ
372  Bool_t kfound = kFALSE;
373  KVHashList* ll = NULL;
374  KVHashList* sub_ll = NULL; //sublist of ll for either type="DeltaE-ToF" or type="AE_AoQ"
375  std::vector<Float_t>* vec;
376 
377  //Si-CsI case
378  if (type == 0) ll = flist_HFcuts_sicsi;
379  //IC-Si case
380  else if (type == 1) ll = flist_HFcuts_icsi;
381  assert(ll);
382 
383  //read the file
384  std::string newline;
385  while ((std::getline(file, newline)) && (!kfound)) {
386  //Ignore comments in the file
387  if (newline.compare(0, 1, "#") != 0) {
388 
389  TString fullpath;
390  TFile* file = NULL;
391  if (gDataSet->SearchKVFile(newline.c_str(), fullpath, gDataSet->GetName())) {
392  file = TFile::Open(fullpath.Data());
393  if (file) {
394 
395  if (fkverbose) {
396  Info("ReadHFCutFileList", "... file %s opened, infos follow ...", file->GetName());
397  file->ls();
398  }
399 
400  //find the TEnv and access to runlist
401  TEnv* env = (TEnv*) file->Get("TEnv");
402  assert(env);
403 
404  if (fkverbose) {
405  Info("ReadHFCutFileList", "... TEnv infos follow ...");
406  env->Print();
407  }
408 
409  const Char_t* runlist = env->GetValue("runlist", "");
410  if (fkverbose) {
411  Info("ReadHFCutFileList", "... Run %d | runlist of the file is '%s' ...",
412  fRunNumber, runlist);
413  }
414 
415  //check if 'fRunNumber' is in run range
416  KVNumberList nl(runlist);
417  if (nl.Contains(fRunNumber)) {
418  Info("ReadHFCutFileList", "... Run %d is in the runlist '%s' of the file '%s' ...",
419  fRunNumber, runlist, file->GetName());
420 
421  //extract the type of observables the cuts refer to
422  //(either 'DeltaE-ToF' cut or 'AE_AoQ' cut)
423  const Char_t* car_type = env->GetValue("type", "");
424  TString ss_type(car_type);
425  if (ss_type == "DeltaE_ToF") {
426  kDE_ToF_found = kTRUE;
427  sub_ll = (KVHashList*) ll->At(0);
428  if (type == 0) vec = &fvec_nHF_DE_ToF_sicsi;
429  else if (type == 1) vec = &fvec_nHF_DE_ToF_icsi;
430  } else if (ss_type == "AE_AoQ") {
431  kAE_AoQ_found = kTRUE;
432  sub_ll = (KVHashList*) ll->At(1);
433  if (type == 0) vec = &fvec_nHF_AE_AoQ_sicsi;
434  else if (type == 1) vec = &fvec_nHF_AE_AoQ_icsi;
435  }
436  assert(sub_ll);
437  assert(vec);
438 
439  //exctract and set cuts list for the specific type
440  TKey* key = NULL;
441  TList* list = file->GetListOfKeys();
442  assert(list);
443  TIter it(list);
444  while ((key = (TKey*) it.Next())) {
445  if ((key->ReadObj())->InheritsFrom("TCutG")) {
446  const TCutG* cutg = (TCutG*) key->ReadObj();
447  TCutG* copy_cut = new TCutG(*cutg);
448  sub_ll->Add(copy_cut);
449 
450  //save associated nHF value
451  //it will work only if the cut saved in the file
452  //has a name of the form: "Cut%d_IDCode%d_nHF%f",ncut, idc, nHF
453  TString ss = cutg->GetName();
454  TObjArray* obj_ar = ss.Tokenize("_");
455 
456  TString str_nn = (((TObjString*) obj_ar->At(0))->String());
457  TObjArray* obj_nn = str_nn.Tokenize("Cut");
458  Int_t nn = (((TObjString*) obj_nn->At(0))->String()).Atoi();
459 
460  TString str_idc = (((TObjString*) obj_ar->At(1))->String());
461  TObjArray* obj_idc = str_idc.Tokenize("IDCode");
462  Int_t idc = (((TObjString*) obj_idc->At(0))->String()).Atoi();
463 
464  TString str_nhf = (((TObjString*) obj_ar->At(2))->String());
465  TObjArray* obj_nhf = str_nhf.Tokenize("nHF");
466  Float_t nhf = (((TObjString*) obj_nhf->At(0))->String()).Atof();
467 
468  vec->push_back(nhf);
469 
470  if (fkverbose) Info("ReadHFCutFileList", "... [%s] cut name = '%s' -> (ncut=%d, idc=%d, nHF=%f) ...", ss_type.Data(), cutg->GetName(), nn, idc, nhf);
471  }
472  }
473  }
474 
475  else if (fkverbose) Info("ReadHFCutFileList", "... Run %d is NOT in the runlist '%s' of the file '%s' ...", fRunNumber, runlist, file->GetName());
476  file->Close();
477  }//end of file is ok
478  }//end of SearchKVFile
479  }//end of line ok
480 
481  if (kDE_ToF_found && kAE_AoQ_found) kfound = kTRUE;
482  }//all lines read
483 }

1774  // direction)

1826  dir.SetY(momentum.Y() / mag);
1827  dir.SetZ(momentum.Z() / mag);

1839  // above.

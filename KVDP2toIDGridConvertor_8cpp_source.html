106 void KVDP2toIDGridConvertor::ReadFile(const Char_t* filename)
107 {
108  // Read grids in file and add to fGrids list
109 
110  if (!fGridClass) {
111  Error(KV__ERROR(ReadFile),
112  "Set class of ID grids to generate first!");
113  return;
114  }
115 
116  ifstream datfile;
117  datfile.open(filename);
118 
119  if (!datfile.good()) {
120  Error(KV__ERROR(ReadFile),
121  "Problem reading file %s", filename);
122  return;
123  }
124  KVString s;
125 
126  s.ReadLine(datfile);
127 
128  while (datfile.good()) {
129 
130  if (s == "") {
131  s.ReadLine(datfile);
132  continue;
133  }
134 
135  if (!s.BeginsWith('#')) { //'#' sign signals comments
136  int ring, modu, frun, lrun, totpoints;
137  if (sscanf(s.Data(), "%d %d %d %d", &ring, &modu, &frun, &lrun) !=
138  4) {
139  Error(KV__ERROR(ReadFile), "Problem reading file %s\nLast line read: %s",
140  filename, s.Data());
141  return;
142  };
143  KVIDGrid* grid = (KVIDGrid*) fGridClass->New(); //create new grid
144 
145  //add to list of grids
146  fGrids->Add(grid);
147 
148  grid->GetParameters()->SetValue("First run", frun);
149  grid->GetParameters()->SetValue("Last run", lrun);
150  grid->GetParameters()->SetValue("Ring min", ring);
151  grid->GetParameters()->SetValue("Ring max", ring);
152  grid->GetParameters()->SetValue("Mod min", modu);
153  grid->GetParameters()->SetValue("Mod max", modu);
154  int nlines;
155  datfile >> nlines;
156  totpoints = 0;
157  //read Z,A and number of points in each line
158  for (int i = 0; i < nlines; i++) {
159  int z, a, npoints;
160  datfile >> z >> a >> npoints;
161  totpoints += 2 * npoints;
162  if (z > 0) {
163  //identification line
164  KVIDLine* line = (KVIDLine*)grid->NewLine("id");
165  line->SetZ(z);
166  line->SetA(a);
167  line->Set(npoints);
168  grid->Add("id", line);
169  } else {
170  //"ok" line
171  KVIDLine* line = (KVIDLine*)grid->NewLine("ok");
172  line->Set(npoints);
173  grid->Add("ok", line);
174  }
175  }
176  ReadLineCoords(grid, datfile);
177  //calculate line widths in grid
178  grid->CalculateLineWidths();
179  }
180  s.ReadLine(datfile);
181  }
182  datfile.close();
183 }

191 void KVDP2toIDGridConvertor::ReadLineCoords(KVIDGrid* grid, ifstream& datfile)
192 {
193  // Read coordinates of lines in the order they were created.
194  // We assume that the 'ID' lines are first, then the 'OK' lines
195 
196  TIter nextID(grid->GetIdentifiers());
197  KVIDentifier* line;
198  while ((line = (KVIDentifier*) nextID())) {
199  //read in points
200  for (int i = 0; i < line->GetN(); i++) {
201  Double_t x, y;
202  datfile >> x >> y;
203  line->SetPoint(i, x, y);
204  }
205  }
206  TIter nextOK(grid->GetCuts());
207  while ((line = (KVIDentifier*) nextOK())) {
208  //read in points
209  for (int i = 0; i < line->GetN(); i++) {
210  Double_t x, y;
211  datfile >> x >> y;
212  line->SetPoint(i, x, y);
213  }
214  }
215 }

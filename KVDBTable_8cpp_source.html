139 void KVDBTable::SetDefaultFormat(const TString& fmt)
140 {
141  // Set a default formatting string for names of numbered records.
142  // e.g. if fmt="Rec#%d", each record which is added to the table will have
143  // its name automatically set to "Rec#1", "Rec#2", etc., according to the
144  // record number.
145  // This allows to replace a slow sequential scan for a record based on number
146  // (GetRecord(Int_t)) with a fast hash-based look-up
147 
148  fDefFormatNumRec = fmt.Data();
149 }

163 KVDBRecord* KVDBTable::GetRecord(Int_t num) const
164 {
165  // Sequential search for record using its number.
166  // To keep things simple, we only search in the top level of the folder structure.
167  //
168  // NOTE: if there are many records, this sequential search can be long.
169  // If a default format for the record name using its number has been defined
170  // (see SetDefaultFormat(const TString&)) the search will be performed using
171  // the resulting formatted object name, which is fast (hash list).
172 
173  if (HasDefaultFormat()) return GetRecord(Form(fDefFormatNumRec.Data(), num));
174 
175  TIter next(GetListOfFolders());
176  KVDBRecord* obj = 0;
177  while ((obj = (KVDBRecord*) next()) && (obj->GetNumber() != num));
178  return (obj ? (obj->GetNumber() == num ? obj : 0) : 0);
179 }

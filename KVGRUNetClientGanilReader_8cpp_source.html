116 void KVGRUNetClientGanilReader::ConnectRawDataParameters()
117 {
118  GEvent* the_event = fGRUAcq->GetEvent();
119  GDataParameters* the_parameters = fGRUAcq->GetDataParameters();
120  Int_t nbparams = the_parameters->GetNbParameters();
121  TString par_name;
122  for (int i = 0; i < nbparams; i++) {
123  par_name = the_parameters->GetParName(i);
124  KVACQParam* par = CheckACQParam(par_name.Data());
125  the_event->Connect(par_name.Data(), par->ConnectData());
126  par->SetNumber(the_parameters->GetIndex(par_name.Data()));
127  par->SetNbBits(the_parameters->GetNbits(par_name.Data()));
128  fParameters->Add(par);
129  }
130 }

179 Bool_t KVGRUNetClientGanilReader::GetNextEvent()
180 {
181  // Get next event from buffer
182  // Returns kTRUE if an event is read.
183  // Returns kFALSE if no acquisition running (check status)
184  // Returns kFALSE if no buffer/event read after fTimeout seconds (check status)
185 
186  // check status of acquisition
187  ((GNetClientGanil*)fGRUAcq->GetDeviceIn())->GetStatusBuffer(0);
188  if (!((GNetClientGanil*)fGRUAcq->GetDeviceIn())->GetAcquisitionOn()) {
189  fStatus = kSTATUS_NOACQ;
190  return kFALSE;
191  }
192  TDatime start;
193  TDatime now;
194  fStatus = kSTATUS_OK;
195  if (fReadFirstBuffer) {
196  // first time we call, we need to read a buffer
197  fReadFirstBuffer = kFALSE;
198  fGRUAcq->GetDeviceIn()->ReadBuffer();
199  fGRUAcq->GetEvent()->RazEvent();
200  while (!fGRUAcq->GetDeviceIn()->GetCurrentBuffer()->IsAeventBuffer()) {
201  fGRUAcq->GetDeviceIn()->ReadBuffer();
202  now.Set();
203  if ((now.Convert() - start.Convert()) >= fTimeout) {
204  fStatus = kSTATUS_NOBUFF;
205  return kFALSE;
206  }
207  }
208  }
209  int Status = fGRUAcq->GetEvent()->NextEvent(fGRUAcq->GetDeviceIn()->GetCurrentBuffer());
210  if (Status != ACQ_OK) {
211  // we have read all events in buffer. get a new buffer.
212  fGRUAcq->GetDeviceIn()->ReadBuffer();
213  fGRUAcq->GetEvent()->RazEvent();
214  start.Set();
215  now.Set();
216  while (!fGRUAcq->GetDeviceIn()->GetCurrentBuffer()->IsAeventBuffer()) {
217  fGRUAcq->GetDeviceIn()->ReadBuffer();
218  now.Set();
219  if ((now.Convert() - start.Convert()) >= fTimeout) {
220  fStatus = kSTATUS_NOBUFF;
221  return kFALSE;
222  }
223  }
224  start.Set();
225  now.Set();
226  while (Status != ACQ_OK) {
227  Status = fGRUAcq->GetEvent()->NextEvent(fGRUAcq->GetDeviceIn()->GetCurrentBuffer());
228  now.Set();
229  if ((now.Convert() - start.Convert()) >= fTimeout) {
230  fStatus = kSTATUS_NOEVENT;
231  return kFALSE;
232  }
233  }
234  }
235  FillFiredParameterList();
236  return kTRUE;
237 }

153 Double_t KVRangeYanezMaterial::DeltaEFunc(Double_t* E, Double_t* p)
154 {
155  // Function parameterising the energy loss of charged particles in this material.
156  // This is simply an interface to the passage() function of the Range C library.
157  // The incident energy E[0] is given in MeV.
158  // The energy loss is calculated in MeV.
159  // To avoid divergences as E->0, for any incident energy E<=1.e-3MeV (i.e. 1keV)
160  // we return dE=E i.e. all particles with E<=1keV are stopped.
161 
162  return (E[0] - EResFunc(E, p));
163 }

175 Double_t KVRangeYanezMaterial::EResFunc(Double_t* E, Double_t*)
176 {
177  // Function parameterising the residual energy of charged particles after traversing this material.
178  // This is simply an interface to the passage() function of the Range C library.
179  // The incident energy E[0] is given in MeV.
180  // The residual energy is calculated in MeV.
181  // To avoid divergences as E->0, for any incident energy E<=1.e-3MeV (i.e. 1keV)
182  // we return Eres=0 i.e. all particles with E<=1keV are stopped.
183 
184  if (E[0] < 1.e-3) return 0.0;
185  return passage(fTableType, Zp, Ap, iabso, fAbsorb[0].z, fAbsorb[0].a, E[0], thickness / KVUnits::mg, &error);
186 }

198 Double_t KVRangeYanezMaterial::RangeFunc(Double_t* E, Double_t*)
199 {
200  // Function parameterising the range of charged particles in this material.
201  // This is simply an interface to the rangen() function of the Range C library.
202  // The incident energy E[0] is given in MeV.
203  // The range is calculated in g/cm**2.
204  // To avoid divergences as E->0, for any incident energy E<=1.e-3MeV (i.e. 1keV)
205  // we return range=0 i.e. all particles with E<=1keV are stopped.
206 
207  if (E[0] < 1.e-3) return 0.;
208  Double_t R = rangen(fTableType, Zp, Ap, iabso, fAbsorb[0].z, fAbsorb[0].a, E[0]);
209  return (R * KVUnits::mg);
210 }

280 TF1* KVRangeYanezMaterial::GetRangeFunction(Int_t Z, Int_t A, Double_t)
281 {
282  // Return function giving range (in g/cm**2) as a function of incident energy (in MeV) for
283  // charged particles (Z,A) traversing this material.
284  // isotopic mass isoAmat argument is not used.
285 
286  PrepareRangeLibVariables(Z, A);
287  // Yanez' range functions tend to have a (negative) minimum at very low energy,
288  // below which the range diverges towards +infty at E=0.
289  // Therefore we limit the range of the function to E>=Emin where Emin
290  // is the energy for which the minimum occurs
291  Double_t minX = fRange->GetMinimumX(1.e-6, 400 * Ap);
292  fRange->SetRange(TMath::Max(1.e-6, minX), 400 * Ap);
293  return fRange;
294 }

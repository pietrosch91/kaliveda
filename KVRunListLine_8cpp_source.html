72 Int_t KVRunListLine::BreakLineIntoFields(const char _delim)
73 {
74  //Break down "line" into fields separated by the delimiter "_delim". Fields are stored in TObjArray of TObjString s (including empty fields
75  //i.e. 2 consecutive delimiters in "line" with nothing between are translated as a field whose content is "". TString::Tokenize() simply skips over such
76  //cases, so we can't use it with our field idxing system). Remove leading/trailing white space from each field.
77  //Return the total number of fields found (including empty fields).
78 
79  Int_t allfields = 0;
80  Int_t start = 0;
81  TString delim(_delim);
82  Int_t end = fLine.Index(delim);
83  TObjArray& tca = *fFields;
84  while (end > -1) {
85 
86  tca[allfields] = new TObjString();
87 
88  if (start < end) {
89  TString _dummy = fLine(start, end - start);
90  KVString dummy(_dummy);
91 
92  //remove trailing/leading tabs
93  dummy.Remove(TString::kBoth, '\t');
94  //remove trailing/leading whitespace
95  dummy.Remove(TString::kBoth, ' ');
96 
97 #if ROOT_VERSION_CODE >= ROOT_VERSION(3,5,0)
98  //since ROOT v3.05 SetString(const char *)
99  ((TObjString*) tca[allfields])->SetString(dummy.Data());
100 #else
101  //before ROOT v3.05 SetString(char *)
102  ((TObjString*) tca[allfields])->SetString((char*) dummy.Data());
103 #endif
104 
105  }
106  allfields++;
107  //check size of TObjArray fFields - if too many fields are found, double size of array
108  if (allfields == fMaxNFields) {
109  fMaxNFields *= 2;
110  tca.Expand(fMaxNFields);
111  }
112  start = end + 1;
113  end = fLine.Index(delim, end + 1);
114  //if last column does not have a delimiter on the right, we take rest of line up to end
115  if (end == -1 && start < fLine.Length())
116  end = fLine.Length();
117  }
118  //does this line define the field names ?
119  if (IsFieldHeader())
120  SetFields();
121  return allfields;
122 }

135 {
136  //set field idxes based on field header line in file identified by IsFieldHeader().

138  //we check that at least one index has been defined at the end

152  //non-null ASCII string ? use it as column header
153  fIndexList.SetValue(field->GetString().Data(), idx);

157  }

339 Int_t KVRunListLine::GetTrigger(const Char_t* field_name,
340  const Char_t* fmt)
341 {
342  //Returns value of "Trigger" field, assuming it is written with format "fmt" i.e. if fmt = "M>=%d" (default) we expect "M>=1", "M>=4" etc.
343  //The actual field name for the trigger is given as first argument (default value = "Trigger")
344  //If trigger field is not present, we return -1 (but no error)
345  //If format is not respected, an error message is printed and -1 is returned.
346 
347  if (!HasFieldValue(field_name))
348  return -1;
349  Int_t trigger = -1;
350  if (sscanf(GetField(field_name), fmt, &trigger) != 1) {
351  Warning(Form("%s::GetTrigger", IsA()->GetName()),
352  "trigger field - %s - unreadable", GetField(field_name));
353  }
354  return trigger;
355 }

167 Bool_t KVINDRARunSheetReader::ReadRunSheet(Int_t run)
168 {
169  //Read runsheet corresponding to run number 'run'
170  //Store the informations we read in the runsheet in the run database.
171  //Create TTree if required and if not already done
172  //Return kTRUE if all goes well.
173 
174  cout <<
175  "<KVINDRARunSheetReader::ReadRunSheet> : Reading runsheet for new run, Run "
176  << run << endl;
177  KVINDRADBRun* indra_db_run = gIndraDB->GetRun(run);
178  if (!indra_db_run) {
179  //create new run and add it to the database
180  indra_db_run = new KVINDRADBRun(run);
181  gIndraDB->AddRun(indra_db_run);
182  }
183 
184  //open runsheet file
185  ifstream read_file(GetRunSheetFileName(run));
186  if (!read_file.good()) {
187  cout << "Error opening file : " << fFilePath.Data() << endl;
188  return kFALSE;
189  }
190  //reset runsheet/TTree variables
191  init_vars();
192 
193  KVString line, dumstr;
194 
195  line.ReadLine(read_file);
196 
197  //read all infos up to list of scalers (which begins with the line "*** Impression des Echelles :"
198  while (read_file.good() && !line.Contains("Echelles")) {
199 
200  if (line.Contains("Run numero"))
201  run_num = (Int_t) GetNumberField(line);
202  if (line.Contains("Debut")) {
203  start_run.SetGanacqDate(GetDateField(line));
204  istart = start_run.Convert();
205  if (!fTIME_START) {
206  //we put the "time origin" at the start of the first run we read
207  fTIME_START = start_run.Convert();
208  }
209  indra_db_run->SetStartDate(start_run.AsGanacqDateString());
210  }
211  if (line.Contains("Fin")) {
212  endrun.SetGanacqDate(GetDateField(line));
213  iend = endrun.Convert();
214  indra_db_run->SetEndDate(endrun.AsGanacqDateString());
215  }
216  if (line.Contains("Length of run")) {
217  sscanf(line.Data(),
218  "Length of run: %lf seconds, Size : %lf MB ( %lf MB/sec)",
219  &len_run, &size, &data_rate);
220  indra_db_run->SetTime(len_run / 60.);
221  indra_db_run->SetSize(size);
222  }
223  if (line.Contains("Stat_eve")) {
224  dumstr = GetStringField(line);
225  stat_eve = GetStringField(dumstr, " ", 0);
226  }
227  if (line.Contains("Buffers sauves")) {
228  dumstr = GetStringField(line);
229  buf_sav = (Int_t) GetNumberField(dumstr, " ", 0);
230  eve_sav = (Int_t) GetNumberField(line, ":", 2);
231  indra_db_run->SetScaler("Buffers", buf_sav);
232  indra_db_run->SetEvents(eve_sav);
233  }
234  if (line.Contains("Evenements lus"))
235  eve_lus = (Int_t) GetNumberField(line);
236  if (line.Contains("Buffers ctrl")) {
237  dumstr = GetStringField(line);
238  buf_ctrl = (Int_t) GetNumberField(dumstr, " ", 0);
239  eve_ctrl = (Int_t) GetNumberField(line, ":", 2);
240  }
241  if (line.Contains("Taux d'acquisition")) {
242  dumstr = GetStringField(line);
243  acq_rate = GetNumberField(dumstr, " ", 0);
244  }
245  if (line.Contains("Taux de traitement")) {
246  dumstr = GetStringField(line);
247  trait_rate = GetNumberField(dumstr, " ", 0);
248  }
249  if (line.Contains("Taux de controle")) {
250  dumstr = GetStringField(line);
251  ctrl_rate = GetNumberField(dumstr, " ", 0);
252  }
253  if (line.Contains("Remplissage")) {
254  dumstr = GetStringField(line);
255  rempli_dlt_pc = GetNumberField(dumstr, " ", 0);
256  rempli_dlt_blocs = (Int_t) GetNumberField(dumstr, " ", 3);
257  }
258 
259  //read next line
260  line.ReadLine(read_file);
261  }
262  if (line.Contains("Echelles")) {
263  //found beginning of scalers
264  line.ReadLine(read_file);
265  //find first scaler line
266  while (read_file.good() && !line.Contains(":")) line.ReadLine(read_file);
267  //now read all scalers
268  while (read_file.good() && line.Contains(":")) {
269  dumstr = GetStringField(line, ":", 0);
270  dumstr.Remove(TString::kBoth, ' ');
271  Int_t scaler = (Int_t)GetNumberField(line);
272  //cout << "SCALER: \"" << dumstr.Data() << "\" = " << scaler << endl;
273  indra_db_run->SetScaler(dumstr.Data(), scaler);
274  line.ReadLine(read_file);
275  if (fMakeTree) {
276  dumstr.ReplaceAll(" ", "_");
277  dumstr.ReplaceAll("-", "_");
278  TBranch* br_sca = fTree->FindBranch(dumstr.Data());
279  Int_t index;
280  //do we need to add a branch for this scaler ?
281  if (!br_sca) {
282  index = fTree->GetListOfBranches()->GetEntries();
283  br_sca = fTree->Branch(dumstr.Data(), &fScalers[index], Form("%s/I", dumstr.Data()));
284  //cout << "br_sca = fTree->Branch( \"" <<dumstr.Data()<<"\", &fScalers["<<index<<"], "<<Form("%s/I", dumstr.Data())<<");"<<endl;
285  } else {
286  index = fTree->GetListOfBranches()->IndexOf(br_sca);
287  }
288  //fill tableau with value
289  fScalers[index] = scaler;
290  //cout << "fScalers["<<index<<"] = "<<scaler<<";"<<endl;
291  }
292  }
293  }
294  //in case events sauves = buffers sauves = 0, we use events lus & buffers ctrl
295  if (eve_sav == 0)
296  indra_db_run->SetEvents(eve_lus);
297  if (buf_sav == 0) {
298  indra_db_run->SetScaler("Buffers", buf_ctrl);
299  indra_db_run->SetSize(buf_ctrl * 16384. / 1024. / 1024.);
300  }
301 
302  temps_mort = 100. * indra_db_run->GetTempsMort();
303  indra_db_run->SetTMpercent(temps_mort);
304 
305  istart -= fTIME_START;
306  iend -= fTIME_START;
307 
308  if (fMakeTree)
309  fTree->Fill();
310 
311  return kTRUE;
312 }

359 TString KVINDRARunSheetReader::GetDateField(TString& line,
360  const Char_t* delim)
361 {
362  //Special case of GetStringField for date-strings with ':' used in between hours, minutes, seconds.
363  //If ':' is also the delimiter of the string, we need to stick the time back together.
364  TObjArray* toks = line.Tokenize(delim);
365 
366  TString date =
367  ((TObjString*) toks->At(1))->GetString().Strip(TString::kLeading,
368  ' ');
369  if (!strcmp(delim, ":")) {
370  for (int i = 2; i < toks->GetEntries(); i++) {
371  date += ":";
372  date += ((TObjString*) toks->At(i))->GetString();
373  }
374  }
375  delete toks;
376  date = date.Strip(TString::kTrailing, ' ');
377  return date;
378 }

388 Int_t KVINDRARunSheetReader::ReadAllRunSheets(Int_t first, Int_t last)
389 {
390  //Read all run sheets in the directory fRunSheetDir from run number 'first' to 'last'
391  //(default values are 1 and 9999).
392  //Returns the number of run sheets actually read.
393 
394  Int_t n_lus = 0; //number of run sheets read
395 
396  if (fMakeTree) CreateTree();
397 
398  for (int run = first; run <= last; run++) {
399 
400  //Check run sheet file exists - if not, skip run
401  if (gSystem->AccessPathName(GetRunSheetFileName(run)))
402  continue;
403 
404  n_lus += (Int_t) ReadRunSheet(run);
405  }
406 
407  if (fMakeTree) fTree->Print();
408  return n_lus;
409 }

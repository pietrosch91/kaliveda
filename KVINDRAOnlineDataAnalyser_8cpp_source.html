152 void KVINDRAOnlineDataAnalyser::PrintControlRate()
153 {
154  TDatime nowtime;
155  int time_passed = nowtime.Convert() - fStart.Convert();
156  if (time_passed < 1.0) {
157  fEventsRefresh *= 2;
158  return;
159  }
160  // check time - we aim for a 5 second refresh
161  fEventsRefresh *= 5. / time_passed;
162  // minimum 5 events
163  fEventsRefresh = TMath::Max(fEventsRefresh, 5);
164  Float_t acq_rate = (1.0 * (events - last_events)) / (1.0 * time_passed);
165  printf("Events read = %d <=> Control rate = %f /sec.\n", events, acq_rate);
166  last_events = events;
167 
168  fStart.Set();
169 
170  if (fDumpEvents) fRunFile->GetFiredDataParameters()->ls();
171 }

179 void KVINDRAOnlineDataAnalyser::ProcessRun()
180 {
181  // Perform treatment of a given run
182 
183  //Open data file
184  fRunFile = (KVRawDataReader*)gDataSet->OpenRunfile(GetDataType().Data(), fRunNumber);
185  if (fRunFile->GetStatus() == KVGRUNetClientGanilReader::kSTATUS_NOHOST) {
186  // cannot connect to ACQ host
187  Error("ProcessRun", "Cannot connect to acquisition host machine. Check KVGRUNetClientGanilReader.AcqHostName");
188  return;
189  }
190  //warning! real number of run may be different from that deduced from file name
191  //we get the real run number from gMultiDetArray and use it to name any new files
192  fRunNumber = gMultiDetArray->GetCurrentRunNumber();
193 
194  fEventNumber = 0; //event number
195 
196  cout << endl << "Reading ONLINE events from RUN# " << fRunNumber << endl;
197 
198  cout << "Starting analysis of data on : ";
199  TDatime now;
200  cout << now.AsString() << endl << endl;
201 
202  preInitRun();
203  //call user's beginning of run
204  InitRun();
205  postInitRun();
206 
207  fDetEv = new KVDetectorEvent;
208 
209  //loop over events in file
210  fGoEventLoop = kTRUE;
211  fDumpEvents = kFALSE;
212  while (fGoEventLoop) {
213  Bool_t gotevent = fRunFile->GetNextEvent();
214 
215  if (gotevent) {
216  fEventNumber++;
217 
218  //reconstruct hit groups
219  KVSeqCollection* fired = fRunFile->GetFiredDataParameters();
220  gMultiDetArray->GetDetectorEvent(fDetEv, fired);
221 
222  preAnalysis();
223  //call user's analysis. stop if returns kFALSE.
224  if (!Analysis()) break;
225  postAnalysis();
226 
227  fDetEv->Clear();
228 
229  if (!((fEventNumber) % 10000)) cout << " ++++ " << fEventNumber << " events read ++++ " << endl;
230  } else {
231  // got no event - why ?
232  Int_t status = fRunFile->GetStatus();
233  switch (status) {
234  case KVGRUNetClientGanilReader::kSTATUS_NOBUFF:
235  Warning("ProcessRun", "Got no buffer from ACQHOST. Sleeping for 2 seconds.");
236  gSystem->Sleep(2000);
237  break;
238 
239  case KVGRUNetClientGanilReader::kSTATUS_NOEVENT:
240  Warning("ProcessRun", "Got no event from ACQHOST. Sleeping for 2 seconds.");
241  gSystem->Sleep(2000);
242  break;
243 
244  case KVGRUNetClientGanilReader::kSTATUS_NOACQ:
245  Warning("ProcessRun", "Acquisition is stopped. Sleeping for 5 seconds.");
246  gSystem->Sleep(5000);
247  break;
248  }
249  }
250  }
251 
252  delete fDetEv;
253 
254  cout << "Ending analysis of run " << fRunNumber << " on : ";
255  TDatime now2;
256  cout << now2.AsString() << endl << endl;
257  cout << endl << "Finished reading " << fEventNumber << " events" << endl << endl;
258 
259  preEndRun();
260  //call user's end of run function
261  EndRun();
262  postEndRun();
263 
264  if (fMessageThread) {
265  fMessageThread->Kill();
266  TThread::Delete(fMessageThread);
267  delete fMessageThread;
268  fMessageThread = 0;
269  }
270 }

332 void KVINDRAOnlineDataAnalyser::HandleCommands(TString& ordre, TSocket* theSocket)
333 {
334  // Gestion des ordres
335  // Returns kTRUE if command was understood & executed
336  // Returns kFALSE if command unknown
337  //
338  // STOP will stop the analysis (calls EndRun() and EndAnalysis())
339  // CLEAR RAZ de tous les spectres
340  //
341  // DUMP=yes start dumping fired data parameters
342  // DUMP=no stop dumping fired data parameters
343  //
344  // SAVE=toto.root will save all histograms in '${ANALYSIS_RESULTS}/toto.root' (does not stop analysis)
345  // SAVE=/home/toto.root will save all histograms in '/home/toto.root' (does not stop analysis)
346  // SAVE will save all histograms in '${ANALYSIS_RESULTS}/[class name]_Run[run number]_[date].root' (does not stop analysis)
347 
348  KVString com(ordre.Data());
349  com.Begin("=");
350  ordre = com.Next();
351  ordre.ToUpper();
352  if (ordre == "STOP") {
353  // arrete l'analyse, en appelant EndRun() et EndAnalysis()
354  fGoEventLoop = kFALSE;
355  theSocket->Send("ok");
356  return;
357  } else if (ordre == "SAVE") {
358  TString tmp_fil = com.Next();
359  TString fil;
360  bool send_name = kFALSE;
361  if (tmp_fil.Contains("/")) fil = tmp_fil;
362  else if (tmp_fil == "") {
363  TDatime save_time;
364  save_time.Set();
365  TString date = save_time.AsString();
366  date.ReplaceAll(" ", "_");
367  fil = Form("${ANALYSIS_RESULTS}/%s_Run%d_%s.root", GetName(), GetRunNumber(), date.Data());
368  send_name = kTRUE;
369  } else fil = Form("${ANALYSIS_RESULTS}/%s", tmp_fil.Data());
370  SaveSpectra(fil.Data());
371  if (send_name) theSocket->Send(fil.Data());
372  else theSocket->Send("ok");
373  return;
374  } else if (ordre == "CLEAR") {
375  ClearAllHistos();
376  theSocket->Send("ok");
377  return;
378  } else if (ordre == "DUMP") {
379  TString tmp_fil = com.Next();
380  tmp_fil.ToLower();
381  if (tmp_fil == "yes")fDumpEvents = kTRUE;
382  else if (tmp_fil == "no")fDumpEvents = kFALSE;
383  else {
384  theSocket->Send("unknown command");
385  return;
386  }
387  theSocket->Send("ok");
388  return;
389  }
390  theSocket->Send("unknown command");
391 }

54 
55  double ag = par[2];
56  double ar = par[3];

58  double e_delta = par[5];

60  double m = float(afit) * u;
61  if (zfit == 2. && afit == 4.) m = 3727.4;
62  if ((zfit == 1.) && (afit == 1.)) m = 938.;
63 
64  double beta = 0.;
65  double fraction = 0.;
66  double dlum = 0.;
67 
68  double beta_delta = sqrt(2.*e_delta / u);
69 
70  beta = sqrt(1. - m * m / ((e + m) * (e + m)));
71 
72  fraction = GetDeltaFraction(beta, beta_delta);
73  if (fraction > 1.) fraction = 1.;
74  if (fraction < 0.) fraction = 0.;
75 
76 
77  double se = sp_e(zfit, afit, e);
78  double sn = sp_n(zfit, afit, e);
79  double deno = 1. + ar * se + an * sn;
80  double deno2 = 1. + (1. - fraction) * ar * se + an * sn;
81  double raps = 1. + sn / se;
82  if (e <= e_delta * afit) {
83  if (se > 0. && raps > 0. && deno > 0.) dlum = (1. + an * sn) / (deno * raps);
84  } else {

86  dlum = (1. + an * sn) * (1. - fraction) / (deno2 * raps) + fraction / raps;
87  }
88  }

171 Double_t KVLightEnergyCsIFull::sp_e(double z, double a, double e)
172 {
173 // if energy E<0.1 keV, return 0
174  if (e < 1.e-4) return 0.;
175 // coefficients for Helium particles (master curves) on Iodium (Z=53), see ref. Ziegler
176  double a1_i = 3.118;
177  double a2_i = 0.6519;
178  double a3_i = 164.9;
179  double a4_i = 1.208;
180  double a5_i = 1.51;
181  double a6_i = 4.614;
182  double a7_i = 0.1043;
183  double a8_i = -0.1315;
184  double a9_i = -0.009175;
185 // coefficients for Helium particles (master curves) on Cesium (Z=55), see ref. Ziegler
186  double a1_cs = 14.4;
187  double a2_cs = 0.3923;
188  double a3_cs = 152.5;
189  double a4_cs = 8.354;
190  double a5_cs = 2.597;
191  double a6_cs = 4.671;
192  double a7_cs = 0.1136;
193  double a8_cs = -0.1298;
194  double a9_cs = -0.009078;
195 // Reference particles for master curves, here alpha particles
196  int zref = 2;
197  int aref = 4;
198  double ei = float(aref) * e / a * 1000.; // energy in keV !
199  double slow_cs = 0.;
200  double shigh_cs = 0.;
201  double se_cs = 0.;
202  double slow_i = 0.;
203  double shigh_i = 0.;
204  double se_i = 0.;
205  double se = 0.;
206  slow_cs = a1_cs * pow(ei, a2_cs);
207  shigh_cs = 1000.*a3_cs * log(1. + 1000.*a4_cs / ei + a5_cs * ei / 1000.) / ei;
208  slow_i = a1_i * pow(ei, a2_i);
209  shigh_i = 1000.*a3_i * log(1. + 1000.*a4_i / ei + a5_i * ei / 1000.) / ei;
210  if (ei <= 10000.) {
211  double fact = 0.997;
212  se_cs = fact * slow_cs * shigh_cs / (slow_cs + shigh_cs);
213  se_i = fact * slow_i * shigh_i / (slow_i + shigh_i);
214  } else {
215  se_cs = exp(a6_cs + a7_cs * log(1000. / ei) + a8_cs * pow(log(1000. / ei), 2.) + a9_cs * pow(log(1000. / ei), 3.));
216  se_i = exp(a6_i + a7_i * log(1000. / ei) + a8_i * pow(log(1000. / ei), 2.) + a9_i * pow(log(1000. / ei), 3.));
217  }
218 // Electronic stopping power for the composite Cs+I = CsI
219  double seref = (127.*se_i + 133.*se_cs) / (127. + 133.);
220 // Get the effective charge
221  se = pow(gamma(z, a, e) * float(z) / (gamma(zref, aref, ei) * float(zref)), 2.) * seref;
222 // Special cases for Z=1 (gamma=1) and Z=2 (reference!)
223  if (z == int(zref)) se = seref;
224  if (z == 1.) se = seref / (pow(float(zref), 2.) * gamma(zref, aref, ei));
225 // Convert the stopping power from eV/(10**15 atoms/cm**2) to MeV/(mg/cm**2)
226  se *= 6.022 / (10.*fAmed);
227 // Convert in MeV/micron, not here !
228  return se;
229  /* if ( e<1.e-4 ) return 0.;
230  Double_t se = fMaterialTable->GetStoppingFunction(z,a)->Eval(e);//-sp_n(z,a,e); // in units of MeV/(g/cm**2)
231  se*=1.e-3; // in units of MeV/(mg/cm**2)
232  return se-sp_n(z,a,e);
233  */
234 }

241 Double_t KVLightEnergyCsIFull::gamma(double z, double a, double e)
242 {
243 // Hubert, Bimbot and Gauvin
244  double g0 = 0.;
245  g0 = (gamma_hbg(z, a, e) + gamma_ziegler(z, a, e)) / 2.;
246 // if E<2.5 meV/n, take effective charge gamma from Ziegler, otherwise take Hubert and Bimbot's formula
247 // A linear adjustment is made to avoid discontinuity between the two prescriptions at E=2.5 MeV/n
248 // For very low energy (E<400 keV/n), take the gamma value from Ziegler's formula at E=400 keV/n
249  if (e < 0.4 * a) g0 = gamma_ziegler(z, a, 0.4 * a);
250  return g0;
251 }

257 Double_t KVLightEnergyCsIFull::gamma_ziegler(double z, double a, double e)
258 {
259  double ei = e;
260  double m = a * u;
261  if (z == 2. && a == 4.) m = 3727.4;
262  double v0 = 7.2983824e-3; // Bohr velocity for electrons
263  double v = sqrt(2.*ei / m);
264  double v1 = 0.886 * (v / v0) * pow(z, -2. / 3.);
265  double v2 = v1 + 0.038 * sin(3.141592654 * v1 / 2.);
266  double xgamma = 1. - (1.034 - 0.17777 * exp(-0.08114 * z)) * exp(-v2);
267  return xgamma;

274 Double_t KVLightEnergyCsIFull::gamma_hbg(double z, double a, double e)
275 {
276  double z2 = fZmed; // Here the atomic number Z for the stopping material
277  double b = 1.658;
278  double c = 0.0517;
279  double d = 1.164 + 0.2319 * exp(-0.0004302 * z2);
280  double x1 = d + b * exp(-c * z);
281  double x2 = 8.144 + 0.09876 * log(z2);
282  double x3 = 0.314 + 0.01072 * log(z2);
283  double x4 = 0.5218 + 0.02521 * log(z2);
284 // Special values for Beryllium
285  if (z == 4) {
286  b = 2.;
287  c = 0.04369;
288  d = 2.045;
289  x1 = d + b * exp(-c * z);
290  x2 = 7.;
291  x3 = 0.2643;
292  x4 = 0.4171;
293  }
294 // Special values for Carbon
295  if (z == 6) {
296  b = 1.91;
297  c = 0.03958;
298  d = 2.584;
299  x1 = d + b * exp(-c * z);
300  x2 = 6.933;
301  x3 = 0.2433;
302  x4 = 0.3969;
303  }
304  double xgamma = 1. - x1 * exp(-x2 * pow(e / a, x3) / pow(z, x4));
305  if (xgamma > 1.) xgamma = 1.;
306  if (xgamma < 0.) xgamma = 0.;
307  return xgamma;

320  double sn = 1.593 * sqrt(ered);

393 {
394  //Calcul de la lumiere totale a partir de Z, A d'une particule et son energie -> copie de KVLightEnergyCsI->CalculeLumiere(...)
395  //
396  // x[0] = energie (MeV)
397  // par[0] = pied
398  // par[1] = a1
399  // par[2] = a2
400  // par[3] = a3
401  // par[4] = a4
402  // par[5] = Z
403  // par[6] = A
404 
405  Double_t energie = x[0];
406 
407  Double_t Z = par[5];
408  Double_t A = par[6];
409  Double_t c1 = par[1];
410  Double_t c2 = Z * Z * A * par[2];
411  Double_t c3 = A * par[3];
412  Double_t c4 = par[4];
413  Double_t T = 8 * A;
414  Double_t c4_new = c4 / (1. + TMath::Exp((c3 - energie) / T));
415  Double_t c0 = c4 / (1. + TMath::Exp(c3 / T));
416 
417  Double_t lumcalc = c1 * energie;
418  if (c2 > 0.0) {
419  Double_t xm = -c1 * c0 * c2 * TMath::Log(c2 / (c2 + c3));
420  lumcalc = lumcalc - c1 * c2 * TMath::Log(1. + energie / c2) + c1 * c2 * c4_new * TMath::Log((energie + c2) / (c3 + c2)) + xm;
421  }
422 
423  return par[0] + lumcalc;
424 }

598 Double_t KVLightEnergyCsIFull::Compute(Double_t light) const

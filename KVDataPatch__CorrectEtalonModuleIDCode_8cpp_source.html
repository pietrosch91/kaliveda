89 void KVDataPatch_CorrectEtalonModuleIDCode::ApplyToParticle(KVNucleus* n)
90 {
91  // Patch is applied to all identified particles stopping in the CsI member of
92  // an etalon module which have general ID code 5.
93  // If the CsI identification is available and was succesful, then either
94  // (1) for gammas we set the correct ID code and set the energy = 0
95  // (2) for charged particles we correct the ID code and the energy
96  // which is probably overestimated (due to taking into account the
97  // Si75/SiLi in the calibration when in fact they were not hit)
98 
99  KVINDRAReconNuc* N = (KVINDRAReconNuc*)n;
100  if (N->IsIdentified() && N->GetRingNumber() > 9 && N->StoppedInCsI() && (N->GetSiLi() || N->GetSi75())
101  && N->GetCodes().TestIDCode(kIDCode5)) {
102  KVIdentificationResult* idr = N->GetIdentificationResult(1);
103  if (idr->IDattempted && idr->IsType("CSI_R_L") && idr->IDOK) {
104  if (idr->IDcode == kIDCode0) {
105  // gamma
106  // just set general IDcode gamma and set energy = 0
107  N->SetIdentification(idr);
108  N->SetEnergy(0.);
109  N->GetParameters()->SetValue("KVDataPatch_CorrectEtalonModuleIDCode", "correction applied");
110  } else {
111  // charged particle
112  // needs re-identifying and re-calibrating
113  KVIdentificationResult ID;
114  Bool_t ok = N->CoherencyChIoCsI(ID);
115  if (ok) {
116  N->SetIdentification(&ID);
117  N->Calibrate();
118  N->GetParameters()->SetValue("KVDataPatch_CorrectEtalonModuleIDCode", "correction applied");
119  } else {
120  Warning("ApplyToParticle", "Rustine failed for the following particle:");
121  N->GetParameters()->SetValue("KVDataPatch_CorrectEtalonModuleIDCode", "correction failed");
122  N->Print();
123  }
124  }
125  // update particle's 'OK' status
126  N->SetIsOK(fEvent->CheckCodes(N->GetCodes()));
127  }
128  }
129 }

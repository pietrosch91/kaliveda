78 inline Bool_t KVRunListLine::HasFieldValue(const Char_t* fname) const
79 {
80  //we check that the field in question exists and has been filled in this line
81  //if the field index is greater than the actual number of fields in the line, we assume the field was not filled (HasFieldValue=kFALSE)
82  //fields which are present but empty or filled with whitespace are not considered to have a field value
83  Int_t index = GetFieldIndex(fname);
84  if (!(index > -1 && index < fFields->GetEntries()))
85  return kFALSE;
86  KVString tmp(((TObjString*)(*fFields)[index])->String());
87  return !tmp.IsWhitespace();
88 }

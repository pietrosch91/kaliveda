120 Bool_t KVIDChIoCorrCsI::Identify(KVIdentificationResult* idr, Double_t x, Double_t y)
121 {
122  idr->SetIDType(GetType());
123  idr->IDattempted = kTRUE;
124 
125  Double_t chIoCorr = (y < 0. ? GetIDMapY() : y);
126  Double_t csiLight = (x < 0. ? GetIDMapX() : x);
127 
128  if (fGrid->IsIdentifiable(csiLight, chIoCorr))
129  fGrid->Identify(csiLight, chIoCorr, idr);
130 
131  if (fGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
132  // only if the final quality code is kICODE8 do we consider that it is
133  // worthwhile looking elsewhere. In all other cases, the particle has been
134  // "identified", even if we still don't know its Z and/or A (in this case
135  // we consider that we have established that they are unknowable).
136  idr->IDcode = kIDCode15;
137  return kFALSE;
138  }
139 
140  if (fGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
141  // if the final quality code is kICODE7 (above last line in grid) then the estimated
142  // Z is only a minimum value (Zmin)
143  idr->IDcode = kIDCode5;
144  return kTRUE;
145  }
146 
147  if (fGrid->GetQualityCode() > KVIDZAGrid::kICODE3 && fGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
148  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
149  // corresponds to a point which is inbetween the lines, i.e. noise
150  idr->IDcode = kIDCode10;
151  return kTRUE;
152  }
153  if (fGrid->GetQualityCode() == KVIDGChIoSi_e494s::k_BelowSeuilChIo) {
154 
155  idr->IDcode = kIDCode15;
156  return kTRUE;
157  }
158 
159  // set general ID code ChIo-CsI
160  idr->IDcode = kIDCode4;
161 
162  return kTRUE;
163 }

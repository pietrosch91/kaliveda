517 Bool_t KVDetector::Fired(Option_t* opt)
518 {
519  // Returns kTRUE if detector was hit (fired) in an event
520  //
521  // The actual meaning of hit/fired depends on the context and the option string opt.
522  //
523  // If the detector is in "simulation mode", i.e. if SetSimMode(kTRUE) has been called,
524  // this method returns kTRUE if the calculated energy loss in the active layer is > 0.
525  //
526  // In "experimental mode" (i.e. IsSimMode() returns kFALSE), depending on the option:
527  //
528  //opt="any" (default):
529  //Returns true if ANY* of the working acquisition parameters associated with the detector were fired in an event
530  //opt="all" :
531  //Returns true if ALL* of the working acquisition parameters associated with the detector were fired in an event
532  //opt="Pany" :
533  //Returns true if ANY* of the working acquisition parameters associated with the detector were fired in an event
534  //and have a value greater than their pedestal value
535  //opt="Pall" :
536  //Returns true if ALL* of the working acquisition parameters associated with the detector were fired in an event
537  //and have a value greater than their pedestal value
538  //
539  // *the actual parameters taken into account can be fine tuned using environment variables such as
540  // [classname].Fired.ACQParameterList.[type]: PG,GG,T
541  // where [classname]=KVDetector by default, or the name of some class
542  // derived from KVDetector which calls the method KVDetector::SetKVDetectorFiredACQParameterListFormatString()
543  // in its constructor.
544  // See KVDetector::SetFiredBitmask() for more details.
545 
546  if (!IsDetecting()) return kFALSE; //detector not working, no answer at all
547  if (IsSimMode()) return (GetActiveLayer()->GetEnergyLoss() > 0.); // simulation mode: detector fired if energy lost in active layer
548 
549  if (opt[0] == 'P') return FiredP(opt + 1);
550 
551  Binary8_t event; // bitmask for event
552  TIter next(fACQParams);
553  KVACQParam* par;
554  Int_t id = 0;
555  while ((par = (KVACQParam*)next())) {
556  if (par->Fired()) event.SetBit(id);
557  else event.ResetBit(id);
558  id++;
559  }
560  Binary8_t ok = fFiredMask & event;
561  // "all" considered parameters fired if ok == mask
562  // "any" considered parameters fired if ok != 0
563  if (!strcmp(opt, "all")) return (ok == fFiredMask);
564  return (ok != "0");
565 }

315 inline Double_t KVVAMOSReconNuc::CalculateEnergy(Int_t Z, Int_t A, Int_t Q, Double_t Brho)
316 {
317  // Calculates kinetic energy (E) in MeV.
318  // Begin_Latex
319  // E = #sqrt{M^{2}+100 c^{2} B#rho^{2} Q^{2}}- M
320  // End_Latex
321  // where
322  // M : mass deduced from Z and A, in MeV/c^2
323  // c : speed of light in cm/ns
324  // Brho : magnetic rigidity in T.m
325  // Q : charge state
326  //
327  // If Z is set lower or equal 0 then the mass excess is null to calculate
328  // M (i.e. M = A*u)
329 
330  assert(Z > 0);
331  assert(A > 0);
332  assert(Q > 0);
333  assert(Brho > 0.);
334 
335  Double_t M = 0.;
336  if (Z > 0) {
337  static KVNucleus nuc;
338  nuc.SetZandA(Z, A);
339  M = nuc.GetMass();
340  } else M = A * u();
341  return TMath::Sqrt(M * M + 100.*TMath::Power(C() * Brho * Q, 2)) - M;
342 }

345 inline Double_t KVVAMOSReconNuc::CalculateEnergy(Int_t Z, Int_t A, Double_t beta)
346 {
347  // Calculates kinetic energy in MeV.
348  // Begin_Latex
349  // E = (#gamma-1)M
350  // End_Latex
351  // where
352  // M : mass deduced from Z and A in, MeV/c^2
353  // gamma : Lorentz factor calculated from beta (velocity/c)
354  //
355  // If Z is set lower or equal 0 then the mass excess is null to calculate
356  // M (i.e. M = A*u)
357 
358  assert(Z > 0);
359  assert(A > 0);
360  assert((beta > 0.) && (beta < 1.));
361 
362  if (beta <= 0) return 0.;
363 
364  Double_t M = 0.;
365  if (Z > 0) {
366  static KVNucleus nuc;
367  nuc.SetZandA(Z, A);
368  M = nuc.GetMass();
369  } else M = A * u();
370  Double_t gamma = 1.0 / TMath::Sqrt(1 - beta * beta);
371  return (gamma - 1) * M ;
372 }

375 inline Double_t KVVAMOSReconNuc::CalculateEnergy(Int_t Q, Double_t Brho, Double_t beta)
376 {
377  // Calculates kinetic energy in MeV from Q, Brho and beta.
378  // First the M/Q is calculated from Brho and beta.
379  // Then the energy is given by
380  // Begin_Latex
381  // E = (#gamma-1)*(M/Q)*Q
382  // End_Latex
383  // where
384  // gamma : Lorentz factor calculated from beta (velocity/c)
385 
386  Double_t M = CalculateRealAoverQ(Brho, beta) * Q;
387  if (M <= 0) return 0.;
388  Double_t gamma = 1.0 / TMath::Sqrt(1 - beta * beta);
389  return (gamma - 1) * M ;
390 }

395 {

426 inline Double_t KVVAMOSReconNuc::CalculateRealAoverQ(Double_t Brho, Double_t beta)
427 {
428  // Calculates the ratio between the mass (MeV/c^2) and the charge state Q
429  // calculated from the magnetic rigitidy Brho and the relativistic beta
430  // (i.e. velocity/c).
431  // The returned value is real.
432  // Begin_Latex
433  // #frac{M}{Q} = 10 C #frac{B_{#rho}}{ #gamma #beta}
434  // End_Latex
435  // where
436  // Brho : magnetic rigidity in T.m
437  // u : atomic mass unit in MeV/c^2
438  // C : speed of light in vacuum in cm/ns
439  // gamma : Lorentz factor calculated from beta
440 
441  assert((beta > 0.) && (beta < 1.));
442  assert(Brho > 0.);
443 
444  if (beta <= 0) return 0.;
445 
446  Double_t gamma = 1.0 / TMath::Sqrt(1 - beta * beta);
447  Double_t tmp = beta * gamma;
448  return Brho * C() * 10. / tmp;
449 }

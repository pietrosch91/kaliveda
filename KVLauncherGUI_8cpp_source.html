111 TGTextButton* KVLauncherGUI::AddButton(const char* name, const char* method, TObject* obj)
112 {
113  // Add a button to the launcher with given "name".
114  // When clicked, this button will call "method" of whatever object obj is pointing to.
115  // - "method" is the name of an argument-less method, without parentheses:
116  // i.e. to call method DummyMethod() when button is clicked, give method="DummyMethod"
117  // If obj=0 (default), 'this' will be used.
118  // If "method" is not a public interface method of the class of 'obj' an error
119  // will be printed and no button added.
120 
121  if (!obj) obj = this;
122  if (!obj->IsA()->GetListOfAllPublicMethods()->FindObject(method)) {
123  Error("AddButton", "%s is not a public method of %s", method, obj->ClassName());
124  return NULL;
125  }
126 
127  ULong_t ucolor;
128  gClient->GetColorByName("#66ccff", ucolor);
129 
130  TGTextButton* fButton = new TGTextButton(this, name);
131  fButton->Connect("Clicked()", obj->ClassName(), obj, Form("%s()", method));
132  fButton->SetName(name);
133  fButton->SetMargins(5, 5, 5, 5);
134  fButton->ChangeBackground(ucolor);
135 
136  AddFrame(fButton, new TGLayoutHints(kLHintsRight | kLHintsExpandX, 10, 10, 5, 5));
137  fButtonList.AddLast(fButton);
138 
139  return fButton;
140 }

138 void KVQuadMoment::Fill(KVNucleus* c)
139 {
140  //Add the particle's contribution to the momentum tensor.
141 
142  TVector3 P = c->GetFrame(fFrame.Data(), kFALSE)->GetMomentum();
143  Double_t P2 = P.Mag2();
144  for (int i = 0; i < 3; i++) {
145  for (int j = i; j < 3; j++) {
146  if (i == j) matrix[i][j] += 3.*P(i) * P(j) - P2;
147  else {
148  Double_t a = 3.*P(i) * P(j);
149  matrix[i][j] += a;
150  matrix[j][i] += a;
151  }
152  }
153  }
154 }

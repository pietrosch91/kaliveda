101 Bool_t KVDBKey::LinkTo(KVDBRecord* rec, Bool_t linkback)
102 {
103 // This function adds a record to the list of cross-references
104 // if fIsUnique is kTRUE we check if there is already an object with the same name:
105 // if so we return kFALSE otherwise the record is added and kTRUE is returned.
106 // if fSingle is kTRUE the list can handle only one object:
107 // if the list is empty the record is added and kTRUE is returned, otherwise kFALSE.
108 
109  // check if more than one object is allowed
110  if (fSingle && (GetLinks()->GetSize() > 0) && (GetLinks()->First())) {
111  Warning("LinkTo(KVDBRecord*)",
112  "%s : Only one cross-reference allowed.\nCurrent cross-reference is:%s\nCan't add new record \"%s\" in list",
113  GetName(), GetLinks()->First()->GetName(), rec->GetName());
114  return kFALSE;
115  }
116  // check if identical names are allowed
117  if (fIsUnique) {
118  if (fLinks->FindObject(rec->GetName())) {
119  Warning("LinkTo(KVDBRecord*)",
120  "%s : Cross-references must have different names.\nA record named %s already exists.\nCan't add new record \"%s\" in list",
121  GetName(), rec->GetName(), rec->GetName());
122  return kFALSE;
123  }
124  }
125  AddLink(rec);
126 
127  if (linkback) {
128  //if linkback is kTRUE then a mirror link to this record will be made in record "rec"
129  //if a key with the same name as the table containing the current record is found
130  //in the record to link , then this key is linked back to the current record.
131  //if no key is found, one will be created, and then the linkback performed.
132 
133  //check parent is set
134  if (!GetParent()) {
135  Error("LinkTo", "Parent table not set for key %s", GetName());
136  return kFALSE;
137  }
138  const Char_t* table_name = GetParent()->GetTable()->GetName();
139 
140  KVDBKey* key = rec->GetKey(table_name);
141  if (!key)
142  key = rec->AddKey(table_name, GetParent()->GetName());
143  key->LinkTo(GetParent(), kFALSE); //linkback=kFALSE otherwise infinite circular linkage results !!!
144  }
145  return kTRUE;
146 }

155 void KVDBKey::Unlink(KVDBRecord* rec, Bool_t linkback)
156 {
157  //This function removes a record from the list of cross-references
158  //if linkback is kTRUE then the reference in record "rec" is also removed.
159 
160  RemoveLink(rec);
161 
162  if (linkback) {
163 
164  //check parent is set
165  if (!GetParent()) {
166  Error("Unlink", "Parent not set for key %s", GetName());
167  return;
168  }
169  const Char_t* table_name = GetParent()->GetTable()->GetName();
170 
171  KVDBKey* key = rec->GetKey(table_name);
172  if (key) key->Unlink(GetParent(), kFALSE); //linkback=kFALSE otherwise infinite circular linkage results !!!
173  }
174 }

62 void KVIDSiCsI_e613::Initialize()
63 {
64  // Initialize telescope for current run.
65  // If there is at least 1 grid, we set fCanIdentify = kTRUE
66  // "Natural" line widths are calculated for KVIDZAGrids.
67 
68  fSi = (KVSilicon*) GetDetector(1);
69  fSiPGPedestal = fSi->GetPedestal("PG");
70  fSiGGPedestal = fSi->GetPedestal("GG");
71  fCsI = (KVCsI*) GetDetector(2);
72  fCsIRPedestal = fCsI->GetPedestal("R");
73  fCsILPedestal = fCsI->GetPedestal("L");
74 
75  fGGgrid = fPGgrid = 0;
76 
77  KVIDGraph* grid = 0;
78  TIter next_grid(GetListOfIDGrids());
79  while ((grid = (KVIDGraph*)next_grid())) {
80  if (!strcmp(grid->GetVarY(), "SI_GG")) fGGgrid = (KVIDZAGrid*)grid;
81  else if (!strcmp(grid->GetVarY(), "SI_PG")) fPGgrid = (KVIDZAGrid*)grid;
82  }
83 
84  if (fGGgrid) {
85  SetBit(kReadyForID);
86  fGGgrid->Initialize();
87  if (fPGgrid) fPGgrid->Initialize();
88  fPIEDESTAL = (KVIDCutLine*)fGGgrid->GetCut("PIEDESTAL");
89  } else ResetBit(kReadyForID);

119 Double_t KVIDSiCsI_e613::GetIDMapY(Option_t* opt)

138 Bool_t KVIDSiCsI_e613::Identify(KVIdentificationResult* idr, Double_t x, Double_t y)
139 {
140  // Particle identification and code setting using identification grids.
141  // perform identification in Si(GG) - CsI(H) map
142  // Sets idr->deltaEpedestal according to position in GG map
143 
144  idr->SetIDType(GetType());
145  idr->IDattempted = kTRUE;
146 
147  Double_t sigg = (y < 0. ? GetIDMapY("GG") : y);
148  Double_t lumtot = (x < 0. ? GetIDMapX() : x);
149 
150  KVIDZAGrid* TheGrid = 0;
151 
152  fGGgrid->Identify(lumtot, sigg, idr);
153  // check if silicon-GG is in pedestal region (possible neutron)
154  if (fPIEDESTAL) {
155  if (fPIEDESTAL->TestPoint(lumtot, sigg)) idr->deltaEpedestal = KVIdentificationResult::deltaEpedestal_NO;
156  else idr->deltaEpedestal = KVIdentificationResult::deltaEpedestal_YES;
157  } else {
158  idr->deltaEpedestal = KVIdentificationResult::deltaEpedestal_UNKNOWN;
159  }
160 
161  TheGrid = (KVIDZAGrid*) fGGgrid;
162 
163  //Info("Identify","OK=%d Z=%d Zmax=%d GG=%d\n",idr->IDOK,idr->Z,TheGrid->GetZmax(),TMath::Nint(sigg));
164  if (idr->IDOK && idr->Z == TheGrid->GetZmax()) {
165  //Gestion des saturations GG
166  //on teste l identification PG
167  if (fPGgrid) {
168  Double_t sipg = (y < 0. ? GetIDMapY("PG") : y);
169  fPGgrid->Identify(lumtot, sipg, idr);
170  //on garde l identification PG si celle ci renvoie un code
171  //de 0 a 4 ou 7
172  if (idr->Zident) {
173  //Info("Identify","On passe de %d a %d",Zgg,Zpg);
174  TheGrid = (KVIDZAGrid*) fPGgrid;
175  }
176  }
177  }
178 
179  if (TheGrid == fGGgrid) {
180  if (fGGgrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fPGgrid) { //we have to try PG grid (if there is one)
181  Double_t sipg = (y < 0. ? GetIDMapY("PG") : y);
182  fPGgrid->Identify(lumtot, sipg, idr);
183  TheGrid = (KVIDZAGrid*) fPGgrid;
184  }
185  }
186  if (TheGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
187  // only if the final quality code is kICODE8 do we consider that it is
188  // worthwhile looking elsewhere. In all other cases, the particle has been
189  // "identified", even if we still don't know its Z and/or A (in this case
190  // we consider that we have established that they are unknowable).
191 
192  return kFALSE;
193  }
194 
195  if (TheGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
196  // if the final quality code is kICODE7 (above last line in grid) then the estimated
197  // Z is only a minimum value (Zmin)
198  idr->IDcode = kIDCode5;
199 
200  return kTRUE;
201  }
202 
203  if (TheGrid->GetQualityCode() > KVIDZAGrid::kICODE3 && TheGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
204  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
205  // corresponds to a point which is inbetween the lines, i.e. noise
206 
207  idr->IDcode = kIDCode10;
208  return kTRUE;
209  }
210 
211  // set general ID code Si-CsI
212  idr->IDcode = kIDCode3;
213  return kTRUE;
214 }

175 void KVFunctionCal::SetExpFormula(const Char_t* formula, Double_t xmin, Double_t xmax)
176 //------------------------------
177 
178 
185 {
186  // Change the formula of the calibration function.
187  //
188  // If the [xmin,xmax] range is not given and if a previous calibration
189  // function was already set then the same range is taken.
190  // The default range is [O,4096].
191 
192  Double_t min = 0., max = 4096.;
193  if (xmin != xmax) {
194  min = xmin;
195  max = xmax;
196  }
197  if (fcalibfunction) {
198  if (xmin == xmax) fcalibfunction->GetRange(min, max);
199  delete fcalibfunction;
200  }
201 
202  fcalibfunction = new TF1(GetName(), formula, min, max);
203  SetNumberParams(fcalibfunction->GetNpar());
204 }

208 Double_t KVFunctionCal::Compute(Option_t* opt) const
209 //------------------------------
210 
216 
217 {
218  // Give the calibration result for the ACQ parameter corresponding to
219  // this calibrator.
220  // If opt = "P" then the currently set pedestal is removed from the
221  // value of the acquisition parameter before the calculation.
222  if (!fACQpar) {
223  Error("Compute", "No ACQ parameter corresponds to the calibrator %s", GetName());
224  return -666;
225  }
226  Double_t ped = (opt[0] ? fACQpar->GetPedestal() : 0.);
227  return fcalibfunction->Eval(fACQpar->GetData() - ped);

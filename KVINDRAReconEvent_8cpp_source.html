164 void KVINDRAReconEvent::Streamer(TBuffer& R__b)
165 {
166  // Stream an object of class KVINDRAReconEvent.
167  // If acceptable ID/E codes have been set with methods AcceptIDCodes()/
168  // AcceptECodes(), we loop over the newly-read particles in order to set
169  // their IsOK() status according to these choices.
170  // If not, it will have already been set according to the default code
171  // selection (defined by [DataSet].INDRA.ReconstructedNuclei.AcceptID/ECodes)
172  // in KVReconstructedNucleus::Streamer
173 
174  if (R__b.IsReading()) {
175  R__b.ReadClassBuffer(KVINDRAReconEvent::Class(), this);
176  if (fCodeMask && !fCodeMask->IsNull()) {
177  KVINDRAReconNuc* par;
178  while ((par = GetNextParticle())) {
179  par->SetIsOK(CheckCodes(par->GetCodes()));
180  }
181  }
182  } else {
183  R__b.WriteClassBuffer(KVINDRAReconEvent::Class(), this);
184  }
185 }

265 void KVINDRAReconEvent::AcceptIDCodes(UShort_t code)
266 {
267  //Define the identification codes that you want to include in your analysis.
268  //Example: to keep only (VEDA) codes 2 and 3, use
269  // event.AcceptIDCodes( kIDCode2 | kIDCode3 );
270  //Particles which have the correct ID codes will have IsOK() = kTRUE
271  //i.e. if( particle.IsOK() ) { // ... analysis ... }
272  //although it is easier to loop over all 'correct' particles using:
273  // while ( particle = GetNextParticle("ok") ){ // ... analysis ... }
274  //
275  //To remove any previously defined acceptable identification codes,
276  // use AcceptIDCodes(0)
277  //
278  //N.B. : this method is preferable to using directly the KVINDRACodeMask pointer
279  //as the 'IsOK' status of all particles of the current event are automatically updated
280  //when the code mask is changed using this method.
281  GetCodeMask()->SetIDMask(code);
282  KVINDRAReconNuc* par = 0;
283  while ((par = GetNextParticle())) {
284  if (CheckCodes(par->GetCodes()))
285  par->SetIsOK();
286  else
287  par->SetIsOK(kFALSE);
288  }

355 void KVINDRAReconEvent::ChangeFragmentMasses(UChar_t mass_formula)
356 {
357  //Changes the mass formula used to calculate A from Z for all nuclei in event
358  //For the values of mass_formula, see KVNucleus::GetAFromZ
359  //
360  //The fragment energy is modified in proportion to its mass, this is due to the
361  //contribution from the CsI light-energy calibration:
362  //
363  // E -> E + E_CsI*( newA/oldA - 1 )
364  //
365  //From an original lunch by Remi Bougault.
366  //
367  //Only particles with 'acceptable' ID & E codes stopping in (or passing through)
368  //a CsI detector are affected; particles whose mass was measured
369  //(i.e. having KVReconstructedNucleus::IsAMeasured()==kTRUE)
370  //are not affected by the change of mass formula.
371 
372  ResetGetNextParticle();
373  KVINDRAReconNuc* frag;
374  while ((frag = (KVINDRAReconNuc*)GetNextParticle("ok"))) {
375 
376  if (!frag->IsAMeasured()) {
377 
378  Float_t oldA = (Float_t)frag->GetA();
379  frag->SetMassFormula(mass_formula);
380 
381  if (frag->GetCsI()) {
382  Float_t oldECsI = frag->GetEnergyCsI();
383  Float_t oldE = frag->GetEnergy();
384  Float_t newECsI = oldECsI * ((Float_t)frag->GetA() / oldA);
385  frag->GetCsI()->SetEnergy(newECsI);
386  frag->SetEnergy(oldE - oldECsI + newECsI);
387  }
388  }
389  }

407 void KVINDRAReconEvent::IdentifyEvent()
408 {
409  // Performs event identification (see KVReconstructedEvent::IdentifyEvent), and then
410  // particles stopping in first member of a telescope (GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) are
411  // labelled with VEDA ID code kIDCode5 (Zmin)
412  //
413  // When CsI identification gives a gamma, we unset the 'analysed' state of all detectors
414  // in front of the CsI and reanalyse the group in order to reconstruct and identify charged particles
415  // stopping in them.
416  //
417  // Unidentified particles receive the general ID code for non-identified particles (kIDCode14)
418 
419  KVReconstructedEvent::IdentifyEvent();
420  KVINDRAReconNuc* d = 0;
421  int mult = GetMult();
422  KVUniqueNameList gammaGroups;//list of groups with gammas identified in CsI
423  ResetGetNextParticle();
424 
425  while ((d = GetNextParticle())) {
426  if (d->IsIdentified() && d->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
427  d->SetIDCode(kIDCode5); // Zmin
428  } else if (d->IsIdentified() && d->GetCodes().TestIDCode(kIDCode0)) {
429  // gamma identified in CsI
430  // reset analysed state of all detectors in front of CsI
431  if (d->GetCsI()) {
432  if (d->GetCsI()->GetAlignedDetectors()) {
433  TIter next(d->GetCsI()->GetAlignedDetectors());
434  KVDetector* det = (KVDetector*)next(); //first detector = CsI
435  while ((det = (KVDetector*)next())) det->SetAnalysed(kFALSE);
436  gammaGroups.Add(d->GetGroup());
437  } else {
438  Error("IdentifyEvent", "particule id gamma, no aligned detectors???");
439  d->Print();
440  }
441  } else {
442  Error("IdentifyEvent", "particule identified as gamma, has no CsI!!");
443  d->Print();
444  }
445  }
446  }
447 
448  // perform secondary reconstruction in groups with detected gammas
449  int ngamG = gammaGroups.GetEntries();
450  if (ngamG) {
451  for (int i = 0; i < ngamG; i++) {
452  gIndra->AnalyseGroupAndReconstructEvent(this, (KVGroup*)gammaGroups.At(i));
453  }
454  }
455  if (GetMult() > mult) {
456  /*Info("IdentifyEvent", "Event#%d: Secondary reconstruction (gammas) -> %d new particles",
457  GetNumber(), GetMult()-mult);*/
458 
459  // identify new particles generated in secondary reconstruction
460  KVReconstructedEvent::IdentifyEvent();
461  ResetGetNextParticle();
462 
463  while ((d = GetNextParticle())) {
464  if (d->IsIdentified() && d->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
465  d->SetIDCode(kIDCode5); // Zmin
466  } else if (!d->IsIdentified()) {
467  d->SetIDCode(kIDCode14);
468  }
469  }
470  /*
471  for(int i=mult+1; i<=GetMult(); i++){
472  d = GetParticle(i);
473  if(d->IsIdentified())
474  printf("\t%2d: Ring %2d Module %2d Z=%2d A=%3d code=%d\n",i,d->GetRingNumber(),
475  d->GetModuleNumber(),d->GetZ(),d->GetA(),d->GetCodes().GetVedaIDCode());
476  else
477  printf("\t%2d: Ring %2d Module %2d UNIDENTIFIED status=%d\n", i,d->GetRingNumber(),
478  d->GetModuleNumber(), d->GetStatus());
479  }
480  */
481  }
482 }

494 void KVINDRAReconEvent::SecondaryIdentCalib()
495 {
496  // Perform identifications and calibrations of particles not included
497  // in first round (methods IdentifyEvent() and CalibrateEvent()).
498  //
499  // Here we treat particles with GetStatus()==KVReconstructedNucleus::kStatusOKafterSub
500  // after subtracting the energy losses of all previously calibrated particles in group from the
501  // measured energy losses in the detectors they crossed.
502 
503  if (!fHitGroups)
504  fHitGroups = new KVUniqueNameList;
505  else
506  fHitGroups->Clear();
507  // build list of hit groups
508  KVINDRAReconNuc* d;
509  while ((d = GetNextParticle())) fHitGroups->Add(d->GetGroup());
510 
511  //loop over hit groups
512  TIter next_grp(fHitGroups);
513  KVGroup* grp;
514  while ((grp = (KVGroup*)next_grp())) {
515  SecondaryAnalyseGroup(grp);
516  }
517 
518  // set "unidentified" code for any remaining unidentified particle
519  ResetGetNextParticle();
520  while ((d = GetNextParticle())) {
521  if (!d->IsIdentified()) {
522  d->SetIDCode(kIDCode14);
523  }
524  }

539 void KVINDRAReconEvent::SecondaryAnalyseGroup(KVGroup* grp)
540 {
541  // Perform identifications and calibrations of particles not included
542  // in first round (methods IdentifyEvent() and CalibrateEvent()).
543  //
544  // Here we treat particles with GetStatus()==KVReconstructedNucleus::kStatusOKafterSub
545  // after subtracting the energy losses of all previously calibrated particles in group from the
546  // measured energy losses in the detectors they crossed.
547 
548  // loop over al identified & calibrated particles in group and subtract calculated
549  // energy losses from all detectors
550  KVINDRAReconNuc* nuc;
551  TList sixparts;
552  TIter parts(grp->GetParticles());
553  while ((nuc = (KVINDRAReconNuc*)parts())) {
554  if (nuc->IsIdentified() && nuc->IsCalibrated()) {
555  nuc->SubtractEnergyFromAllDetectors();
556  // reconstruct particles from pile-up in silicon detectors revealed by coherency CsIR/L - SiCsI
557  if (nuc->IsSiPileup() && nuc->GetSi()->GetEnergy() > 0.1) {
558  KVINDRAReconNuc* SIX = AddParticle();
559  SIX->Reconstruct(nuc->GetSi());
560  sixparts.Add(SIX);
561  }
562  // reconstruct particles from pile-up in si75 detectors revealed by coherency
563  if (nuc->IsSi75Pileup()) {
564  KVINDRAReconNuc* SIX = AddParticle();
565  SIX->Reconstruct(nuc->GetSi75());
566  sixparts.Add(SIX);
567  }
568  // reconstruct particles from pile-up in sili detectors revealed by coherency
569  if (nuc->IsSiLiPileup()) {
570  KVINDRAReconNuc* SIX = AddParticle();
571  SIX->Reconstruct(nuc->GetSiLi());
572  sixparts.Add(SIX);
573  }
574 
575  // reconstruct particles from pile-up in ChIo detectors revealed by coherency CsIR/L - ChIoCsI
576  if (nuc->IsChIoPileup() && nuc->GetChIo()->GetEnergy() > 1.0) {
577  KVINDRAReconNuc* SIX = AddParticle();
578  SIX->Reconstruct(nuc->GetChIo());
579  sixparts.Add(SIX);
580  }
581  }
582  }
583  // reanalyse group
584  KVReconstructedNucleus::AnalyseParticlesInGroup(grp);
585 
586  Int_t nident = 0; //number of particles identified in each step
587  if (sixparts.GetEntries()) { // identify any particles added by coherency CsIR/L - SiCsI
588  KVINDRAReconNuc* SIX;
589  TIter nextsix(&sixparts);
590  while ((SIX = (KVINDRAReconNuc*)nextsix())) {
591  if (SIX->GetStatus() == KVReconstructedNucleus::kStatusOK) {
592  SIX->Identify();
593  if (SIX->IsIdentified()) {
594  nident++;
595  if (SIX->GetCodes().TestIDCode(kIDCode5)) SIX->SetIDCode(kIDCode7);
596  else SIX->SetIDCode(kIDCode6);
597  SIX->Calibrate();
598  if (SIX->IsCalibrated()) SIX->SubtractEnergyFromAllDetectors();
599  } else {
600  // failure of ChIo-Si identification: particle stopped in ChIo ?
601  // estimation of Z (minimum) from energy loss (if detector is calibrated)
602  UInt_t zmin = ((KVDetector*)SIX->GetDetectorList()->Last())->FindZmin(-1., SIX->GetMassFormula());
603  if (zmin) {
604  SIX->SetZ(zmin);
605  SIX->SetIsIdentified();
606  SIX->SetIDCode(kIDCode7);
607  // "Identifying" telescope is taken from list of ID telescopes
608  // to which stopping detector belongs
609  SIX->SetIdentifyingTelescope((KVIDTelescope*)SIX->GetStoppingDetector()->GetIDTelescopes()->Last());
610  SIX->Calibrate();
611  }
612  }
613  }
614  }
615  }
616  if (nident) { // newly-identified particles may change status of others in group
617  // reanalyse group
618  KVReconstructedNucleus::AnalyseParticlesInGroup(grp);
619  nident = 0;
620  }
621 
622  TIter parts2(grp->GetParticles()); // list may have changed if we have added particles
623  // identify & calibrate any remaining particles with status=KVReconstructedNucleus::kStatusOK
624  while ((nuc = (KVINDRAReconNuc*)parts2())) {
625  if (!nuc->IsIdentified() && nuc->GetStatus() == KVReconstructedNucleus::kStatusOK) {
626  nuc->ResetNSegDet();
627  nuc->Identify();
628  if (nuc->IsIdentified()) {
629  nident++;
630  nuc->Calibrate();
631  if (nuc->IsCalibrated()) nuc->SubtractEnergyFromAllDetectors();
632  }
633  }
634  }
635  if (nident) { // newly-identified particles may change status of others in group
636  // reanalyse group
637  KVReconstructedNucleus::AnalyseParticlesInGroup(grp);
638  nident = 0;
639  }
640 
641  // any kStatusOKafterShare particles ?
642  TList shareChIo;
643  parts2.Reset();
644  while ((nuc = (KVINDRAReconNuc*)parts2())) {
645  if (!nuc->IsIdentified() && nuc->GetStatus() == KVReconstructedNucleus::kStatusOKafterShare) {
646  shareChIo.Add(nuc);
647  }
648  }
649  Int_t nshares = shareChIo.GetEntries();
650  if (nshares) {
651  KVChIo* theChIo = ((KVINDRAReconNuc*)shareChIo.At(0))->GetChIo();
652  if (theChIo && nshares > 1) {
653  // divide chio energy equally
654  Double_t Eshare = theChIo->GetEnergyLoss() / nshares;
655  theChIo->SetEnergyLoss(Eshare);
656  // modify PG and GG of ChIo according to new energy loss
657  Double_t volts = theChIo->GetVoltsFromEnergy(Eshare);
658  Double_t GG = theChIo->GetCanalGGFromVolts(volts);
659  Double_t PG = theChIo->GetCanalPGFromVolts(volts);
660  theChIo->GetACQParam("PG")->SetData(TMath::Min(4095, (Int_t)PG));
661  theChIo->GetACQParam("GG")->SetData(TMath::Min(4095, (Int_t)GG));
662  }
663  // now try to identify
664  TIter nextSh(&shareChIo);
665  while ((nuc = (KVINDRAReconNuc*)nextSh())) {
666  nuc->SetNSegDet(10);
667  nuc->Identify();
668  if (nuc->IsIdentified()) {
669  nuc->SetIDCode(kIDCode8);
670  nuc->Calibrate();
671  }
672  }
673  }
674 
675  // any remaining stopped in first stage particles ?
676  parts2.Reset();
677  while ((nuc = (KVINDRAReconNuc*)parts2())) {
678  if (!nuc->IsIdentified() && nuc->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
679  // estimation of Z (minimum) from energy loss (if detector is calibrated)
680  UInt_t zmin = ((KVDetector*)nuc->GetDetectorList()->Last())->FindZmin(-1., nuc->GetMassFormula());
681  if (zmin) {
682  nuc->SetZ(zmin);
683  nuc->SetIsIdentified();
684  nuc->SetIDCode(kIDCode5);
685  // "Identifying" telescope is taken from list of ID telescopes
686  // to which stopping detector belongs
687  nuc->SetIdentifyingTelescope((KVIDTelescope*)nuc->GetStoppingDetector()->GetIDTelescopes()->Last());
688  nuc->Calibrate();
689  }
690  }
691  }
692 }

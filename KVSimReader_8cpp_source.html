16 //
17 // Specific classes derived from this one can be used to read a variety of simulation
18 // output files and transform them into ROOT files containing a TTree with KVSimEvent
19 // objects in a branch called 'Simulated_events'.
20 //
21 // To see the list of available classes/models:
22 //
23 // kaliveda[0] KVBase::GetListOfPlugins("KVSimReader")
24 // (const Char_t *) "KVSimReader_ELIE KVSimReader_ELIE_asym KVSimReader_HIPSE KVSimReader_HIPSE_asym ... "
25 //
26 // Usually for each model there is a class [XXX] and a class [XXX]_asym.
27 // The first one is for reading "primary" events (e.g. before secondary decay),
28 // the second one is for converting "asymptotic" events (e.g. just before detection).
29 // To generate an instance of the appropriate reader object, use the static method
30 // KVSimReader::MakeReader("[model name]_[asym]")
31 //
32 // Example 1
33 // To convert asymptotic events from an MMM simulation in file "mmm.out", do:
34 //
35 // kaliveda[0] KVSimReader::MakeReader("MMM_asym")->ConvertAndSaveEventsInFile("mmm.out")
36 //
37 // Example 2
38 // To convert primary events (before secondary decay) from an ELIE simulation in file "elie.out", do:
39 //
40 // kaliveda[0] KVSimReader::MakeReader("ELIE")->ConvertAndSaveEventsInFile("elie.out")

172 void KVSimReader::DeclareTree(Option_t* option)
173 {
174  // Use root_file_name as basename for file
175  // If multiple files are being read, they will be named
176  // [root_file_name].1
177  // [root_file_name].2
178  // [root_file_name].etc
179  // Files are written in fOutputDirectory (default: "./")
180 
181  TString filename;
182  filename = root_file_name;
183  if (fMultiFiles) filename += Form(".%d", fFileIndex);
184  if (!fOutputDirectory.EndsWith("/")) fOutputDirectory += "/";
185  filename.Prepend(fOutputDirectory);
186  Info("DeclareTree", "Ouverture du fichier de stockage %s", filename.Data());
187  file = new TFile(filename, option);
188 
189  tree = new TTree(tree_name, tree_title);
190  KVEvent::MakeEventBranch(tree, branch_name, "KVSimEvent", &evt);
191 }

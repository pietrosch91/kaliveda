254 void KVIonRangeTableMaterial::Initialize()
255 {
256  // Correctly initialize material ready for use
257  // For compound or mixed materials, calculate normalised weights of components,
258  // effective Z and A, and molar weight of substance
259 
260  fMoleWt = 0.;
261  if (IsCompound() || IsMixture()) {
262  // mixture or compound
263  // calculate molar weight and effective Z & A
264  fZmat = fAmat = 0;
265  TIter next(fComposition);
266  KVNameValueList* nvl;
267  Double_t totW = 0;
268  while ((nvl = (KVNameValueList*)next())) {
269  Double_t arw = nvl->GetDoubleValue("Ar*Weight");
270  Double_t poid = nvl->GetDoubleValue("Weight");
271  fMoleWt += arw;
272  totW += poid;
273  fZmat += poid * nvl->GetIntValue("Z");
274  fAmat += poid * nvl->GetIntValue("A");
275  }
276  fZmat /= totW;
277  fAmat /= totW;
278  next.Reset();
279  while ((nvl = (KVNameValueList*)next())) {
280  Double_t prop = nvl->GetDoubleValue("Weight");
281  nvl->SetValue("NormWeight", prop / totW);
282  }
283  } else {
284  // isotopically-pure elemental material
285  // get mass of 1 mole of element
286  KVNucleus n(fZmat, fAmat);
287  fMoleWt = n.GetAtomicMass();
288  }
289 }

338 void KVIonRangeTableMaterial::PrintRangeTable(Int_t Z, Int_t A, Double_t isoAmat, Double_t units, Double_t T, Double_t P)
339 {
340  // Print range of element (in g/cm**2) as a function of incident energy (in MeV).
341  // For solid elements, print also the linear range (in cm). To change the default units,
342  // set optional argument units (e.g. to have linear range in microns, call with units = KVUnits::um).
343  // For gaseous elements, give the temperature (in degrees) and the pressure (in torr)
344  // in order to print the range in terms of length units.
345 
346  GetRangeFunction(Z, A, isoAmat);
347  printf(" **** %s Range Table ****\n\n", GetTable()->GetName());
348  ls();
349  printf(" Element: Z=%d A=%d\n\n", Z, A);
350  printf("\tENERGY (MeV)\t\tRANGE (g/cm**2)");
351  if (!IsGas() || (IsGas() && T > 0 && P > 0)) printf("\t\tLIN. RANGE");
352  SetTemperatureAndPressure(T, P);
353  printf("\n\n");
354  for (Double_t e = 0.1; (e <= 1.e+4 && e <= GetEmaxValid(Z, A)); e *= 10) {
355  printf("\t%10.5g\t\t%10.5g", e, fRange->Eval(e));
356  if (!IsGas() || (IsGas() && T > 0 && P > 0)) printf("\t\t\t%10.5g", fRange->Eval(e) / GetDensity() / units);
357  printf("\n");
358  }

683  // with the properties of this material.

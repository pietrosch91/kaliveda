195 Double_t KVSeD::GetCalibT(const Char_t* type)
196 {
197  // Calculates calibrated time in ns of type 'type' (SED1_HF, SED2_HF,
198  // SED1_SED2, ...) for coder values.
199  // Returns 0 if calibration not ready or time ACQ parameter not fired.
200  // (we require that the acquisition parameter has a value
201  // greater than the current pedestal value).
202  //
203  // The returned value 'T' is:
204  // T = Tns + T0 + corrT0
205  // where 'Tns' is the first-converted time in ns, 'T0' is the zero-time and
206  // 'corrT0' is a correction of 'T0'. For the moment the correction is
207  // performed from the raw Xf-position measured in this SeD ( see method
208  // GetXfCorrectorOfT0(...) ).
209 
210  // standard calibrated time: Tns + T0
211  Double_t t = KVVAMOSDetector::GetCalibT(type);
212 
213  // correction of T0 as a function of Xf (raw value)
214  KVFunctionCal* xfcorr = GetXfCorrectorOfT0(type);
215  if (xfcorr && xfcorr->GetStatus()) {
216  Double_t xraw = GetRawPosition('X');
217  if (xraw > 0) t += xfcorr->Compute(xraw);
218  }
219 
220  return t;
221 }

249 Float_t KVSeD::CalculateQThreshold(Char_t dir)
250 {
251  // Calculate the noise in order to remove it on strips for X and
252  // Y positions.
253 
254  //Study the noise and the maximum value for the charge.
255  //At this stage the quantities noise_*** contain also the signal!
256  Float_t QTmp, noise_mean, noise_var, noise_stdev, noise_sum;
257  Float_t Q_max;
258  Q_max = noise_mean = noise_var = noise_sum = noise_stdev = 0.;
259 
260  UShort_t Q_max_Nr;
261  Q_max_Nr = 1000 ;
262 
263  TH1F* hQcal = GetQHisto(dir);
264  if (!hQcal) return 0;
265 
266  for (Int_t j = 1; j <= hQcal->GetNbinsX(); j++) { //loop over strips
267  QTmp = hQcal->GetBinContent(j);
268  if (QTmp > 0) {
269  noise_mean += QTmp;
270  noise_var += QTmp * QTmp;
271  noise_sum += 1.;
272  }
273  if (QTmp > Q_max) {
274  Q_max = QTmp;
275  Q_max_Nr = j;
276  }
277  }
278 
279  //Study the surrounding of the maximum charge
280  Float_t max_mean, max_var, max_sum;
281  max_mean = max_var = max_sum = 0.;
282  for (Int_t j = -Border; j <= Border; j++) {
283  Int_t k = Q_max_Nr + j;
284  if ((0 < k) && (k <= hQcal->GetNbinsX())) {
285  QTmp = hQcal->GetBinContent(k);
286  if (QTmp > 0) {
287  max_mean += QTmp;
288  max_var += QTmp * QTmp;
289  max_sum += 1.;
290  }
291  }
292  }
293  //The signal of the maximum charge is removed from the quantities noise_***
294  noise_mean -= max_mean;
295  noise_var -= max_var;
296  noise_sum -= max_sum;
297  //Calculate statistical quntities related to the noise and fix the threshold
298  //as the mean of the noise + 1 time its standart deviation
299  if (noise_sum < 2. || noise_var == 0) {
300  noise_mean = noise_var = 0;
301  } else {
302  noise_mean /= noise_sum;
303  noise_var = noise_var / noise_sum - noise_mean * noise_mean;
304  noise_stdev = TMath::Sqrt(noise_var);
305  }
306  Float_t threshold = noise_mean + noise_stdev;
307 
308  return threshold;
309 }

317 TH1F* KVSeD::GetCleanQHisto(Char_t dir)
318 {
319  // Retruns the X or Y position histogram with calibrated charges (Q)
320  // after the substration of noise.
321 
322  Int_t i = IDX(dir);
323  if ((i < 0) || (i > 2)) return NULL;
324  if (!fQ[2][i]) return NULL;
325  if (fQ[2][i]->GetEntries()) return fQ[2][i];
326 
327  TH1F* hQcal = GetQHisto(dir);
328  if (!hQcal) return NULL;
329 
330  Float_t threshold = CalculateQThreshold(dir);
331 
332  for (Int_t j = 1; j <= hQcal->GetNbinsX(); j++) { //loop over strips
333  Double_t QTmp = hQcal->GetBinContent(j);
334  if (QTmp > threshold) {
335  fQ[2][i]->SetBinContent(j, QTmp - threshold);
336  }
337  }
338 
339  return fQ[2][i];
340 }

511 void KVSeD::SetACQParams()
512 {
513  // Setup the position acquistion parameters to the SeD.
514  // These ACQ parameters have the type 'Q' (for charge).
515  // Their name has the form: [detector type][detector number]_[X,Y]_[128,48]
516  // where 128 is the number of strips for the X measurment and 48
517  // the number of 48 channels, each connecting 3 wires for the Y
518  // measurment (A. Drouart et al Nuc. Instr. Meth A 579 (2007) 1090).
519  //
520  // Number of each ACQ parameters:
521  // SED1_X_001 --> 1001
522  // SED1_X_002 --> 1002
523  // :
524  // SED1_X_128 --> 1128
525  // SED1_Y_001 --> 2001
526  // SED1_Y_002 --> 2002
527  // :
528  // SED1_Y_128 --> 2128
529 
530 
531  // Charge ACQ parameters
532  TString name, title;
533  KVACQParam* par = NULL;
534  Int_t Nstrips[] = { 128, 48 };
535  const Char_t* extraname[] = { "raw", "calib", "clean"};
536  const Char_t* extratitle[] = { "", "", "without noise"};
537  const Char_t* units[] = {"channel", "Volt", "Volt"};
538  for (Int_t i = 0; i < 2; i++) {
539  for (Int_t num = 1; num <= Nstrips[i]; num++) {
540  par = new KVACQParam;
541  name.Form("%s_%c_%03d", GetArrayName(), DIRECTION(i), num);
542  par->SetName(name);
543  par->SetType("Q");
544  par->SetLabel(Form("%c", DIRECTION(i)));
545  par->SetNumber(num);
546  par->SetUniqueID(CalculateUniqueID(par));
547  AddACQParam(par);
548  }
549 
550  // build Q histograms for X and Y position
551  for (Int_t j = 0; j < 3; j++) {
552  // Qraw histogram
553  if (fQ[j][i]) SafeDelete(fQ[j][i]);
554  name.Form("Q%s_%s_%c", extraname[j], GetArrayName(), DIRECTION(i));
555  title.Form("%c position in %s %s; Strip number; Q%s (%s)", DIRECTION(i), GetArrayName(), extratitle[j], extraname[j], units[j]);
556  fQ[j][i] = new TH1F(name.Data(), title.Data(), Nstrips[i], .5, Nstrips[i] + .5);
557  fQ[j][i]->SetDirectory(0);
558  fQ[j][i]->SetLineColor(j + 1);
559  }
560  }
561 }

578 {
579  // Setup the calibrators for this SeD detector. Call once name
580  // has been set.
581  // The calibrators are KVFunctionCal (see KVVAMOSDetector::SetCalibrators())
582  // except for the position calibration (position->cm) where a

585  // Etra calibrators are set to correct the zero-time T0 of times of flight

588

591

595  if (!AddCalibrator(c)) delete c;
596  else fPosCalib = c;
597 
598 
599  // loop over Time ACQ-parameters associated to this detector
600  // to build Xf-corrector of T0

694 Double_t KVSeD::GetRawPosition2(Char_t dir, Double_t min_amp, Double_t min_sigma, Double_t max_sigma, Int_t maxNpeaks)
695 {
696  // Return the position (strip) deduced from the histogram representing
697  // the calibrated charge versus strip number. First the method searchs
698  // peaks. If there is to many peaks (>maxNpeaks) the method returns -1
699  // else the function
700  // f(x) = [0]*exp(-0.5*((x-[1])/[2])**2)+[3]
701  // is fitted to the histogram by setting initial parameters to
702  // fit the main peak found. The position is given by the parameter[1].
703  // argument: min_amp - acceptable minimum for the amplitude
704  // parameter [0]
705  // min_sigma and max_sigma - acceptable minimum and maximum
706  // for the sigma parameter [2]
707  //
708  // If these two parameter are not in the limits then the method returns -1.
709 
710  IGNORE_UNUSED(maxNpeaks);
711 
712  Int_t idx = IDX(dir);
713  if (fRawPos[ idx ] > -500) return fRawPos[ idx ];
714  fRawPos [ idx ] = -1;
715  fERawPos[ idx ] = -1;
716  TH1F* hh = GetQHisto(dir);
717  if (!hh) return fRawPos[ idx ];
718 
720 // if( !fSpectrum ) fSpectrum = new TSpectrum(15);
721  if (!fPeakFunction) {
722  fPeakFunction = new TF1(Form("KVSeD:%s:PeakFunction", GetName()), "gaus+[3]", 0, 129);
723  fPeakFunction->SetNpx(128);
724  fPeakFunction->SetLineColor(kBlue);
725  }
726 
727  Double_t par[4];
728  TFitResultPtr r = hh->Fit("pol0", "QSNC");
729  if (!r.Get()) return fRawPos[ idx ];
730  par[3] = r->Parameter(0); // noise
731 // Int_t nfound = fSpectrum->Search(hh,2,"goff",0.2);
732 // if( (nfound<1) || (maxNpeaks< nfound) ) return -1;
733 // Double_t *xpeaks = fSpectrum->GetPositionX();
734 // Info("GetRawPosition2","Number of peaks found: %d", nfound);
735  par[2] = 2.; // sigma
736  Int_t locmax = hh->GetMaximumBin();
737  Double_t max = hh->GetBinContent(locmax);
738 
739 // par[1] = xpeaks[0]; // position
740  par[1] = (Double_t)locmax; // position
741  par[0] = TMath::Abs(max - par[3]); // amplitude
742  fPeakFunction->SetParameters(par);
743 
744  fPeakFunction->SetParLimits(0, 0., max);
745  fPeakFunction->SetParLimits(1, 1., hh->GetNbinsX());
746  fPeakFunction->SetParLimits(2, 0.5, max_sigma * 1.1);
747  fPeakFunction->SetParLimits(3, 0., max);
748 
749  fPeakFunction->SetParError(0, par[0] * 0.2);
750  fPeakFunction->SetParError(1, 1);
751  fPeakFunction->SetParError(2, 0.5);
752  fPeakFunction->SetParError(3, max * .2);
753 
754  hh->Fit(fPeakFunction, "QN", "", locmax - Border, locmax + Border);
755 // hh->Fit(fPeakFunction,"","",locmax-Border,locmax+Border);
756 
757  par[0] = fPeakFunction->GetParameter(0);
758  par[1] = fPeakFunction->GetParameter(1);
759  par[2] = fPeakFunction->GetParameter(2);
760 
761  if (par[0] < min_amp) return fRawPos[ idx ];
762  if ((par[2] < min_sigma) || (max_sigma < par[2])) return fRawPos[ idx ];
763 
764  fRawPos [ idx ] = par[1];
765  fERawPos[ idx ] = par[2] * 1.2;
766  return fRawPos[ idx ];
767 }

776 Double_t KVSeD::GetRawPosition(Char_t dir, Int_t /* num */)
777 {
778  // Return the position (strip) deduced from the histogram representing
779  // the calibrated charge versus strip number. Faster method compare to
780  // the method GetRawPosition2. The resolution is less good but sufficient.
781 
782  Int_t idx = IDX(dir);
783  if (fRawPos[ idx ] > -500) return fRawPos[ idx ];
784  fRawPos [ idx ] = -1;
785  fERawPos[ idx ] = -1;
786  TH1F* hh = GetCleanQHisto(dir);
787  if (!hh) return fRawPos[ idx ];
788 
789  if (!hh->GetEntries()) return fRawPos[ idx ];
790 
792 
793  Int_t binMax = hh->GetMaximumBin();
794  Int_t min;
795  for (min = binMax; min > 1; min--) {
796  if (hh->GetBinContent(min) <= 0.) {
797  min++;
798  break;
799  }
800  Double_t deltaQ = hh->GetBinContent(min) - hh->GetBinContent(min - 1);
801  if (deltaQ < 0) break;
802  }
803  Int_t max;
804  for (max = binMax; max < hh->GetNbinsX(); max++) {
805  if (hh->GetBinContent(max) <= 0.) {
806  max--;
807  break;
808  }
809  Double_t deltaQ = hh->GetBinContent(max) - hh->GetBinContent(max + 1);
810  if (deltaQ < 0) break;
811  }

852 UChar_t KVSeD::GetPosition(Double_t* XYZf, Char_t /* dir */, Int_t /* idx */)
853 {
854  // Get calibrated and deviation-corrected positions Xf, Yf and Zf (in cm)
855  // in the focal plan reference frame from the raw positions in channel
856  // obtained with GetRawPosition(...). The argument 'XYZf' has to be an
857  // array of size 3 i.e. XYZf[3].
858  // The 3 first bits of the returned UChar_t value give an information about
859  // the coordinates well determined, calibrated and corrected. For example is the Y
860  // coordinate, with the indice 1, is good then the bit 1 is set to 1.
861  // See documentation of KVSeDPositionCal for more details about
862  // calibration algorithm.
863 
864  XYZf[0] = XYZf[1] = XYZf[2] = -666;
865  // Nothing is done if there is no calibrator for the position
866  if (!IsPositionCalibrated()) return 0;
867 
868  Double_t Xraw = GetRawPosition('X');
869  Double_t Yraw = GetRawPosition('Y');
870 
871  // Calibration is performed with the calibrator KVSeDPositionCal
872  // The calibrator return kFALSE if at least one of the raw coordinates is
873  // less of equal to zero
874 
875  Bool_t OK = kFALSE;
876  UChar_t rvalue = 0; // returned value;
877  XYZf[2] = 0.; // Z = 0 in the reference frame of the
878  // active volule
879 
880  if ((Xraw > 0) && (Yraw > 0)) {
881  OK = (*fPosCalib)(Xraw, Yraw, XYZf[0], XYZf[1]);
882  OK = (OK && ActiveVolumeToFocal(XYZf, XYZf));
883  rvalue = 7; // X,Y,Z are OK
884  } else if (Xraw > 0) {
885  Yraw = 24; // we impose the center in Y (48/2) for the X calibration
886  OK = (*fPosCalib)(Xraw, Yraw, XYZf[0], XYZf[1]);
887  OK = (OK && ActiveVolumeToFocal(XYZf, XYZf));
888  XYZf[1] = -666;
889  rvalue = 5; // X and Z are OK
890  } else if (Yraw > 0) {
891  Xraw = 64; // we impose the center in X (128/2) for the Y calibration
892  OK = (*fPosCalib)(Xraw, Yraw, XYZf[0], XYZf[1]);
893  OK = (OK && ActiveVolumeToFocal(XYZf, XYZf));
894  XYZf[0] = -666;
895  rvalue = 6; // Y and Z are OK
896  }
897 
898  if (OK) return rvalue;
899 
900  // No coordinates are OK
901  XYZf[0] = XYZf[1] = XYZf[2] = -666;
902  return 0;
903 }

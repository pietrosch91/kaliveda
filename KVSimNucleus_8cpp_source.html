128 KVSimNucleus KVSimNucleus::operator+(const KVSimNucleus& rhs)
129 {
130  // KVNucleus addition operator.
131  // Add two nuclei together to form a compound nucleus whose Z, A, momentum
132  // and excitation energy are calculated from energy and momentum conservation.
133 
134  KVSimNucleus& lhs = *this;
135  Int_t ztot = lhs.GetZ() + rhs.GetZ();
136  Int_t atot = lhs.GetA() + ((KVNucleus&) rhs).GetA();
137  KVSimNucleus CN(ztot, atot);
138 
139  Double_t etot = lhs.E() + rhs.E();
140  TVector3 ptot = lhs.GetMomentum() + rhs.GetMomentum();
141  TLorentzVector q(ptot, etot);
142  CN.Set4Mom(q);
143 
144  TVector3 pos;
145  const TVector3* prhs = rhs.GetPosition();
146  pos.SetX(prhs->X()*rhs.GetA() + position.X()*lhs.GetA());
147  pos.SetY(prhs->Y()*rhs.GetA() + position.Y()*lhs.GetA());
148  pos.SetZ(prhs->Z()*rhs.GetA() + position.Z()*lhs.GetA());
149 
150  pos *= 1. / (atot);
151  CN.SetPosition(pos);
152 
153  return CN;
154 
155 }

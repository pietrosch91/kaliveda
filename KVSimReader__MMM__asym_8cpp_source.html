91 Bool_t KVSimReader_MMM_asym::ReadEvent()
92 {
93 
94  evt->Clear();
95  Int_t mult = 0, natt = 0;
96 
97  ReadLine(" ");
98 
99  Int_t res = GetNparRead();
100  switch (res) {
101  case 0:
102  return kFALSE;
103  default:
104  //nlus = toks->GetEntries();
105  idx = 0;
106  mult = GetIntReadPar(idx++);
107  natt = 5 * mult + 1;
108  if (natt != res) {
109  Info("ReadEvent", "Nombre de parametres (%d) different de celui attendu (%d)", res, natt);
110  return kFALSE;
111  }
112  for (Int_t mm = 0; mm < mult; mm += 1) {
113  nuc = (KVSimNucleus*)evt->AddParticle();
114  ReadNucleus();
115  }
116 
117  evt->SetNumber(nevt);
118  nevt += 1;
119  return kTRUE;
120  }

128 Bool_t KVSimReader_MMM_asym::ReadNucleus()
129 {
130 
131  Int_t aa = GetIntReadPar(idx++);
132  Int_t zz = GetIntReadPar(idx++);
133 
134  nuc->SetZ(zz);
135  nuc->SetA(aa);
136 
137  Double_t px = GetDoubleReadPar(idx++);
138  Double_t py = GetDoubleReadPar(idx++);
139  Double_t pz = GetDoubleReadPar(idx++);
140 
141  nuc->SetMomentum(px, py, pz);
142 
143  if (fApplyBoost) {
144  // add particle twice to the event
145  // the first one (labelled "Prov=QP") with the particle's momentum
146  // in the QP Frame
147  // the second one (labelled "Prov=QC") with the particle's momentum
148  // in the QC Frame
149  nuc->ChangeFrame(KVFrameTransform(fBoostQP, kTRUE));
150  nuc->GetParameters()->SetValue("Prov", "QP");
151 
152  nuc = (KVSimNucleus*)evt->AddParticle();
153  nuc->SetZandA(zz, aa);
154  nuc->SetMomentum(px, py, pz);
155  nuc->ChangeFrame(KVFrameTransform(fBoostQC, kTRUE));
156  nuc->GetParameters()->SetValue("Prov", "QC");
157  }
158 
159  return kTRUE;

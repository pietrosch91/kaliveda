27 // KVDBSystem
28 //
29 // Database class used to store information on different colliding systems studied during an experiment.
30 // A 'system' is a unique combination of projectile, target and beam energy.
31 //
32 // Informations on Z & A of projectile and target and beam energy are here (GetZbeam(), GetZtarget(), etc.)
33 //
34 // GetTarget() returns a pointer to a KVTarget object describing the physical characteristics of the
35 // target for this system : such an object can be used to calculate energy losses of incident charged
36 // particles etc.
37 // NOTE: if the current run being analysed corresponds to this system, the target of INDRA will have
38 // been set using this method. i.e. you can use gIndra->GetTarget() to obtain the KVTarget object of the current
39 // run.
40 //
41 // GetKinematics() returns a pointer to a KV2Body object which can be used to obtain information on the kinematics
42 // of the colliding system. See the KV2Body class for more details.
43 // Some of the more commonly used kinematical quantities are also available directly here: GetPtot(), GetZVtot(),
44 // GetECM() etc.

129 KV2Body* KVDBSystem::GetKinematics()
130 {
131  // Create (if it doesn't already exist) and return pointer to a KV2Body object initialised
132  // with the entrance channel corresponding to this system. Use this to obtain information
133  // such as the recoil velocity of the CM, available energy, etc. (see KV2Body).
134  //
135  // If no projectile and/or target are defined for the system, we return 0x0.
136 
137  if (GetZbeam()*GetZtarget() == 0) return nullptr;
138 
139  if (!fCinema) {
140  fCinema = new KV2Body();
141  fCinema->SetProjectile(GetZbeam(), GetAbeam());
142  fCinema->SetTarget(GetZtarget(), GetAtarget());
143  fCinema->GetNucleus(1)->SetEnergy(GetEbeam() * GetAbeam());
144  fCinema->CalculateKinematics();
145  }
146  return fCinema;
147 }

234 Int_t KVDBSystem::Compare(const TObject* obj) const
235 {
236  //Function used to sort lists of systems.
237  //Systems are sorted according to the number of the first run in the
238  //(sorted) list of runs associated to the system.
239  //Systems with lower first run numbers appear earlier in the list.
240 
241  if (!GetRuns())
242  return 0;
243  KVDBSystem* other_sys =
244  dynamic_cast < KVDBSystem* >(const_cast < TObject* >(obj));
245  if (!other_sys)
246  return 0;
247  KVList* other_runs;
248  if (!(other_runs = other_sys->GetRuns()))
249  return 0;
250  Int_t first = ((KVDBRecord*) fRunlist->At(0))->GetNumber();
251  Int_t other_first = ((KVDBRecord*) other_runs->At(0))->GetNumber();
252  return (first == other_first ? 0 : (other_first > first ? -1 : 1));
253 }

351 void KVDBSystem::Load(istream& f)
352 {
353  //Read and set informations on system in the format used in Systems.dat files:
354  //
355  //+155Gd + 238U 36 MeV/A '+' followed by name of system
356  //155 64 238 92 36.0 Aproj Zproj Atarg Ztarg Ebeam
357  //Target: 3 0.0 target with 3 layers, angle 0 degrees
358  //C 0.02 1st layer : carbon, 20 ug/cm2
359  //238U 0.1 2nd layer : uranium-238, 100 ug/cm2
360  //C 0.023 3rd layer : carbon, 23 ug/cm2
361  //Runs: 770-804 list of runs in KVNumberList format
362 
363  TString line;
364  Float_t target_thickness;
365  fAbeam = fZbeam = fAtarget = fZtarget = 0;
366  fEbeam = target_thickness = 0;
367  KVNumberList runlist;
368  //'peek' at first character of next line
369  char next_char = f.peek();
370  if (next_char == '+') {
371  line.ReadLine(f, kFALSE);
372  line.Remove(0, 1);
373  SetName(line.Data());
374  cout << "New System : " << line.Data() << endl;
375  } else {
376  Error("Load", "Should read system name : %s\n", line.Data());
377  return;
378  }
379  next_char = f.peek();
380  while (next_char != '+' && f.good() && !f.eof()) {
381  if ((next_char >= '0') && (next_char <= '9')) {
382  line.ReadLine(f, kFALSE);
383  if (sscanf(line.Data(), "%u %u %u %u %f %f", &fAbeam, &fZbeam, &fAtarget, &fZtarget, &target_thickness, &fEbeam) == 6) {
384  cout << "Zproj = " << fZbeam << " Ztarg = " << fZtarget << " targ_thick = " << target_thickness << " Ebeam = " << fEbeam << endl;
385  } else if (sscanf(line.Data(), "%u %u %u %u %f", &fAbeam, &fZbeam, &fAtarget, &fZtarget, &fEbeam) == 5) {
386  cout << "Zproj = " << fZbeam << " Ztarg = " << fZtarget << " Ebeam = " << fEbeam << endl;
387  }
388  } else {
389  line.ReadLine(f, kFALSE);
390  if (line.BeginsWith("Target")) {
391  fTarget = new KVTarget;
392  line.Remove(0, line.Index(":") + 1);
393  Int_t nlay;
394  Float_t angle;
395  sscanf(line.Data(), "%d %f", &nlay, &angle);
396  Char_t mat[10];
397  Float_t thick;
398  for (int i = 0; i < nlay; i++) {
399  line.ReadLine(f);
400  sscanf(line.Data(), "%s %f", mat, &thick);
401  fTarget->AddLayer(mat, thick);
402  }
403  fTarget->SetAngleToBeam(angle);
404  fTarget->Print();
405  } else if (line.BeginsWith("Runs")) {
406  line.Remove(0, line.Index(":") + 1);
407  runlist.SetList(line.Data());
408  cout << "Runs : " << line.Data() << endl;
409  } else if (line.BeginsWith("Run Range")) {
410  line.Remove(0, line.Index(":") + 1);
411  Int_t frun, lrun;
412  sscanf(line.Data(), "%i %i", &frun, &lrun);
413  runlist.Add(Form("%i-%i", frun, lrun));
414  cout << "Run range : " << line.Data() << endl;
415  }
416  }
417  next_char = f.peek();
418  }
419  if (runlist.GetNValues()) {
420  SetRuns(runlist);
421  }
422  //set target if not already done (old versions)
423  if (!fTarget && target_thickness > 0 && fZtarget > 0) {
424  KVNucleus n(fZtarget, fAtarget);
425  fTarget = new KVTarget(n.GetSymbol(), target_thickness);
426  fTarget->Print();
427  }
428 }

437 void KVDBSystem::SetRuns(KVNumberList& rl)
438 {
439  //Associate this system with the runs in the list
440  //Any previously associated runs are first removed (links in the runs will be removed too)
441  Info("SetRuns", "Setting runs for system %s : %s", GetName(), rl.AsString());
442  RemoveAllRuns();
443  rl.Begin();
444  KVDBTable* runtable = GetRunsTable();
445  Int_t run_number;
446  while (!rl.End()) {
447  run_number = rl.Next();
448  KVDBRun* run = (KVDBRun*)runtable->GetRecord(run_number);
449  if (run) {
450  if (run->GetSystem()) {
451  Error("SetRuns", "Associating run %d with system \"%s\" : run already associated with system \"%s\"",
452  run_number, GetName(), run->GetSystem()->GetName());
453  }
454  if (AddLink("Runs", run)) {
455  //use name of system as title of run
456  run->SetTitle(GetName());
457  } else {
458  Info("SetRuns", "Could not add link for run %d", run_number);
459  }
460  } else {
461  //Info("SetRuns", "Run %d not found in database", run_number);
462  }
463  }

474 void KVDBSystem::AddRun(KVDBRecord* rec)
475 {
476  //Associate the given run with this system.
477  //If the run was previously associated with another system, this association
478  //will be removed.
479  if (!rec) return;
480  if (!rec->InheritsFrom("KVDBRun")) {
481  Error("AddRun", "Called with pointer to an object of class %s; should inherit from KVDBRun!",
482  rec->ClassName());
483  return;
484  }
485  KVDBRun* run = (KVDBRun*)rec;
486  if (run->GetSystem()) run->GetSystem()->RemoveRun(run);
487  if (AddLink("Runs", run)) {
488  Info("AddRun", "Added link for run %d", run->GetNumber());
489  //use name of system as title of run
490  run->SetTitle(GetName());
491  } else {
492  Info("AddRun", "Could not add link for run %d", run->GetNumber());
493  }

597 KVDBTable* KVDBSystem::GetRunsTable()
598 {
599  //Deduce path to runs table in database from full path to parent table of this record.
600  //The systems are stored in a table called "Systems"
601  //The runs are stored in a table called "Runs"

605  TString path = fFullPathTable.Data();

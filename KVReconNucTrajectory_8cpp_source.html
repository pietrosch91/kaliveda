44 KVReconNucTrajectory::KVReconNucTrajectory(const KVGeoDNTrajectory* tr, const KVGeoDetectorNode* n) :
45  KVGeoDNTrajectory(), fIndependentIdentifications(0)
46 {
47  // Build a reconstructed trajectory on tr starting from node n
48 
49  fAddToNodes = kFALSE;
50  KVUniqueNameList* idtlist = dynamic_cast<KVUniqueNameList*>(AccessIDTelescopeList());
51 
52  tr->SaveIterationState();// in case an iteration was already underway
53  // add all nodes starting at n
54  tr->IterateFrom(n);
55  KVGeoDetectorNode* _n;
56  while ((_n = tr->GetNextNode())) {
57  AddLast(_n);
58  }
59  // add all ID telescopes from parent trajectory which contain only
60  // detectors on this trajectory
61  TIter next(tr->GetIDTelescopes());
62  KVIDTelescope* idt;
63  while ((idt = (KVIDTelescope*)next())) {
64  if (ContainsAll(idt->GetDetectors())) {
65  idtlist->Add(idt);
66  if (idtlist->ObjectAdded()) fIndependentIdentifications += (Int_t)idt->IsIndependent();
67  }
68  }
69 
70  // unique name for fast look-up in hash table
71  SetName(Form("%s_%s", tr->GetTitle(), n->GetName()));
72  tr->RestoreIterationState();
73 }

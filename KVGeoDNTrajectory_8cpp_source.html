17 // A KVGeoDNTrajectory represents one possible trajectory that particles produced in
18 // an event may take to travel from the target through the detectors of the multidetector
19 // array. Each trajectory is made up of a series of KVGeoDetectorNode objects which
20 // are each associated with a detector in the geometry. As the trajectories are mostly
21 // used in particle reconstruction & identification, each trajectory starts from the last
22 // detector in a stack and moves towards the target.
23 //
24 // NAME, NUMBER & TITLE
25 // GetName() - The name of each trajectory is "DET1/DET2/DET3/..." made up from the names of the
26 // detectors/nodes on the trajectory.
27 // GetNumber() - The unique number of the trajectory.
28 // GetTitle() - The title of each trajectory is "DET1/DET2/DET3/..." made up from the names of the
29 // detectors/nodes on the trajectory.
30 //
31 // ITERATE OVER TRAJECTORY
32 // In order to visit each detector/node of the trajectory in order, use the iterators:
33 //
34 // KVGeoDNTrajectory* trajectory; // pointer to some valid trajectory
35 // trajectory->IterateFrom(); // default: start from first node of trajectory
36 // // [i.e. the detector furthest from the target]
37 // KVGeoDetectorNode* node;
38 // while( (node = trajectory->GetNextNode()) ){
39 //
40 // KVDetector* detector = node->GetDetector(); // detector associated to node
41 // /* your code */
42 //
43 // }
44 //
45 // You can also start from any node on the trajectory:
46 //
47 // KVGeoDetectorNode* node_on_traj = trajectory->GetNodeAt(1); // start from second node
48 // trajectory->IterateFrom(node_on_traj);
49 // while( (node = trajectory->GetNextNode()) ){
50 //
51 // /* your code */
52 //
53 // }
54 //
55 // To iterate in the other direction i.e. away from the target:
56 //
57 // trajectory->IterateBackFrom(); // default: start from last node of trajectory
58 // // [i.e. the detector closest to the target]
59 // trajectory->IterateBackFrom(node_on_traj);

199 void KVGeoDNTrajectory::rebuild_title()
200 {
201  // called every time a new node is added to the trajectory
202  // to update the title with the new node name
203  // Dynamically constructed title: DET1/DET2/DET3/...
204  // if fPathInTitle = kFALSE, we change the name not the title
205 
206  TString t;
207  TIter next(&fNodes);
208  KVGeoDetectorNode* n;
209  while ((n = (KVGeoDetectorNode*)next())) {
210  t += n->GetName();
211  t += "/";
212  }
213  if (fPathInTitle) SetTitle(t);
214  else SetName(t);
215 }

263 void KVGeoDNTrajectory::ReverseOrder()
264 {
265  // Reverse the order of the nodes in the trajectory
266 
267  int idx = fNodes.GetEntries();
268  TObjArray tmp(idx);
269  int N = idx;
270  TIter it(&fNodes);
271  KVGeoDetectorNode* node;
272  while ((node = (KVGeoDetectorNode*)it())) tmp.AddAt(node, --idx);
273  fNodes.Clear();
274  for (idx = 0; idx < N; ++idx) fNodes.AddAt(tmp[idx], idx);
275  rebuild_title();
276 }

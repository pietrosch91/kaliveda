43  void SetType(Int_t t)
44  {
45  // Set type of functional to use:
46  // t=0->basic functional t<>0->extended functional
47  // * For the basic formula :
48  // yy = ((g*E)**(mu+1)+lambda**(mu+1)*Z**2*A**mu)**(1/(mu+1))-g*E + pdy
49  // * For the extended formula :
50  // yy = ((g*E)**(mu+nu+1)+(lambda*Z**alpha*A**beta)**(mu+nu+1)+
51  // xi*A**mu*(g*E)**nu)**(1/(mu+mu+1))-g*E + pdy
52  fType = t;
53  };
54  Int_t GetType() const

56  return fType;
57  };
58  void SetLight(Int_t ih)
59  {
60  // Set whether to calculate non-linear light response or not
61  // (if X-coordinate is CsI total light response).
62  // * If ih=0 no non-linear light response : E=xx-pdx
63  // * If ih<>0 non-linear light response included :
64  // E = sqrt(h**2+2*rho*h*(1+log(1+h/rho)))
65  // rho=eta*Z**2*A and h=xx-pdx

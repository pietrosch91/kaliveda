86 void KVCoulombPropagator::CalcDerivs(Double_t, Double_t* Y, Double_t* DYDX)
87 {
88 
89  for (int i = 1; i <= fMult; ++i) {
90  for (int j = 0; j < 3; ++j) {
91  DYDX[particle_position_offset(i) + j] = Y[particle_velocity_offset(i) + j];
92  DYDX[particle_velocity_offset(i) + j] = 0;
93  }
94  }
95  for (int i = 1; i < fMult; ++i) {
96  KVSimNucleus* Ni = static_cast<KVSimNucleus*>(theEvent->GetParticle(i));
97 
98  for (int j = i + 1; j <= fMult; ++j) {
99  KVSimNucleus* Nj = static_cast<KVSimNucleus*>(theEvent->GetParticle(j));
100  TVector3 Rij(Y[particle_position_offset(i)] - Y[particle_position_offset(j)],
101  Y[particle_position_offset(i) + 1] - Y[particle_position_offset(j) + 1],
102  Y[particle_position_offset(i) + 2] - Y[particle_position_offset(j) + 2]);
103  TVector3 rij = Rij.Unit();
104  Double_t F = Ni->GetZ() * Nj->GetZ() * KVNucleus::e2 / Rij.Mag2();
105  TVector3 Fij = F * rij;
106  Double_t mi = Ni->GetMass();
107  Double_t mj = Nj->GetMass();
108 
109  for (int k = 0; k < 3; ++k) {
110  DYDX[particle_velocity_offset(i) + k] += Fij[k] / mi;
111  DYDX[particle_velocity_offset(j) + k] -= Fij[k] / mj;
112  }
113  }
114  }
115 }

121 Double_t KVCoulombPropagator::TotalPotentialEnergy() const

126

128  KVSimNucleus* Nj = static_cast<KVSimNucleus*>(theEvent->GetParticle(j));
129  TVector3 Rij = Ni->GetPosition() - Nj->GetPosition();

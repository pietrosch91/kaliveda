128 void KVFAZIA::SetNameOfDetectors(KVEnv& env)
129 {
130  //define the format of detectors name
131  // label-index
132  // where index = block*100+quartet*10+telescope
133  // example :
134  // SI1-123 is the Silicon 1 of the block 1, the quartet 2 and the telescope 3
135  //
136 
137  for (Int_t bb = fStartingBlockNumber; bb < fNblocks; bb += 1) {
138  for (Int_t qq = 1; qq <= 4; qq += 1) {
139  for (Int_t tt = 1; tt <= 4; tt += 1) {
140  fDetectorLabels.Begin(",");
141  while (!fDetectorLabels.End()) {
142  KVString sdet = fDetectorLabels.Next();
143  env.SetValue(
144  Form("BLOCK_%d_QUARTET_%d_%s-T%d", bb, qq, sdet.Data(), tt),
145  Form("%s-%d", sdet.Data(), bb * 100 + qq * 10 + tt)
146  );
147  }
148  }
149  }
150  }

267 void KVFAZIA::GetDetectorEvent(KVDetectorEvent* detev, TSeqCollection* signals)
268 {
269  // First step in event reconstruction based on current status of detectors in array.
270  // Fills the given KVDetectorEvent with the list of all groups which have fired.
271  // i.e. loop over all groups of the array and test whether KVGroup::Fired() returns true or false.
272  //
273  // If the list of fired acquisition parameters 'signals' is given, KVMultiDetArray::GetDetectorEvent
274  // is called
275  //
276 
277  if (signals) {
278  // list of fired acquisition parameters given
279  TIter next_par(signals);
280 
281  KVSignal* par = 0;
282  KVDetector* det = 0;
283  KVGroup* grp = 0;
284  while ((par = (KVSignal*)next_par())) {
285  if (!(par->GetN() > 0))
286  Info("GetDetectorEvent", "%s empty", par->GetName());
287  par->DeduceFromName();
288 // if (!(det = GetDetector(par->GetDetectorName()))) {
289 // det = GetDetector(KVFAZIADetector::GetNewName(par->GetDetectorName()));
290 // }
291  det = GetDetector(par->GetDetectorName());
292  if (det) {
293  ((KVFAZIADetector*)det)->SetSignal(par, par->GetType());
294  if ((!(((KVFAZIADetector*)det)->GetSignal(par->GetType())->GetN() > 0)))
295  Warning("Error", "%s %s empty signal is returned", det->GetName(), par->GetType());
296  if ((grp = det->GetGroup()) && !detev->GetGroups()->FindObject(grp)) {
297  detev->AddGroup(grp);
298  }
299  }
300  else {
301  Error("GetDetectedEvent", "Unknown detector %s !!!", par->GetDetectorName());
302  }
303  }
304  }
305  else {
306  KVMultiDetArray::GetDetectorEvent(detev, 0);
307  }

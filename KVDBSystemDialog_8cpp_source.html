229 void KVDBSystemDialog::CreateNewSystem()
230 {

323 void KVDBSystemDialog::Undo()
324 {
325  //Revert unsaved changes to system (if fNeedSave=kTRUE), or
326  //revert to previously saved version of Systems.dat file
327 
328  //to remove unsaved changes, we simply rebuild the database from
329  //the existing files
330  //if there are no unsaved changes, we replace the current Systems.dat
331  //with the previous version
332  if (!NeedSave()) {
333  //is their a previous version of the Systems.dat ?
334  KVString old_vers = fUndo->GetPreviousVersion(fCurrentSystemsFile.Data());
335  if (old_vers == "") return; // no older version : cannot undo
336  //get full path to previous version
337  KVString fullpath_backup;
338  KVBase::SearchKVFile(old_vers.Data(), fullpath_backup, gDataSet->GetDataSetDir());
339  //get full path to Systems.dat file
340  KVString fullpath_systemsdat;
341  KVBase::SearchKVFile(gDataSet->GetDataSetEnv("INDRADB.Systems"), fullpath_systemsdat, gDataSet->GetDataSetDir());
342  //save current Systems.dat with timestamp if it was not already an old version
343  if (fCurrentSystemsFile == gDataSet->GetDataSetEnv("INDRADB.Systems")) {
344  KVBase::BackupFileWithDate(fullpath_systemsdat.Data());
345  }
346  //now make a copy of the previous version, with the name Systems.dat
347  gSystem->CopyFile(fullpath_backup.Data(), fullpath_systemsdat.Data(), kTRUE);
348  fCurrentSystemsFile = old_vers;
349  }
350  //now rebuild the database
351  gDataSet->GetDataBase("update")->cd();
352  SetNeedSave(0);
353  //update list of systems
354  FillSystemList();
355  UpdateRunlist();
356 }

714 void KVDBSystemDialog::RemoveTargetLayer()
715 {
716  //Removes currently selected layer from target.
717  //In fact we create a new target which does not have the offending layer
718 
719  cout << "Removing layer : " << fLayer->GetName() << " from target !" << endl;
720  if (fTarget->NumberOfLayers() == 1) {
721  //last layer in target - destroy target
722  delete fTarget;
723  fTarget = 0;
724  if (fSystem) fSystem->SetTarget(0);
725  fLayer = 0;
726  } else {
727  KVTarget* new_target = new KVTarget;
728  TIter next(fTarget->GetLayers());
729  KVMaterial* mat;
730  while ((mat = (KVMaterial*)next())) {
731  if (mat != fLayer) {
732  new_target->AddLayer(mat->GetType(), mat->GetAreaDensity());
733  }
734  }
735  new_target->SetAngleToBeam(fTarget->GetAngleToBeam());
736  delete fTarget;
737  fTarget = 0;
738  if (fSystem) fSystem->SetTarget(new_target);
739  fLayer = 0;
740  }
741  UpdateTargetProperties();
742  SetNeedSave(1);
743 }

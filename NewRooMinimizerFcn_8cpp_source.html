152 Bool_t NewRooMinimizerFcn::Synchronize(std::vector<ROOT::Fit::ParameterSettings>& parameters,
153  Bool_t optConst, Bool_t verbose)
154 {
155 
156  // Internal function to synchronize TMinimizer with current
157  // information in RooAbsReal function parameters
158 
159  Bool_t constValChange(kFALSE) ;
160  Bool_t constStatChange(kFALSE) ;
161 
162  Int_t index(0) ;
163 
164  // Handle eventual migrations from constParamList -> floatParamList
165  for (index = 0; index < _constParamList->getSize() ; index++) {
166 
167  RooRealVar* par = dynamic_cast<RooRealVar*>(_constParamList->at(index)) ;
168  if (!par) continue ;
169 
170  RooRealVar* oldpar = dynamic_cast<RooRealVar*>(_initConstParamList->at(index)) ;
171  if (!oldpar) continue ;
172 
173  // Test if constness changed
174  if (!par->isConstant()) {
175 
176  // Remove from constList, add to floatList
177  _constParamList->remove(*par) ;
178  _floatParamList->add(*par) ;
179  _initFloatParamList->addClone(*oldpar) ;
180  _initConstParamList->remove(*oldpar) ;
181  constStatChange = kTRUE ;
182  _nDim++ ;
183 
184  if (verbose) {
185  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: parameter "
186  << par->GetName() << " is now floating." << endl ;
187  }
188  }
189 
190  // Test if value changed
191  if (par->getVal() != oldpar->getVal()) {
192  constValChange = kTRUE ;
193  if (verbose) {
194  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: value of constant parameter "
195  << par->GetName()
196  << " changed from " << oldpar->getVal() << " to "
197  << par->getVal() << endl ;
198  }
199  }
200 
201  }
202 
203  // Update reference list
204  *_initConstParamList = *_constParamList ;
205 
206  // Synchronize MINUIT with function state
207  // Handle floatParamList
208  for (index = 0; index < _floatParamList->getSize(); index++) {
209  RooRealVar* par = dynamic_cast<RooRealVar*>(_floatParamList->at(index)) ;
210 
211  if (!par) continue ;
212 
213  Double_t pstep(0) ;
214  Double_t pmin(0) ;
215  Double_t pmax(0) ;
216 
217  if (!par->isConstant()) {
218 
219  // Verify that floating parameter is indeed of type RooRealVar
220  if (!par->IsA()->InheritsFrom(RooRealVar::Class())) {
221  oocoutW(_context, Minimization) << "NewRooMinimizerFcn::fit: Error, non-constant parameter "
222  << par->GetName()
223  << " is not of type RooRealVar, skipping" << endl ;
224  _floatParamList->remove(*par);
225  index--;
226  _nDim--;
227  continue ;
228  }
229 
230  // Set the limits, if not infinite
231  if (par->hasMin())
232  pmin = par->getMin();
233  if (par->hasMax())
234  pmax = par->getMax();
235 
236  // Calculate step size
237  pstep = par->getError();
238  if (pstep <= 0) {
239  // Floating parameter without error estitimate
240  if (par->hasMin() && par->hasMax()) {
241  pstep = 0.1 * (pmax - pmin);
242 
243  // Trim default choice of error if within 2 sigma of limit
244  if (pmax - par->getVal() < 2 * pstep) {
245  pstep = (pmax - par->getVal()) / 2 ;
246  } else if (par->getVal() - pmin < 2 * pstep) {
247  pstep = (par->getVal() - pmin) / 2 ;
248  }
249 
250  // If trimming results in zero error, restore default
251  if (pstep == 0) {
252  pstep = 0.1 * (pmax - pmin);
253  }
254 
255  } else {
256  pstep = 1 ;
257  }
258  if (verbose) {
259  oocoutW(_context, Minimization) << "NewRooMinimizerFcn::synchronize: WARNING: no initial error estimate available for "
260  << par->GetName() << ": using " << pstep << endl;
261  }
262  }
263  } else {
264  pmin = par->getVal() ;
265  pmax = par->getVal() ;
266  }
267 
268  // new parameter
269  if (index >= Int_t(parameters.size())) {
270 
271  if (par->hasMin() && par->hasMax()) {
272  parameters.push_back(ROOT::Fit::ParameterSettings(par->GetName(),
273  par->getVal(),
274  pstep,
275  pmin, pmax));
276  } else {
277  parameters.push_back(ROOT::Fit::ParameterSettings(par->GetName(),
278  par->getVal(),
279  pstep));
280  if (par->hasMin())
281  parameters.back().SetLowerLimit(pmin);
282  else if (par->hasMax())
283  parameters.back().SetUpperLimit(pmax);
284  }
285 
286  continue;
287 
288  }
289 
290  Bool_t oldFixed = parameters[index].IsFixed();
291  Double_t oldVar = parameters[index].Value();
292  Double_t oldVerr = parameters[index].StepSize();
293  Double_t oldVlo = parameters[index].LowerLimit();
294  Double_t oldVhi = parameters[index].UpperLimit();
295 
296  if (par->isConstant() && !oldFixed) {
297 
298  // Parameter changes floating -> constant : update only value if necessary
299  if (oldVar != par->getVal()) {
300  parameters[index].SetValue(par->getVal());
301  if (verbose) {
302  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: value of parameter "
303  << par->GetName() << " changed from " << oldVar
304  << " to " << par->getVal() << endl ;
305  }
306  }
307  parameters[index].Fix();
308  constStatChange = kTRUE ;
309  if (verbose) {
310  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: parameter "
311  << par->GetName() << " is now fixed." << endl ;
312  }
313 
314  } else if (par->isConstant() && oldFixed) {
315 
316  // Parameter changes constant -> constant : update only value if necessary
317  if (oldVar != par->getVal()) {
318  parameters[index].SetValue(par->getVal());
319  constValChange = kTRUE ;
320 
321  if (verbose) {
322  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: value of fixed parameter "
323  << par->GetName() << " changed from " << oldVar
324  << " to " << par->getVal() << endl ;
325  }
326  }
327 
328  } else {
329  // Parameter changes constant -> floating
330  if (!par->isConstant() && oldFixed) {
331  parameters[index].Release();
332  constStatChange = kTRUE ;
333 
334  if (verbose) {
335  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: parameter "
336  << par->GetName() << " is now floating." << endl ;
337  }
338  }
339 
340  // Parameter changes constant -> floating : update all if necessary
341  if (oldVar != par->getVal() || oldVlo != pmin || oldVhi != pmax || oldVerr != pstep) {
342  parameters[index].SetValue(par->getVal());
343  parameters[index].SetStepSize(pstep);
344  if (par->hasMin() && par->hasMax())
345  parameters[index].SetLimits(pmin, pmax);
346  else if (par->hasMin())
347  parameters[index].SetLowerLimit(pmin);
348  else if (par->hasMax())
349  parameters[index].SetUpperLimit(pmax);
350  }
351 
352  // Inform user about changes in verbose mode
353  if (verbose) {
354  // if ierr<0, par was moved from the const list and a message was already printed
355 
356  if (oldVar != par->getVal()) {
357  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: value of parameter "
358  << par->GetName() << " changed from " << oldVar << " to "
359  << par->getVal() << endl ;
360  }
361  if (oldVlo != pmin || oldVhi != pmax) {
362  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: limits of parameter "
363  << par->GetName() << " changed from [" << oldVlo << "," << oldVhi
364  << "] to [" << pmin << "," << pmax << "]" << endl ;
365  }
366 
367  // If oldVerr=0, then parameter was previously fixed
368  if (oldVerr != pstep && oldVerr != 0) {
369  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: error/step size of parameter "
370  << par->GetName() << " changed from " << oldVerr << " to " << pstep << endl ;
371  }
372  }
373  }
374  }
375 
376  if (optConst) {
377  if (constStatChange) {
378 
379  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::CollectErrors) ;
380 
381  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: set of constant parameters changed, rerunning const optimizer" << endl ;
382  _funct->constOptimizeTestStatistic(RooAbsArg::ConfigChange) ;
383  } else if (constValChange) {
384  oocoutI(_context, Minimization) << "NewRooMinimizerFcn::synchronize: constant parameter values changed, rerunning const optimizer" << endl ;
385  _funct->constOptimizeTestStatistic(RooAbsArg::ValueChange) ;
386  }
387 
388  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::PrintErrors) ;
389 
390  }
391 
392  updateFloatVec() ;
393 
394  return 0 ;
395

467 void NewRooMinimizerFcn::BackProp(const ROOT::Fit::FitResult& results)
468 {
469  // Transfer MINUIT fit results back into RooFit objects
470 
471  for (Int_t index = 0; index < _nDim; index++) {
472  Double_t value = results.Value(index);
473  SetPdfParamVal(index, value);
474 
475  // Set the parabolic error
476  Double_t err = results.Error(index);
477  SetPdfParamErr(index, err);
478 
479  Double_t eminus = results.LowerError(index);
480  Double_t eplus = results.UpperError(index);
481 
482  if (eplus > 0 || eminus < 0) {
483  // Store the asymmetric error, if it is available
484  SetPdfParamErr(index, eminus, eplus);
485  } else {
486  // Clear the asymmetric error
487  ClearPdfParamAsymErr(index) ;
488  }
489  }

594 double NewRooMinimizerFcn::DoEval(const double* x) const
595 {
596 
597  // Set the parameter values for this iteration
598  for (int index = 0; index < _nDim; index++) {
599  if (_logfile)(*_logfile) << x[index] << " " ;
600  SetPdfParamVal(index, x[index]);
601  }
602 
603  // Calculate the function for these parameters
604  RooAbsReal::setHideOffset(kFALSE) ;
605  double fvalue = _funct->getVal();
606  RooAbsReal::setHideOffset(kTRUE) ;
607 
608  if (RooAbsPdf::evalError() || RooAbsReal::numEvalErrors() > 0 || fvalue > 1e30) {
609 
610  if (_printEvalErrors >= 0) {
611 
612  if (_doEvalErrorWall) {
613  oocoutW(_context, Minimization) << "NewRooMinimizerFcn: Minimized function has error status." << endl
614  << "Returning maximum FCN so far (" << _maxFCN
615  << ") to force MIGRAD to back out of this region. Error log follows" << endl ;
616  } else {
617  oocoutW(_context, Minimization) << "NewRooMinimizerFcn: Minimized function has error status but is ignored" << endl ;
618  }
619 
620  TIterator* iter = _floatParamList->createIterator() ;
621  RooRealVar* var ;
622  Bool_t first(kTRUE) ;
623  ooccoutW(_context, Minimization) << "Parameter values: " ;
624  while ((var = (RooRealVar*)iter->Next())) {
625  if (first) {
626  first = kFALSE ;
627  } else ooccoutW(_context, Minimization) << ", " ;
628  ooccoutW(_context, Minimization) << var->GetName() << "=" << var->getVal() ;
629  }
630  delete iter ;
631  ooccoutW(_context, Minimization) << endl ;
632 
633  RooAbsReal::printEvalErrors(ooccoutW(_context, Minimization), _printEvalErrors) ;
634  ooccoutW(_context, Minimization) << endl ;
635  }
636 
637  if (_doEvalErrorWall) {
638  fvalue = _maxFCN + 1 ;
639  }
640 
641  RooAbsPdf::clearEvalError() ;
642  RooAbsReal::clearEvalErrorLog() ;
643  _numBadNLL++ ;
644  } else if (fvalue > _maxFCN) {
645  _maxFCN = fvalue ;
646  }
647 
648  // Optional logging
649  if (_logfile)
650  (*_logfile) << setprecision(15) << fvalue << setprecision(4) << endl;
651  if (_verbose) {
652  cout << "\nprevFCN" << (_funct->isOffsetting() ? "-offset" : "") << " = " << setprecision(10)
653  << fvalue << setprecision(4) << " " ;
654  cout.flush() ;
655  }
656 
657  _evalCounter++ ;
658 
659  return fvalue;
660 }

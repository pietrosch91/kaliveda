138 Double_t KVGaus2D::f2D(Double_t* xx, Double_t* para)
139 {
140 
141 //Function gaussienne 2D
142 //
143 // E1 vs E2
144 
145  userdefined = kTRUE;
146 
147  Double_t e2 = xx[0];
148  Double_t e1 = xx[1];
149 
150  Double_t E2 = para[0];
151  Double_t sigE2 = para[1];
152  Double_t E1 = para[2];
153  Double_t sigE1 = para[3];
154  Double_t rho = para[4];
155  Double_t norm = para[5];
156 
157  Double_t dE1 = E1 - e1;
158  Double_t dE2 = E2 - e2;
159 
160  Double_t A = -0.5 / ((1. - TMath::Power(rho, 2.)) * TMath::Power(sigE1 * sigE2, 2.)) * (TMath::Power(dE1 * sigE2, 2.) + TMath::Power(dE2 * sigE1, 2.) - 2.*rho * dE1 * dE2 * sigE1 * sigE2);
161  Double_t det = (1. - TMath::Power(rho, 2.)) * TMath::Power(sigE1 * sigE2, 2.);
162  Double_t P = norm * 1. / det * exp(A);
163 
164  return P;
165 
166 }

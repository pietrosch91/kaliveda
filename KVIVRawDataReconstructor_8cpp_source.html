92 void KVIVRawDataReconstructor::InitRun()
93 {
94  // Creates new ROOT file with TTree for reconstructed/calibrated events.
95  // By default this file will be written in the same data repository as the raw data file we are reading.
96  // This can be changed by setting the environment variable(s):
97  //
98  // Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
99  // [name of dataset].Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
100  //
101  // If no value is set for the current dataset (second variable), the value of the
102  // first variable will be used. If neither is defined, the new file will be written in the same repository as
103  // the raw file (if possible, i.e. if repository is not remote).
104 
105  // Create new KVIVReconEvent used to reconstruct & store events
106  // The condition used to seed new reconstructed particles (see KVReconstructedEvent::AnalyseTelescopes)
107  // is set by reading the value of the environment variables:
108  // Reconstruction.DataAnalysisTask.ParticleSeedCond: [all/any]
109  // [name of dataset].Reconstruction.DataAnalysisTask.ParticleSeedCond: [all/any]
110  // If no value is set for the current dataset (second variable), the value of the
111  // first variable will be used.
112 
113  if (!recev) recev = fIVevent = new KVIVReconEvent;
114  KVINDRARawDataReconstructor::InitRun();
115 
116  //Remove the first branch 'INDRAReconEvent' of tree created by KVINDRARawDataReconstructor::InitRun()
117  //and replace it by new branch 'IVReconEvent' with a KVIVReconEvent object
118  TObjArray* branches = tree->GetListOfBranches();
119  delete branches->RemoveAt(0);
120  branches->Compress();
121  KVEvent::MakeEventBranch(tree, "IVReconEvent", "KVIVReconEvent", &fIVevent);
122 
123  //Detector events for INDRA and VAMOS
124  fINDRADetEv = new KVDetectorEvent;
125  fVAMOSDetEv = new KVDetectorEvent;
126 
127  //initialise number of reconstructed VAMOS events
128  fNbVAMOSrecon = 0;
129 }

138 void KVIVRawDataReconstructor::preAnalysis()
139 {
140  // Before the reconstruction of a new event:
141  // - Fill the detector events for INDRA and VAMOS.
142  // - Initialize VAMOS spectrometer
143 
144  KVINDRARawDataReconstructor::preAnalysis();
145 
146  KVSeqCollection* fired = fRunFile->GetFiredDataParameters();
147  gIndra->GetDetectorEvent(fINDRADetEv, fired);
148  gVamos->GetDetectorEvent(fVAMOSDetEv, fired);
149 
150  // INDRA event reconstruction

152  // returns the INDRA event for KVINDRARawDataReconstructor::Analysis()
153  KVDetectorEvent* tmp = fDetEv;
154  fDetEv = fINDRADetEv;
155  fINDRADetEv = tmp;
156 
157 
158  gVamos->Initialize();

172 Bool_t KVIVRawDataReconstructor::Analysis()
173 {
174  // Analysis of event measured with the INDRA multidetector array and the VAMOS
175  // spectrometer.
176  //
177  // First, the reconstruction of the VAMOS event is done and then the INDRA event is
178  // reconstructed by calling the Analysis method of the mother class
179  // KVINDRARawDataReconstructor;
180 
181  // VAMOS event reconstruction
182  if (fVAMOSDetEv->GetMult() > 0) {
183  fIVevent->ReconstructVAMOSEvent(gVamos, fVAMOSDetEv);
184  fNbVAMOSrecon++;
185  }
186 
187  // INDRA event reconstruction
188  return KVINDRARawDataReconstructor::Analysis();
189 }

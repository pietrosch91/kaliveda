70 inline Int_t KVIDZALine::Compare(const TObject* obj) const
71 {
72  //Used to sort lists of KVIDZALines.
73  //Sorts the lines as a function of increasing Z, then increasing A
74  //(i.e. p,d,t,3He,4He,6He, etc. etc.)
75 
76  //First compare Z : only if Z1=Z2 do we need to compare A
77 
78  int zline1 = GetZ();
79  int zline2 = ((KVIDZALine*)obj)->GetZ();
80  if (zline1 > zline2)
81  return 1;
82  if (zline1 < zline2)
83  return -1;
84  // same Z: compare A
85  int aline1 = GetA();
86  int aline2 = ((KVIDZALine*)obj)->GetA();
87  if (aline1 > aline2)
88  return 1;
89  if (aline1 < aline2)
90  return -1;
91  return 0;
92 }

76 void KVINDRAUpDater_e475s::SetCalibrationParameters(UInt_t run)
77 {
78 
79 
80  //Set calibration parameters for this run.
81  //This will:
82  // remove all the calibrators of all the detectors ready to receive the calibrators for the run (handled by child classes),
83  // set calibration parameters for the run
84  // set pedestals for the run
85 
86  cout << "Setting calibration parameters of INDRA array for run " << run << ":" <<
87  endl;
88  KVDBRun* kvrun = gIndraDB->GetRun(run);
89  if (!kvrun) {
90  Error("SetParameters(UInt_t)", "Run %u not found in database!", run);
91  return;
92  }
93  //Reset all calibrators of all detectors first
94  TIter next(gIndra->GetListOfDetectors());
95  KVDetector* kvd;
96  while ((kvd = (KVDetector*) next())) {
97  if (kvd->InheritsFrom("KVSiLi") || kvd->InheritsFrom("KVSi75")) {
98  if (kvd->GetListOfCalibrators())
99  kvd->RemoveCalibrators();
100  kvd->SetCalibrators();
101  } else {
102  if (kvd->GetListOfCalibrators()) {
103  kvd->RemoveCalibrators();
104  TIter lacq(kvd->GetACQParamList());
105  KVACQParam* acq = 0;
106  while ((acq = (KVACQParam*)lacq())) {
107  acq->SetPedestal(0);
108  }
109  }
110  }
111  }
112  SetCalibParameters(kvrun);
113  SetPedestals(kvrun);

124 {
125 
126  KVRList* param_list = kvrun->GetLinks("Calibrations");
127 
128  if (!param_list)

131  return;
132 
133  //KVDetector *kvd;
134  KVDBParameterSet* kvps;
135  TIter next_ps(param_list);
136  TString str;
137 
138  // Setting all calibration parameters available
139  while ((kvps = (KVDBParameterSet*) next_ps())) { // boucle sur les parametres
140  str = kvps->GetName();
141  if (gIndra->GetDetector(str.Data())) {
142 
143  if (gIndra->GetDetector(str.Data())->InheritsFrom("KVSilicon_e475s")) {
144  ((KVSilicon_e475s*)gIndra->GetDetector(str.Data()))->SetCalibrator(kvps);
145  } else if (gIndra->GetDetector(str.Data())->InheritsFrom("KVChIo_e475s")) {
146  ((KVChIo_e475s*)gIndra->GetDetector(str.Data()))->SetCalibrator(kvps);
147  } else if (gIndra->GetDetector(str.Data())->InheritsFrom("KVCsI_e475s")) {
148  ((KVCsI_e475s*)gIndra->GetDetector(str.Data()))->SetCalibrator(kvps);
149  } else {
150  Warning("SetCalibParameters(KVDBRun*)",
151  "Calibrator %s %s not found ! ",
152  kvps->GetName(), kvps->GetTitle());
153  }
154  } //detector found
155  } //boucle sur les parameters

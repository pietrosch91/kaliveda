15 // Use this list when you need to check as each object is added that it is not already in
16 // the list (assuming that objects are the same if they have the same name), in order to
17 // avoid the same object being added several times.
18 //
19 // Using a standard TList/KVList, implementing such a check can become VERY time-consuming
20 // as the list becomes longer, because for each new object we add we have to perform a
21 // sequential scan of all objects in the list and check the name of each one. However, using
22 // a hash list based on object names, the number of objects to check is kept small, even
23 // as the list becomes very long, thanks to automatic rehashing of the list.
24 //
25 // By default, if an object with the same name is already in the list, the
26 // 'new' object is not added (no warning, no error).
27 //
28 // If option 'replace' is used, then adding an object with the same name
29 // as an existing object will replace the previous object with the new one.
30 //
31 // In addition, if the list owns its objects and 'replace' is used, the previous
32 // object will be deleted as well as being replaced.
33 //
34 // After calling Add(TObject*), you can use method ObjectAdded() to know if the object
35 // was effectively added to the list or not.
36 //
37 // ============== EXAMPLES ===============
38 //
39 // TNamed* a = new TNamed("A1", "an object");
40 // TNamed* b = new TNamed("B1", "another object");
41 // TNamed* c = a;
42 // TNamed* d = new TNamed("A1", "an object with the same name as another");
43 //
44 // === Example 0 - normal TList
45 //
46 // TList aList;
47 // aList.Add(a);
48 // aList.Add(b);
49 // aList.Add(c);
50 // aList.Add(d);
51 // aList.ls();

227 Bool_t KVUniqueNameList::checkObjInList(TObject* obj)
228 {
229  // Return kTRUE if 'obj' can be added to list.
230  // If fReplace==kFALSE, there must be no object in the list with
231  // same name as obj->GetName().
232  // If fReplace==kTRUE the previous object is removed from the list.
233  // If IsOwner()==kTRUE the previous object is removed AND deleted.
234  // In the latter two cases the new object can always be added (returns kTRUE)
235 
236  TObject* old_obj = FindObject(obj->GetName());
237  if (!old_obj) return kTRUE;
238  else {
239  if (fReplace) {
240  Remove(old_obj);
241  if (IsOwner()) delete old_obj;
242  return kTRUE;
243  }
244  }
245  return kFALSE;
246 }

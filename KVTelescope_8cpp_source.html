17 // KVTelescope
18 //
19 // An assembly of one or more layers of detectors, usually any particle
20 // traversing the layer closest to the target will then traverse all
21 // subsequent layers of the telescope (if it has sufficient energy).
22 //
23 // Telescope structures own their detectors, and will delete them when
24 // the telescope is deleted.
25 //
26 // As detectors are added, we automatically set the KVDetectorNode
27 // information as they are supposed to be placed one behind the other.
28 //
29 // In imported ROOT geometries (see KVGeoImport), you can trigger the creation
30 // of a KVTelescope structure by using a node name of the type
31 //
32 // STRUCT_TELESCOPE_[number]
33 //

148 void KVTelescope::DetectParticle(KVNucleus* kvp, KVNameValueList* nvl)
149 {
150  //Simulates the passage of a charged particle through all detectors of the telescope, in the order in which they
151  //were added to it (corresponding to rank given by GetDetectorRank()).
152  //Begin_Html
153  //<img src="http://indra.in2p3.fr/KaliVedaDoc/images/kvtelescope_detectparticle.gif">
154  //End_Html
155  //It should be noted that
156  // (1) the small variations in effective detector thickness due to the particle's angle of incidence are not, for the moment, taken into account
157  // (2) the simplified description of detector geometry used here does not take into account trajectories such as that marked "b" shown in the figure.
158  // All particles impinging on the first detector of the telescope are assumed to pass through all subsequent detectors as in "c"
159  // (unless they stop in one of the detectors)
160  //
161  //The KVNameValueList, if it's defined, allows to store
162  //the energy loss in the different detectors the particle goes through
163  //exemple : for a Silicon-CsI telescope named SI_CSI_0401 , you will obtain:
164  // {
165  // KVNucleus nn(6,12); nn.SetKE(1000);
166  // KVTelescope* tel = gMultiDetArray->GetTelescope("SI_CSI_0401");
167  // KVNameValueList* nvl = new KVNameValueList;
168  // tel->DetectParticle(&nn,nvl);
169  // nvl->Print();
170  // }
171  // Collection name='KVNameValueList', class='KVNameValueList', size=2
172  // OBJ: TNamed SI_0401 8.934231
173  // OBJ: TNamed CSI_0401 991.065769
174  //The energy loss in each detector corresponds to those lost in active layer
175 
176  KVDetector* obj;
177 
178  TIter next(GetDetectors());
179  while ((obj = (KVDetector*) next())) {
180  Double_t ebefore = obj->GetEnergy(); //Energie dans le detecteur avant passage
181  obj->DetectParticle(kvp); //Detection de la particule
182  ebefore -= obj->GetEnergy(); //la difference d energie avant et apres passage (donc l energie laissee par la particule)
183  if (nvl)
184  nvl->SetValue(obj->GetName(), TMath::Abs(ebefore)); //Enregistrement de la perte d energie
185  if (kvp->GetEnergy() <= 0.0)
186  break;
187  }
188 }

302 TGeoVolume* KVTelescope::GetGeoVolume()
303 {
304  // Create and return TGeoVolume representing detectors in this telescope.
305 
306  int no_of_dets = GetDetectors()->GetEntries();
307  if (no_of_dets == 1) {
308  // single detector "telescope": just return detector volume
309  return GetDetector(1)->GetGeoVolume();
310  }
311  TGeoVolume* mother_vol = gGeoManager->MakeVolumeAssembly(Form("%s_TEL", GetName()));
312  // total length of telescope = depth of last detector + thickness of last detector
313  Double_t tot_len_tel = GetTotalLengthInCM();
314  //**** BUILD & ADD DETECTOR VOLUMES ****
315  TIter next(GetDetectors());
316  KVDetector* det;
317  while ((det = (KVDetector*)next())) {
318  TGeoVolume* det_vol = det->GetGeoVolume();
319  // position detector in telescope
320  Double_t dist = -tot_len_tel / 2. + det->GetDepthInTelescope() + det->GetTotalThicknessInCM() / 2.;
321  TGeoTranslation* tran = new TGeoTranslation(0., 0., dist);
322  mother_vol->AddNode(det_vol, 1, tran);
323  }
324  return mother_vol;
325 }

332 void KVTelescope::AddToGeometry()
333 {
334  // Construct and position a TGeoVolume shape to represent this telescope in the current geometry
335  if (!gGeoManager) return;
336 
337  // get volume for telescope
338  TGeoVolume* vol = GetGeoVolume();
339 
340  // rotate telescope to orientation corresponding to (theta,phi)
341  Double_t theta = GetTheta();
342  Double_t phi = GetPhi();
343  TGeoRotation rot1, rot2;
344  rot2.SetAngles(phi + 90., theta, 0.);
345  rot1.SetAngles(-90., 0., 0.);
346  Double_t tot_len_tel = GetTotalLengthInCM();
347  // distance to telescope centre = distance to telescope + half total lenght of telescope
348  Double_t dist = GetDistance() + tot_len_tel / 2.;
349  // translate telescope to correct distance from target (note: reference is CENTRE of telescope)
350  Double_t trans_z = dist;
351 
352  TGeoTranslation tran(0, 0, trans_z);
353  TGeoHMatrix h = rot2 * tran * rot1;
354  TGeoHMatrix* ph = new TGeoHMatrix(h);
355 
356  // add telescope volume to geometry
357  gGeoManager->GetTopVolume()->AddNode(vol, 1, ph);
358 }

27 <h2>KVElasticScatter</h2>
28 <h4>Calculate elastic scattering spectra in multidetector arrays</h4>
29 
30 <p>
31 Use this class to calculate the energy losses of elastically scattered nuclei
32 in the detectors of a multidetector array. It is assumed that the following
33 global pointers have been initialised before using this class:
34 </p>
35 <ul>
36  <li><pre><a href="KVMultiDetArray.html">gMultiDetArray</a></pre> - points to object describing multidetector array</li>
37  <li><pre><a href="KVDataBase.html">gDataBase</a></pre> - points to database of run-dependent information for experimental dataset</li>
38 </ul>
39 <p>
40 In other words, we assume that the user has chosen a dataset on which she wants
41 to work:
42 </p>
43 <pre><a href="KVDataSetManager.html">gDataSetManager</a>-><a href="KVDataSetManager.html#KVDataSetManager:GetDataSet">GetDataSet</a>("name_of_dataset")-><a href="KVDataSet.html#KVDataSet:cd">cd</a>()</pre>
44 <p>
45 and then initialised the description of the experimental configuration for the dataset:
46 </p>
47 <pre><a href="KVDataSet.html">gDataSet</a>-><a href="KVDataSet.html#KVDataSet:BuildMultiDetector">BuildMultiDetector</a>()</pre>
48 <h3>Setting up the calculation</h3>
49 <p>Create a new elastic scattering object:</p>
50 <pre>KVElasticScatter* es = new KVElasticScatter</pre>
51 <p>
52 Then call any of the following methods in any order in order to set up the calculation:</p>
53 <ul>
54  <li><pre>es-><a href="#KVElasticScatter:SetProjectile">SetProjectile</a>(...)</pre></li>
55  <li><pre>es-><a href="#KVElasticScatter:SetEnergy">SetEnergy</a>(...)</pre></li>
56  <li><pre>es-><a href="#KVElasticScatter:SetDetector">SetDetector</a>(...)</pre></li>
57  <li><pre>es-><a href="#KVElasticScatter:SetRun">SetRun</a>(...)</pre></li>
58 </ul>
59 <p>
60 The SetRun(...) method uses the experimental database in order to determine the target
61 for the run, detector state (gas pressures etc.).
62 </p>
63 <h3>Multilayer targets</h3>
64 <p>
65 When multilayer targets are used, the user can restrict the scattering calculation to one
66 specific component of the target, i.e. the scattering takes place between the projectile
67 and one of the nuclei of the specified layer. Energy losses before and after the scattering
68 in the other layers of the target are of course still taken into account. To set the part of
69 the target where scattering takes place use
70 </p>
71 <pre>es-><a href="#KVElasticScatter:SetTargetScatteringLayer">SetTargetScatteringLayer</a>(...)</pre>
72 <p>
73 with the name of the layer you require (this is normally the name of the element
74 making up the layer - see <a href="KVTarget.html">KVTarget</a> for details).
75 </p>
76 <h3>Inelastic scattering</h3>
77 <p>
78 Calculations can also be performed for inelastic scattering i.e. when the target nucleus is left
79 in an excited state (N.B. we still use the Rutherford elastic scattering cross-section for
80 weighting the energy loss distributions). In order to do this, call
81 </p>
82 <pre>es-><a href="#KVElasticScatter:SetTargetExcitedState">SetTargetExcitedState</a>(...)</pre>
83 <p>
84 with the energy of the excited state of the target after scattering.
85 </p>

262 void KVElasticScatter::SetDetector(const Char_t* det)
263 {
264  //Set name of detector which will detect particle
265  fDetector = gMultiDetArray->GetDetector(det);
266  fTelescope = (KVTelescope*)fDetector->GetParentStructure("TELESCOPE");
267  //get list of all detectors particle must pass through to get to required detector
268  fAlignedDetectors =
269  fDetector->GetAlignedDetectors(KVGroup::kForwards);
270  //we store the association between detector type and index in list
271  if (!fDetInd)
272  fDetInd = new KVNameValueList;
273  else
274  fDetInd->Clear();
275  KVDetector* d;
276  TIter n(fAlignedDetectors);
277  Int_t i = 0;
278  while ((d = (KVDetector*) n())) {
279  //check same type is not already present
280  if (fDetInd->HasParameter(d->GetType())) {
281  //detetors with same type will be called "Type_1", "Type_2" etc.
282  TString newname;
283  int j = 1;
284  newname.Form("%s_%d", d->GetType(), j++);
285  while (fDetInd->HasParameter(newname.Data()))
286  newname.Form("%s_%d", d->GetType(), j++);
287  fDetInd->SetValue(newname.Data(), i);
288  } else {
289  fDetInd->SetValue(d->GetType(), i);
290  }
291  i++;
292  }
293  //store number of aligned detectors
294  fNDets = i;
295 }

343 void KVElasticScatter::CalculateScattering(Int_t N)
344 {
345  //Perform scattering 'N' times for current values
346  //of particle Z, A and energy, target excited state, and detector.
347 
348  if (!fProj) {
349  cout <<
350  "<KVElasticScatter::CalculateScattering> : Set projectile properties first"
351  << endl;
352  return;
353  }
354  if (!fEnergy) {
355  cout <<
356  "<KVElasticScatter::CalculateScattering> : Set projectile energy first"
357  << endl;
358  return;
359  }
360  if (!fDetector) {
361  cout <<
362  "<KVElasticScatter::CalculateScattering> : Set detector first" <<
363  endl;
364  return;
365  }
366  if (!fTarget) {
367  cout <<
368  "<KVElasticScatter::CalculateScattering> : No target set. Set run first."
369  << endl;
370  return;
371  }
372 
373  /* -------------------------------------------------------------------------------------------------------------------------- */
374 
375  //set up histograms
376 
377  /* -------------------------------------------------------------------------------------------------------------------------- */
378 
379  // -- histograms for debugging: depth in target and angular distribution
380  if (fDepth)
381  delete fDepth;
382  if (fTheta)
383  delete fTheta;
384  fDepth =
385  new TH1F("hDepth", "Depth (mg/cm2)", 500, 0.,
386  fTarget->GetTotalEffectiveThickness());
387  fTheta = new TH1F("hTheta", "Theta (deg.)", 500, 0., 0.);
388 
389  /* -------------------------------------------------------------------------------------------------------------------------- */
390 
391  //set up histograms for all detectors particle passes through
392  if (!fHistos) {
393  fHistos = new TObjArray(fAlignedDetectors->GetSize());
394  fHistos->SetOwner(); //will delete histos it stores
395  } else {
396  fHistos->Clear(); //delete any previous histograms
397  }
398  KVDetector* d;
399  TIter n(fAlignedDetectors);
400  while ((d = (KVDetector*) n())) {
401  fHistos->
402  Add(new
403  TH1F(Form("hEloss_%s", d->GetName()), "Eloss (MeV)", fBinE, 0., 0.));
404  }
405 
406  /* -------------------------------------------------------------------------------------------------------------------------- */
407 
408  //set up kinematics
409  if (!fKinematics)
410  fKinematics = new KV2Body;
411  fProj->SetEnergy(fEnergy);
412  fProj->SetTheta(0);
413 
414  /* -------------------------------------------------------------------------------------------------------------------------- */
415 
416  //set random interaction point for scattering
417  if (fIntLayer) {
418  fTarget->SetInteractionLayer(fIntLayer, fBeamDirection);
419  } else {
420  fTarget->GetInteractionPoint(fProj);
421  }
422 
423  /* -------------------------------------------------------------------------------------------------------------------------- */
424 
425  //get target nucleus properties from scattering layer
426  TVector3 IP = fTarget->GetInteractionPoint();
427  KVMaterial* targ_mat = fTarget->GetLayer(IP);
428  if (!fTarg)
429  fTarg = new KVNucleus;
430  fTarg->SetZ((Int_t) targ_mat->GetZ());
431  fTarg->SetA((Int_t) targ_mat->GetMass());
432  fKinematics->SetProjectile(fProj);
433  fKinematics->SetTarget(fTarg);
434  fKinematics->SetOutgoing(fProj);
435 
436  /* -------------------------------------------------------------------------------------------------------------------------- */
437 
438  //set excited state of target nucleus - in other words, dissipated energy for
439  //reaction due to excitation of target
440  fKinematics->SetEDiss(fExx);
441 
442  /* -------------------------------------------------------------------------------------------------------------------------- */
443 
444  Double_t xsec;
445  for (int i = 0; i < N; i++) {
446  //calculate slowing of incoming projectile
447  fTarget->SetIncoming();
448  fTarget->DetectParticle(fProj);
449  fKinematics->CalculateKinematics();
450  //set random direction of outgoing projectile
451 
452  double th, ph;
453  th = ph = 0.;
454  fDetector->GetRandomAngles(th, ph);
455  fProj->SetEnergy(fProj->GetEnergy());
456  fProj->SetTheta(th);
457  fProj->SetPhi(ph);
458 
459 // fProj->SetRandomMomentum(fProj->GetEnergy(),
460 // fTelescope->GetThetaMin(),
461 // fTelescope->GetThetaMax(),
462 // fTelescope->GetPhiMin(),
463 // fTelescope->GetPhiMax(), "random");
464  //set energy of scattered nucleus
465  //WARNING: for inverse kinematics reactions, their are two energies for
466  //each angle below the maximum scattering angle.
467  //We only use the highest energy corresponding to the most forward CM angle.
468  Double_t e1, e2;
469  fKinematics->GetELab(3, fProj->GetTheta(), 3, e1, e2);
470  fProj->SetEnergy(TMath::Max(e1, e2));
471  xsec = TMath::Abs(fKinematics->GetXSecRuthLab(fProj->GetTheta()));
472  fTheta->Fill(fProj->GetTheta(), xsec);
473  //slowing of outgoing projectile in target
474  fTarget->SetOutgoing();
475  fTarget->DetectParticle(fProj);
476  //now detect particle in detector(s)
477  fAlignedDetectors->R__FOR_EACH(KVDetector, DetectParticle)(fProj);
478  //fill histograms
479  fDepth->Fill(IP.z());
480  int j = 0;
481  n.Reset();
482  while ((d = (KVDetector*) n())) {
483  ((TH1F*)(*fHistos)[j++])->Fill(d->GetEnergy(), xsec);
484  //prepare for next round: set energy loss to zero
485  d->Clear();
486  }
487  fProj->SetEnergy(fEnergy);
488  fProj->SetTheta(0);
489  //set random interaction point for scattering
490  if (fIntLayer) {
491  fTarget->SetInteractionLayer(fIntLayer, fBeamDirection);
492  } else {
493  fTarget->GetInteractionPoint(fProj);
494  //if target is multilayer and the interaction layer is not fixed,
495  //the layer & hence the target nucleus may change
496  if (fMultiLayer) {
497  targ_mat = fTarget->GetLayer(fTarget->GetInteractionPoint());
498  fTarg->SetZ((Int_t) targ_mat->GetZ());
499  fTarg->SetA((Int_t) targ_mat->GetMass());
500  fKinematics->SetTarget(fTarg);
501  fKinematics->SetOutgoing(fProj);
502  }
503  }
504  IP = fTarget->GetInteractionPoint();
505  }
506 }

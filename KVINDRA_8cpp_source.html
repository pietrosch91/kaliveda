177 
178  TString path = Form("indra-struct.%s.env", fDataSet.Data());
179  TString path2;
180  SearchKVFile(path.Data(), path2, "data");
181  if (path2 == "") {
182  path = "indra-struct.env";
183  SearchKVFile(path.Data(), path2, "data");
184  }
185  fStrucInfos.ReadFile(path2, kEnvAll);
186 
187  KVString lruns = fStrucInfos.GetValue("AddOnForRuns", "");
188  //test if additional geometrical specification exists
189  if (lruns != "") {
190  lruns.Begin(",");
191  while (!lruns.End()) {
192  KVString sruns = lruns.Next();
193  KVNumberList nlr(sruns.Data());
194  //the current run needs specific geometry
195  if (nlr.Contains(fCurrentRun)) {
196  path = fStrucInfos.GetValue(sruns.Data(), "");
197  Info("BuildGeometry", "Additional geometry for run=%d in file #%s#", fCurrentRun, path.Data());
198  SearchKVFile(path.Data(), path2, "data");
199  if (path2 == "") {
200  Warning("BuildGeometry", "fichier %s inconnu", path.Data());
201  }
202  else {
203  fStrucInfos.ReadFile(path2, kEnvChange);
204  }
205  }
206  }
207  }
208 
209  SetName(fStrucInfos.GetValue("INDRA.Name", ""));
210  SetTitle(fStrucInfos.GetValue("INDRA.Title", ""));

213  layers.Begin(" ");
214  while (!layers.End()) BuildLayer(layers.Next());

247 KVRing* KVINDRA::BuildRing(Int_t number, const Char_t* prefix)
248 {
249  // Build ring with infos in file "$KVROOT/KVFiles/data/indra-struct.[dataset].env"
250 
251  KVRing* ring = new KVRing;
252  Info("BuildRing", "Building ring %d (%s)", number, prefix);
253 
254  Double_t thmin = fStrucInfos.GetValue(Form("%s.ThMin", prefix), 0.0);
255  Double_t thmax = fStrucInfos.GetValue(Form("%s.ThMax", prefix), 0.0);
256  Double_t phi_0 = fStrucInfos.GetValue(Form("%s.Phi0", prefix), 0.0);
257  Int_t ntel = fStrucInfos.GetValue(Form("%s.Ntel", prefix), 0);
258  Int_t step = fStrucInfos.GetValue(Form("%s.Step", prefix), 1);
259  Int_t num_first = fStrucInfos.GetValue(Form("%s.Nfirst", prefix), 1);
260  Double_t dphi = fStrucInfos.GetValue(Form("%s.Dphi", prefix), -1.0);
261  KVNumberList absent = fStrucInfos.GetValue(Form("%s.Absent", prefix), "");
262 
263  ring->SetPolarMinMax(thmin, thmax);
264  KVINDRATelescope* kvt = BuildTelescope(prefix, num_first);
265  Double_t phi_min = phi_0 - 0.5 * (kvt->GetAzimuthalWidth());
266  phi_min = (phi_min < 0. ? phi_min + 360. : phi_min);
267  dphi = (dphi < 0. ? 360. / ntel : dphi);
268  Double_t phi_max =
269  phi_0 + ((ntel - 1.0) * dphi) + 0.5 * (kvt->GetAzimuthalWidth());
270  phi_max = (phi_max > 360. ? phi_max - 360. : phi_max);
271  ring->SetAzimuthalMinMax(phi_min, phi_max);
272  ring->SetNumber(number);
273 
274  Double_t phi = phi_0;
275  UInt_t counter = num_first;
276  for (int i = 0; i < ntel; i++) {
277  kvt->SetPolarMinMax(thmin, thmax);
278  kvt->SetPhi(phi);
279  phi = (phi + dphi > 360. ? (phi + dphi - 360.) : (phi + dphi));
280  kvt->SetNumber(counter);
281  if (!absent.Contains(counter)) ring->Add(kvt);
282  else delete kvt;
283  counter += step;
284  if (i + 1 < ntel) kvt = BuildTelescope(prefix, counter);
285  }
286  ring->SetName(Form("RING_%d", number));
287  return ring;
288 }

296 KVINDRATelescope* KVINDRA::BuildTelescope(const Char_t* prefix, Int_t module)
297 {
298  // Build telescope from infos in file "$KVROOT/KVFiles/data/indra-struct.[dataset].env"
299 
300  //Info("BuildTelescope", "Building telescope %s",name);
301  KVINDRATelescope* kvt = new KVINDRATelescope;
302  KVString telescopes = fStrucInfos.GetValue(Form("%s.Telescope", prefix), "");
303  telescopes.Begin(" ");
304  KVString name;
305  while (!telescopes.End()) {
306  name = telescopes.Next();
307  KVNumberList mods = fStrucInfos.GetValue(Form("%s.Telescope.%s.Modules", prefix, name.Data()), "1-100");
308  if (mods.Contains(module)) break;
309  }
310 
311  KVString detectors = fStrucInfos.GetValue(Form("INDRA.Telescope.%s.Detectors", name.Data()), "");
312  Double_t aziwidth = fStrucInfos.GetValue(Form("INDRA.Telescope.%s.AziWidth", name.Data()), 1.0);
313  Double_t dist = fStrucInfos.GetValue(Form("%s.Dist", prefix), 0.0);
314  kvt->SetDistance(dist);
315  detectors.Begin(" ");
316  Int_t idet = 1;
317  while (!detectors.End()) {
318  KVString detector = detectors.Next();
319  detector.Begin("()");
320  KVString dettype = detector.Next();
321  Double_t detthick = detector.Next().Atof();
322  KVDetector* det = KVDetector::MakeDetector(Form("%s.%s", fDataSet.Data(), dettype.Data()), detthick);
323  kvt->Add(det);
324  Double_t depth = fStrucInfos.GetValue(Form("INDRA.Telescope.%s.Depth.%s", name.Data(), dettype.Data()), 0.);
325  kvt->SetDepth(idet, depth);
326  idet++;
327  }
328  kvt->SetAzimuthalWidth(aziwidth);
329  return kvt;
330 }

343 void KVINDRA::FillTrajectoryIDTelescopeLists()
344 {
345  // Kludge to make INDRA ROOT geometry work like any other
346  // Normally ID telescopes are deduced from successive detectors on the different trajectories
347  // Each trajectory then possesses its list of ID telescopes
348  // These lists are then used when reconstruction trajectories are calculated
349  //
350  // When INDRA ROOT geometry is used, trajectory lists need to be filled
351  // by hand before reconstruction trajectories are calculated
352 
353  TIter it_traj(GetTrajectories());
354  KVGeoDNTrajectory* tr;
355  while ((tr = (KVGeoDNTrajectory*)it_traj())) {
356  // get first detector on trajectory
357  KVDetector* det = tr->GetNodeAt(0)->GetDetector();
358  // list of 'aligned' id telescopes
359  TIter it_idt(det->GetAlignedIDTelescopes());
360  KVIDTelescope* idt;
361  while ((idt = (KVIDTelescope*)it_idt())) {
362  if (tr->ContainsAll(idt->GetDetectors())) { // all detectors on trajectory
363  if (idt->GetDetectors()->GetEntries() > 1) {
364  // make sure dE and E detector are consecutive on trajectory
365  // i.e. dE has to be immediately in front of E on the trajectory
366  if (tr->GetNodeInFront(idt->GetDetector(2)->GetNode()) == idt->GetDetector(1)->GetNode())
367  tr->AccessIDTelescopeList()->Add(idt);
368  }
369  else
370  tr->AccessIDTelescopeList()->Add(idt);//single-detector telescope
371  }
372  }
373  }

1027 void KVINDRA::SetGGtoPGConversionFactors()
1028 {
1029  // Sets the parameters for linear conversion of silicon & ChIo coder values
1030  // between GG and PG, using the following formula:
1031  //
1032  // PG = alpha + beta*(GG - GG_0) + PG_0
1033  //
1034  // where GG_0 and PG_0 are respectively GG and PG pedestals
1035  //
1036  // We look for the file whose name is given by the .kvrootrc variable
1037  // [dataset].INDRADB.GGtoPGFactors:
1038  // or by default
1039  // INDRADB.GGtoPGFactors:
1040  // and expect to find in it a line for each detector of the form:
1041  // Det_Name alpha beta
1042  // Comments in the file can be written on lines beginning with the character '#'
1043 
1044  ifstream datfile;
1045  if (!gDataSet->OpenDataSetFile(gDataSet->GetDataSetEnv("INDRADB.GGtoPGFactors", ""), datfile)) {
1046 
1047  Info("SetGGtoPGConversionFactors", "Cannot open file with parameters for conversion (%s).",
1048  gDataSet->GetDataSetEnv("INDRADB.GGtoPGFactors", ""));
1049  return;
1050  }
1051  else {
1052  Info("SetGGtoPGConversionFactors", "Reading parameters from file %s",
1053  gDataSet->GetDataSetEnv("INDRADB.GGtoPGFactors", ""));
1054 
1055  Char_t detname[30];
1056  Double_t a, b;
1057  TString aline;
1058  aline.ReadLine(datfile);
1059  while (datfile.good()) {
1060 
1061  if (aline[0] != '#') { //skip comments
1062 
1063  sscanf(aline.Data(), "%s %lf %lf", detname, &a, &b);

1067  Error("SetGGtoPGConversionFactors", "Unknown detector : %s", detname);
1068  }
1069  else {
1070  det->SetGGtoPGConversionFactors(a, b);
1071  //Info("SetGGtoPGConversionFactors", "%s : PG = %f + %f * GG", detname, a, b);

1073  }

1077  }

1094 void KVINDRA::CreateROOTGeometry()
1095 {
1096  // Overrides KVASMultiDetArray::CreateGeoManager in order to use INDRAGeometryBuilder
1097  // which builds the TGeo representation of INDRA using the Y. Huguet CAO data.
1098  //
1099  // The optional arguments (dx,dy,dz) are the half-lengths in centimetres of the "world"/"top" volume
1100  // into which all the detectors of the array are placed. This should be big enough so that all detectors
1101  // fit in. The default values of 500 give a "world" which is a cube 1000cmx1000cmx1000cm (with sides
1102  // going from -500cm to +500cm on each axis).
1103  //
1104  // If closegeo=kFALSE we leave the geometry open for other structures to be added.
1105 
1106  if (!IsBuilt()) {
1107  Error("CreateROOTGeometry", "gIndra has to be build first");
1108  return;
1109  }
1110  if (!GetNavigator()) {
1111  //Error("CreateROOTGeometry","No existing navigator"); return;
1112  SetNavigator(new KVRangeTableGeoNavigator(gGeoManager, KVMaterial::GetRangeTable()));
1113  GetNavigator()->SetNameCorrespondanceList("INDRA.names");
1114  }
1115 
1116  // set up shape & matrix pointers in detectors
1117  Info("CreateROOTGeometry", "Scanning geometry shapes and matrices...");
1118  KVGeoImport gimp(gGeoManager, KVMaterial::GetRangeTable(), this, kFALSE);
1119  gimp.SetNameCorrespondanceList("INDRA.names");
1120  KVEvent* evt = new KVEvent();
1121  KVNucleus* nuc = evt->AddParticle();
1122  nuc->SetZAandE(1, 1, 1);
1123  KVDetector* det;
1124  TIter next(GetDetectors());
1125  Int_t nrootgeo = 0;
1126  while ((det = (KVDetector*)next())) {
1127  nuc->SetTheta(det->GetTheta());
1128  nuc->SetPhi(det->GetPhi());
1129  gimp.SetLastDetector(0);
1130  gimp.PropagateEvent(evt);
1131  if (!(det->GetActiveLayerShape() && det->GetActiveLayerMatrix())) {
1132  Info("CreateROOTGeometry", "Volume checking for %s", det->GetName());
1133  Double_t theta0 = det->GetTheta();
1134  Double_t phi0 = det->GetPhi();
1135  for (Double_t TH = theta0 - 0.5; TH <= theta0 + 0.5; TH += 0.1) {
1136  for (Double_t PH = phi0 - 10; PH <= phi0 + 10; PH += 1) {
1137  nuc->SetTheta(TH);
1138  nuc->SetPhi(PH);
1139  gimp.SetLastDetector(0);
1140  gimp.PropagateEvent(evt);
1141  if (det->GetActiveLayerShape() && det->GetActiveLayerMatrix()) break;
1142  }
1143  if (det->GetActiveLayerShape() && det->GetActiveLayerMatrix()) break;
1144  }
1145  }
1146  if (!(det->GetActiveLayerShape() && det->GetActiveLayerMatrix())) {
1147  Info("CreateROOTGeometry", "Volume checking failed for : %s", det->GetName());
1148  }
1149  // check etalon trajectories
1150  if (det->GetActiveLayerShape() && det->GetActiveLayerMatrix()
1151  && (det->IsCalled("CSI_1002") || det->IsCalled("CSI_1102")
1152  || det->IsCalled("CSI_1202") || det->IsCalled("CSI_1304")
1153  || det->IsCalled("CSI_1403") || det->IsCalled("CSI_1503")
1154  || det->IsCalled("CSI_1602") || det->IsCalled("CSI_1702"))
1155  && det->GetNode()->GetNDetsInFront() < 2) {
1156  Info("CreateROOTGeometry", "Trajectory checking for %s", det->GetName());
1157  Double_t theta0 = det->GetTheta();
1158  Double_t phi0 = det->GetPhi();
1159  for (Double_t TH = theta0 - 0.5; TH <= theta0 + 0.5; TH += 0.1) {
1160  for (Double_t PH = phi0 - 10; PH <= phi0 + 10; PH += 1) {
1161  nuc->SetTheta(TH);
1162  nuc->SetPhi(PH);
1163  gimp.SetLastDetector(0);
1164  gimp.PropagateEvent(evt);
1165  if (det->GetNode()->GetNDetsInFront() == 2) break;
1166  }
1167  if (det->GetNode()->GetNDetsInFront() == 2) break;
1168  }
1169  }
1170  nrootgeo += (det->GetActiveLayerShape() && det->GetActiveLayerMatrix());
1171  }
1172  delete evt;
1173 
1174  Info("CreateROOTGeometry", "ROOT geometry initialised for %d/%d detectors", nrootgeo, GetDetectors()->GetEntries());
1175 
1176  // Set up trajectories
1177  TIter it(GetDetectors());
1178  KVDetector* d;
1179  while ((d = (KVDetector*)it())) d->GetNode()->RehashLists();// make sure detector nodes are correct
1180  AssociateTrajectoriesAndNodes();
1181  FillTrajectoryIDTelescopeLists();
1182  CalculateReconstructionTrajectories();
1183  GetNavigator()->AbsorbDetectorPaths(&gimp);
1184 }

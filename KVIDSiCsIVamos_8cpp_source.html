113 Bool_t KVIDSiCsIVamos::Identify(KVIdentificationResult* idr, Double_t csi, Double_t si)
114 {
115  //Particle identification and code setting using identification grids.
116 
117 
118  //perform identification in Si - CsI map, in Vamos FP
119 
120  idr->SetIDType(GetType());
121  idr->IDattempted = kTRUE;
122 
123  KVIDGrid* theIdentifyingGrid = 0;
124 
125  fgrid->Identify(csi, si, idr);
126  theIdentifyingGrid = (KVIDGrid*)fgrid;
127 
128  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
129  // only if the final quality code is kICODE8 do we consider that it is
130  // worthwhile looking elsewhere. In all other cases, the particle has been
131  // "identified", even if we still don't know its Z and/or A (in this case
132  // we consider that we have established that they are unknowable).
133  return kFALSE;
134  }
135 
136  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
137  // if the final quality code is kICODE7 (above last line in grid) then the estimated
138  // Z is only a minimum value (Zmin)
139  idr->IDcode = 5;
140  return kTRUE;
141  }
142 
143 
144  // set general ID code SiLi-CsI
145  idr->IDcode = 3;
146  return kTRUE;
147 }

145 KVTGID* KVRTGIDManager::GetTGID(const Char_t*,
146  const Char_t* id_type,
147  const Char_t* grid_type)
148 {
149  //Overrides the same method of KVTGIDManager. This method
150  //Retrieve the identification object using:
151  // id_type = type of identification ("Z", "A", "A_for_Z=3", etc.)
152  // grid_type = type of identification grid ("GG", "PG1", etc.)
153  // but the name of ID telescope is not used anymore because identification
154  // object can be associated to several ID telescopes.
155  // If any object is found then it returns the first:
156  // - KVTGIDZA object if id_type contains "A"
157  // - KVTGIDZ object else.
158 
159  Char_t cname[9] = "KVTGIDZ";
160  TString buff = id_type;
161 
162  if (buff.Contains("A")) strcpy(cname, "KVTGIDZA");
163 
164  buff.Form("_%s_%s", id_type, grid_type);
165 
166  TIter next(&fIDList);
167  KVTGID* tgid = NULL;
168  KVTGID* tgid2 = NULL;
169  TString name;
170 
171  while ((tgid = (KVTGID*)next())) {
172  name = tgid->GetName();
173  if (name.Contains(buff)) return tgid;
174  if (!tgid2 && !strcmp(tgid->ClassName(), cname)) tgid2 = tgid;
175  }
176 
177  return tgid2;
178 }

270 Bool_t KVRTGIDManager::ReadAsciiFile(const Char_t* filename)
271 {
272 
273  //Read file, create KVTGID fits corresponding to information in file.
274  //Comment line beginning by '#' is skipped.
275 
276  ifstream fitfile(filename);
277  if (!fitfile.good()) {
278  Error("KVRTGIDManager::ReadAsciiFile", "File %s cannot be opened", filename);
279  return kFALSE;
280  }
281  KVString s, cname;
282  while (fitfile.good()) {
283  //read a line
284  s.ReadLine(fitfile);
285 
286  if (s.BeginsWith("#")) continue;
287  if (s.BeginsWith("++")) {
288  //New fit
289  //Get name of class by stripping off the '+' at the start of the line
290  s.Remove(0, 2);
291  cname = s;
292  cname.Remove(s.Index("::"));
293  TClass* cl = TClass::GetClass(cname.Data());
294  if (cl && cl->InheritsFrom("KVTGID")) {
295 
296  s.Remove(0, s.Index("::") + 2);
297 
298  // Make new identification function
299  KVTGID* fit = NULL;
300  fit = KVTGID::ReadFromAsciiFile(s.Data(), fitfile);
301  AddTGIDToGlobalList(fit);
302 
303  // when mass identification is possible, make a copy
304  // for Z identification
305  if (!fit->GetZorA()) {
306  KVTGIDZ* fitz = new KVTGIDZ(*fit);
307  fitz->SetTitle(Form("COPY from %p", fit));
308  AddTGIDToGlobalList(fitz);
309  }
310  }
311  }
312  }
313  fitfile.close();
314  return kTRUE;
315 }

367 void KVRTGIDManager::BuildGridForAllTGID(const Char_t* idtype, Double_t xmin, Double_t xmax, Int_t ID_min, Int_t ID_max, Int_t npoints, Bool_t logscale)
368 {
369  // Build a grid (KVTGIDGrid) for all the identification functions
370  // of the global list. The new grids are automatically loaded in
371  // gIDGridManager and are visible in the Grid Manager GUI.
372  // If a function is already associated to a grid then a new grid
373  // is not built. No grid is built for copies of KVTGID's made in
374  // the method ReadAsciiFile(...).
375  //
376  // Inputs: idtype - type of the identification for which the
377  // grids will be built (CI-SI, SI-CSI, CI-CSI,
378  // SI75-SILI, ...). By default, all grids are
379  // built
380  // xmin
381  // xmax
382  // ID_min
383  // ID_max
384  // npoints
385  // logscale - see KVTGIDGrid::Generate(...)
386 
387  if (!fIDGlobalList) return;
388 
389  // First make a sublist of TGID found in object inheriting
390  // from KVTGIDGrid in gIDGridManager
391  TList tgid_list;
392  GetTGIDfromIDGridManager(&tgid_list);
393  KVIDGridManager* gm = gIDGridManager;
394  KVList* grid_list = NULL;
395  if (gm) {
396  grid_list = gm->GetGrids();
397  grid_list->Disconnect("Modified()", gm, "Modified()");
398  }
399  // If the TGID of the global list is not in the sublist then
400  // build grid
401  TIter next(fIDGlobalList);
402  Bool_t IDtypeOK = strcmp(idtype, "");
403  KVTGID* tgid = NULL;
404  while ((tgid = (KVTGID*)next())) {
405  if (tgid_list.FindObject(tgid)) continue;
406 
407  if (IDtypeOK) {
408  KVBase* idt = NULL;
409  TSeqCollection* idt_list = (TSeqCollection*)GetIDTelescopesForTGID(tgid);
410  if (!idt_list) continue;
411  if (!(idt = (KVBase*)idt_list->First())) continue;
412  SafeDelete(idt_list);
413  if (strcmp(idtype, idt->GetLabel())) continue;
414  }
415  // Not built grid for a KVTGID copy
416  TString tmp = tgid->GetTitle();
417  if (tmp.Contains("COPY")) {
418  tmp.Remove(0, tmp.Index("0x"));
419  KVTGID* tmp_tgid = reinterpret_cast<KVTGID*>((Int_t)tmp.Atof());
420  Warning("KVRTGIDManager::BuildGridForAllTGID", "No grid built for %s (%s, %p) because it is a copy of %s (%s, %p)"
421  , tgid->GetName(), tgid->ClassName(), tgid
422  , tmp_tgid->GetName(), tmp_tgid->ClassName(), tmp_tgid);
423  continue;
424  }
425  KVTGIDGrid* grid = new KVTGIDGrid(tgid);
426  grid->SetOnlyZId((Bool_t)tgid->GetZorA());
427  if (tgid->GetZorA()) grid->SetMassFormula(tgid->GetMassFormula());
428  grid->Generate(xmax, xmin, ID_min, ID_max, npoints, logscale);
429  Info("KVRTGIDManager::BuildGridForAllTGID", "grid built from its TGID function %s (%s, %p)"
430  , tgid->GetName(), tgid->ClassName(), tgid);
431  }
432  if (grid_list) grid_list->Connect("Modified()", "KVIDGridManager", gm, "Modified()");
433  gm->Modified();
434 }

456 void KVRTGIDManager::SetIDFuncInTelescopes(UInt_t run)
457 {
458  // For each identification function of the global list which is valid for this run, we add it in the associated telescope which
459  // have to inherits from KVTGIDManager and from a KVIDTelescope.
460 
461  TIter next(fIDGlobalList);
462  KVTGID* tgid = NULL;
463 
464  while ((tgid = (KVTGID*)next())) {
465 
466  if (!tgid->IsValidForRun(run)) continue;
467 
468  // this list have to be deleted after use
469  TCollection* lidtel = GetIDTelescopesForTGID(tgid);
470  TIter nextidt(lidtel);
471  TObject* idt = NULL;
472  KVRTGIDManager* tgidm = NULL;
473 
474  while ((idt = nextidt())) {
475  if (!idt->InheritsFrom("KVRTGIDManager")) {
476  Error("KVRTGIDManager::SetIDFuncInTelescopes", "The IDtelescope %s does not inherit from KVRTGIDManager", idt->GetName());
477  continue;
478  }
479  tgidm = (KVRTGIDManager*)idt->IsA()->DynamicCast(KVRTGIDManager::Class(), idt);
480  tgidm->SetTGID(tgid);
481  }
482  delete lidtel;
483  }
484 }

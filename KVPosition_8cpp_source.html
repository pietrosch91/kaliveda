39 //KVPosition
40 //
41 //Base class used for handling geometry in a multidetector array.
42 //
43 //Each KVPosition object represents an element of solid angle, with a domain in
44 //polar angle (theta) and azimuthal angle (phi). All methods necessary to
45 //inquire about relative positions of such elements (overlaps, inclusion, etc.)
46 //are provided.
47 //
48 //The coordinate system has the target position for its centre. The beam
49 //direction defines the positive z axis, with respect to which polar angles are
50 //measured. The +ve x-axis is taken to be vertical, and azimuthal angles
51 //increase when going clockwise from the +ve x-axis, looking towards the +ve
52 //z direction:
53 //
54 // phi=0
55 // (x) beam |
56 // |
57 // |
58 // +------- phi=90
59 //
60 //All angles are in degrees.
61 //
62 //Polar angles (theta) vary between 0 and 180 degrees
63 //Azimuthal angles (phi) vary between 0 and 359.999... degrees.
64 //
65 //An element is defined by theta-min/max and phi-min/max:
66 //Theta-min is the polar angle of the edge closest the beam (+ve z) axis
67 //Theta-max is the polar angle of the edge furthest from the beam (+ve z) axis
68 //Phi-min is the phi angle of the edge which is the most anticlockwise when
69 // looking as in the figure shown above
70 //Phi-max is the phi angle of the edge which is the most clockwise when
71 // looking as in the figure shown above
72 //
73 //The above definitions mean that phi-min is not necessarily smaller than
74 //phi-max: we redefine "smaller" for phi angles to mean "more anticlockwise than"
75 //or "less clockwise than". Some examples:
76 // an element with phi-min = 20 and phi-max = 40 has an azimuthal width of 20
77 // degrees;
78 // an element with phi-min = 40 and phi-max = 20 has an azimuthal width of 340
79 // degrees;
80 // an element with phi-min = 350 and phi-max = 10 has an azimuthal width of
81 // 20 degrees.
82 //
83 //Examples of use of KVPosition objects:
84 //
85 //Generating a random unit vector within a given angular range
86 //
87 // KVPosition pos; |
88 // pos.SetPolarMinMax(20, 30); | -- or: KVPosition pos(20, 30, 0, 90);
89 // pos.SetAzimuthalMinMax(0, 90); |
90 // TVector3 dir1 = pos.GetDirection();//unit vector in direction theta=25, phi=45
91 // TVector3 dir2 = pos.GetRandomDirection();//isotropic direction within angular limits of pos
92 // TVector3 dir3 = pos.GetRandomDirection("random");//random direction
93 //

151 void KVPosition::SetPolarAngle(Double_t th)
152 {
153  //Sets the polar angle corresponding to the centre of this telescope/solid angle element/etc.
154  //If the polar width has been set already (KVPosition::SetPolarWidth),
155  //the limits theta-min and theta-max are calculated.
156 
157  fTheta = th;
158  if (fTheta_min == -180.) { // polar width of detector already set, calculate theta_min
159  // and theta_max
160  fTheta_min = fTheta - .5 * (fTheta_max + 180.);
161  fTheta_max = 2. * fTheta - fTheta_min;
162  }

172 void KVPosition::SetAzimuthalAngle(Double_t ph)
173 {
174  //Sets the azimuthal angle corresponding to the centre of this telescope/solid angle element/etc.
175  //If the azimuthal width has been set already (KVPosition::SetAzimuthalWidth),
176  //the limits phi-min and phi-max are calculated.
177 
178  fPhi = ph;
179  //make sure phi between 0 and 360
180  if (fPhi < 0.)
181  fPhi += 360.;
182  if (fPhi_min == -180.) { // azimuthal width of detector already set, calculate phi_min
183  // and phi_max
184  fPhi_min = fPhi - .5 * (fPhi_max + 180.);
185  fPhi_max = 2. * fPhi - fPhi_min;
186  //make sure phimin and phimax are between 0 and 360
187  if (fPhi_min < 0.)
188  fPhi_min += 360.;
189  if (fPhi_max >= 360.)
190  fPhi_max -= 360.;
191  }

295 TVector3 KVPosition::GetRandomDirection(Option_t* t)
296 {
297  // Returns a unit vector in a random direction corresponding to this detector.
298  // Depending on the optional option string, the direction is either drawn at
299  // "random" among the corresponding angles, or "isotropic".
300  // By default, the direction is "isotropic".
301  //
302  // * ROOT Geometry *
303  // Direction corresponds to a random position on the detector's entrance window.
304  // The "isotropic" option has no effect.
305 
306  if (ROOTGeo()) {
307  // * ROOT Geometry *
308  TVector3 r = GetRandomPoint();
309  return r.Unit();
310  }
311 
312  Double_t dtor = TMath::DegToRad();
313  Double_t th, ph;
314  GetRandomAngles(th, ph, t);
315  TVector3 dummy;
316  dummy.SetMagThetaPhi(1.0, th * dtor, ph * dtor); // a unit vector
317 
318  return dummy;
319 }

333 void KVPosition::GetRandomAngles(Double_t& th, Double_t& ph, Option_t* t)
334 {
335  // Set th and ph to random values between the max and min limits defining the
336  // solid angle element.
337  // Depending on the optional option string, the direction is either drawn at
338  // "random" among the corresponding angles, or "isotropic".
339  // By default, the direction is "isotropic".
340  //
341  // * ROOT Geometry *
342  // th and ph correspond to a random position on the detector's entrance window.
343  // The "isotropic" option has no effect.
344 
345  Double_t dtor = TMath::DegToRad();
346 
347  if (ROOTGeo()) {
348  // * ROOT Geometry *
349  TVector3 r = GetRandomPoint();
350  th = r.Theta() / dtor;
351  ph = r.Phi() / dtor;
352  if (ph < 0) ph += 360.;
353  return;
354  }
355 
356  Double_t phmin = fPhi_min; //phimin in radians
357  Double_t phmax = fPhi_max; //phimax in radians
358  if (phmax < phmin)
359  phmax += 360;
360 
361  Double_t dphi = phmax - phmin;
362  Double_t thmin, thmax, dtheta;
363 
364  if (!strcmp(t, "random")) {
365  thmin = fTheta_min * dtor;
366  thmax = fTheta_max * dtor;
367  dtheta = thmax - thmin;
368  th = gRandom->Uniform(dtheta) + thmin;
369  } else {
370  thmin = TMath::Cos(fTheta_min * dtor);
371  thmax = TMath::Cos(fTheta_max * dtor);
372  dtheta = thmin - thmax;
373  th = TMath::ACos(gRandom->Uniform(dtheta) + thmax);
374  }
375  ph = gRandom->Uniform(dphi) + phmin;
376  if (ph > 360)
377  ph -= 360;
378  th /= dtor;
379 
380 }

388 Bool_t KVPosition::IsInPhiRange(const Double_t phi)
389 {
390  //kTRUE if given angle phi is within the azimuthal range of this solid
391  //angle element
392 
393  ROOT_GEO_DO_NOT_USE("IsInPhiRange", kFALSE)
394 
395  Double_t phimintest = fPhi_min;
396  Double_t phimaxtest = fPhi_max;
397  if (phimintest > phimaxtest) {
398  phimaxtest += 360.;
399  }
400  if (phi >= phimintest && phi <= phimaxtest) {
401  return kTRUE;
402  }
403  if ((phi + 360.) >= phimintest && (phi + 360.) <= phimaxtest) {
404  return kTRUE;
405  }
406  return kFALSE;
407 }

460 Bool_t KVPosition::IsOverlappingWith(KVPosition* other)
461 {
462  //kTRUE if there is at least partial overlap between two solid angle elements
463 
464  ROOT_GEO_DO_NOT_USE("IsOverlappingWith", kFALSE)
465  return (
466  //overlapping corners - case 1
467  (IsInPolarRange(other->GetThetaMin())
468  || IsInPolarRange(other->GetThetaMax()))
469  && (IsInPhiRange(other->GetPhiMin())
470  || IsInPhiRange(other->GetPhiMax()))
471  ) || (
472  //overlapping corners - case 2
473  (other->IsInPolarRange(GetThetaMin())
474  || other->IsInPolarRange(GetThetaMax()))
475  && (other->IsInPhiRange(GetPhiMin())
476  || other->IsInPhiRange(GetPhiMax()))
477  ) || (
478  //case where this is phi-contained by the other, but the other is theta-contained by this
479  (other->IsInPolarRange(GetThetaMin())
480  || other->IsInPolarRange(GetThetaMax()))
481  && (IsInPhiRange(other->GetPhiMin())
482  || IsInPhiRange(other->GetPhiMax()))
483  ) || (
484  //case where this is phi-contained by the other, but the other is theta-contained by this
485  (IsInPolarRange(other->GetThetaMin())
486  || IsInPolarRange(other->GetThetaMax()))
487  && (other->IsInPhiRange(GetPhiMin())
488  || other->IsInPhiRange(GetPhiMax()))
489  );

548 void KVPosition::GetCornerCoordinates(TVector3* corners, Double_t depth)
549 {
550  // Fill the array (TVector3 corner[4]) with the coordinates of the 4 'corners' of the solid angle element.
551  //
552  // These 'corners' are the points of intersection between the plane defined by the normal
553  // to the centre of the solid angle (direction: theta,phi), at a distance fDistance [cm] from the
554  // origin, and the four lines starting at the origin with directions (thetamin,phimin),
555  // (thetamax,phimin), (thetamax,phimax), (thetamin,phimax).
556  //
557  // If optional argument 'depth' [cm] is given, the coordinates are calculated for the plane
558  // situated at distance (fDistance+depth) from the origin.
559  //
560  // The order of the 4 corners is as follows:
561  // corners[3] : theta-min, phi-min
562  // corners[2] : theta-max, phi-min
563  // corners[1] : theta-max, phi-max
564  // corners[0] : theta-min, phi-max
565  //
566  // Coordinates are in CENTIMETRES
567 
568  ROOT_GEO_DO_NOT_USE("GetCornerCoordinates",)
569 
570  // calculate unit vector normal to solid angle
571  Double_t pthR = GetTheta() * TMath::DegToRad();
572  Double_t pphR = GetPhi() * TMath::DegToRad();
573  TVector3 normal_to_plane(sin(pthR)*cos(pphR), sin(pthR)*sin(pphR), cos(pthR));
574 
575  // the four directions/lines
576  corners[3].SetMagThetaPhi(1.0, TMath::DegToRad()*GetThetaMin(), TMath::DegToRad()*GetPhiMin());
577  corners[2].SetMagThetaPhi(1.0, TMath::DegToRad()*GetThetaMax(), TMath::DegToRad()*GetPhiMin());
578  corners[1].SetMagThetaPhi(1.0, TMath::DegToRad()*GetThetaMax(), TMath::DegToRad()*GetPhiMax());
579  corners[0].SetMagThetaPhi(1.0, TMath::DegToRad()*GetThetaMin(), TMath::DegToRad()*GetPhiMax());
580 
581  // calculate intersection points
582  for (int i = 0; i < 4; i++) {
583  Double_t t = corners[i] * normal_to_plane;
584  if (t <= 0.0) corners[i].SetXYZ(0, 0, 0);
585  else corners[i] *= (fDistance + depth) / t;
586  }
587 }

596 void KVPosition::GetCornerCoordinatesInOwnFrame(TVector3* corners, Double_t depth)
597 {
598  // Like GetCornerCoordinates(), except that the coordinates correspond to a reference frame
599  // in which the +ve z-axis goes through the centre of the solid angle
600 
601  ROOT_GEO_DO_NOT_USE("GetCornerCoordinatesInOwnFrame",)
602  GetCornerCoordinates(corners, depth);
603  TRotation rot_to_frame;
604  rot_to_frame.SetYEulerAngles(-GetPhi()*TMath::DegToRad(), -GetTheta()*TMath::DegToRad(), 0.);
605  TVector3 displZ(0, 0, fDistance + depth);
606  for (int i = 0; i < 4; i++) {
607  corners[i] = rot_to_frame * corners[i] - displZ;
608  }
609 }

619 Double_t KVPosition::GetSolidAngle(void)
620 {
621  // Return values of the solid angle (in msr) seen by the geometric ensemble
622  // For simple geometries defined by theta_min/max etc., this is exact.
623  // For ROOT geometries we calculate the area of the entrance window and divide
624  // it by the square of the distance to the detector.
625 
626  if (ROOTGeo()) {
627  return fSolidAngle;
628  }
629 
630  return (-1.*cos(GetThetaMax() * TMath::DegToRad()) +
631  cos(GetThetaMin() * TMath::DegToRad()))
632  * (GetAzimuthalWidth() * TMath::DegToRad()) * 1.e3;
633 
634 }

655 Double_t KVPosition::GetAzimuthalWidth(Double_t phmin, Double_t phmax) const
656 {
657  //Calculate azimuthal width taking phi-min as the most anticlockwise point of the
658  //element and phi-max the most clockwise.
659  //If no arguments are given, width calculated for this object
660  //Otherwise, width calculated for given phi-min and phi-max
661 
662  ROOT_GEO_DO_NOT_USE("GetAzimuthalWidth", 0.)
663 
664  if (phmin == -1.)
665  phmin = GetPhiMin();
666  if (phmax == -1.)
667  phmax = GetPhiMax();
668  if (phmin < phmax)
669  return (phmax - phmin);
670  else
671  return (phmax - phmin + 360.);
672 }

682 void KVPosition::GetWidthsFromDimension(Double_t lin_dim)
683 {
684  //Calculate azimuthal and polar widths for a square element placed at a
685  //given distance from the origin with linear dimension 'lin_dim' (in mm).
686  //SetDistance, SetTheta and SetPhi must already have been called.
687 
688  if (GetDistance() <= 0.0) {
689  Error("GetWidthsFromDimension", "Distance not set");
690  return;
691  }
692  Double_t d__th =
693  2. * TMath::RadToDeg() * TMath::ATan(lin_dim /
694  (2. * GetDistance()));
695  Double_t d__ph =
696  TMath::RadToDeg() * lin_dim / (GetDistance() * GetSinTheta());
697  SetPolarWidth(d__th);
698  SetAzimuthalWidth(d__ph);
699 }

708 TRotation KVPosition::GetRandomIsotropicRotation()
709 {
710  // Generates a rotation which, if applied to a unit vector in the Z-direction,
711  // will transform it into an isotropically-distributed vector in this
712  // angular range.
713 
714  TRotation rr2;
715  ROOT_GEO_DO_NOT_USE("GetRandomIsotropicRotation", rr2)
716  Double_t a1min, a1max, a2min, a2max, a3min, a3max;
717  a1min = 0;
718  a1max = 2 * TMath::Pi();
719  a2min = GetThetaMin() * TMath::DegToRad();
720  a2max = GetThetaMax() * TMath::DegToRad();
721  a3min = GetPhiMin() * TMath::DegToRad();
722  a3max = GetPhiMax() * TMath::DegToRad();
723  a3min += TMath::Pi() / 2.;
724  a3max += TMath::Pi() / 2.;
725  rr2.SetXEulerAngles(gRandom->Uniform(a1min, a1max),
726  TMath::ACos(gRandom->Uniform(cos(a2max), cos(a2min))),
727  gRandom->Uniform(a3min, a3max));
728  return rr2;
729 }

742 void KVPosition::SetMatrix(const TGeoHMatrix* m)
743 {
744  // * ROOT Geometry *
745  // Set the global transformation matrix for this detector
746  // If shape has been set, we set the (theta,phi) angles
747  // corresponding to the centre of the entrance window, and
748  // the distance from the target.
749 
750  if (fMatrix) delete fMatrix;
751  fMatrix = new TGeoHMatrix(*m);
752  if (ROOTGeo()) {
753  TVector3 centre = GetCentre();
754  SetTheta(centre.Theta()*TMath::RadToDeg());
755  SetPhi(centre.Phi()*TMath::RadToDeg());
756  SetDistance(centre.Mag());
757  Double_t area = GetShape()->GetFacetArea(1);
758  fSolidAngle = area / pow(GetDistance(), 2.) * 1.e3;
759  }

832 {
833  // * ROOT Geometry *
834  // Generate a vector in the world (laboratory) frame from the origin
835  // to a random point on the entrance window of the detector.
836  //
837  // It is assumed that the detector volume was defined in such a way
838  // that the entrance window corresponds to the facet in the X-Y plane
839  // placed at -dZ.
840  //
841  // NOTE: we force the use of TGeoBBox::GetPointsOnFacet.
842  // For TGeoArb8, the method has been overridden and does nothing.
843  // We use the TGeoBBox method, and then use TGeoShape::Contains
844  // to check that the point does actually correspond to the TGeoArb8.
845 
846  if (!ROOTGeo()) {
847  ::Error("KVPosition::GetRandomPointOnEntranceWindow",
848  "ROOT Geometry has not been initialised");
849  return TVector3();
850  }
851  Double_t master[3 * 50];
852  Double_t points[3 * 50];
853  const Double_t* origin = GetShape()->GetOrigin();
854  Double_t dz = GetShape()->GetDZ();
855  // This will generate a point on the (-DZ) face of the bounding box of the shape
856  Bool_t ok1 = GetShape()->TGeoBBox::GetPointsOnFacet(1, 50, points);
857  // We move the point slightly inside the volume to test if it actually corresponds
858  // to a point on the shape's facet
859  points[2] += dz / 100.;
860  // Correct for offset of centre of shape
861  for (int i = 0; i < 3; i++) points[i] += origin[i];
862  Bool_t ok2 = GetShape()->Contains(points);
863  if (!ok1) {
864  ::Error("KVPosition::GetRandomPoint",
865  "TGeoBBox::GetPointsOnFacet returns kFALSE for shape %s. Returning coordinates of centre.", GetShape()->ClassName());
866  return GetCentre();
867  }
868  Int_t np = 0;
869  if (!ok2) {
870  // try to find a point that works
871  np++;
872  while (np < 50) {
873  Double_t* npoint = points + 3 * np;
874  npoint[2] += dz / 100.;
875  // Correct for offset of centre of shape
876  for (int i = 0; i < 3; i++) npoint[i] += origin[i];
877  ok2 = GetShape()->Contains(npoint);
878  if (ok2) break;
879  np++;
880  }
881  if (!ok2) {
882  ::Error("KVPosition::GetRandomPoint",
883  "Cannot generate points for shape %s. Returning coordinates of centre.", GetShape()->ClassName());
884  return GetCentre();
885  }
886  }
887  Double_t* npoint = points + 3 * np;
888  npoint[2] -= dz / 100.;
889  GetMatrix()->LocalToMaster(npoint, master);
890  return TVector3(master);
891 }

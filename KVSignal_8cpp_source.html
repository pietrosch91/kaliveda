695 Bool_t KVSignal::ComputeMeanAndSigma(Int_t start, Int_t stop, Double_t& mean, Double_t& sigma)
696 //----------------------------------
697 
698 
706 {
707  //compute mean value and sigma values
708  //between "start" point included and "stop" point excluded
709  //for example :
710  // ComputeMeanAndSigma(0,50,mean,sigma)
711  // compute values for the first 50 points
712  //
713  Int_t np = 0;
714  Double_t xx;
715  mean = 0;
716  Double_t mean2 = 0;
717  if (stop > GetN()) {
718  Warning("ComputeMeanAndSigma",
719  "stop position greater than number of samples %d/%d, set stop to %d", GetN(), stop, GetN());
720  stop = GetN();
721  }
722  if (start < 0) {
723  Warning("ComputeMeanAndSigma",
724  "start position unrealistic %d, set start to 0", start);
725  start = 0;
726  }
727 
728  for (Int_t ii = start; ii < stop; ii += 1) {
729  xx = fAdc.At(ii);
730  mean += xx;
731  mean2 += xx * xx;
732  np += 1;
733  }
734 
735  if (np == 0) {
736  Error("ComputeMeanAndSigma", "values cannot be computed with 0 sample");
737  return kFALSE;
738  }
739  mean /= np;
740  mean2 /= np;
741  sigma = TMath::Sqrt(mean2 - mean * mean);
742  return kTRUE;
743 }

149 void KVTGIDGrid::Generate(Double_t xmax, Double_t xmin, Int_t ID_min, Int_t ID_max, Int_t npoints, Bool_t logscale)
150 {
151  //(Re)generate ID grid from the functional using current values of parameters.
152  //
153  // xmin, xmax - min and max values of 'x' coordinate used in corresponding
154  // identification map
155  // ID_min, ID_max - min and max 'ID' of lines in grid
156  // npoints - number of points in each line
157  //
158  //DEFAULT ARGUMENT VALUES:
159  //=========================
160  // xmin = 0.
161  // If the grid gets "ugly" for small x, you might try setting xmin = fitted X-pedestal
162  // This is the value of parameter 7 in most cases....
163  //
164  // ID_min, ID_max = min and max ID of lines on which functional was fitted
165  // npoints = 100
166  //
167  // if logscale=kTRUE (default is kFALSE) lines are generated with more points at
168  // the beginning of the lines than at the end.
169 
170  if (!GetTGID()) {
171  Error("Generate", "No parameter stored to initialize a functional !");
172  return;
173  }
174 
175  ID_min = (ID_min ? ID_min : (Int_t) fTGID->GetIDmin());
176  ID_max = (ID_max ? ID_max : (Int_t) fTGID->GetIDmax());
177 
178  // clear existing lines
179  Bool_t was_drawn = kFALSE;
180  TVirtualPad* last_pad = fPad;
181  if (fPad) {
182  UnDraw();
183  was_drawn = kTRUE;
184  }
185 
186  // delete all previous identification lines
187  fIdentifiers->Delete();
188  fXmin = fYmin = fXmax = fYmax = 0;
189 
190  for (Int_t ID = ID_min; ID <= ID_max; ID++) {
191  fTGID->AddLineToGrid(this, ID, npoints, xmin, xmax, logscale);
192  }
193  if (was_drawn) {
194  last_pad->cd();
195  Draw();
196  }
197  FindAxisLimits();
198 }

21 // This class returns an index determined via a linear interpolation between

26 // returned value is
27 //
28 // (x-xmin)*nb_cells
29 // 1+(Int_t)-------------------
30 // xmax-xmin
31 //
32 // if x<xmin it returns 1
33 // if x>xmax it returns nb_cells
34 //
35 // virtual Int_t GetNumCase(void *,...) return -1,not to use.
36 //
37 // BEWARE! : the index value ranges between 1 and nb_cells.
38 //
39 // The Xmin and Xmax values can be adjusted with the methods
40 // virtual void SetXmin(Double_t x)

216 void KVTrieurLin::SetNomsCases(void)
217 {
218 //
219 // On affecte les noms aux cases
220 //
221  TString nomt;
222  if (noms_cases) {
223  Double_t xpas = (xmax - xmin) / nb_cases;
224  for (Int_t i = 0; i < nb_cases; i++) {
225  if (i == 0) {
226  nomt.Form("%s < %f", nom_var, xmin + (i + 1) * xpas);
227  } else if (i == nb_cases - 1) {
228  nomt.Form("%f #leq %s", xmin + i * xpas, nom_var);
229  } else {
230  nomt.Form("%f #leq %s < %f", xmin + i * xpas, nom_var,
231  xmin + (i + 1) * xpas);
232  }
233  TNamed* nom = (TNamed*) noms_cases->At(i);
234  nom->SetTitle(nomt);
235  }
236  }
237 }

417 Int_t KVTrieurLin::GetNumCase(Double_t x)
418 {
419 //
420 // On retourne le numero de case
421 //
422  Int_t i = 0;
423  if (xmax > xmin) {
424  i = (Int_t)((x - xmin) / (xmax - xmin) * nb_cases) + 1;
425  if (i < 1)
426  i = 1;
427  if (i > nb_cases)
428  i = nb_cases;
429  } else {
430  Warning("GetNumCase(Double_t x)", "Xmin >= Xmax !");
431  i = 0;
432  }
433  return i;
434 }

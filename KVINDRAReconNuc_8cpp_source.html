45 //KVINDRAReconNuc
46 //
47 //Nuclei reconstructed from data measured in the INDRA array.
48 //Most useful methods are already defined in parent classes KVReconstructedNucleus,
49 //KVNucleus and KVParticle. Here we add a few useful INDRA-specific methods :
50 //
51 // GetRingNumber(), GetModuleNumber()
52 // GetTimeMarker() - returns time-marker of detector in which the particle stopped
53 //
54 // StoppedInChIo(), StoppedInSi(), StoppedInCsI() - information on the detector in
55 // which the particle stopped
56 // ** Note1: to access the detector in question, use GetStoppingDetector() (see KVReconstructedNucleus)
57 // ** Note2: for a general test of the type of detector in which the particle stopped,
58 // you can do one of the following e.g. to test if it stopped in an
59 // ionisation chamber (assuming a particle pointer 'part'):
60 //
61 // if( !strcmp(part->GetStoppingDetector()->GetType(), "CI") ) { ... }
62 // //test the type of the detector - see INDRA detector classes for the different types
63 // if( part->GetStoppingDetector()->InheritsFrom("KVChIo") ) { ... }
64 // //test the inheritance of the class of the stopping detector
65 //
66 // GetChIo(), GetSi(), GetCsI() - return pointers to the detectors through which the particle passed
67 // GetEnergyChIo(), GetEnergySi(), GetEnergyCsI() - return the calculated contribution of each detector to the
68 // particle's energy
69 //
70 //Identification/Calibration status and codes
71 //-------------------------------------------
72 //
73 //Identified particles have IsIdentified()=kTRUE.
74 //Calibrated particles have IsCalibrated()=kTRUE.
75 //
76 //The KVINDRACodes object fCodes is used to store information on the identification and
77 //calibration of the particle. You can access this object using GetCodes() and then use the
78 //methods of KVINDRACodes/KVINDRACodeMask to obtain the information.
79 //
80 //For example, you can obtain the VEDA identification code of a particle 'part' using
81 //
82 // part.GetCodes().GetVedaIDCode() [ = 0, 1, 2, etc.]
83 //
84 //Information on whether the particle's mass was measured :
85 //
86 // part.IsAMeasured() [ = kTRUE or kFALSE]
87 //
88 //Information on whether the particle's charge was measured :
89 //
90 // part.IsZMeasured() [ = kTRUE or kFALSE]
91 //
92 //You can also access information on the status codes returned
93 //by the different identification telescopes used to identify the particle:
94 //see GetIDSubCode() and GetIDSubCodeString().
95 //
96 //Masses of reconstructed nuclei
97 //------------------------------
98 //
99 //When the nucleus' A is not measured, we estimate it from the identified Z.
100 //By default the mass formula used is that of R.J. Charity (see KVNucleus::GetAFromZ).
101 // ** Note: for data previous to the 5th campaign converted to the KaliVeda format
102 // ** from old DSTs, we keep the masses from the Veda programme, corresponding to
103 // ** KVNucleus mass option kVedaMass.
104 //IN ALL CASES THE RETURNED VALUE OF GetA() IS POSITIVE
105 //

587

589  if (strcmp(id_tel_type, "")) ir = GetIdentificationResult(id_tel_type);
590  else ir = GetIdentificationResult(GetIdentifyingTelescope());
591  if (!ir) return -1;
592  return ir->IDquality;

653 Bool_t KVINDRAReconNuc::CoherencyChIoSiCsI(KVIdentificationResult theID)
654 {
655  // we check that the ChIo contribution is sane:
656  // if no other particles hit this group, the Z given by the ChIoSi
657  // must be <= the Z found from Si-CsI or CsI-RL identification
658  //
659  // in this case the measured energy loss of the ChIo can be solely attributed to this particle
660  // and we return kTRUE;
661 
662  // ChIo was hit by more than one particle in group
663  if (GetChIo() && GetChIo()->GetNHits() > 1) {
664  return kFALSE;
665  }
666 
667  KVIdentificationResult* IDchiosi = GetIdentificationResult(3);
668  if (!IDchiosi->IDattempted) {
669  // no ChIo-Si identification ? assume coherency ?
670  return kTRUE;
671  }
672 
673  // if we have a successful ChIo-Si id with a Z > Z given by CsI or Si-CsI id,
674  // then we consider that there is a pile-up in the ChIo
675  // note that we consider Z_ChIoSi = 1 to mean the ChIo is in the pedestal i.e. nothing seen
676  // we also require the chio-si identification to be above the punch-through line
677  if (IDchiosi->IDOK && IDchiosi->Z > 1 && IDchiosi->IDquality != KVIDGChIoSi::k_BelowPunchThrough && IDchiosi->Z > theID.Z) {
678  return kFALSE;
679  }
680 
681  return kTRUE;
682 }

700 Bool_t KVINDRAReconNuc::CoherencySiCsI(KVIdentificationResult& theID)
701 {
702  // Called by Identify() for particles stopping in CsI detectors on rings 1-9,
703  // which have a Silicon detector just in front of them.
704  //
705  // coherent = kFALSE if CsI-R/L and Si-CsI identifications are not coherent.
706  // this is a warning, the CsI identification is kept, either the Si signal
707  // was not good (particle hitting dead zone), or it actually corresponds
708  // to two particles reaching the CsI at the same time
709  // pileup = kTRUE means that the particle identified in CsI-R/L is correct,
710  // and there is probably a second particle which stopped in the silicon
711  // detector at the same time, to be identified in ChIo-Si after
712  // subtraction of the Silicon contribution
713 
714  KVIdentificationResult* IDcsi = GetIdentificationResult(1);
715  KVIdentificationResult* IDsicsi = GetIdentificationResult(2);
716  fCoherent = kTRUE;
717  fPileup = kFALSE;
718 
719  // Unsuccessful/no CsI id attempt with successful Si-CsI id ?
720  // Then use Si-CsI identification result
721  if (IDsicsi->IDOK && !IDcsi->IDOK) {
722  theID = *IDsicsi;
723  return kTRUE;
724  }
725 
726  // check coherency of CsI-R/L and Si-CsI identifications
727  if (IDcsi->IDOK) {
728  // gammas
729  if (IDcsi->IDcode == kIDCode0) {
730  theID = *IDcsi;
731  return kTRUE;
732  }
733  // Neutrons have no energy loss in Si detector (pedestal), have a reaction in the CsI and create a Z=1
734  // or Z=2 which is identified in CsI R-L, while they show up in Si-CsI maps as a horizontal
735  // band around the Si pedestal for low energies (energies where proton dE is significantly larger than
736  // the pedestal).
737  if ((IDcsi->Z == 1 || IDcsi->Z == 2) && GetSi()) {
738  if (IDsicsi->deltaEpedestal == KVIdentificationResult::deltaEpedestal_UNKNOWN) {
739  // no explicit treatment of 'neutron-like' particles with a cut in Si-CsI id grid
740  // First we check that we are in the domain where proton dE can be distinguished from pedestal.
741  // If so, if the measured dE is below [ped + factor*(dE_exp - ped)], then we label the particle as a neutron.
742  // 'factor' depends on the Si-CsI telescope: if it has mass identification, factor=0.3; if not, factor=0.1
743  // (these empirical values correspond to 5th campaign data)
744  KVIDTelescope* idt = (KVIDTelescope*)GetIDTelescopes()->FindObjectByType(IDsicsi->GetType());
745  if (idt) {
746  Double_t ped = idt->GetPedestalY();
747  Int_t status;
748  Double_t dE_exp = idt->GetMeanDEFromID(status, 1, 1);
749  if (status == KVIDTelescope::kMeanDE_OK) { // proton/Z=1 line exists, and we are in its energy range
750  if (dE_exp > ped + 5.) { // arbitrary choice, must have expected dE at least 5 channels above pedestal
751 
752  // if Si-CsI has no isotopic identification, reduce factor
753  Double_t factor = (idt->HasMassID() ? 0.3 : 0.1);
754  if (idt->GetIDMapY() < (ped + factor * (dE_exp - ped))) {
755  theID = *IDsicsi;
756  theID.IDOK = kTRUE;
757  theID.Zident = kTRUE;
758  theID.Aident = kTRUE;
759  theID.Z = 0;
760  theID.A = 1;
761  theID.IDcode = kIDCode1; // general code for neutrons
762  return kTRUE;
763  }
764  }
765  }
766  }
767  } else if (IDsicsi->deltaEpedestal == KVIdentificationResult::deltaEpedestal_YES) {
768  // explicit treatment of 'neutron-like' particles with a cut in Si-CsI id grid
769  theID = *IDsicsi;
770  theID.IDOK = kTRUE;
771  theID.Zident = kTRUE;
772  theID.Aident = kTRUE;
773  theID.Z = 0;
774  theID.A = 1;
775  theID.IDcode = kIDCode1; // general code for neutrons
776  return kTRUE;
777  }
778  }
779 
780  // We check the coherency of the mass and charge identifications
781  // If a successful Si-CsI identification is available we check:
782  // if Si-CsI gives A & Z - must have same Z, A within +/-1 unit
783  // if Z or A smaller => incoherency : pile-up of two particles in CsI ?
784  // or bad signal from Si detector (dead zones) ?
785  // if Z or A larger, CsI identification is good,
786  // assume another particle stopped in Si (identify in ChIo-Si)
787  // if Si-CsI gives just Z - must have same Z
788  // if Z smaller => incoherency : pile-up of two particles in CsI ?
789  // or bad signal from Si detector (dead zones) ?
790  // if Z larger => CsI identification is good,
791  // assume another particle stopped in Si (identify in ChIo-Si)
792  //
793  // If CsI identification gives code KVIDGCsI::kICode1 ou KVIDGCsI::kICode3 and the
794  // Si-CsI gives the same Z but A = Acsi + 1, we use the Si-CsI identification.
795  //
796  // If CsI identification gives code KVIDGCsI::kICode2 ou KVIDGCsI::kICode3 and the
797  // Si-CsI gives the same Z but A = Acsi - 1, we use the Si-CsI identification.
798  //
799  // N.B. if CsI-R/L identification gives "8Be" (2 alpha particles) then there are two correct possibilities:
800  // 1) Si-CsI identification gives 7Li => CsI identification is correct ("8Be")
801  // 2) Si-CsI identification gives 8He => the particle is 8He (falls on same R/L line as 2*alpha)
802  // Seeing the very low statistics for 8He compared to 8Be/2alpha, we assume that if Si-CsI id
803  // gives something above 8He it is either incoherent (below 7Li) or 8Be + something else in ChIo-Si
804  // (above 7Li).
805  if (IDsicsi->IDOK) {
806  theID = *IDcsi;
807  Int_t Zref = IDcsi->Z;
808  Int_t Aref = IDcsi->A;
809  if (IDsicsi->Aident) { // Si-CsI provides mass identification
810 
811  if (IDcsi->Z == 4 && IDcsi->A == 8) {
812  // traitement special 8Be
813  // if sicsi => 7Li, it is 8Be (2alpha)
814  // if sicsi => 8He, it is 8He
815  if (IDsicsi->Z < 2 || (IDsicsi->Z == 2 && IDsicsi->A < 7)) {
816  fCoherent = kFALSE;
817  IDsicsi->SetComment("CsI-R/L & Si-CsI identifications not coherent");
818  return kTRUE;
819  } else if (IDsicsi->Z == 2 && IDsicsi->A > 6 && IDsicsi->A < 10) {
820  // accept helium-7,8,9 as 8He
821  theID = *IDsicsi;
822  return kTRUE;
823  } else if ((IDsicsi->Z == 2 && IDsicsi->A > 9) || (IDsicsi->Z == 3 && IDsicsi->A < 6)) {
824  fCoherent = kFALSE;
825  IDsicsi->SetComment("CsI-R/L & Si-CsI identifications not coherent");
826  return kTRUE;
827  } else if (IDsicsi->Z == 3 && IDsicsi->A > 5 && IDsicsi->A < 9) {
828  // accept lithium-6,7,8 as 7Li
829  return kTRUE;
830  } else if ((IDsicsi->Z == 3 && IDsicsi->A > 8) || IDsicsi->Z > 3) {
831  fPileup = kTRUE;
832  IDsicsi->SetComment("Second particle stopping in Si, identification ChIo-Si required");
833  return kTRUE;
834  }
835  }
836  // if CsI says A could be bigger and Si-CsI gives same Z and A+1, use Si-CsI
837  if ((IDsicsi->Z == Zref) && (IDsicsi->A == (Aref + 1))
838  && (IDcsi->IDquality == KVIDGCsI::kICODE1 || IDcsi->IDquality == KVIDGCsI::kICODE3)) {
839  theID = *IDsicsi;
840  return kTRUE;
841  }
842  // if CsI says A could be smaller and Si-CsI gives same Z and A-1, use Si-CsI
843  if ((IDsicsi->Z == Zref) && (IDsicsi->A == (Aref - 1))
844  && (IDcsi->IDquality == KVIDGCsI::kICODE2 || IDcsi->IDquality == KVIDGCsI::kICODE3)) {
845  theID = *IDsicsi;
846  return kTRUE;
847  }
848  // everything else - Z must be same, A +/- 1 unit
849  if (IDsicsi->Z == Zref && TMath::Abs(IDsicsi->A - Aref) < 2) {
850  return kTRUE;
851  } else if (IDsicsi->Z < Zref || (IDsicsi->Z == Zref && IDsicsi->A < (Aref - 1))) {
852  fCoherent = kFALSE;
853  IDsicsi->SetComment("CsI-R/L & Si-CsI identifications not coherent");
854  return kTRUE;
855  } else if (IDsicsi->Z > Zref || (IDsicsi->Z == Zref && IDsicsi->A > (Aref + 1))) {
856  fPileup = kTRUE;
857  IDsicsi->SetComment("Second particle stopping in Si, identification ChIo-Si required");
858  return kTRUE;
859  }
860  } else { // only Z identification from Si-CsI
861  if (IDcsi->Z == 4 && IDcsi->A == 8) {
862  // traitement special 8Be
863  // we ask for Z to be equal 3 in SiCsI, but with no mass identification
864  // we do not try for 8He identification
865  if (IDsicsi->Z < 3) {
866  fCoherent = kFALSE;
867  IDsicsi->SetComment("CsI-R/L & Si-CsI identifications not coherent");
868  return kTRUE;
869  } else if (IDsicsi->Z == 3) {
870  return kTRUE;
871  } else {
872  fPileup = kTRUE;
873  IDsicsi->SetComment("Second particle stopping in Si, identification ChIo-Si required");
874  return kTRUE;
875  }
876  }
877  // everything else
878  if (IDsicsi->Z == Zref) {
879  return kTRUE;
880  } else if (IDsicsi->Z < Zref) {
881  fCoherent = kFALSE;
882  IDsicsi->SetComment("CsI-R/L & Si-CsI identifications not coherent");
883  return kTRUE;
884  } else {
885  fPileup = kTRUE;
886 
887  IDsicsi->SetComment("Second particle stopping in Si, identification ChIo-Si required");
888  return kTRUE;
889  }
890  }
891  }
892  // in all other cases accept CsI identification
893  theID = *IDcsi;
894  return kTRUE;
895  }
896  return kFALSE;

1394 void KVINDRAReconNuc::CalibrateRings1To9()
1395 {
1396  // Special calibration for particles in rings 1 to 9
1397  // We set the energy calibration code for the particle here
1398  // kECode0 = no calibration (e.g. gammas)
1399  // kECode1 = everything OK
1400  // kECode2 = small warning, for example if energy loss in a detector is calculated
1401  // kECode15 = bad, calibration is no good
1402  // The contributions from ChIo, Si, and CsI are stored in member variables fEChIo, fESi, fECsI
1403  // If the contribution is calculated rather than measured (see below), it is stored as a negative value.
1404  // NOTE: in no case do we ever calculate an energy for uncalibrated detector using measured energy
1405  // loss in the detector placed in front (i.e. calculate ECsI from deSi, or calculate ESi
1406  // from deChIo) as this gives wildly varying (mostly false) results.
1407  //
1408  // For gammas (IDCODE=0): no calibration performed, energy will be zero, ECODE=kECode0
1409  // For neutrons (IDCODE=1): if CsI is calibrated, we use the CsI light response to calculate
1410  // the equivalent energy for a proton.
1411  // For particles stopping in the CsI detector (IDCODE=2 or 3):
1412  // - for "8Be" we use half the CsI light output to calculate 4He kinetic energy which is then doubled
1413  // (we assume two 4He of identical energy), and ECODE=kECode2 (calculated)

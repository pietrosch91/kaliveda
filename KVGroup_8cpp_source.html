171 TList* KVGroup::GetAlignedDetectors(KVDetector* det, UChar_t dir)
172 {
173  //Fill TList with all detectors aligned with "det" which are closer to the target.
174  //These are the detectors through which any particle stopping in "det" will have
175  //to pass. By default (dir=KVGroup::kBackwards) the list starts with "det" and
176  //goes towards the target. Use dir=KVGroup::kForwards to have the list in the
177  //order seen by an impinging particle.
178  //
179  //Delete TList after use.
180 
181  TList* tmp = new TList;
182 
183  while (det) {
184  tmp->Add(det);
185  KVGeoDetectorNode* node = det->GetNode();
186  KVSeqCollection* infront = node->GetDetectorsInFront();
187  if (!infront) break;
188  if (infront->GetEntries() > 1) {
189  Warning("GetAlignedDetectors",
190  "No unique solution. There are %d detectors in front of %s.",
191  infront->GetEntries(), det->GetName());
192  infront->ls();
193  }
194  det = (KVDetector*)infront->First();
195  }
196 
197  if (dir == kForwards) {
198  TIter next(tmp, kIterBackward);
199  TList* tmp2 = new TList;
200  while ((det = (KVDetector*)next())) tmp2->Add(det);
201  delete tmp;
202  tmp = tmp2;
203  }
204  return tmp;
205 }

231 Int_t KVGroup::CalculateReconstructionTrajectories()
232 {
233  // Calculate all possible (sub-)trajectories
234  // for particle reconstruction (GetReconTrajectories())
235  //
236  // Returns number of (unique) trajectories in group
237 
238  fReconTraj.Clear();
239  TIter next_traj(GetTrajectories());
240  KVGeoDNTrajectory* traj;
241  while ((traj = (KVGeoDNTrajectory*)next_traj())) { // loop over all trajectories
242 
243  traj->IterateFrom(); // from furthest-out to closest-in detector
244 
245  KVGeoDetectorNode* N;
246  while ((N = traj->GetNextNode())) {
247  fReconTraj.Add(KVGeoDNTrajectory::Factory("KVReconNucTrajectory", traj, N));
248  }
249  }
250 
251  // There may be trajectories with different names but identical titles
252  // (=physically same trajectories)
253  // We find the duplicates, delete them, and set up a map between the names of the
254  // duplicates and the name of the one remaining trajectory in the list
255  TList toRemove;
256  KVUniqueNameList unique_trajectories(kFALSE);//no replace
257  unique_trajectories.SetOwner();
258  fReconTrajMap.Clear();
259  TIter nxtRT(GetReconTrajectories());
260  KVGeoDNTrajectory* rnt;
261  while ((rnt = (KVGeoDNTrajectory*)nxtRT())) {
262 
263  TNamed* n = new TNamed(rnt->GetTitle(), rnt->GetName());
264  unique_trajectories.Add(n);
265  TNamed* orig = n;
266  if (!unique_trajectories.ObjectAdded()) {
267  orig = (TNamed*)unique_trajectories.FindObject(rnt->GetTitle());
268  toRemove.Add(rnt);
269  delete n;
270  }
271  // set up mapping from duplicate trajectory name to orginal trajectory name
272  fReconTrajMap.SetValue(rnt->GetName(), orig->GetTitle());
273 
274  }
275 
276  // now remove & delete the duplicates
277  TIter nxtDel(&toRemove);
278  while ((rnt = (KVGeoDNTrajectory*)nxtDel())) {
279  fReconTraj.Remove(rnt);
280  }
281  toRemove.Delete();
282 
283  return fReconTraj.GetEntries();
284 }

104 Bool_t KVIDQA::Identify(KVIdentificationResult* IDR, const Char_t* tof_name, Double_t realAoQ, Double_t realA)
105 {
106  //Q and A identification and code setting using identification grid,
107  // corresponding to Q and A/Q quantities calculated from the given
108  // time of flight (tof_name). First this grid is loaded in fQAGrid
109  // before identification.
110 
111  if (IDR && (realAoQ > 0.) && (realA > 0.)) {
112  //look for a grid for the given ToF
113  fQAGrid = (KVIDQAGrid*)GetListOfIDGrids()->FindObjectWithMethod(tof_name, "GetToF");
114  if (fQAGrid) {
115  Double_t Y = (fQAGrid->IsAvsAoQ() ? realA : realA / realAoQ);
116  return Identify(IDR, realAoQ, Y);
117  }
118  //else Error("Identify","no grid available for Q-A identification with %s time of flight",tof_name);
119  }
120  return kFALSE;
121 }

131 void KVIDQA::Initialize()
132 {
133  // Initialisation of telescope before identification.
134  // This method MUST be called once before any identification is attempted.
135  // Initialisation of grid is performed here.
136  // IsReadyForID() will return kTRUE if a grid is associated to this telescope for the current run.
137 
138  ResetBit(kReadyForID);
139  TIter next(fIDGrids);
140  TObject* obj = NULL;
141  while ((obj = next())) {
142  if (!obj->InheritsFrom(KVIDQAGrid::Class())) fIDGrids->Remove(obj);
143  else {
144  fQAGrid = (KVIDQAGrid*)obj;
145  fQAGrid->Initialize();
146  SetBit(kReadyForID);
147  }
148  }
149 }

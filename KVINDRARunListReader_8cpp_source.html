122 Bool_t KVINDRARunListReader::ReadRLLine()
123 {
124  //Read a line from current runlist file. If not a comment line (i.e. does not begin with 'fComment' character)
125  //then analyse it with fLineReader object.
126  //Returns kFALSE if no file is open or if file is not good for i/o
127  //Returns kFALSE if fLineReader has not been initialised
128  //Returns kFALSE if line cannot be read / end of file is reached
129  if (!fRunsStream.is_open()) {
130  Error(Form("%s::ReadRLLine", IsA()->GetName()),
131  "No file open for reading");
132  return kFALSE;
133  }
134  if (!fRunsStream.good()) {
135  Error(Form("%s::ReadRLLine", IsA()->GetName()),
136  "File %s is not good for reading", fRunsFile.Data());
137  return kFALSE;
138  }
139  fLine.ReadLine(fRunsStream);
140 
141  if (!fRunsStream.good())
142  return kFALSE;
143 
144  if (fLine.BeginsWith("Version")) {
145  //read version number in file
146  if (sscanf(fLine.Data(), "Version=%d", &fVersion) == 1) {
147  //set flag to force reading runlist with new method
148  fNewRunList = kTRUE;
149  return kFALSE;
150  }
151  } else if (!fLine.BeginsWith(fComment)) {
152  fLineReader->SetLine(fLine);
153  fLineReader->BreakLineIntoFields(fSeparator);
154  } else {
155  //comment line - reset line reader
156  fLineReader->Clear();
157  }
158  return kTRUE;
159 }

171 void KVINDRARunListReader::ReadRunList(const Char_t* t)
172 {
173  //This method will open the runlist file 't' and read its entire contents, calling user-overridable method "GoodRunLine(KVRunListLine*)"
174  //every time a line defining a new run is read from the file.
175  //If 't' is not given, we assume that the name of the run list file was passed to the constructor, and so the file should
176  //already be open in this case.
177  //Before calling this method, you should call SetRunListLineType() to define the class used to read each line, and
178  //SetRLCommentChar()/SetRLSeparatorChar() if you want to override the default comment/separator characters.
179 
180  if (!strcmp(t, "")) {
181  //no filename given - file should have been opened by OpenRLFile - check !
182  if (!fRunsStream.is_open()) {
183  Error(Form("%s::ReadRunList", IsA()->GetName()),
184  "No runlist file open for reading");
185  return;
186  }
187  } else {
188  //open new file - check OK
189  //close any previously opened file
190  CloseRLFile();
191  if (!OpenRLFile(t))
192  return; //error message will be given by OpenRLFile
193  }
194  //reset last trigger read
195  fTrigger = -1;
196  fCurrentLine = 1;
197  while (ReadRLLine()) { // read lines in file while we can
198 
199  if (fLineReader->GoodRunLine()) {
200  //call user's function
201  GoodRunLine();
202  }
203  //reset line reader
204  fLineReader->Clear();
205  fCurrentLine++;
206  }
207 
208  //close run file
209  CloseRLFile();
210 }

238 Int_t KVINDRARunListReader::GetRunListTrigger(const Char_t* field,
239  const Char_t* fmt)
240 {
241  //Get trigger multiplicity for current line.
242  //Give field name which contains trigger information and format string 'fmt' containing e.g. "M>=%d"
243  //if trigger field info is written in form "M>=4" etc. etc.
244  //The last value read is kept (fTrigger), so that, if no trigger value is given for the current line,
245  //but one was given on a previous line, we assume that the previous value is valid until a new one is given.
246  //A value "-1" means no trigger has been set for any line in the file so far.
247 
248  Int_t trig = fLineReader->GetTrigger(field, fmt);
249  if (trig < 0 && fLineReader->HasFieldValue(field))
250  cout << "...apparently on line " << GetRLLineNumber() << endl;
251  return (trig < 0 ? fTrigger : fTrigger = trig);
252 }

118  inline Double_t GetPhiI() const
119  {
120  //Returns the angle (in degree) between the X-axis and the projection of the velocity vector of the trajectory
121  //on the XY-plane (i.e. azimuth angle in spherical coordinates) in the laboratory frame of reference
122  //and in the VAMOS coordinate system.
123  //It varies between 0... to 359.999... degrees.
124 
125  //Convert phi angle in VAMOS standard (in rad) to INDRA standard (in rad)
126  Double_t PhiINDRA = GetPhiL() * TMath::DegToRad();
127  //- convert phi (VAMOS ref. frame ) range from [-Pi; Pi] to [0, 2*Pi]
128  if (PhiINDRA < 0) PhiINDRA += 2.*TMath::Pi();
129  //- convert phi in VAMOS frame (and range [0, 2*Pi]) to INDRA ref. frame
130  PhiINDRA -= 1.*TMath::Pi() / 2.;
131  if (PhiINDRA < 0) PhiINDRA += 2.*TMath::Pi();
132 
133  return PhiINDRA * TMath::RadToDeg();
134  }

157 Double_t KVTGIDFunctions::fede(Double_t* x, Double_t* par)
158 {
159  // General Tassan-Got functional used with KVTGIDFitter
160  // It gives Y as a function of X and Z or A (depending on parameters).
161  //
162  // par[0] = ixt -----> type of functional
163  // * ixt : =0->basic functional <>0->extended functional
164  // * * For the basic formula :
165  // * yy = ((g*E)**(mu+1)+lambda**(mu+1)*Z**2*A**mu)**(1/(mu+1))-g*E + pdy
166  // * * For the extended formula :
167  // * yy = ((g*E)**(mu+nu+1)+(lambda*Z**alpha*A**beta)**(mu+nu+1)+
168  // * xi*Z*Z*A**mu*(g*E)**nu)**(1/(mu+nu+1))-g*E + pdy
169  //
170  // par[1] = ih -----> treatment of CsI total light output
171  // * ih : =0->no non-linear light response <>0->non-linear light response included
172  // * * If ih=0 no non-linear light response : E=xx-pdx
173  // * * If ih<>0 non-linear light response included :
174  // * E = sqrt(h**2+2*rho*h*(1+log(1+h/rho)))
175  // * rho=eta*Z**2*A and h=xx-pdx
176  //
177  // par[2] = ZorA -----> whether x[0]=Z or x[0]=A
178  // * ZorA : =0->x[0]=A <>0->x[0]=Z
179  // * * If ZorA=1, we calculate A using the mass formula indicated
180  // * by the value of MassForm=par[3]
181  // * * If ZorA=0, Z is given by the value of par[3]
182  //
183  // par[3] = mass formula or Z
184  // * If ZorA=1, par[3] = mass formula, for values see KVNucleus::GetAFromZ
185  // * If ZorA=0, par[3] = Z
186  //
187  // par[4] = X coordinate
188  // par[5] = Y coordinate
189  //
190  // The number & order of remaining parameters depend on ixt and ih.
191  // * ixt=0 ih=0 5 parameters: lambda, mu, g, pdx, pdy
192  // * ixt=0 ih<>0 6 parameters: lambda, mu, g, pdx, pdy, eta
193  // * ixt<>0 ih=0 9 parameters: lambda, alpha, beta, mu, nu,
194  // * xi, g, pdx, pdy
195  // * ixt<>0 ih<>0 10 parameters: lambda, alpha, beta, mu, nu,
196  // * xi, g, pdx, pdy, eta
197  //
198  // Therefore:
199  //
200  // ixt=0, ih=0: Double_t par[11];
201  // ============
202  // par[6] = lambda
203  // par[7] = mu
204  // par[8] = g
205  // par[9] = pdx
206  // par[10] = pdy
207  //
208  // ixt=0, ih<>0: Double_t par[12];
209  // ============
210  // par[6] = lambda
211  // par[7] = mu
212  // par[8] = g
213  // par[9] = pdx
214  // par[10] = pdy
215  // par[11] = eta
216  //
217  // ixt<>0, ih=0: Double_t par[15];
218  // ============
219  // par[6] = lambda
220  // par[7] = alpha
221  // par[8] = beta
222  // par[9] = mu
223  // par[10] = nu
224  // par[11] = xi
225  // par[12] = g
226  // par[13] = pdx
227  // par[14] = pdy
228  //
229  // ixt<>0, ih<>0: Double_t par[16];
230  // ============
231  // par[6] = lambda
232  // par[7] = alpha
233  // par[8] = beta
234  // par[9] = mu
235  // par[10] = nu
236  // par[11] = xi
237  // par[12] = g
238  // par[13] = pdx
239  // par[14] = pdy
240  // par[15] = eta
241 
242  Int_t ixt, ih, ZorA;
243  ixt = (Int_t)par[0];
244  ih = (Int_t)par[1];
245  ZorA = (Int_t)par[2];
246 
247  Double_t A = 1, Z = 1;
248  if (ZorA == 0) {
249  // A given as function argument, Z is parameter
250  A = x[0];
251  Z = par[3];
252  } else if (ZorA == 1) {
253  // Z given as function argument, calculate A
254  Z = x[0];
255  A = KVNucleus::GetRealAFromZ(Z, (Int_t)par[3]);
256  }
257 
258  Double_t E, pdx, pdy, eta, lambda, mu, g, yy;
259  eta = 0.;
260  lambda = par[6];
261  if (ixt) {
262  //extended formula
263  pdx = par[13];
264  pdy = par[14];
265  mu = par[9];
266  g = par[12];
267  Double_t alpha = par[7];
268  Double_t beta = par[8];
269  Double_t nu = par[10];
270  Double_t xi = par[11];
271  Double_t gamma = mu + nu + 1.;
272  if (ih) {
273  eta = par[15];
274  //calculate energy from total light
275  Double_t h = par[4] - pdx;
276  if (h > 0) {
277  Double_t rho = eta * Z * Z * A;
278  if (rho > 0) {
279  E = TMath::Sqrt(h * h + 2 * rho * h * (1 + TMath::Log(1 + h / rho)));
280  } else
281  E = h;
282  } else {
283  E = 0;
284  }
285  } else {
286  E = par[4] - pdx;
287  }
288  Double_t gE = g * E;
289  //yy = ((g*E)**(mu+nu+1)+(lambda*Z**alpha*A**beta)**(mu+nu+1)+
290  // xi*Z**2*A**mu*(g*E)**nu)**(1/(mu+nu+1))-g*E + pdy
291  yy = TMath::Power(gE, gamma)
292  + TMath::Power((lambda * TMath::Power(Z, alpha) * TMath::Power(A, beta)), gamma)
293  + xi * Z * Z * TMath::Power(A, mu) * TMath::Power(gE, nu);
294  yy = TMath::Power(yy, 1. / gamma) - gE + pdy;
295  } else {
296  //"standard" formula
297  mu = par[7];
298  g = par[8];
299  pdx = par[9];
300  pdy = par[10];
301  Double_t muplus = mu + 1.;
302  if (ih) {
303  eta = par[11];
304  //calculate energy from total light
305  Double_t h = par[4] - pdx;
306  if (h > 0) {
307  Double_t rho = eta * Z * Z * A;
308  if (rho > 0) {
309  E = TMath::Sqrt(h * h + 2 * rho * h * (1 + TMath::Log(1 + h / rho)));
310  } else
311  E = h;
312  } else {
313  E = 0;
314  }
315  } else {
316  E = par[4] - pdx;
317  }
318  Double_t gE = g * E;
319  //yy = ((g*E)**(mu+1)+lambda**(mu+1)*Z**2*A**mu)**(1/(mu+1))-g*E + pdy
320  yy = TMath::Power(gE, muplus) + TMath::Power(lambda, muplus) * Z * Z * TMath::Power(A, mu);
321  yy = TMath::Power(yy, 1. / muplus) - gE + pdy;
322  }
323  return yy - par[5];
324 }

464 Double_t KVTGIDFunctions::pichon_Z(Double_t* x, Double_t* par)
465 {
466  //Tassan-Got formula (10 parameters) used for Z identification in Si-CsI or ChIo-CsI
467  //5th campaign telescopes - 12 parameters
468  //As per Matthieu Pichon's these, we use A = 2Z for the mass.
469  //Returns difference between measured ChIo/Si PG/GG channel
470  //and that calculated for given (Z, CsI light output).
471  //
472  //Corresponds to Eq.11 of NIM B194 (2002) 503 with Eq.13 for the
473  //light-energy dependence
474  //
475  //Arguments are
476  //x[0] Z of nucleus considered
477  // par[0] : lambda
478  // par[1] : alpha (1: 0.5 - 1.5)
479  // par[2] : beta (0.5: 0.2 - 1)
480  // par[3] : mu (1: 0.2 - 1.5)
481  // par[4] : nu (1: 0.1 - 4)
482  // par[5] : zeta (?? squiggly epsilon thingy ??) (>0)
483  // par[6] : g
484  // par[7] : pedestal of x-coordinate
485  // par[8] : pedestal of y-coordinate
486  // par[9] : eta
487  //par[10] measured total light output of CsI - X coord
488  //par[11] measured ChIo/Si PG or GG channel - Y coord
489 
490  Double_t Z = x[0];
491  Double_t A = 2. * Z; //Deduce A from Z, A=2Z
492 
493  Double_t gamma = par[3] + par[4] + 1.0;//gamma=mu+nu+1
494  Double_t h = par[10] - par[7];
495  Double_t de = par[11] - par[8];
496 
497  Double_t rho = par[9] * TMath::Power(Z, 2) * A;//=eta*Z^2*A
498  Double_t E;
499  if (h > 0) {
500  if (rho > 0) {
501  E = TMath::Sqrt(TMath::Power(h, 2) + 2 * rho * h * (1 + TMath::Log(1 + h / rho)));
502  } else
503  E = h;
504  } else
505  E = 0;
506 
507  Double_t gE = par[6] * E;//=g*E
508  Double_t gE_to_the_gamma = TMath::Power(gE, gamma);
509  Double_t gE_to_the_nu = TMath::Power(gE, par[4]);
510  Double_t lambda_Z_alpha_A_beta =
511  par[0] * TMath::Power(Z, par[1]) * TMath::Power(A, par[2]);
512  Double_t zeta_Z2_A_mu = par[5] * TMath::Power(Z, 2) * TMath::Power(A, par[3]);
513  Double_t one_over_gamma = 1. / gamma;
514  Double_t Delta_E = gE_to_the_gamma + TMath::Power(lambda_Z_alpha_A_beta, gamma)
515  + zeta_Z2_A_mu * gE_to_the_nu;
516  Delta_E = TMath::Power(Delta_E, one_over_gamma) - gE;
517  return (Delta_E - de);
518 }

536 {
537  //Tassan-Got formula (13 parameters) for A identification (known Z)
538  //Returns difference between measured ChIo/Si PG/GG channel and that calculated for given
539  //(Z, A, CsI light output)
540  //
541  //Arguments are
542  //x[0] A of nucleus considered
543  //par[0], ..., par[9] 10 parameters of corrected Tassan-Got formula
544  //par[10] Z of nucleus considered
545  //par[11] measured total light output of CsI - X coord
546  //par[12] measured ChIo/Si PG or GG channel - Y coord
547 
548  Double_t gamma = par[3] + par[4] + 1.0;
549  Double_t h = par[11] - par[7];
550  Double_t de = par[12] - par[8];
551

553  Double_t E;
554  if (h > 0) {

565  Power((par[0] * TMath::Power(par[10], par[1]) *
566  TMath::Power(x[0], par[2])), gamma)

569  par[3])

589 Double_t KVTGIDFunctions::pawlowski_Z(Double_t* x, Double_t* par)
590 {
591  //Tassan-Got formula with Pawlowski correction (11th parameter) for Z identification
592  //A = 2*Z + 1
593  //13 parameters in all.
594  //Returns difference between measured Si PG/GG channel
595  //and that calculated for given (Z, CsI light output).
596  //
597  //Arguments are
598  //x[0] Z of nucleus considered
599  //par[0], ..., par[10] 11 parameters of corrected Tassan-Got formula
600  //par[11] measured total light output of CsI - X coord
601  //par[12] measured ChIo/Si PG or GG channel - Y coord
602 
603  Double_t A = 2.*x[0] + 1; //Deduce A from Z
604 
605  Double_t gamma = par[3] + par[4] + 1.0;//gamma=mu+nu+1
606  Double_t h = par[11] - par[7];
607  h = 0.5 * (h + TMath::Sqrt(TMath::Power(h, 2) + par[10])); //Correction P. Pawlowski
608  Double_t de = par[12] - par[8];
609 
610  Double_t rho = par[9] * TMath::Power(x[0], 2) * A;//=eta*Z^2*A
611  Double_t E;
612  if (h > 0) {
613  if (rho > 0) {
614  E = TMath::Sqrt(TMath::Power(h, 2) + 2 * rho * h * (1 + TMath::Log(1 + h / rho)));
615  } else
616  E = h;
617  } else
618  E = 0;
619 
620  Double_t gE = par[6] * E;//=g*E
621  Double_t gE_to_the_gamma = TMath::Power(gE, gamma);
622  Double_t gE_to_the_nu = TMath::Power(gE, par[4]);
623  Double_t lambda_Z_alpha_A_beta =
624  par[0] * TMath::Power(x[0], par[1]) * TMath::Power(A, par[2]);
625  Double_t zeta_Z2_A_mu = par[5] * TMath::Power(x[0], 2) * TMath::Power(A, par[3]);
626  Double_t one_over_gamma = 1. / gamma;
627  Double_t Delta_E = gE_to_the_gamma + TMath::Power(lambda_Z_alpha_A_beta, gamma)
628  + zeta_Z2_A_mu * gE_to_the_nu;
629  Delta_E = TMath::Power(Delta_E, one_over_gamma) - gE;
630  return (Delta_E - de);
631 }

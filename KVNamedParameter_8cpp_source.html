266 const Char_t* KVNamedParameter::GetString() const
267 {
268  // Returns value of parameter as a string, whatever the type
269  // (integer or floating values are converted to a string,

271

398 Bool_t KVNamedParameter::HasSameValueAs(const KVNamedParameter& other) const
399 {
400  // Returns kTRUF if the two parameters have the same type and the
401  // same value (don't care about parameter names)
402  if (other.GetType() != GetType()) return kFALSE;
403  switch (GetType()) {
404  case kIsString:
405  if (fTitle == other.fTitle) return kTRUE;
406  break;
407 
408  case kIsInt:
409  if (other.GetInt() == GetInt()) return kTRUE;
410  break;
411 
412  case kIsDouble:
413  return KVBase::AreEqual(other.GetDouble(), GetDouble());
414  break;
415 
416  case kIsBool:
417  if (other.GetBool() == GetBool()) return kTRUE;
418  break;
419 
420  default:
421  return kFALSE;
422  }
423  return kFALSE;
424 }

449 void KVNamedParameter::ls(Option_t* option) const
450 {
451  // compact listing of parameter name & value, used by KVNameValueList::Print
452  // option controls what is printed:
453  // "" (default) : all parameters
454  // "int" : only integer parameters
455  // "bool" : only boolean parameters
456  // "double" : only double parameters
457  // "string" : only string parameters
458 
459  Bool_t can_print = kTRUE;
460  if (strcmp(option, "")) {
461  TString opt(option);
462  opt.ToLower();
463  if (opt == "int" && !IsInt()) can_print = kFALSE;
464  else if (opt == "bool" && !IsBool()) can_print = kFALSE;
465  else if (opt == "double" && !IsDouble()) can_print = kFALSE;
466  else if (opt == "string" && !IsString()) can_print = kFALSE;
467  }
468  TROOT::IndentLevel();
469  if (IsString()) {
470  if (can_print) cout << "<" << GetName() << "=" << GetTitle() << ">" << endl;
471  } else {
472  switch (GetType()) {
473  case kIsInt:
474  if (can_print) cout << "<" << GetName() << "=" << GetInt() << ">" << endl;
475  break;
476 
477  case kIsBool:
478  if (can_print) cout << "<" << GetName() << "=" << GetString() << ">" << endl;
479  break;
480 
481  case kIsDouble:
482  if (can_print) cout << "<" << GetName() << "=" << GetDouble() << ">" << endl;
483  break;
484 
485  default:
486  break;
487  }
488  }

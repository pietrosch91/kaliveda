162  // set mass formula of grid (and identifiers)

167  par = fPar->GetParameter(i);
168  TString parname(par->GetName());
169  if (!parname.Contains("ScaleFactor"))
170  grid.fPar->SetValue(*par);
171  }
172  //restore scaling to this grid if there was one, and apply it to the copied grid
173  if (fLastScaleX != 1 || fLastScaleY != 1) {
174  const_cast < KVIDGraph* >(this)->Scale(fLastScaleX, fLastScaleY);
175  grid.SetXScaleFactor(fLastScaleX);
176  grid.SetYScaleFactor(fLastScaleY);
177  }

457 void KVIDGraph::WriteToAsciiFile(ofstream& gridfile)
458 {
459  //Write info on this grid in already-open ascii file stream 'gridfile'.
460  //This begins with the line
461  //++ClassName
462  //where '"ClassName'" is the name of the class of the actual instance of this object
463  //followed by informations on the grid and its objects.
464  //
465  //The names associated to the X & Y axis of the identification map are written
466  //on lines beginning with
467  //<VARX>
468  //<VARY>
469  //Any associated parameters are written before the objects, starting with
470  //<PARAMETER>
471  //then the name of the parameter and its value
472  //e.e.
473  //<PARAMETER> ChIo pressure=45.5
474  //A single '!' on a line by itself signals the end of the informations for this grid.
475  //
476  //Scaling factors
477  //Any scaling factors are written in the file as
478  //<PARAMETER> XScaleFactor=...
479  //<PARAMETER> YScaleFactor=...
480  //However the coordinates written in the file are the unscaled factors
481  //When the file is read back again, the unscaled coordinates are read in and
482  //then the scaling is applied.
483 
484  gridfile << "# ASCII file generated by " << ClassName() <<
485  "::WriteToAsciiFile" << endl;
486  gridfile << "# ID Graph Name : " << GetName() << endl;
487  gridfile << "# This file can be read using " << ClassName() <<
488  "::ReadFromAsciiFile" << endl;
489  gridfile << "# " << endl;
490  gridfile << "++" << ClassName() << endl;
491 
492  // write name if given
493  if (fName != "") gridfile << "<NAME> " << fName.Data() << endl;
494  //write X & Y axis names
495  gridfile << "<VARX> " << GetVarX() << endl;
496  gridfile << "<VARY> " << GetVarY() << endl;
497  //write parameters
498  WriteParameterListOfIDTelescopes();
499  KVNamedParameter* par = 0;
500  for (int i = 0; i < fPar->GetNpar(); i++) { //loop over all parameters
501  par = fPar->GetParameter(i);
502  gridfile << "<PARAMETER> " << par->GetName() << "=" << par->GetString() << endl;
503  }
504 
505  //write fOnlyZId & mass formula
506  if (OnlyZId()) gridfile << "OnlyZId " << GetMassFormula() << endl;
507 
508  //remove scaling if there is one
509  if (fLastScaleX != 1 || fLastScaleY != 1)
510  Scale(1. / fLastScaleX, 1. / fLastScaleY);
511 
512  //Write identifiers
513  TIter next_IDline(fIdentifiers);
514  KVIDentifier* line;
515  while ((line = (KVIDentifier*) next_IDline())) {
516  line->WriteAsciiFile(gridfile, "ID");
517  }
518  //Write cuts
519  TIter next_OKline(fCuts);
520  while ((line = (KVIDentifier*) next_OKline())) {
521  line->WriteAsciiFile(gridfile, "OK");
522  }
523  gridfile << "!" << endl;
524 
525  //restore scaling if there is one
526  if (fLastScaleX != 1 || fLastScaleY != 1)
527  Scale(fLastScaleX, fLastScaleY);
528 
529  // if a back-up copy had previously been created (by starting the editor)
530  // we replace it by the version read from file
531  if (fLastSavedVersion) UpdateLastSavedVersion();
532 }

604 void KVIDGraph::ReadFromAsciiFile(ifstream& gridfile)
605 {
606  //Read info from already open ascii file stream containing each line of this grid.
607  //Any previous lines in the grid are destroyed.
608  //Comment lines begin with #
609  //Grid info begins with
610  //++ClassName
611  //where ClassName is actual class name of instance of this object
612  //The names associated to the X & Y axis of the identification map are written
613  //on lines beginning with
614  //<VARX>
615  //<VARY>
616  //Parameter lines begin with <PARAMETER>
617  //New lines begin with +name_of_class
618  //'!' signals end of grid info
619  //Any scaling factors are applied once the coordinates have been read
620 
621  //reset grid - destroy old lines, axis limits...
622  Clear();
623  //clear parameters
624  fPar->Clear();
625 
626  KVString s;
627  int line_no = 0; // counter for lines read
628  SetOnlyZId(kFALSE);
629  Int_t mass_formula = -1;
630 
631  while (gridfile.good()) {
632  //read a line
633  s.ReadLine(gridfile);
634  if (s.BeginsWith('!')) {
635  //end of grid signalled
636  break;
637  } else if (s.BeginsWith("++")) { //will only happen if we are reading a file using ReadAsciiFile(const char*)
638  //check classname
639  s.Remove(0, 2);
640  s.Remove(TString::kBoth, ' '); //remove whitespace
641  if (s != ClassName())
642  Warning("ReadFromAsciiFile(ofstream&)",
643  "Class name in file %s does not correspond to this class (%s)",
644  s.Data(), ClassName());
645  } else if (s.BeginsWith("<NAME>")) {
646  //name of grid
647  s.Remove(0, 7);
648  s.Remove(TString::kBoth, ' '); //remove whitespace
649  SetName(s.Data());
650  } else if (s.BeginsWith("<VARX>")) {
651  //x-axis definition
652  s.Remove(0, 7);
653  s.Remove(TString::kBoth, ' '); //remove whitespace
654  SetVarX(s.Data());
655  } else if (s.BeginsWith("<VARY>")) {
656  //y-axis definition
657  s.Remove(0, 7);
658  s.Remove(TString::kBoth, ' '); //remove whitespace
659  SetVarY(s.Data());
660  } else if (s.BeginsWith("<PARAMETER>")) {
661  //parameter definition
662  s.Remove(0, 11);
663  //split into tokens separated by '='
664  TObjArray* toks = s.Tokenize('=');
665  if (toks->GetEntries() > 1) {
666  TString name =
667  ((TObjString*) toks->At(0))->GetString().
668  Strip(TString::kBoth);
669  KVString value(((TObjString*) toks->At(1))->GetString());
670  value.Remove(TString::kBoth, ' '); //remove whitespace
671  if (name != "" && value != "") {
672  if (value.IsFloat())
673  fPar->SetValue(name.Data(), value.Atof());
674  else if (value.IsDigit())
675  fPar->SetValue(name.Data(), value.Atoi());
676  else
677  fPar->SetValue(name.Data(), value.Data());
678  }
679  }
680  delete toks; //must clean up
681  } else if (s.BeginsWith("<LIST>")) { // number list definition
682  s.Remove(0, 6); // get rid of "<LIST>"
683  //split into tokens separated by '='
684  TObjArray* toks = s.Tokenize('=');
685  if (toks->GetEntries() > 1) {
686  TString name =
687  ((TObjString*) toks->At(0))->GetString().
688  Strip(TString::kBoth);
689  TString list(((TObjString*) toks->At(1))->GetString());
690  //set appropriate list
691  if (name == "Runs")
692  SetRuns(list.Data());
693  else
694  fPar->SetValue(name.Data(), list.Data());
695  }
696  delete toks; //must clean up
697  } else if (s.BeginsWith("OnlyZId")) {
698  SetOnlyZId(kTRUE);
699  s.ReplaceAll("OnlyZId", "");
700  s.Remove(TString::kBoth, ' ');
701  if (s != "") { //older versions did not write mass formula after OnlyZId
702  mass_formula = s.Atoi();
703  if (mass_formula > -1) SetMassFormula(mass_formula);
704  }
705  } else if (s.BeginsWith('+')) {
706  //New line
707  line_no++;
708  //Get name of class by stripping off the '+' at the start of the line
709  s.Remove(0, 1);
710  TString lineclass = s;
711  //next line is type ("ID" or "OK") followed by ':', followed by name of line
712  s.ReadLine(gridfile);
713  //split into tokens separated by ':'
714  TObjArray* toks = s.Tokenize(':');
715  TString type = ((TObjString*) toks->At(0))->GetString();
716  TString name = "";
717  // make sure the line actually has a name !
718  if (toks->GetEntries() > 1) name = ((TObjString*) toks->At(1))->GetString();
719  else {
720  // print warning allowing to find nameless line in file
721  Warning("ReadFromAsciiFile",
722  "In graph : %s\nIdentifier with no name. Identifier type = %s. Line number in graph = %d",
723  GetName(), type.Data(), line_no);
724  }
725  delete toks; //must clean up
726  ReadIdentifierFromAsciiFile(name, type, lineclass, gridfile);
727  }
728  }
729  fLastScaleX = GetXScaleFactor();
730  fLastScaleY = GetYScaleFactor();
731  if (fLastScaleX != 1 || fLastScaleY != 1) {
732  Scale(fLastScaleX, fLastScaleY);
733  }
734  BackwardsCompatibilityFix();
735  //set runlist
736  if (fPar->HasParameter("Runlist")) SetRuns(fPar->GetStringValue("Runlist"));
737  else SetRuns("");
738 
739  // if a back-up copy had previously been created (by starting the editor)
740  // we replace it by the version read from file
741  if (fLastSavedVersion) UpdateLastSavedVersion();
742 }

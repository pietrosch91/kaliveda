90 char* GTDataParameters::CopyParam(char* Dest, char* Source) const
91 {
92  // Small utility routine to copy a char string.
93  char c;
94  do {
95  c = *Source++;
96  if ((c == ',') || (c == 0x0d))

112 int GTDataParameters::Fill(const char* buffParam)
113 {
114  // Data from the parameter buffet (buffParam) have been read from disk and is
115  // parsed by this routine who create the list of labels and corresponding
116  // index.
117  //Returns largest parameter number (not necessarily same as total number of parameters)
118 
119  char* CurrPointer;
120  char NomParam[20], chaine[20];
121  int NumPar, NbBits;
122 
123  CurrPointer = (char*)buffParam;
124 
125  fMaxIndex = 0;
126 
127  while (strncmp(CurrPointer, " ", 4) != 0 &&
128  strncmp(CurrPointer, "!!!!", 4) != 0) {
129  CurrPointer = CopyParam(NomParam, CurrPointer);
130  CurrPointer = CopyParam(chaine, CurrPointer);
131  sscanf(chaine, "%d", &NumPar);
132  CurrPointer = CopyParam(chaine, CurrPointer);
133  sscanf(chaine, "%d", &NbBits);
134  GTDataPar* par = new GTDataPar(NomParam, NumPar, NbBits);
135  fList->Add(par);
136  fMaxIndex = TMath::Max(par->Index(), fMaxIndex);
137  }
138  return (fMaxIndex);
139 }

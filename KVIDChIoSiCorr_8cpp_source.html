67 Double_t KVIDChIoSiCorr::GetIDMapY(Option_t* opt)
68 {
69  // This method gives the Y-coordinate in a 2D identification map
70  // associated whith the ChIo-Si identification telescope.
71  // The Y-coordinate is the ionisation chamber's current petit gain coder data minus the petit gain pedestal. If the grand gain coder
72  // data is less than 3900 then the petit gain value is calculated
73  // from the current grand gain coder data (see KVINDRADetector::GetPGFromGG())
74  // We set the pedestal to zero in order to not cut the physics
75  // as the pedestal GG seems to be too high.
76  // PG-GG conversion has been calculated without ChIo GG pedestal.
77 
78  fchio->SetPedestal("GG", 0.);
79  return GetIDMapXY((KVINDRADetector*)fchio, opt);
80 }

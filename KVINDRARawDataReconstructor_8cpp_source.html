98 void KVINDRARawDataReconstructor::InitRun()
99 {
100  // Creates new ROOT file with TTree for reconstructed/calibrated events.
101  // By default this file will be written in the same data repository as the raw data file we are reading.
102  // This can be changed by setting the environment variable(s):
103  //
104  // Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
105  // [name of dataset].Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
106  //
107  // If no value is set for the current dataset (second variable), the value of the
108  // first variable will be used. If neither is defined, the new file will be written in the same repository as
109  // the raw file (if possible, i.e. if repository is not remote).
110 
111  // Create new KVINDRAReconEvent used to reconstruct & store events
112  // The condition used to seed new reconstructed particles (see KVReconstructedEvent::AnalyseTelescopes)
113  // is set by reading the value of the environment variables:
114  // Reconstruction.DataAnalysisTask.ParticleSeedCond: [all/any]
115  // [name of dataset].Reconstruction.DataAnalysisTask.ParticleSeedCond: [all/any]
116  // If no value is set for the current dataset (second variable), the value of the
117  // first variable will be used.
118 
119  if (!recev) recev = new KVINDRAReconEvent;
120  recev->SetPartSeedCond(gDataSet->GetDataSetEnv("Reconstruction.DataAnalysisTask.ParticleSeedCond"));
121 
122  // get dataset to which we must associate new run
123  KVDataSet* OutputDataset =
124  gDataRepositoryManager->GetDataSet(gDataSet->GetOutputRepository(taskname), gDataSet->GetName());
125 
126  file = OutputDataset->NewRunfile(datatype.Data(), fRunNumber);
127 
128  cout << "Writing \"" << datatype.Data() << "\" events in ROOT file " << file->GetName() << endl;
129 
130  //tree for raw data
131  rawtree = new TTree("RawData", Form("%s : %s : raw data",
132  gIndraDB->GetRun(fRunNumber)->GetName(), gIndraDB->GetRun(fRunNumber)->GetTitle()));
133  rawtree->Branch("RunNumber", &fRunNumber, "RunNumber/I");
134  rawtree->Branch("EventNumber", &fEventNumber, "EventNumber/I");
135 
136  // the format of the raw data tree must be "arrays" : we depend on it in KVINDRAReconDataAnalyser
137  // in order to read the raw data and set the detector acquisition parameters
138  TString raw_opt = "arrays";
139  GetRawDataReader()->SetUserTree(rawtree, raw_opt.Data());
140  Info("InitRun", "Created raw data tree (%s : %s). Format: %s",
141  rawtree->GetName(), rawtree->GetTitle(), raw_opt.Data());
142 
143 #if ROOT_VERSION_CODE > ROOT_VERSION(5,25,4)
144 #if ROOT_VERSION_CODE < ROOT_VERSION(5,26,1)
145  // The TTree::OptimizeBaskets mechanism is disabled, as for ROOT versions < 5.26/00b
146  // this lead to a memory leak
147  rawtree->SetAutoFlush(0);
148 #endif
149 #endif
150 
151 
152  //tree for reconstructed events
153  tree = new TTree("ReconstructedEvents", Form("%s : %s : %s events created from raw data",
154  gIndraDB->GetRun(fRunNumber)->GetName(),
155  gIndraDB->GetRun(fRunNumber)->GetTitle(),
156  datatype.Data())
157  );
158 #if ROOT_VERSION_CODE > ROOT_VERSION(5,25,4)
159 #if ROOT_VERSION_CODE < ROOT_VERSION(5,26,1)
160  // The TTree::OptimizeBaskets mechanism is disabled, as for ROOT versions < 5.26/00b
161  // this lead to a memory leak
162  tree->SetAutoFlush(0);
163 #endif
164 #endif
165 
166 
167  //leaves for reconstructed events
168  KVEvent::MakeEventBranch(tree, "INDRAReconEvent", "KVINDRAReconEvent", &recev);
169 
170  Info("InitRun", "Created reconstructed data tree %s : %s", tree->GetName(), tree->GetTitle());
171 
172  //tree for gene data
173  genetree = new TTree("GeneData", Form("%s : %s : gene data",
174  gIndraDB->GetRun(fRunNumber)->GetName(), gIndraDB->GetRun(fRunNumber)->GetTitle()));
175 
176  //we add to the 'gene tree' a branch for every acquisition parameter of the detector
177  genetree->Branch("RunNumber", &fRunNumber, "RunNumber/I");
178  genetree->Branch("EventNumber", &fEventNumber, "EventNumber/I");
179  KVACQParam* acqpar;
180  TIter next_acqpar(gIndra->GetACQParams());
181  while ((acqpar = (KVACQParam*)next_acqpar())) {
182  genetree->Branch(acqpar->GetName(), *(acqpar->ConnectData()), Form("%s/S", acqpar->GetName()));
183  }
184 #if ROOT_VERSION_CODE > ROOT_VERSION(5,25,4)
185 #if ROOT_VERSION_CODE < ROOT_VERSION(5,26,1)
186  // The TTree::OptimizeBaskets mechanism is disabled, as for ROOT versions < 5.26/00b
187  // this lead to a memory leak
188  genetree->SetAutoFlush(0);
189 #endif
190 #endif
191 
192 
193  Info("InitRun", "Created pulser/laser data tree (%s : %s) for %d parameters",
194  genetree->GetName(), genetree->GetTitle(), genetree->GetNbranches());
195  //initialise number of reconstructed events
196  nb_recon = 0;
197 
198 }

211 Bool_t KVINDRARawDataReconstructor::Analysis()
212 {
213  // Analysis of event.
214  // RawData TTree is filled with values of parameters for event.
215  // If event has INDRA trigger information (IsIndraEvent()==kTRUE)
216  // then
217  // *) event reconstruction is performed for 'Physics' events
218  // *) or the GeneTree is filled with pulser/laser data for 'Gene' events
219 
220  recev->SetNumber(GetEventNumber());
221 
222  if (gIndra->GetTriggerInfo()->IsINDRAEvent()) {
223  if (gIndra->GetTriggerInfo()->IsPhysics()) {
224  gIndra->ReconstructEvent(recev, (KVDetectorEvent*)GetDetectorEvent());
225  nb_recon++;
226  ExtraProcessing();
227  } else {
228  genetree->Fill();
229  }
230  }
231  // ReconstructedEvents tree must be filled for every event, even ones where
232  // no event has been reconstructed. This is so that when reading back we can make
233  // RawData a 'friend' TTree and keep everything in synch.
234  tree->Fill();
235  recev->Clear();
236 
237  return kTRUE;
238 }

262 }

57 Double_t KVIDChIoCorrCsI_e494s::GetIDMapY(Option_t* opt)
58 {
59  // This method gives the Y-coordinate in a 2D identification map
60  // associated whith the ChIo-CsI identification telescope.
61  // The Y-coordinate is the ionisation chamber's current petit gain coder data minus the petit gain pedestal. If the grand gain coder
62  // data is less than 3900 then the petit gain value is calculated
63  // from the current grand gain coder data (see KVINDRADetector::GetPGFromGG())
64  // We set the pedestal to zero in order to not cut the physics
65  // as the pedestal GG seems to be too high.
66  // PG-GG conversion has been calculated without ChIo GG pedestal.
67 
68  IGNORE_UNUSED(opt);
69 
70  fChIo->SetPedestal("GG", 0.);
71  if (fChIo->GetGG() < 3900.) return fChIo->GetPGfromGG() - fChIo->GetPedestal("PG");
72  return fChIo->GetPG() - fChIo->GetPedestal("PG");
73 }

6 // MonoMinimiserImpl.cpp
7 //
8 // Description
9 //
10 // Standard implementation of the silicon energy minimiser. This class
11 // implements the minimisation by using two while loops operating in different
12 // directions (in terms of A value), both starting from the same initial ansatz
13 // that A is most likely to be around to 2Z. It is essentially the single
14 // threaded version of the ThreadedMassEstimator, operating to reduce the
15 // difference between the observed and simulated silicon detector energies in
16 // order to determine the most likely value for A.
17 //
18 // The bisection method previously used has been replaced by this "brute force"
19 // method. The reasoning behind this decision is that the bisection method
20 // relies upon the function being well behaved and passing through delta = 0
21 // with positive values on one side of the minimum and negative ones on the
22 // other. However due to the large A range being used, this does not seem to be
23 // guaranteed. For example if one takes a value in A which is too large, the
24 // energy calculation can result in something quite extreme and using the
25 // bisection method you will lose 1/2 of the available solution space from this
26 // single error.
27 //
28 // On my desktop this algorithm processes at about 85 events/s compared with 220
29 // events/s for the threaded version (ThreadedMassEstimator).
30 //
31 // Note: It is important that you keep the public interface (public member
32 // functions) of the implementation classes in agreement with those in this
33 // class, as all functions are simple passed off to the implementation class.
34 // For example SiliconEnergyMinimiser::SetIDTelescope() simply calls
35 // MonoMinimiserImpl::SetIDTelescope().
36 //
37 // Peter C. Wigg <peter.wigg.314159@gmail.com>
38 // Sun 17 Jan 20:43:44 2016
39 //

183 Int_t MonoMinimiserImpl::Minimise(
184  UInt_t z_value, Double_t si_energy, Double_t csi_light,
185  MinimiserData* const data)
186 {
187  assert(kInitialised_);
188 
189  // These assertions are paired with the user input validation tests in
190  // SiliconEnergyMinimiser::Minimise so you should never encounter them.
191  assert((z_value != 0) && (z_value <= 120));
192  assert(si_energy > 0.);
193  assert(csi_light > 0.);
194 
195  if (!kTelescopeSet_) {
196  Error("MonoMinimiserImpl::Minimise",
197  "You must call MonoMinimiserImpl::SetIDTelescope() "
198  "before calling this function");
199  return kTelescopeNotSet;
200  }
201 
202 #if __cplusplus < 201103L
203  MonoMinimiserData* mono_data(NULL);
204 #else
205  MonoMinimiserData* mono_data(nullptr);
206 #endif
207 
208  if (data) {
209  // Optional data storage pointer provided, cast it to make sure we can
210  // provide the extra data. It is assumed that the true nature of the
211  // pointer being passed in is a MonoMinimiserData pointer.
212  mono_data = static_cast<MonoMinimiserData*>(data);
213  assert(mono_data);
214  mono_data->SetZ(z_value);
215  mono_data->SetStatusCode(0);
216  }
217 
218  sim_parameters_->z = z_value;
219  sim_parameters_->si_energy = si_energy;
220  sim_parameters_->csi_light = csi_light;
221 
222  UInt_t starting_a_value(2 * z_value);
223 
224  UInt_t a_value(starting_a_value);
225  Double_t delta(100000.);
226  Double_t delta_last(100000.);
227 
228  Double_t best_forward_delta(100000.);
229  Double_t best_backward_delta(100000.);
230  UInt_t best_forward_a_value(0);
231  UInt_t best_backward_a_value(0);
232 
233  // Records the number of steps for which we have been consistently
234  // maximising.
235  UInt_t num_maximising(0);
236 
237  // Same starting point (A = 2Z) for both directions (while loops)
238  sim_parameters_->a = a_value;
239 #if __cplusplus < 201103L
240  if (stack_->Simulate(sim_parameters_, sim_results_)) {
241 #else
242  if (stack_->Simulate(sim_parameters_.get(), sim_results_.get())) {
243 #endif
244  delta_last = sim_results_->delta;
245  } else {
246  delta_last = 100000.;
247  }
248 
249  best_forward_delta = delta_last;
250  best_backward_delta = delta_last;
251  best_forward_a_value = a_value;
252  best_backward_a_value = a_value;
253 
254  Bool_t forward_status(kTRUE);
255  Bool_t backward_status(kTRUE);
256 
257  // -------------------------------------
258  // Iterate Forwards (Increasing A value)
259  // -------------------------------------
260 
261  a_value = starting_a_value + 1;
262  UInt_t n(1);
263  while (n < max_iterations_) {
264 
265  // Go forward from 2Z
266 
267  if (a_value > 299) {
268  forward_status = kFALSE;
269  break;
270  }
271 
272  sim_parameters_->a = a_value;
273 #if __cplusplus < 201103L
274  if (stack_->Simulate(sim_parameters_, sim_results_)) {
275 #else
276  if (stack_->Simulate(sim_parameters_.get(), sim_results_.get())) {
277 #endif
278  delta = sim_results_->delta;
279  } else {
280  delta = 100000.;
281  }
282 
283  if (delta > (delta_last + (tolerance_ * delta_last))) {
284  // The value for delta in this iteration is greater than the last by
285  // the tolerance setting (default 5%)
286 
287  if (num_maximising >= 2) {
288  // We have been maximising at greater than the tolerance setting
289  // (default 5%) for at least 2 iterations, it is perhaps safe to
290  // assume that this trend will continue and so we stop
291  // processing
292 
293  break;
294  }
295  ++num_maximising;
296  } else {
297  num_maximising = 0;
298  }
299 
300  if (delta < best_forward_delta) {
301  best_forward_delta = delta;
302  best_forward_a_value = a_value;
303  }
304 
305  delta_last = delta;
306  ++n;
307  ++a_value;
308  }
309 
310  if (mono_data) mono_data->SetForwardCounter(n);
311 
312  if (n >= max_iterations_) {
313  //Warning("MonoMinimiserImpl::Minimise",
314  // "Forward loop exceeded the maximum number of iterations");
315  if (mono_data) mono_data->SetStatusCode(kForwardExceeded);
316  return kForwardExceeded;
317  }
318 
319  // --------------------------------------
320  // Iterate Backwards (Decreasing A value)
321  // --------------------------------------
322 
323  num_maximising = 0;
324  a_value = starting_a_value - 1;
325  n = 1;
326  while (n < max_iterations_) {
327 
328  // Go backward from 2Z
329 
330  if (a_value < 1) {
331  backward_status = kFALSE;
332  break;
333  }
334 
335  sim_parameters_->a = a_value;
336 #if __cplusplus < 201103L
337  if (stack_->Simulate(sim_parameters_, sim_results_)) {
338 #else
339  if (stack_->Simulate(sim_parameters_.get(), sim_results_.get())) {
340 #endif
341  delta = sim_results_->delta;
342  } else {
343  delta = 100000.;
344  }
345 
346  if (delta > (delta_last + (tolerance_ * delta_last))) {
347  // The value for delta in this iteration is greater than the last by
348  // the tolerance setting (default 5%)
349 
350  if (num_maximising >= 2) {
351  // We have been maximising at greater than the tolerance setting
352  // (default 5%) for at least 2 iterations, it is perhaps safe to
353  // assume that this trend will continue and so we stop
354  // processing
355 
356  break;
357  }
358  ++num_maximising;
359  } else {
360  num_maximising = 0;
361  }
362 
363  if (delta < best_backward_delta) {
364  best_backward_delta = delta;
365  best_backward_a_value = a_value;
366  }
367 
368  delta_last = delta;
369  ++n;
370  --a_value;
371 
372  }
373 
374  if (mono_data) mono_data->SetBackwardCounter(n);
375 
376  if (n >= max_iterations_) {
377  //Warning("MonoMinimiserImpl::Minimise",
378  // "Backward loop exceeded the maximum number of iterations");
379  if (mono_data) mono_data->SetStatusCode(kBackwardExceeded);
380  return kBackwardExceeded;
381  }
382 
383  // ----------------------------------
384  // Select and return the best A value
385  // ----------------------------------
386 
387  if (forward_status && !backward_status) {
388  if (mono_data) {
389  mono_data->SetA(best_forward_a_value);
390  mono_data->SetDelta(best_forward_delta);
391  }
392  return best_forward_a_value;
393 
394  } else if (backward_status && !forward_status) {
395  if (mono_data) {
396  mono_data->SetA(best_backward_a_value);
397  mono_data->SetDelta(best_backward_delta);
398  }
399  return best_backward_a_value;
400 
401  } else if (!backward_status && !forward_status) {
402  if (mono_data) {
403  mono_data->SetA(-1);
404  mono_data->SetDelta(100000.);
405  }
406  return kNoValidResult;
407 
408  } else {
409  if (best_backward_delta < best_forward_delta) {
410  if (mono_data) {
411  mono_data->SetA(best_backward_a_value);
412  mono_data->SetDelta(best_backward_delta);
413  }
414  return best_backward_a_value;
415 
416  } else {
417  if (mono_data) {
418  mono_data->SetA(best_forward_a_value);
419  mono_data->SetDelta(best_forward_delta);
420  }
421  return best_forward_a_value;
422 
423  }
424  }
425 }

158 TGeoVolume* KVRing::GetGeoVolume()
159 {
160  // Create and return TGeoVolume representing detectors in this ring.
161 
162  TString name(GetName());
163  name.ToUpper();
164  name.ReplaceAll(" ", "_");
165  TGeoVolume* mother_vol = gGeoManager->MakeVolumeAssembly(name.Data());
166  //**** BUILD & ADD TELESCOPEs ****
167  TIter next(GetTelescopes());
168  KVTelescope* det;
169  while ((det = (KVTelescope*)next())) {
170  TGeoVolume* det_vol = det->GetGeoVolume();
171  // position telescope in ring
172  // rotate telescope to orientation corresponding to (theta,phi)
173  Double_t theta = det->GetTheta();
174  Double_t phi = det->GetPhi();
175  TGeoRotation rot1, rot2;
176  rot2.SetAngles(phi + 90., theta, 0.);
177  rot1.SetAngles(-90., 0., 0.);
178  Double_t tot_len_tel = det->GetTotalLengthInCM();
179  // distance to telescope centre = distance to telescope + half total length of telescope
180  Double_t dist = det->GetDistance() + tot_len_tel / 2.;
181  // translate telescope to correct distance from target (note: reference is CENTRE of telescope)
182  Double_t trans_1 = dist;
183  // translate telescope so that centre of ring is on origin
184  Double_t trans_2 = dist * TMath::Cos(theta * TMath::DegToRad());
185  // set distance for ring = distance between origin and centre of ring
186  SetDistance(trans_2);// distance in cm
187 
188  TGeoTranslation tran1(0, 0, trans_1);
189  TGeoTranslation tran2(0, 0, -trans_2);
190  TGeoHMatrix h = tran2 * rot2 * tran1 * rot1;
191  TGeoHMatrix* ph = new TGeoHMatrix(h);
192 
193  mother_vol->AddNode(det_vol, 1, ph);
194  }
195  return mother_vol;
196 }

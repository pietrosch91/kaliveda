22 // KVIDGChIoSi
23 //
24 // Z-only identification grids for INDRA Ionisation chamber-Silicon telescopes.
25 // To identify a particle with this grid, do:
26 // if( grid->IsIdentifiable(x,y) ){
27 // grid->Identify(x,y,nuc);
28 // }
29 //
30 // After attempting identification with method Identify, GetQualityCode() method returns
31 // one of the following status codes:
32 //
33 // KVIDZAGrid::kICODE0, OK
34 // KVIDZAGrid::kICODE1, slight ambiguity of Z, which could be larger
35 // KVIDZAGrid::kICODE2, slight ambiguity of Z, which could be smaller
36 // KVIDZAGrid::kICODE3, slight ambiguity of Z, which could be larger or smaller
37 // KVIDZAGrid::kICODE4, point is in between two lines of different Z, too far from either to be considered well-identified
38 // KVIDZAGrid::kICODE5, point is in between two lines of different Z, too far from either to be considered well-identified
39 // KVIDZAGrid::kICODE6, (x,y) is below first line in grid
40 // KVIDZAGrid::kICODE7, (x,y) is above last line in grid
41 // KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid
42 // KVIDGChIoSi::k_BelowPunchThrough "warning: point below punch-through line",
43 // KVIDGChIoSi::k_BelowSeuilSi "point to identify left of Si threshold line (bruit/arret ChIo?)",
44 // KVIDGChIoSi::k_LeftOfBragg "point to identify below Bragg curve. Z given is a Zmin",
45 // KVIDGChIoSi::k_RightOfEmaxSi "point to identify has E_Si > Emax_Si i.e. codeur is saturated. Unidentifiable",
46 //
47 // Correctly-identified particles with no ambiguity have quality codes < KVIDZAGrid::kICODE4.
48 // Particles with quality code KVIDGChIoSi::k_BelowPunchThrough are also in principle correctly identified,
49 // but we give a warning that the point is below the line 'Punch_through' delimiting the region in which the
50 // identification lines can be mixed with particles punching through the silicon if no additional condition
51 // is placed on e.g. the CsI detector behind the silicon not firing.
52 // Particles with quality code KVIDGChIoSi::k_LeftOfBragg are below the Bragg curve and the Z attributed
53 // is a minimum value.
54 // Particles with code KVIDZAGrid::kICODE7 are (far) above the last line of the grid, their Z is also a minimum.

155 Bool_t KVIDGChIoSi::IsIdentifiable(Double_t x, Double_t y) const
156 {
157  // Sets fStatus for particle depending on its position in the grid.
158  //
159  // Returns kTRUE if the particle corresponding to point (x=esi,y=echio) in the ChIo-Si map
160  // is identifiable i.e. if it is contained within the region delimited by
161  // - the line indicating the limit of particles stopping in the ChIo (KVIDCutLine:"Seuil_Si", accpeted:"right")
162  // - the line indicating the maximum canal/energy Silicium before saturation (KVIDCutLine:"Emax_Si", accepted:"left")
163  // Particles which do not satisfy one of the following cuts:
164  // - the line indicating the lower limit of Z identification, or Bragg curve (KVIDCutLine:"Bragg_line", accepted:"right")
165  // - the line indicating the limit of particles punching through to the CsI (KVIDCutLine:"Punch_through", accepted:"above")
166  // are identified with a quality code warning.
167 
168  Bool_t can_id = kTRUE;
169  //must be right of arret chio/seuil si line
170  if (fSeuil) {
171  can_id = fSeuil->WhereAmI(x, y, "right");
172  if (!can_id) {
173  const_cast < KVIDGChIoSi* >(this)->fICode = k_BelowSeuilSi;
174  return kFALSE;
175  };
176  }
177  //must be left of Emax_Si line
178  if (fEmaxSi) {
179  can_id = fEmaxSi->WhereAmI(x, y, "left");
180  if (!can_id) {
181  const_cast < KVIDGChIoSi* >(this)->fICode = k_RightOfEmaxSi;
182  return kFALSE;
183  };
184  }
185  return can_id;
186 }

227 void KVIDGChIoSi::BackwardsCompatibilityFix()
228 {
229  // Called after reading a grid from an ascii file.
230  // Tries to convert information written by an old version of the class:
231  //
232  //<PARAMETER> Ring min=... ----> <PARAMETER> IDTelescopes=...
233  //<PARAMETER> Ring max=...
234  //<PARAMETER> Mod min=...
235  //<PARAMETER> Mod max=...
236 
237  KVIDZAGrid::BackwardsCompatibilityFix();
238  if (fPar->HasParameter("IDTelescopes")) return;
239 
240  Warning("BackwardsCompatibilityFix",
241  "This fix no longer works correctly. Dummy ID telescopes will be associated with this grid. There will be problems.");
242  if (fPar->HasParameter("Rings")) { // && gIndra ) <== SHOULD NOT DEPEND ON KVINDRA!!!
243  KVNumberList Rings(fPar->GetStringValue("Rings"));
244  KVNumberList Modules(fPar->GetStringValue("Modules"));
245  int r, m;
246  Rings.Begin();
247  while (!Rings.End()) {
248  Modules.Begin();
249  r = Rings.Next();
250  while (!Modules.End()) {
251  m = Modules.Next();
252  KVIDTelescope* id = new KVIDTelescope();
253  id->SetName(Form("CI_SI_%02d%02d", r, m)); // gIndra->GetIDTelescope( Form("CI_SI_%02d%02d", r,m) ); <== SHOULD NOT DEPEND ON KVINDRA!!!
254  if (id) AddIDTelescope(id);
255  }
256  }
257  WriteParameterListOfIDTelescopes();
258  fPar->RemoveParameter("Rings");
259  fPar->RemoveParameter("Modules");
260  }
261  fSeuil = (KVIDLine*)GetCut("Seuil_Si");
262  fBragg = (KVIDLine*)GetCut("Bragg_line");
263  fPunch = (KVIDLine*)GetCut("Punch_through");
264  fEmaxSi = (KVIDLine*)GetCut("Emax_Si");
265  if (fSeuil)((KVIDCutLine*)fSeuil)->SetAcceptedDirection("right");
266  if (fBragg)((KVIDCutLine*)fBragg)->SetAcceptedDirection("right");
267  if (fPunch)((KVIDCutLine*)fPunch)->SetAcceptedDirection("above");
268  if (fEmaxSi)((KVIDCutLine*)fEmaxSi)->SetAcceptedDirection("left");
269  SetVarY("CI-E/PG/GG");
270  SetVarX("SI-E/PG/GG");
271  SetOnlyZId();
272 }

325 void KVIDGChIoSi::Identify(Double_t x, Double_t y, KVIdentificationResult* idr) const
326 {
327  // After identification of the particle, we adjust the quality code
328  // (if the particle was well-identified by KVIDZAGrid::Identify, i.e. with
329  // fICode<KVIDZAGrid::kICODE4) if:
330  // the particle is below the 'Bragg_line' => quality code KVIDGChIoSi::k_LeftOfBragg
331  // in this case the Z given is a minimum value
332  // the particle is below the 'Punch_through' line
333  // => quality code KVIDGChIoSi::k_BelowPunchThrough, but the particle
334  // is in principle well identified, if a condition has been applied
335  // to make sure that the detector behind the silicon (i.e. CsI) did not
336  // fire.
337 
338  KVIDZAGrid::Identify(x, y, idr);
339  // check Bragg & punch through for well identified particles
340  if (fICode < KVIDZAGrid::kICODE4) {
341  //identified particles below (left of) Bragg line : Z is a Zmin
342  if (fBragg && fBragg->WhereAmI(x, y, "left")) {
343  const_cast<KVIDGChIoSi*>(this)->fICode = k_LeftOfBragg;
344  idr->SetComment("Point to identify below Bragg curve. Z given is a Zmin");
345  }
346  //if a particle is well-identified (i.e. not too far from the identification lines)
347  //but it lies below the 'Punch_through' line, we give it a warning code
348  if (fPunch && fPunch->WhereAmI(x, y, "below")) {
349  const_cast<KVIDGChIoSi*>(this)->fICode = k_BelowPunchThrough;
350  idr->SetComment("warning: point below punch-through line");
351  }
352  idr->IDquality = fICode;
353  } else if (fICode == KVIDZAGrid::kICODE7) {
354  // for particles above last line in grid, check if we are in fact in the Bragg zone
355  if (fBragg && fBragg->WhereAmI(x, y, "left")) {
356  const_cast<KVIDGChIoSi*>(this)->fICode = k_LeftOfBragg;
357  idr->SetComment("Point to identify below Bragg curve. Z given is a Zmin");
358  idr->IDOK = kTRUE;
359  idr->IDquality = fICode;
360  }
361 
362  }
363 }

19 // ***** Base class for Global variables with 1 value ******
20 //
21 // This class is a base abstact class for the management of global variable with only
22 // one value. A global variable class deriving from KVVarGlob1 has to provide the following methods:
23 // virtual void Init(void) initiates the internal fields
24 // virtual void Reset(void) resets internal fields before the treatment
25 // virtual void Fill(KVNucleus *c) tells how to compute the global variable using a KVNucleus
26 //
27 // A call to the protected method FillVar(Double_t v) adds v to the current value.
28 // A call to the GetValue() method returns the value of the variable.
29 // Have a look to the KVEkin class to have an example of how to implement a global variable deriving from
30 // KVVarGlob1. Look at KVVarGlob class to have an example of use.
31 //

284 void KVVarGlob1::MakeClass(const Char_t* classname, const Char_t* classdesc, int type)
285 {
286  //Creates skeleton '.h' and '.cpp' files for a new global variable class which
287  //inherits from this class. Give a name for the new class and a short description
288  //which will be used to document the class.
289  //
290  // By default the new class will be of type 1-body. A Fill(KVNucleus*) method
291  // will be generated which the user should complete.
292  // For a 2-body variable, call MakeClass with type = KVVarGlob::kTwoBody.
293  // A skeleton Fill2(KVNucleus*,KVNucleus*) method will be generated.
294  // For a N-body variable, call MakeClass with type = KVVarGlob::kNBody.
295  // A skeleton FillN(KVEvent*) method will be generated.
296 
297  // basic class template
298  KVClassFactory cf(classname, classdesc, "KVVarGlob1", kTRUE);
299 
300  KVString body;
301 
302  // add 'init' method
303  KVVarGlob::AddInitMethod(classname, cf, body, type);
304 
305  // add 'Fill', 'Fill2', or 'FillN' method
306  KVVarGlob::AddFillMethod(cf, type);
307 
308  // body of 'Fill', 'Fill2', or 'FillN' method
309  KVVarGlob1::FillMethodBody(body, type);
310 
311  // add body of method
312  KVVarGlob::AddFillMethodBody(cf, body, type);
313 
314  cf.GenerateCode();
315 }

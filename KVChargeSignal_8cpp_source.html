419 Double_t KVChargeSignal::GetMaxFluctuationsWindow(Double_t* window, Int_t width)
420 {
421  if (!window) return 0;
422  TH1F* h2 = new TH1F("windows", "windows", (TMath::Nint(GetAmplitude()) + 1), GetYmin() - 0.5, GetYmax() + 0.5);
423  TGraph* gmoy = new TGraph();
424  TGraph* grms = new TGraph();
425  if (bidim) delete bidim;
426  bidim = new TGraph();
427  Double_t xx, yy;
428  Int_t nsamples = GetN() / width;
429 
430  printf("npoints=%d - nsamples=%d - width=%d\n", GetN(), nsamples, width);
431 
432  Double_t rms_max = 0;
433  Int_t irms_max = -1;
434  for (Int_t ii = 0; ii < nsamples; ii += 1) {
435  h2->Reset();
436  Double_t xinf = 0;
437  for (Int_t nn = 0; nn < width; nn += 1) {
438  GetPoint(width * ii + nn, xx, yy);
439  if (nn == 0) xinf = xx;
440  h2->Fill(yy);
441  }
442  Double_t xmoy = xinf;
443  Double_t rms = h2->GetRMS();
444  Double_t mean = h2->GetMean();
445  printf("%d %lf %lf %lf\n", ii, xmoy, rms, mean);
446  gmoy->SetPoint(ii, xmoy, mean);
447  grms->SetPoint(ii, xmoy, rms);
448  bidim->SetPoint(ii, mean, rms);
449  if (rms > rms_max) {
450  //endroit ou les fluctuations sont maximums
451  rms_max = h2->GetRMS();
452  irms_max = ii;
453  }
454  }
455 
456  Int_t ideb = irms_max;
457  Double_t ybefore = rms_max;
458  Double_t yafter = -1;
459  Int_t iparcours = ideb - 1;
460 
461  grms->GetPoint(iparcours, xx, yafter);
462  while (yafter < ybefore) {
463  ybefore = yafter;
464  iparcours -= 1;
465  grms->GetPoint(iparcours, xx, yafter);
466  }
467  Double_t xgauche = 0;
468  grms->GetPoint(iparcours + 1, xgauche, yy);
469  //xgauche-=width/2.;
470  Double_t ybas = 0;
471  gmoy->GetPoint(iparcours + 1, xx, ybas);
472 
473  ideb = irms_max;
474  ybefore = rms_max;
475  iparcours = ideb + 1;
476 
477  grms->GetPoint(iparcours, xx, yafter);
478  while (yafter < ybefore) {
479  ybefore = yafter;
480  iparcours += 1;
481  grms->GetPoint(iparcours, xx, yafter);
482  }
483  Double_t xdroite = 0;
484  grms->GetPoint(iparcours - 1, xdroite, yy);
485  //xdroite+=width/2.;
486  Double_t yhaut = 0;
487  gmoy->GetPoint(iparcours - 1, xx, yhaut);
488 
489  window[0] = xgauche;
490  window[1] = xdroite;
491  window[2] = ybas;
492  window[3] = yhaut;
493 
494  delete h2;
495  delete gmoy;
496  delete grms;
497  return rms_max;
498 }

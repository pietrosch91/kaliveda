101 void KVElementDensityTable::Initialize()
102 {
103 
104  TString dfile;
105  dfile.Form("%s.DataFile", GetName());
106  TString cl_path;
107  if (!KVBase::SearchKVFile(gEnv->GetValue(dfile.Data(), ""), cl_path, "data")) {
108  Error("Initialize", "No file found for %s", GetName());
109  return;
110  } else {
111 // Info("Initialize","%s will be read",gEnv->GetValue(dfile.Data(),""));
112  }
113  SetTitle(gEnv->GetValue(dfile.Data(), ""));
114 
115  Int_t ntot = 0;
116  nucMap = new TMap(50, 2);
117  KVFileReader* fr = new KVFileReader();
118  fr->OpenFileToRead(cl_path.Data());
119 
120  //Premier passage
121  //Lecture du nombre de noyaux a enregistrer
122  while (fr->IsOK()) {
123 
124  fr->ReadLine(" \t");
125  if (fr->GetCurrentLine().IsNull()) {
126  break;
127  } else if (fr->GetNparRead() == 0) {
128  break;
129  } else if (fr->GetReadPar(0).BeginsWith("//")) {
130 
131  kcomments += fr->GetCurrentLine();
132  kcomments += "\n";
133 
134  } else {
135  Int_t zz = fr->GetIntReadPar(4);
136  Int_t aa = 2 * zz + 1;
137  GiveIndexToNucleus(zz, aa, ntot);
138  ntot += 1;
139  }
140  }
141 
142  if (!fr->PreparForReadingAgain()) return;
143 
144 // Info("Initialize","Set up map for %d nuclei", ntot);
145  CreateTable(ntot);
146 
147  KVElementDensity* lf = 0;
148  ntot = 0;
149  while (fr->IsOK()) {
150  fr->ReadLine(" \t");
151  if (fr->GetCurrentLine().IsNull()) {
152  break;
153  } else if (fr->GetNparRead() == 0) {
154  break;
155  } else if (fr->GetReadPar(0).BeginsWith("//")) { }
156  else {
157  CreateElement(ntot);
158  lf = (KVElementDensity*)GetCurrent();
159 
160  KVString unit = fr->GetReadPar(1);
161  Double_t units = KVUnits::g / KVUnits::cc;
162  if (unit == "g/L") {
163  lf->SetIsGas();
164  units = KVUnits::g / KVUnits::litre;
165  }
166  Double_t val = fr->GetDoubleReadPar(0);
167  lf->SetValue(val * units);
168  lf->SetElementSymbol(fr->GetReadPar(3));
169  lf->SetElementName(fr->GetReadPar(2));
170  lf->SetZ(fr->GetIntReadPar(4));
171  ntot += 1;
172  }
173 
174  }
175 
176 // Info("Initialize","table initialised correctly for %d/%d nuclei", ntot,GetNumberOfNuclei());
177  fr->CloseFile();
178  delete fr;

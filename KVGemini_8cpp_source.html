97 void KVGemini::DecaySingleNucleus(KVSimNucleus& toDecay, KVSimEvent* decayProducts)
98 {
99  // Calculate decay products of excited nucleus toDecay
100  // Takes into account Z, A, E*, v and angular momentum of nucleus.
101  // Adds all decay products to decayProducts event:
102  // - call decayProducts->Clear() before calling this method
103  // if you just want to keep the products of a single nucleus
104  //
105  // If there is a problem with the decay of the nucleus,
106  // we throw an exception of type gemini_bad_decay
107 
108  CNucleus CN(toDecay.GetZ(), toDecay.GetA());
109  try {
110  CN.setCompoundNucleus(toDecay.GetExcitEnergy(), toDecay.GetAngMom().Mag());
111  // set velocity
112  CN.setVelocityCartesian(toDecay.GetVelocity().X(), toDecay.GetVelocity().Y(), toDecay.GetVelocity().Z());
113 
114  // set angle of spin axis
115  CAngle ang(toDecay.GetAngMom().Theta(), toDecay.GetAngMom().Phi());
116  CN.setSpinAxis(ang);
117 
118  CN.decay();
119  }
120  catch (std::exception& e) {
121  Info("DecaySingleNucleus", "Caught std::exception: %s", e.what());
122  Info("DecaySingleNucleus", "While decaying: Z=%d A=%d E*=%g S=%g", toDecay.GetZ(), toDecay.GetA(), toDecay.GetExcitEnergy(), toDecay.GetAngMom().Mag());
123  CN.reset();
124  throw gemini_bad_decay();
125  }
126  catch (...) {
127  Info("DecaySingleNucleus", "Caught unknown exception");
128  Info("DecaySingleNucleus", "While decaying: Z=%d A=%d E*=%g S=%g", toDecay.GetZ(), toDecay.GetA(), toDecay.GetExcitEnergy(), toDecay.GetAngMom().Mag());
129  CN.reset();
130  throw gemini_bad_decay();
131  }
132 
133  if (CN.abortEvent) { //problem with decay?
134  // throw exception on gemini++ error
135  CN.reset();
136  Info("DecaySingleNucleus", "Bad Gemini decay (CNucleus::abortEvent=true)");
137  Info("DecaySingleNucleus", "While decaying: Z=%d A=%d E*=%g S=%g", toDecay.GetZ(), toDecay.GetA(), toDecay.GetExcitEnergy(), toDecay.GetAngMom().Mag());
138  throw gemini_bad_decay();
139  }
140 
141  CNucleus* nuc = CN.getProducts(0);
142 
143  while (nuc) {
144 
145  KVNucleus* N = decayProducts->AddParticle();
146  N->SetParameter("GEMINI_PARENT_INDEX", part_index); //set parameter with index of parent nucleus in 'hot' event
147  N->SetZ(nuc->iZ);
148  N->SetA(nuc->iA);
149  N->SetExcitEnergy(nuc->fEx);
150  float* v = nuc->getVelocityVector();
151  TVector3 vv(v);
152  N->SetVelocity(vv);
153  nuc = CN.getProducts();
154 
155  }
156 
157  CN.reset();//<-- essential!
158 }

170 void KVGemini::DecayEvent(const KVSimEvent* hot, KVSimEvent* cold)
171 {
172  // Perform statistical decay of all nuclei in 'hot' event
173  // Takes into account Z, A, E*, v and spin of nuclei.
174  // Adds all decay products to 'cold' output event
175  //
176  // If there is a problem with the decay of any of the nuclei in the event,
177  // we throw an exception of type gemini_bad_decay
178 
179  cold->Clear();
180  KVSimNucleus* hotnuc;
181  part_index = 1;
182  while ((hotnuc = (KVSimNucleus*)const_cast<KVSimEvent*>(hot)->GetNextParticle())) {
183  try {
184  DecaySingleNucleus(*hotnuc, cold);
185  ++part_index;
186  }
187  catch (...) {
188  // rethrow any exceptions
189  throw;
190  }
191  }
192 }

201 void KVGemini::FillTreeWithEvents(KVSimNucleus& toDecay, Int_t nDecays, TTree* theTree, TString branchname)
202 {
203  // Perform nDecays decays of nucleus toDecay and write the events
204  // containing all decay products of each decay in theTree
205  // If no branchname is given, branch is called "gemini"
206 
207  if (branchname == "") branchname = "gemini";
208  KVSimEvent* decayProducts = new KVSimEvent;
209  KVEvent::MakeEventBranch(theTree, branchname, "KVSimEvent", &decayProducts);
210 
211  while (nDecays--) {
212  decayProducts->Clear();
213  try {
214  DecaySingleNucleus(toDecay, decayProducts);
215  }
216  catch (exception& e) {
217  continue;
218  }
219  theTree->Fill();
220  std::cout << "\xd" << "Gemini++ processing, " << nDecays << " decays left ..." << std::flush;
221  }
222  std::cout << std::endl;
223 
224 }

235 void KVGemini::FillTreeWithArrays(KVSimNucleus& toDecay, Int_t nDecays, TTree* theTree, TString mode)
236 {
237  // Perform nDecays decays of nucleus toDecay and write the events
238  // containing all decay products of each decay in theTree
239  // in the form of simple arrays.
240  // mode = "EThetaPhi" (default) or "V" and decides what informations are
241  // in the arrays (see KVEvent::FillArraysV and KVEvent::FillArraysEThetaPhi)
242 
243  Bool_t ethetaphi = (mode == "EThetaPhi");
244 #define MAX_GEMINI_ARRAY_SIZE 200
245  Int_t mult, Z[MAX_GEMINI_ARRAY_SIZE], A[MAX_GEMINI_ARRAY_SIZE];
246  Double_t x[MAX_GEMINI_ARRAY_SIZE], y[MAX_GEMINI_ARRAY_SIZE], z[MAX_GEMINI_ARRAY_SIZE];
247  KVSimEvent* decayProducts = new KVSimEvent;
248 
249  theTree->Branch("Multiplicity", &mult, "mult/I");
250  theTree->Branch("Z", Z, "Z[mult]/I");
251  theTree->Branch("A", A, "A[mult]/I");
252  if (ethetaphi) {
253  theTree->Branch("Energy", x, "Energy[mult]/D");
254  theTree->Branch("Theta", y, "Theta[mult]/D");
255  theTree->Branch("Phi", z, "Phi[mult]/D");
256  }
257  else {
258  theTree->Branch("Vx", x, "Vx[mult]/D");
259  theTree->Branch("Vy", y, "Vy[mult]/D");
260  theTree->Branch("Vz", z, "Vz[mult]/D");
261  }
262  while (nDecays--) {
263  decayProducts->Clear();
264  try {
265  DecaySingleNucleus(toDecay, decayProducts);
266  }
267  catch (exception& e) {
268  continue;
269  }
270  if (ethetaphi) decayProducts->FillArraysEThetaPhi(mult, Z, A, x, y, z);
271  else decayProducts->FillArraysV(mult, Z, A, x, y, z);
272  theTree->Fill();
273  std::cout << "\xd" << "Gemini++ processing, " << nDecays << " decays left ..." << std::flush;
274  }
275  std::cout << std::endl;
276 }

284 Float_t KVGemini::GetMaxSpinWithFissionBarrier(int z, int a)
285 {
286  // Maximum angular momentum with non-zero fission barrier (Sierk model)
287  // Only for 19<=Z<=111
288  if (z < 19 || z > 111) {
289  Error("GetMaxSpinWithFissionBarrier", "Only use for 19<=Z<=111");
290  return -1;
291  }
292  float amin = 1.2 * z + 0.01 * pow(z, 2);
293  float amax = 5.8 * z - 0.024 * pow(z, 2);
294 
295  if (a < amin || a > amax) {
296  Error("GetMaxSpinWithFissionBarrier", "Only valid for Z=%d with %d<=A<=%d",
297  z, (int)amin, (int)amax);
298  return -1;
299  }
300  return CYrast::instance()->getJmaxSierk(z, a);
301 }

72 void KVIVUpDater::SetIDGrids(UInt_t run)
73 {
74  // Set identification grid or identification function (KVTGID)
75  // for all ID telescopes for this run.
76  // The global ID grid manager gIDGridManager is used to set the
77  // grids. The static function KVRTGIDManager::SetIDFuncInTelescopes
78  // is used to set the ID functions. First, any previously set grids/functions are removed.
79  // Then all grids/functions for current run are set in the associated ID telescopes.
80 
81  Info("KVIVUpDater::SetIDGrids", "Setting Identification Grids/Functions");
82  TIter next_idt(gMultiDetArray->GetListOfIDTelescopes());
83 
84  KVIDTelescope* idt = NULL;
85 #ifdef WITH_FITLTG
86  KVRTGIDManager* rtgidm = NULL;
87 #endif
88 
89  while ((idt = (KVIDTelescope*) next_idt())) {
90  idt->RemoveGrids();
91 #ifdef WITH_FITLTG
92  if (idt->InheritsFrom("KVRTGIDManager")) {
93  rtgidm = (KVRTGIDManager*)idt->IsA()->DynamicCast(KVRTGIDManager::Class(), idt);
94  rtgidm->RemoveAllTGID();
95  }
96 #endif
97 
98  }
99  gMultiDetArray->SetGridsInTelescopes(run);
100 #ifdef WITH_FITLTG
101  KVRTGIDManager::SetIDFuncInTelescopes(run);
102 #endif
103 }

127 void KVIVUpDater::SetParameters(UInt_t run)
128 {
129  //Set the parameters of INDRA and VAMOS for this run.
130  //
131  //For INDRA, this will:
132  // set the multiplicity trigger of gIndra using the database value for the run
133  // set special detector gains for run (if any)
134  // set the target corresponding to the run
135  // set the ChIo pressures for the run
136  // set calibration parameters for the run
137  // set the correction of pedestals
138  // set identification parameters for the run
139  //
140  //For VAMOS, this will reset and set:
141  // the rotation angle of VAMOS around the target for the run
142  // the high frequency of the beam for the run
143  // the position and the type of the stripping foil for the run
144  // reference magnetic rigidity for the run
145  // calibration parameters for the run
146  // identification parameters for the run
147 
148  KVINDRAUpDater::SetParameters(run);
149  SetChVoltRefGains();
150  KVDBRun* kvrun = gIndraDB->GetRun(run);
151  if (!kvrun) return;
152  SetPedestalCorrections(kvrun);
153 
154  if (!gVamos) {
155  Error("KVIVUpDater::SetParameters", "VAMOS is not found ( gVamos = NULL )");
156  return;
157  }
158 
159  gVamos->ResetParameters();
160  SetVamosCalibAndConfParams(kvrun);
161 }

171 void KVIVUpDater::SetPedestalCorrections(KVDBRun* run)
172 {
173  // For any detectors, having a pedestal correction defined
174  // for their associated QDC and one of their signals, will have
175  // their correction corresponding to the signal, set to the value
176  // for the run.
177 
178  Info("KVIVUpDater::SetPedestalCorrections", "Setting pedestal corrections");
179 
180  KVRList* dp_list = run->GetLinks("DeltaPedestal");
181  if (!dp_list) return;
182 
183  KVDBParameterSet* dp = NULL;
184  TIter next_dp(dp_list);
185  TString QDCnum;
186 
187  // Loop over INDRA detectors
188  KVINDRADetector* det = NULL;
189  TIter next_det(gMultiDetArray->GetDetectors());
190  while ((det = (KVINDRADetector*)next_det())) {
191 
192  // Initializing each ACQ parameter pedestal correction for
193  // all detectors
194  KVACQParam* acqpar = NULL;
195  TIter next_acqp(det->GetACQParamList());
196  while ((acqpar = (KVACQParam*)next_acqp()))
197  acqpar->SetDeltaPedestal(0.);
198 
199  // Set the pedestal correction if a value exists for
200  // the QDC associated to this detector
201  next_dp.Reset();
202  while ((dp = (KVDBParameterSet*)next_dp())) {
203  QDCnum = dp->GetName();
204  QDCnum.Remove(0, 3);
205  if (det->GetNumeroCodeur() == QDCnum.Atoi()) {
206  acqpar = det->GetACQParam(dp->GetTitle());
207  acqpar->SetDeltaPedestal(dp->GetParameter());
208  }
209  }
210  }
211 }

223 void KVIVUpDater::SetVamosCalibAndConfParams(KVDBRun* run)
224 {
225  // For a given run:
226  // - set calibration parameters to the calibrators of the detectors
227  // placed at the focal plane of VAMOS (HarpeeSi, HarpeeIC, SeD, ...);
228  // - set parameter of a detector by calling setter method of the
229  // detector (pedestal, Moulton PHD parameters, ...).
230  // - set configuration parameters (Brho, rotation angle, ...) of VAMOS.
231 
232  Info("KVIVUpDater::SetVamosCalibAndConfParams", "Setting VAMOS calibration parameters");
233 
234  if (!gVamos) {
235  Error("KVIVUpDater::SetVamosCalibAndConfParams", "VAMOS is not found ( gVamos = NULL )");
236  return;
237  }
238 
239  KVRList* list = run->GetLinks("VAMOS calib. & conf.");
240  if (!list) return;
241 
242  TObject* obj = NULL;
243  KVDBParameterSet* par = NULL;
244  TIter next(list);
245  TString formula, partype;
246 
247  while ((par = (KVDBParameterSet*)next())) {
248  formula = par->GetTitle();
249  partype = formula;
250  Int_t fidx = formula.Index("f(x)=");
251  if (fidx > 0) {
252  formula.Remove(0, fidx + 5);
253  partype.Remove(fidx - 1);
254  }
255  partype.Append(" ");
256  partype.Append(par->GetName());
257  KVCalibrator* cal = gVamos->GetCalibrator(partype.Data());
258  // Case 1: parameter associated to a calibrator
259  if (cal) {
260  if (cal->InheritsFrom("KVFunctionCal"))
261  ((KVFunctionCal*)cal)->SetExpFormula(formula.Data());
262 
263  if (cal->GetNumberParams() != par->GetParameter(0)) {
264  Error("KVIVUpDater::SetVamosCalibAndConfParams",
265  "In the database, different number of parameters for the calibrator %s (%s)", cal->ClassName(), cal->GetType());
266  continue;
267  }
268 
269  for (Int_t i = 0; i < par->GetParameter(0); i++) {
270  cal->SetParameter(i, TString(par->GetParamName(i + 1)).Atof());
271  cal->SetStatus(kTRUE);
272  }
273 
274  continue;
275  }
276 
277 
278  // Case 2: parameter associated to a setter method of a detector or of VAMOS
279  if (!(obj = gVamos->GetDetector(par->GetName()))) {
280  if (!strcmp(gVamos->GetName(), par->GetName())) obj = gVamos;
281  else {
282  Error("KVIVUpDater::SetVamosCalibAndConfParams",
283  "The parameter %s is not associated to an existing calibrator, detector or VAMOS",
284  partype.Data());
285  continue;
286  }
287  }
288 
289  TString meth, arg;
290  meth.Form("Set%s", par->GetTitle());
291  for (Int_t i = 0; i < par->GetParameter(0); i++) {
292  // arg+=par->GetParameter( i+1 );
293  arg += par->GetParamName(i + 1);
294  arg += ",";
295  }
296  arg.Remove(TString::kTrailing, ',');
297  if (!obj->IsA()->GetMethod(meth.Data(), arg.Data())) {
298 
299  Error("KVIVUpDater::SetVamosCalibAndConfParams",
300  "Impossible to call %s::%s(%s) for detector %s",
301  obj->ClassName(), meth.Data(), arg.Data(), obj->GetName());
302  continue;
303  }
304  obj->Execute(meth.Data(), arg.Data());
305  }
306 }

496 void KVIVUpDater::SetChVoltRefGains()
497 {
498  // For any detector, the gain is already taken into account
499  // in the Channel->Volt calibration, then the reference gain of the
500  // calibrator (KVChannelVolt) must be equal to the gain. (see
501  // KVChannelVolt::Compute()).
502 
503  Info("KVIVUpDater::SetChVoltRefGains", "Setting channel->Volt calibrator reference gain = detector gain");
504 
505  KVDetector* det = NULL;
506  TIter next_det(gMultiDetArray->GetDetectors());
507  KVSeqCollection* sublist = NULL;
508  while ((det = (KVDetector*)next_det())) {
509  if (!det->GetListOfCalibrators()) continue;
510  sublist = det->GetListOfCalibrators()->GetSubListWithClass("KVChannelVolt");
511  sublist->R__FOR_EACH(KVChannelVolt, SetGainRef)(det->GetGain());
512  SafeDelete(sublist);
513  }
514 }

62 void KVTenseur3::Diago(void)
63 {
64 //
65 // Routine de diagonalisation en utilisant TMatrix
66 //
67  TMatrix matrice(3, 3);
68  TVector vapro(3);
69  TMatrix vepro(3, 3);
70  Int_t i, j;
71 
72 #if ROOT_VERSION_CODE >= ROOT_VERSION(4,0,0)
73 
74  Double_t check_sum = 0.0;
75  //check_sum used to make sure matrix has non-zero elements
76  //otherwise call to matrice.Eigenvectors never returns !!
77  for (i = 0; i < 3; i++)
78  for (j = 0; j < 3; j++) {
79  matrice(i, j) = tenseur[3 * i + j];
80  check_sum += tenseur[3 * i + j];
81  }
82 
83  if (check_sum > 0.0)
84  vepro = matrice.EigenVectors(vapro);
85 
86  //On va trier les valeurs propres par odre decroissant
87  int itri[3] = { 0, 1, 2 };
88  Bool_t ok = kFALSE;
89  while (!ok) {
90  ok = kTRUE;
91  for (Int_t ii = 0; ii < 2; ii++)
92  if (vapro(itri[ii]) > vapro(itri[ii + 1])) {
93  int tmp = itri[ii];
94  itri[ii] = itri[ii + 1];
95  itri[ii + 1] = tmp;
96  ok = kFALSE;
97  }
98  }
99  //
100 
101  for (Int_t ii = 0; ii < 3; ii++) {
102  i = itri[ii];
103  for (j = 0; j < 3; j++)
104  vep[3 * ii + j] = vepro(j, i);
105  vap[ii] = vapro(i);
106  }
107 
108  is_diago = 1;
109 #else
110 
111  Double_t check_sum = 0.0;
112  //check_sum used to make sure matrix has non-zero elements
113  //otherwise call to matrice.Eigenvectors never returns !!
114  for (i = 0; i < 3; i++)
115  for (j = 0; j < 3; j++) {
116  matrice(i, j) = tenseur[3 * i + j];
117  check_sum += tenseur[3 * i + j];
118  }
119 
120  if (check_sum > 0.0)
121  vepro = matrice.EigenVectors(vapro);
122 
123  for (i = 0; i < 3; i++)
124  for (j = 0; j < 3; j++)
125  vep[3 * j + i] = vepro(i, 2 - j);
126  for (i = 0; i < 3; i++)
127  vap[i] = vapro(2 - i);
128 
129  is_diago = 1;
130 #endif
131 }

456 Double_t KVTenseur3::GetVapNorm(Int_t i)
457 {

621 void KVTenseur3::GetRotation(TRotation& rot)
622 {
623  //Sets rotation matrix corresponding to eigenvectors of diagonalized tensor.
624  //According to ROOT physics vectors convention, this is an 'active' rotation
625  //(see TRotation class description). Note that KVParticle::SetFrame automatically
626  //uses this matrix in such a way as to describe particles in the coordinate frame
627  //corresponding to the rotated axes.
628  //
629  //By convention, the three eigenvectors (V1, V2, V3) are numbered in increasing
630  //order of size of the corresponding eigenvalue (V3 <--> largest eignevalue <--> "flow" axis).
631  //The new coordinate axes are defined (arbitrarily) as follows :
632  // V1 ---> X'
633  // V2 ---> Y'
634  // V3 ---> Z'
635 
636  //make orthonormal basis vectors from 3 eignevectors
637  TVector3 V1 = GetVep(1);
638  TVector3 V2 = GetVep(2);
639  TVector3 V3 = GetVep(3);
640  if (V3.Z() < 0)
641  V3 = -V3;
642  rot.MakeBasis(V1, V2, V3);
643  //set rotation matrix
644  rot.SetToIdentity();
645  rot.RotateAxes(V1, V2, V3);
646 }

25 <h2>KVParticleCondition</h2>
26 <h4>Implements parser of particle selection criteria</h4>
27 
28 These must be valid C++ expressions using _NUC_ instead and in place of
29 a pointer to the particle to be tested. Note that the methods used in the selection
30 do not have to be limited to the KVNucleus class. The 'real' class of the object
31 passed to Test() will be used to cast the base pointer (KVNucleus*) up to the
32 required pointer type at execution. In this case, you must call the method
33 SetParticleClassName() with the name of the class to use in the cast.
34 
35 Conditions can be combined using Boolean '&&' and '||' operators.
36 
37 Note that the first time Test() is called we generate and compile a class derived
38 from KVParticleCondition which implements the required test. For all subsequent
39 calls it is this compiled code which is used, greatly increasing speed of
40 execution.

265 void KVParticleCondition::Set(const Char_t* cond)
266 {
267  //Set particle condition criteria.
268  //
269  //These must be valid C++ expressions using _NUC_ instead and in place of
270  //a pointer to the particle to be tested. Note that the methods used in the selection
271  //do not have to be limited to the KVNucleus class. The 'real' class of the object
272  //passed to Test() will be used to cast the base (KVNucleus) pointer up to the
273  //required pointer type at execution.
274 
275  fCondition = cond;
276  Ssiz_t ind = fCondition.Index(";");
277  if (ind < 0) {
278  fCondition_raw = fCondition;
279  fCondition += ";"; //we add a ";" if there isn't already
280  } else {
281  fCondition_raw = fCondition.Strip(TString::kTrailing, ';');
282  }
283  fCondition_brackets = "(" + fCondition_raw + ")";
284 }

432 void KVParticleCondition::AddExtraInclude(const Char_t* inc_file)
433 {
434  //Optimisation of KVParticleCondition::Test implies the automatic generation
435  //of a new class which implements the selection required by the user (see Optimize).
436  //
437  //If the user's condition depends on objects of classes other than the family
438  //of particle classes (TLorentVector <- KVParticle <- KVNucleus ...) there will
439  //not be by default the necessary '#include' directive for the classes in question
440  //in the generated class; the required plugin for Test() to function will not
441  //load. In this case, the user should call this method with the name of each
442  //'#include' file to be added to the class implementation.
443  //
444  //Example:
445  // KVParticleCondition p("_NUC_->GetVpar()>=gDataAnalyser->GetKinematics()->GetNucleus(1)->GetVpar()");
446  //
447  //Optimization will not work, as the 'gIndra' pointer is declared in KVINDRA.h, which
448  //is not a default '#include' file when the optimised class is generated; similarly,
449  //gIndra->GetSystem() returns a pointer to a KVDBSystem : => #include "KVDBSystem.h";
450  //and KVDBSystem::GetKinematics() returns a pointer to a KV2Body : => #include "KV2Body.h".
451  //Therefore, for this condition to work, the user must first call the methods :
452  //
453  // p.AddExtraInclude("KVINDRA.h");
454  // p.AddExtraInclude("KV2Body.h");
455  // p.AddExtraInclude("KVDBSystem.h");
456  //
457  //before the first call to p.Test() (when optimization occurs).
458 
459  CreateClassFactory();
460  cf->AddImplIncludeFile(inc_file);
461 }

503 void KVParticleCondition::Optimize()
504 {
505  //Generate a new class which inherits from KVParticleCondition but having a Test()
506  //method which tests explicitly the condition which is set by the user.
507  //
508  //If needed, the KVNucleus pointer argument will be upcasted to the type given to SetParticleClassName().
509  //
510  //The new class is added to the list of plugins of type KVParticleCondition,
511  //then an instance of the class is generated and a pointer to it stored in
512  //member KVParticleCondition::fOptimal.
513  //
514  //This object is then used in the Test() method of this object to test the condition.
515  //
516  //If compilation fails, the condition will evaluate to kFALSE for all subsequent calls.
517 
518  /* check that the same condition has not already been optimized */
519  fOptimal = (KVParticleCondition*)fgOptimized.FindObject(GetName());
520  if (fOptimal) {
521  //Info("Optimize", "Using existing optimized condition %p", fOptimal);
522  fOptimal->fNUsing++;
523  fOptOK = kTRUE;
524  return;
525  }
526  Info("Optimize", "Optimization of KVParticleCondition : %s", fCondition.Data());
527 
528  CreateClassFactory();
529  //add Test() method
530  cf->AddMethod("Test", "Bool_t");
531  cf->AddMethodArgument("Test", "KVNucleus*", "nuc");
532  cf->AddHeaderIncludeFile("KVNucleus.h");
533 
534  //write body of method
535  KVString body(" //Optimized Test method for particle condition\n");
536  KVString pointer = "nuc";
537  if (fClassName != "") {
538  pointer.Form("((%s*)nuc)", fClassName.Data());
539  //upcasting pointer - we need to add corresponding #include to '.cpp' file
540  cf->AddImplIncludeFile(Form("%s.h", fClassName.Data()));
541  }
542  KVString tmp;
543  tmp = fCondition;
544  tmp.ReplaceAll("_NUC_", pointer.Data());
545  body += " return ";
546  body += tmp;
547 
548  cf->AddMethodBody("Test", body);
549 
550  //generate .cpp and .h for new class
551  cf->GenerateCode();
552 
553  //add plugin for new class
554  gROOT->GetPluginManager()->AddHandler("KVParticleCondition", cf->GetClassName(), cf->GetClassName(),
555  Form("%s+", cf->GetImpFileName()), Form("%s()", cf->GetClassName()));
556  //load plugin
557  TPluginHandler* ph;
558  if (!(ph = LoadPlugin("KVParticleCondition", cf->GetClassName()))) {
559  Error("Optimize", " *** Optimization failed for KVParticleCondition : %s", fCondition.Data());
560  Error("Optimize", " *** Use method AddExtraInclude(const Char_t*) to give the names of all necessary header files for compilation of your condition.");
561  Fatal("Optimize", " *** THIS CONDITION WILL BE EVALUATED AS kFALSE FOR ALL PARTICLES!!!");
562  delete cf;
563  cf = 0;
564  //we set fOptimal to a non-zero value to avoid calling Optimize
565  //every time that Test() is called subsequently.
566  fOptimal = this;
567  fOptOK = kFALSE;
568  return;
569  }
570  fOptOK = kTRUE;
571  delete cf;
572  cf = 0;
573  //execute constructor
574  fOptimal = (KVParticleCondition*) ph->ExecPlugin(0);
575 
576  Info("Optimize", "fOptimal = %p", fOptimal);
577  if (!fOptimal) {
578  Error("Optimize", " *** Optimization failed for KVParticleCondition : %s", fCondition.Data());
579  Error("Optimize", " *** Use method AddExtraInclude(const Char_t*) to give the names of all necessary header files for compilation of your condition.");
580  Fatal("Optimize", " *** THIS CONDITION WILL BE EVALUATED AS kFALSE FOR ALL PARTICLES!!!");
581  //we set fOptimal to a non-zero value to avoid calling Optimize
582  //every time that Test() is called subsequently.
583  fOptimal = this;
584  fOptOK = kFALSE;
585  }
586  // add to list of optimized conditions
587  fOptimal->SetName(GetName());
588  fgOptimized.Add(fOptimal);
589  fOptimal->fNUsing++;
590  Info("Optimize", "Success");
591 }

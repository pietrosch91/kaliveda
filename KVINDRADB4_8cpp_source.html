97 void KVINDRADB4::GoodRunLine()
98 {
99  //For each "good run line" in the run list file, we:
100  // add a KVINDRADBRun to the database if it doesn't already exist
101  // set properties of run objects
102  //kFirstRun & kLastRun are set
103 
104  KVRunListLine* csv_line = GetLineReader();
105 
106  //run number
107  Int_t run_n = csv_line->GetIntField("RUN");
108 
109  if (!run_n) {
110  cout << "run_n = 0 ????????? line number =" << GetRLLineNumber() <<
111  endl;
112  GetLineReader()->Print();
113  return;
114  }
115  kLastRun = TMath::Max(kLastRun, run_n);
116  kFirstRun = TMath::Min(kFirstRun, run_n);
117 
118 
119  /*********************************************
120  WE CREATE A NEW RUN AND ADD
121  IT TO THE DATABASE. WE SET ALL
122  AVAILABLE INFORMATIONS ON
123  RUN FROM THE FILE. ERROR IF
124  DBASE RUN ALREADY EXISTS =>
125  SAME RUN APPEARS TWICE
126  *********************************************/
127  KVINDRADBRun* run = (KVINDRADBRun*) GetRun(run_n);
128  if (!run) {
129 
130  run = new KVINDRADBRun(run_n);
131  AddRun(run);
132 
133  if (csv_line->HasFieldValue("EVENTS"))
134  run->SetEvents(csv_line->GetIntField("EVENTS"));
135  else
136  Error("GoodRunLine", "No events field ? run=%d", run_n);
137  if (csv_line->HasField("TRIGGER"))
138  run->SetTrigger(GetRunListTrigger("TRIGGER", "M>=%d"));
139  else
140  Error("GoodRunLine", "No TRIGGER field ? run=%d", run_n);
141  if (csv_line->HasFieldValue("RUNCOMMENT"))
142  run->SetComments(csv_line->GetField("RUNCOMMENT"));
143  if (csv_line->HasFieldValue("TSTART"))
144  run->SetStartDate(csv_line->GetField("TSTART"));
145 
146  } else {
147  Error("GoodRunLine", "Run %d already exists", run_n);
148  }
149 }

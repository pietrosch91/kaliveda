72 Bool_t KVRemoteDataSetManager::OpenAvailableDatasetsFile()
73 {
74  //Transfers file $KVROOT/KVFiles/[repository name].available.datasets
75  //from remote machine containing info on available datasets and
76  //associated subdirectories in remote data repository.
77  //Copies to local working directory and opens for reading, if all goes
78  //well (returns kTRUE).
79  //Returns kFALSE in case of problems.
80  //
81  //N.B. If 'curl' is used to transfer the file, we check for HTML code in the file
82  //It can happen that 'curl' itself does not give an error when the web server
83  //is down, but transfers a file which looks like this:
84 // <!DOCTYPE HTML PUBLIC "-//IETF//DTD HTML 2.0//EN">
85 // <html><head>
86 // <title>403 Forbidden</title>
87 // </head><body>
88 // <h1>Forbidden</h1>
89 // <p>You don't have permission to access /KaliVedaDoc/ccali.available.datasets
90 // on this server.</p>
91 // <p>Additionally, a 403 Forbidden
92 // error was encountered while trying to use an ErrorDocument to handle the request.</p>
93 // <hr>
94 // <address>Apache/2.0.53 (Unix) mod_ssl/2.0.53 OpenSSL/0.9.7a DAV/2 PHP/5.1.6 mod_python/3.2.8 Python/2.4.2 mod_jk/1.2.15 mod_perl/2.0.2 Perl/v5.8.7 Server at indra.in2p3.fr Port 80</address>
95 // </body></html>
96 
97 
98  TString http =
99  gEnv->GetValue(Form("%s.DataRepository.RemoteAvailableRuns.url",
100  fRepository->GetName()), "");
101  if (http == "") {
102  Warning("OpenAvailableDatasetsFile()",
103  "%s.DataRepository.RemoteAvailableRuns.url is not defined. See $KVROOT/KVFiles/.kvrootrc",
104  fRepository->GetName());
105  return kFALSE;
106  }
107  TString url;
108  url.Form("%s/%s", http.Data(), fCacheFileName.Data());
109 
110  //Find executable 'curl' or programme indicated by
111  //value of DataRepository.RemoteAvailableRuns.protocol
112  fCurl = gEnv->GetValue(Form
113  ("%s.DataRepository.RemoteAvailableRuns.protocol",
114  fRepository->GetName()), "curl");
115  if (!KVBase::FindExecutable(fCurl)) {
116  Error("OpenAvailableDatasetsFile",
117  "Executable for file transfer \"%s\" not found. You cannot use remote data repository.",
118  fCurl.Data());
119  return kFALSE;
120  }
121  TString cmd;
122  cmd.Form("%s -oremote.available.datasets %s", fCurl.Data(), url.Data());
123  if (gSystem->Exec(cmd.Data()))
124  return kFALSE; // problem with curl
125  fDatasets.open("remote.available.datasets");
126  if (fCurl.Contains("curl")) {
127  //check contents of file for HTML code
128  KVString line;
129  line.ReadLine(fDatasets);
130  if (line.Contains("<")) {
131  Error("OpenAvailableDatasetsFile",
132  "Remote available datasets file transferred by curl contains HTML code :");
133  //print contents of file
134  cout << line.Data() << endl;
135  line.ReadLine(fDatasets);
136  while (fDatasets.good()) {
137  cout << line.Data() << endl;
138  line.ReadLine(fDatasets);
139  }
140  Error("OpenAvailableDatasetsFile",
141  "You cannot use this remote data repository");
142  return kFALSE;
143  } else { // no HTML in file : OK
144  //reset stream to beginning of file
145  fDatasets.clear(); // clear any error flags (EOF etc.)
146  fDatasets.seekg(0, ios::beg); // set file buffer pointer to beginning of file
147  }
148  }
149  return kTRUE;
150 }

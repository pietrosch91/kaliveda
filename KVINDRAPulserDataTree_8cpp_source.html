22 <h2>KVINDRAPulserDataTree</h2>
23 <h4>Handles TTree with mean pulser data for every run</h4>
24  Create and fill tree with pulser data.
25  We look for the following two directories:
26 <pre>
27  $KVROOT/KVFiles/name_of_dataset/gene_detecteurs
28  $KVROOT/KVFiles/name_of_dataset/gene_pins
29 </pre>
30  If not found, we look for the following compressed archives:
31 <pre>
32  $KVROOT/KVFiles/name_of_dataset/gene_detecteurs.tgz
33  $KVROOT/KVFiles/name_of_dataset/gene_pins.tgz
34 </pre>
35  and if found, uncompress them ('tar -zxf').
36 <br>
37  The default names of these directories are defined in .kvrootrc by:
38 <pre>
39  KVINDRAPulserDataTree.GeneDetDir: gene_detecteurs
40  KVINDRAPulserDataTree.GenePinDir: gene_pins
41 </pre>
42  Dataset-dependent alternatives can be defined using:
43 <pre>
44  dataset_name.KVINDRAPulserDataTree.GeneDetDir: dataset_specific_value
45 </pre>
46  The first directory (gene_detecteurs) must contain 1 file per run with names like:
47 <pre>
48  run8820.gene
49 </pre>
50  These files are generated using example analysis class GetGeneMean
51  (see Examples). They contain the mean value of every acquisition parameter
52  associated with a detector in the run:
53 <pre>
54  CI_0201_GG 3095.28
55  CI_0201_PG 275.626
56  CI_0203_GG 2863.66
57  CI_0203_PG 263.308
58  CI_0205_GG 3042.83
59  etc.
60 </pre>
61  i.e. 'name of acquisition parameter' 'mean value for run'
62 <br>
63  The second directory (gene_pins) must contain 1 file per run with names like:
64 <pre>
65  run8820.genepin
66  OR run8820.laserpin
67  OR run8820.genelaserpin
68 </pre>
69  These files are generated using example analysis class GetGeneMeanPin
70  (see Examples). They contain the mean values of the acquisition parameters
71  associated with pin diodes in the run:
72 <pre>
73  PILA_01_PG_gene 792.616
74  PILA_01_PG_laser 1747.18
75  PILA_01_GG_gene 0
76  etc.
77 </pre>
78  We create a TTree with 1 branch for each detector acquisition parameter.
79  For PILA and SI_PIN parameters, we create a 'gene' and a 'laser' branch for each.

242 void KVINDRAPulserDataTree::Build()
243 {
244  // Create and fill tree with pulser data.
245  // We look for the following two directories:
246  //
247  // $KVROOT/KVFiles/name_of_dataset/gene_detecteurs
248  // $KVROOT/KVFiles/name_of_dataset/gene_pins
249  //
250  // If not found, we look for the following compressed archives:
251  //
252  // $KVROOT/KVFiles/name_of_dataset/gene_detecteurs.tgz
253  // $KVROOT/KVFiles/name_of_dataset/gene_pins.tgz
254  //
255  // and if found, uncompress them ('tar -zxf').
256  // [[ N.B. in this case the extracted directories will be deleted after reading,
257  // [[ ensuring that if new archives are supplied,
258  // [[ we always use the latest versions of files.
259  //
260  // The default names of these directories are defined in .kvrootrc by:
261  //
262  // KVINDRAPulserDataTree.GeneDetDir: gene_detecteurs
263  // KVINDRAPulserDataTree.GenePinDir: gene_pins
264  //
265  // Dataset-dependent alternatives can be defined using:
266  //
267  // dataset_name.KVINDRAPulserDataTree.GeneDetDir: dataset_specific_value
268  //
269  // The first directory (gene_detecteurs) must contain 1 file per run with names like:
270  //
271  // run8820.gene
272  //
273  // These files are generated using example analysis class GetGeneMean
274  // (see Examples). They contain the mean value of every acquisition parameter
275  // associated with a detector in the run:
276  //
277  // CI_0201_GG 3095.28
278  // CI_0201_PG 275.626
279  // CI_0203_GG 2863.66
280  // CI_0203_PG 263.308
281  // CI_0205_GG 3042.83
282  // etc.
283  //
284  // i.e. 'name of acquisition parameter' 'mean value for run'
285  //
286  // The second directory (gene_pins) must contain 1 file per run with names like:
287  //
288  // run8820.genepin
289  // OR run8820.laserpin
290  // OR run8820.genelaserpin
291  //
292  // These files are generated using example analysis class GetGeneMeanPin
293  // (see Examples). They contain the mean values of the acquisition parameters
294  // associated with pin diodes in the run:
295  //
296  // PILA_01_PG_gene 792.616
297  // PILA_01_PG_laser 1747.18
298  // PILA_01_GG_gene 0
299  // etc.
300  //
301  // We create a TTree with 1 branch for each detector acquisition parameter.
302  // For PILA and SI_PIN parameters, we create a 'gene' and a 'laser' branch for each.
303 
304  fGeneDir = new KVTarArchive(GetDirectoryName("GeneDetDir"), gDataSet->GetDataSetDir());
305  fPinDir = new KVTarArchive(GetDirectoryName("GenePinDir"), gDataSet->GetDataSetDir());
306  if (!fGeneDir->IsOK() && !fPinDir->IsOK()) {
307  Info("Build", "No data available to build pulser data tree");
308  return;
309  }
310  CreateTree();
311  ReadData();
312  delete fGeneDir;
313  delete fPinDir;
314 }

353 void KVINDRAPulserDataTree::CreateTree()
354 {
355  // Create new TTree with
356  // 1 branch 'Run' with run number
357  // 1 branch for each acquisition parameter of every detector (except time markers)
358  // 2 branches for each 'PILA_...' or 'SI_PIN...' parameter, suffixed with '_laser' and '_gene'
359  //
360  // NB if multidetector has not been built, it will be built by this method
361 
362  fArb = new TTree("PulserData", "Created by KVINDRAPulserDataTree");
363  fArb->SetDirectory(0);
364 
365  fArb->Branch("Run", &fRun, "Run/I");
366 
367  if (!gIndra) KVMultiDetArray::MakeMultiDetector(gDataSet->GetName());
368 
369  KVSeqCollection* acq_pars = gIndra->GetACQParams();
370 
371  fTab_siz = acq_pars->GetEntries() + 20;
372  fVal = new Float_t[fTab_siz];
373  fIndex = new THashTable(20, 5);
374  fIndex->SetOwner(kTRUE);
375 
376  TIter nxtACQ(acq_pars);
377  KVACQParam* ap = 0;
378  Int_t ap_num = 0;
379  KVBase* iob;
380  while ((ap = (KVACQParam*)nxtACQ())) {
381 
382  TString ap_name(ap->GetName());
383  TString ap_type(ap->GetType());
384  if (ap_name.BeginsWith("PILA") || ap_name.BeginsWith("SI_PIN")) {
385  ap_name += "_laser";
386  iob = new KVBase(ap_name.Data());
387  iob->SetNumber(ap_num);
388  fIndex->Add(iob);
389  fArb->Branch(ap_name.Data(), &fVal[ap_num++], Form("%s/F", ap_name.Data()));
390  ap_name.Form("%s%s", ap->GetName(), "_gene");
391  iob = new KVBase(ap_name.Data());
392  iob->SetNumber(ap_num);
393  fIndex->Add(iob);
394  fArb->Branch(ap_name.Data(), &fVal[ap_num++], Form("%s/F", ap_name.Data()));
395  } else if (ap_type != "T") {
396  iob = new KVBase(ap_name.Data());
397  iob->SetNumber(ap_num);
398  fIndex->Add(iob);
399  fArb->Branch(ap_name.Data(), &fVal[ap_num++], Form("%s/F", ap_name.Data()));
400  }
401  if (ap_num > fTab_siz - 2) {
402  Error("CreateTree",
403  "Number of branches to create is greater than estimated (%d). Not all parameters can be treated.",
404  fTab_siz);
405  return;
406  }
407 
408  }
409  //keep number of used 'slots' in array
410  fTab_siz = ap_num;
411 }

519 void KVINDRAPulserDataTree::ReadData()
520 {
521  // Read data for every run in dataset
522 
523  if (!fRunlist) {
524  Error("ReadData", "Must set list of runs first using SetRunList(TList*)");
525  return;
526  }
527  Info("ReadData", "Reading pulser and laser data for all runs");
528  TIter Nxt_r(fRunlist);
529  KVINDRADBRun* run = 0;
530  KVNumberList missing1, missing2;
531  while ((run = (KVINDRADBRun*)Nxt_r())) {
532  Int_t run_num = run->GetNumber();
533  //reset all array members to -1
534  for (int i = 0; i < fTab_siz; i++) fVal[i] = -1.0;
535  std::cout << "\rInfo in <KVINDRAPulserDataTree::ReadData>: Reading data for run " << run_num << std::flush;
536  UChar_t msg = ReadData(run_num);
537  if (msg & 1) missing1.Add(run_num);
538  if (msg & 2) missing2.Add(run_num);
539  }
540  std::cout << std::endl;
541  if (missing1.GetEntries())
542  Warning("ReadData", "Missing file 'run[run_num].gene' for runs: %s", missing1.GetList());
543  if (missing2.GetEntries())
544  Warning("ReadData", "Missing file 'run[run_num].[gene][laser]pin' for runs: %s", missing2.GetList());
545 }

594 Float_t KVINDRAPulserDataTree::GetMean(const Char_t* param, Int_t run)
595 {
596  // Return mean value of pulser/laser for given parameter and run.
597  // For detectors, param should be name of an acquisition parameter
598  // e.g. CI_0201_PG, CSI_1301_L, etc.
599  // For pin laser diodes, param should be name of associated acquisition parameter
600  // with either '_laser' or '_gene' appended
601  // e.g. PILA_05_PG_laser, SI_PIN1_PG_gene
602  //
603  // Returns -1.0 if no data available for this parameter/run.
604 
605  if (!fArb) return -1.0;
606 
607  //find corresponding branch
608  TBranch* br = fArb->GetBranch(param);
609  if (!br) {
610  //no branch found - wrong name given ?
611  Error("GetMean", "No branch found with name %s", param);
612  return -1.0;
613  }
614  //enable branch
615  fArb->SetBranchStatus(param, 1);
616  //connect variable to branch
617  Float_t value = -1.0;
618  br->SetAddress(&value);
619  //read entry corresponding to run
620  Int_t bytes = fArb->GetEntryWithIndex(run);
621  if (bytes < 0) {
622  //unknown run number
623  Error("GetMean", "Unknown run %d", run);
624  return -1.0;
625  }
626  //disable branch
627  fArb->SetBranchStatus(param, 0);
628 
629  return value;
630 }

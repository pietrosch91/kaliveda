43 //Description of atomic nuclei
44 //
45 //Examples of use:
46 //________________
47 //
48 //Create an alpha particle, and find its binding energy, mass, and mass excess:
49 //
50 // KVNucleus n(2,4); // create nucleus with Z=2 and A=4
51 // cout << "Binding energy of alpha = " << n.GetBindingEnergy() << " MeV" << endl;
52 // cout << "Mass = " << n.GetMass() << " = 4 * " << KVNucleus::kAMU << " + " << n.GetMassExcess() << endl;
53 //
54 //Using the same object we can also obtain binding energies and mass excesses
55 //for all other nuclei:
56 //
57 // n.GetBindingEnergy(1,2);//binding energy of deuteron
58 // n.GetMassExcess(0,1);//mass excess of neutron
59 //
60 //NUCLEAR MASSES
61 //-----------------------
62 //We can also create nuclei by only specifying the atomic number Z.
63 //In this case the mass number A is calculated from Z:
64 //
65 // KVNucleus nuc(82);//create Pb nucleus
66 // nuc.GetA();//by default this is the A corresponding to beta-stability, i.e. 208
67 // nuc.SetMassFormula(KVNucleus::kVedaMass);//change the mass formula used for this nucleus
68 // nuc.GetA();//now the value is that calculated according to the Veda formula, 202
69 // nuc.SetMassFormula(KVNucleus::kEALMass);//Evaporation Attractor Line from R.J. Charity
70 // nuc.GetA();//gives 186
71 // nuc.SetMassFormula(KVNucleus::kBetaMass);//restore the default mass formula
72 //
73 //Z and A can be specified separately:
74 //
75 // KVNucleus a;//no A or Z specified
76 // a.SetZ(10);//at this moment the nucleus' mass number is 20 (beta-stability)
77 // a.SetA(24);//now this represents a 24Ne nucleus
78 //
79 //However, be careful not to use SetZ() AFTER SetA(), because the mass number
80 //is always automatically calculated from Z after a call to SetZ().
81 //
82 //The value of the atomic mass unit, u, is given by KVNucleus::kAMU or KVNucleus::u()
83 //It is 931.494 043 x 10**6 eV, as per the 2002 CODATA recommended values (Reviews of Modern Physics 77, 1-107 (2005)).
84 //
85 // Nuclear Arithmetic & Calorimetry
86 // ----------------------------------------
87 //The '+', '-' and '=' operators have been redefined for the KVNucleus class.
88 //One can therefore perform "nuclear arithmetic".
89 //Example:
90 //
91 // KVNucleus c = a + b; //'a' and 'b' are also KVNucleus objects
92 //
93 //The Z, A, momentum and excitation energy of 'c' are calculated from the
94 //appropriate conservation laws. The mass excesses of the 3 nuclei are obviously
95 //taken into consideration.
96 //If 'a' and 'b' are projectile and target, 'c' is the compound nucleus after
97 //fusion.
98 //In order to perform calorimetry (calculation of source characteristics from
99 //daughter nuclei) one need only sum all nuclei associated with the source.
100 //The resulting nucleus is the source nucleus with its Z, A, momentum and
101 //excitation energy.
102 //
103 // Note for the excitation energy, if one define an excitation energy for a nucleus
104 // using the SetExcitEnergy, the mass and the total energy is modified (M = Mgs + Ex)
105 // when excitation energy is set, one can access to the ground state mass via GetMassGS()
106 //
107 //The subtraction operator allows to perform energy balance for a binary
108 //splitting of a nucleus.
109 //Example:
110 //
111 // KVNucleus d = c - b;
112 //
113 //In this case, the resulting nucleus 'd' should be identical to 'a' in the first
114 //example. One could also imagine
115 //
116 // KVNucleus e = c - alpha;
117 //
118 //where 'alpha' is a KVNucleus alpha-particle, for which we specify the
119 //momentum after emission. The resulting nucleus 'e' is the residue of the
120 //fusion compound after evaporation of an alpha particle.
121 //
122 //The operators '+=' and '-=' also exist. 'a+=b' means 'a = a + b' etc.
123 //
124 //Mass Excess Table
125 //==========
126 //Different mass tables can be implemented using classes derived from
127 //KVMassTable. The mass table to be used is defined by environment variable
128 //
129 // KVNucleus.MassExcessTable: MyMassExcessTable
130 //
131 //where 'MyMassExcessTable' must be defined in terms of a KVNuclDataTable plugin:
132 //
133 //+Plugin.KVNuclDataTable: MyMassExcessTable MyMassExcessTable MyMassExcessTable.cpp+ " MyMassExcessTable()"

325 Int_t KVNucleus::IsMassGiven(const Char_t* isotope)
326 {
327  // test if the given string corresponds to the name of an isotope/element,
328  // and whether or not a mass is specified.
329  // isotope = symbol for element isotope, "C", "natSn", "13N", etc.
330  // if the mass of the isotope is given ("13N", "233U") we return the given mass
331  // if this is a valid element but no mass is given we return 0
332  // if this is not a valid isotope/element, we return -1
333 
334  Int_t A;
335  Char_t name[5];
336  TString tmp(isotope);
337  if (tmp.BeginsWith("nat"))
338  tmp.Remove(0, 3);
339  if (sscanf(tmp.Data(), "%d%s", &A, name) == 2) {
340  //name given in form "208Pb"
341  Int_t z = GetZFromSymbol(name);
342  if (z < 0) return z;
343  return A;
344  }
345  Int_t z = GetZFromSymbol(tmp);
346  if (z < 0) return z;
347  return 0;
348 }

604 Double_t KVNucleus::GetRealAFromZ(Double_t Z, Char_t mt)
605 {
606  //Calculate nuclear mass number from the element's atomic number Z.
607  //This value is not rounded off, we just return the result of one of the following formulae:
608  //
609  //mt = KVNucleus::kVedaMass
610  //__________________________
611  //Veda - A calculated using the formula
612  // fA = (1.867*fZ+.016*fZ*fZ-1.07E-4*fZ*fZ*fZ);
613  // This corresponds to the amass.f subroutine of the old INDRA Veda
614  // calibration programme. This formula was supposed to represent
615  // the Z-dependence of isotope masses in the beta-stability valley,
616  // but is in fact a rather poor approximation, especially for large Z.
617  //
618  //mt = KVNucleus::kBetaMass
619  //_________________________
620  //Beta (default) - An improved parametrisation of the beta-stability valley,
621  // correct even for heavy nuclei up to 238U. The formula is the result
622  // of a fit to 8 stable nuclear masses from Ne20 up to U238.
623  // fA = (.2875 + 1.7622 *Z + .013879 * Z * Z - .000054875 * Z * Z * Z);
624  //
625  //mt = KVNucleus::kEALMass
626  //________________________
627  //EAL - parametrisation of the Evaporation Attractor Line (residue corridor)
628  // due to R.J. Charity (PRC 58(1998)1073) (eq 2)
629  // fA = (2.072*Z + 2.32E-03 * Z*Z) ;
630  //
631  //mt = KVNucleus::kEALResMass
632  //________________________
633  //EALRes - R.J. Charity ---- improvement of EAL parametrisation for
634  // Heavy Residue (QP for instance) (PRC 58(1998)1073) (eq 7)
635  // fA = (2.045*Z + 3.57E-03 * Z*Z) ;
636  //
637  //mt = any other value: A=2*Z
638 
639  Double_t A;
640  switch (mt) {
641 
642  case kVedaMass:
643  A = (1.867 * Z + .016 * TMath::Power(Z, 2.) -
644  1.07E-4 * TMath::Power(Z, 3.));
645  break;
646 
647  case kBetaMass:
648  A = (.2875 + 1.7622 * Z + .013879 * TMath::Power(Z, 2.) -
649  .000054875 * TMath::Power(Z, 3.));
650  break;
651 
652  case kEALMass:
653  A = (2.072 * Z + 2.32E-03 * TMath::Power(Z, 2.));
654  break;
655 
656  case kEALResMass:
657  A = (2.045 * Z + 3.57E-03 * TMath::Power(Z, 2.));
658  break;
659 
660  default:
661  A = 2. * Z;
662  }
663 
664  return A;
665 }

723 Int_t KVNucleus::GetAFromZ(Double_t Z, Char_t mt)
724 {
725  //Calculate nuclear mass number from the element's atomic number Z.
726  //Used by default to set fA and fMass if fA not given.
727  //For light nuclei (Z<6) the values are given (not calculated) and
728  //correspond to: p, alpha, 7Li, 9Be, 11B.
729  //For heavier nuclei, several prescriptions are available
730  //by giving one of the following values to argument mt:
731  //
732  //mt = KVNucleus::kVedaMass
733  //__________________________
734  //Veda - A calculated using the formula
735  // fA = (Int_t)(1.867*fZ+.016*fZ*fZ-1.07E-4*fZ*fZ*fZ) + 1;
736  // This corresponds to the amass.f subroutine of the old INDRA Veda
737  // calibration programme. This formula was supposed to represent
738  // the Z-dependence of isotope masses in the beta-stability valley,
739  // but is in fact a rather poor approximation, especially for large Z.
740  //
741  //mt = KVNucleus::kBetaMass
742  //_________________________
743  //Beta (default) - An improved parametrisation of the beta-stability valley,
744  // correct even for heavy nuclei up to 238U. The formula is the result
745  // of a fit to 8 stable nuclear masses from Ne20 up to U238. From carbon-12 onwards,
746  // the mass is calculated using
747  // fA = (Int_t) (.2875 + 1.7622 *Z + .013879 * Z * Z - .000054875 * Z * Z * Z) + 1;
748  //
749  //mt = KVNucleus::kEALMass
750  //________________________
751  //EAL - parametrisation of the Evaporation Attractor Line (residue corridor)
752  // due to R.J. Charity (PRC 58(1998)1073).
753  // fA = (Int_t)(2.072*Z + 2.32E-03 * Z*Z) + 1; (eq 2)
754  //
755  //mt = KVNucleus::kEALResMass
756  //________________________
757  //EALRes - R.J. Charity ---- improvement of EAL parametrisation for
758  // Heavy Residues (QP for instance) (PRC 58(1998)1073) (eq 7)
759  // fA = (Int_t)(2.045*Z + 3.57E-03 * Z*Z) + 1 ;
760  //
761  //mt = any other value: A=2*Z
762 
763  Int_t A = 0;
764  Int_t z = (Int_t) Z;
765  switch (z) { // masses for lightest nuclei
766  case 1:
767  A = 1;
768  break;
769  case 2:
770  A = 4;
771  break;
772  case 3:
773  A = 7;
774  break;
775  case 4:
776  A = 9;
777  break;
778  case 5:
779  A = 11;
780  break;
781  default:
782  A = (Int_t) KVNucleus::GetRealAFromZ(Z, mt) + 1;
783  }
784  return A;
785 }

1207  // rct=2 (kELTON)take into account the finite surfacethickness

1232  Double_t r2 = -0.548;

106 Bool_t KVRemoteAvailableRunsFile::OpenAvailableRunsFile()
107 {
108  //Initialise fRunlist so that it can be used to read the available runs file from the beginning.
109  //
110  //If the file has already been opened, this means resetting the ifstream to the start of the file.
111  //
112  //If not already open, open the file containing the list of available runs for the dataset.
113  //What this actually means is:
114  // - read value of DataRepository.RemoteAvailableRuns.url for the data repository
115  // - use 'curl' or programme indicated by value of
116  // DataRepository.RemoteAvailableRuns.protocol to make local copy
117  // of file
118  // - open copy
119  //Returns kFALSE in case of problems.
120 
121  //already open ?
122  if (IsFileOpen()) {
123  fRunlist.clear(); // clear any error flags (EOF etc.)
124  fRunlist.seekg(0, ios::beg); // set file buffer pointer to beginning of file
125  //if we can get a lock on the file, all is well. if not, we need to fetch a new
126  //copy
127  if (runlist_lock.Lock()) return kTRUE;
128  }
129  TString http =
130  gEnv->GetValue(Form("%s.DataRepository.RemoteAvailableRuns.url",
131  GetDataSet()->GetRepository()->GetName()), "");
132  if (http == "") {
133  Warning("OpenAvailableRunsFile(ifstream& runlist)",
134  "%s.DataRepository.RemoteAvailableRuns.url is not defined. See $KVROOT/KVFiles/.kvrootrc",
135  GetDataSet()->GetRepository()->GetName());
136  return kFALSE;
137  }
138  TString url;
139  url.Form("%s/%s", http.Data(), GetFileName());
140  //remote file will be copied to $TEMP directory with name
141  //repository.available_runs....
142  fFilePath = GetFileName();
143  KVBase::GetTempFileName(fFilePath);
144  TString cmd;
145  cmd.Form("%s -o%s %s", fCurl.Data(), fFilePath.Data(), url.Data());
146  //lock temp file - just in case
147  runlist_lock.Lock(fFilePath.Data());
148  if (gSystem->Exec(cmd.Data())) {
149  runlist_lock.Release();
150  return kFALSE; // problem with curl
151  }
152  fRunlist.open(fFilePath.Data());
153  if (!fRunlist.good() || !fRunlist.is_open()) {
154  Error("OpenAvailableRunsFile",
155  "Cannot open temp file to copy remote runlist file");
156  runlist_lock.Release();
157  return kFALSE;
158  }
159  return kTRUE;
160 }

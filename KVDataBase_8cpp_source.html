196 Bool_t KVDataBase::AddTable(KVDBTable* tab)
197 {
198 // Add a table to the list of available tables and return kTRUE
199 // it's added. As each table must have a unique name we check if the new table's name
200 // already exists, if not the table is added otherwise it's not added
201 // and the method return kFALSE.
202 //
203 
204  //table with this name already in DB ?
205  if (GetTable(tab->GetName()))
206  return kFALSE;
207 
208  Add(tab);
209  tab->SetFullPath(Form("//root/%s/%s", fDataSet.Data(), tab->GetName()));
210  return kTRUE;
211 }

221 KVDBTable* KVDataBase::AddTable(const Char_t* name, const Char_t* title,
222  Bool_t unique)
223 {
224 // Add a table to the list of tables and check if the new table's name already exists.
225 // If it exists the table is not added and null pointer is returned, otherwise a new
226 // table is created and added to the list and its address is returned.
227 
228  //Table with this name already in DB ?
229  if (GetTable(name))
230  return 0;
231 
232  KVDBTable* table = new KVDBTable(name, title, unique);
233  Add(table);
234  table->SetFullPath(Form("//root/%s/%s", fDataSet.Data(), name));
235  return table;
236 }

318 KVDataBase* KVDataBase::MakeDataBase(const Char_t* name, const Char_t* datasetdir)
319 {
320  //Static function which will create and 'Build' the database object corresponding to 'name'
321  //These are defined as 'Plugin' objects in the file $KVROOT/KVFiles/.kvrootrc :
322  //
323  // Plugin.KVDataBase: INDRA_camp1 KVDataBase1 KVIndra "KVDataBase1()"
324  // +Plugin.KVDataBase: INDRA_camp2 KVDataBase2 KVIndra "KVDataBase2()"
325  // +Plugin.KVDataBase: INDRA_camp4 KVDataBase4 KVIndra "KVDataBase4()"
326  // +Plugin.KVDataBase: INDRA_camp5 KVDataBase5 KVIndra5 "KVDataBase5()"
327  //
328  //The 'name' ("INDRA_camp1" etc.) corresponds to the name of a dataset in $KVROOT/KVFiles/manip.list
329  //This name is stored in member variable fDataSet.
330  //The constructors/macros used have arguments (const Char_t* name)
331 
332  //does plugin exist for given name ?
333  TPluginHandler* ph;
334  if (!(ph = KVBase::LoadPlugin("KVDataBase", name))) {
335  return 0;
336  }
337  //execute constructor/macro for database
338  KVDataBase* mda = (KVDataBase*) ph->ExecPlugin(1, name);
339  mda->SetDataSetDir(datasetdir);
340  //call Build() method
341  mda->Build();
342  return mda;
343 }

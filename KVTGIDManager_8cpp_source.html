16 //KVTGIDManager
17 //
18 //Handles a set of Tassan-Got functional-based identifications
19 //(KVTGID objects) for use by a KVIDTelescope.
20 //
21 //IdentZ and IdentA are, respectively, Z and A identification procedures.
22 //Their purpose is to retrieve the correct KVTGID identification object
23 //from the manager's list, and then use KVTGID::GetIdentification.
24 //
25 //Status code for last identification with IdentZ and IdentA can be
26 //obtained using GetStatus/GetStatusString.
27 //
28 // 0 KVTGIDManager::kStatus_OK "ok",
29 // 1 KVTGIDManager::kStatus_noTGID "no KVTGID for requested identification",
30 // 2 KVTGIDManager::kStatus_OutOfIDRange "point to identify outside of identification range of function",
31 // 3 KVTGIDManager::kStatus_ZtooSmall "IdentA called with Z<1",
32 // 4 KVTGIDManager::kStatus_ZtooLarge "IdentA called with Z larger than max Z defined for KVTGIDZA isotopic identification object"

107 void KVTGIDManager::RemoveTGID(const Char_t* name)
108 {
109  //Delete the identification object with name
110  KVTGID* tgid = 0;
111  if ((tgid = (KVTGID*)fIDList.FindObject(name))) {
112  fIDList.Remove(tgid);
113  //recalculate fID_max if needed
114  if (tgid->GetIDmax() == fID_max) {
115  fID_max = 0.;
116  TIter n(&fIDList);
117  KVTGID* t;
118  while ((t = (KVTGID*)n())) fID_max = TMath::Max(fID_max, t->GetIDmax());
119  }
120  delete tgid;
121  }
122 }

192 Double_t KVTGIDManager::IdentZ(const Char_t* idtelescope_name, Double_t id_map_X, Double_t id_map_Y, Double_t& funLTG,
193  const Char_t* grid_type)
194 {
195  //Identification in Z of particle for ID telescope idt using Tassan-Got functional corresponding
196  //to grid of type "grid_type". The "signal_type" argument is passed to GetIDMapX/Y(signal_type),
197  //in case "grid_type" and "signal_type" are not exactly the same.
198  //
199  //funLTG gives distance to identification line (absolute value of LTG function for the Z we find)
200  // funLTG = 0 -> identification parfaite
201  // funLTG = -1 -> no identification
202  //If the point to identify is outside of the range of the identification
203  //for this 'grid_type' (not between Zmin and Zmax) then we return -1 and funLTG=-1.
204  //if no identification of the required type exists, we return -1. and funLTG=-1.
205 
206  funLTG = -1.;
207  fStatus = kStatus_OK;
208 
209  //get Z identification object from KVTGIDManager
210  KVTGID* _tgid = GetTGID(idtelescope_name, "Z", grid_type);
211 
212  //if no identification object exists, return -1
213  if (!_tgid) {
214  fStatus = kStatus_noTGID;
215  return (-1.0);
216  }
217  //Enter current point coordinates into Z identification function
218  _tgid->SetParameter("X", id_map_X);
219  _tgid->SetParameter("Y", id_map_Y);
220 
221  Double_t Zfound = -1.;
222  Double_t Zmin = _tgid->GetIDmin();
223  Double_t Zmax = _tgid->GetIDmax();
224  if ((Int_t) Zmax == (Int_t) GetIDmax()) {
225  //If this functional corresponds to the grid with the highest ID lines,
226  //(i.e. for higher ID there are no more grids) then we need to be able to identify all
227  //particles with this grid, even above the last ID line drawn by hand.
228 
229  Zmax = 100.;
230  }
231  //we need to lower and raise the limits in Z in order to pass the Eval(Zmin)*Eval(Zmax)<0
232  //condition. E.G. for GG, Zmin=1, which means that any points below the Z=1 line are not
233  //identified, whereas we should allow Zreal between 0.5 and 1.5
234  //also when several fits are used for different ranges of Z, often the ranges of validity
235  //of two successive fits do not overlap, i.e. Z=1-5 in GG, then Z=6-20 in PG etc.
236  //in this case somewhere in between Z=5 & Z=6 particles will be left unidentified
237  //as they will neither work in GG nor in PG.
238  Zmin -= 1.;
239  Zmax += 1.;
240 
241  Zfound = _tgid->GetIdentification(Zmin, Zmax, funLTG);
242 
243  if (_tgid->GetStatus() == KVTGID::kStatus_NotBetween_IDMin_IDMax)
244  fStatus = kStatus_OutOfIDRange;
245 
246  return Zfound;
247 }

263 Double_t KVTGIDManager::IdentA(const Char_t* idtelescope_name, Double_t id_map_X, Double_t id_map_Y, Double_t& funLTG,
264  const Char_t* grid_type, Int_t Z)
265 {
266  //Get A identification (we already know Z) for ID telescope idt using grid "grid_type".
267  //The "signal_type" argument is passed to GetIDMapX/Y(signal_type),
268  //in case "grid_type" and "signal_type" are not exactly the same.
269  //
270  //funLTG is the ABSOLUTE value of the Tassan-Got function for the Z and the calculated
271  //A i.e. in an ideal world it should equal zero. This gives an idea of the
272  //quality of the identification.
273  // funLTG = 0 -> identification parfaite
274  // funLTG = -1 -> no identification
275 
276  funLTG = -1.;
277  fStatus = kStatus_OK;
278 
279  if (Z < 1) {
280  fStatus = kStatus_ZtooSmall;
281  return -1.;
282  }
283  //get A identification object from KVTGIDManager
284  KVTGID* _tgid = GetTGID(idtelescope_name, "A", grid_type);
285  //if no identification object exists, return -1
286  if (!_tgid) {
287  fStatus = kStatus_noTGID;
288  return (-1.0);
289  }
290  //check Z is not too big - no isotopic identification for Z > Zmax of grid
291  if (Z > _tgid->GetIDmax()) {
292  fStatus = kStatus_ZtooLarge;
293  return -1.;
294  }
295  //Initialise parameters of A identification function
296  _tgid->SetParameter("X", id_map_X);
297  _tgid->SetParameter("Y", id_map_Y);
298  _tgid->SetParameter("Z", Z);
299 
300  Double_t Afound = -1., Amin, Amax;
301  // reasonable limits for nuclear masses:
302  if (Z == 1) {
303  Amin = 1.;
304  Amax = 3.;
305  } else if (Z == 2) {
306  Amin = 3;
307  Amax = 8;
308  } else if (Z == 3) {
309  Amin = 6;
310  Amax = 11;
311  } else if (Z == 4) {
312  Amin = 7;
313  Amax = 14;
314  } else if (Z == 5) {
315  Amin = 8;
316  Amax = 17;
317  } else {
318  //formula for Amin reasonable for 6<Z<20
319  Amin = TMath::Max((Z + 1.), (1.8 * (Z - 2.) + 1.));
320  Amax = 2.*Z + 8.;
321  }
322 
323  // same limit trick as in IdentZ
324  Amin -= 0.5;
325  Amax += 0.5;
326 
327  Afound = _tgid->GetIdentification(Amin, Amax, funLTG);
328 
329  if (_tgid->GetStatus() == KVTGID::kStatus_NotBetween_IDMin_IDMax)
330  fStatus = kStatus_OutOfIDRange;
331 
332  return Afound;
333 }

646  // sorted according to increasing run number

649

651  // sort runs

654  fLastRun = ((KVINDRADBRun*)fRunList->GetLastInList())->GetNumber();

657  fRunList->Sort(0);
658  fFirstRun = ((KVINDRADBRun*)fRunList->GetFirstInList())->GetNumber();

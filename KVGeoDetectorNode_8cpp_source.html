19 // Each detector of the multidetector geometry has an associated node which can
20 // be accessed using
21 //
22 // KVDetector* det; // pointer to a detector
23 // KVGeoDetectorNode* node = det->GetNode(); // access detector's node
24 //
25 // Note that the relation is reciprocal, i.e. one can retrieve the detector
26 // associated to a node:
27 //
28 // KVDetector* d = node->GetDetector(); // in this case, d==det !
29 //
30 // The nodes are set up when the multidetector geometry is imported from a ROOT
31 // geometry using KVGeoImport. All possible trajectories from the target through
32 // the detectors of the array are explored. As each detector/node is traversed,
33 // we store information on the detectors which are immediately in front of or
34 // behind it, and which trajectories (see KVGeoDNTrajectory class) pass through
35 // which node. This information can then be obtained from the KVGeoDetectorNode
36 // associated to each detector.
37 //
38 // FORWARDS/BACKWARDS, IN FRONT/BEHIND
39 // The convention is that movement towards the target is FORWARDS;
40 // a detector D1 is IN FRONT of detector D2 if D1 is closer to the target.
41 // Conversely, movement away from the target is BACKWARDS;
42 // a detector D1 is BEHIND detector D2 if D1 is further from the target.

100 void KVGeoDetectorNode::CalculateForwardsTrajectories()
101 {
102  // Fill list with all trajectories going forwards from this node
103 
104  fNTrajForwards = 0;
105  if (GetNTraj()) {
106  TIter next(GetTrajectories());
107  KVGeoDNTrajectory* t;
108  while ((t = (KVGeoDNTrajectory*)next())) {
109  if (!t->EndsAt(this)) {
110  fNTrajForwards++;
111  if (!fTrajF) fTrajF = new KVUniqueNameList;
112  fTrajF->Add(t);
113  }
114  }
115  }

444 void KVGeoDetectorNode::BuildTrajectoriesForwards(TSeqCollection* list)
445 {
446  // Add this node to each trajectory in list
447  // Then continue trajectories for each node in front of this one
448  // If more than one node is in front, a new trajectory is created
449  // and added to the list for each extra node
450  //
451  // N.B. we are building trajectories starting from nodes furthest from
452  // target and moving towards it. Trajectories always go from the stopping
453  // detector towards the target.
454  // Therefore we add each new node to the end of each trajectory.
455 
456  if (!list->GetEntries()) {
457  // no trajectories in list
458  // add new trajectory starting here
459  list->Add(new KVGeoDNTrajectory(this));
460  } else {
461  // add this node to each trajectory in list
462  list->R__FOR_EACH(KVGeoDNTrajectory, AddLast)(this);
463  }
464  // add each trajectory to list of trajectories through this node
465  TIter nextT(list);
466  KVGeoDNTrajectory* traj;
467 
468 
469  // if no nodes in front of this one, stop
470  if (!GetNDetsInFront()) return;
471 
472  nextT.Reset();
473  TList* newTrajectories = new TList;
474  while ((traj = (KVGeoDNTrajectory*)nextT())) {
475  KVGeoDNTrajectory baseTraj(*traj);
476  // for each trajectory in list
477  // for first node in front of this one, continue existing trajectory
478  // for each subsequent node in front, create new copy of existing trajectory
479  // and continue it
480  TIter nextN(fInFront);
481  KVGeoDetectorNode* node;
482  KVDetector* det;
483  Int_t node_num = 1;
484  while ((det = (KVDetector*)nextN())) {
485  node = det->GetNode();
486  if (node_num == 1) node->BuildTrajectoriesForwards(list);
487  else {
488  KVGeoDNTrajectory* newTraj = new KVGeoDNTrajectory(baseTraj);
489  newTrajectories->Add(newTraj);
490  node->BuildTrajectoriesForwards(newTrajectories);
491  }
492  node_num++;
493  }
494  }
495  if (newTrajectories->GetEntries()) {
496  list->AddAll(newTrajectories);
497  }
498  delete newTrajectories;
499 }

25 // This class will read the contents of a directory and automatically list the simulated and/or
26 // filtered simulated data it contains. To use:
27 //
28 // KVSimDir sim("MySimultations", "/path/to/my/simulations");
29 // sim.AnalyseDirectory();
30 //
31 // When method AnalyseDirectory() is called, each ROOT file in the directory given to the
32 // constructor will be opened and analysed in turn, by method AnalyseFile().
33 // If there is a TTree in the file, then we look at all of its branches until we find one
34 // containing objects which derive from KVEvent:
35 //
36 // -- if they inherit from KVSimEvent, we add the file to the list of simulated data:
37 // * a KVSimFile is created. The title of the TTree where data were found will
38 // be used as 'Information' on the nature of the simulation.
39 // -- if they inherit from KVReconstructedEvent, we add the file to the list of filtered data.
40 // * a KVSimFile is created. Informations on the filtered data are extracted from
41 // TNamed objects in the file with names 'Dataset', 'System', 'Run', 'Geometry'
42 // (type of geometry used, 'ROOT' or 'KV'), 'Origin' (i.e. the name of the simulation
43 // file which was filtered). These objects are automatically created when data is
44 // filtered using KVEventFiltering.
45 //
46 // Analysis of the file stops after the first TTree with a branch satisfying one of the
47 // two criteria is found (it is assumed that in each file there is only one TTree containing
48 // either simulated or filtered data).

186 void KVSimDir::AnalyseDirectory()
187 {
188  // Read contents of directory given to ctor.
189  // Each ROOT file will be analysed by AnalyseFile().
190 
191  Info("AnalyseDirectory", "Analysing %s...", GetDirectory());
192  fSimData.Clear();
193  fFiltData.Clear();
194  //loop over files in current directory
195  TSystemDirectory thisDir(".", GetDirectory());
196  unique_ptr<TList> fileList(thisDir.GetListOfFiles());
197  TIter nextFile(fileList.get());
198  TSystemFile* aFile = 0;
199  while ((aFile = (TSystemFile*)nextFile())) {
200 
201  if (aFile->IsDirectory()) continue;
202 
203  KVString fileName = aFile->GetName();
204  if (!fileName.EndsWith(".root") && !fileName.Contains(".root.")) continue; /* skip non-ROOT files */
205 
206  AnalyseFile(fileName);
207 
208  }
209 }

234 void KVSimDir::AnalyseFile(const Char_t* filename)
235 {
236  // Analyse ROOT file given as argument.
237  // If there is a TTree in the file, then we look at all of its branches until we find one
238  // containing objects which derive from KVEvent:
239  //
240  // -- if they inherit from KVSimEvent, we add the file to the list of simulated data:
241  // * a KVSimFile is created. The title of the TTree where data were found will
242  // be used as 'Information' on the nature of the simulation.
243  // -- if they inherit from KVReconstructedEvent, we add the file to the list of filtered data.
244  // * a KVSimFile is created. Informations on the filtered data are extracted from
245  // TNamed objects in the file with names 'Dataset', 'System', 'Run', 'Geometry'
246  // (type of geometry used, 'ROOT' or 'KV'), 'Origin' (i.e. the name of the simulation
247  // file which was filtered), 'Filter' (type of filter: Geo, GeoThresh or Full).
248  // These objects are automatically created when data is filtered using KVEventFiltering.
249  //
250  // Analysis of the file stops after the first TTree with a branch satisfying one of the
251  // two criteria is found (it is assumed that in each file there is only one TTree containing
252  // either simulated or filtered data).
253 
254  //Info("AnalyseFile", "Analysing file %s...", filename);
255  TString fullpath;
256  AssignAndDelete(fullpath, gSystem->ConcatFileName(GetDirectory(), filename));
257  unique_ptr<TFile> file(TFile::Open(fullpath));
258  if (!file.get() || file->IsZombie()) return;
259  // look for TTrees in file
260  TIter next(file->GetListOfKeys());
261  TKey* key;
262  while ((key = (TKey*)next())) {
263  TString cn = key->GetClassName();
264  if (cn == "TTree") {
265  // look for branch with KVEvent objects
266  TTree* tree = (TTree*)file->Get(key->GetName());
267  TSeqCollection* branches = tree->GetListOfBranches();
268  TIter nextB(branches);
269  TBranchElement* branch;
270  while ((branch = (TBranchElement*)nextB())) {
271  TString branch_classname = branch->GetClassName();
272  TClass* branch_class = TClass::GetClass(branch_classname, kFALSE, kTRUE);
273  if (branch_class && branch_class->InheritsFrom("KVEvent")) {
274  if (branch_class->InheritsFrom("KVReconstructedEvent")) {
275  // filtered data. there must be TNamed called 'Dataset', 'System', & 'Run' in the file.
276  unique_ptr<TNamed> ds((TNamed*)file->Get("Dataset"));
277  unique_ptr<TNamed> orig((TNamed*)file->Get("Origin"));
278  unique_ptr<TNamed> sys((TNamed*)file->Get("System"));
279  unique_ptr<TNamed> r((TNamed*)file->Get("Run"));
280  unique_ptr<TNamed> g((TNamed*)file->Get("Geometry"));
281  unique_ptr<TNamed> f((TNamed*)file->Get("Filter"));
282  TString dataset;
283  if (ds.get()) dataset = ds->GetTitle();
284  TString system;
285  if (sys.get()) system = sys->GetTitle();
286  TString run;
287  if (r.get()) run = r->GetTitle();
288  TString origin;
289  if (orig.get()) origin = orig->GetTitle();
290  TString geometry;
291  if (g.get()) geometry = g->GetTitle();
292  TString filter;
293  if (f.get()) filter = f->GetTitle();
294  Int_t run_number = run.Atoi();
295  KVSimFile* fff = new KVSimFile(this, filename, tree->GetTitle(), tree->GetEntries(), tree->GetName(), branch->GetName(),
296  dataset, system, run_number, geometry, origin, filter);
297  fFiltData.Add(fff);
298  unique_ptr<TNamed> gem((TNamed*)file->Get("Gemini++"));
299  if (gem.get()) {
300  if (!strcmp(gem->GetTitle(), "yes")) fff->SetGemini();
301  unique_ptr<TNamed> gemdec((TNamed*)file->Get("GemDecayPerEvent"));
302  if (gemdec.get()) {
303  TString gemdecperev = gemdec->GetTitle();
304  fff->SetGemDecayPerEvent(gemdecperev.Atoi());
305  }
306  }
307  return;
308  }
309  else {
310  fSimData.Add(new KVSimFile(this, filename, tree->GetTitle(), tree->GetEntries(), tree->GetName(), branch->GetName()));
311  return;
312  }
313  }
314  }
315  }
316  }

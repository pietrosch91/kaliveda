104 NewRooFitResult* NewRooFitResult::lastMinuitFit(const RooArgList& varList)
105 {
106  // Import the results of the last fit performed by gMinuit, interpreting
107  // the fit parameters as the given varList of parameters.
108 
109  // Verify length of supplied varList
110  if (varList.getSize() > 0 && varList.getSize() != gMinuit->fNu) {
111  oocoutE((TObject*)0, InputArguments) << "NewRooFitResult::lastMinuitFit: ERROR: supplied variable list must be either empty " << endl
112  << " or match the number of variables of the last fit (" << gMinuit->fNu << ")" << endl ;
113  return 0 ;
114  }
115 
116  // Verify that all members of varList are of type RooRealVar
117  TIterator* iter = varList.createIterator() ;
118  RooAbsArg* arg ;
119  while ((arg = (RooAbsArg*)iter->Next())) {
120  if (!dynamic_cast<RooRealVar*>(arg)) {
121  oocoutE((TObject*)0, InputArguments) << "NewRooFitResult::lastMinuitFit: ERROR: variable '" << arg->GetName() << "' is not of type RooRealVar" << endl ;
122  return 0 ;
123  }
124  }
125  delete iter ;
126 
127  NewRooFitResult* r = new NewRooFitResult("lastMinuitFit", "Last MINUIT fit") ;
128 
129  // Extract names of fit parameters from MINUIT
130  // and construct corresponding RooRealVars
131  RooArgList constPars("constPars") ;
132  RooArgList floatPars("floatPars") ;
133 
134  Int_t i ;
135  for (i = 1; i <= gMinuit->fNu; ++i) {
136  if (gMinuit->fNvarl[i - 1] < 0) continue;
137  Int_t l = gMinuit->fNiofex[i - 1];
138  TString varName(gMinuit->fCpnam[i - 1]) ;
139  Bool_t isConst(l == 0) ;
140 
141  Double_t xlo = gMinuit->fAlim[i - 1];
142  Double_t xhi = gMinuit->fBlim[i - 1];
143  Double_t xerr = gMinuit->fWerr[l - 1];
144  Double_t xval = gMinuit->fU[i - 1] ;
145 
146  RooRealVar* var ;
147  if (varList.getSize() == 0) {
148 
149  if ((xlo < xhi) && !isConst) {
150  var = new RooRealVar(varName, varName, xval, xlo, xhi) ;
151  } else {
152  var = new RooRealVar(varName, varName, xval) ;
153  }
154  var->setConstant(isConst) ;
155  } else {
156 
157  var = (RooRealVar*) varList.at(i - 1)->Clone() ;
158  var->setConstant(isConst) ;
159  var->setVal(xval) ;
160  if (xlo < xhi) {
161  var->setRange(xlo, xhi) ;
162  }
163  if (varName.CompareTo(var->GetName())) {
164  oocoutI((TObject*)0, Eval) << "NewRooFitResult::lastMinuitFit: fit parameter '" << varName
165  << "' stored in variable '" << var->GetName() << "'" << endl ;
166  }
167 
168  }
169 
170  if (isConst) {
171  constPars.addOwned(*var) ;
172  } else {
173  var->setError(xerr) ;
174  floatPars.addOwned(*var) ;
175  }
176  }
177 
178  Int_t icode, npari, nparx ;
179  Double_t fmin, edm, errdef ;
180  gMinuit->mnstat(fmin, edm, errdef, npari, nparx, icode) ;
181 
182  r->setConstParList(constPars) ;
183  r->setInitParList(floatPars) ;
184  r->setFinalParList(floatPars) ;
185  r->setMinNLL(fmin) ;
186  r->setEDM(edm) ;
187  r->setCovQual(icode) ;
188  r->setStatus(gMinuit->fStatus) ;
189  r->fillCorrMatrix() ;
190 
191  return r ;
192 }

203 {
204  // Return true if this fit result is identical to other within tolerance 'tol' on fitted values
205  // and tolerance 'tolCor' on correlation coefficients
206 
207  Bool_t ret = kTRUE ;
208 
209  if (fabs(_minNLL - other._minNLL) >= tol) {
210  cout << "NewRooFitResult::isIdentical: minimized value of -log(L) is different " << _minNLL << " vs. " << other._minNLL << endl ;
211  ret = kFALSE ;
212  }
213 
214  for (Int_t i = 0 ; i < _constPars->getSize() ; i++) {
215  RooAbsReal* ov = static_cast<RooAbsReal*>(other._constPars->find(_constPars->at(i)->GetName())) ;

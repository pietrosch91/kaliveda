79 void KVedaLossRangeFitter::SetMaterial(KVIonRangeTableMaterial* m)
80 {
81  // Sets range table to fit. Also finds material with closest Z in VEDALOSS library.
82  fMaterial = m;
83  Double_t zmat = fMaterial->GetZ();
84  TObjArray* mats = VEDALOSS.GetListOfMaterials();
85  TString closest;
86  Double_t closestZ = 100;
87  TNamed* mat;
88  TIter next(mats);
89  while ((mat = (TNamed*)next())) {
90  KVIonRangeTableMaterial* y = VEDALOSS.GetMaterial(mat->GetName());
91  Double_t yZ = y->GetZ();
92  if (TMath::Abs(yZ - zmat) < TMath::Abs(closestZ - zmat)) {
93  closestZ = yZ;
94  closest = mat->GetName();
95  }
96  }
97  fClosestVedaMat = (KVedaLossMaterial*)VEDALOSS.GetMaterial(closest);
98  Info("SetMaterial", "Initial fit parameters will be taken from:");
99  fClosestVedaMat->Print();
100  delete mats;
101 }

128 TGraph* KVedaLossRangeFitter::GenerateRangeTable(Int_t Z)
129 {
130  // Create TGraph containing log(range) vs. log (E/A) for ions of given
131  // atomic number (the mass used is the reference mass found in the VEDALOSS
132  // tables).
133  // Convert range to mg/cm**2 (units of internal VEDALOSS range function)
134 
135  TGraph* r = new TGraph;
136  r->SetMarkerStyle(24);
137  r->SetTitle(Form("Z = %d", Z));
138  int i = 0, ip = 0;
139  while (1) {
140  if (energy[i] < 0.) break;
141  Double_t logE = log(energy[i]);
142  Double_t R = fMaterial->GetRangeOfIon(Z, Aref, Aref * energy[i]) / KVUnits::mg;
143  if (R > 0.) {
144  r->SetPoint(ip++, logE, log(R));
145  }
146  i++;
147  }
148  return r;
149 }

199 void KVedaLossRangeFitter::DoFits(TString output_file, Int_t Zmin, Int_t Zmax)
200 {
201  // Perform fits to range tables for elements from Zmin to Zmax
202  TGraph* g = 0;
203  TGraph* v = 0;
204  new TCanvas;
205 
206  ofstream results(output_file.Data());
207 
208  PrintFitHeader(results);
209  for (int Z = Zmin; Z <= Zmax; Z++) {
210  gPad->Clear();
211  SetInitialParameters(Z);
212  if (g) delete g;
213  if (v) delete v;
214  g = GenerateRangeTable(Z);
215  g->Draw("AP");
216  Int_t fitStatus = FitRangeTable(g);
217  v = GenerateVEDALOSSRangeTable(Z);
218  v->Draw("P");
219  printf("Z = %d : fit status ", Z);
220  if (fitStatus == 0) {
221  printf("OK\n");
222  } else
223  printf("PROBLEM (%d)\n", fitStatus);
224  PrintFitParameters(Z, results);
225  gPad->Modified();
226  gPad->Update();
227  gSystem->Sleep(500);
228  }
229  results.close();
230 }

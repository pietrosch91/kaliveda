84 void KVIDChIoSi_e613::Initialize()
85 {
86  // Initialisation of telescope before identification.
87  // This method MUST be called once before any identification is attempted.
88  // Initialisation of grid is performed here.
89  // IsReadyForID() will return kTRUE if a grid is associated to this telescope for the current run.
90 
91  fchio = GetDetector(1);
92  fsi = GetDetector(2);
93 
94  ChIoSiGrid = FromFitChIoSiGrid = 0;
95  TIter next(GetListOfIDGrids());
96  KVIDGraph* grid;
97  while ((grid = (KVIDGraph*)next())) {
98  if (!strcmp(grid->GetVarY(), "CI-E")) ChIoSiGrid = (KVIDGChIoSi*)grid;
99  else if (!strcmp(grid->GetVarY(), "CI-E-Fit")) FromFitChIoSiGrid = (KVIDGChIoSi*)grid;
100  }
101  if (ChIoSiGrid) {
102  SetBit(kReadyForID);
103  ChIoSiGrid->Initialize();
104  if (FromFitChIoSiGrid) FromFitChIoSiGrid->Initialize();

106  ResetBit(kReadyForID);

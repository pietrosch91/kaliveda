158 void KVINDRAUpDater::CheckStatusOfDetectors(KVDBRun* kvrun)
159 {
160 
161  KVRList* absdet = kvrun->GetLinks("Absent Detectors");
162  KVRList* oooacq = kvrun->GetLinks("OoO ACQPars");
163  KVRList* ooodet = kvrun->GetLinks("OoO Detectors");
164 
165  TIter next(gIndra->GetListOfDetectors());
166  KVDetector* det;
167  KVACQParam* acq;
168 
169  Int_t ndet_absent = 0;
170  Int_t ndet_ooo = 0;
171  Int_t nacq_ooo = 0;
172 
173  while ((det = (KVDetector*)next())) {
174  //Test de la presence ou non du detecteur
175  if (!absdet) {
176  det->SetPresent();
177  } else {
178  if (absdet->FindObject(det->GetName(), "Absent Detector")) {
179  det->SetPresent(kFALSE);
180  ndet_absent += 1;
181  } else {
182  det->SetPresent();
183  }
184  }
185  if (det->IsPresent()) {
186  //Test du bon fonctionnement ou non du detecteur
187  if (!ooodet) {
188  det->SetDetecting();
189  } else {
190  if (ooodet->FindObject(det->GetName(), "OoO Detector")) {
191  det->SetDetecting(kFALSE);
192  ndet_ooo += 1;
193  } else {
194  det->SetDetecting();
195  }
196  }
197  //Test du bon fonctionnement ou non des parametres d acquisition
198  if (det->IsDetecting()) {
199  TIter next_acq(det->GetACQParamList());
200  if (!oooacq) {
201  while ((acq = (KVACQParam*)next_acq())) {
202  acq->SetWorking();
203  }
204  } else {
205  Int_t noff = 0;
206  while ((acq = (KVACQParam*)next_acq())) {
207  if (oooacq->FindObject(acq->GetName(), "OoO ACQPar")) {
208  acq->SetWorking(kFALSE);
209  noff += 1;
210  nacq_ooo += 1;
211  } else {
212  acq->SetWorking();
213  }
214  }
215  if (noff == 3) {
216  det->SetDetecting(kFALSE);
217  ndet_ooo += 1;
218  nacq_ooo -= 3;
219  }
220  }
221  }
222  }
223  }
224 
225  Info("KVINDRAUpDater", "%d detecteurs absents", ndet_absent);
226  Info("KVINDRAUpDater", "%d detecteurs ne fonctionnent pas", ndet_ooo);
227  Info("KVINDRAUpDater", "%d parametres d acquisition ne fonctionnent pas", nacq_ooo);
228

298 void KVINDRAUpDater::SetChIoPressures(KVDBRun* kvrun)
299 {
300  //Update ChIo pressures for this run with values in database (if any)
301  //
302  //if pressure is equal to 0 (no gas)
303  //mark the corresponding ChIo's as non detecting detector (see KVDetector::SetDetecting())
304  //
305 
306  KVRList* param_list = kvrun->GetLinks("ChIo Pressures");
307  if (!param_list) {
308  return;
309  }
310  KVChIo* kvd;
311  KVDBChIoPressures* kvps;
312  TIter next_ps(param_list);
313  KVSeqCollection* chios = gIndra->GetListOfChIo();
314  if (!chios) {
315  Error("SetChIoPressures",
316  "gIndra->GetListOfChIo() returns null list pointer");
317  return;
318  }
319  cout << "--> Setting ChIo pressures" << endl;
320  TIter next_chio(chios);
321  while ((kvps = (KVDBChIoPressures*) next_ps())) {
322  cout << " Ring 2/3: " << kvps->
323  GetPressure(CHIO_2_3) << "mbar" << endl;
324  cout << " Ring 4/5: " << kvps->
325  GetPressure(CHIO_4_5) << "mbar" << endl;
326  cout << " Ring 6/7: " << kvps->
327  GetPressure(CHIO_6_7) << "mbar" << endl;
328  cout << " Ring 8/12: " << kvps->
329  GetPressure(CHIO_8_12) << "mbar" << endl;
330  cout << " Ring 13/17: " << kvps->
331  GetPressure(CHIO_13_17) << "mbar" << endl;
332  while ((kvd = (KVChIo*) next_chio())) {
333  if (!strcmp(kvd->GetType(), "CI")) {
334  //check detector type: ="CI" for standard INDRA chio
335  if (kvd->GetRingNumber() == 2)
336  kvd->SetPressure(kvps->GetPressure(CHIO_2_3));
337  if (kvd->GetRingNumber() == 4)
338  kvd->SetPressure(kvps->GetPressure(CHIO_4_5));
339  if (kvd->GetRingNumber() == 6)
340  kvd->SetPressure(kvps->GetPressure(CHIO_6_7));
341  if (kvd->GetRingNumber() >= 8 && kvd->GetRingNumber() <= 12)
342  kvd->SetPressure(kvps->GetPressure(CHIO_8_12));
343  if (kvd->GetRingNumber() >= 13 && kvd->GetRingNumber() <= 17)
344  kvd->SetPressure(kvps->GetPressure(CHIO_13_17));
345  if (kvd->GetPressure() == 0.0) {
346  kvd->SetDetecting(kFALSE);
347  } else {
348  kvd->SetDetecting(kTRUE);
349  }
350  }
351  }
352  next_chio.Reset();
353  }

484 void KVINDRAUpDater::SetCsIGainCorrectionParameters(KVDBRun* kvrun)
485 {
486  // Sets KVCsI::fGainCorrection data member, used by KVCsI::GetCorrectedLumiereTotale
487  // to return the total light output corrected by a run-dependent factor.
488  // We set all detectors' correction to 1, then set the corrections defined for this
489  // run, if any.
490 
491  TIter next_csi(gIndra->GetListOfCsI());
492  KVCsI* csi;
493  while ((csi = (KVCsI*)next_csi())) {
494  csi->SetTotalLightGainCorrection(1.0);
495  }
496 
497  KVRList* param_list = kvrun->GetLinks("CsIGainCorr");
498  if (!param_list) {
499  return;
500  }
501  if (!param_list->GetSize()) {
502  return;
503  }
504 
505  TIter next_ps(param_list);
506  KVDBParameterSet* dbps;
507  while ((dbps = (KVDBParameterSet*)next_ps())) {
508 
509  csi = (KVCsI*)gIndra->GetDetector(dbps->GetName());
510  if (!csi) {
511  // the name of the parameter set should be the name of the detector;
512  // however, it may be the name of an acquisition parameter associated with
513  // the detector!
514  KVACQParam* a = gIndra->GetACQParam(dbps->GetName());
515  if (a) csi = (KVCsI*)a->GetDetector();
516  // still no good ?
517  if (!csi) {
518  Warning("SetCsIGainCorrectionParameters",
519  "Cannot find detector associated with %s", dbps->GetName());
520  continue;
521  }
522  }
523  csi->SetTotalLightGainCorrection(dbps->GetParameter());
524  Info("SetCsIGainCorrectionParameters", "%s gain correction = %f", csi->GetName(), csi->GetTotalLightGainCorrection());
525  }

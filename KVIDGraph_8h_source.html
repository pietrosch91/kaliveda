276  virtual void AddIdentifier(KVIDentifier* id)
277  {
278  // Add identifier to the graph. It will be deleted by the graph.
279  fIdentifiers->Add(id);
280  id->SetParent(this);
281  id->SetVarX(GetVarX());
282  id->SetVarY(GetVarY());
283  id->SetOnlyZId(OnlyZId());
284  //if grid is Z-identification only, set mass formula for line
285  //according to mass formula of grid
286  if (OnlyZId()) id->SetMassFormula(GetMassFormula());
287  //Modified();
288  };

54 inline Double_t KVIDLine::DistanceToLine(Double_t px, Double_t py,
55  Int_t& i_near)
56 {
57  //Compute the closest distance of approach from point px,py to this line.
58  //The units of px, py are the same as the coordinates of the graph
59  //
60  //WARNING: result can be NEGATIVE - see below.
61  //
62  //The distance from the point to each segment is calculated in turn.
63  //
64  //If the point lies between the endpoints of at least one segment, the returned
65  //distance will be the closest of these segments. In this case i_near gives the
66  //index of the first point of the closest segment.
67  //
68  //If not then the distance is the smallest distance between the endpoint
69  //of a segment and the point. In this case i_near gives the index of the
70  //closest endpoint.
71  //If the closest segment-endpoint is not one of the endpoints of the whole line,
72  //the returned distance is positive.
73  //On the other hand, if the closest part of the line to the point is one of the
74  //two endpoints of the line, the returned distance is NEGATIVE.
75 
76  Double_t distance, dist2;
77  Int_t i;
78  Double_t d;
79  distance = dist2 = 9999.;
80  Int_t i_nearest_point = 0, inear1 = 0, inear2 = 0;
81  //distance = closest approach to a segment with point lying between endpoints
82  //dist2 = closest approach to a segment with point outside endpoints
83  //if distance == 9999 at end, then dist2 is used
84 
85  for (i = 0; i < fNpoints - 1; i++) {
86  d = DistanceToLine(px, py, fX[i], fY[i], fX[i + 1], fY[i + 1],
87  i_nearest_point);
88  //cout << "i=" << i << " X1="<<fX[i]<<" Y1="<<fY[i]<<" X2="<<fX[i+1]<<" Y2="<<fY[i+1]<<" --- d = "<<d<<endl;
89  if (d >= 0.) {
90  if (d < distance) {
91  distance = d;
92  inear1 = i;
93  }
94  } else {
95  //point not between endpoints of segment
96  if (-d < dist2) {
97  dist2 = -d;
98  inear2 = i + i_nearest_point;
99  }
100  }
101  }
102 
103  i_near = inear1;
104  if (distance < 9999.)
105  return distance;
106  i_near = inear2;
107  //closest point is an 'internal' point of line - POSITIVE distance
108  if (inear2 > 0 && inear2 < (fNpoints - 1))
109  return dist2;
110  //closest point is one of endpoints of line - NEGATIVE distance
111  return -dist2;
112 }

117 inline Double_t KVIDLine::DistanceToLine(Double_t px, Double_t py,
118  Double_t xp1, Double_t yp1,
119  Double_t xp2, Double_t yp2,
120  Int_t& i_nearest_point)
121 {
122  //Given a line segment defined by endpoints (xp1,yp1) and (xp2,yp2) find the
123  // shortest distance between point (px,py) and the line.
124  //
125  // M
126  // P1 (xp1,yp1) x---------------------------------------x P2 (xp2,yp2)
127  // |
128  // |
129  // |
130  // |
131  // |
132  // x
133  // P (px,py)
134  //This is simply the magnitude of the component of vector P1->P (or P2->P)
135  //perpendicular to P1->P2.
136  //
137  //If the point is indeed between the two endpoints as shown, then
138  //P1M + P2M = P1P2
139  //If not, P1M + P2M > P1P2. In this case the closest distance is that to the nearer
140  //of the two endpoints, but the value returned is negative; i_nearest_point gives the index (0 or 1)
141  //of the nearer endpoint
142 
143  TVector2 P1(xp1, yp1), P2(xp2, yp2), P(px, py);
144  TVector2 P1P2 = P2 - P1;
145  TVector2 P1P = P - P1;
146  TVector2 P2P = P - P2;
147  TVector2 MP = P1P.Norm((P1P2));
148  i_nearest_point = 0;
149  //check point lies between endpoints
150  Double_t sum = (P1P - MP).Mod() + (P2P - MP).Mod();
151  //cout << "sum = "<<sum<<" P1P2 = " << P1P2.Mod() << " difference = " << sum-P1P2.Mod() << endl;
152  if (TMath::Abs(sum - P1P2.Mod()) > 1.e-06) {
153  //cout << "take endpoint" << endl;
154  if (P1P.Mod() < P2P.Mod()) {
155  i_nearest_point = 0;
156  return -(P1P.Mod());
157  }
158  i_nearest_point = 1;
159  return -(P2P.Mod());
160  }
161  //cout << "point is close to segment" << endl;
162  return MP.Mod();
163 }

168 inline Bool_t KVIDLine::PosRelToLine(Option_t* opt, Double_t px,
169  Double_t py, Double_t xp1,
170  Double_t yp1, Double_t xp2,
171  Double_t yp2)
172 {
173  //Given a line segment defined by endpoints (xp1,yp1) and (xp2,yp2) test the
174  // direction of (px,py) with respect to the line in the (x,y) plane.
175  //
176  // M
177  // P1 (xp1,yp1) x---------------------------------------x P2 (xp2,yp2)
178  // |
179  // |
180  // |
181  // |
182  // |
183  // x
184  // P (px,py)
185  //This is given by the angle phi between vector MP (component of vector P1->P (or P2->P)
186  //perpendicular to P1->P2) and the +ve x-axis.
187  //Point is left of line if 90<phi<270
188  //Point is right of line if 270<phi<90
189  //Point is above line if 0<phi<180
190  //Point is below line if 180<phi<360
191  //The option string is "left", "right", "above" or "below", and the result is either kTRUE or kFALSE
192  //depending on the relative position of the point and the line.
193  //E.g. in the diagram shown above, PosRelToLine("below", ...) would give kTRUE - the point is
194  //below the line.
195 
196  TVector2 P1(xp1, yp1), P2(xp2, yp2), P(px, py);
197  TVector2 P1P2 = P2 - P1;
198  TVector2 P1P = P - P1;
199  TVector2 MP = P1P.Norm((P1P2));
200  Double_t phi = MP.Phi() * TMath::RadToDeg();
201  //Point is left of line if 90<phi<270
202  //Point is right of line if 270<phi<90
203  //Point is above line if 0<phi<180
204  //Point is below line if 180<phi<360
205  Bool_t result = kFALSE;
206  if (!strcmp(opt, "left")) {
207  result = (phi > 90. && phi < 270.);
208  } else if (!strcmp(opt, "right")) {
209  result = (phi < 90. || phi > 270.);
210  } else if (!strcmp(opt, "above")) {
211  result = (phi > 0. && phi < 180.);
212  } else if (!strcmp(opt, "below")) {
213  result = (phi > 180. && phi < 360.);
214  }
215  return result;
216 }

220 inline Bool_t KVIDLine::WhereAmI(Double_t px, Double_t py, Option_t* opt)
221 {
222  //The relative position of point (px,py) with respect to the line is tested.
223  //The option string can be "left", "right", "above" or "below".
224  //WhereAmI( px, py, "above") returns kTRUE if the point is above the line etc. etc.
225  //
226  //First of all, the closest segment/point of the line to the point is found.
227  //Then the relative position of the point and this segment/point is tested.
228 
229  Double_t* XX, *YY;
230  Double_t xx, yy;
231  Int_t sign = 1;
232 
233  if (!strcmp(opt, "left")) {
234  XX = fY;
235  xx = py;
236  YY = fX;
237  yy = px;
238  } else if (!strcmp(opt, "right")) {
239  XX = fY;
240  xx = py;
241  YY = fX;
242  yy = px;
243  sign = -1;
244  } else if (!strcmp(opt, "above")) {
245  XX = fX;
246  xx = px;
247  YY = fY;
248  yy = py;
249  sign = -1;
250  } else if (!strcmp(opt, "below")) {
251  XX = fX;
252  xx = px;
253  YY = fY;
254  yy = py;
255  } else return kFALSE;
256 
257  Int_t i_start = 0;
258  Int_t i_stop = fNpoints - 1;
259  Int_t prev_i_stop = 0;
260  Bool_t same_sign = TMath::Sign(1., XX[i_start] - xx) == TMath::Sign(1., XX[i_stop] - xx);
261  while ((i_start < i_stop - 1) || same_sign) {
262 
263  if (same_sign && (prev_i_stop == 0)) break;
264  else if (same_sign) {
265  i_start = i_stop;
266  i_stop = prev_i_stop;
267  } else {
268  prev_i_stop = i_stop;
269  i_stop = (Int_t)((i_start + i_stop) / 2 + 0.5);
270  }
271 
272  same_sign = TMath::Sign(1., XX[i_start] - xx) == TMath::Sign(1., XX[i_stop] - xx);
273  }
274 
275  Double_t a = (YY[i_stop] - YY[i_start]) / (XX[i_stop] - XX[i_start]);
276  Double_t b = YY[i_start] - a * XX[i_start];
277 
278  Bool_t res = (sign * yy < sign * (a * xx + b));
279 
280  return res;
281 }

309 inline Bool_t KVIDLine::IsBetweenEndPoints(Double_t x, Double_t y,
310  const Char_t* axis) const
311 {
312  //Returns kTRUE for point (x,y) if :
313  // axis = "" (default) and both x and y lie inside the endpoints of the line x1 < x < x2, y1 < y < y2
314  // axis = "x" and x lies inside the endpoints x1 < x < x2
315  // axis = "y" and y lies inside the endpoints y1 < y < y2
316 
317  TString ax(axis);
318  ax.ToUpper();
319  Double_t x1, y1, x2, y2;
320  GetStartPoint(x1, y1);
321  GetEndPoint(x2, y2);
322  Double_t xmin = TMath::Min(x1, x2);
323  Double_t xmax = TMath::Max(x1, x2);
324  Double_t ymin = TMath::Min(y1, y2);
325  Double_t ymax = TMath::Max(y1, y2);
326 
327  Bool_t in_range_x = (x <= xmax && x >= xmin);
328  Bool_t in_range_y = (y <= ymax && y >= ymin);
329  if (ax == "X") {
330  return in_range_x;
331  } else if (ax == "Y") {
332  return in_range_y;
333  }
334 
335  return (in_range_x && in_range_y);
336 }

60 Double_t KVChannelEnergyAlpha::Compute(Double_t chan) const
61 {
62  Double_t gain = 1.;
63  KVDetector* det = GetDetector();
64  if (det)
65  gain = det->GetGain();
66  //Calculate the calibrated signal strength in volts for a given channel number.
67  if (fReady) {
68  return (fPar[0] + fPar[1] * chan + fPar[2] * chan * chan) / gain;
69  } else {
70  return 0.;
71  }
72 }

124 Double_t KVChannelEnergyAlpha::Invert(Double_t volts)
125 {
126  //Given the calibrated (or simulated) signal amplitude in volts,
127  //calculate the corresponding channel number according to the
128  //calibration parameters (useful for filtering simulations).
129 
130  Double_t gain = 1.;
131  KVDetector* det = GetDetector();
132  if (det)
133  gain = det->GetGain();
134  Int_t channel = 0;
135 
136  if (fReady) {
137  if (fPar[2]) {
138  // quadratic transfer function
139  Double_t c;
140  c = fPar[1] * fPar[1] - 4. * fPar[2] * (fPar[0] - gain * volts);
141  if (c < 0.0)
142  return -1;
143  c = (-fPar[1] + TMath::Sqrt(c)) / (2.0 * fPar[2]);
144  if (c < 0.0
145  && ((-fPar[1] - TMath::Sqrt(c)) / (2.0 * fPar[2])) > 0.0) {
146  c = (-fPar[1] - TMath::Sqrt(c)) / (2.0 * fPar[2]);
147  }
148  channel = (Int_t)(c + 0.5);
149  } else {
150  // linear transfer function
151  channel = (Int_t)(0.5 + (gain * volts - fPar[0]) / fPar[1]);
152  }
153  } else {
154  Warning("Compute", "Parameters not correctly initialized");
155  }
156  return (Double_t) channel;
157 }

65 void KVIDSiLiCsI_camp5::Initialize()
66 {
67  // Initialize telescope for current run.
68  // Pointers to grids for run are set, and if there is at least 1 (GG) grid,
69  // we set IsReadyForID = kTRUE
70 
71  fSiLi = GetDetector(1);
72  fCsI = GetDetector(2);
73  fGGgrid = fPGgrid = 0;
74  TIter next(fIDGrids);
75  KVIDGrid* grid = 0;
76  while ((grid = (KVIDGrid*)next())) {
77  if (!strcmp(grid->GetVarY(), "SILI_GG")) fGGgrid = (KVIDZAGrid*)grid;
78  else if (!strcmp(grid->GetVarY(), "SILI_PG")) fPGgrid = (KVIDZAGrid*)grid;
79  }
80  if (fGGgrid) {
81  SetBit(kReadyForID);
82  fGGgrid->Initialize();
83  if (fPGgrid) fPGgrid->Initialize();
84  } else
85  ResetBit(kReadyForID);
86 }

121 Bool_t KVIDSiLiCsI_camp5::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
122 {
123  //Particle identification and code setting using identification grids.
124 
125  //perform identification in SiLi(GG) - CsI(R) map
126  IDR->SetIDType(GetType());
127  IDR->IDattempted = kTRUE;
128 
129  Double_t sili = (y < 0. ? GetIDMapY("GG") : y);
130  Double_t csir = (x < 0. ? GetIDMapX() : x);
131 
132  KVIDGrid* theIdentifyingGrid = 0;
133 
134  fGGgrid->Identify(csir, sili, IDR);
135  theIdentifyingGrid = (KVIDGrid*)fGGgrid;
136 
137  if (fGGgrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fPGgrid) { //we have to try PG grid (if there is one)
138 
139  // try Z & A identification in SiLi(PG)-CsI(R) map
140  sili = (y < 0. ? GetIDMapY("PG") : y);
141  fPGgrid->Identify(csir, sili, IDR);
142  theIdentifyingGrid = (KVIDGrid*)fPGgrid;
143  }
144 
145 
146  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
147  // only if the final quality code is kICODE8 do we consider that it is
148  // worthwhile looking elsewhere. In all other cases, the particle has been
149  // "identified", even if we still don't know its Z and/or A (in this case
150  // we consider that we have established that they are unknowable).
151  return kFALSE;
152  }
153 
154  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
155  // if the final quality code is kICODE7 (above last line in grid) then the estimated
156  // Z is only a minimum value (Zmin)
157  IDR->IDcode = kIDCode5;
158  return kTRUE;
159  }
160 
161  if (theIdentifyingGrid->GetQualityCode() > KVIDZAGrid::kICODE3 &&
162  theIdentifyingGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
163  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
164  // corresponds to a point which is inbetween the lines, i.e. noise
165  IDR->IDcode = kIDCode10;
166  return kTRUE;
167  }
168 
169  // set general ID code SiLi-CsI
170  IDR->IDcode = kIDCode3;
171  return kTRUE;
172 }

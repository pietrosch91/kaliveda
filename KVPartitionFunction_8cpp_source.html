12 <h2>KVPartitionFunction</h2>
13 <h4>Calculates number of partitions of (A,Z,M)</h4>
14 These functions calculate the total number of partitions and the number of partitions
15 per multiplicity either for a single component system (corresponding to the partition function
16 of number theory, i.e. the number of ways to decompose an integer into distinct sums of other
17 integers), or for a system made of two kinds of objects, i.e. protons and neutrons, with A=N+Z.
18 These are exact results, the numbers correspond to each and every partition appearing
19 once and only once.
20 
21 <h3>Single component system</h3>
22 Double_t PartFunc(Int_t A, Int_t M) calculates the total number of ways to partition an integer
23 A into the sum of M integers (1<=M<=A). We use the recursion relation given by J.P. Bondorf in
24 Nucl. Phys. A443, 321 (1985), Eq. (2.6).
25 
26 Double_t PartSum(Int_t A) returns the total number of partitions summed over all multiplicities.
27 
28 <h3>Two component system</h3>
29 Double_t PartFunc(Int_t A, Int_t Z, Int_t M) calculates the total number of ways to
30 partition Z protons and (A-Z) neutrons into M fragments, using the method given by K. Sneppen
31 in Nucl. Phys. A470, 213 (1987), Eqs. (4)-(6).
32 
33 Double_t PartSum(Int_t A, Int_t Z) returns the total number of partitions summed over all multiplicities.

105 Double_t KVPartitionFunction::PartFunc(int A, int M)
106 {
107  // recursive calculation of number of partitions of A nucleons
108  // into M fragments of size 1, 2, ..., A
109  // from Eq. (2.6) of Bondorf et al., Nucl. Phys. A443, 321 (1985):
110  /* "We now consider the division of the partition space into the subsets characterized
111  by having a common multiplicity. Let P(A,, M) be the number of partitions in one
112  of such subsets. To compute P(A,, M) it is sufficient to observe that it should be
113  equal to the number of partitions having at least one fragment composed of just
114  one nucleon (which number is immediately shown to equal P(A0 - 1, M - 1)) plus
115  those not having any unitary fragment. The number of this last portion can be
116  computed by observing that, if we take one nucleon from each of its fragments one
117  has the set of all partitions of A0 - M nucleons into M fragments. We thus arrive at
118  P(A0,M) = P(A0-1,M-l) + P(A0-M, M) Eq. (2.5)"
119  */
120 // The expression below follows from recursive use of this relation.
121 
122  if (!A) return 0;
123  if (A == M) return 1;
124  if (M == 1) return 1;
125  Double_t p = 0;
126  int klim = A / M - 1;
127  for (int k = 0; k <= klim; k++) {
128  p += PartFunc(A - k * M - 1, M - 1);
129  }
130  return p;
131 }

194 Double_t KVPartitionFunction::calc_sneppen_Nclass(int A, int Z, int M, int B)
195 {
196  // calculates Eqs. 5a-d
197 
198  if (M <= 0) return 0;
199  if (A <= 0) return 0;
200  if (Z < 0) return 0;
201  if (B < 0) return 0;
202  if (B > A - Z) return 0;
203  if (Z == 0 && B == 0) return 0;
204  if ((M - 1) > (A - B)) return 0;
205  if (M == A) {
206  if (Z == A && B > 0) return 0;
207  if (B != 1) return 0;
208  }
209  if (Z == 0) {
210  if (A - B * M > 0) {
211  Double_t sn1 = sneppen_Np(A - (B - 1) * M, 0, M);
212  Double_t sn2 = sneppen_Np(A - B * M, 0, M);
213  return sn1 - sn2;
214  }
215  return 0;
216  }
217  if (B == 0) {
218  if (A - M >= M && Z - M >= 0) {
219  Double_t snp, snc;
220  snp = sneppen_Np(A - M, Z - M, M);
221  snc = sneppen_Nclass(A - 1, Z - 1, M - 1, 0);
222  return snp + snc;
223  }
224  return 0;
225  }
226  Double_t ncl = 0;
227  if (A - B > 0 && M - 1 > 0 && B > 1) {
228  for (int i = 1; i <= B - 1; i++) {
229  ncl += sneppen_Nclass(A - B, Z, M - 1, i);
230  }
231  }
232  ncl = sneppen_Np(A - B, Z, M - 1) - ncl;
233  return ncl;
234 }

311 Double_t KVPartitionFunction::MeanNA(int A0, int A)
312 {
313  // Calculate the mean number of clusters of size A when a system of size A0
314  // fragments in all possible ways with equal probability
315  // Using Eq. (3) of K. Sneppen Nucl. Phys. A470, 213 (1987)
316  //
317  // Correction for A=A0:
318  // there is always exactly 1 partition of M=1 composed of one fragment with A=A0
319  // the mean multiplicity of A=A0 is therefore 1/PartSum(A0)
320 
321  if (A == A0) return 1. / PartSum(A0);
322  Double_t NA = 0;
323  Int_t imax = (Int_t)(1.*A0 / (1.*A));
324  for (int i = 1; i <= imax; i++) {
325  Int_t A1 = A0 - i * A;
326  if (A1 > 0) NA += PartSum(A1);
327  }
328  NA /= PartSum(A0);
329  return NA;
330 }

352 Double_t KVPartitionFunction::MeanNA_M(int A0, int A, int M)
353 {
354  // Calculate the mean number of clusters of size A when a system of size A0
355  // breaks up into M fragments, each partition having equal probability
356  // Using Eqs. (2b) & (3) of K. Sneppen Nucl. Phys. A470, 213 (1987)
357  //
358  // Correction for M=1:
359  // if A==A0, mean multiplicity is 1; otherwise 0.
360  // Correction for M=2 and even values of A0:
361  // when an odd-A0 splits into M=2, there are PartFunc(A0,M) partitions
362  // in which each A between 1 and A0-1 occurs once only, therefore
363  // the mean multiplicity for any A is 1/PartFunc(A0,M)
364  // when an even-A0 splits into M=2, one of the PartFunc(A0,M) partitions
365  // is the symmetric split (A0/2, A0/2). Therefore the mean multiplicity
366  // of A=A0/2 is twice that of the other A, i.e. 2/PartFunc(A0,M)
367  // Correction for M=A0:
368  // in this case only one partition exists, made of M=A0 monomers A=1
369  // therefore mean multiplicity of A=1 is M, for all other A it is 0.
370 
371  if (M == 1) {
372  if (A == A0) return 1.;
373  else return 0.;
374  }
375  if (M == 2) {
376  if (!(A0 % 2) && A == A0 / 2) return 2. / PartFunc(A0, M);
377  else if (A >= 1 && A < A0) return 1. / PartFunc(A0, M);
378  else return 0.;
379  }
380  if (M == A0) {
381  if (A == 1) return M;
382  else return 0.;
383  }
384  Double_t NA = 0;
385  Int_t imax = (Int_t)(1.*A0 / (1.*A));
386  imax = TMath::Min(imax, M - 1);
387  for (int i = 1; i <= imax; i++) {
388  Int_t A1 = A0 - i * A;
389  if (A1 > 0)NA += PartFunc(A1, M - i);
390  }
391  NA /= PartFunc(A0, M);
392  return NA;
393 }

406 {
407  // Calculate the mean number of clusters of mass A and charge Z
408  // when a two-component system (A0,Z0) fragments in all possible ways with equal probability
409  // Using Eq. (7) of K. Sneppen Nucl. Phys. A470, 213 (1987)
410 
411  Double_t NAZ = 0;
412  Int_t imax = (Int_t)(1.*A0 / (1.*A));
413  for (int i = 1; i <= imax; i++) {
414  Int_t A1 = A0 - i * A;
415  Int_t Z1 = Z0 - i * Z;
416  if (A1 > 0) NAZ += PartSum(A1, Z1);
417  }
418  NAZ /= PartSum(A0, Z0);
419  return NAZ;

464 Double_t KVPartitionFunction::MeanM(int A0)
465 {
466  // Calculate the mean multiplicity of all partitions of a single-component
467  // system of mass A0.
468  Double_t mult = 0.;
469  Double_t part_sum = 0;
470  Double_t part_func = 0;
471  for (int i = 1; i <= A0; i++) {
472  part_func = PartFunc(A0, i);
473  mult += i * part_func;
474  part_sum += part_func;
475  }
476  mult /= part_sum;
477  return mult;

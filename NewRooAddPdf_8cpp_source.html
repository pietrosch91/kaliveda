853 RooAbsReal* NewRooAddPdf::createNLL(RooAbsData& data, const RooCmdArg& arg1, const RooCmdArg& arg2, const RooCmdArg& arg3, const RooCmdArg& arg4,

855 {
856  // Construct representation of -log(L) of PDFwith given dataset. If dataset is unbinned, an unbinned likelihood is constructed. If the dataset
857  // is binned, a binned likelihood is constructed.
858  //
859  // The following named arguments are supported
860  //
861  // ConditionalObservables(const RooArgSet& set) -- Do not normalize PDF over listed observables
862  // Extended(Bool_t flag) -- Add extended likelihood term, off by default
863  // Range(const char* name) -- Fit only data inside range with given name
864  // Range(Double_t lo, Double_t hi) -- Fit only data inside given range. A range named "fit" is created on the fly on all observables.
865  // Multiple comma separated range names can be specified.
866  // SumCoefRange(const char* name) -- Set the range in which to interpret the coefficients of RooAddPdf components
867  // NumCPU(int num, int strat) -- Parallelize NLL calculation on num CPUs
868  //
869  // Strategy 0 = RooFit::BulkPartition (Default) --> Divide events in N equal chunks
870  // Strategy 1 = RooFit::Interleave --> Process event i%N in process N. Recommended for binned data with
871  // a substantial number of zero-bins, which will be distributed across processes more equitably in this strategy
872  // Strategy 2 = RooFit::SimComponents --> Process each component likelihood of a RooSimultaneous fully in a single process
873  // and distribute components over processes. This approach can be benificial if normalization calculation time
874  // dominates the total computation time of a component (since the normalization calculation must be performed
875  // in each process in strategies 0 and 1. However beware that if the RooSimultaneous components do not share many
876  // parameters this strategy is inefficient: as most minuit-induced likelihood calculations involve changing
877  // a single parameter, only 1 of the N processes will be active most of the time if RooSimultaneous components
878  // do not share many parameters
879  // Strategy 3 = RooFit::Hybrid --> Follow strategy 0 for all RooSimultaneous components, except those with less than
880  // 30 dataset entries, for which strategy 2 is followed.
881  //
882  // Optimize(Bool_t flag) -- Activate constant term optimization (on by default)
883  // SplitRange(Bool_t flag) -- Use separate fit ranges in a simultaneous fit. Actual range name for each
884  // subsample is assumed to by rangeName_{indexState} where indexState
885  // is the state of the master index category of the simultaneous fit
886  // Constrain(const RooArgSet&pars) -- For p.d.f.s that contain internal parameter constraint terms, only apply constraints to given subset of parameters
887  // ExternalConstraints(const RooArgSet& ) -- Include given external constraints to likelihood
888  // GlobalObservables(const RooArgSet&) -- Define the set of normalization observables to be used for the constraint terms.
889  // If none are specified the constrained parameters are used
890  // GlobalObservablesTag(const char* tagName) -- Define the set of normalization observables to be used for the constraint terms by a string attribute
891  // associated with pdf observables that match the given tagName
892  // Verbose(Bool_t flag) -- Constrols RooFit informational messages in likelihood construction
893  // CloneData(Bool flag) -- Use clone of dataset in NLL (default is true)
894  // Offset(Bool_t) -- Offset likelihood by initial value (so that starting value of FCN in minuit is zero). This
895  // can improve numeric stability in simultaneously fits with components with large likelihood values
896  //
897  //
898 
899  RooLinkedList l ;
900  l.Add((TObject*)&arg1) ;
901  l.Add((TObject*)&arg2) ;
902  l.Add((TObject*)&arg3) ;
903  l.Add((TObject*)&arg4) ;
904  l.Add((TObject*)&arg5) ;
905  l.Add((TObject*)&arg6) ;
906  l.Add((TObject*)&arg7) ;
907  l.Add((TObject*)&arg8) ;
908  return createNLL(data, l) ;

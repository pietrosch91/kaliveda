14 // This class fills, diagonalizes the momentum tensor for particles in an
15 // event and returns several useful values. This is based on the KVTenseur3 class.
16 //
17 //By default, only particles with Z >= Zmin (default value=3) are included in
18 //the calculation. Use SetZmin(0) or SetZmin(1) to include all particles.
19 //
20 //Use SetPartGroup() in order to use only particles belonging to a given group
21 //(see KVParticle::AddGroup and KVParticle::BelongsToGroup).
22 //Note that the condition on the particle's Z is still used, thus
23 //if e.g., SetZmin(10) and SetPartGroup("alpha") are used, only
24 //particles belonging to group "alpha" and Z>=10 will be included.
25 //
26 //By default the tensor is calculated from particle momenta
27 //in the default (laboratory) frame.
28 //However, this can be changed by giving the name of the desired frame
29 //to either the constructor
30 // KVTensP* tp = new KVTensP("my_tensor", "CM"); // calculate in "CM" frame
31 //or by using method SetFrame
32 // tp->SetFrame( "QP" ); // calculate in "QP" frame
33 //
34 // The names of the variables and the corresponding indices are as follows:
35 //
36 // Name Index Meaning
37 //----------------------------------------------------------------------------
38 // ThetaFlot 0 (default) Theta flow (in degrees between 0 and +90)
39 // Sphericite 1 Sphericity
40 // Coplanarite 2 Coplanarity
41 // PhiPlan 3 Phi of the reaction plane (in degrees between 0 and 360)
42 // Vap1 4 Value of the 1st eigenvalue
43 // Vap2 5 Value of the 2nd eigenvalue
44 // Vap3 6 Value of the 3rd eigenvalue
45 //
46 // All these values can be obtained by calling the GetValuePtr() method which
47 // returns an array of Double_t containing the values.
48 //
49 // Here are examples on how to obtain the values.
50 //
51 // KVTensP *ptens=new KVTensP();
52 // KVTensP tens;
53 // ...
54 // Double_t tf=ptens->GetValue(); |
55 // tf=ptens->GetValue(0); |--> Theta Flow
56 // tf=tens("ThetaFlot"); |
57 // Double_t phiplane=tens(1);---------> Phi of reaction plane
58 // Double_t copla=ptens->GetValue("Coplanarite");--> Coplanarity
59 //
60 // Double_t *values=ptens->GetValuePtr();
61 // Double_t s=values[1] --------------> Sphericity
62 // tf=values[0] -------------> Theta Flow
63 //
64 // Look at KVVarGlob class to have an example of use.
65 //
66 //

356 void KVTensP::Fill(KVNucleus* c)
357 {
358  //Add the particle's contribution to the momentum tensor.
359  //
360  //This particle will be rejected if:
361  //(1) it has Z < Zmin
362  //(2) it does not belong to the group defined with SetPartGroup**
363  //
364  // **if no group defined, only Z>=Zmin condition is used.
365  //
366  //This uses the 3 cartesian components of the particle's velocity vector
367  //with a weight equal to the particle's A.
368  //
369  //By default the velocity vector in the default (laboratory) frame is used.
370  //To calculate in another frame, use KVVarGlob::SetFrame() before
371  //calculation begins
372 
373  // if method is called by FillWithCondition i.e. by KVGVList::CalculateGlobalVariables
374  // (conditioned_fill==kTRUE), all selections have already been tested
375  // if not i.e. Fill method called "by hand", we need to test here
376  if (conditioned_fill || fSelection->Test(c)) {
377  Double_t poids = (Double_t) c->GetA();
378  tenseurP->Fill(c->GetFrame(fFrame.Data(), kFALSE)->GetV(), poids);
379  }
380 }

70 void KVSimDirAnalyser::SubmitTask()
71 {
72  // Set up and run the analysis
73 
74  BuildChain();
75 
76  Bool_t read_all_events = (GetNbEventToRead() == 0);
77  Long64_t nevents = (!read_all_events ? GetNbEventToRead() : fAnalysisChain->GetEntries());
78  Long64_t update_interval = (nevents > 10 ? nevents / 10 : 1);
79  TString results_file_name;
80  KVSimFile* first_file = (KVSimFile*)fListOfSimFiles->First();
81  results_file_name.Form("%s_%s", GetUserClass(), first_file->GetName());
82  TString options;
83  options.Form("SimFileName=%s,SimTitle=%s,OutputDir=%s,EventsReadInterval=%lld,BranchName=%s,CombinedOutputFile=%s,SimulationInfos=%s",
84  first_file->GetName(), fAnalysisChain->GetTitle(), first_file->GetSimDir()->GetDirectory(),
85  update_interval, first_file->GetBranchName(),
86  results_file_name.Data(), first_file->GetTitle());
87  if (first_file->IsFiltered()) options += Form(",DataSet=%s,System=%s,Run=%d",
88  first_file->GetDataSet(), first_file->GetSystem(), first_file->GetRun());
89  // Add any user-defined options
90  if (GetUserClassOptions() != "") {
91  options += ",";
92  options += GetUserClassOptions();
93  }
94 
95  // Check compilation of user class & run
96  if (CheckIfUserClassIsValid()) {
97  Info("SubmitTask", "Beginning TChain::Process...");
98 #ifdef WITH_CPP11
99  if (GetProofMode() != KVDataAnalyser::EProofMode::None) fAnalysisChain->SetProof(kTRUE);
100 #else
101  if (GetProofMode() != KVDataAnalyser::None) fAnalysisChain->SetProof(kTRUE);
102 #endif
103  TString analysis_class;
104  if (GetAnalysisTask()->WithUserClass()) analysis_class.Form("%s%s", GetUserClassImp().Data(), GetACliCMode());
105  else analysis_class = GetUserClass();
106  if (read_all_events) {
107  fAnalysisChain->Process(analysis_class, options.Data());
108  } else {
109  fAnalysisChain->Process(analysis_class, options.Data(), GetNbEventToRead());
110  }
111  }
112  delete fAnalysisChain;
113  fAnalysisChain = nullptr;
114 }

139 void KVSimDirAnalyser::WriteBatchEnvFile(const Char_t* jobname, Bool_t sav)
140 {
141  //Save (in the TEnv fBatchEnv) all necessary information on analysis task which can be used to execute it later
142  //(i.e. when batch processing system executes the job).
143  //If save=kTRUE (default), write the information in a file whose name is given by ".jobname"
144  //where 'jobname' is the name of the job as given to the batch system.
145 
146  KVDataAnalyser::WriteBatchEnvFile(jobname, kFALSE);
147  KVSimFile* simF = (KVSimFile*)fListOfSimFiles->First();
148  KVSimDir* simD = simF->GetSimDir();
149  GetBatchInfoFile()->SetValue("SimDir", simD->GetDirectory());
150  if (simF->IsFiltered()) GetBatchInfoFile()->SetValue("SimFile.Type", "filtered");
151  else GetBatchInfoFile()->SetValue("SimFile.Type", "simulated");
152  GetBatchInfoFile()->SetValue("SimFiles", simF->GetName());
153  if (fListOfSimFiles->GetEntries() > 1) {
154  TIter next(fListOfSimFiles);
155  next();
156  while ((simF = (KVSimFile*)next())) GetBatchInfoFile()->SetValue("+SimFiles", simF->GetName());
157  }
158  if (sav) GetBatchInfoFile()->SaveLevel(kEnvUser);
159 }

168 Bool_t KVSimDirAnalyser::ReadBatchEnvFile(const Char_t* filename)
169 {
170  //Read the batch env file "filename" and initialise the analysis task using the
171  //informations in the file
172  //Returns kTRUE if all goes well
173 
174  Bool_t ok = kFALSE;
175 
176  if (!KVDataAnalyser::ReadBatchEnvFile(filename)) return ok;
177 
178  KVString simdir = GetBatchInfoFile()->GetValue("SimDir", "");
179  if (simdir == "") return ok;
180 
181  fSimDir = new KVSimDir("SIMDIR", simdir);
182  fSimDir->AnalyseDirectory();
183 
184  KVString filetype = GetBatchInfoFile()->GetValue("SimFile.Type", "");
185  if (filetype == "") return ok;
186 
187  KVString simfiles = GetBatchInfoFile()->GetValue("SimFiles", "");
188  if (simfiles == "") return ok;
189 
190  DeleteSimFilesListIfOurs();
191  fListOfSimFiles = new TList;
192  fListOfSimFiles->SetName("toDelete");
193 
194  simfiles.Begin(" ");
195  while (!simfiles.End()) {
196  if (filetype == "simulated") fListOfSimFiles->Add(fSimDir->GetSimDataList()->FindObject(simfiles.Next()));
197  else if (filetype == "filtered") fListOfSimFiles->Add(fSimDir->GetFiltDataList()->FindObject(simfiles.Next()));
198  }
199 
200  ok = kTRUE;
201 
202  return ok;
203 }

210 void KVSimDirAnalyser::BuildChain()
211 {
212  // Build a TChain with all files/trees to be analysed
213 
214  TIter next(fListOfSimFiles);
215  KVSimFile* file;
216  while ((file = (KVSimFile*)next())) {
217  if (!fAnalysisChain) {
218  fAnalysisChain = new TChain(file->GetTreeName());
219  }
220  TString fullpath;
221  AssignAndDelete(fullpath, gSystem->ConcatFileName(file->GetSimDir()->GetDirectory(), file->GetName()));
222  fAnalysisChain->Add(fullpath);
223  }
224 }

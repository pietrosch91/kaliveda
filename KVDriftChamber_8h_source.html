155 inline void KVDriftChamber::SetZOffsets(Float_t X1, Float_t X2, Float_t Y)
156 {
157  // Sets the values of Z offsets for X1, X2 and Y position measurements.
158  // By default these offsets are given by the position in the detector of :
159  // - the first line of pads of the cathode ( -2.5 cm ) for X1
160  // - the second line of pads of the cathode ( +2.5 cm ) for X2
161  // - the center of the detector ( 0 cm ) for Y
162 
163  fOffsetZ[0] = Y;
164  fOffsetZ[1] = X1;
165  fOffsetZ[2] = X2;
166 }

169 inline void KVDriftChamber::SetXOffsets(Float_t X1, Float_t X2)
170 {
171  // Sets the values of X1 and X2 offsets in fraction of the strip width.
172  // By default the The two cathode plans are offset by half a strip to
173  // reduce the non linearity of the position measurement in between the
174  // strips. The center of the detector is at the middle of the strip 32
175  // of the range 2 and between strips 32 and 33 of the range 1 (offsets 0,0).
176  // With this command you can change the strip position with respect to
177  // the center of the detector for each plan.
178  //
179  // For example, if you want to move the plan 2 in order to have its strip
180  // 35 at the center and keep the range 1 at its default position then use
181  // SetXOffsets( 0., 3.);
182  fOffsetX[0] = X1;
183  fOffsetX[1] = X2;
184 }

187 inline Int_t KVDriftChamber::GetPadMax(Int_t row_num)
188 {
189  // Returns the pad with the maximum charge from the histogram representing
190  // the calibrated charge versus strip number for the chosen row of the drift
191  // chamber.
192  // row_num = {1,2}
193 
194  if (row_num < 3 && row_num > 0) return fPadMax[ row_num - 1 ];
195  else return -1;
196 }

59 void KVIDSi150CsI_camp5::Initialize()
60 {
61  // Initialize telescope for current run.
62  // If there is at least 1 grid, we set IsReadyForID = kTRUE
63  // "Natural" line widths are calculated for grids.
64 
65  fSi = (KVSilicon*)GetDetector(1);
66  fSIPG = fSi->GetACQParam("PG");
67  fCsI = (KVCsI*)GetDetector(2);
68  fZGrid = fZAGrid = 0;
69  TIter next(fIDGrids);
70  KVIDGrid* grid = 0;
71  while ((grid = (KVIDGrid*)next())) {
72  if (grid->OnlyZId()) fZGrid = (KVIDZAGrid*)grid;
73  else fZAGrid = (KVIDZAGrid*)grid;
74  }
75  if (fZAGrid) {
76  SetBit(kReadyForID);
77  fZAGrid->Initialize();
78  if (fZGrid) fZGrid->Initialize();
79  } else
80  ResetBit(kReadyForID);
81 }

125 Bool_t KVIDSi150CsI_camp5::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
126 {
127  //Particle identification and code setting using identification grids
128 
129  IDR->SetIDType(GetType());
130  IDR->IDattempted = kTRUE;
131  KVIDGrid* theIdentifyingGrid = 0;
132 
133  // try full isotopic identification
134  Double_t sili = (y < 0. ? GetIDMapY() : y);
135  Double_t csi = (x < 0. ? GetIDMapX() : x);
136  fZAGrid->Identify(csi, sili, IDR);
137  theIdentifyingGrid = fZAGrid;
138 
139  if (fZAGrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fZGrid) {
140 
141  // particle is above Z&A grid: try Z only ID
142  fZGrid->Identify(csi, sili, IDR);
143  theIdentifyingGrid = fZGrid;
144  }
145 
146 
147  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
148  // only if the final quality code is kICODE8 do we consider that it is
149  // worthwhile looking elsewhere. In all other cases, the particle has been
150  // "identified", even if we still don't know its Z and/or A (in this case
151  // we consider that we have established that they are unknowable).
152  return kFALSE;
153  }
154 
155  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
156  // if the final quality code is kICODE7 (above last line in grid) then the estimated
157  // Z is only a minimum value (Zmin)
158  IDR->IDcode = kIDCode5;
159  return kTRUE;
160  }
161 
162  if (theIdentifyingGrid->GetQualityCode() > KVIDZAGrid::kICODE3 &&
163  theIdentifyingGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
164  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
165  // corresponds to a point which is inbetween the lines, i.e. noise
166  IDR->IDcode = kIDCode10;
167  return kTRUE;
168  }
169 
170  // set general ID code Si150-CsI
171  IDR->IDcode = kIDCode3;
172  return kTRUE;
173 }

16 // Calibration object dedicated to handle the position calibration
17 // (conversion position in channel -> cm) and to correct the deviation of
18 // the position in SeD detectors of VAMOS. For this purpose, an aluminium
19 // mask with calibrated holes of 5mm diameter is put in front of the emissive
20 // foil. The positions are corrected with a bidimensional polynomial of
21 // order 4 (method of P. Napolitani and S. Leblond) in two rounds. Let X and Y
22 // be the raw values (in channel) of positions. The calibration process is the
23 // following:
24 // - First round
25 // Begin_Latex X_{1} = #sum_{i=0}^{4} #sum_{j=0}^{4} C_{X1;i,j} X^{i}Y^{j} End_Latex Begin_Latex Y_{1} = #sum_{i=0}^{4} #sum_{j=0}^{4} C_{Y1;i,j} X_{1}^{j}Y^{i} End_Latex
26 // - Second round
27 // Begin_Latex X_{2} = #sum_{i=0}^{4} #sum_{j=0}^{4} C_{X2;i,j} X_{1}^{i}Y_{1}^{j} End_Latex Begin_Latex Y_{2} = #sum_{i=0}^{4} #sum_{j=0}^{4} C_{Y2;i,j} X_{2}^{j}Y_{1}^{i} End_Latex
28 // where Begin_Latex X_{2} End_Latex and Begin_Latex Y_{2} End_Latex are the calibrated positions obtained at the end
29 // of the process. This calibrator has then 100 parameters:
30 // - parameters 0 to 24 correspond to Begin_Latex C_{X1;i,j} End_Latex
31 // - parameters 25 to 49 correspond to Begin_Latex C_{Y1;i,j} End_Latex
32 // - parameters 50 to 74 correspond to Begin_Latex C_{X2;i,j} End_Latex
33 // - parameters 75 to 99 correspond to Begin_Latex C_{Y2;i,j} End_Latex
34 //
35 // The type of the calibrator is "position->cm Detector_name" if the detector
36 // is set in the constructor or "position->cm" if not.
37 //
38 // Calibrator used for the INDRA_VAMOS experiment e494s.

156 Bool_t KVSeDPositionCal::Compute(Double_t X, Double_t Y, Double_t& Xcal, Double_t& Ycal)
157 {
158  //Calculate calibrated X and Y position (Xcal, Ycal) in cm for the given raw X and Y values.
159 
160  Double_t Xtmp[2];
161  Xtmp[0] = X;
162  Xtmp[1] = Y;
163 
164  Xcal = Ycal = -666;
165  if (Xtmp[0] <= 0 || Xtmp[1] <= 0) return kFALSE;
166 
167  UShort_t num = 0;
168  for (Int_t h = 0; h < 2; h++) { // calibration process performed two times
169  for (Int_t i = 0; i < 2; i++) { // Correction in X and in Y
170 
171  // calibration on the SeD# plane in cm
172  Double_t cor = Xtmp[i];
173  Double_t other = Xtmp[ i ? 0 : 1 ];
174  Xtmp[i] = 0.;
175  for (Int_t j = 0; j <= SEDPOSCAL_FIT_ORDER; j++) {
176  for (Int_t k = 0; k <= SEDPOSCAL_FIT_ORDER; k++) {
177  if (GetParameter(num) != 0)
178  Xtmp[i] += GetParameter(num) * TMath::Power(cor, j) * TMath::Power(other, k);
179  num++;
180  }
181  }
182  }
183  }
184 
185  Xcal = Xtmp[0];
186  Ycal = Xtmp[1];
187 
188  return kTRUE;
189 }

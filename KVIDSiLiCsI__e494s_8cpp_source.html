58 void KVIDSiLiCsI_e494s::Initialize()
59 {
60  // Initialisation of telescope before identification.
61  // This method MUST be called once before any identification is attempted.
62  // IsReadyForID() will return kTRUE if KVTGID objects are associated
63  // to this telescope for the current run.
64 
65  fSiLi = (KVINDRADetector*)GetDetector(1);
66  fCsI = (KVCsI*)GetDetector(2);
67 
68  Bool_t ok = fSiLi && fCsI && GetListOfIDFunctions().GetEntries();
69  if (ok) {
70  KVTGID* idf = (KVTGID*)GetListOfIDFunctions().First();
71  fThresholdX = GetThesholdFromVar(idf->GetVarX());
72  fThresholdY = GetThesholdFromVar(idf->GetVarY());
73  // Info("Initialize","Thresholds for %s: X= %f, Y= %f", GetName(), fThresholdX, fThresholdY);
74  }
75 
76  SetBit(kReadyForID, ok);
77 }

109 Double_t KVIDSiLiCsI_e494s::GetIDMapY(Option_t* opt)
110 {
111  // This method gives the Y-coordinate in a 2D identification map
112  // associated with the SiLi-CsI identification telescope.
113  // The Y-coordinate is the SiLi current low gain coder data minus the
114  // low gain pedestal correction (see KVACQParam::GetDeltaPedestal()).
115  // If the high gain coder data is less than 3900 the the low gain value
116  // is calculated from the current high gain coder data minus the high
117  // gain pedestal correction (see KVINDRADetector::GetPGfromGG()).
118 
119  IGNORE_UNUSED(opt);
120 
121  if (fSiLi->GetGG() <= 3900.5)
122  return fSiLi->GetPGfromGG(fSiLi->GetGG() - fSiLi->GetACQParam("GG")->GetDeltaPedestal());
123 
124  return fSiLi->GetPG() - fSiLi->GetACQParam("PG")->GetDeltaPedestal();
125 }

143 Bool_t KVIDSiLiCsI_e494s::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
144 {
145  //Identification of particles using SiCorrelated-CsI matrices for E503/E494s
146  //First of all, Z identification is attempted with KVIDSiLiCsI_e494s::IdentZ.
147  //If successful, if this telescope has mass identification capabilities
148  //(HasMassID()=kTRUE), then if the identified Z is not too large for the
149  //GG grid Z&A identification, we identify the mass of the particle.
150  //In this case, a further check is made that the resulting nucleus is not
151  //too bizarre, i.e. it must be in the KVNucleus mass table (KVNucleus::IsKnown = kTRUE).
152  //If not, or if mass identification is not possible, the mass of the nucleus
153  //is deduced from Z using the default mass formula of class KVNucleus.
154  //
155  // Note that optional arguments (x,y) for testing identification are not used.
156 
157  Double_t X = (x < 0. ? GetIDMapX() : x);
158  Double_t Y = (y < 0. ? GetIDMapY() : y);
159 
160  Double_t funLTG_Z = -1;
161  Double_t funLTG_A = -1;
162  Double_t mass = -1;
163  Double_t Z = -1.;
164 
165  Int_t ia = -1;
166  Int_t iz = -1;
167 
168  IDR->SetIDType(GetType());
169  IDR->IDattempted = kTRUE;
170  IDR->IDquality = 15;
171 
172  // set general ID code
173  IDR->IDcode = fIDCode;
174 
175  const Bool_t inRange = (fThresholdX < X) && (X < 4090.) && (fThresholdY < Y) && (Y < 4090.);
176 
177  if (inRange) Z = IdentZ(GetName(), X, Y, funLTG_Z, ""); //IdentZ(this, funLTG_Z, "", "");
178  else return kFALSE;
179 
180  //use KVTGIDManager::GetStatus value for IdentZ as identification subcode
181  IDR->IDquality = GetStatus();
182 
183  if (GetStatus() == KVTGID::kStatus_NotBetween_IDMin_IDMax) return kFALSE;
184 
185  if (GetStatus() != KVTGIDManager::kStatus_OK) return kFALSE; // no ID
186 
187  iz = TMath::Nint(Z);
188 
189  if (iz < 1) {
190  IDR->IDquality = KVTGIDManager::kStatus_ZtooSmall;
191  return kFALSE;
192  }
193 
194  IDR->IDOK = kTRUE;
195 
196  //is mass identification a possibility ?
197  if (iz < 9) {
198 
199  mass = IdentA(GetName(), X, Y, funLTG_A, "", iz); //IdentA(this, funLTG_A, "", "", iz);
200 
201  if (GetStatus() != KVTGIDManager::kStatus_OK) { //mass ID not good ?
202 
203  //only Z identified
204  IDR->Z = iz;
205  IDR->PID = Z;
206  IDR->Zident = kTRUE;
207  //subcode says "Z ok but A failed because..."
208  IDR->IDquality = GetStatus();
209  } else { //both Z and A successful ?
210 
211  ia = TMath::Nint(mass);
212  // fix 16B
213  if (iz == 5 && ia == 16) ia = (mass > 16.0 ? 17 : 15);
214  // fix 9B
215  if (iz == 5 && ia == 9) ia = (mass > 9.0 ? 10 : 8);
216  // fix 8Be
217  if (iz == 4 && ia == 8) ia = (mass > 8.0 ? 9 : 7);
218  // fix 13Be
219  if (iz == 4 && ia == 13) ia = (mass > 13.0 ? 14 : 12);
220  // fix 10Li
221  if (iz == 3 && ia == 10) ia = (mass > 10.0 ? 11 : 9);
222  // fix 5He
223  if (iz == 2 && ia == 5) ia = (mass > 5.0 ? 6 : 4);
224  // fix 7He
225  if (iz == 2 && ia == 7) ia = (mass > 7.0 ? 8 : 6);
226  // check that mass is not too bizarre
227  KVNucleus tmp(iz, ia);
228  if (!tmp.IsKnown()) {
229  // if ia seems too light, we try with iz-1
230  // if ia seems too heavy, we try with iz+1
231  Info("Identify", "%s : initial ID Z=%d A=%d funLTG=%g", GetName(), iz, ia, funLTG_A);
232  Int_t iz2 = (ia < 2 * iz ? iz - 1 : iz + 1);
233  if (iz2 > 0) {
234  Double_t old_funLTG_A = funLTG_A;
235  Double_t new_mass = IdentA(GetName(), X, Y, funLTG_A, "", iz2); //IdentA(this, funLTG_A, "", "", iz2);
236  // is this a better solution ?
237  if (GetStatus() == KVTGIDManager::kStatus_OK) {
238  Int_t new_ia = TMath::Nint(new_mass);
239  tmp.SetZ(iz2);
240  tmp.SetA(new_ia);
241  Info("Identify", "%s : new ID Z=%d A=%d funLTG=%g", GetName(), iz2, new_ia, funLTG_A);
242  if (tmp.IsKnown() && (TMath::Abs(funLTG_A) < TMath::Abs(old_funLTG_A))) {
243  // new nucleus is known and point is closer to line
244  iz = iz2;
245  ia = new_ia;
246  mass = new_mass;
247  Info("Identify", "%s : accepted new ID", GetName());
248  } else {
249  Info("Identify", "%s : rejected new ID", GetName());
250  }
251  } else {
252  Info("Identify", "%s : failed to obtain new ID with Z=%d", GetName(), iz2);
253  }
254  }
255  }
256  IDR->Z = iz;
257  IDR->A = ia;
258  IDR->PID = mass;
259  IDR->Zident = kTRUE;
260  IDR->Aident = kTRUE;
261  }
262  } else {
263  //only Z identified
264  //ID subcode remains 'Zstatus'
265  IDR->Z = iz;
266  IDR->PID = Z;
267  IDR->Zident = kTRUE;
268 
269  }
270 
271  return kTRUE;
272 }

284 Bool_t KVIDSiLiCsI_e494s::SetIdentificationParameters(const KVMultiDetArray*)
285 {
286  //Initialise the identification parameters (grids, etc.) of ALL identification telescopes of this
287  //kind (label) in the multidetector array. Therefore this method need only be called once, and not
288  //called for each telescope. The kind/label (returned by GetLabel) of the telescope corresponds
289  //to the URI used to find the plugin class in $KVROOT/KVFiles/.kvrootrc.
290  //Parameters are read from the file with name given by the environment variable
291  //INDRA_camp5.IdentificationParameterFile.SI-CSI: [filename]
292 
293  TString filename = gDataSet->GetDataSetEnv(Form("IdentificationParameterFile.%s", GetLabel()));
294  if (filename == "") {
295  Warning("SetIdentificationParameters",
296  "No filename defined. Should be given by %s.IdentificationParameterFile.%s",
297  gDataSet->GetName(), GetLabel());
298  return kFALSE;
299  }
300  TString path;
301  if (!SearchKVFile(filename.Data(), path, gDataSet->GetName())) {
302  Error("SetIdentificationParameters",
303  "File %s not found. Should be in $KVROOT/KVFiles/%s",
304  filename.Data(), gDataSet->GetName());
305  return kFALSE;
306  }
307  //read grids from file
308  Info("SetIdentificationParameters", "Using file %s", path.Data());
309  return ReadAsciiFile(path.Data());
310 }

330 Double_t KVIDSiLiCsI_e494s::GetThesholdFromVar(const Char_t* var)
331 {
332  // returns the threshold deduced from VarX or VarY.
333  // If the string 'var' contains the character '>', we consider
334  // that the threshold is given in the string after this character.
335 
336  TString thresh = var;
337  Int_t idx = thresh.Index(">");
338  if (idx > -1) {
339  thresh.Remove(0, idx + 1);
340  return (thresh.IsFloat() ? thresh.Atof() : 0.);
341  }
342  return 0.;
343 }

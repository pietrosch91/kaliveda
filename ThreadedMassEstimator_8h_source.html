6 // ThreadedMassEstimator.h
7 //
8 // Description
9 //
10 // This class estimates the A value of a fragment detected in the
11 // Si:Isobutane:CsI detector stack, given the experimentally measured data (Z,
12 // Silicon Energy, CsI Light).
13 //
14 // It achieves this by minimising the absolute magnitude of the difference
15 // between the experimentally measured silicon energy and the silicon energy
16 // simulated in a set of test nuclei (with fixed Z) to determine which nucleus
17 // (A value) provides the closest match.
18 //
19 // Peter C. Wigg <peter.wigg.314159@gmail.com>
20 // Wed 20 Jan 15:19:33 2016

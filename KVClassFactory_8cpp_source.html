21 //Factory class for generating skeleton files for new classes.
22 //It can generate basic '.h' and '.cpp' files for
23 // - base classes (base_class="")
24 // - inherited classes (base_class!="")
25 // - classes based on template files
26 //
27 //There are two ways to use KVClassFactory in order to generate code:
28 //
29 //1. Using the static (stand-alone) method
30 //
31 // KVClassFactory::MakeClass(classname, classdesc, base_class, withTemplate, templateFile);
32 //
33 //2. Create and configure a KVClassFactory object and then call the GenerateCode() method:
34 //
35 // KVClassFactory fact(classname, classdesc, base_class, withTemplate, templateFile);
36 // fact.GenerateCode();
37 //
38 //The second method is more flexible and allows to add methods to classes before
39 //code generation, even if the class is created from a predefined template.
40 //
41 //For example, let us suppose that we want to add the following method to our class:
42 //
43 // declaration:
44 // virtual const Char_t* GetName(Option_t* = "") const;
45 //
46 // implementation:
47 // const Char_t* MyNewClass::GetName(Option_t* opt) const
48 // {
49 // //A new method
50 // if( strcmp(opt,"") ) cout << "Option : " << opt << endl;
51 // else cout << fName.Data() << endl;
52 // return fName.Data();
53 // }
54 //
55 //This can be done as follows:
56 //
57 // fact.AddMethod("GetName", "const Char_t*", kTRUE, kTRUE);
58 // fact.AddMethodArgument("GetName", "Option_t*", "opt", "\"\"");
59 // KVString body;
60 // body += " //A new method\n";
61 // body += " if( strcmp(opt,\"\") ) cout << \"Option : \" << opt << endl;\n";
62 // body += " else cout << fName.Data() << endl;\n";
63 // body += " return fName.Data();";
64 // fact.AddMethodBody("GetName", body);
65 //
66 //The addition of a new method may mean that it is necessary to add an '#include'
67 //directive to either the header or the implementation file of the new class. For example,
68 //in this case, the use of 'cout', 'endl' etc. may require to add an '#include "Riostream.h"'
69 //to the '.cpp' file of the new class. This can be done as follows:
70 //
71 // fact.AddImplIncludeFile("Riostream.h");
72 //

202  //Create a new class with the following characteristics:
203  //
204  // classname = name of new class
205  // classdesc = short (one line) description of class
206  // base_class = name of base class(es)* (if creating a derived class)
207  // withTemplate = kTRUE if template files for the '.h' and '.cpp' are to be used
208  // templateFile = base name of template files
209  //
210  // *(if the class has several base classes, give a comma-separated list)

213  //By default we create a new base class without use of template files.
214  //

218  //The template for the class structure is defined in a '.h' and a '.cpp' file,
219  //as follows:
220  //
221  // if templateFile="" (default), we expect base_class!="", and template files with names base_classTemplate.h and base_classTemplate.cpp
222  // must be present in either $TEMPLATEDIR, $HOME or $PWD directories.
223  // the dummy classname "base_classTemplate" will be replaced everywhere by 'classname'
224  //
225  // if templateFile="/absolute/path/classTemplate" we use classTemplate.h & classTemplate.cpp in the given directory.
226  // the dummy classname "classTemplate" will be replaced everywhere by 'classname'
227  //
228  // if templateFile="classTemplate" we look for classTemplate.h & classTemplate.cpp in $TEMPLATEDIR, $HOME or $PWD directories.
229  // the dummy classname "classTemplate" will be replaced everywhere by 'classname'
230

232  SetClassName(classname);
233  SetClassDesc(classdesc);
234  SetBaseClass(base_class);

948 Ssiz_t KVClassFactory::FindNextUncommentedLine(TString& file, Ssiz_t beg)
949 {
950  // Return position of beginning of next uncommented line in file starting
951  // from position beg (default = 0).
952  // An uncommented line is a line outside of a "/* ... */" block
953  // containing non-whitespace material before any '/*' or '//'.
954  // If no uncommented line is found we return -1.
955 
956  Ssiz_t pos = beg;
957  Ssiz_t beg_cur_line = beg;
958  Bool_t nonWS = kFALSE;
959 
960  while (pos < file.Length()) {
961 
962  if (file(pos) == ' ' || file(pos) == '\t') {
963  // skip whitespace & tab
964  pos++;
965  } else if (file(pos) == '\n') {
966  // reached newline without seeing comment
967  // if there is non-whitespace stuff on line, this is OK
968  if (nonWS) return beg_cur_line;
969  pos++;
970  beg_cur_line = pos;
971  } else if (file(pos) == '/') {
972 
973  if (file(pos + 1) == '*') {
974  // beginning of comment block
975  // was there non-whitespace stuff before this ?
976  if (nonWS) return beg_cur_line;
977  Ssiz_t len = file.Index("*/", pos + 2); // find end of comment
978  if (len > -1) pos = len + 2;
979  else {
980  // end of comment block not found in rest of file
981  // return -1
982  pos = -1;
983  break;
984  }
985  } else if (file(pos + 1) == '/') {
986  // beginning of 1-line comment
987  // was there non-whitespace stuff before this ?
988  if (nonWS) return beg_cur_line;
989  Ssiz_t len = file.Index('\n', pos + 2); // find end of line
990  if (len > 0) {
991  pos = len + 1;
992  beg_cur_line = pos;
993  } else {
994  // end of comment not found in rest of file
995  // return -1
996  pos = -1;
997  break;
998  }
999  }
1000  } else {
1001  //non-whitespace stuff
1002  nonWS = kTRUE;
1003  pos++;
1004  }
1005  }
1006  // no new line at end of first uncommented file which is also the last line in the file
1007  if (nonWS && pos > -1) return beg_cur_line;
1008  return pos;
1009 }

25 //Light-energy calibration for INDRA CsI detectors.
26 //
27 //The function Double_t CalculLumiere(Double_t * x, Double_t * par) (in the source file KVLightEnergyCsI.cpp)
28 //is used to parameterize the total light output as a function of deposited energy (in MeV).
29 //
30 //This function has 3 parameters, a1, a2 & a3. In order to correctly reproduce the light-energy
31 //relationship for all ions, two parameterizations have to be used: one for Z=1 and another for Z>1.
32 //The parameter a3 normally has a fixed value (a3=6), but this is not "hard-coded" : it should be fixed
33 //when fitting data.
34

66 Double_t CalculLumiere(Double_t* x, Double_t* par)
67 {
68  //Calcul de la lumiere totale a partir de Z, A d'une particule et son energie
69  //
70  // x[0] = energie (MeV)
71  // par[0] = a1
72  // par[1] = a2
73  // par[2] = a3
74  // par[3] = a4
75  // par[4] = Z
76  // par[5] = A
77 
78  Double_t Z = par[4];
79  Double_t A = par[5];
80 
81  Double_t energie = x[0];
82  Double_t c1 = par[0];
83  Double_t c2 = Z * Z * A * par[1];
84  Double_t c3 = A * par[2];
85  Double_t c4 = par[3];
86  Double_t T = 8 * A;
87  Double_t c4_new = c4 / (1. + TMath::Exp((c3 - energie) / T));
88  Double_t c0 = c4 / (1. + TMath::Exp(c3 / T));
89 
90  Double_t lumcalc = c1 * energie;
91  if (c2 > 0.0) {
92  Double_t xm = -c1 * c0 * c2 * TMath::Log(c2 / (c2 + c3));
93  lumcalc = lumcalc - c1 * c2 * TMath::Log(1. + energie / c2) + c1 * c2 * c4_new * TMath::Log((energie + c2) / (c3 + c2)) + xm;
94  }
95 
96  return lumcalc;
97 }

145 Double_t KVLightEnergyCsI::Compute(Double_t light) const
146 {
147  // Calculate the calibrated energy (in MeV) for a given total light output.
148  // The Z and A of the particle should be given first using SetZ, SetA.
149  // By default, Z=A=1 (proton).
150  //
151  // This is done by inversion of the light-energy function using TF1::GetX.
152 
153  SetParametersOfLightEnergyFunction();
154 
155  //invert light vs. energy function to find energy
156  Double_t xmin, xmax;
157  fLight.GetRange(xmin, xmax);
158  Double_t energy = fLight.GetX(light, xmin, xmax);
159 
160  return energy;
161 }

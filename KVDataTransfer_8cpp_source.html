22 // Transfers data between data repositories
23 //
24 // If the user has defined two or more repositories in her $HOME/.kvrootrc file,
25 // e.g. as in the example #3 given in $KVROOT/KVFiles/.kvrootrc:
26 //
27 // #Case 3: user has data files on his machine or on a locally-mounted disk, and can access
28 // #a remote data repository using xrootd via an SSH tunnel on port 10000 (the example given
29 // #is for the xrootd server at ccali centre de calcul).
30 //
31 // #definition of local repository
32 // DataRepository: default
33 // default.DataRepository.Type: local
34 // default.DataRepository.RootDir: $(HOME)/Data
35 //
36 // #definition of remote repository
37 // +DataRepository: ccali
38 // ccali.DataRepository.Type: remote
39 // ccali.DataRepository.RootDir: cchpssindra:/hpss/in2p3.fr/group/indra
40 // ccali.DataRepository.ReadProtocol: root
41 // ccali.DataRepository.XRDServer: localhost:10000
42 // ccali.DataRepository.XRDRootDir: /hpss/in2p3.fr/group/indra
43 // ccali.DataRepository.RemoteAvailableRuns.protocol: curl
44 // ccali.DataRepository.RemoteAvailableRuns.url: http://indra.in2p3.fr/KaliVedaDoc
45 // ccali.DataRepository.FileTransfer.type: bbftp
46 // ccali.DataRepository.FileTransfer.server: ccbbftp.in2p3.fr
47 // #if your login name is not the same as for ccali, give it here
48 // ccali.DataRepository.FileTransfer.user:
49 //
50 // then it is possible to transfer files from one to the other by running one of the following
51 // commands on the local machine:
52 //
53 // KVDataTransfer::NewTransfer("ccali", "default")->Run() // transfer from ccali to local machine
54 // KVDataTransfer::NewTransfer("default", "ccali")->Run() // transfer from local machine to ccali
55 //
56 // The same menu-driven approach as for submitting analysis tasks is used to
57 // allow the user to choose among the available datasets, systems and runs to
58 // transfer. The transferred files are copied in to the target repository, creating any
59 // missing directories which may be needed.
60 //
61 // It is also possible to use a KVDataTransfer object interactively, in this case
62 // you should use:
63 //
64 // KVDataTransfer* trans = KVDataTransfer::NewTransfer("ccali", "default");
65 // trans->SetDataSet(...);
66 // trans->SetDataType(...);
67 // trans->SetRuns(...);
68 // trans->TransferRuns();
69 //
70 // To use the BBFTP transfer facility, a bbftp client must be installed on the user's machine
71 // If the client is not located in the user's PATH, he should give the full path to the executable
72 // in the configuration, e.g. on WinXP with cygwin version of bbftp client:
73 //
74 // ccali.DataRepository.FileTransfer.type: c:\cygwin\bin\bbftp.exe

180 KVDataTransfer* KVDataTransfer::NewTransfer(const Char_t* source_rep,
181  const Char_t* target_rep)
182 {
183  //Creates a new data transfer object to transfer files between the two named data repositories.
184  //The type of the created object depends on the values of the environment variables
185  //(defined in .kvrootrc file):
186  //
187  //source_rep.DataRepository.FileTransfer.type
188  //target_rep.DataRepository.FileTransfer.type
189  //
190  //If either of these = bbftp, we create a KVDataTransferBBFTP object.
191  //If source_rep.DataRepository.FileTransfer.type = xrd, we create a KVDataTransferXRD object.
192  //By default (no type given), we use a KVDataTransferSFTP object.
193  //In fact, the type of object created is defined in .kvrootrc by the following plugins:
194  //
195  // # Plugins for data transfer between repositories
196  // # Used by KVDataTransfer::NewTransfer
197  // Plugin.KVDataTransfer: sftp KVDataTransferSFTP KVMultiDet "KVDataTransferSFTP()"
198  // +Plugin.KVDataTransfer: bbftp KVDataTransferBBFTP KVMultiDet "KVDataTransferBBFTP()"
199  // +Plugin.KVDataTransfer: xrd KVDataTransferXRD KVMultiDet "KVDataTransferXRD()"
200 
201  KVDataRepository* SR =
202  gDataRepositoryManager->GetRepository(source_rep);
203  if (!SR) {
204  ::Error("KVDataTransfer::NewTransfer",
205  "Unknown source data repository: %s", source_rep);
206  return 0;
207  }
208  KVDataRepository* TR =
209  gDataRepositoryManager->GetRepository(target_rep);
210  if (!TR) {
211  ::Error("KVDataTransfer::NewTransfer",
212  "Unknown target data repository: %s", target_rep);
213  return 0;
214  }
215  TString uri = "sftp"; //default plugin
216  //if either one has transfer type 'bbftp'...
217  if (!strcmp(SR->GetFileTransferType(), "bbftp")
218  || !strcmp(TR->GetFileTransferType(), "bbftp"))
219  uri = "bbftp";
220  //if source repository has transfer type 'xrd'...
221  else if (!strcmp(SR->GetFileTransferType(), "xrd"))
222  uri = "xrd";
223  TString transfer_exec = SR->GetFileTransferExec();
224  if (!strcmp(TR->GetFileTransferType(), "bbftp")) transfer_exec = TR->GetFileTransferExec();
225  //if transfer_exec=="" then we cannot perform the transfer
226  if (transfer_exec == "") {
227  ::Error("KVDataTransfer::NewTransfer",
228  "No file transfer executable to perform transfer");
229  return 0;
230  }
231 
232  //check and load plugin library
233  TPluginHandler* ph;
234  if (!(ph = KVBase::LoadPlugin("KVDataTransfer", uri)))
235  return 0;
236 
237  KVDataTransfer* tran = (KVDataTransfer*) ph->ExecPlugin(0);
238  //set target and source repositories
239  tran->fSourceRep = SR;
240  tran->fTargetRep = TR;
241  //set transfer client executable path
242  tran->SetTransferExec(transfer_exec);
243  //initialise transfer
244  tran->init();
245  return tran;
246 }

259 void KVDataTransfer::Run()
260 {
261  //Perform file transfer
262  //The user selects runs which are available in the source repository in the same way as
263  //when performing data analysis
264  //Note that this will make the 'source' repository the current active repository
265  //(i.e. gDataRepository and gDataSetManager will correspond to source repository
266  //after execution of this command).
267 
268  if (!fOK) {
269  Error("KVDataTransfer::Run",
270  "Status not OK. Data transfer aborted.");
271  return;
272  }
273  //make 'source' repository the 'active' repository
274  fSourceRep->cd();
275 
276  SetMenus();
277  SetQuit(kFALSE);
278  SetChooseDataSet();
279  SetChooseSystem(kFALSE);
280  SetChooseTask(kFALSE);
281  SetChooseRuns(kFALSE);
282  SetSubmit(kFALSE);
283 
284  set_dataset_pointer(nullptr);
285  SetDataType("");
286  ClearRunList();
287  SetSystem(nullptr);
288 
289  //choose dataset, data type, system, runs
290  while (!IsQuit()) {
291 
292  if (IsChooseDataSet())
293  ChooseDataSet();
294  else if (IsChooseTask())
295  ChooseDataType();
296  else if (IsChooseSystem())
297  ChooseSystem();
298  else if (IsChooseRuns())
299  ChooseRuns();
300  else if (IsSubmit())
301  TransferRuns();
302 
303  }
304 }

315 void KVDataTransfer::TransferRuns()
316 {
317  //Based on information gathered from user (see Run()),
318  //perform the transfer of files.
319  //Any missing directories in the target repository are created beforehand.
320  //After transfer, the available runlist for the target repository is updated
321 
322  SetSubmit(kFALSE);
323 
324  if (!GetDataSet()) {
325  Error("KVDataTransfer::Transfer",
326  "No dataset defined for transfer. Choose dataset first.");
327  SetChooseDataSet();
328  return;
329  }
330  if (GetDataType() == "") {
331  Error("KVDataTransfer::Transfer",
332  "No data type defined for transfer. Choose type of data to transfer first.");
333  SetChooseTask();
334  return;
335  }
336  if (GetRunList().IsEmpty()) {
337  Error("KVDataTransfer::Transfer",
338  "No runlist defined for transfer. Choose runs to transfer first.");
339  SetChooseRuns();
340  return;
341  }
342 
343  CheckTargetRepository();
344 
345  WriteTransferScript();
346 
347  ExecuteCommand();
348 
349  //delete temporary command file if present
350  if (fCmdFile != "") gSystem->Unlink(fCmdFile.Data());
351 
352  //update available run list for target repository
353  fTargetRep->GetDataSetManager()->GetDataSet(GetDataSet()->GetName())->UpdateAvailableRuns(GetDataType());
354 
355  //force user to choose new system
356  SetChooseSystem();
357 }

367 void KVDataTransfer::CheckTargetRepository()
368 {
369  //Make sure that the target repository has the necessary
370  //dataset/subdirectory to receive the transferred files.
371  //If not, we create the new dataset/subdir.
372 
373  Bool_t update = kFALSE;
374  if (!fTargetRep->GetDataSetManager()->GetDataSet(GetDataSet()->GetName())->IsAvailable()) {
375  //add dataset directory to target repository
376  fTargetRep->MakeSubdirectory(GetDataSet());
377  update = kTRUE;
378  }
379  if (!fTargetRep->GetDataSetManager()->GetDataSet(GetDataSet()->GetName())->HasDataType(GetDataType())) {
380  //add subdirectory for new data type to dataset directory
381  fTargetRep->MakeSubdirectory(GetDataSet(), GetDataType());
382  update = kTRUE;
383  }
384  if (update) {
385  //update dataset manager of target repository with new state of available datasets/datatypes
386  fTargetRep->GetDataSetManager()->Update();
387  }
388 }

436 void KVDataTransfer::set_dataset_pointer(KVDataSet* ds)
437 {
438  //Set dataset to be used for transfer.
439  //If the chosen dataset is not available, an error message is printed
440  //Only datasets which are available in the source repository can be transferred
441  //If the pointer actually corresponds to a dataset in the target repository,
442  //we replace it with a pointer to the dataset with the same name in the source
443  //repository.
444 
445  //allow user to reset dataset pointer to 0
446  _set_dataset_pointer(ds);
447  if (!ds)
448  return;
449 
450  //check repository
451  if (ds->GetRepository() != fSourceRep) {
452  _set_dataset_pointer(fSourceRep->GetDataSetManager()->GetDataSet(ds->GetName()));
453  }
454  //check availablility
455  if (!ds->IsAvailable()) {
456  Error("SetDataSet",
457  "Dataset %s is not available for analysis", ds->GetName());
458  _set_dataset_pointer(nullptr);
459  }

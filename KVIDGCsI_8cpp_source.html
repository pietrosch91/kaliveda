140 void KVIDGCsI::Identify(Double_t x, Double_t y, KVIdentificationResult* idr) const
141 {
142  // Set Z and A of nucleus based on position in R-L grid
143  // The identification of gammas (kICODE10) and charged particles is performed
144  // Note:
145  // for isotopically identified particles, the integer A (KVNucleus::GetA) is the mass assigned to the closest line
146  // [unless the closest line is the IMF line, in which case we use the closest identifier line],
147  // whereas the floating-point A (KVReconstructedNucleus::GetRealA) is calculated by interpolation.
148  // the integer A is not necessarily = nint(floating-point A): for example, if no 5He line is drawn in the grid
149  // (which is usually the case), there will be no isotopically-identified particle with GetA()=5, although
150  // there may be particles with GetRealA() between 4.5 and 5.5
151 
152  // before first use of this method, we add the IMF line to the list of identifiers
153  // which is necessary for IdentZA to work correctly
154  if (!fIMFlineadded) {
155  if (IMFLine) fIdentifiers->AddLast(IMFLine);
156  const_cast < KVIDGCsI* >(this)->fIMFlineadded = kTRUE;
157  }
158 
159  if (!IsIdentifiable(x, y)) {
160  //point below gamma line
161  const_cast < KVIDGCsI* >(this)->fICode = kICODE10;
162  idr->IDquality = fICode;
163  idr->Z = 0;
164  idr->A = 0;
165  idr->IDOK = kTRUE;
166  idr->SetComment("gamma");
167  return;
168  }
169  if (!const_cast<KVIDGCsI*>(this)->FindFourEmbracingLines(x, y, "above")) {
170  //no lines corresponding to point were found
171  const_cast < KVIDGCsI* >(this)->fICode = kICODE8; // Z indetermine ou (x,y) hors limites
172  idr->IDquality = fICode;
173  idr->SetComment("no identification: (x,y) out of range covered by grid");
174  return;
175  }
176  Int_t Z;
177  Double_t A;
178  const_cast < KVIDGCsI* >(this)->IdentZA(x, y, Z, A);
179  idr->Z = Z;
180  idr->A = Aint;
181  idr->PID = A;
182  idr->IDquality = fICode;
183  switch (fICode) {
184 
185  case kICODE0:
186  idr->SetComment("ok");
187  break;
188  case kICODE1:
189  idr->SetComment("Z ok, mass may be greater than A");
190  break;
191  case kICODE2:
192  idr->SetComment("Z ok, mass may be smaller than A");
193  break;
194  case kICODE3:
195  idr->SetComment("Z ok, mass may be greater or smaller than A");
196  break;
197  case kICODE4:
198  idr->SetComment("Z ok, mass out of range, >=A");
199  break;
200  case kICODE5:
201  idr->SetComment("Z ok, mass out of range, <=A");
202  break;
203  case kICODE6:
204  idr->SetComment("point above IMF line, Z is a minimum value");
205  break;
206  case kICODE7:
207  idr->SetComment("point is left of IMF line, Z is the most probable lower limit");
208  break;
209  case kICODE8:
210  idr->SetComment("no identification: (x,y) out of range covered by grid");
211  break;
212  case kICODE9:
213  idr->SetComment("no identification for this module");
214  break;
215  default:
216  idr->SetComment("no identification: (x,y) out of range covered by grid");
217  }
218 
219  if (fICode < kICODE4) {
220  idr->IDOK = kTRUE;
221  idr->Zident = kTRUE;
222  idr->Aident = kTRUE;
223  }
224 }

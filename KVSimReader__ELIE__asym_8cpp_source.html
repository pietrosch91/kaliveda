15 // This class is for reading asymptotic events, i.e. output files with names
16 // like: elie_58_197_32_1000_secondary.output
17 // The events will be stored in a ROOT file with a name like:
18 // ELIE_[PROJ]_[TARG]_[EBEAM]AMeV_ASYM.root
19 // This file will contain a TTree called "ELIE" with a branch "Simulated_evts"
20 // All the informations in the header of the ELIE output file are stored in
21 // a KVNameValueList in the TTree:
22 // ELIE->GetUserInfo()->FindObject("ELIE Parameters")->ls()
23 //KVNameValueList::ELIE Parameters : (0x55b2067172f0)
24 // <ievt=50>
25 // <projectile=Ni>
26 // <target=Au>
27 // <a_proj=58>
28 // <z_proj=28>
29 // <a_targ=197>
30 // <z_targ=79>
31 // <ebeam=32>
32 // <ilevel=0>
33 // <medium_effect=0>
34 // <g_al=1>
35 // <fraction_ex=0.5>
36 // <tempe_limite=5.5>
37 // <decay_in_flight=1>
38 // <lab_frame=0>
39 // <bminimum=0>
40 // <bmaximum=1>
41 // <choix_ex=0>
42 // <fermi=1>
43 // <version=1000>
44 // <periph_trick=0>
45 // <b_limite=0.75>
46 // <freeze_out=3>
47 // <proba_free_nucleon=0.001>
48 // <tole_pourcent=0.01>
49 // <discrete=1>
50 // <minimisation=0>
51 // <geometry=0>
52 // <free_nucleon=0>
53 // <lambda_ex=1.3>
54 //
55 // If parameter lab_frame=1, we transform the events to the CM frame.
56 // In this way, for further analysis/filtering, ELIE events are always in the CM frame.

164 void KVSimReader_ELIE_asym::ConvertEventsInFile(KVString filename)
165 {
166  if (!OpenFileToRead(filename)) return;
167  if (!ReadHeader()) return;
168  define_output_filename();
169  tree_title.Form("ELIE secondary events %s + %s %.1f MeV/nuc.",
170  proj.GetSymbol(), targ.GetSymbol(), ebeam);
171  Run();
172  CloseFile();
173 }

186 Bool_t KVSimReader_ELIE_asym::ReadNucleus()
187 {
188  // numero_particule,z, a, teta, phi, energie, origine de la particule, energie d'ecitation finale
189  // 0 27 57 3.27897003986 230.52425244 1109.37002505 0 0
190  // 1 80 198 176.726338776 129.481056376 319.364098122 1 0
191  //
192  // The origin of secondary decay particles is stored in a parameter named "ORIGIN"
193  // As particles in a KVEvent are numbered 1,2,... we add 1 to the value read in
194 
195  Int_t res = ReadLineAndCheck(8, " ");
196  switch (res) {
197  case 0:
198  Info("ReadNucleus", "case 0 line est vide");
199  return kFALSE;
200 
201  case 1:
202  nuc->SetZ(GetIntReadPar(1));
203  nuc->SetA(GetIntReadPar(2));
204  nuc->SetExcitEnergy(GetDoubleReadPar(7));
205  nuc->SetEnergy(GetDoubleReadPar(5));
206  nuc->SetTheta(GetDoubleReadPar(3));
207  nuc->SetPhi(GetDoubleReadPar(4));
208  nuc->SetParameter("ORIGIN", GetIntReadPar(6) + 1);
209  return kTRUE;
210  break;
211 
212  default:
213  return kFALSE;
214  }
215 
216  return kFALSE;
217 
218 }

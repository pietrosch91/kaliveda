67 void KVDatedFileManager::ReadFiles()
68 {
69  //Read all versions of file with given basename in directory and put them
70  //in a list sorted according to their timestamps
71  gSystem->ExpandPathName(fDirectory);
72  TSystemDirectory DIR("dir", fDirectory.Data());
73  TList* files = DIR.GetListOfFiles();
74  if (!files) {
75  Error(KV__ERROR(ReadFiles), "Cannot read directory %s", fDirectory.Data());
76  return;
77  }
78  TIter next(files);
79  TObject* file;
80  while ((file = next())) {
81  KVString nom(file->GetName());
82  if (nom.BeginsWith(fBaseName)) fFileList.Add(new KVSortableDatedFile(nom, fBaseName));
83  }
84  delete files;
85  if (fFileList.GetEntries()) fFileList.Sort();
86 }

98 const Char_t* KVDatedFileManager::GetPreviousVersion(const Char_t* name)
99 {
100  //Returns name of file with timestamp immediately prior to the given name.
101  //Example:
102  // to know the name of the most recent backup version, give the base name of the file
103  //
104  //If no older version exists, returns empty string
105 
106  TObject* obj = fFileList.FindObject(name);
107  if (!obj) {
108  Error(KV__ERROR(GetPreviousVersion), "File %s does not exist", name);
109  return "";
110  }
111  Int_t index = fFileList.IndexOf(obj);
112  if (index == (fFileList.GetEntries() - 1)) {
113  //file is already oldest version
114  Warning(KV__ERROR(GetPreviousVersion), "File %s is oldest known version", name);
115  return "";
116  }
117  return fFileList.At(index + 1)->GetName();
118 }

168 KVSortableDatedFile::KVSortableDatedFile(const Char_t* filename, const Char_t* basename)
169  : TNamed(filename, basename)
170 {
171  //filename = full name of file (including timestamp)
172  //basename = basename of file
173  //Example:
174  // KVSortableDatedFile("Runlist.csv.2007-05-02_14:15:58", "Runlist.csv")
175  //The timestamp extension is extracted from the filename.
176  //If the filename given is the same as the basename (i.e. no timestamp in name),
177  //the timestamps is set to the current date & time
178  KVString f(filename), b(basename);
179  if (f != b) {
180  f.Remove(0, b.Length() + 1);
181  f.ReplaceAll("_", " ");
182  fTimestamp.SetSQLDate(f.Data());
183  }
184 }

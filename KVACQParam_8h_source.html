141 inline Float_t KVACQParam::GetData() const
142 {
143  //Return value of data as floating-point number, by taking the value of
144  //GetCoderData() and adding a random number in the range [-0.5,+0.5]
145  //The range of values returned will thus go from -0.5 to MAX+0.5, where MAX is the maximum
146  //of the coder's range.
147  //GetData() = -1 if the parameter was not present in the event (coder not fired)
148 
149  if (GetCoderData() >= 0)
150  return (Float_t)(GetCoderData()) + gRandom->Uniform(-0.5, 0.5);
151  return -1.;
152 }
153 
154 inline Short_t KVACQParam::GetCoderData() const
155 {
156  //Returns value of parameter as read back from raw acquisition event
157  //GetCoderData() = -1 if the parameter was not present in the event (coder not fired)
158  if (fChannel)
159  return (Short_t)(*fChannel);
160  return (Short_t) fData;
161 }

169 inline Bool_t KVACQParam::Fired(Option_t* what) const
170 {
171  // Returns kTRUE if acquisition parameter fired in event.
172  // Different definitions of "fired" can be used:
173  //
174  // what = "" (default) : acquisition parameter present in event (i.e. not equal to -1)
175  // what = "P" : acquisition parameter present in event and greater than currently set pedestal
176 
177  return (what[0] ? ((Float_t)GetCoderData() > fPied) : (GetCoderData() != -1));
178 };

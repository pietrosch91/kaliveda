61 void KVFAZIAReconDataAnalyser::SubmitTask()
62 {
63  //Run the interactive analysis
64  if (gDataSet != GetDataSet()) GetDataSet()->cd();
65 
66  KVEventSelector es;
67  es.SetOption(GetUserClassOptions());
68  es.ParseOptions();
69  fLinkRawData = es.IsOptGiven("ReadRawData");
70 
71  //loop over runs
72  GetRunList().Begin();
73  while (!GetRunList().End()) {
74  fRunNumber = GetRunList().Next();
75  Info("SubmitTask", "treatment of run# %d", fRunNumber);
76  ProcessRun();
77  }
78 }

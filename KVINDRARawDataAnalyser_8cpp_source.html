151 KVNumberList KVINDRARawDataAnalyser::PrintAvailableRuns(KVString& datatype)
152 {
153  //Prints list of available runs, sorted according to multiplicity
154  //trigger, for selected dataset, data type/analysis task, and system
155  //Returns list containing all run numbers
156 
157  KVNumberList all_runs =
158  GetDataSet()->GetRunList(datatype.Data(), GetSystem());
159  KVINDRADBRun* dbrun;
160 
161  //first read list and find what triggers are available
162  vector<int> triggers;
163  all_runs.Begin();
164  while (!all_runs.End()) {
165  dbrun = (KVINDRADBRun*)GetDataSet()->GetDataBase()->GetTable("Runs")->GetRecord(all_runs.Next());
166  if (triggers.size() == 0
167  || std::find(triggers.begin(), triggers.end(), dbrun->GetTrigger()) != triggers.end()) {
168  triggers.push_back(dbrun->GetTrigger());
169  }
170  }
171  //sort triggers in ascending order
172  std::sort(triggers.begin(), triggers.end());
173 
174  for (std::vector<int>::iterator it = triggers.begin(); it != triggers.end(); ++it) {
175  cout << " ---> Trigger M>" << *it << endl;
176  all_runs.Begin();
177  while (!all_runs.End()) {
178  dbrun = (KVINDRADBRun*)GetDataSet()->GetDataBase()->GetTable("Runs")->GetRecord(all_runs.Next());
179  if (dbrun->GetTrigger() == *it) {
180  cout << " " << Form("%4d", dbrun->GetNumber());
181  cout << Form("\t(%7d events)", dbrun->GetEvents());
182  cout << "\t[File written: " << dbrun->GetDatime().
183  AsString() << "]";

185  cout << "\t" << dbrun->GetComments();

187  }
188  }

190  }
191  return all_runs;

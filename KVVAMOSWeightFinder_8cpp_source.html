218 void KVVAMOSWeightFinder::ReadInformations(std::ifstream& file)
219 {
220  // Read informations in run list file and save them only for

222

411 void KVVAMOSWeightFinder::ReadTransCoefListFile(std::ifstream& file)
412 {
413  //Read the trans. coef. list file. for the data set
414  //For each file listed in this list file we:
415  //- save in 'fvec_TCfiles' triplet (VamosAngle, IDCode, file_name) containing the VamosAngle, the associated IDCode and its associated file
416  //- read the file TEnv and save steps informations in 'fvec_TCsteps'
417  //- Add the tree in fchain
418  //
419  //NOTE: for e503 ChIo-Si IDCode are 4 (above punch-through line) or 11 (below punch-through line)
420  // the acceptance is the same for IDCode 4 or 11
421 
422  std::string newline;
423  while (std::getline(file, newline)) {
424  //Ignore comments in the file
425  if (newline.compare(0, 1, "#") != 0) {
426  TString ss(newline.c_str());
427  TObjArray* obj_arr = ss.Tokenize(" ");
428  obj_arr->ls();
429 
430  TString str_va = ((TObjString*) obj_arr->At(0))->GetString();
431  TString str_id = ((TObjString*) obj_arr->At(1))->GetString();
432  TString str_na = ((TObjString*) obj_arr->At(2))->GetString();
433  std::vector<TString> vec;
434  vec.push_back(str_va);
435  vec.push_back(str_id);
436  vec.push_back(str_na);
437  fvec_TCfiles.push_back(vec);
438 
439  TString fullpath;
440  TFile* file = NULL;
441  if (gDataSet->SearchKVFile(str_na.Data(), fullpath, gDataSet->GetName())) {
442  file = TFile::Open(fullpath.Data());
443  if (file) {
444 
445  if (fkverbose) {
446  Info("ReadTransCoefListFile", "... file %s opened, infos follow ...", file->GetName());
447  file->ls();
448  }
449 
450  //find the TEnv and fill the vector with step values
451  TEnv* env = (TEnv*) file->Get("TEnv");
452  assert(env);
453  std::vector<Float_t> line;
454  line.push_back(env->GetValue("ThetaVamos_deg", (Float_t) - 1));
455  line.push_back(env->GetValue("delta_min", (Float_t) - 1));
456  line.push_back(env->GetValue("delta_max", (Float_t) - 1));
457  line.push_back(env->GetValue("delta_step", (Float_t) - 1));
458  line.push_back(env->GetValue("delta_Nstep", (Float_t) - 1));
459  line.push_back(env->GetValue("thetaI_min_rad", (Float_t) - 1));
460  line.push_back(env->GetValue("thetaI_max_rad", (Float_t) - 1));
461  line.push_back(env->GetValue("thetaI_step_rad", (Float_t) - 1));
462  line.push_back(env->GetValue("thetaI_Nstep_rad", (Float_t) - 1));
463  line.push_back(env->GetValue("MonteCarlo_stat", (Float_t) - 1));
464 
465  //IDCode case
466  KVNumberList buff_nl(env->GetValue("IDCode", (const Char_t*) ""));
467  if (buff_nl.Contains(3)) line.push_back(3);
468  else if (buff_nl.Contains(4))line.push_back(4);
469 
470  fvec_TCsteps.push_back(line);
471 
472  //add the TTree in fchain
473  fchain->Add(file->GetName());
474 
475  //close file
476  //file->Close();
477  }
478  }
479  }
480  }//all lines read

493 Bool_t KVVAMOSWeightFinder::CheckTransCoefSteps()
494 {
495  //Internal check of all trans. coef. files steps and infos.
496  //Use the steps informations extracted from all trans. coef. files
497  //and saved in fvec_TCsteps vector.
498  //Check if the steps in delta, the number of values and even the Monte-Carlo statistic
499  //used to generate the trans. coef. *.root files are the same for all used
500  //files saved in the dataset directory.
501  Bool_t ok = kFALSE;
502 
503  //just one line
504  if (fvec_TCsteps.size() == 1) ok = kTRUE;
505 
506  //more than one line
507  else {
508  for (unsigned i = 0; i < fvec_TCsteps.size() - 1; i++) {
509  std::vector<Float_t> line0 = fvec_TCsteps.at(i);
510  std::vector<Float_t> line1 = fvec_TCsteps.at(i + 1);
511 
512  Int_t nmc0 = (Int_t) line0.at(9);
513  Int_t nmc1 = (Int_t) line1.at(9);
514  Int_t nd0 = (Int_t) line0.at(4);
515  Int_t nd1 = (Int_t) line1.at(4);
516  Int_t nt0 = (Int_t) line0.at(8);
517  Int_t nt1 = (Int_t) line1.at(8);
518 
519  //Ndelta, Ntheta and NMonteCarlo is OK
520  if ((nmc0 == nmc1) && (nd0 == nd1) && (nt0 == nt1)) {
521  Float_t dmin0 = line0.at(1);
522  Float_t dmax0 = line0.at(2);
523  Float_t dst0 = line0.at(3);
524  Float_t dmin1 = line1.at(1);
525  Float_t dmax1 = line1.at(2);
526  Float_t dst1 = line1.at(3);
527  Float_t epsd = TMath::Min(dst0, dst1);
528 
529  //Delta steps and ranges should be the same
530  if ((TMath::Abs(dmin0 - dmin1) <= epsd / 2.) && (TMath::Abs(dmax0 - dmax1) <= epsd / 2.) && (TMath::Abs(dst0 - dst1) <= epsd / 2.)) {
531  ok = kTRUE;
532  }
533 
534  else if (fkverbose) {
535  Error("ChecTransCoefSteps", "... error in delta ranges ...\n (dmin0=%f, dmin1=%f) | (dmax0=%f, dmax1=%f) | (dstep0=%f, dstep1=%f)",
536  dmin0, dmin1, dmax0, dmax1, dst0, dst1);
537  }
538  }
539 
540  //Ndelta, Ntheta and NMonteCarlo is not OK
541  else if (fkverbose) {
542  Error("ChecTransCoefSteps", "... error in Nsteps ...\n (nMC0=%d, nMC1=%d) | (nDelta0=%d, nDelta1=%d) | (nTheta0=%d, nTheta1=%d)",
543  nmc0, nmc1, nd0, nd1, nt0, nt1);
544  }
545  }
546  }
547 
548  return ok;
549 }

563 std::pair<Float_t, Float_t> KVVAMOSWeightFinder::GetTransCoef(Float_t VamosAngle_deg, Int_t idc, Float_t delta, Float_t thetaI_rad)
564 {
565  //Find the trans. coef. for a given (ThetaVamos, delta, thetaI) triplet.
566  //First find the path to the file coresponding to the given ThetaVamos
567  //(precision of 0.1 deg) and containing a TTree of its associated trans. coefs.
568  //Then use the associated steps to extract the correct entry number of the
569  //couple (delta, thetaI) in the TTree.
570  //
571  //Return a pair of <trans. coef., trans. coef. variance>
572 
573  //Find (VamosAngle, IDCode) position in fvec_TCfiles
574  Int_t pos = 0;
575  Int_t nn = 0;
576  for (std::vector< std::vector<TString> >::iterator it = fvec_TCfiles.begin(); it != fvec_TCfiles.end(); ++it) {
577  std::vector<TString> par = *it;
578  Float_t angle = (par.at(0)).Atof();
579  Int_t IDC = (par.at(1)).Atoi();
580  TString path = par.at(2);
581 
582  //Check the angle and IDCode
583  if ((TMath::Abs(VamosAngle_deg - angle) <= 0.1) && (idc == IDC)) {
584  pos = nn;
585  if (fkverbose) {
586  Info("GetTransCoef", "... found (VamosAngle (=%f), IDcode (=%d) ) position in vector -> position =%d ...\n",
587  VamosAngle_deg, idc, pos);
588  }
589  }
590 
591  nn++;
592  }
593 
594  //Find, using the associated (VamosAngle, IDCode) TEnv saved in fvec_TCsteps:
595  //- delta position in its range
596  //- thetaI position in its range
597  //- total number of delta
598  //- total number of thetaI
599  //Also check if the VamosAngle and IDCode are OK.
600  std::vector<Float_t> line = fvec_TCsteps.at(pos);
601  if ((TMath::Abs(line.at(0) - VamosAngle_deg) > 0.1) || (idc != (Int_t) line.at(10))) {
602  if (fkverbose) Error("GetTransCoef", "... found [VamosAngle(=%f), IDCode(=%d)] in steps vector is different from given [angle(=%f), IDCode(=%d)] ...",
603  line.at(0), (Int_t) line.at(10), VamosAngle_deg, idc);
604  return std::make_pair(-666., -666.);
605  }
606 
607  Int_t ndelta = GetNValue(delta, line.at(1), line.at(2), line.at(3));
608  Int_t ntheta = GetNValue(thetaI_rad, line.at(5), line.at(6), line.at(7));
609  //Int_t ntot_delta = line.at(4)+1;
610  Int_t ntot_delta = line.at(4);
611  Int_t ntot_theta = line.at(8) + 1;
612 
613  if (fkverbose) {
614  Info("GetTransCoef", "... found delta/theta positions in their associated ranges -> ndelta=%d, ntheta=%d, ntot_delta=%d, ntot_theta=%d ...",
615  ndelta, ntheta, ntot_delta, ntot_theta);
616  }
617 
618  //Use the found values to exctract the entry in fchain corresponding
619  //to the wanted trans. coef.
620  if (ndelta >= 0 && ntheta >= 0 && ntot_delta > 0 && ntot_theta > 0) {
621  Long64_t pos_tree = (ndelta * ntot_theta) + ntheta; //entry in the tree
622  Long64_t pos_chain = pos * (ntot_delta * ntot_theta) + pos_tree; //entry in the chain
623 
624  if (fkverbose) Info("GetTransCoef", "... found entry in tree (=%lli) and entry in chain (=%lli / %lli) ...", pos_tree, pos_chain, fchain->GetEntries());
625 
626  Float_t tc, dtc, dd, tt;
627  fchain->SetBranchAddress("TransCoef", &tc);
628  fchain->SetBranchAddress("DeltaTransCoef", &dtc); //variance of trans. coef. from MC
629  fchain->SetBranchAddress("Zdelta", &dd);
630  fchain->SetBranchAddress("ZthetaI", &tt);
631  fchain->GetEntry(pos_chain, 1);
632 
633  //debug
634  if (fkverbose) {
635  Info("GetTransCoef", "... input: (VamosAngle=%f, IDCode=%d, delta=%f, thetaI=%f) | closest: (delta=%f, thetaI=%f) | chain_entry: (delta=%f, thetaI=%f) ...",
636  VamosAngle_deg, idc, delta, thetaI_rad, GetClosestValue(delta, line.at(3)), GetClosestValue(thetaI_rad, line.at(7)), dd, tt);
637  }
638 
639  return std::make_pair(tc, dtc);
640  }
641 
642  else return std::make_pair(-666., -666.);
643 }

653 Int_t KVVAMOSWeightFinder::GetNValue(Float_t val_exp, Float_t val_min, Float_t val_max, Float_t val_step)
654 {
655  //Return position of the experimental observable 'val_exp'
656  //in its associated range '[val_min, val_max]'
657  //and with the step 'val_step'.
658 
659  //Get the closest value considering the step
660  Float_t cl_val = GetClosestValue(val_exp, val_step);
661 
662  //If the closest delta is out of the given range
663  if ((cl_val > val_max + val_step / 2.) || (cl_val < val_min - val_step / 2.)) {
664  if (fkverbose) {
665  if (cl_val > val_max + val_step / 2.) Error("GetNValue", "... closest value (=%f) above val_max+step/2. (=%f) ...", cl_val, val_max + val_step / 2.);
666  if (cl_val < val_min - val_step / 2.) Error("GetNValue", "... closest value (=%f) below val_min-step/2. (=%f) ...", cl_val, val_min - val_step / 2.);
667  }
668  return -1;
669  }
670 
671  //Closest value is in given range
672  else {
673  Int_t val_num = TMath::Nint((cl_val - val_min) / val_step);
674  return val_num;
675  }

791 Float_t KVVAMOSWeightFinder::GetWeight(Float_t brho, Float_t thetaI, Int_t idcode)
792 {
793  //Compute the weight for a given experimental VAMOS event,
794  //using the identified VAMOS nucleus informations, which means:
795  //(bro_exp, thetaI_rad) where 'brho' is the reconstructed magnetic rigidity of
796  //the nucleus (in T.m), and 'thetaI' is the experimental theta angle (in rad)
797  //in lab. ref. frame (in INDRA convention) of the nucleus (and also the current
798  //run number this events belongs to, set as an argument of the class constructor).
799  //
800  //The weight will be computed from:
801  //- the chosen run list (see SetRunList() method) and the associated experimental
802  // (BrhoRef, AngleVamos, Scalers_INDRA) values from this run list.
803  //- the transmission coefficient obtained from ZGOUBI simulations (normalised over
804  // 2.*TMath::Pi() in phi angle in lab. ref. frame and INDRA convention, depending on
805  // the (brho, thetaI) of the nucleus.
806  //- the experimental Dead Time (DT) associated to fRunNumber, the run number the VAMOS
807  // events belongs to.
808  //
809  //The computed weight is similar to a weighted average on VAMOS events only defined by their
810  //(Brho, thetaI) values. In this way we are able to use ALL the statistics without
811  //applying cuts on overlaping experimental Brho.
812  //
813  //The returned total weight 'W' will be:
814  //
815  // Begin_Latex
816  // #W(B #rho , #theta _{I}) = #frac{ #sum_{i}^{RunList}Scaler^{i} }{ #sum_{i}^{RunList}Scaler^{i} #cdot T(#delta ^{i}, #theta_{I} , #theta ^{i}_{V})}
817  // End_Latex
818  //
819  // and:
820  // Begin_Latex
821  // DT_{corr}(run_number) = #frac{1}{1-DT(run_number)}
822  // End_Latex
823  //
824  // and:
825  // run_number = current run number which the identified VAMOS nucleus belongs to
826  // delta = brho/brho_ref_i
827  // brho = experimental brho in T.m
828  // brho_ref_i = reference brho of VAMOS for the run i
829  // thetaI = experimental theta angle in rad and in lab. ref. frame in INDRA convention
830  // Scaler_i = value of INDRA scalers for the run i
831  // theta_V = rotation angle of VAMOS in theta around beam trajectory
832 
833  if (fkverbose) Info("GetWeight", "... starting weight computation for VAMOS event with (brho=%f [T.m], thetaI=%f [rad], IDCode=%d, run=%d) ...", brho, thetaI, idcode, fRunNumber);
834 
835  Float_t num = 0.;
836  Float_t denum = 0.;
837  Float_t dt_run = GetDeadTime(fRunNumber); //default '-666.' value will be returned if not found
838 
839  if (dt_run >= 0) {//DT found from the event run_number
840  //Compute W(Brho, ThetaI) from runlist
841  fRunList.Begin();
842  while (!fRunList.End()) {
843  Int_t next_run = fRunList.Next();
844  Float_t brho_ref = GetBrhoRef(next_run);
845  Float_t scaler = GetScalerINDRA(next_run);
846  Float_t thetav = GetThetaVamos(next_run);
847  Float_t density = GetTargetDensity(next_run);
848  std::pair<Float_t, Float_t> pair_tc = GetTransCoef(thetav, idcode, brho / brho_ref, thetaI);
849  Float_t tc = pair_tc.first; // trans. coef.
850  Float_t dtc = pair_tc.second; //variance of trans. coef.
851 
852  if (scaler > 0) {
853  //if trans. coef. doesn't exist for the given (ThetaVamos, delta, thetaI),
854  //we still consider the scaler in the numerator (i.e tc=0.)
855  num += scaler;
856 
857  //if (brho_ref, scaler) is OK for the given run and trans. coef. exists for the given (ThetaVamos, delta, thetaI, IDCode)
858  if ((brho_ref > 0) && (tc > 0.) && (dtc > 0.) && (density > 0.)) {
859  //if limits on trans.coef. were set by the user
860  if (AreTransCoefLimitsSet()) {
861  if ((tc > ftc_min) && (tc < ftc_max)) denum += scaler * tc * dt_run;
862  }
863  //no limits set
864  else denum += scaler * tc * dt_run;
865  }
866  }
867 
868  if (fkverbose) {
869  Info("GetWeight", "... (next_run=%d, brho=%lf, thetaI=%lf) -> (brho_ref=%lf, thetav=%lf, scaler=%lf, tc=%lf, var_tc=%lf) ...",
870  next_run, brho, thetaI, brho_ref, thetav, scaler, tc, dtc);
871  Info("GetWeight", "... numerator=%lf, denumerator=%lf\n", num, denum);
872  }
873  }
874 
875  //return Weight
876  Float_t weight = (denum > 0. ? num / denum : -666.);
877  return weight;
878  }
879 
880  //DT not found for the current run, weight can't be computed
881  //default value -666. will be returned...
882  else {
883  if (fkverbose) {
884  Info("GetWeight", "... !!! DT(current_run=%d)=%f <0. !!! can't compute the weight of the event ...", fRunNumber, dt_run);
885  }
886 
887  return -666.;
888  }
889 }

113 void KVDBRun::WriteRunListLine(ostream& outstr, Char_t) const
114 {
115  //Write informations on run in format used for runlists, i.e. a line of fields separated by the
116  //separator character '|' (the 'delim' argument is obsolete and is not used)
117 
118  TString _delim = " | ";
119  KVString s;
120  //write run number
121  outstr << GetNumber() << _delim.Data();
122  //write all scalers (integer values)
123  for (int i = 0; i < fParameters.GetNpar(); i++) {
124  if (fParameters.GetParameter(i)->IsInt()) {
125  s.Form("%s=%d", fParameters.GetParameter(i)->GetName(), fParameters.GetParameter(i)->GetInt());
126  outstr << s.Data() << _delim.Data();
127  }
128  }
129  //write all floating point values
130  for (int i = 0; i < fParameters.GetNpar(); i++) {
131  if (fParameters.GetParameter(i)->IsDouble()) {
132  s.Form("%s=%f", fParameters.GetParameter(i)->GetName(), fParameters.GetParameter(i)->GetDouble());
133  outstr << s.Data() << _delim.Data();
134  }
135  }
136  //write all string values
137  for (int i = 0; i < fParameters.GetNpar(); i++) {
138  if (fParameters.GetParameter(i)->IsString()) {
139  // as we write all strings as 'parname=value', there is a problem if the string 'value'
140  // itself contains the '=' symbol !!
141  // therefore we replace any '=' in the string by '\equal' before writing in the file.
142  // when reading back (see ReadRunListLine), we replace '\equal' by '='
143  TString tmp(fParameters.GetParameter(i)->GetString());
144  tmp.ReplaceAll("=", "\\equal");
145  s.Form("%s=%s", fParameters.GetParameter(i)->GetName(), tmp.Data());
146  outstr << s.Data() << _delim.Data();
147  }
148  }
149  outstr << endl;
150 }

160 void KVDBRun::ReadRunListLine(const KVString& line)
161 {
162  //Set run characteristics by reading informations in 'line' (assumed to have been written
163  //by WriteRunListLine).
164 
165  //Break line into fields using delimiter '|'
166  TObjArray* fields = line.Tokenize('|');
167  if (fields->GetEntries() < 1) {
168  //not a valid line for run
169  delete fields;
170  return;
171  }
172 
173  //first field is run number
174  KVString kvs = ((TObjString*)fields->At(0))->String().Remove(TString::kBoth, ' ');
175  if (kvs.IsDigit()) {
176  SetNumber(kvs.Atoi());
177  } else {
178  //not a valid line for run
179  delete fields;
180  return;
181  }
182 
183 // cout << GetName() << endl;
184 
185  //loop over other fields
186  for (int i = 1; i < fields->GetEntries(); i++) {
187 
188  //each field is of the form "parameter=value"
189  KVString kvs = ((TObjString*)fields->At(i))->String().Remove(TString::kBoth, ' ');
190  TObjArray* toks = kvs.Tokenize('=');
191  if (toks->GetEntries() == 2) {
192  KVString parameter = ((TObjString*)toks->At(0))->String().Remove(TString::kBoth, ' ');
193  KVString value = ((TObjString*)toks->At(1))->String().Remove(TString::kBoth, ' ');
194  //set parameter based on value
195  if (value.IsDigit()) {
196  //only true for non-floating point i.e. scaler values
197  SetScaler(parameter.Data(), value.Atoi());
198 // cout << " -- SCA " << parameter.Data() << " = " << GetScaler(parameter.Data()) << endl;
199  } else if (value.IsFloat()) {
200  Set(parameter.Data(), value.Atof());
201  // cout << " -- FLO " << parameter.Data() << " = " << Get(parameter.Data()) << endl;
202  } else { // string value
203  // as we write all strings as 'parname=value', there is a problem if the string 'value'
204  // itself contains the '=' symbol !!
205  // therefore we replace any '=' in the string by '\equal' before writing in the file
206  // (see WriteRunListLine).
207  // when reading back, we replace '\equal' by '='
208  value.ReplaceAll("\\equal", "=");
209  Set(parameter.Data(), value.Data());
210  // cout << " -- STR " << parameter.Data() << " = " << GetString(parameter.Data()) << endl;
211  }
212  }
213  delete toks;
214  }
215 
216  delete fields;
217 }

270 void KVDBRun::SetSystem(KVDBSystem* system)
271 {
272  //Set system for run. Any previous system is unassociated (run will be removed from system's list)
273  if (!GetKey("Systems")) {
274  KVDBKey* key = AddKey("Systems", "Physical system used");
275  key->SetUniqueStatus(kTRUE);
276  key->SetSingleStatus(kTRUE);
277  } else {
278  UnsetSystem();

16 <h2>KVRecombination</h2>
17 <h4>Silicon PHD described by electron-hole recombination model</h4>
18 
19 <p>
20 The static TF1 "fParlog" calculates the PHD of a particle of incident energy
21 E, charge Z and Mass A calculated according to Parlog's formula (see Phys. Res. A 428 (1999) 379):
22 </p>
23 
24 <p>
25  PHD(Ed) = 1/2 * Ed *( 1-1/X * ln|1+X| + X * ln|1+1/X| )
26  with X = a*A*Z**2/Ed
27  Ed = energy lost by particle in detector (=E if particle stops)
28 </p>
29 <p>Only one parameter (a) can be set using method<br>
30 SetParameters(a) or SetParameter(0,a).
31 The charge and the mass numbers have to be set first the calculation using<br>
32 method SetZandA( Z, A).
33 </p>

78 Double_t KVRecombination::PHDParlog(Double_t* x, Double_t* par)
79 {
80  //Returns Parlog PHD for given E, Z and A:
81  //
82  // PHD(E) = 1/2 * Ed *( 1-1/X * ln|1+X| + X * ln|1+1/X| )
83  // with X = a*A*Z**2/Ed
84  // Ed = energy lost by particle in detector (=E if particle stops)
85  //
86  // x[0] = E (MeV)
87  // par[0] = Z
88  // par[1] = A
89 
90  Int_t Z = par[0];
91  Int_t A = par[1];
92  Double_t a = GetParameter(0);
93  Double_t E = x[0] > 0 ? x[0] : 0.000001;
94  Double_t X = a * A * Z * Z / E;
95 
96  return (1. - 1. / X * TMath::Log(TMath::Abs(1. + X))
97  + X * TMath::Log(TMath::Abs(1. + 1. / X))
98  ) * E / 2;
99 }

158 Double_t KVRecombination::Compute(Double_t energy) const
159 {
160  //Calculate the pulse height defect (in MeV) for a given energy loss in the detector,
161  //a given Z and a given A (Z and A of the particle should be set first using method SetZandA)
162  //The Parlog formula is used:
163  //
164  // PHD(E) = 1/2 * Ed *( 1-1/X * ln|1+X| + X * ln|1+1/X| )
165  // with X = a*A*Z**2/Ed
166  // Ed = energy lost by particle in detector (=E if particle stops)
167 
168  return const_cast<KVRecombination*>(this)->GetParlogPHDFunction(GetZ(), GetA())->Eval(energy);
169 }

194 TF1* KVRecombination::GetELossFunction(Int_t Z, Int_t A, Bool_t Wrong)
195 {
196  // Return pointer to TF1 giving energy loss in active layer of detector minus
197  // the pulse height defect for a given nucleus (Z,A).
198  //
199  // If Wrong=kTRUE (default:kFALSE) this will be calculated incorrectly
200  // (if the particle does not stop in the detector) by using the Parlog formula
201  // with the incident energy of the particle instead of the calculated energy
202  // loss of the particle.
203 
204  wrong = Wrong;
205 
206  if (!fDeltaEphd) {
207  fDeltaEphd = new TF1(Form("KVRecombination:%s:ELossActive", GetDetector()->GetName()),
208  this, &KVRecombination::ELossActive, 0., 1.e+04, 2, "KVRecombination", "ELossActive");
209  fDeltaEphd->SetNpx(gEnv->GetValue("KVPulseHeightDefect.EnergyLoss.Npx", 20));
210  }
211  fDeltaEphd->SetParameters((Double_t)Z, (Double_t)A);
212  fDeltaEphd->SetRange(0., GetDetector()->GetSmallestEmaxValid(Z, A));
213  fDeltaEphd->SetTitle(Form("PHD dependent energy loss [MeV] in detector %s for Z=%d A=%d", GetDetector()->GetName(), Z, A));
214  GetParlogPHDFunction(Z, A);
215  return fDeltaEphd;
216 }

226 Double_t KVRecombination::ELossActive(Double_t* x, Double_t* par)
227 {
228  // Calculate energy lost in active layer of detector minus the Parlog PHD
229  // x[0] = incident energy
230  // par[0] = Z
231  // par[1] = A
232 
233  Double_t e = x[0];
234  TIter next(GetDetector()->GetListOfAbsorbers());
235  KVMaterial* mat;
236  while ((mat = (KVMaterial*)next()) != GetDetector()->GetActiveLayer()) {
237  // calculate energy losses in absorbers before active layer
238  e = mat->GetERes(par[0], par[1], e); //residual energy after layer
239  if (e <= 0.)
240  return 0.; // return 0 if particle stops in layers before active layer
241  }
242  // calculate energy loss in active layer
243  Double_t dE = mat->GetDeltaE(par[0], par[1], e);
244  // calculate Parlog PHD corresponding to energy lost in active layer
245  Double_t phd;
246  if (wrong) phd = PHDParlog(&e, par); //incorrect calculation using incident energy
247  else phd = PHDParlog(&dE, par);
248 
249  return dE - phd;
250 }

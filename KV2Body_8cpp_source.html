28 //KV2Body
29 //
30 //This class calculates various useful characteristics of binary nuclear reactions.
31 //It is based on a Fortran programme by Jean-Luc Charvet (cibi.for).
32 //It allows to calculate:
33 // - centre of mass recoil velocity and available energy
34 // - laboratory grazing angle of projectile and target
35 // - maximum laboratory scattering angles
36 // - Q-values
37 // - CM and laboratory Rutherford cross-sections for elastic scattering
38 // - CM and laboratory angles of ejectiles after elastic or inelastic scattering
39 //The values are calculated using relativistic kinematics.
40 //
41 //
42 //1. Defining the entrance channel.
43 //
44 //Either use the constructor with arguments to specify projectile and target:
45 // KV2Body kin( proj, targ );//'proj' and 'targ' are pointers to KVNucleus objects
46 //or declare them afterwards:
47 // KV2Body kin;
48 // kin.SetProjectile( proj ); kin.SetTarget( targ );
49 //One can also specify the projectile & target using their Z and A:
50 // kin.SetProjectile( z1, a1 );
51 // kin.SetTarget( z2, a2 );
52 //(In each case a KVNucleus is created which will be deleted with the KV2Body object).
53 //If you need to further define the properties of these nuclei, you can get a pointer to
54 //them using GetNucleus(int i):
55 // kin.GetNucleus(1)->SetMomentum(...);//set momentum of projectile
56 //
57 //2. Defining the exit channel
58 //
59 //If no exit channel is defined, it is assumed to be identical to the entrance channel,
60 //i.e. elastic scattering.
61 //You can define a different exit channel, either using the constructor:
62 // KV2Body kin( proj, targ, QP );// QP is a KVNucleus
63 // KV2Body kin( proj, targ, QP, Ediss );// Ediss = energy dissipated in inelastic reaction
64 //or using the methods:
65 // kin.SetOutgoing( QP );
66 // kin.SetEDiss( Ediss ); OR kin.SetExcitEnergy( Ediss );
67 //
68 //3. Calculating and obtaining reaction information
69 //
70 //Once the entrance channel (and, if necessary, the exit channel) is(are) defined,
71 //the kinematical calculation is carried out by:
72 // kin.CalculateKinematics();
73 //For a general print-out of the reaction characteristics:
74 // kin.Print();
75 // kin.Print("lab");//laboratory scattering angles and energies
76 // kin.Print("ruth");//Rutherford scattering angles, energies and cross-sections
77 //
78 //4. Extracting information on Quasi-Projectile and/or target
79 //
80 //Several methods are defined and calculate energy/velocity/angle in CM or lab frame
81 //for projectile (3) and/or target (4)
82 //The two arguments are : the angle and the angular range of which nucleus you are interested in
83 //
84 //For example : the Double_t GetThetaCMTarget(Double_t ThetaLab,Int_t OfNucleus=3)
85 //give the polar angle in the CM of reaction of the target
86 // depending on the polar angle in the lab of the projectile (OfNucleus=3) or of the target (OfNucleus=4)
87 //
88 //

211 KV2Body::KV2Body(const Char_t* systemname) : fNuclei(4, 1)
212 {
213  //Set up calculation defining entrance channel following
214  //this prescription :
215  //[Projectile_Symbol]+[Target_Symbol]@[Incident_Energy]MeV/A
216  // Example :
217  // 129Xe+119Sn@50.0MeV/A
218  // U+U@5MeV/A
219  // Ta+Zn
220  init();
221  KVString sener = "";
222  Double_t ee = 0;
223  KVString syst(systemname);
224  syst.ReplaceAll(" ", "");
225  KVString scouple(syst);
226 
227  if (syst.Contains("@")) {
228  syst.Begin("@");
229  scouple = syst.Next();
230  sener = syst.Next();
231  }
232  if (sener != "") {
233  sener.ReplaceAll("MeV/A", "");
234  ee = sener.Atof();
235  }
236  scouple.Begin("+");
237  fNuclei[1] = new KVNucleus(scouple.Next(), ee);
238  fNuclei[2] = new KVNucleus(scouple.Next());
239 }

251 KV2Body::KV2Body(KVNucleus* proj, KVNucleus* cib, KVNucleus* proj_out, Double_t Ediss): fNuclei(4, 1)
252 {
253  //Set up calculation defining entrance channel (projectile & target nuclei or
254  //single decaying nucleus), exit channel (if necessary, the remaining nucleus of the
255  //exit channel is deduced by conservation of mass, charge, momentum) and energy dissipated in reaction.
256  //
257  //By default the dissipated energy is zero (elastic reaction).
258  //If only nucleus 'proj' is defined and Ediss=0, we use the excitation energy of 'proj' as -Ediss
259 
260  init();
261  fNuclei[1] = proj;
262  fNuclei[2] = cib;
263  fEDiss = Ediss;
264  if (!cib) {
265  // break-up kinematics
266  if (Ediss == 0.0) {
267  fEDiss = -proj->GetExcitEnergy();

272  SetOutgoing(proj_out);
273  }

401 void KV2Body::Set4thNucleus()
402 {
403  // Private method, used to deduce 4th nucleus (target-like) from projectile, target
404  // and outgoing projectile using conservation of mass, momentum and energy.
405  // This nucleus will be deleted with the object.
406  //
407  // if the exit channel is a the compund nucleus, there is no 4th nucleus defined
408  // and the excitation energy of the fusion reaction is set in the fEdiss variable

411

414  if (GetNucleus(4))

422  fDeleteN4 = kFALSE;
423  }

577 Double_t KV2Body::GetLabGrazingAngle(Int_t i) const
578 {
579  //Calculate laboratory grazing angle.
580  // i = 1 (default) : projectile
581  // i = 2 : target
582 
583  if (i < 1 || i > 2) {
584  Warning("GetLabGrazingAngle(Int_t i)",
585  "i should be 1 (proj) or 2 (targ)");
586  return 0.0;
587  }
588  if (!GetNucleus(2)) {
589  Warning("GetLabGrazingAngle(Int_t i)",
590  "No target defined for reaction");
591  return 0.0;
592  }
593  Double_t RP = 1.28 * TMath::Power(GetNucleus(1)->GetA(), 1. / 3.)
594  - 0.76 + 0.8 * TMath::Power(GetNucleus(1)->GetA(), -1. / 3.);
595  Double_t RT = 1.28 * TMath::Power(GetNucleus(2)->GetA(), 1. / 3.)
596  - 0.76 + 0.8 * TMath::Power(GetNucleus(2)->GetA(), -1. / 3.);
597  Double_t CP = RP * (1. - TMath::Power(1. / RP, 2.));
598  Double_t CT = RT * (1. - TMath::Power(1. / RT, 2.));
599  Double_t RINT = CP + CT + 4.49 - (CT + CP) / 6.35;
600  Double_t BAR =
601  1.44 * GetNucleus(1)->GetZ() * GetNucleus(2)->GetZ() / RINT;
602  if (GetCMEnergy() < BAR) {
603  //below Coulomb barrier
604  switch (i) {
605  case 1:
606  return 180.;
607  break;
608  case 2:
609  return 90.;
610  break;
611  }
612  }
613  Double_t X = 2. * RINT * GetCMEnergy();
614  Double_t Y =
615  X / (GetNucleus(1)->GetZ() * GetNucleus(2)->GetZ() * 1.44) - 1.;
616  Double_t U = 1. / Y;
617  Double_t GR = 2. * TMath::ASin(U);
618  Double_t GPART = TMath::Pi() - GR;
619  Double_t ARAZ =
620  TMath::Sin(GR) / (TMath::Cos(GR) +
621  GetNucleus(1)->GetA() / (1.0 *
622  GetNucleus(2)->GetA()));
623  Double_t GRAZ = TMath::ATan(ARAZ);
624  if (GRAZ <= 0.)
625  GRAZ += TMath::Pi();
626 
627  switch (i) {
628  case 1: //projectile
629  return (GRAZ * TMath::RadToDeg());
630  break;
631  case 2: //target
632  return (GPART * TMath::RadToDeg() / 2.);
633  break;
634  }
635  return -99.;
636 }

647 void KV2Body::CalculateKinematics()
648 {
649  // Called to make kinematics calculation for all nuclei, use once properties of
650  // entrance-channel (and, if necessary, exit-channel) nuclei have been defined.
651  // If no exit-channel nuclei are defined, we assume elastic scattering (i.e. identical
652  // outgoing nuclei)
653 
654  KVNucleus* Nuc1 = GetNucleus(1);
655  KVNucleus* Nuc2 = GetNucleus(2);
656 
657  // make sure E* of nuclei is zero
658  // this is because the E*/Ediss is handled separately here
659  Nuc1->SetExcitEnergy(0.);
660  if (Nuc2) Nuc2->SetExcitEnergy(0.);
661 
662  // call SetOutgoing if not already done
663  if (!fSetOutgoing) SetOutgoing(Nuc1);
664  //fSetOutgoing = kFALSE;
665 
666  // set everything to zero
667  WLT = WCT = BCM = 0.;
668  for (int i = 0; i < 5; i++) {
669  WC[i] = 0.;
670  VC[i] = 0.;
671  EC[i] = 0.;
672  K[i] = 0.;
673  TETAMAX[i] = 0.;
674  TETAMIN[i] = 0.;
675  }
676  VCM.SetXYZ(0., 0., 0.);
677 
678  //total energy (T + m) of entrance channel
679  WLT = Nuc1->E();
680  //velocity of CM
681  VCM = Nuc1->GetMomentum();
682  if (Nuc2) {
683  WLT += Nuc2->E();
684  VCM += Nuc2->GetMomentum();
685  }
686 
687  //beta of CM
688  BCM = VCM.Mag() / WLT;
689 
690  VCM *= (1. / WLT) * KVParticle::C();
691  Nuc1->SetFrame("CM", KVFrameTransform(VCM, kFALSE));
692  if (Nuc2)
693  Nuc2->SetFrame("CM", KVFrameTransform(VCM, kFALSE));
694 
695  //total energy in CM
696  WCT = (GetCMGamma() > 0.0 ? WLT / GetCMGamma() : 0.);
697  if (WCT == 0.0)
698  return;
699 
700  //total energy proj in CM
701  WC[1] = Nuc1->GetFrame("CM")->E();
702  //kinetic energy proj in CM
703  EC[1] = Nuc1->GetFrame("CM")->GetKE();
704  VC[1] = Nuc1->GetFrame("CM")->GetVelocity().Mag();
705  K[1] = VCM.Mag() / VC[1];
706  if (Nuc2) {
707  WC[2] = Nuc2->GetFrame("CM")->E();
708  EC[2] = Nuc2->GetFrame("CM")->GetKE();
709  VC[2] = Nuc2->GetFrame("CM")->GetVelocity().Mag();
710  K[2] = VCM.Mag() / VC[2];
711  }
712 
713  Double_t AM3 = GetNucleus(3)->GetMass();
714  Double_t AM4 = (GetNucleus(4) ? GetNucleus(4)->GetMass() : 0.0);
715  //total cm energy of nucleus 3 (quasiproj)
716  WC[3] = (WCT - GetEDiss()) / 2. + (AM3 - AM4) * (AM3 + AM4)
717  / (2. * (WCT - GetEDiss()));
718  VC[3] = GetVelocity(AM3, WC[3]);
719  //cm kinetic energy
720  EC[3] = WC[3] - AM3;
721  if (VC[3] > 0.) K[3] = VCM.Mag() / VC[3];
722 
723  Double_t T3MAX = 0.;
724  if (TMath::AreEqualAbs(K[3], 1., 1.e-16))
725  T3MAX = TMath::PiOver2();
726  else if (K[3] < 1.)
727  T3MAX = TMath::Pi();
728  else
729  T3MAX = TMath::ATan((1. / TMath::Sqrt(K[3] * K[3] - 1.)) / GetCMGamma());
730  TETAMAX[3] = T3MAX * TMath::RadToDeg();
731 
732  if (!GetNucleus(4))
733  return; //only valid for binary channels
734  //total cm energy of nucleus 4 (quasitarg)
735  WC[4] = WCT - GetEDiss() - WC[3];
736  VC[4] = GetVelocity(AM4, WC[4]);
737  //cm kinetic energy
738  EC[4] = WC[4] - AM4;
739  if (VC[4] > 0.) K[4] = VCM.Mag() / VC[4];
740 
741  Double_t T4MAX = 0.;
742  if (TMath::AreEqualAbs(K[4], 1., 1.e-16))
743  T4MAX = TMath::PiOver2();
744  else if (K[4] < 1.)
745  T4MAX = TMath::Pi();
746  else
747  T4MAX = TMath::ATan((1. / TMath::Sqrt(K[4] * K[4] - 1.)) / GetCMGamma());
748  if (TMath::Abs(GetQReaction()) < 1.E-08 && K[4] < 1.)
749  T4MAX = TMath::PiOver2();
750  TETAMAX[4] = T4MAX * TMath::RadToDeg();
751 }

1563 Double_t KV2Body::EqbmChargeState(Double_t* t, Double_t*)
1564 {
1565  // Calculate the mean charge state of the projectile after passage through the target,
1566  // assuming that the equilibrium charge state distribution is achieved*.
1567  // t[0] = energy of projectile after the target (in MeV/nucleon)
1568  //
1569  // We use the empirical parameterization of Leon et al., At. Dat. and Nucl. Dat. Tab. 69, 217 (1998)
1570  // developed for heavy ions in the GANIL energy range (it represents a fit to data measured using
1571  // GANIL beams).
1572  //
1573  // *N.B. Concerning equilibrium charge state distributions, it is not easy to know whether, for a given
1574  // combination of projectile, projectile energy, target, and target thickness, the equilibrium
1575  // distribution is reached or not. Here are some comments from the paper cited above which
1576  // may give some guidelines:
1577  //
1578  // "The energies available at the GANIL accelerator range from 24 to 95 MeV/u. Within this energy range,
1579  // the equilibrium charge state is reached only for fairly thick targets (~1 mg/cm2 for C foils)."
1580  //
1581  // "Mean Charge State as a Function of the Target Thickness
1582  // A typical example of the variation of the mean charge as a function of the foil thickness is shown ... It is seen
1583  // that the mean charge initially increases due to the ionization process. Then, the equilibrium state is reached at
1584  // a certain thickness, the so-called equilibrium thickness, due to the equilibration of electron loss and
1585  // capture processes. Finally, for foils thicker than the equilibrium thickness, the mean charge decreases due to the
1586  // slowing down of the ions in matter leading to higher capture cross sections."
1587  //
1588  // It should be noted that, according to the data published in this and other papers, the equilibrium thickness
1589  // decreases with increasing atomic number of the target, and increases with increasing energy of the projectile.
1590 
1591  KVNucleus* proj = (KVNucleus*)fNuclei[1];
1592  Double_t Zp = proj->GetZ();
1593  proj->SetEnergy(t[0]*proj->GetA());
1594  Double_t beta = proj->Beta();
1595  Double_t vp = beta * KVParticle::C();
1596  Double_t Zt = ((KVNucleus*)fNuclei[2])->GetZ();
1597 
1598  Double_t q = Zp * (1. - TMath::Exp(-83.275 * beta / pow(Zp, 0.477)));
1599 
1600  // correction for target Z
1601  Double_t g = 0.929 + 0.269 * TMath::Exp(-0.16 * Zt) + (0.022 - 0.249 * TMath::Exp(-0.322 * Zt)) * vp / pow(Zp, 0.477);
1602  q *= g;
1603 
1604  if (Zp > 54) {
1605  // f(Zp) - correction for projectiles with Z>54
1606  Double_t f = 1. - TMath::Exp(-12.905 + 0.2124 * Zp - 0.00122 * pow(Zp, 2));
1607  q *= f;
1608  }
1609 
1610  return q;
1611 }

22 // Classe héritée de KVVarGlob avec d'ajouter à ces membres, deux KVNameValueList permettant
23 // gérer les ingrédients/variables et les paramètres de manière complètement progressive.
24 // Ces deux listes sont accessibles via GetList(Option_t* option = "ing" ou "par")
25 //
26 // PRINCIPE
27 // KVCaloBase fait la somme des Z (Zsum), A (Asum), Ek (Eksum) et Q (Qsum) des noyaux considérés (méthode Fill(KVNucleus* ))
28 // Ces ingrédients permettent ensuite de calculer l'énergie d'excitation en utilisant
29 // le bilan énergétique suivant:
30 // Exci + Qini = Eksum + Qsum -> Exci = Eksum + Qsum - Qini
31 // A noter : Si l'utilisateur définit un repère via KVVarGlob::SetFrame(const Char_t* ) les énergies cinétiques
32 // des noyaux sont prises dans ce référentiel ( KVNucleus::GetFrame("nom_du_referentiel)"->GetKE() )
33 //------------------
34 // Exemple d utilisation :
35 //------------------
36 //
37 //KVNucleus alpha(2,4,10); //definition des noyaux
38 //KVNucleus triton(1,3);
39 //KVNucleus azote(7,16,40);
40 //
41 //KVCaloBase ca;
42 //
43 //ca.Fill(&alpha); //Remplissage de la variable
44 //ca.Fill(&triton);
45 //ca.Fill(&azote);
46 //
47 //ca.Calculate(); //Calcul
48 //ca.Print("ing"); //Print
49 //Ingredients, 7 stored:
50 //0 | Zsum | 10.00000 Sum of charges
51 //1 | Asum | 23.00000 Sum of masses
52 //2 | Eksum | 50.0000 Sum of kinetic energiex (MeV)
53 //3 | Qsum | 23.05840 Sum of mass excess (MeV)
54 //4 | Msum | 3.000000 Multiplicity
55 //5 | Qini | -5.15400 Mass Excess of the initial state (reconstructed sourceeeeeee)
56 //6 | Exci | 78.21240 Excitation energy (MeV)
57 //
58 //ca.GetValue(0)
59 //10.0000
60 //ca.GetValue("Exci")
61 //78.21240
62 //
63 //------------------------ //Fin de l'exemple
64 //
65 // ACCES AUX VARIABLES
66 //
67 //L'accès aux variables se fait via les deux méthodes :
68 // - GetValue(Int_t) choix par index
69 // - GetValue(const Char_t* ) choix par nom
70 // - GetValuePtr(), renvoie un tableau de Double_t*, de la dimension correspondant au nombre de valeurs (méthode GetNumberOfValues())

671 {
672  //Réalisation de la calorimétrie
673  //Calcul de l'énergie d'excitation
674  //appel de SumUp()

710 Bool_t KVCaloBase::RootSquare(Double_t aa, Double_t bb, Double_t cc)
711 {
712  // protected method
713  //
714  // calcul les racines du polynome d'ordre 2 : aa*x*x + bb*xx + cc = 0
715  // les racines sont accessibles par les variables kracine_min et kracine_max
716  //
717  // kroot_status>=0 -> tout c'est bien passe la fonction retourne kTRUE
718  // =0 2 racines reelles distinctes
719  // =1 2 racines reelles egales (aa==0)
720  //
721  // kroot_status<0 les deux racines sont mises a zero la fonction retourne kFALSE
722  // =-1 2 racines imaginaires (Delta<0)
723  // =-2 aa=bb=0
724  // le calcul n'est alors pas poursuivi, la méthode Calculate() retournera kFALSE
725  // la cause peut être discriminée en appelant la méthode GetValue("RootStatus")
726  //
727  kracine_max = 0, kracine_min = 0;
728  Double_t x1, x2;
729  kroot_status = 0;
730  if (aa != 0) {
731  Double_t Delta = TMath::Power(bb, 2.) - 4.*aa * cc;
732  if (Delta < 0) {
733  //Warning("RootSquare","Delta<0 - Solutions imaginaires");
734  kroot_status = -1;
735  SetIngValue("RootStatus", kroot_status);
736  } else {
737  Double_t racDelta = TMath::Sqrt(Delta);
738  x1 = (-1.*bb + racDelta) / (2.*aa);
739  x2 = (-1.*(bb + racDelta)) / (2.*aa);
740  kroot_status = 0;
741  if (x1 > x2) {
742  kracine_max = x1;
743  kracine_min = x2;
744  } else {
745  kracine_max = x2;
746  kracine_min = x1;
747  }
748  }
749  } else {
750  if (bb != 0) {
751  kroot_status = 1;
752  kracine_max = kracine_min = -1.*cc / bb;
753  } else {
754  kroot_status = -2;
755  kracine_max = kracine_min = 0;
756  SetIngValue("RootStatus", kroot_status);
757  }
758  }
759  if (kroot_status < 0) {
760  SetIngValue("RootStatus", kroot_status);
761  return kFALSE;
762  } else {
763  return kTRUE;
764  }
765 
766 }

119 void KVASGroup::SetDimensions(KVPosition* p1, KVPosition* p2)
120 {
121  //Adjust angular dimensions of group according to theta-min/max, phi-min/max
122  //of p1 and p2, where p1 and p2 are either two telescopes or a group (most
123  //probably this group) and a telescope.
124  //For theta-min/max, it is the smallest/largest angle which is used for the group dimension.
125  //For azimuthal/phi angles, all 4 combinations of "min" and "max" are tried and the one
126  //which gives the greatest azimuthal width to the group is kept.
127 
128  Float_t theta_min = TMath::Min(p1->GetThetaMin(), p2->GetThetaMin());
129  Float_t theta_max = TMath::Max(p1->GetThetaMax(), p2->GetThetaMax());
130  SetPolarMinMax(theta_min, theta_max);
131 
132  KVPosition* pp[2]; //array of pointers
133  pp[0] = p1;
134  pp[1] = p2;
135  Float_t max_width = 0.;
136  for (int i = 0; i < 2; i++) {
137  for (int j = 0; j < 2; j++) {
138  if (GetAzimuthalWidth(pp[i]->GetPhiMin(), pp[j]->GetPhiMax()) >
139  max_width) {
140  SetAzimuthalMinMax(pp[i]->GetPhiMin(), pp[j]->GetPhiMax());
141  max_width = GetAzimuthalWidth();
142  }
143  }
144  }
145 }

174 KVNameValueList* KVASGroup::DetectParticle(KVNucleus* part)
175 {
176  //Calculate energy losses of a charged particle traversing the telescopes of the group.
177  //This method return a list of TNamed where each detector which is throught in the particle
178  //are written with the corrresponding energy loss
179  //WARNING : this KVNameValueList has to be deleted by the user
180  // after use
181  //return 0 if no telescope are on the path of the particle (DEAD zone)
182  TList* d = GetTelescopesWithAngles(part->GetTheta(), part->GetPhi());
183  if (d) {
184  TIter nextd(d);
185  KVTelescope* t = 0;
186  KVNameValueList* nvl = new KVNameValueList();
187  while ((t = (KVTelescope*) nextd())) {
188  t->DetectParticle(part, nvl);
189  if (part->GetEnergy() <= 0.0) {
190  delete d;
191  return nvl;
192  }
193  }
194  delete d;
195  return nvl;
196  }
197  return 0;
198 }

209 {

262 void KVASGroup::CountLayers()
263 {
264  //Count the number of different layers to which telescopes in the group belong.
265  //This is based on different layers having different numbers.
266  //The layer closest to the target is assumed to have the smallest layer number,
267  //the layer furthest from the target is assumed to have the largest layer number.
268  fNumberOfLayers = 0;
269  fLayNumMin = 99;
270  fLayNumMax = 0;
271  if (GetTelescopes()) {
272  TIter ntel(GetTelescopes());
273  KVTelescope* tel;
274  UInt_t laynums[10];
275  while ((tel = (KVTelescope*) ntel())) {
276  if (fNumberOfLayers) {
277  Bool_t found = kFALSE;
278  //Check to make sure layer number not already in array
279  for (UInt_t i = 0; i < fNumberOfLayers; i++) {
280  if (tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber() == laynums[i])
281  found = kTRUE;
282  }
283  if (!found) {
284  laynums[fNumberOfLayers++] = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
285  if (fNumberOfLayers > 9) {
286  Warning("CountLayers", "Too many layers in group");
287  }
288  if (tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber() > fLayNumMax)
289  fLayNumMax = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
290  if (tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber() < fLayNumMin)
291  fLayNumMin = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
292  }
293  } else {
294  laynums[fNumberOfLayers++] = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
295  if (fNumberOfLayers > 9) {
296  Warning("CountLayers", "Too many layers in group");
297  }
298  if (tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber() > fLayNumMax)
299  fLayNumMax = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
300  if (tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber() < fLayNumMin)
301  fLayNumMin = tel->GetParentStructure("RING")->GetParentStructure("LAYER")->GetNumber();
302  }
303  }
304  }
305 }

182 void KVIDQAMarker::UpdateXandY()
183 {
184  // Set X and Y of this marker calculated from the coordinates of low point, up point, and fDelta.
185  // fDelta is the distance between this marker end the low point,
186  // normalized to the distance between low and up points;
187  if (!fParent) return;
188  Double_t x_low, y_low;
189  if (fParent->GetPoint(fIdx, x_low, y_low) > -1) {
190  if (fDelta) { // marker is between low and up points
191  Double_t x_up, y_up;
192  if (fParent->GetPoint(fIdx + 1, x_up, y_up) > -1) {
193  SetX(x_low + (x_up - x_low)*fDelta);
194  SetY(y_low + (y_up - y_low)*fDelta);
195  }
196  } else { // marker is over one point
197  SetX(x_low);
198  SetY(y_low);
199  }
200  }
201 }

207 void KVIDQAMarker::ExecuteEvent(Int_t event, Int_t px, Int_t py)
208 {
209 
210  if (fParent && !fParent->IsEditable()) return;
211 
212  static Double_t x_prev;
213  static Bool_t motion = kFALSE;
214 
215  // keep memory of the current opaque moving and reset it
216  // because if gPad->OpaqueMoving() is true then it is impossible
217  // to move this marker everywhere you want
218  Bool_t opaque = gPad->OpaqueMoving();
219  gPad->GetCanvas()->MoveOpaque(0);
220 
221  TMarker::ExecuteEvent(event, px, py);
222 
223  switch (event) {
224  case kButton1Down:
225  x_prev = GetX();
226  break;
227 
228  case kButton1Motion:
229  motion = kTRUE;
230  break;
231  case kButton1Up:
232  if (motion) {
233  motion = kFALSE;
234  if ((fIdx > -1)) {
235  if (fDelta) {// marker is between low and up points
236  fIdx = fParent->InsertPoint(fIdx + 1, GetX(), GetY(), x_prev);
237  fDelta = 0;
238  fParent->GetMarkers()->R__FOR_EACH(KVIDQAMarker, UpdateXandY)();
239  } else {
240  fParent->SetPoint(fIdx, GetX(), GetY());
241  fParent->GetMarkers()->R__FOR_EACH(KVIDQAMarker, UpdateXandY)();
242  }
243  gPad->Modified(kTRUE);
244  gPad->Update();
245  }
246  }
247  break;
248  }
249 
250  // set the initial opaque moving status
251  gPad->GetCanvas()->MoveOpaque(opaque);
252 }

261 void KVIDQAMarker::SetPointIndexAndX(Int_t idx, Double_t x)
262 {
263  // Set index of the low neighbour point and the X coordinate of this
264  // marker. Then fDelta and Y coordinates will be calculated and set to
265  // this marker.
266 
267  if (!fParent) return;
268  if (fParent->GetN() - 2 < idx) return;
269 
270  fIdx = idx;
271 
272  Double_t x_low, y_low, x_up, y_up;
273  if ((fParent->GetPoint(idx, x_low, y_low) > -1) && (fParent->GetPoint(idx + 1, x_up, y_up) > -1)) {
274  Double_t y;
275  if (x_low == x_up) y = y_low;
276  else y = y_up + (x - x_up) * (y_low - y_up) / (x_low - x_up);
277 
278  Double_t L = TMath::Sqrt(TMath::Power(x_up - x_low, 2) + TMath::Power(y_up - y_low, 2));
279  Double_t l = TMath::Sqrt(TMath::Power(x - x_low, 2) + TMath::Power(y - y_low, 2));
280 
281  if (L == 0) fDelta = 0.;
282  else fDelta = l / L;
283 
284  SetX(x);
285  SetY(y_low + (y_up - y_low)*fDelta);
286  }
287 }

324 Double_t KVIDQAMarker::DistanceToMarker(Double_t px, Double_t py) const
325 {
326  // Computes the closest distance of approach from point (px,py) to this marker.
327  // This distance is calculated on a A/Q scale i.e. if the Y coordinate is A
328  // then the Y coordinate will be divide by GetQ(). If the Y coordinate is
329  // Q then Y is change by GetA()/Y. Keep in mind that the X coordinate is supposed
330  // to give the ratio A/Q (mass number over the charge state).
331 
332  IGNORE_UNUSED(py);
333 
334  if (!fParent) return -1.;
335  Double_t X = px - fX;
336  Double_t Y = 0.;
337  /*
338  if( fParent->IsAvsAoQ() ){
339  if( GetQ() ) Y = (py-fY)/GetQ();
340  else return -1;
341  }
342  else{
343  if( py*fY ) Y = GetA()*(fY-py)/(fY*py);
344  else return -1.;
345  }
346  */
347  return TMath::Sqrt(X * X + Y * Y);
348 }

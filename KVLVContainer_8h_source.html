49 public:
50  KVLVColumnData(TClass* cl, const Char_t* name, const Char_t* method = "")
51  : fName(name), fMethod(method), result(""), fDate(kFALSE), fFmt(KVDatime::kCTIME), fIsKVDatime(kFALSE), fIsBoolean(kFALSE)
52  {
53  if (fMethod == "") fMethod.Form("Get%s", name);
54  fMethCall = new TMethodCall(cl, fMethod.Data(), "");
55  if (!fMethCall->IsValid()) {
56  std::cout << "Error in <KVLVColumnData::KVLVColumnData> : method " << fMethod.Data()
57  << " is not valid for class " << cl->GetName() << std::endl;
58  }
59  fRetType = (Int_t)fMethCall->ReturnType();
60  fDataFormat = "";
61  switch (fRetType) {
62  case (Int_t)TMethodCall::kLong :
63  fDataFormat = "%ld";
64  break;
65 
66  case (Int_t)TMethodCall::kDouble :
67  fDataFormat = "%lf";
68  break;
69 
70  default:
71  break;
72  }
73  if (!strcmp(fMethCall->GetMethod()->GetReturnTypeName(), "bool")
74  || !strcmp(fMethCall->GetMethod()->GetReturnTypeName(), "Bool_t")) fIsBoolean = kTRUE;
75  else fIsBoolean = kFALSE;
76  };

78  {
79  delete fMethCall;
80  };

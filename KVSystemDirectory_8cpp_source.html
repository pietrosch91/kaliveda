119 void KVSystemDirectory::GetListings()
120 {
121  // Fill lists of files and directories
122 
123  void* dir = gSystem->OpenDirectory(GetTitle());
124  if (!dir) {
125  Error("GetListings", "Cannot open directory %s", GetTitle());
126  return;
127  }
128 
129  const char* file = 0;
130  fContents = new TList;
131  fContents->SetOwner();
132  fSubdirs = new TList;
133  fSubdirs->SetOwner();
134  while ((file = gSystem->GetDirEntry(dir))) {
135  if (IsItDirectory(file)) {
136  TString sdirpath = "";
137  if (file[0] == '.' && file[1] == '\0')
138  continue;
139  else if (file[0] == '.' && file[1] == '.' && file[2] == '\0')
140  continue;
141  else {
142  sdirpath = GetTitle();
143  if (!sdirpath.EndsWith("/"))
144  sdirpath += "/";
145  sdirpath += file;
146  }
147  if (sdirpath != "") {
148  KVSystemDirectory* sd = new KVSystemDirectory(file, sdirpath.Data());
149  fSubdirs->Add(sd);
150  fTotDirs += sd->GetTotalSize();
151  }
152  } else {
153  KVSystemFile* sf = new KVSystemFile(file, GetTitle());
154  fContents->Add(sf);
155  fTotFiles += sf->GetSize();
156  }
157  }
158  gSystem->FreeDirectory(dir);
159 }

167 void KVSystemDirectory::ls(Option_t* opt) const
168 {
169  // if opt="nosubdirs", do not list contents of subdirectories
170  // default is to list recursively contents of all subdirectories
171 
172  TROOT::IndentLevel();
173  printf("%s :\n[total : %lld / files : %lld bytes]\n\n",
174  GetTitle(), GetTotalSize(), GetTotalFiles());
175  TROOT::IncreaseDirLevel();
176  TROOT::IncreaseDirLevel();
177  GetListOfFiles()->R__FOR_EACH(KVSystemFile, ls)();
178  if (!strcmp(opt, "nosubdirs")) {
179  TIter next(GetListOfDirectories());
180  KVSystemDirectory* sd;
181  while ((sd = (KVSystemDirectory*)next())) {
182  TROOT::IndentLevel();
183  printf("%-60s [total : %lld / files : %lld bytes]\n",
184  sd->GetTitle(), sd->GetTotalSize(), sd->GetTotalFiles());
185  }
186  } else {
187  GetListOfDirectories()->R__FOR_EACH(KVSystemDirectory, ls)();
188  }
189  TROOT::DecreaseDirLevel();
190  TROOT::DecreaseDirLevel();
191  printf("\n");
192 }

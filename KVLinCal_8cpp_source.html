24 //KVLinCal
25 //
26 //Linear calibration function:
27 //
28 // E = par(1) + par(2) * channel
29 //
30 //The channel number used is taken from the acquisition parameter
31 //whose type is set using the SetChannelParameter() method.
32 //
33 //If WithPedestalCorrection(kTRUE) is called, the corresponding
34 //pedestal value for the acquisition parameter in question will be
35 //subtracted from the channel number, as
36 //
37 // E = par(1) + par(2) * (channel - pedestal)
38 //
39 //By default, no pedestal correction is performed.

112 Double_t KVLinCal::Compute(Double_t chan) const
113 {
114  //Calculate the calibrated energy.
115  //if chan<0 channel number is taken directly from the chosen acquisition parameter of the associated detector
116  // - returns -99 if the parameter has not fired (KVACQParam::Fired==kFALSE)
117  //if chan>=0 we use value of chan
118  //
119  //Returns -99 if calibrator is not ready (parameters not set)
120 
121  if (fReady) {
122  Double_t chan_corr;
123  if (chan < 0) {
124  if (!GetParam()->Fired()) return -99.;
125  chan_corr = GetParam()->GetData();
126  } else chan_corr = chan;
127 
128  if (fPedCorr) chan_corr -= GetParam()->GetPedestal();
129 
130  return (GetParameter(0) + GetParameter(1) * chan_corr);
131  }
132  return -99.;
133 }

170 Double_t KVLinCal::Invert(Double_t energy)
171 {
172  //Given the calibrated (or simulated) energy in MeV,
173  //calculate the corresponding channel number according to the
174  //calibration parameters (useful for filtering simulations).
175 
176  KVDetector* det = GetDetector();
177  Int_t channel = 0;
178 
179  if (fReady) {
180  // linear transfer function
181  if (fPedCorr) {
182  channel = (Int_t)(0.5 + (energy - GetParameter(0)) / GetParameter(1)
183  + det->GetPedestal("GG"));
184  } else {
185  channel = (Int_t)(0.5 + (energy - GetParameter(0)) / GetParameter(1));
186  }
187  }
188  return (Double_t) channel;
189 }

18 <h2>KVPulseHeightDefect</h2>
19 <h4>Silicon PHD described by Moulton formula</h4>
20 
21 <p>
22 The static TF1 "fMoulton" calculates the PHD of a particle of incident energy
23 E and charge Z calculated according to Moulton's formula:
24 </p>
25 
26 <p>
27 log_10(PHD) = b(Z) + a(Z)*log_10(Ed) for Z > 2,<br>
28 PHD = 0 for Z <= Zmin<br>
29  with a(Z) = a_1*(Z**2/1000) + a_2<br>
30  b(Z) = b_1*(100/Z) + b_2<br>
31  Ed = energy lost by particle in detector (=E if particle stops)
32 </p>
33 <p>The five parameters (a_1, a_2, b_1, b_2) can be set using method<br>
34 SetParameters(a_1,a_2,b_1,b_2).
35 It should be noted that all the PHD of all INDRA silicon detectors
36 is calibrated with the fixed values of a_1=0.0223 and a_2=0.5682.
37 </p>
38 <!-- */

85 Double_t KVPulseHeightDefect::PHDMoulton(Double_t* x, Double_t* par)
86 {
87  //Returns Moulton PHD for given E and Z:
88  //
89  // log_10(PHD) = b(Z) + a(Z)*log_10(E), for Z > Zmin
90  //
91  // PHD = 0, for Z <= Zmin
92  //
93  // with a(Z) = a_1*(Z**2/1000) + a_2
94  // b(Z) = b_1*(100/Z) + b_2
95  // E = energy lost by particle in detector
96  //
97  // x[0] = E (MeV)
98  // par[0] = Z
99 
100  Int_t Z = par[0];
101  // by definition, no PHD for Z=2 or less
102  if (Z < 3) return 0;
103  Double_t a_1 = GetParameter(0);
104  Double_t a_2 = GetParameter(1);
105  Double_t b_1 = GetParameter(2);
106  Double_t b_2 = GetParameter(3);
107 
108  Double_t a = a_1 * Z * Z / 1000. + a_2;
109  Double_t b = b_1 * 100. / Z + b_2;
110  return (TMath::Power(10, b) * TMath::Power(x[0], a));
111 }

178 Double_t KVPulseHeightDefect::Compute(Double_t energy) const
179 {
180  //Calculate the pulse height defect (in MeV) for a given energy loss in the detector
181  //and a given Z (the Z of the particle should be set first using method SetZ)
182  //The Moulton formula is used:
183  //
184  // log_10(PHD) = b(Z) + a(Z)*log_10(E)
185  //
186  // PHD = 0, for Z <= Zmin
187  //
188  // with a(Z) = a_1*(Z**2/1000) + a_2
189  // b(Z) = b_1*(100/Z) + b_2
190  // E = energy loss of particle
191 
192  return const_cast<KVPulseHeightDefect*>(this)->GetMoultonPHDFunction(GetZ())->Eval(energy);
193 }

219 TF1* KVPulseHeightDefect::GetELossFunction(Int_t Z, Int_t A, Bool_t Wrong)
220 {
221  // Return pointer to TF1 giving energy loss in active layer of detector minus
222  // the pulse height defect for a given nucleus (Z,A).
223  //
224  // If Wrong=kTRUE (default:kFALSE) this will be calculated incorrectly
225  // (if the particle does not stop in the detector) by using the Moulton formula
226  // with the incident energy of the particle instead of the calculated energy
227  // loss of the particle.
228 
229  wrong = Wrong;
230 
231  if (!fDeltaEphd) {
232  fDeltaEphd = new TF1(Form("KVPulseHeightDefect:%s:ELossActive", GetDetector()->GetName()),
233  this, &KVPulseHeightDefect::ELossActive, 0., 1.e+04, 2, "KVPulseHeightDefect", "ELossActive");
234  fDeltaEphd->SetNpx(gEnv->GetValue("KVPulseHeightDefect.EnergyLoss.Npx", 20));
235  }
236  fDeltaEphd->SetParameters((Double_t)Z, (Double_t)A);
237  fDeltaEphd->SetRange(0., GetDetector()->GetSmallestEmaxValid(Z, A));
238  fDeltaEphd->SetTitle(Form("PHD dependent energy loss [MeV] in detector %s for Z=%d A=%d", GetDetector()->GetName(), Z, A));
239  GetMoultonPHDFunction(Z);
240  return fDeltaEphd;
241 }

251 Double_t KVPulseHeightDefect::ELossActive(Double_t* x, Double_t* par)
252 {
253  // Calculate energy lost in active layer of detector minus the Moulton PHD
254  // x[0] = incident energy
255  // par[0] = Z
256  // par[1] = A
257 
258  Double_t e = x[0];
259  TIter next(GetDetector()->GetListOfAbsorbers());
260  KVMaterial* mat;
261  while ((mat = (KVMaterial*)next()) != GetDetector()->GetActiveLayer()) {
262  // calculate energy losses in absorbers before active layer
263  e = mat->GetERes(par[0], par[1], e); //residual energy after layer
264  if (e <= 0.)
265  return 0.; // return 0 if particle stops in layers before active layer
266  }
267  // calculate energy loss in active layer
268  Double_t dE = mat->GetDeltaE(par[0], par[1], e);
269  // calculate Moulton PHD corresponding to energy lost in active layer
270  Double_t phd;
271  if (wrong) phd = PHDMoulton(&e, par); //incorrect calculation using incident energy
272  else phd = PHDMoulton(&dE, par);
273 
274  return dE - phd;
275 }

86 void DMSAvailableRunsFile::Update(Bool_t no_existing_file)
87 {
88  //Examine the contents of the DMS container corresponding to this datatype
89  //for parent dataset fDataSet.
90  //For each file found which was not already in the list of available runs and
91  //which corresponds to a run in the database gDataBase,
92  //we add an entry to the available runlist file:
93  // [run number]|[date of modification]|[name of file]
94  // For "old" runs we keep the existing informations (including KV version & username)
95  //
96  // When no_existing_file=kTRUE we are making an available runs file
97  // for the first time. There is no pre-existing file.
98 
99  TString runlist = GetFullPathToAvailableRunsFile();
100 
101  if (!no_existing_file) {
102  // read all existing informations
103  ReadFile();
104  //use "lockfile" to make sure nobody else tries to modify available_runs file
105  //while we are working on it
106  if (!runlist_lock.Lock(runlist.Data())) return;
107  }
108  //open temporary file
109  TString tmp_file_path(GetFileName());
110  ofstream tmp_file;
111  KVBase::OpenTempFile(tmp_file_path, tmp_file);
112 
113  KVDataRepository* repository = GetDataSet()->GetRepository();
114  //Info("Update","repository : %s",repository->IsA()->GetName());
115  cout << endl << "Updating runlist : " << flush;
116  //get directory listing from repository
117  KVUniqueNameList* dir_list =
118  repository->GetDirectoryListing(GetDataSet(), GetDataType());
119  if (!dir_list) {
120  //Info("Update","Sort au milieu");
121  return;
122  }
123  TIter next(dir_list);
124  DMSFile_t* objs;
125  //progress bar
126  Int_t ntot = dir_list->GetSize();
127  Int_t n5pc = TMath::Max(ntot / 20, 1);
128  Int_t ndone = 0;
129  KVDBTable* run_table = GetDataSet()->GetDataBase()->GetTable("Runs");
130  KVNumberList lnewruns; //list containing run with a valid file but not store in database
131  while ((objs = (DMSFile_t*) next())) { // loop over all entries in directory
132 
133  //Info("Update","objs->GetName()=%s",objs->GetName());
134  Int_t run_num;
135  //is this the correct name of a run in the repository ?
136  if ((run_num = IsRunFileName(objs->GetName()))) {
137 
138  KVDatime filedate;
139  KVDBRun* run = (KVDBRun*) run_table->GetRecord(run_num);
140  if (run) {
141  //runfile exists in repository
142  //check in case it is possible to extract a date from the name of the file
143  //the file may be much older than its DMS modtime (=date of DMS import)
144  if (!ExtractDateFromFileName(objs->GetName(), filedate))
145  filedate = objs->GetModTime();
146 
147  if (!no_existing_file) {
148  // was there already an entry for exactly the same file in the previous file ?
149  Int_t occIdx = 0;
150  KVNameValueList* prevEntry = RunHasFileWithDateAndName(run->GetNumber(), objs->GetName(), filedate, occIdx);
151  if (prevEntry) {
152  // copy infos of previous entry
153  tmp_file << run->GetNumber() << '|' << filedate.AsSQLString() << '|' << objs->GetName();
154  if (prevEntry->HasParameter(Form("KVVersion[%d]", occIdx))) {
155  tmp_file << "|" << prevEntry->GetStringValue(Form("KVVersion[%d]", occIdx)) << "|" << prevEntry->GetStringValue(Form("Username[%d]", occIdx));
156  }
157  tmp_file << endl;
158  } else {
159  // New Entry - write in temporary runlist file '[run number]|[date of modification]|[name of file]
160  tmp_file << run->GetNumber() << '|' << filedate.AsSQLString() << '|' << objs->GetName() << endl;
161  }
162  } else { // no previous existing file
163  // New Entry in a new file - write in temporary runlist file '[run number]|[date of modification]|[name of file]
164  tmp_file << run->GetNumber() << '|' << filedate.AsSQLString() << '|' << objs->GetName() << endl;
165  }
166  } else {
167  lnewruns.Add(run_num);
168  filedate = objs->GetModTime();
169  tmp_file << run_num << '|' << filedate.AsSQLString() << '|' << objs->GetName() << endl;
170  }
171  }
172 
173  ndone++;
174  if (!(ndone % n5pc))
175  cout << '>' << flush;
176  }
177 
178  cout << " DONE" << endl;
179 
180  // If runs not present in the database were found, emit an informational message
181  if (lnewruns.GetNValues())
182  Info("Update", "Found %d runs not in database :\n\tlist:%s", lnewruns.GetNValues(), lnewruns.AsString());
183 
184  delete dir_list;
185  //close temp file
186  tmp_file.close();
187 
188  if (CheckDirectoryForAvailableRunsFile()) {
189  if (no_existing_file) {
190  //use "lockfile" to make sure nobody else tries to modify available_runs file
191  //while we are working on it
192  if (!runlist_lock.Lock(runlist.Data())) return;
193  }
194 
195  //copy temporary file to KVFiles directory, overwrite previous
196  gSystem->CopyFile(tmp_file_path, runlist, kTRUE);
197  //set access permissions to 664
198  gSystem->Chmod(runlist.Data(), CHMODE(6, 6, 4));
199  }
200  //remove lockfile
201  runlist_lock.Release();
202 
203  //delete temp file
204  gSystem->Unlink(tmp_file_path);
205 }

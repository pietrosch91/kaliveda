153  Int_t events_read = 0;

167  for (int k = 0; k < kmax; k++) {
168  x = gRandom->Uniform(x0 - .5 * wx, x0 + .5 * wx);
169  y = gRandom->Uniform(y0 - .5 * wy, y0 + .5 * wy);

171  fGrid->Identify(x, y, idr);
172  Float_t PID = idr->PID;
173  if (idr->Aident) PID = (idr->Z + 0.1 * (idr->PID - 2. * idr->Z));
174  fLinearHisto->Fill(x, PID, weight);
175  // fConvertHisto->SetPoint(fConvertHisto->GetN(),x0, nuc.GetPID(), y0);
176  }

178  IncrementLinear((Float_t) events_read);

543 {

547  if (zmax < 0) zmax = ((KVIDentifier*)fGrid->GetIdentifiers()->Last())->GetZ();
548 
549  int ww = 10;

551  if (z > 4) ww = 20;

553  if (z > 10) ww = 40;
554  if (z > 12) ww = 50;

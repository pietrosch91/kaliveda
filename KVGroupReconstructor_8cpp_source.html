112 void KVGroupReconstructor::Reconstruct()
113 {
114  // Reconstruct the particles in the group
115 
116  GetEventFragment()->Clear();
117 
118  TIter nxt_traj(GetGroup()->GetTrajectories());
119  KVGeoDNTrajectory* traj;
120 
121  // loop over trajectories
122  while ((traj = (KVGeoDNTrajectory*)nxt_traj())) {
123  cout << "on trajectory:" << traj->GetTitle() << endl;
124 
125  // Work our way along the trajectory, starting from furthest detector from target,
126  // start reconstruction of new detected particle from first fired detector.
127  //
128  // More precisely: If detector has fired*,
129  // making sure fired detector hasn't already been used to reconstruct
130  // a particle, then we create and fill a new detected particle.
131  //
132  // *change condition by calling SetPartSeedCond("any"): in this case,
133  // particles will be reconstructed starting from detectors with at least 1 fired parameter.
134 
135  traj->IterateFrom();
136  KVGeoDetectorNode* node;
137  while ((node = traj->GetNextNode())) {
138 
139  KVDetector* d = node->GetDetector();
140  // if d has fired, has not already been analysed, and is not the last
141  // node of several trajectories, reconstruct particle
142  if ((d->Fired(GetEventFragment()->GetPartSeedCond()) && !d->IsAnalysed()
143  && !(traj->EndsAt(node) && node->GetBackwardTrajectories()->GetEntries() > 1))) {
144  cout << d->GetName() << " fired: reconstructing particle" << endl;
145 
146  KVReconstructedNucleus* kvdp = GetEventFragment()->AddParticle();
147  //add all active detector layers in front of this one
148  //to the detected particle's list
149  ReconstructParticle(kvdp, traj, node);
150 
151  //set detector state so it will not be used again
152  d->SetAnalysed(kTRUE);
153  }
154  }
155 
156  }
157 
158  if (GetEventFragment()->GetMult()) AnalyseParticles();
159 
160 }

196 void KVGroupReconstructor::AnalyseParticles()
197 {
198  cout << "GetNUnidentifiedInGroup()=" << GetNUnidentifiedInGroup() << endl;
199 
200  if (GetNUnidentifiedInGroup() > 1) { //if there is more than one unidentified particle in the group
201 
202  Int_t n_nseg_1 = 0;
203  KVReconstructedNucleus* nuc;
204  //loop over particles counting up different cases
205  while ((nuc = GetEventFragment()->GetNextParticle())) {
206  //ignore identified particles
207  if (nuc->IsIdentified())
208  continue;
209 
210  if (nuc->GetNSegDet() >= 2) {
211  //all part.s crossing 2 or more independent detectors are fine
212  nuc->SetStatus(KVReconstructedNucleus::kStatusOK);
213  } else if ((nuc->GetNSegDet() <= 1) && (nuc->GetReconstructionTrajectory()->GetNumberOfIdentifications())) {
214  //1 or less independent detector hit => depends on what's in the rest of the group
215  ++n_nseg_1;
216  } else {
217  //part.s crossing 0 independent detectors (i.E. arret ChIo)
218  //can not be reconstructed
219  nuc->SetStatus(KVReconstructedNucleus::kStatusStopFirstStage);
220  }
221  }
222  //loop again, setting status
223  while ((nuc = GetEventFragment()->GetNextParticle())) {
224  if (nuc->IsIdentified())
225  continue; //ignore identified particles
226 
227  if ((nuc->GetNSegDet() <= 1) && (nuc->GetReconstructionTrajectory()->GetNumberOfIdentifications())) {
228  if (n_nseg_1 == 1) {
229  //just the one ? then we can get it no problem
230  //after identifying the others and subtracting their calculated
231  //energy losses from the "dependent"/"non-segmented" detector
232  nuc->SetStatus(KVReconstructedNucleus::kStatusOKafterSub);
233  } else {
234  //more than one ? then we can make some wild guess by sharing the
235  //"non-segmented" contribution between them, but
236  //I wouldn't trust it as far as I can spit
237  nuc->SetStatus(KVReconstructedNucleus::kStatusOKafterShare);
238  }
239  //one possibility remains: the particle may actually have stopped e.g.
240  //in the DE detector of a DE-E telescope, in which case AnalStatus = 3
241  if (!nuc->GetReconstructionTrajectory()->GetNumberOfIdentifications()) {
242  //no ID telescopes with which to identify particle
243  nuc->SetStatus(KVReconstructedNucleus::kStatusStopFirstStage);
244  }
245  }
246  }
247  } else if (GetNUnidentifiedInGroup() == 1) {
248  //only one unidentified particle in group: just need an idtelescope which works
249 
250  //loop over particles looking for the unidentified one
251  KVReconstructedNucleus* nuc;
252  while ((nuc = GetEventFragment()->GetNextParticle())) {
253  if (!nuc->IsIdentified()) break;
254  }
255  cout << "nuc->GetNSegDet()=" << nuc->GetNSegDet() << endl;
256  if (nuc->GetReconstructionTrajectory()->GetNumberOfIdentifications()) {
257  //OK no problem
258  nuc->SetStatus(KVReconstructedNucleus::kStatusOK);
259  } else {
260  //dead in the water
261  nuc->SetStatus(KVReconstructedNucleus::kStatusStopFirstStage);
262  }
263  }
264 }

277 void KVGroupReconstructor::IdentifyParticle(KVReconstructedNucleus* PART)
278 {
279  // Try to identify this nucleus by calling the Identify() function of each
280  // ID telescope crossed by it, starting with the telescope where the particle stopped, in order
281  // - only attempt identification in ID telescopes containing the stopping detector.
282  // - only telescopes which have been correctly initialised for the current run are used,
283  // i.e. those for which KVIDTelescope::IsReadyForID() returns kTRUE.
284  // This continues until a successful identification is achieved or there are no more ID telescopes to try.
285  // The identification code corresponding to the identifying telescope is set as the identification code of the particle.
286 
287 
288  const KVSeqCollection* idt_list = PART->GetReconstructionTrajectory()->GetIDTelescopes();
289 
290  if (idt_list->GetEntries() > 0) {
291 
292  KVIDTelescope* idt;
293  TIter next(idt_list);
294  Int_t idnumber = 1;
295  Int_t n_success_id = 0;//number of successful identifications
296  while ((idt = (KVIDTelescope*) next())) {
297 
298  KVIdentificationResult* IDR = PART->GetIdentificationResult(idnumber++);
299 
300  if (idt->IsReadyForID()) { // is telescope able to identify for this run ?
301 
302  IDR->IDattempted = kTRUE;
303  idt->Identify(IDR);
304 
305  if (IDR->IDOK) n_success_id++;
306  } else
307  IDR->IDattempted = kFALSE;
308 
309  if (n_success_id < 1 &&
310  ((!IDR->IDattempted) || (IDR->IDattempted && !IDR->IDOK))) {
311  // the particle is less identifiable than initially thought
312  // we may have to wait for secondary identification
313  Int_t nseg = PART->GetNSegDet();
314  PART->SetNSegDet(TMath::Max(nseg - 1, 0));
315  //if there are other unidentified particles in the group and NSegDet is < 2
316  //then exact status depends on segmentation of the other particles : reanalyse
317  if (PART->GetNSegDet() < 2 && GetNUnidentifiedInGroup() > 1) {
318  AnalyseParticles();
319  return;
320  }
321  //if NSegDet = 0 it's hopeless
322  if (!PART->GetNSegDet()) {
323  AnalyseParticles();
324  return;
325  }
326  }
327 
328  }
329 
330  }

352 void KVGroupReconstructor::CalibrateParticle(KVReconstructedNucleus* PART)
353 {
354  //Calculate and set the energy of a (previously identified) reconstructed particle,
355  //including an estimate of the energy loss in the target.
356  //
357  //Starting from the detector in which the particle stopped, we add up the
358  //'corrected' energy losses in all of the detectors through which it passed.
359  //Whenever possible, for detectors which are not calibrated or not working,
360  //we calculate the energy loss. Measured & calculated energy losses are also
361  //compared for each detector, and may lead to new particles being seeded for
362  //subsequent identification. This is done by KVIDTelescope::CalculateParticleEnergy().
363  //
364  //For particles whose energy before hitting the first detector in their path has been
365  //calculated after this step we then add the calculated energy loss in the target,
366  //using gMultiDetArray->GetTargetEnergyLossCorrection().
367 
368  KVIDTelescope* idt = PART->GetIdentifyingTelescope();
369  idt->CalculateParticleEnergy(PART);
370  if (idt->GetCalibStatus() != KVIDTelescope::kCalibStatus_NoCalibrations) {
371  PART->SetIsCalibrated();
372  //add correction for target energy loss - moving charged particles only!
373  Double_t E_targ = 0.;
374  if (PART->GetZ() && PART->GetEnergy() > 0) {
375  E_targ = GetEventReconstructor()->GetTargetEnergyLossCorrection(PART);
376  PART->SetTargetEnergyLoss(E_targ);
377  }
378  Double_t E_tot = PART->GetEnergy() + E_targ;
379  PART->SetEnergy(E_tot);
380  // set particle momentum from telescope dimensions (random)
381  PART->GetAnglesFromStoppingDetector();
382  }
383 }

395 void KVGroupReconstructor::Identify()
396 {
397  //All particles which have not been previously identified (IsIdentified=kFALSE), and which
398  //may be identified independently of all other particles in their group according to the 1st
399  //order coherency analysis (KVReconstructedNucleus::GetStatus=0), will be identified.
400  //Particles stopping in first member of a telescope (KVReconstructedNucleus::GetStatus=3) will
401  //have their Z estimated from the energy loss in the detector (if calibrated).
402 
403  KVReconstructedNucleus* d;
404  while ((d = GetEventFragment()->GetNextParticle())) {
405  if (!d->IsIdentified()) {
406  if (d->GetStatus() == KVReconstructedNucleus::kStatusOK) {
407  // identifiable particles
408  IdentifyParticle(d);
409  } else if (d->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
410  // particles stopped in first member of a telescope
411  // estimation of Z (minimum) from energy loss (if detector is calibrated)
412  Int_t zmin = d->GetStoppingDetector()->FindZmin(-1., d->GetMassFormula());
413  if (zmin) {
414  d->SetZ(zmin);
415  d->SetIsIdentified();
416  // "Identifying" telescope is taken from list of ID telescopes
417  // to which stopping detector belongs
418  //d->SetIdentifyingTelescope( (KVIDTelescope*)d->GetStoppingDetector()->GetIDTelescopes()->At(0) );
419  }
420  }
421  }
422  }

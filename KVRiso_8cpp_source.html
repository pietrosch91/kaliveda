14 // Global variable returning the Riso variable. It is defined as follow:
15 //
16 // Etrans
17 // Riso= ----------
18 // 2*Epar
19 //
20 //
21 //with Etrans=Sum(etrans(i)) : sum of the transverse energies of the
22 // particules and fragments
23 // Epar=Sum(epar(i)) : sum of the parallel energies of the
24 // particules and fragments
25 //
26 //By default Riso is calculated from particle momenta in the default (laboratory) frame.
27 //However, this can be changed by giving the name of the desired frame
28 //with method SetFrame() :
29 // riso->SetFrame( "QP" ); // calculate in "QP" frame
30 //
31 // The names of the variables and the corresponding indexes are as follows:
32 //
33 // Name Index Meaning
34 //----------------------------------------------------------------------------
35 // Riso 0 (default) Riso
36 // Epar 1 Sum of parallel kinetic energies
37 // Eperp 2 Sum of transverse kinetic energies
38 //
39 //
40 // All these values can be obtained by calling the GetValuePtr() method which
41 // returns an array of Double_t containing the values.
42 //
43 // Here are examples on how obtaining values.
44 //
45 // KVRiso *priso=new KVRiso();
46 // KVRiso riso;
47 // ...
48 // Double_t r=priso->GetValue(); |
49 // r=prsio->GetValue(0); |--> Riso
50 // r=priso("Riso"); |
51 // Double_t epar=riso(1);---------> Sum of parallel kinetic energies
52 // Double_t etrans=priso->GetValue("Eperp");--> Sum of transverse kinetic
53 // energies
54 // Double_t *values=riso.GetValuePtr();
55 // r=values[0]; --> Riso
56 // epar=values[1]; --> Epar
57 // etrans=values[2]; --> Eperp
58 //
59 //
60 // Look at KVVarGlob class to have an example of use.
61 //
62 //
63 /* */

336 void KVRiso::CalculateRatio(void)
337 {
338  //Calculation of isotropy ratio from sums of parallel and
339  //transverse energies
340  //Limit to 100
341 
342  if (Riso < 0) {
343  if (Epar > 0) {

45 void KVWilckeReactionParameters::init()
46 {
47  ZC = ZP + ZT;
48  NP = AP - ZP;
49  NT = AT - ZT;
50  NC = NP + NT;
51  AC = AP + AT;
52  MU = AP * AT / (1.*AC);
53  RINT = InteractionRadius(AP, AT);
54  R0 = r0_Wilcke(AP, AT);
55  RP = SharpRadius(AP);
56  RT = SharpRadius(AT);
57  CP = MatterHalfDensityRadius(AP);
58  CT = MatterHalfDensityRadius(AT);
59  CBAR = CP * CT / (CP + CT);
60  RCP = ChargeRadius_Myers(ZP, AP);
61  RCT = ChargeRadius_Myers(ZT, AT);
62  RCTOTAL = RCP + RCT;
63  V0 = BSS_V0(ZP, AP, ZT, AT);
64  BSS_K = V0 - e2_Wilcke * ZP * ZT / RCTOTAL;
65  BSS_N = e2_Wilcke * ZP * ZT / RCTOTAL / BSS_K;
66  BSS_K /= pow(RCTOTAL, BSS_N);
67 
68  FISSIONTKE = TKESymFiss(ZC, AC);
69  ASYMMFISSIONTKE = FISSIONTKE * 4.*ZP * ZT / (1.*ZC * ZC);
70  GAMMA = NLDSurfaceTensionCoefficient(ZC, AC);
71  // Factor to convert the dimensionless proximity potential function of
72  // J. Blocki et al., Ann. Phys. (N. Y.) 105, 427 (1977)
73  // into a nuclear potential V_N
74  PROXFACTOR = 4 * TMath::Pi() * GAMMA * CBAR;
75  RBARRIER = SWaveFusionBarrierRadius(ZP, AP, ZT, AT);
76  // Maximum critical angular momentum for fusion
77  Double_t S = CT + CP + 0.3;
78  Double_t PHI = -0.96;
79  LCRIT = PROXFACTOR * PHI + e2_Wilcke * ZP * ZT / S / S;
80  LCRIT *= (-MU * mu_Wilcke * S * S * S / KVNucleus::hbar / KVNucleus::hbar);
81  LCRIT = 7. / 5.*pow(LCRIT, 0.5);
82 
83  if (!fBSS) fBSS = new TF1("BSS", this, &KVWilckeReactionParameters::VC, 0, 25, 0, "KVWilckeReactionParameters", "VC");
84  VC_RINT = fBSS->Eval(RINT);
85  if (!fCMThetaQuart) fCMThetaQuart = new TF1("ThetaQuart", this, &KVWilckeReactionParameters::QuarterPointAngle, 0, 100, 0, "KVWilckeReactionParameters", "QuarterPointAngle");
86  if (!fLmax) fLmax = new TF1("LMAX", this, &KVWilckeReactionParameters::Lmax, 0, 100, 0, "KVWilckeReactionParameters", "Lmax");
87  if (!fSigmaR) fSigmaR = new TF1("SIGMA-R", this, &KVWilckeReactionParameters::SigmaR, 0, 100, 0, "KVWilckeReactionParameters", "SigmaR");
88  if (!fProx) fProx = new TF1("PROX", this, &KVWilckeReactionParameters::ProxPot, 0, 25, 0, "KVWilckeReactionParameters", "ProxPot");
89  if (!fPotential) fPotential = new TF1("HIPOT", this, &KVWilckeReactionParameters::Potential, 0, 25, 0, "KVWilckeReactionParameters", "Potential");
90  if (!fCentPot) {
91  fCentPot = new TF1("CENTPOT", this, &KVWilckeReactionParameters::CentrifugalPotential, 0, 25, 1, "KVWilckeReactionParameters", "CentrifugalPotential");
92  fCentPot->SetNpx(1000);
93  fCentPot->SetParName(0, "l [hbar]");
94  }
95  VRB = fPotential->Eval(RBARRIER);
96 
97  if (!fSigmaFus) fSigmaFus = new TF1("SIGMA-FUS", this, &KVWilckeReactionParameters::SigmaFus, 0, 1.e+02, 0, "KVWilckeReactionParameters", "SigmaFus");
98 }

152 Double_t KVWilckeReactionParameters::epsilon_bar_Myers(Int_t Z, Int_t A)
153 {
154  // epsilon_bar, Eq.(7) in W.D. Myers, Phys. Lett. B 30, 451 (1969)
155  Double_t eps = (-2.*a2_Myers * pow(A, -THIRD) +
156  L_Myers * pow(delta_bar_Myers(Z, A), 2) +
157  c1_Myers * Z * Z * pow(A, -4. / 3.)) / K_Myers;
158  return eps;
159 }

166 Double_t KVWilckeReactionParameters::delta_bar_Myers(Int_t Z, Int_t A)
167 {
168  // delta_bar, Eq.(8) in W.D. Myers, Phys. Lett. B 30, 451 (1969)
169  Double_t I = (A - 2.*Z) / (1.*A);
170  Double_t delta = (I + (3 * c1_Myers / 8. / Q_Myers) * Z * Z * pow(A, -5. / 3.)) /
171  (1. + (9.*J_Myers / 4. / Q_Myers) * pow(A, -THIRD));
172  return delta;
173 }

183 Double_t KVWilckeReactionParameters::RLDCriticalAngularMomentum(Int_t z, Int_t a)
184 {
185  // S. Cohen, F. Plasil and W.J. Swiatecki, Ann. Phys. (N. Y.) 82, 557 (1974)
186  // Rotating Liquid Drop model critical angular momentum
187  // Only y_I is returned (normally valid for fissility parameter > 0.81)
188  // as no parameterisation for y_II is given in the paper ???
189 
190  Double_t Esurf = (17.9439 / 0.9517) * NLDSurfaceTensionCoefficient(z, a) * pow(a, 2. / 3.);
191  Double_t Ecoul = 0.7053 * z * z / pow(a, THIRD);
192  Double_t x = Ecoul / 2. / Esurf; // fissility parameter
193  Double_t y_I;
194  if (x < 0.75) y_I = 0.2829 - 0.3475 * x - 0.0016 * x * x + 0.0501 * x * x * x;
195  else y_I = (7. / 5.) * pow((1. - x), 2) - 4.5660 * pow(1. - x, 3.) + 6.7443 * pow(1 - x, 4.);
196  return y_I;
197 }

259 Double_t KVWilckeReactionParameters::PotentialPocketRadius(Double_t l)
260 {
261  // Find position (radial distance between centres) at which total potential has a
262  // minimum for the given angular momentum. Returns -1.0 if no pocket.
263 
264  TF1Derivative df(GetCentrifugalPotential(l));
265  TF1Derivative df2(GetCentrifugalPotential(l), 2);
266 
267  // look for extrema
268  Double_t rmax = 20.;
269  Double_t rmin = -1.0;
270  for (int i = 2; i; --i) {
271  rmin = df.GetX(0., 1., rmax);
272  if (rmin == rmax) return -1.0; // no extrema
273  // Test min/max ?
274  if (df2.Eval(rmin) > 0) return rmin; // found minimum
275  // That was a maximum. Try again.
276  rmax = rmin - 0.1;
277  }
278  return -1.0;
279 }

371 Double_t KVWilckeReactionParameters::GetBassReactionCrossSection(Double_t e_sur_a)
372 {
373  // Bass reaction cross-section [mb] for incident energy [MeV/nucleon]
374  Double_t R1 = 1.12 * pow(AP, 1. / 3.) - 0.94 * pow(AP, -1. / 3.);
375  Double_t R2 = 1.12 * pow(AT, 1. / 3.) - 0.94 * pow(AT, -1. / 3.);
376  Double_t Rint = R1 + R2 + 3.2;
377  Double_t VRINT = 1.44 * ZP * ZT / Rint - R1 * R2 / (R1 + R2);
378  return 10 * TMath::Pi() * pow(Rint, 2) * (1. - VRINT / ECM(e_sur_a));
379 }

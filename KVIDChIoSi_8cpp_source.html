71 Bool_t KVIDChIoSi::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
72 {
73  //Particle identification and code setting using identification grid KVIDGChIoSi
74 
75  IDR->SetIDType(GetType());
76  IDR->IDattempted = kTRUE;
77  //identification
78  Double_t chio = (y < 0. ? GetIDMapY() : y);
79  Double_t si = (x < 0. ? GetIDMapX() : x);
80 
81  if (ChIoSiGrid->IsIdentifiable(si, chio))
82  ChIoSiGrid->Identify(si, chio, IDR);
83  Int_t quality = ChIoSiGrid->GetQualityCode();
84  IDR->IDquality = quality;
85 
86  // set general ID code
87  IDR->IDcode = kIDCode4;
88  //if point lies above Zmax line, we give Zmax as Z of particle (real Z is >= Zmax)
89  //general ID code = kIDCode5 (Zmin)
90  if (quality == KVIDZAGrid::kICODE7) {
91  IDR->IDcode = kIDCode5;
92  }
93  //Identified particles with subcode kID_LeftOfBragg are given
94  //general ID code kIDCode5 (Zmin).
95  if (quality == KVIDGChIoSi::k_LeftOfBragg) {
96  IDR->IDcode = kIDCode5;
97  }
98  //unidentifiable particles with subcode kID_BelowSeuilSi are given
99  //general ID code kIDCode5 (Zmin) and we estimate Zmin from energy
100  //loss in ChIo
101  if (quality == KVIDGChIoSi::k_BelowSeuilSi) {
102  IDR->IDcode = kIDCode5;
103  IDR->Z = fchio->FindZmin();
104  IDR->SetComment("point to identify left of Si threshold line (bruit/arret ChIo?)");
105  }
106  if (quality == KVIDGChIoSi::k_RightOfEmaxSi) IDR->SetComment("point to identify has E_Si > Emax_Si i.e. codeur is saturated. Unidentifiable");
107 
108  return kTRUE;
109 }

150 void KVIDChIoSi::Initialize()
151 {
152  // Initialisation of telescope before identification.
153  // This method MUST be called once before any identification is attempted.
154  // Initialisation of grid is performed here.
155  // IsReadyForID() will return kTRUE if a grid is associated to this telescope for the current run
156  // or if no calib/ident parameters are available for current dataset.
157 
158  fchio = GetDetector(1);
159  fsi = GetDetector(2);
160  fsipgped = fsi->GetPedestal("PG");
161  fchiopgped = fchio->GetPedestal("PG");
162  ChIoSiGrid = (KVIDGChIoSi*) GetIDGrid();
163  if (ChIoSiGrid) {
164  ChIoSiGrid->Initialize();
165  SetBit(kReadyForID);
166  } else
167  ResetBit(kReadyForID);
168  if (!gDataSet->HasCalibIdentInfos()) SetBit(kReadyForID);
169 }

178 Bool_t KVIDChIoSi::CheckTheoreticalIdentificationThreshold(KVNucleus* ION, Double_t)
179 {
180  // Overrides KVIDTelescope method
181  // We lower the calculated energy threshold to be better in agreement with data
182  // EINC argument not used as ChIo-Si is always first telescope
183 
184  Double_t emin = (1.9 / 2.5) * (fchio->GetEIncOfMaxDeltaE(ION->GetZ(), ION->GetA()));
185  return (ION->GetEnergy() > emin);
186

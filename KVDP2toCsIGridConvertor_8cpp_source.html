66 void KVDP2toCsIGridConvertor::Convert(const Char_t* id_and_imf_file, const Char_t* gamma_file)
67 {
68  //Read ID and Gamma-line files for CsI R-L identifications.
69  //Initialise grids in KVIDCsI telescopes
70 
71  if (fGrids) fGrids->Clear();
72  else fGrids = new TList;
73  ReadFile(id_and_imf_file);
74  // set name of all IMF lines
75  TIter nextGrid(fGrids);
76  KVIDGrid* grid;
77  KVIDentifier* line;
78  while ((grid = (KVIDGrid*)nextGrid())) {
79  TIter nextOK(grid->GetCuts());
80  while ((line = (KVIDentifier*) nextOK())) {

83  }
84  ReadGammaFile(gamma_file);

87 }

95 void KVDP2toCsIGridConvertor::ReadGammaFile(const Char_t* gammafile)
96 {
97  ifstream gamfile;
98  gamfile.open(gammafile);
99 
100  if (!gamfile.good()) {
101  Error(KV__ERROR(ReadGammaFile),
102  "Problem reading file %s", gammafile);
103  return;
104  }
105  KVString s;
106 
107  s.ReadLine(gamfile);
108 
109  while (gamfile.good()) {
110 
111  if (s == "") {
112  s.ReadLine(gamfile);
113  continue;
114  }
115 
116  if (!s.BeginsWith('#')) { //'#' sign signals comments
117  int ring, modu, frun, lrun;
118  if (sscanf(s.Data(), "%d %d %d %d", &ring, &modu, &frun, &lrun) !=
119  4) {
120  Error(KV__ERROR(ReadGammaFile), "Problem reading file %s\nLast line read: %s",
121  gammafile, s.Data());
122  return;
123  };
124  //get grid for this ring,mod
125  TString name;
126  name.
127  Form
128  ("CsI R-L Grid First run=%d Last run=%d Ring min=%d Ring max=%d Mod min=%d Mod max=%d",
129  frun, lrun, ring, ring, modu, modu);
130  KVIDGrid* grid = (KVIDGrid*)fGrids->FindObject(name.Data());
131  if (!grid) {
132  Error(KV__ERROR(ReadGammaFile), "Can't find grid %s", name.Data());
133  }
134  int npoints;
135  gamfile >> npoints;
136  //found gamma line
137  KVIDLine* line = 0;
138  if (grid) {
139  line = grid->NewLine("ok");
140  line->SetName("gamma_line");
141  line->Set(npoints);
142  grid->Add("ok", line);
143  }
144  //read in points
145  for (int i = 0; i < npoints; i++) {
146  Double_t x, y;
147  gamfile >> x >> y;
148  if (grid)
149  line->SetPoint(i, x, y);
150  }
151  }
152  s.ReadLine(gamfile);
153  }
154  gamfile.close();
155 }

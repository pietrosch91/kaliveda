28 //Handles text files containing list of available runs for different datasets and types of data
29 //
30 //For each type of data associated with each dataset (type="raw", "recon", "ident" or "root")
31 //we maintain runlist files which contain the run numbers of datafiles which are physically

35 //
36 //The name of each file has the following format:
37 //
38 // [repository].available_runs.[dataset subdir].[type of data]

161 const Char_t* KVAvailableRunsFile::GetFilePath() const
162 {
163  // Returns the full path to the directory where the available runs file is stored.
164  // This is by default the same directory where the dataset files are stored.
165  // However, if KaliVeda was installed using a GNU-style directory layout
166  // (possibly in the system directories, e.g. Ubuntu packages)
167  // then the path will be the user's working directory + dataset name
168  //
169  // Alternatively, by setting the config variable
170  // KVAvailableRunsFile.RunFileDirectory: /some/path
171  //
172  // files will be stored/looked for in
173  // /some/path/[dataset]
174  //
175  // You can use environment variables - $(VAR) - in the path
176 
177  static TString filepath;
178  if (!fDataSet) {
179  Error("GetFilePath", "Dataset has not been set for this file.");
180  filepath = "";
181  } else {
182  TString p = gEnv->GetValue("KVAvailableRunsFile.RunFileDirectory", "");
183  if (p == "") {
184  // no user-defined run file directory
185  if (is_gnuinstall()) {
186  // GNU-style install: use working directory
187  filepath = GetWORKDIRFilePath(fDataSet->GetName());
188  } else
189  filepath = fDataSet->GetDataSetDir();
190  } else {
191  gSystem->ExpandPathName(p);
192  AssignAndDelete(filepath, gSystem->ConcatFileName(p.Data(), fDataSet->GetName()));
193  }
194  }
195  return filepath.Data();
196 }

272 Int_t KVAvailableRunsFile::IsRunFileName(const Char_t* filename)
273 {
274  //This method tests the string given as 'filename' to see if it could be the name of a runfile
275  //of the datatype of this available runs file (GetDataType()), for its parent dataset, fDataSet.
276  //Any protocol and/or path information in the filename is first removed, i.e. if
277  //e.g. "run8805.ident.root.2006-12-06_19:41:56" is a valid run filename, then the following
278  //are also valid and will return the run number 8805:
279  //
280  // "/hpss/in2p3.fr/group/indra/campagne5/ident/run8805.ident.root.2006-12-06_19:41:56"
281  // "root://ccxroot.in2p3.fr:1999//hpss/in2p3.fr/group/indra/campagne5/ident/run8805.ident.root.2006-12-06_19:41:56"
282  // "rfio:cchpssindra:/hpss/in2p3.fr/group/indra/campagne5/ident/run8805.ident.root.2006-12-06_19:41:56"
283  //
284  //The decision as to whether the filename is valid or not is based on whether it corresponds
285  //to one of the formats defined in $KVROOT/KVFiles/.kvrootrc, e.g. :
286  //
287  //# Default formats for runfile names (run number is used to replace integer format)
288  //DataSet.RunFileName.raw: run%d.raw
289  //DataSet.RunFileName.recon: run%d.recon.root
290  //DataSet.RunFileName.ident: run%d.ident.root
291  //DataSet.RunFileName.root: run%d.root
292  //
293  //We also test, for each format, "format%*", in case the filename has the file creation date
294  //string appended to it.
295  //
296  //If a working format is found, we return the run number deduced from the filename.
297  //If not, we return 0.
298 
299  //get format string for current dataset
300  TString fmt =
301  fDataSet->
302  GetDataSetEnv(Form("DataSet.RunFileName.%s", GetDataType()));
303  if (fmt == "") {
304  Error("IsRunFileName", "No default format set for datatype: %s",
305  GetDataType());
306  return 0;
307  }
308  Int_t run;
309  //Test format of filename
310  KVString _file(gSystem->BaseName(filename)); //Remove protocol and/or path
311  if (_file.Sscanf(fmt.Data(), &run))
312  return run;
313  //wtih date at end ?
314  fmt += "%*";
315  if (_file.Sscanf(fmt.Data(), &run))
316  return run;
317  return 0;
318 }

329 Bool_t KVAvailableRunsFile::ExtractDateFromFileName(const Char_t* name, KVDatime& date)
330 {
331  // We assume that 'name' is the name of a run of the type of this available runs file.
332  // We attempt several methods to try to extract a date from 'name'.
333  // If successful, we return kTRUE and 'date' contains the result.
334 
335  //get format string for current dataset
336  TString fmt =
337  fDataSet->
338  GetDataSetEnv(Form("DataSet.RunFileName.%s", GetDataType()));
339  Int_t rubbish = fmt.Index("%*");
340  if (rubbish > -1) fmt.Remove(rubbish, fmt.Length());
341  Int_t run = IsRunFileName(name);
342  if (!run) return kFALSE;
343  TString basename;
344  basename.Form(fmt.Data(), run);
345  TString fullname(name);
346  if (!fullname.Contains(basename)) return kFALSE;
347  if (fullname.Length() == basename.Length()) return kFALSE;
348  fullname.Remove(0, basename.Length() + 1);
349  fullname.ReplaceAll("_", " ");
350  fullname.ToUpper();
351  if (KVDatime::IsSQLFormat(fullname.Data())) {
352  date.SetSQLDate(fullname.Data());
353  return kTRUE;
354  } else {
355  fullname.ReplaceAll(".", " ");
356  if (KVDatime::IsGANACQFormat(fullname.Data())) {
357  date.SetGanacqDate(fullname.Data());
358  return kTRUE;
359  } else if (KVDatime::IsGANACQ2010Format(fullname.Data())) {
360  date.SetGanacq2010Date(fullname.Data());
361  return kTRUE;
362  }
363  }
364  return kFALSE;
365 }

382 void KVAvailableRunsFile::Update(Bool_t no_existing_file)
383 {
384  // Examine the contents of the repository directory corresponding to this datatype
385  // for parent dataset fDataSet.
386  // For each file found which was not already in the list of available runs and
387  // which corresponds to a run in the database gDataBase,
388  // we add an entry to the available runlist file:
389  // [run number]|[date of modification]|[name of file]
390  // For "old" runs we keep the existing informations (including KV version & username)
391  //
392  // When no_existing_file=kTRUE we are making an available runs file
393  // for the first time. There is no pre-existing file.
394 
395 
396  TString runlist = GetFullPathToAvailableRunsFile();
397 
398  if (!no_existing_file) {
399  // read all existing informations
400  ReadFile();
401  //use "lockfile" to make sure nobody else tries to modify available_runs file
402  //while we are working on it
403  if (!runlist_lock.Lock(runlist.Data())) return;
404  }
405  //open temporary file
406  TString tmp_file_path(GetFileName());
407  ofstream tmp_file;
408  KVBase::OpenTempFile(tmp_file_path, tmp_file);
409 
410  KVDataRepository* repository = fDataSet->GetRepository();
411 
412  cout << endl << "Updating runlist : " << flush;
413  //get directory listing from repository
414  KVUniqueNameList* dir_list =
415  repository->GetDirectoryListing(fDataSet, GetDataType());
416  if (!dir_list)
417  return;
418 
419  TIter next(dir_list);
420  KVBase* objs;
421  //progress bar
422  Int_t ntot = dir_list->GetSize();
423  Int_t n5pc = TMath::Max(ntot / 20, 1);
424  Int_t ndone = 0;
425  KVDBTable* run_table = 0;
426  KVDataBase* db = fDataSet->GetDataBase();
427  if (!db) {
428  db = new KVDataBase();
429  db->AddTable("Runs", "List of Runs");
430  }
431  run_table = db->GetTable("Runs");
432  while ((objs = (KVBase*) next())) { // loop over all entries in directory
433 
434  Int_t run_num;
435  //is this the correct name of a run in the repository ?
436  if ((run_num = IsRunFileName(objs->GetName()))) {
437 
438  KVDBRun* run = (KVDBRun*) run_table->GetRecord(run_num);
439  if (run) {
440  FileStat_t fs;
441  //get file modification date
442  if (repository->
443  GetFileInfo(fDataSet, GetDataType(),
444  objs->GetName(), fs)) {
445  //runfile exists in repository
446  TDatime modt(fs.fMtime);
447  if (!no_existing_file) {
448  // was there already an entry for exactly the same file in the previous file ?
449  Int_t occIdx = 0;
450  KVNameValueList* prevEntry = RunHasFileWithDateAndName(run->GetNumber(), objs->GetName(), modt, occIdx);
451  if (prevEntry) {
452  // copy infos of previous entry
453  tmp_file << run->GetNumber() << '|' << modt.AsSQLString() << '|' << objs->GetName();
454  if (prevEntry->HasParameter(Form("KVVersion[%d]", occIdx))) {
455  tmp_file << "|" << prevEntry->GetStringValue(Form("KVVersion[%d]", occIdx)) << "|" << prevEntry->GetStringValue(Form("Username[%d]", occIdx));
456  }
457  tmp_file << endl;
458  } else {
459  // New Entry - write in temporary runlist file '[run number]|[date of modification]|[name of file]
460  tmp_file << run->GetNumber() << '|' << modt.AsSQLString() << '|' << objs->GetName() << endl;
461  }
462  } else { // no previous existing file
463  // New Entry in a new file - write in temporary runlist file '[run number]|[date of modification]|[name of file]
464  tmp_file << run->GetNumber() << '|' << modt.AsSQLString() << '|' << objs->GetName() << endl;
465  }
466  }
467  } else {
468  Info("Update", "the current run [%s] is not in database", objs->GetName());
469  FileStat_t fs;
470  if (repository->GetFileInfo(fDataSet, GetDataType(), objs->GetName(), fs)) {
471  TDatime modt(fs.fMtime);
472  // New Entry in a new file - write in temporary runlist file '[run number]|[date of modification]|[name of file]
473  tmp_file << run_num << '|' << modt.AsSQLString() << '|' << objs->GetName() << endl;
474  } else {
475  Warning("Update", "%s GetFileInfo return kFALSE", objs->GetName());
476  }
477  }
478  }
479 
480  ndone++;
481  if (!(ndone % n5pc))
482  cout << '>' << flush;
483  }
484 
485  cout << " DONE" << endl;
486  delete dir_list;
487  //close temp file
488  tmp_file.close();
489 
490  if (CheckDirectoryForAvailableRunsFile()) {
491 
492  if (no_existing_file) {
493  //use "lockfile" to make sure nobody else tries to modify available_runs file
494  //while we are working on it
495  if (!runlist_lock.Lock(runlist.Data())) return;
496  }
497 
498  //copy temporary file to available runs file directory, overwrite previous
499  gSystem->CopyFile(tmp_file_path, runlist, kTRUE);
500  //set access permissions to 664
501  gSystem->Chmod(runlist.Data(), CHMODE(6, 6, 4));
502  }
503 
504  //remove lockfile
505  runlist_lock.Release();
506 
507  //delete temp file
508  gSystem->Unlink(tmp_file_path);
509 }

521 Bool_t KVAvailableRunsFile::GetRunInfo(Int_t run, TDatime& modtime,
522  TString& filename)
523 {
524  //Look for a given run number in the file, and read the file's modification date/time and filename
525  //If run not found, returns kFALSE
526  //If available runs file does not exist, Update() is called to create it.
527  //If there are multiple versions of the same run (abnormal situation),
528  //we print a warning and give info on the most recent file.
529 
530  KVList filenames, dates;
531  //get infos for all versions of run
532  GetRunInfos(run, &dates, &filenames);
533  if (filenames.GetEntries() == 1) { //only one version
534  filename = ((TObjString*) filenames.First())->String();
535  modtime = ((TObjString*) dates.First())->String().Data();
536  return kTRUE;
537  } else if (filenames.GetEntries() > 1) { //several versions
538  Warning("GetRunInfo",
539  "Multiple versions of this runfile exist in the repository. Infos for most recent file will be returned.");
540  Warning("GetRunInfo",
541  "You should clean the repository using KVDataSet::CleanMultiRunfiles.");
542  //find most recent version
543  TDatime most_recent("1998-12-25 00:00:00");
544  Int_t i_most_recent = 0;
545  for (int i = 0; i < dates.GetEntries(); i++) {
546  //check if run is most recent
547  TDatime rundate(((TObjString*) dates.At(i))->String().Data());
548  if (rundate > most_recent) {
549  most_recent = rundate;
550  i_most_recent = i;
551  }
552  }
553  filename = ((TObjString*) filenames.At(i_most_recent))->String();
554  modtime = ((TObjString*) dates.At(i_most_recent))->String().Data();
555  return kTRUE;
556  }
557  return kFALSE;
558 }

572 void KVAvailableRunsFile::GetRunInfos(Int_t run, KVList* dates,
573  KVList* files)
574 {
575  //Look for a given run number in the file, and read the file's modification date/time and filename
576  //These informations are stored in the two TList as TObjString objects (these objects belong to the
577  //lists and will be deleted by them).
578  //We do not stop at the first run found, but continue until the end of the file, adding
579  //informations for every occurence of the run in the file.
580  //If available runs file does not exist, Update() is called to create it.
581 
582  //does runlist exist ?
583  if (!OpenAvailableRunsFile()) {
584  Error("GetRunInfos", "Error opening available runs file");
585  return;
586  }
587  //clear lists - delete objects
588  dates->Delete();
589  files->Delete();
590 
591  //loop over lines in fRunlist file
592  //look for line beginning with 'run|'
593  TString line;
594  line.ReadLine(fRunlist);
595  while (fRunlist.good()) {
596  if (line.BeginsWith(Form("%d|", run))) {
597 
598  //found it
599  TObjArray* toks = line.Tokenize('|'); // split into fields
600  // check date is not identical to a previous entry
601  // i.e. there are spurious duplicate entries
602  TObjString* rundate = (TObjString*)toks->At(1)->Clone();
603  if (dates->FindObject(rundate->GetName())) {
604  delete toks;
605  delete rundate;
606  line.ReadLine(fRunlist);
607  continue;
608  }
609  //add date string
610  dates->Add(toks->At(1)->Clone());
611 
612  //backwards compatibility
613  //an old available_runs file will not have the filename field
614  //in this case we assume that the name of the file is given by the
615  //dataset's base file name (i.e. with no date/time suffix)
616  if (toks->GetEntries() > 2) {
617  files->Add(toks->At(2)->Clone());
618  } else {
619  files->
620  Add(new
621  TObjString(fDataSet->
622  GetBaseFileName(GetDataType(), run)));
623  }
624  delete toks;
625 
626  }
627  line.ReadLine(fRunlist);
628  }
629  CloseAvailableRunsFile();
630 }

1347 void KVAvailableRunsFile::ReadFile()
1348 {
1349  // Read all infos in available runs file and store as KVNameValueList objects in fAvailableRuns.
1350  // For each run in the file we add a KVNameValueList with the following fields:
1351  //
1352  // Name = run number
1353  // Occurs = number of times run appears in file
1354  // Filename[0] = name of first file for run
1355  // Filename[1] =
1356  // ...
1357  // Filename[Occurs-1] = name of last file for run
1358  // Date[0] = date & time of generation of first file etc.
1359  // KVVersion[0] = name of KaliVeda version used to generate first file etc. (if known)
1360  // Username[0] = name of user who generated first file etc. (if known)
1361 
1362  //does runlist exist ?
1363  if (!OpenAvailableRunsFile()) {
1364  Error("ReadFile", "Cannot open available runs file");
1365  return;
1366  }
1367 
1368  if (fAvailableRuns) delete fAvailableRuns;
1369  fAvailableRuns = new KVHashList;
1370  fAvailableRuns->SetOwner(kTRUE);
1371 
1372  TString fLine;
1373  Int_t line_number = 1;
1374  fLine.ReadLine(fRunlist);
1375 
1376  KVNumberList duplicate_lines;
1377 
1378  Int_t fRunNumber;
1379 
1380  while (fRunlist.good()) {
1381 
1382  TObjArray* toks = fLine.Tokenize('|'); // split into fields
1383 
1384  // number of fields can vary
1385  // nfields = 2: run number, date
1386  // nfields = 3: run number, date, filename
1387  // nfields = 5: run number, date, filename, KaliVeda version, username
1388  Int_t nfields = toks->GetEntries();
1389  KVString kvs(((TObjString*) toks->At(0))->GetString());
1390  fRunNumber = kvs.Atoi();
1391  if (nfields < 2) {
1392  Warning("ReadFile", "Less than 2 fields in entry for run %d (line:%d)???", fRunNumber, line_number);
1393  toks->ls();
1394  continue;
1395  }
1396  //get date string
1397  KVString datestring(((TObjString*) toks->At(1))->GetString());
1398 
1399  // is run already in list ?
1400  KVNameValueList* NVL = (KVNameValueList*)fAvailableRuns->FindObject(kvs);
1401  Int_t Occurs = (NVL ? NVL->GetIntValue("Occurs") : 0);
1402  if (!NVL) {
1403  NVL = new KVNameValueList(kvs);
1404  fAvailableRuns->Add(NVL);
1405  } else {
1406  // check date for run is different to any others
1407  Bool_t ok = kTRUE;
1408  for (Int_t ii = 0; ii < Occurs; ii++) {
1409  KVString olddate = NVL->GetStringValue(Form("Date[%d]", ii));
1410  if (olddate == datestring) {
1411  ok = kFALSE;
1412  duplicate_lines.Add(line_number);
1413  break;
1414  }
1415  }
1416  if (!ok) {
1417  delete toks;
1418  line_number++;
1419  fLine.ReadLine(fRunlist);
1420  continue;
1421  }
1422  }
1423  Occurs++;
1424  NVL->SetValue("Occurs", Occurs);
1425 
1426  NVL->SetValue(Form("Date[%d]", Occurs - 1), datestring.Data());
1427 
1428  //backwards compatibility
1429  //an old available_runs file will not have the filename field
1430  //in this case we assume that the name of the file is given by the
1431  //dataset's base file name (i.e. with no date/time suffix)
1432  KVString filename;
1433  if (nfields > 2) {
1434  filename = ((TObjString*) toks->At(2))->GetString();
1435  } else {
1436  filename = fDataSet->GetBaseFileName(GetDataType(), fRunNumber);
1437  }
1438  NVL->SetValue(Form("Filename[%d]", Occurs - 1), filename.Data());
1439  KVString kvversion, username;
1440  if (nfields > 4) {
1441  kvversion = ((TObjString*) toks->At(3))->GetString();
1442  username = ((TObjString*) toks->At(4))->GetString();
1443  NVL->SetValue(Form("KVVersion[%d]", Occurs - 1), kvversion.Data());
1444  NVL->SetValue(Form("Username[%d]", Occurs - 1), username.Data());
1445  }
1446  delete toks;
1447 
1448  line_number++;
1449  fLine.ReadLine(fRunlist);
1450  }
1451 
1452  CloseAvailableRunsFile();
1453 
1454  if (duplicate_lines.GetNValues()) {
1455  Info("ReadFile", "There were %d duplicate entries in available runs file, they will be removed", duplicate_lines.GetNValues());
1456  RemoveDuplicateLines(duplicate_lines);
1457  }
1458 
1459 }

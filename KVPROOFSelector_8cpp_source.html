574 
575  Bool_t justopened = kFALSE;
576 
577  TFile* file = 0;
578  TDirectory* pwd = gDirectory;
579  //if filename correspond to an already opened file, write in it

581  //and write in it
582  if (!(file = (TFile*)gROOT->GetListOfFiles()->FindObject(histo_file_name.Data()))) {
583  file = new TFile(histo_file_name.Data(), option);

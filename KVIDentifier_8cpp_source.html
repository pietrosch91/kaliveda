338 void KVIDentifier::Scale(Double_t sx, Double_t sy)
339 {
340  //Scale the coordinates of every point of the cut according to
341  // X_0 ---> sx*X_0
342  // Y_0 ---> sy*Y_0
343  //Default argument value '-1' means leave coordinate unchanged
344  if (TMath::Abs(sx) == 1 && TMath::Abs(sy) == 1)
345  return;
346  for (int i = 0; i < fNpoints; i++) {
347  if (sx > 0.)
348  fX[i] *= sx;
349  if (sy > 0.)
350  fY[i] *= sy;
351  }

367 void KVIDentifier::Scale(TF1* sx, TF1* sy)
368 {
369  // Apply to the coordinates of every point the transformation
370  // X_0 ---> sx(X_0,Y_0)
371  // Y_0 ---> sy(X_0,Y_0)
372  // if sx & sy are two-dimensional functions (inherits from TF2),
373  // otherwise we apply the transformation
374  // X_0 ---> sx(X_0)
375  // Y_0 ---> sy(Y_0)
376  // for sx & sy one-dimensionsal (TF1) functions.
377 
378  if (!sx && !sy) return; // if the two functions are NULL we leave coordinate unchanged
379  Bool_t sx_2d, sy_2d;
380  sx_2d = (sx && sx->InheritsFrom("TF2"));
381  sy_2d = (sy && sy->InheritsFrom("TF2"));
382  if (sx && sy) {
383  for (Int_t ii = 0; ii < fNpoints; ii++) {
384  Double_t fXp = 0.; // valeur intermediaire pour calculer X' et Y'
385  Double_t fYp = 0.; //a partir des coordonnees initiales
386  if (sx_2d) fXp = sx->Eval(fX[ii], fY[ii]);
387  else fXp = sx->Eval(fX[ii]);
388  if (sy_2d) fYp = sy->Eval(fX[ii], fY[ii]);
389  else fYp = sy->Eval(fY[ii]);
390  fX[ii] = fXp;
391  fY[ii] = fYp;
392  }
393  } else if (sx) {
394  for (Int_t ii = 0; ii < fNpoints; ii++) {
395  if (sx_2d) fX[ii] = sx->Eval(fX[ii], fY[ii]);
396  else fX[ii] = sx->Eval(fX[ii]);
397  }
398  } else {
399  for (Int_t ii = 0; ii < fNpoints; ii++) {
400  if (sy_2d) fY[ii] = sy->Eval(fX[ii], fY[ii]);
401  else fY[ii] = sy->Eval(fY[ii]);
402  }
403  }
404 }

440 void KVIDentifier::ExtendLine(Double_t Limit, Option_t* Direction)
441 {
442  // Right-clicking an identifier and selecting this method from the context
443  // menu allows to add a straight-line segment at the end or the beginning
444  // of the line (whichever is closest to the mouse).
445  //
446  // Direction = "" (default) - continue in the direction of first/last segment
447  // Direction = "H", "h", "hori", "HORI" etc. - add horizontal segment
448  // Direction = "v", "V", "vert", "VERT" etc. - add vertical segment
449 
450  if (!GetEditable()) return;
451 
452  // find closest point
453  Int_t px = gPad->GetEventX(); // mouse position
454  Int_t py = gPad->GetEventY(); // mouse position
455  Int_t dpx = px - gPad->XtoAbsPixel(gPad->XtoPad(fX[0])); // first point
456  Int_t dpy = py - gPad->YtoAbsPixel(gPad->XtoPad(fY[0])); // first point
457  Int_t dist_1 = dpx * dpx + dpy * dpy;
458  dpx = px - gPad->XtoAbsPixel(gPad->XtoPad(fX[fNpoints - 1])); // last point
459  dpy = py - gPad->YtoAbsPixel(gPad->XtoPad(fY[fNpoints - 1])); // last point
460  Int_t dist_N = dpx * dpx + dpy * dpy;
461  Int_t ipoint = (dist_N < dist_1 ? fNpoints - 1 : 0);
462 
463  // coordinates of new point
464  TString opt(Direction);
465  opt.ToUpper();
466  Double_t newX = fX[ipoint];
467  Double_t newY = fY[ipoint];
468  if (opt.BeginsWith("H")) newX = Limit;
469  else if (opt.BeginsWith("V")) newY = Limit;
470  else {
471  if (fNpoints < 2) {
472  Error("ExtendLine", "Cannot extend line, need at least one segment!");
473  return;
474  }
475  // trouver equation de la droite du dernier/premier segment
476  Double_t u, v, uu, vv;
477  Int_t iipoint = (ipoint == 0 ? 1 : fNpoints - 2);
478  u = fX[ipoint];
479  uu = fX[iipoint];
480  v = fY[ipoint];
481  vv = fY[iipoint];
482  Double_t a = vv - v;
483  Double_t b = u - uu;
484  Double_t c = -(b * v + a * u);
485  // use 'Limit' as x-coordinate of new point
486  newX = Limit;
487  newY = -(a * newX + c) / b;
488  }
489 
490  // add point
491  Int_t iend = (ipoint > 0 ? fNpoints : 0);
492  Double_t** ps = ExpandAndCopy(fNpoints + 1, iend);
493  CopyAndRelease(ps, ipoint, fNpoints++, ipoint + 1);
494 
495  // To avoid redefinitions in descendant classes
496  FillZero(iend, iend + 1);
497 
498  if (ipoint > 0) ipoint = fNpoints - 1;
499  fX[ipoint] = newX;
500  fY[ipoint] = newY;
501  gPad->Modified();
502 }

554 
555  Int_t px = gPad->GetEventX();
556  Int_t py = gPad->GetEventY();
557 
558  Info("InsertPoint", "x=%d y=%d", px, py);
559  Int_t ifound = -1;
560  //while ( gPad->XtoAbsPixel(gPad->XtoPad(fX[ii]))>px && ii<fNpoints-1) ii+=1;
561  //Recherche en pixel absolu
562  //pour trouver le point a partir duquel le TGraph

566  Info("InsertPoint", "np=%d ii=%d %lf %lf %d\n",
567  fNpoints,
568  ii,
569  fX[ii],
570  gPad->XtoPad(fX[ii]),
571  gPad->XtoAbsPixel(gPad->XtoPad(fX[ii]))
572  );

574  ifound = ii;
575  break;
576  }
577  }
578 
579  Info("InsertPoint", "point ifound=%d", ifound);
580 
581  Double_t deltaX = fX[ifound] - fX[ifound - 1];
582  Double_t aa = (fY[ifound] - fY[ifound - 1]) / (deltaX);
583  Double_t bb = fY[ifound] - fX[ifound] * aa;
584  //Conversion du px pour le TGraph et extrapole le new Y
585  Double_t newX = gPad->PadtoX(gPad->AbsPixeltoX(px));
586  Double_t newY = aa * newX + bb;
587  Info("InsertPoint", "nouveau point %lf %lf", newX, newY);
588 
589  //Dernier point du graph
590  Double_t lastX = fX[fNpoints - 1];
591  Double_t lastY = fY[fNpoints - 1];
592  //Le dernier point est ajoute en fin de graph
593  //fNpoint -> fNpoint+1
594  SetPoint(fNpoints, lastX, lastY);
595  //On deplace les autres points
596  for (Int_t ii = fNpoints - 2; ii >= ifound; ii -= 1)
597  SetPoint(ii, fX[ii - 1], fY[ii - 1]);
598  //Ajout du nouveau point
599  SetPoint(ifound, newX, newY);
600 
601  Print("");
602 
603  gPad->Modified();
604 
605  return ifound;
606 }

669 void KVIDentifier::ChechHierarchy(KVIDentifier* gr)
670 {
671  //Check if the line has been draw from left to right
672  //or right to left
673  //In this last case, we invert the filling order
674

676  Double_t* xx = gr->GetX();
677  Double_t* yy = gr->GetY();
678  Double_t Delta = xx[np - 1] - xx[0];
679  if (Delta > 0) return;
680 
681  Info("ChechHierarchy", "Invert filling order");

683  Double_t* yp = new Double_t[np];
684  for (Int_t nn = 0; nn < np; nn += 1) {
685  xp[nn] = xx[nn];
686  yp[nn] = yy[nn];
687  }
688 
689  for (Int_t ii = 1; ii <= np; ii += 1) {
690  gr->SetPoint(ii - 1, xp[np - ii], yp[np - ii]);
691  }
692  delete [] xp;
693  delete [] yp;

705 Int_t KVIDentifier::AddPointAtTheEnd()
706 {
707  // Insert a new point at the end
708  // its position is extrapolated from the two last points
709  // assuming linear evolution (yy = a*xx + bb)
710  // The KVIDentifier has to have at least 2 points
711  if (!GetEditable()) return -2;
712  if (fNpoints < 2) return -3;
713 
714  Int_t np = fNpoints;
715  Double_t deltaX = fX[np - 1] - fX[np - 2];
716  Double_t aa = (fY[np - 1] - fY[np - 2]) / (deltaX);
717  Double_t bb = fY[np - 1] - fX[np - 1] * aa;
718  //yy = aa*xx+bb -> bb = yy-aa*xx
719  Double_t newX = fX[np - 1] + deltaX;
720  Double_t newY = aa * newX + bb;
721  SetPoint(np, newX, newY);
722  gPad->Modified();
723  return np + 1;
724 
725 }

149 Double_t KVIDHarpeeICSi_e503::GetIDMapY(Option_t* opt)
150 {
151  // GetIDMapY (Public Interface Method)
152  //
153  // Calculates the current Y coordinate for identification. It is the
154  // calibrated energy of the fired acquisition parameter of HarpeeIC (i.e.
155  // the energy loss in the crossed segment) for which we found an ID grid. If
156  // more than one fired acq. parameter is found with their associated ID
157  // grid then zero is returned. The found ID grid is loaded in the pointer
158  // fGrid to be used in the method Identify.
159 
160  IGNORE_UNUSED(opt);
161 
162  fGrid = NULL;
163  Double_t E = 0;
164  TIter next(fDEdet->GetListOfCalibrators());
165  KVFunctionCal* cal = NULL;
166  while ((cal = (KVFunctionCal*)next())) {
167 
168  //Energy only coded in ACQParam with label 'A'
169  if (cal->GetLabel()[0] != 'A') continue;
170 
171  // look at only the ACQParam having its value greater than its pedestal
172  if (cal->GetACQParam()->Fired("P")) {
173 
174  // find if a grid exists in the list for the fired acq. parameter
175  // of the IC
176  KVIDZAGrid* tmp = FindGridForSegment(cal->GetNumber());
177 
178  // more than one grid correspond to fired acq. parameters
179  if (tmp && fGrid) {
180  fGrid = NULL;
181  return 0.;
182  } else if (tmp) {
183  // Load the grid associated to the fired ACQ Parameter
184  // for identification
185  fGrid = tmp;
186 
187  //find the punch-through line in the associated grid
188  //it would be better to directly use a KVIDGChIoSi grid
189  //but we do so for the moment...
190  fPunchThrough = (KVIDLine*) fGrid->GetCut("Punch_through");
191 
192  // returns the calibrated ACQ parameter i.e. energy measured
193  // in the fired section of HarpeeIC
194  if (cal->GetStatus()) E = cal->Compute();
195  }
196  }
197  }
198 
199  return E;
200 }

213 KVIDZAGrid* KVIDHarpeeICSi_e503::FindGridForSegment(Int_t num)
214 {
215  // Find the grid in the list fIDGrids for which its Y variable (
216  // grid->GetVarY() ) contains the name corresponding to the HarpeeIC segment
217  // ( i.e. the energy acq. parameter ) with number 'num'. We assume that each
218  // name in Y variable is separated by a comma and that the segment number is
219  // given by the last charater of the name, for example:
220  // VarY="CHI_01,CHI_2,IC3" means the the grid is associate to the segment 1,
221  // 2 and 3
222 
223  TIter next(fIDGrids);
224  KVIDZAGrid* grid = NULL;
225  while ((grid = (KVIDZAGrid*)next())) {
226  static KVString vars;
227  vars = grid->GetVarY();
228  vars.Begin(",");
229  while (!vars.End()) {
230  static KVString var;
231  var = vars.Next(kTRUE);
232  if (var.Remove(0, var.Length() - 1).Atoi() == num) return grid;
233  }
234  }
235  return NULL;
236 }

244 Bool_t KVIDHarpeeICSi_e503::Identify(
245  KVIdentificationResult* idr,
246  Double_t x,
247  Double_t y
248 )
249 {
250  // At the moment this class does nothing more than the basic KVIDHarpeeICSi
251  // identification with some additional logic checks.
252 
253  assert(idr);
254 
255  //IMPORTANT !!! For the moment we need to reset the kInitialised flag
256  //each time the Identify method is called.
257  //Else each time the same KVIDHarpeeICSi_e503 is called (for ex: VID_CHI_SI_05) more
258  //than once (typically in a KVIVSelector), it will be considered as already initialised and
259  //the base_id_result_ will keep the information of the former identification !!!
260  kInitialised_ = kFALSE;
261  Init();
262 
263  idr->IDOK = kFALSE;
264  idr->IDattempted = kTRUE;
265 
266  KVHarpeeIC* ic = static_cast<KVHarpeeIC*>(fDEdet);
267  assert(ic);
268 
269  if (ic->GetFiredSegNumber() == 0) {
270  // Mutliple segments (or none at all) have been fired in the ionisation
271  // chamber and so we abandon this identification attempt.
272  idr->IDquality = kSegmentMisfire;
273  return kFALSE;
274  }
275 
276  // Enforce ic number boundary
277  assert((ic->GetFiredSegNumber() > 0) && (ic->GetFiredSegNumber() < 8));
278 
279  // Base class performs the preliminary identification routine. We do this
280  // before the initialisation check as that is only applicable to the derived
281  // class. This way we will store the data from the base class
282  // identification even if the derived class initialisation fails.
283 
284  Bool_t status = KVIDHarpeeICSi::Identify(idr, x, y);
285 
286  //if a particle is well-identified (i.e. not too far from the identification lines)
287  //but it lies below the 'Punch_through' line, we give it a warning code
288  if (idr->IDquality < KVIDZAGrid::kICODE4) {
289  if (fPunchThrough) {
290  if (fverbose) Info("Identify", "... puch-through line found for detector %s (ICSegment=%d) ...", GetName(), ic->GetFiredSegNumber());
291 
292  if (fPunchThrough->WhereAmI(GetIDMapX(), GetIDMapY(), "below")) {
293  idr->IDquality = kBelowPunchThrough;
294  idr->SetComment("warning: point below punch-through line");
295  if (fverbose) Info("Identify", "... point(%lf, %lf) is below punch-through line ...", GetIDMapX(), GetIDMapY());
296  } else if (fverbose) Info("Identify", "... point(%lf, %lf) is above punch-through line ...", GetIDMapX(), GetIDMapY());
297 
298  } else if (fverbose) Info("Identify", "... no puch-through line found for detector %s (ICSegment=%d)...", GetName(), ic->GetFiredSegNumber());
299  }
300 
301  // Set the idcode and type for this telescope
302  idr->IDcode = fIDCode;
303  idr->SetIDType(GetType());
304 
305  return status;
306 }

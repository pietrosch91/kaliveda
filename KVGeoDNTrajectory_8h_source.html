172  void IterateFrom(const KVGeoDetectorNode* node0 = nullptr) const
173  {
174  // FORWARD ITERATION: moving towards the target
175  // Start an iteration over the trajectory nodes, starting from node0.
176  // If node0=nullptr (default value), start from the first node.
177  // After calling this method, calling method GetNextNode()
178  // will return each node of the trajectory starting with node0
179  // to the last one, after which it returns 0x0
180 
181  fIter_idx = (node0 == nullptr ? 0 : Index(node0));
182  fIter_delta = 1;
183  fIter_limit = GetN() - 1;
184  }
185  void IterateBackFrom(const KVGeoDetectorNode* node0 = nullptr) const
186  {
187  // BACKWARD ITERATION: moving away from the target
188  // Start an iteration over the trajectory nodes, starting from node0.
189  // If node0=nullptr (default value), start from the last node.
190  // After calling this method, calling method GetNextNode()
191  // will return each node of the trajectory starting with node0
192  // to the last one, after which it returns 0x0
193 
194  fIter_idx = (node0 == nullptr ? GetN() - 1 : Index(node0));
195  fIter_delta = -1;
196  fIter_limit = 0;
197  }

199  KVGeoDetectorNode* GetNextNode() const
200  {
201  // Get next node in iteration over trajectory.
202  // See IterateFrom(KVGeoDetectorNode*) and IterateBackFrom(KVGeoDetectorNode*)
203 
204  if (fIter_idx > -1) {
205  if (fIter_delta * (fIter_limit - fIter_idx) > 0) {
206  Int_t idx = fIter_idx;
207  fIter_idx += fIter_delta;
208  return GetNodeAt(idx);
209  } else {
210  // last node
211  fIter_idx = -1;
212  return GetNodeAt(fIter_limit);
213  }
214  }
215  return nullptr;
216  }

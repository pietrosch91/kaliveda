63 void KVNDTManager::init()
64 {
65  // Initialize Nuclear Data Table Manager
66  // We automatically instantiate a data table of each class which is
67  // declared as a "KVNuclDataTable" plugin
68  // If a new class is added to the .kvrootrc, there is no need to alter the code.
69 
70  Arange = 0;
71  Zrange = 0;
72 
73  KVString plugins = KVBase::GetListOfPlugins("KVNuclDataTable");
74  plugins.Begin(" ");
75  while (!plugins.End()) {
76  Add((KVNuclDataTable*)TClass::GetClass(plugins.Next())->New());
77  }
78 
79  Execute("Initialize", "");
80 
81 }

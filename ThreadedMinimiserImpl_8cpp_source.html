6 // ThreadedMinimiserImpl.cpp

8 // Description
9 //
10 // Threaded implementation of the silicon energy minimiser. This class
11 // implements the minimisation using the ThreadedMassEstimator.
12 //
13 // Warning: This implementation does not include a tolerance or a consistent
14 // maximisation check (maximisation is declared as soon as delta increases, i.e.
15 // we do not wait to see if delta is maximising consistently). It is recommended
16 // that you use the standard implementation (MonoMinimiserImpl) if you can as it
17 // does not require pthreads, is less complex and probably does the job well
18 // enough. If not then you are free to implement the aforementioned changes :)

159 Int_t ThreadedMinimiserImpl::Minimise(
160  UInt_t z_value, Double_t si_energy, Double_t csi_light,
161  MinimiserData* const data
162 )
163 {
164  assert(kInitialised_);
165 
166  // These assertions are paired with the user input validation tests in
167  // SiliconEnergyMinimiser::Minimise so you should never encounter them.
168  assert((z_value != 0) && (z_value <= 120));
169  assert(si_energy > 0.);
170  assert(csi_light > 0.);
171 
172  if (!kTelescopeSet_) {
173  Error("ThreadedMinimiserImpl::Minimise",
174  "You must call ThreadedMinimiserImpl::SetIDTelescope() "
175  "before calling this function");
176  return kTelescopeNotSet;
177  }
178 
179 #if __cplusplus < 201103L
180  ThreadedMinimiserData* threaded_data(NULL);
181 #else
182  ThreadedMinimiserData* threaded_data(nullptr);
183 #endif
184 
185  if (data) {
186  // Optional data storage pointer provided, cast it to make sure we can
187  // provide the extra data. It is assumed that the true nature of the
188  // pointer being passed in is a ThreadedMinimiserData pointer.
189  threaded_data = static_cast<ThreadedMinimiserData*>(data);
190  assert(threaded_data);
191  threaded_data->SetZ(z_value);
192  threaded_data->SetStatusCode(0);
193  }
194 
195  assert(estimator_input_);
196  assert(possible_a_values_);
197  assert(estimator_result_);
198  assert(mass_estimator_);
199 
200 #if __cplusplus < 201103L
201  mass_estimator_->GetPossibleAValues(z_value, possible_a_values_);
202 #else
203  mass_estimator_->GetPossibleAValues(z_value, possible_a_values_.get());
204 #endif
205 
206  // We should now also have a sorted vector of potential A values for the Z
207  // value returned by the base class identification.
208 
209  // Pack the struct with the information needed by our mass estimator
210 
211  estimator_input_->z = z_value;
212  estimator_input_->si_energy = si_energy;
213  estimator_input_->csi_light = csi_light;
214 
215  // We now estimate the value of A by simulating the passage of a nucleus
216  // (with various A values) through the Si:Isobutane:CsI detector stack and
217  // look for the point which minimises the difference between simulated and
218  // real silicon energy (This nucleus should have the correct A value).
219  //
220  // The mass estimator does over-estimate A by one in the low csi light
221  // regime as the isotopic bands converge and enhance the effect of
222  // simulation errors and approximations. This however should not be a
223  // problem as this is the first stage estimate.
224 
225  // TODO: Pass tolerance value into the mass estimator and implement
226  // consistent maximisation condition
227 #if __cplusplus < 201103L
228  if (!mass_estimator_->EstimateA(estimator_input_, possible_a_values_,
229  estimator_result_)) {
230  if (threaded_data) threaded_data->SetStatusCode(kEstimateAFailed);
231  return kEstimateAFailed;
232  }
233 #else
234  if (!mass_estimator_->EstimateA(estimator_input_.get(),
235  possible_a_values_.get(),
236  estimator_result_.get())) {
237  if (threaded_data) threaded_data->SetStatusCode(kEstimateAFailed);
238  return kEstimateAFailed;
239  }
240 #endif
241 
242  if (estimator_result_->status <= 0) {
243  // Negative status values indicate problems in the estimator
244  if (threaded_data) threaded_data->SetStatusCode(kEstimateAErrors);
245  return kEstimateAErrors;
246  }
247 
248  if (threaded_data) {
249  threaded_data->SetA(estimator_result_->a_value);
250  threaded_data->SetDelta(estimator_result_->delta);
251  threaded_data->SetForwardCounter(estimator_result_->forward_counter);
252  threaded_data->SetBackwardCounter(estimator_result_->backward_counter);
253  }
254  return estimator_result_->a_value;
255 }

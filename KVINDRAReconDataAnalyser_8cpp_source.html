112 void KVINDRAReconDataAnalyser::SubmitTask()
113 {
114  //Run the interactive analysis
115 
116  //make the chosen dataset the active dataset ( = gDataSet; note this also opens database
117  //and positions gDataBase & gIndraDB).
118  GetDataSet()->cd();
119  fSelector = nullptr;
120  fOldSelector = nullptr;
121 
122  theChain = new TChain("ReconstructedEvents");
123  theChain->SetDirectory(0); // we handle delete
124 
125  GetRunList().Begin();
126  Int_t run;
127 
128  // open and add to TChain all required files
129  // we force the opening of the files to avoid problems with xrootd which sometimes
130  // seems to have a little difficulty
131  while (!GetRunList().End()) {
132  run = GetRunList().Next();
133  TString fullPathToRunfile = gDataSet->GetFullPathToRunfile(GetDataType(), run);
134  cout << "Opening file " << fullPathToRunfile << endl;
135  TFile* f = (TFile*)gDataSet->OpenRunfile(GetDataType(), run);
136  cout << "Adding file " << fullPathToRunfile;
137  cout << " to the TChain." << endl;
138  dynamic_cast<TChain*>(theChain)->Add(fullPathToRunfile);
139  if (f && !f->IsZombie()) {
140  // update run infos in available runs file if necessary
141  KVAvailableRunsFile* ARF = gDataSet->GetAvailableRunsFile(GetDataType());
142  if (ARF->InfosNeedUpdate(run, gSystem->BaseName(fullPathToRunfile))) {
143  if (!((TTree*)f->Get("ReconstructedEvents"))) {
144  Error("SubmitTask", "No tree named ReconstructedEvents is present in the current file");
145  delete theChain;
146  return;
147  }
148  TEnv* treeInfos = (TEnv*)((TTree*)f->Get("ReconstructedEvents"))->GetUserInfo()->FindObject("TEnv");
149  if (treeInfos) {
150  TString kvversion = treeInfos->GetValue("KVBase::GetKVVersion()", "");
151  TString username = treeInfos->GetValue("gSystem->GetUserInfo()->fUser", "");
152  if (kvversion != "") ARF->UpdateInfos(run, gSystem->BaseName(fullPathToRunfile), kvversion, username);
153  } else {
154  Info("SubmitTask", "No TEnv object associated to the tree");
155  }
156  }
157  }
158  }
159  TotalEntriesToRead = theChain->GetEntries();
160  TString option("");
161 
162  // Add the total run list in option
163  if (!(option.IsNull())) option += ",";
164  option += Form("FullRunList=%s", GetFullRunList().GetList());
165 
166  // Add any user-defined options
167  if (GetUserClassOptions() != "") {
168  option += ",";
169  option += GetUserClassOptions();
170  }
171 
172  //debug
173  //Info("SubmitTask", "Option=%s", option.Data());
174 
175 
176  // for backwards compatibility, we allow user class to inherit from
177  // KVOldINDRASelector instead of KVINDRAEventSelector
178  TObject* new_selector = GetInstanceOfUserClass("KVOldINDRASelector");
179 
180  if (!new_selector || !new_selector->InheritsFrom("TSelector")) {
181  cout << "The selector \"" << GetUserClass() << "\" is not valid." << endl;
182  cout << "Process aborted." << endl;
183  SafeDelete(new_selector);
184  } else {
185  SafeDelete(new_selector);
186  Info("SubmitTask", "Beginning TChain::Process...");
187 #ifdef WITH_CPP11
188  if (GetProofMode() != KVDataAnalyser::EProofMode::None) dynamic_cast<TChain*>(theChain)->SetProof(kTRUE);
189 #else
190  if (GetProofMode() != KVDataAnalyser::None) dynamic_cast<TChain*>(theChain)->SetProof(kTRUE);
191 #endif
192  TString analysis_class;
193  if (GetAnalysisTask()->WithUserClass()) analysis_class.Form("%s%s", GetUserClassImp().Data(), GetACliCMode());
194  else analysis_class = GetUserClass();
195 
196  if (GetNbEventToRead()) {
197  theChain->Process(analysis_class, option.Data(), GetNbEventToRead());
198  } else {
199  theChain->Process(analysis_class, option.Data());
200  }
201  }
202  delete theChain;
203  fSelector = nullptr; //deleted by TChain/TTreePlayer
204  fOldSelector = nullptr; //deleted by TChain/TTreePlayer
205 }

236 KVNumberList KVINDRAReconDataAnalyser::PrintAvailableRuns(KVString& datatype)
237 {
238  //Prints list of available runs, sorted according to multiplicity
239  //trigger, for selected dataset, data type/analysis task, and system
240  //Returns list containing all run numbers
241 
242  KVNumberList all_runs =
243  GetDataSet()->GetRunList(datatype.Data(), GetSystem());
244  KVINDRADBRun* dbrun;
245 
246  //first read list and find what triggers are available
247  vector<int> triggers;
248  all_runs.Begin();
249  while (!all_runs.End()) {
250  dbrun = (KVINDRADBRun*)GetDataSet()->GetDataBase()->GetTable("Runs")->GetRecord(all_runs.Next());
251  if (triggers.size() == 0
252  || std::find(triggers.begin(), triggers.end(), dbrun->GetTrigger()) != triggers.end()) {
253  triggers.push_back(dbrun->GetTrigger());
254  }
255  }
256  //sort triggers in ascending order
257  std::sort(triggers.begin(), triggers.end());
258 
259  for (std::vector<int>::iterator it = triggers.begin(); it != triggers.end(); ++it) {
260  cout << " ---> Trigger M>" << *it << endl;
261  all_runs.Begin();
262  while (!all_runs.End()) {
263  dbrun = (KVINDRADBRun*)GetDataSet()->GetDataBase()->GetTable("Runs")->GetRecord(all_runs.Next());
264  if (dbrun->GetTrigger() == *it) {
265  cout << " " << Form("%4d", dbrun->GetNumber());
266  cout << Form("\t(%7d events)", dbrun->GetEvents());
267  cout << "\t[File written: " << dbrun->GetDatime().
268  AsString() << "]";
269  if (dbrun->GetComments())
270  cout << "\t" << dbrun->GetComments();
271  cout << endl;
272  }
273  }
274  cout << endl;
275  }
276  return all_runs;
277 }

333  // We call gIndra->SetParameters for the current run.

342  cout << endl << " =================== New Run =================== " <<
343  endl << endl;
344 
345  CurrentRun->Print();

349  }
350 
351  cout << endl << " ================================================= " <<
352  endl << endl;
353 
354  ConnectRawDataTree();
355  ConnectGeneDataTree();

391 void KVINDRAReconDataAnalyser::preAnalysis()
392 {
393  // Read and set raw data for the current reconstructed event
394  // Any required data patches ("rustines") are applied.
395 
396  if (!theRawData) return;
397  // all recon events are numbered 1, 2, ... : therefore entry number is N-1
398  Long64_t rawEntry = GetRawEntryNumber();
399 
400  gIndra->GetACQParams()->R__FOR_EACH(KVACQParam, Clear)();
401 
402  theRawData->GetEntry(rawEntry);
403  for (int i = 0; i < NbParFired; i++) {
404  KVACQParam* par = gIndra->GetACQParam((*parList)[ParNum[i]]->GetName());
405  if (par) {
406  par->SetData(ParVal[i]);
407  }
408  }
409  gIndra->SetMinimumOKMultiplicity(GetReconstructedEvent());
410 
411  // as rustines often depend on a knowledge of the original raw data,
412  // we apply them after it has been read in
413  KVINDRAReconEvent* event = (KVINDRAReconEvent*)GetReconstructedEvent();
414  if (fRustines.HasActivePatches()) fRustines.Apply(event);
415 }

426 void KVINDRAReconDataAnalyser::ConnectRawDataTree()
427 {
428  // Called by preInitRun().
429  // When starting to read a new run (=new file), we look for the TTree "RawData" in the
430  // current file (it should have been created by KVINDRARawDataReconstructor).
431  // If found, it will be used by ReadRawData() to set the values of all acquisition parameters
432  // for each event.
433 
434  theRawData = (TTree*)theChain->GetCurrentFile()->Get("RawData");
435  if (!theRawData) {
436  Warning("ConnectRawDataTree", "RawData tree not found in file; raw data parameters of detectors will not be available in analysis");
437  return;
438  } else
439  Info("ConnectRawDataTree", "Found RawData tree in file");
440  Int_t maxNopar = theRawData->GetMaximum("NbParFired");
441  if (ParVal) delete [] ParVal;
442  if (ParNum) delete [] ParNum;
443  ParVal = new UShort_t[maxNopar];
444  ParNum = new UInt_t[maxNopar];
445  parList = (TObjArray*)theRawData->GetUserInfo()->FindObject("ParameterList");
446  theRawData->SetBranchAddress("NbParFired", &NbParFired);
447  theRawData->SetBranchAddress("ParNum", ParNum);
448  theRawData->SetBranchAddress("ParVal", ParVal);
449  Info("ConnectRawDataTree", "Connected raw data parameters");
450  Entry = 0;
451 }

66 void KVIDSi75SiLi_camp5::Initialize()
67 {
68  // Initialize telescope for current run.
69  // If there is at least 1 (GG) grid, we set IsReadyForID = kTRUE
70  // "Natural" line widths are calculated for grids.
71 
72  fSiLi = GetDetector(2);
73  fSi75 = GetDetector(1);
74  fGGgrid = fPGgrid = fPGZgrid = 0;
75  TIter next(fIDGrids);
76  KVIDGrid* grid = 0;
77  while ((grid = (KVIDGrid*)next())) {
78  if (!strcmp(grid->GetVarY(), "SI75_GG")) fGGgrid = (KVIDZAGrid*)grid;
79  else if (!strcmp(grid->GetVarY(), "SI75_PG")) {
80  if (grid->OnlyZId()) fPGZgrid = (KVIDZAGrid*)grid;
81  else fPGgrid = (KVIDZAGrid*)grid;
82  }
83  }
84  if (fGGgrid) {
85  SetBit(kReadyForID);
86  fGGgrid->Initialize();
87  if (fPGgrid) fPGgrid->Initialize();
88  if (fPGZgrid) fPGZgrid->Initialize();
89  } else
90  ResetBit(kReadyForID);
91 }

126 Bool_t KVIDSi75SiLi_camp5::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
127 {
128  //Particle identification and code setting using identification grids.
129 
130  //perform identification in Si75(GG) - SiLi(PG) map
131  IDR->SetIDType(GetType());
132  IDR->IDattempted = kTRUE;
133 
134  Double_t si75 = (y < 0. ? GetIDMapY("GG") : y);
135  Double_t sili = (x < 0. ? GetIDMapX() : x);
136 
137  KVIDGrid* theIdentifyingGrid = 0;
138 
139  fGGgrid->Identify(sili, si75, IDR);
140  theIdentifyingGrid = (KVIDGrid*)fGGgrid;
141 
142  if (fGGgrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fPGgrid) { //we have to try PG grid (if there is one)
143 
144  // try Z & A identification in Si75(PG)-SiLi(PG) map
145  si75 = (y < 0. ? GetIDMapY("PG") : y);
146  fPGgrid->Identify(sili, si75, IDR);
147  theIdentifyingGrid = (KVIDGrid*)fPGgrid;
148 
149  if (fPGgrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fPGZgrid) { //we have to try PGZ grid (if there is one)
150 
151  fPGZgrid->Identify(sili, si75, IDR);
152  theIdentifyingGrid = (KVIDGrid*)fPGZgrid;
153  }
154  }
155 
156  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
157  // only if the final quality code is kICODE8 do we consider that it is
158  // worthwhile looking elsewhere. In all other cases, the particle has been
159  // "identified", even if we still don't know its Z and/or A (in this case
160  // we consider that we have established that they are unknowable).
161  return kFALSE;
162  }
163 
164  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
165  // if the final quality code is kICODE7 (above last line in grid) then the estimated
166  // Z is only a minimum value (Zmin)
167  IDR->IDcode = kIDCode5;
168  return kTRUE;
169  }
170 
171  if (theIdentifyingGrid->GetQualityCode() > KVIDZAGrid::kICODE3 &&
172  theIdentifyingGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
173  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
174  // corresponds to a point which is inbetween the lines, i.e. noise
175  IDR->IDcode = kIDCode10;
176  return kTRUE;
177  }
178 
179  // set general ID code Si75-SiLi
180  IDR->IDcode = kIDCode3;
181  return kTRUE;
182 }

207 void KVUpDater::SetCalibrationParameters(UInt_t run)
208 {
209  //Set calibration parameters for this run.
210  //This will:
211  // reset all the calibrators of all the detectors ready to receive the calibrators for the run (handled by child classes),
212  // set calibration parameters for the run
213 
214  cout << "Setting calibration parameters of multidetector array for run " << run << ":" <<
215  endl;
216  KVDBRun* kvrun = dynamic_cast<KVDBRun*>(gDataBase->GetTable("Runs")->GetRecord(run));
217  if (!kvrun) {
218  Error("SetParameters(UInt_t)", "Run %u not found in database!", run);
219  return;
220  }
221  //Reset all calibrators of all detectors first
222  TIter next(gMultiDetArray->GetDetectors());
223  KVDetector* kvd;
224  KVCalibrator* kvc;
225  while ((kvd = (KVDetector*) next())) {
226  if (kvd->GetListOfCalibrators()) {
227  TIter next_cal(kvd->GetListOfCalibrators());
228  while ((kvc = (KVCalibrator*) next_cal())) {
229  kvc->Reset();
230  }
231  }
232  }
233  SetCalibParameters(kvrun);
234 }

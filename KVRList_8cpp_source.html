113 KVBase* KVRList::FindObject(const Char_t* name, const Char_t* type) const
114 {
115 //
116 // Find an object in this list using its type and name. Requires a sequential
117 // scan till the object has been found. Returns 0 if object with specified
118 // name is not found.
119 //
120 
121  if (type && name) {
122  if (GetSize()) { //make sure list contains some objects
123  for (Int_t i = 0; i < GetSize(); i++) {
124  KVBase* obj = (KVBase*) At(i);
125  if (!strcmp(type, obj->GetType())
126  && !strcmp(name, obj->GetName()))
127  return obj;
128  }
129  }
130  }
131 // Warning("GetObject",KVRLIST_NOT_WITH_THIS_NAME_AND_TYPE,name,type);

194 {

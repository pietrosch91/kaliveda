58 void KVIDChIoSi75_camp5::Initialize()
59 {
60  // Initialize telescope for current run.
61  // If there is at least 1 grid, we set fCanIdentify = kTRUE
62  // "Natural" line widths are calculated for KVIDZAGrids.
63 
64  fchio = GetDetector(1);
65  fsi = GetDetector(2);
66  fGGgrid = fPGgrid = 0;
67  TIter next(fIDGrids);
68  KVIDGrid* grid = 0;
69  while ((grid = (KVIDGrid*)next())) {
70  if (!strcmp(grid->GetVarY(), "CIGG")) fGGgrid = (KVIDZAGrid*)grid;
71  else if (!strcmp(grid->GetVarY(), "CIPG")) fPGgrid = (KVIDZAGrid*)grid;
72  }
73  if (fGGgrid || fPGgrid) {
74  SetBit(kReadyForID);
75  if (fGGgrid) fGGgrid->Initialize();
76  if (fPGgrid) fPGgrid->Initialize();
77  } else
78  ResetBit(kReadyForID);
79 }

117 Bool_t KVIDChIoSi75_camp5::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
118 {
119  //Particle identification and code setting using identification grids.
120 
121  IDR->SetIDType(GetType());
122  IDR->IDattempted = kTRUE;
123  KVIDGrid* theIdentifyingGrid = 0;
124 
125  if (fGGgrid) {
126  Double_t cigg = (y < 0. ? GetIDMapY("GG") : y);
127  Double_t sigg = (x < 0. ? GetIDMapX("GG") : x);
128  fGGgrid->Identify(sigg, cigg, IDR);
129  theIdentifyingGrid = (KVIDGrid*)fGGgrid;
130 
131  }
132  if ((fGGgrid && fGGgrid->GetQualityCode() > KVIDZAGrid::kICODE6) || !fGGgrid) { //we have to try PG grid (if there is one)
133 
134  if (fPGgrid) {
135  Double_t cipg = (y < 0. ? GetIDMapY("PG") : y);
136  Double_t sipg = (x < 0. ? GetIDMapX("PG") : x);
137  fPGgrid->Identify(sipg, cipg, IDR);
138  theIdentifyingGrid = (KVIDGrid*)fPGgrid;
139  }
140  }
141 
142  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
143  // if the final quality code is kICODE7 (above last line in grid) then the estimated
144  // Z is only a minimum value (Zmin)
145  IDR->IDcode = kIDCode5;
146  return kTRUE;
147  }
148 
149  if (theIdentifyingGrid->GetQualityCode() > KVIDZAGrid::kICODE3 &&
150  theIdentifyingGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
151  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
152  // corresponds to a point which is inbetween the lines, i.e. noise
153  IDR->IDcode = kIDCode10;
154  return kTRUE;
155  }
156 
157  // set general ID code ChIo-Si
158  IDR->IDcode = kIDCode4;
159  return kTRUE;
160 }

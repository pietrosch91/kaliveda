15 //
16 // Z-only identification grids for INDRA Ionisation chamber-Silicon telescopes.
17 // To identify a particle with this grid, do:
18 // if( grid->IsIdentifiable(x,y) ){
19 // grid->Identify(x,y,nuc);
20 // }
21 //
22 // After attempting identification with method Identify, GetQualityCode() method returns
23 // one of the following status codes:
24 //
25 // KVIDZAGrid::kICODE0, OK
26 // KVIDZAGrid::kICODE1, slight ambiguity of Z, which could be larger
27 // KVIDZAGrid::kICODE2, slight ambiguity of Z, which could be smaller
28 // KVIDZAGrid::kICODE3, slight ambiguity of Z, which could be larger or smaller
29 // KVIDZAGrid::kICODE4, point is in between two lines of different Z, too far from either to be considered well-identified
30 // KVIDZAGrid::kICODE5, point is in between two lines of different Z, too far from either to be considered well-identified
31 // KVIDZAGrid::kICODE6, (x,y) is below first line in grid
32 // KVIDZAGrid::kICODE7, (x,y) is above last line in grid
33 // KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid
34 // KVIDGChIoSi::k_BelowPunchThrough "warning: point below punch-through line",
35 // KVIDGChIoSi::k_BelowSeuilSi "point to identify left of Si threshold line (bruit/arret ChIo?)",
36 // KVIDGChIoSi::k_LeftOfBragg "point to identify below Bragg curve. Z given is a Zmin",
37 // KVIDGChIoSi::k_RightOfEmaxSi "point to identify has E_Si > Emax_Si i.e. codeur is saturated. Unidentifiable",
38 // KVIDGChIoSi_e494s::k_BelowSeuilChIo "point to identify was below of "ChIo threshold" line,
39 //
40 // Correctly-identified particles with no ambiguity have quality codes < KVIDZAGrid::kICODE4.
41 // Particles with quality code KVIDGChIoSi::k_BelowPunchThrough are also in principle correctly identified,
42 // but we give a warning that the point is below the line 'Punch_through' delimiting the region in which the
43 // identification lines can be mixed with particles punching through the silicon if no additional condition
44 // is placed on e.g. the CsI detector behind the silicon not firing.
45 // Particles with quality code KVIDGChIoSi::k_LeftOfBragg are below the Bragg curve and the Z attributed
46 // is a minimum value.
47 // Particles with code KVIDZAGrid::kICODE7 are (far) above the last line of the grid, their Z is also a minimum.

133 void KVIDGChIoSi_e494s::Identify(Double_t x, Double_t y, KVIdentificationResult* idr) const
134 {
135  // After identification of the particle, we adjust the quality code
136  // (if the particle was well-identified by KVIDGChIoSi::Identify, i.e. with
137  // fICode<KVIDZAGrid::kICODE4) if:
138  // the particle is below the 'ChIo threshold line' => quality code KVIDGChIoSi_e494s::k_BelowSeuilChIo
139 
140  KVIDGChIoSi::Identify(x, y, idr);
141 
142  if (fICode < kICODE4) {
143 
144  if (fChIoSeuil && fChIoSeuil->WhereAmI(x, y, "below")) {
145  const_cast<KVIDGChIoSi_e494s*>(this)->fICode = k_BelowSeuilChIo;
146  idr->SetComment("warning: point below ChIo threshold line");
147  }
148  Int_t ZValidityvalue = -1;
149  ZValidityvalue = const_cast<KVIDGChIoSi_e494s*>(this)->GetParameters()->GetIntValue("ZValidity");
150  if (ZValidityvalue > -1 && idr->Z > ZValidityvalue) {
151  const_cast<KVIDGChIoSi_e494s*>(this)->fICode = kICODE9;
152  }
153  idr->IDquality = fICode;
154  }
155 }

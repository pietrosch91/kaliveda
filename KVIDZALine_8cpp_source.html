28 //KVIDZALine
29 //
30 //Base class for identification ridge lines corresponding to different nuclear species.
31 //
32 //Each line is named "Z=x A=y" where x is the atomic number of the corresponding element
33 //(can be set with SetZ()) and y is the mass number (SetA). If only Z is set, the mass number is
34 //calculated according to one of the mass formulae defined in KVNucleus class. The default is
35 //the default for KVNucleus; you can change it with SetMassFormula.
36 //
37 //The sorting function Compare() sorts the lines as a function of increasing Z,
38 //then increasing A (i.e. p,d,t,3He,4He,6He, etc. etc.)

206 TGraphErrors* KVIDZALine::GetLineWithWidth()
207 {
208  //Return pointer to TGraphError object which can be used to visualise the natural width

211

218  fLineWithWidth->SetFillStyle(3001);
219  fLineWithWidth->SetLineStyle(2);

222  fLineWithWidth->SetMarkerSize(0.8);
223  fLineWithWidth->SetEditable(kFALSE);
224  for (int i = 0; i < GetN(); i++) {
225  fLineWithWidth->SetPointError(i, 0., 0.5 * (GetWidth() < 16000 ? GetWidth() : 0));

241 void KVIDZALine::SetAsymWidth(Double_t d_l, Double_t d_r)
242 {
243  // Set the natural width of the line depending on its asymptotic separation
244  // a gauche (d_l) and a droite (d_r) from a neighbouring line.
245  // This method is used by KVIDGrid::CalculateLineWidths
246  //
247  // For ordinary Z/(Z,A) lines in a dE-E grid, this is just the average of
248  // the two asymptotic distances.
249 
250  SetWidth((d_l + d_r) / 2.);
251 }

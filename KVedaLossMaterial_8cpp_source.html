99 Bool_t KVedaLossMaterial::ReadRangeTable(FILE* fp)
100 {
101  // Read Z- & A-dependent range parameters for material
102  //
103  // For each material we create 4 TF1 objects:
104  // KVedaLossMaterial:[type]:Range - gives range in g/cm**2 as a function of particle energy
105  // KVedaLossMaterial:[type]:StoppingPower - gives dE/dx in MeV/(g/cm**2) as a function of particle energy
106  // KVedaLossMaterial:[type]:EnergyLoss - gives dE as a function of particle energy
107  // KVedaLossMaterial:[type]:ResidualEnergy - gives energy after material (0 if particle stops)
108  //
109  // The TF1::fNpx parameter for these functions is defined by the environment variables
110  //
111  // KVedaLoss.Range.Npx: 20 /* also used for StoppingPower */
112  // KVedaLoss.EnergyLoss.Npx: 50
113  // KVedaLoss.ResidualEnergy.Npx: 20
114  //
115 
116  char line[132];
117 
118  //look for energy limits to calculation validity
119  if (!fgets(line, 132, fp)) {
120  Warning("ReadRangeTable", "Problem reading energy limits in range table file for %s (%s)",
121  GetName(), GetType());
122  return kFALSE;
123  } else {
124  if (!strncmp(line, "COMPOUND", 8)) {
125  // material is compound. read composition for TGeoMixture.
126  if (fgets(line, 132, fp)) {}
127  int nel = atoi(line); // read number of elements
128  for (int el = 0; el < nel; el++) {
129  if (fgets(line, 132, fp)) {}
130  int z, a, w;
131  sscanf(line, "%d %d %d", &z, &a, &w);
132  AddCompoundElement(z, a, w);
133  }
134  } else if (!strncmp(line, "MIXTURE", 7)) {
135  // material is mixture. read composition for TGeoMixture.
136  if (fgets(line, 132, fp)) {}
137  int nel = atoi(line); // read number of elements
138  for (int el = 0; el < nel; el++) {
139  if (fgets(line, 132, fp)) {}
140  int z, a, nat;
141  float w;
142  sscanf(line, "%d %d %d %f", &z, &a, &nat, &w);
143  AddMixtureElement(z, a, nat, w);
144  }
145  }
146  }
147  if (!fgets(line, 132, fp)) {
148  Warning("ReadRangeTable", "Problem reading energy limits in range table file for %s (%s)",
149  GetName(), GetType());
150  return kFALSE;
151  } else {
152  while (line[0] == 'Z') {
153  Int_t z1, z2;
154  Float_t e1, e2;
155  sscanf(line, "Z = %d,%d %f < E/A < %f MeV", &z1,
156  &z2, &e1, &e2);
157  if (fgets(line, 132, fp)) {}
158  for (int i = z1; i <= z2; i++) {
159  fEmin[i - 1] = e1;
160  fEmax[i - 1] = e2;
161  }
162  }
163  }
164 
165  // get require Npx value from (user-defined) environment variables
166  Int_t my_npx = gEnv->GetValue("KVedaLoss.Range.Npx", 100);
167 
168  fRange = new TF1(Form("KVedaLossMaterial:%s:Range", GetType()), this, &KVedaLossMaterial::RangeFunc,
169  0., 1.e+03, 0, "KVedaLossMaterial", "RangeFunc");
170  fRange->SetNpx(my_npx);
171 
172  fStopping = new TF1(Form("KVedaLossMaterial:%s:StoppingPower", GetType()), this, &KVedaLossMaterial::StoppingFunc,
173  0., 1.e+03, 0, "KVedaLossMaterial", "StoppingFunc");
174  fStopping->SetNpx(my_npx);
175 
176  my_npx = gEnv->GetValue("KVedaLoss.EnergyLoss.Npx", 100);
177  fDeltaE = new TF1(Form("KVedaLossMaterial:%s:EnergyLoss", GetType()), this, &KVedaLossMaterial::DeltaEFunc,
178  0., 1.e+03, 0, "KVedaLossMaterial", "DeltaEFunc");
179  fDeltaE->SetNpx(my_npx);
180 
181  my_npx = gEnv->GetValue("KVedaLoss.ResidualEnergy.Npx", 100);
182  fEres = new TF1(Form("KVedaLossMaterial:%s:ResidualEnergy", GetType()), this, &KVedaLossMaterial::EResFunc,
183  0., 1.e+03, 0, "KVedaLossMaterial", "EResFunc");
184  fEres->SetNpx(my_npx);
185 
186  for (int count = 0; count < ZMAX_VEDALOSS; count++) {
187 
188  if (sscanf(line, "%lf %lf %lf %lf %lf %lf %lf %lf",
189  &fCoeff[count][0], &fCoeff[count][1],
190  &fCoeff[count][2], &fCoeff[count][3],
191  &fCoeff[count][4], &fCoeff[count][5],
192  &fCoeff[count][6], &fCoeff[count][7])
193  != 8) {
194  Error("ReadRangeTable", "problem reading coeffs 0-7 in range table for %s (%s)", GetName(), GetType());
195  return kFALSE;
196  }
197  if (!fgets(line, 132, fp)) {
198  Warning("ReadRangeTable", "file too short ??? %s (%s)", GetName(), GetType());
199  return kFALSE;
200  } else {
201  if (sscanf(line, "%lf %lf %lf %lf %lf %lf",
202  &fCoeff[count][8], &fCoeff[count][9],
203  &fCoeff[count][10], &fCoeff[count][11],
204  &fCoeff[count][12], &fCoeff[count][13])
205  != 6) {
206  Error("ReadRangeTable", "problem reading coeffs 8-13 in range table for %s (%s)", GetName(), GetType());
207  return kFALSE;
208  }
209  }
210  if (fNoLimits) {
211  // if we ignore nominal validity limits on incident energy, we must still use energy limits
212  // such that all range functions increase monotonically in the energy interval
213  GetRangeFunction(fCoeff[count][0], fCoeff[count][1])->SetRange(GetEminValid(fCoeff[count][0], fCoeff[count][1]), VERY_BIG_ENERGY);
214  Double_t emax = fRange->GetMaximumX() - 1;
215  emax /= fCoeff[count][1];
216  Double_t original_emax = fEmax[count];
217  // the new emax is only accepted if it is > than the nominal emax (400 or 250 AMeV),
218  // and at most 1 GeV/nucleon
219  fEmax[count] = TMath::Min(TMath::Max(original_emax, emax), 1000.);
220  // we may further reduce the upper limit to correspond to the minimum of stopping,
221  // if one exists
222  GetStoppingFunction(fCoeff[count][0], fCoeff[count][1])->SetRange(GetEminValid(fCoeff[count][0], fCoeff[count][1]), GetEmaxValid(fCoeff[count][0], fCoeff[count][1]));
223  emax = fStopping->GetMinimumX();
224  emax /= fCoeff[count][1];
225  // again, the new emax is only accepted if it is > than the nominal emax (400 or 250 AMeV),
226  // and at most 1 GeV/nucleon
227  fEmax[count] = TMath::Min(TMath::Max(original_emax, emax), 1000.);
228  //if(fEmax[count]!=original_emax) Info("ReadRangeTable", "Max. incident E for Z=%d ===> E/A = %f", count+1, fEmax[count]);
229  }
230  if (fgets(line, 132, fp)) {}
231  }
232 
233  return kTRUE;
234 }

260 Double_t KVedaLossMaterial::EResFunc(Double_t* E, Double_t*)
261 {
262  // Function parameterising the residual energy of charged particles in this material.
263  // The incident energy E[0] is given in MeV.
264  // The residual energy is calculated in MeV.
265 
266  // if range < thickness, particle stops: Eres=0
267  Double_t R0 = RangeFunc(E, nullptr);
268  if (R0 < thickness) {
269  fRangeOfLastDE = R0;
270  return 0.0;
271  } else
272  fRangeOfLastDE = thickness;
273 
274  // calculate energy after absorber - invert range function to find Eres corresponding to (R0 - thickness)
275  R0 -= thickness;
276 
277  // invert range function to get energy after absorber
278  if (fgTable->IsUseNewRangeInversion()) {
279  return static_cast<KVedaLossInverseRangeFunction*>(fInvRange[RF_Z])->GetEnergyPerNucleon(R0, riso) * RF_A;
280  }
281  return fRange->GetX(R0);

292 TF1* KVedaLossMaterial::GetRangeFunction(Int_t Z, Int_t A, Double_t isoAmat)
293 {
294  // Return function giving range (in g/cm**2) as a function of energy (in MeV) for
295  // charged particles (Z,A) in this material.
296  // If required, the isotopic mass of the material can be given.
297 
298  RF_Z = Z;
299  RF_A = A;
300  // get parameters for this Z
301  par = &fCoeff[Z - 1];
302  // set up polynomial
303  Double_t x1 = TMath::Log(0.1);
304  Double_t x2 = TMath::Log(0.2);
305  ran = 0.0;
306  for (int j = 2; j < 7; j++)
307  ran += (*par)[j + 1] * TMath::Power(x2, (Double_t)(j - 1));
308  ran += (*par)[2];
309  Double_t y2 = ran;
310  ran = 0.0;
311  for (int jj = 2; jj < 7; jj++)
312  ran += (*par)[jj + 1] * TMath::Power(x1, (Double_t)(jj - 1));
313  ran += (*par)[2];
314  Double_t y1 = ran;
315  adm = (y2 - y1) / (x2 - x1);
316  adn = (y1 - adm * x1);
317  riso = RF_A / (*par)[1];
318  if (isoAmat > 0.0) riso *= (isoAmat / fAmat);
319 
320  fRange->SetRange(0., GetEmaxValid(Z, A));
321 
322  /*
323  * New inversion of range-energy curve (if KVedaLoss::fgNewRangeInversion=kTRUE)
324  */
325  if (fgTable->IsUseNewRangeInversion()) {
326  if (!fInvRange[Z]) {
327  fInvRange[Z] = new KVedaLossInverseRangeFunction(fRange, A, riso);
328  }
329  }
330 
331  return fRange;
332 }

379 Double_t KVedaLossMaterial::RangeFunc(Double_t* E, Double_t*)
380 {
381  // Function parameterising the range of charged particles in this material.
382  // The energy E[0] is given in MeV.
383  // The range is calculated in units of g/cm**2
384 
385  eps = E[0] / RF_A;
386  dleps = TMath::Log(eps);
387  if (eps < 0.1)
388  ran = adm * dleps + adn;
389  else {
390  DLEP = dleps;
391  ran = (*par)[2] + (*par)[3] * DLEP;
392  ran += (*par)[4] * (DLEP *= dleps);
393  ran += (*par)[5] * (DLEP *= dleps);
394  ran += (*par)[6] * (DLEP *= dleps);
395  ran += (*par)[7] * (DLEP *= dleps);
396  }
397 
398  // range in g/cm**2
399  return riso * TMath::Exp(ran) * KVUnits::mg;
400 }

409 Double_t KVedaLossMaterial::StoppingFunc(Double_t* E, Double_t*)
410 {
411  // Function parameterising the stopping of charged particles in this material.
412  // The energy E[0] is given in MeV.
413  // The stopping power is calculated in units of MeV/(g/cm**2)
414 
415  eps = E[0] / RF_A;
416  dleps = TMath::Log(eps);
417  if (eps < 0.1) {
418  ran = adm * dleps + adn;
419  drande = E[0] / (riso * TMath::Exp(ran) * KVUnits::mg) / adm;
420  return drande;
421  }
422  DLEP = dleps;
423  ran = (*par)[2] + (*par)[3] * DLEP;
424  drande = (*par)[3];
425  for (int i = 4; i < 8; i++) {
426  drande += (i - 2) * (*par)[i] * DLEP;
427  ran += (*par)[i] * (DLEP *= dleps);
428  }
429  // range in g/cm**2
430  return E[0] / (riso * TMath::Exp(ran) * KVUnits::mg) / drande;
431 }

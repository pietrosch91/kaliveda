16 // Use a KVGeoNodeIterator in order to iterate over all nodes on all trajectories
17 // starting from a given node:
18 //
19 // KVGeoDetectorNode* start; // node to start iteration from
20 //
21 // KVGeoNodeIterator next_node(start);
22 // KVGeoDetectorNode* node;
23 //
24 // while( (node = next_node()) ){
25 // /* do something to each node */
26 // }
27 //
28 // Note that the start node for the iteration is included, and will be iterated
29 // over at the beginning of each trajectory. To exclude the start node from
30 // the iteration, call method ExcludeStartNode(kTRUE) before iteration begins.
31 //
32 // Default direction for iteration is FORWARDS i.e. moving towards the target.
33 // To change direction, call method IterateBackwards(kTRUE) before iteration begins.

80 void KVGeoNodeIterator::Reset(KVGeoDetectorNode* start, KVGeoDNTrajectory* trajectory)
81 {
82  // Reset the iterator in order to perform a new iteration:
83  //
84  // it.Reset(); => iterate from same node on same trajectory(ies)
85  // i.e. if you originally specified a trajectory to iterate over,
86  // it will be used again.
87  // If you want to iterate from the same node but now
88  // lifting the restriction on the trajectory, call it.ResetTrajectory()
89  //
90  // it.Reset(start); => iterate from a new node, using all trajectories
91  //
92  // it.Reset(start,trajectory) => iterate from a new node using given trajectory
93 
94  if (start == nullptr) {
95  current_node = start_node;
96  next_trajectory->Reset();
97  } else {
98  current_node = start_node = start;
99  iter_on_traj = trajectory;
100  delete next_trajectory;
101  next_trajectory = new TIter(start->GetTrajectories());
102  }
103  current_trajectory = nullptr;
104  begin_iteration_on_next_trajectory();
105 }

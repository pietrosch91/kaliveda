17 BEGIN_HTML
18 <h2>KVPartitionGenerator</h2>
19 <h4>Permet de determiner numeriquement et exactement un ensemble de partitions d'entrees
20 remplissante certaines conditions</h4>
21 END_HTML
22 ----
23 Les differentes recontrees grandeurs sont :
24 - Ztot (Zt) -> Charge/Taille totale a repartir au sein de la partition
25 - Mtot (Mt) -> Nombre de total d'element (fragment, cluster) de la partition (Zi) i=1,Mtot
26 - Zmax (Zm) -> Charge/Taille du plus gros element de la partition
27 - Zinf (Zi) -> Charge minimale des elements de la partition (Zi) Zi>=Zinf quelque soit i (tous les calculs necessitent cet argument)
28 
29 Plusieurs methodes correspondant conditions initiales possibles :
30 - BreakUsing_Ztot_Zinf_Criterion (la methode la plus generale), donnant toutes les partitions pour une taille initiale donnee
31 - BreakUsing_Ztot_Zinf_Criterion, donnant toutes les partitions pour une taille initiale donnee
32 - BreakUsing_Ztot_Zmax_Zinf_Criterion, donnant toutes les partitions pour une taille initiale donnee, avec une contrainte sur Zmax
33 - BreakUsing_Mtot_Zmax_Zinf_Criterion, donnant toutes les partitions remplissant les contraintes sur le Zmax et sur le Mtot
34 
35 Toutes ces methodes utilisent la protected methode MakePartitions() qui calcule les partitions
36 pour un couple (Ztot,Mtot) donné qui sauvegarde ces partitions dans un fichier ROOT via un arbre, contenant
37 l'information minimum :
38 - branche zt
39 - branche mt
40 - branche tabz [mt], tableau d'entier comprenant les differentes charges (Zi), trie par ordre decroissant
41 Le chemin ou les arbres sont ecrits peut etre definis via la methode SetPathForFile, par defaut les arbres seront
42 ecrits dans le repertoire courant, attention car pour des nombres importants (Ztot>100) la place necessaire devient
43 non negligeable (ex : toutes les partition de Ztot=200 et Zinf=5 -> 20GB environ)
44 
45 A chaque fois que la methode MakePartitions() est appelle par les methodes BreakUsing...(), on garde
46 en memoire l'ensemble des fichiers contenant chacun un arbre, pour a la fin du processus
47 ecrire dans un fichier ROOT, une TChain permettant de travailler rapidement et facilement sur les partitions generees
48 ----
49 
50 Un petit exemple de routine simple :
51 Cette routine produit en sortie un fichier root Partitions_Zf60_Zi1.root
52 contenant la TChain de tous les arbres ...
53 
54 void test{
55 
56  KVPartitionGenerator* pg = new KVPartitionGenerator();
57  pg->SetPathForFile("/space/bonnet/");
58  pg->BreakUsing_Ztot_Zinf_Criterion(60,1);
59 
60  Info in <BreakUsing_Ztot_Zinf_Criterion>: 966467.000000 partitions crees en 5 seconds
61  delete pg;
62 
63 }

67 Attaching file Partitions_Zf60_Zi1.root as _file0...
68 root [2] .ls
69 TFile** Partitions_Zf60_Zi1.root
70  TFile* Partitions_Zf60_Zi1.root
71  KEY: TChain PartitionTree;1
72 root [3] PartitionTree->Draw("tabz")

340 void KVPartitionGenerator::Process(void)
341 {
342  //protected method
343  //Determine toute les partitions pour un
344  //couple donné de parametres : (Ztot, Mtot, Zinf)
345  //avec :
346  // - Ztot -> taille totale de la partition
347  // - Mtot -> multiplicité de partition
348  // - Zinf -> taille minimale pour les fragments de la partition
349  //
350 
351  if (kmt == 1) {
352  tabz[0 + mshift] = kzt;
353  TreatePartition();
354  return;
355  }
356 
357  Int_t zutilise = (kmt * kzinf);
358  Int_t zdispo = kzt - zutilise;
359 
360  Int_t nb_cassure = kmt - 1;
361  KVCouple* coup[nb_cassure];
362  Int_t ncouple[nb_cassure];
363  Int_t niter[nb_cassure];
364  for (Int_t nc = 0; nc < nb_cassure; nc += 1) {
365  ncouple[nc] = 0;
366  coup[nc] = 0;
367  niter[nc] = 0;
368  }
369 
370  Int_t nc;
371  Int_t zsup = zdispo;
372  for (nc = 0; nc < nb_cassure; nc += 1) {
373 
374  if (!coup[nc]) {
375  coup[nc] = new KVCouple(zdispo, zsup, kmt - nc);
376 
377  ncouple[nc] = coup[nc]->GetNbreCouples();
378  niter[nc] = 0;
379  }
380  if (niter[nc] < ncouple[nc]) {
381 
382  zdispo = coup[nc]->GetZ2(niter[nc]);
383  zsup = coup[nc]->GetZ1(niter[nc]);
384  kcurrent[nc] = coup[nc]->GetZ1(niter[nc]);
385  kcurrent[nc + 1] = coup[nc]->GetZ2(niter[nc]);
386 
387  }
388  }
389 
390  Bool_t finish = kFALSE;
391  while (!finish) {
392 
393  nc = nb_cassure - 1;
394  while (niter[nc] < ncouple[nc]) {
395 
396  kcurrent[nc] = coup[nc]->GetZ1(niter[nc]);
397  kcurrent[nc + 1] = coup[nc]->GetZ2(niter[nc]);
398 
399  for (Int_t ii = 0; ii < kmt; ii += 1)
400  tabz[ii + mshift] = kcurrent[ii] + kzinf;
401 
402  if (to_be_checked) {
403  if (tabz[0] >= tabz[mshift]) {
404  TreatePartition();
405  }
406  } else {
407  TreatePartition();
408  }
409 
410  niter[nc] += 1;
411  }
412  delete coup[nc];
413  coup[nc] = 0;
414 
415  Int_t previous = nc - 1;
416  while (niter[previous] == (ncouple[previous] - 1) && previous >= 0) {
417 
418  delete coup[previous];
419  coup[previous] = 0;
420  previous -= 1;
421 
422  if (previous < 0) break;
423  }
424 
425  if (previous < 0) {
426  finish = kTRUE;
427  } else {
428  niter[previous] += 1;
429  zdispo = coup[previous]->GetZ2(niter[previous]);
430  zsup = coup[previous]->GetZ1(niter[previous]);
431 
432  kcurrent[previous] = coup[previous]->GetZ1(niter[previous]);
433 
434  for (Int_t ncbis = previous + 1; ncbis < nb_cassure; ncbis += 1) {
435  if (!coup[ncbis]) {
436  coup[ncbis] = new KVCouple(zdispo, zsup, kmt - ncbis);
437 
438  ncouple[ncbis] = coup[ncbis]->GetNbreCouples();
439  niter[ncbis] = 0;
440  } else { }
441  if (niter[ncbis] < ncouple[ncbis]) {
442 
443  zdispo = coup[ncbis]->GetZ2(niter[ncbis]);
444  zsup = coup[ncbis]->GetZ1(niter[ncbis]);
445  kcurrent[ncbis] = coup[ncbis]->GetZ1(niter[ncbis]);
446  kcurrent[ncbis + 1] = coup[ncbis]->GetZ2(niter[ncbis]);
447  }
448  }
449  }
450  }
451 
452  Info("Process", "ztot:%d en mtot:%d avec zinf:%d donne %lf combinaisons (total %lf)", kzt, kmt, kzinf, npar_zt_mt, npar);

454

23 <h2>KVTGIDFitter</h2>
24 <h4>Fit of E-DE functional</h4>
25 <h3>Laurent TASSAN-GOT 17/03/2001</h3>
26 This class interfaces the Fortran package written by Laurent Tassan-Got
27 to fit KVIDGrid objects using his functionals (see <a href="http://arxiv.org/abs/nucl-ex/0103004">Nucl.Instrum.Meth. B194 (2002) 503-512</a>).
28 The following is the documentation from Laurent's code.
29 <br>
30 <pre>
31  ******************
32  * Automatic fit *
33  ******************
34 
35  globede : user-friendly routine in which only the data have to be supplied.
36  The routine estimates good initial values, reasonnable limits,
37  automatically from the data and manages the whole fit procedure
38 
39  call globede(npts,zd,ad,xd,yd,ixt,ih,par,istate,irc)
40 
41  INPUT :
42  npts : [I] number of data points really used in zd, ad, xd, yd
43  zd : [R] array of Z's (npts values)
44  ad : [R] array of A's (npts values)
45  xd : [R] array of E's (npts values)
46  yd : [R] array of DE's (npts values)
47  ixt : [I] =0->basic functional <>0->extended functional
48  * For the basic formula :
49  yy = ((g*E)**(mu+1)+lambda**(mu+1)*Z**2*A**mu)**(1/(mu+1))-g*E + pdy
50  * For the extended formula :
51  yy = ((g*E)**(mu+nu+1)+(lambda*Z**alpha*A**beta)**(mu+nu+1)+
52  xi*Z**2*A**mu*(g*E)**nu)**(1/(mu+nu+1))-g*E + pdy
53 
54  * If ih=0 no non-linear light response : E=xx-pdx
55  * If ih<>0 non-linear light response included :
56  E = sqrt(h**2+2*rho*h*(1+log(1+h/rho)))
57  rho=eta*Z**2*A and h=xx-pdx
58  ih : [I] =0->no non-linear light response
59  <>0->non-linear light response included
60  OUTPUT :
61  par : [R] array of parameters resulting from the fit, according to the
62  ixt, ih combination the numbers and orders of parameters are :
63  ixt=0 ih=0 5 parameters: lambda, mu, g, pdx, pdy
64  ixt=0 ih<>0 6 parameters: lambda, mu, g, pdx, pdy, eta
65  ixt<>0 ih=0 9 parameters: lambda, alpha, beta, mu, nu,
66  xi, g, pdx, pdy
67  ixt<>0 ih<>0 10 parameters: lambda, alpha, beta, mu, nu,
68  xi, g, pdx, pdy, eta
69  istate : [I] array of status of parameters
70  =0 -> free parameter
71  =1 -> parameter constrained by the lower limit
72  =2 -> parameter constrained by the upper limit
73  =3 -> constant parameter (bl(i)=bu(i)=par(i))
74  irc : [I] return code of the fit :
75  0 -> convergence reached
76  1 -> convergence reached, but not well marked minimum
77  2 -> too many iterations, convergence not reached
78  -1 -> no identification line with at least 2 points
79  -2 -> too few data points
80  -3 -> addressing problem between Fortran and C
81 </pre>

195 void KVTGIDFitter::Fit(KVIDGraph* theGrid)
196 {
197  // Fit the grid using the functional chosen with SetType and SetLight.
198  // Status of fit after this call can be retrieved with GetFitStatus().
199 
200  // must inherit from KVIDZAGrid!
201  if (!theGrid->InheritsFrom(KVIDZAGrid::Class())) {
202  Error("Fit(KVIDGraph*)",
203  "Can only fit graphs inheriting from KVIDZAGrid");
204  fGrid = 0;
205  return;
206  }
207  fGrid = theGrid;
208  fZorA = (Int_t)theGrid->OnlyZId();
209  // prepare new parameter array and status array
210  if (fPar) delete [] fPar;
211  Int_t npar = KVTGID::GetNumberOfLTGParameters(fType, fLight);
212  fPar = new Float_t[npar];
213  if (istate) delete [] istate;
214  istate = new Int_t[npar];
215 
216  //initialise grid - this should sort lines in order of Z & A
217  theGrid->Initialize();
218 
219  //limit points according to abscissa ?
220  Bool_t with_xlimits = (fXmax > fXmin);
221 
222  // calculate total number of points to fit
223  Int_t npts = 0;
224  TIter next_id(theGrid->GetIdentifiers());
225  KVIDentifier* id;
226  while ((id = (KVIDentifier*)next_id())) {
227  if (!with_xlimits) {
228  npts += id->GetN(); // count all points
229  } else {
230  Double_t x, y;
231  for (int i = 0; i < id->GetN(); i++) {
232  id->GetPoint(i, x, y);
233  if (x >= fXmin && x <= fXmax) ++npts; // only points between [fXmin,fXmax]
234  }
235  }
236  }
237  cout << "Points to fit = " << npts << endl;
238  if (npts < 2) {
239  Error("Fit(KVIDGraph*)",
240  "Too few points for fit");
241  irc = -2;
242  return;
243  }
244 
245  // prepare and fill arrays with Z, A, X & Y
246  if (zd) delete [] zd;
247  if (ad) delete [] ad;
248  if (xd) delete [] xd;
249  if (yd) delete [] yd;
250  zd = new Float_t[npts];
251  ad = new Float_t[npts];
252  xd = new Float_t[npts];
253  yd = new Float_t[npts];
254  next_id.Reset();
255  npts = 0;
256  Double_t x, y;
257  while ((id = (KVIDentifier*)next_id())) {
258  for (int i = 0; i < id->GetN(); i++) {
259  id->GetPoint(i, x, y);
260  if ((!with_xlimits) || (x >= fXmin && x <= fXmax)) {
261  xd[npts] = (Float_t)x;
262  yd[npts] = (Float_t)y;
263  zd[npts] = id->GetZ();
264  ad[npts] = id->GetA();
265  ++npts;
266  }
267  }
268  }
269 
270  /*** FIT FONCTIONNELLE ***/
271  irc = globede_c(npts, zd, ad, xd, yd, fType, fLight, fPar, istate);
272 
273  //generate TGID corresponding to fit
274  MakeTGID();
275 }

350 void KVTGIDFitter::MakeTGID()
351 {
352  // Make a KVTGID out of fit result, if fit converged (irc<2)
353 
354  if (irc < 0 || irc > 1) return;
355  if (!fGrid) return;
356 
357  fTGID = KVTGID::MakeTGID(Form("%s_fit", fGrid->GetName()), fType, fLight, fZorA,
358  fGrid->GetMassFormula());
359  fTGID->SetLTGParameters(fPar);
360  fTGID->SetIDmin(((KVIDentifier*)fGrid->GetIdentifiers()->First())->GetZ());
361  fTGID->SetIDmax(((KVIDentifier*)fGrid->GetIdentifiers()->Last())->GetZ());
362  fTGID->SetValidRuns(fGrid->GetRuns());
363  fTGID->SetVarX(fGrid->GetVarX());
364  fTGID->SetVarY(fGrid->GetVarY());
365  fTGID->SetIDTelescopes(fGrid->GetIDTelescopes());
366 }

381 void KVTGIDFitter::FitPanel(Int_t functional_type, Bool_t with_csi_light_energy,
382  Int_t first_Z, Int_t last_Z, Double_t xmin, Double_t xmax)
383 {
384  // GUI method used to fit grid previously set with SetGrid(KVIDGraph*).
385  // functional_type = 0 (standard) or 1 (extended functional)
386  // with_csi_light_energy = kTRUE (with) or kFALSE (without CsI light-energy relation for 'X')
387  // first_Z, last_Z: set minimum & maximum Z for which fit is valid in KVTGID object
388  // resulting from fit. (default: -1, take first and last Z of grid)
389  // xmin, xmax: only fit points with abscissa between limits [xmin,xmax]
390  // (default: use all points regardless of abscissa)
391 
392  if (!fGrid) {
393  Error("FitPanel", "Set grid to fit with SetGrid(KVIDGraph*)");
394  return;
395  }
396  SetType(functional_type);
397  SetLight((Int_t)with_csi_light_energy);
398  fXmin = xmin;
399  fXmax = xmax;
400  Fit(fGrid);
401  if (irc > 1 || irc < 0) {
402  Warning("FitPanel", "Fit failed. Status of fit = %d", irc);
403  return;
404  }
405  if (!fTGID) {
406  Error("FitPanel", "Fit failed : %s", GetFitStatusString());
407  return;
408  }
409 
410  if (first_Z != -1) fTGID->SetIDmin(first_Z);
411  if (last_Z != -1) fTGID->SetIDmax(last_Z);
412 
413  if (fGrid->GetXmin() == fGrid->GetXmax()) fGrid->FindAxisLimits();
414  // generate grid representing fit
415  KVTGIDGrid* fitgr = new KVTGIDGrid(fTGID, (KVIDZAGrid*)fGrid);
416  // make fitted grid 'onlyzid' if parent grid was
417  fitgr->SetOnlyZId(fGrid->OnlyZId());
418  if (fGrid->OnlyZId()) fitgr->SetMassFormula(fGrid->GetMassFormula());
419  fitgr->Generate(fGrid->GetXmax(), fGrid->GetXmin());
420  gIDGridManager->Modified();
421 
422  if (fPad) {
423  // draw fitted grid in same pad as original
424  fPad->cd();
425  fitgr->Draw();
426  fPad->Modified();
427  fPad->Update();
428  }
429 }

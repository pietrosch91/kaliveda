123 Double_t KVChargeStateDistrib::GetQsig()
124 {
125 
126  if (!fNuc) return 0.;
127 
128  Int_t Z = fNuc->GetZ();
129  Double_t W = fNuc->GetEnergyPerNucleon();
130  Double_t Qmean0 = GetQmean0();
131  Double_t Y = Qmean0 / Z;
132  Double_t Qsig = 0.;
133 
134  if (Z >= 54 && W > 1.3) Qsig = TMath::Sqrt(Qmean0 * (0.07535 + 0.19 * Y - 0.2654 * Y * Y));
135  else Qsig = 0.5 * TMath::Sqrt(Qmean0 * (1. - TMath::Power(Y, 1.67)));
136 
137  return Qsig;
138 }

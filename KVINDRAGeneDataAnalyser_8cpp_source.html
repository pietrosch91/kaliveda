134 
135

137  cout << " ---> Trigger M>" << *it << endl;
138  all_runs.Begin();
139  while (!all_runs.End()) {
140  dbrun = (KVINDRADBRun*)GetDataSet()->GetDataBase()->GetTable("Runs")->GetRecord(all_runs.Next());
141  if (dbrun->GetTrigger() == *it) {
142  cout << " " << Form("%4d", dbrun->GetNumber());

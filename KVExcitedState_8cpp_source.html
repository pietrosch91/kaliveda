102 void KVExcitedState::computeParity()
103 {
104  TString jpi = fJPi.Data();
105  jpi.ReplaceAll("(", "");
106  jpi.ReplaceAll(")", "");
107  if (jpi.EqualTo("") || jpi.Contains("GE")) {
108  fParity = 1;
109  return;
110  }
111  if ((jpi.Contains("-")) && (jpi.Contains("+"))) {
112  if (jpi.Index("+") < jpi.Index("-")) fParity = 1;
113  else fParity = -1;
114  } else if (jpi.Contains("-")) fParity = -1;
115  else fParity = 1;
116 }

188 Double_t KVExcitedState::eval(Double_t excit)
189 {
190  double xx = excit;
191  double JJ = fSpin;
192  if (fIsOdd) JJ *= 0.5;
193  double yy = (2.*JJ + 1.) * fWidth / ((xx - fEnergy) * (xx - fEnergy) + 0.25 * fWidth * fWidth);
194  return 0.5 * yy / TMath::Pi();
195 }

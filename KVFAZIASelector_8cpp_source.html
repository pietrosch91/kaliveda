438  //had the option SetAutomaticManagementOfOutputs(kTRUE)
439  //
440  TString output_path = gSystem->Getenv("FAZIASELECTOR_OUTPUTS");
441  TString op;
442  op.Form("%s/%s/%s", output_path.Data(), kvsname, dsname);

499  slist.Data()

503  op.Data(),
504  sname.Data(),
505  op.Data(),
506  lpersyst->First()
507  );
508  }

638  //static method
639  //
640  //Get the list of output batch files
641  //with the format [kvsname]_R*.o*
642  //for example :
643  // CheckCsIIdentification_R14748-14767.o110904
644  // Extract information on CPU, Memory consuming
645  // and the list of ended/not ended files
646  //
647  // It is more adapted than the analyse_logs
648  // script when several runs are in a same job
649  //
650  // output TEnv file is produced with the following information
651  // with the following format : [kvsname].joblog
652  // example with CheckCsIIdentification.joblog
653  //
654  // Nfiles: 21
655  // Runlist: 14588-14905
656  // CpuSys(s): 4.51931
657  // CpuUser(s): 494.547
658  // CpuPerRun: 24.7273
659  // ResMem(MB): 474.09
660  // VirtMem(MB): 1870.77
661  // MissingRuns:
662  //
663  // if eraseafter=kTRUE, (kFALSE by default),
664  // log files are erased after the checking
665  //
666  KVFileReader fr;
667  KVString lfiles = gSystem->GetFromPipe(Form("ls %s_R*.o*", kvsname));
668  lfiles += "\n";
669  lfiles += gSystem->GetFromPipe(Form("ls %s.o*", kvsname));
670 
671  lfiles.Begin("\n");
672  Double_t max_sys_time = 0, max_cpu_time = 0, max_res_mem = 0, max_virt_mem = 0, max_ratio = 0;
673  Double_t sys_time = 0, cpu_time = 0, res_mem = 0, virt_mem = 0;
674  KVNumberList linit_all;
675  KVNumberList lstarted_all;
676  KVNumberList lended_all;
677  Int_t nf = 0;
678  while (!lfiles.End()) {
679  KVString file = lfiles.Next();
680  fr.OpenFileToRead(file.Data());
681 
682  KVNumberList linit;
683  KVNumberList lstarted;
684  KVNumberList lended;
685  Int_t num = -1;
686  while (fr.IsOK()) {
687  fr.ReadLine(0);
688  KVString line = fr.GetCurrentLine();
689  //Info in <KVDataAnalyser::SetRuns>: Accepted runs : 14788-14807
690  if (line.BeginsWith("Info in <KVFAZIAReconDataAnalyser::SetRuns>: Accepted runs :")) {
691  line.ReplaceAll("Info in <KVFAZIAReconDataAnalyser::SetRuns>: Accepted runs :", "");
692  linit.Add(line.Data());
693  linit_all.Add(line.Data());
694  } else if (line.BeginsWith("Info in <KVFAZIAReconDataAnalyser::SubmitTask>: treatment of run#")) {
695  line.ReplaceAll("Info in <KVFAZIAReconDataAnalyser::SubmitTask>: treatment of run#", "");
696  num = line.Atoi();
697  lstarted.Add(num);
698  lstarted_all.Add(num);
699  } else if (line.BeginsWith(Form("Info in <%s::Process>: End of file reached after ", kvsname))) {
700  lended.Add(num);
701  lended_all.Add(num);
702  } else if (line.BeginsWith("CpuSys = ")) {
703  sscanf(line.Data(), " CpuSys = %lf s. CpuUser = %lf s. ResMem = %lf MB VirtMem = %lf MB",
704  &sys_time, &cpu_time, &res_mem, &virt_mem);
705 
706  }
707  }
708  fr.CloseFile();
709  if (linit.GetNValues()) {
710  Double_t ratio = 0;
711  if (lended.GetNValues() > 0)
712  ratio = cpu_time / lended.GetNValues();
713  if (ratio > max_ratio) max_ratio = ratio;
714 
715  if (sys_time > max_sys_time) max_sys_time = sys_time;
716  if (cpu_time > max_cpu_time) max_cpu_time = cpu_time;
717  if (res_mem > max_res_mem) max_res_mem = res_mem;
718  if (virt_mem > max_virt_mem) max_virt_mem = virt_mem;
719 
720  linit.Remove(lstarted);
721  if (linit.GetNValues() > 0)
722  printf("in %s : runs not started : %s\n", file.Data(), linit.AsString());
723 
724  lstarted.Remove(lended);
725  if (lstarted.GetNValues() > 0)
726  printf("in %s : runs not ended : %s\n", file.Data(), lstarted.AsString());
727  } else {
728  printf("in %s : initial run list is empty\n", file.Data());
729  }
730  nf += 1;
731 
732  }
733 
734  if (nf > 0) {
735  TEnv env;
736  env.SetValue("Nfiles", nf);
737  env.SetValue("Runlist", linit_all.AsString());
738  env.SetValue("CpuSys(s)", max_sys_time);
739  env.SetValue("CpuUser(s)", max_cpu_time);
740  env.SetValue("CpuPerRun", max_ratio);
741  env.SetValue("ResMem(MB)", max_res_mem);
742  env.SetValue("VirtMem(MB)", max_virt_mem);
743

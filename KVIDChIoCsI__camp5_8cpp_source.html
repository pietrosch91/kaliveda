66 void KVIDChIoCsI_camp5::Initialize()
67 {
68  // Initialize telescope for current run.
69  // If there is at least 1 grid, we set fCanIdentify = kTRUE
70  // "Natural" line widths are calculated for KVIDZAGrids.
71 
72  KVIDChIoCsI::Initialize();
73  fCsIRPedestal = fCsI->GetPedestal("R");
74  fCsILPedestal = fCsI->GetPedestal("L");
75  fGGgrid = fPGgrid = 0;
76  TIter next(GetListOfIDGrids());
77  KVIDGraph* grid;
78  while ((grid = (KVIDGraph*)next())) {
79  if (!strcmp(grid->GetVarY(), "CHIO-GG")) fGGgrid = (KVIDZAGrid*)grid;
80  else if (!strcmp(grid->GetVarY(), "CHIO-PG")) fPGgrid = (KVIDZAGrid*)grid;
81  }
82  if (fGGgrid) {
83  SetBit(kReadyForID);
84  fGGgrid->Initialize();
85  if (fPGgrid) fPGgrid->Initialize();
86  } else ResetBit(kReadyForID);
87 }

99 Double_t KVIDChIoCsI_camp5::GetIDMapX(Option_t*)
100 {
101  //Calculates current X coordinate for identification.
102  //It is the CsI detector's total light output calculated from current values of 'R' and 'L'
103  //raw data without pedestal correction (because identification maps were drawn without
104  //correcting).
105  //'opt' has no effect.
106  Double_t rapide = (Double_t)fCsI->GetR() + fCsIRPedestal;
107  Double_t lente = (Double_t)fCsI->GetL() + fCsILPedestal;
108  Double_t h = (Double_t)fCsI->GetLumiereTotale(rapide, lente);
109  return h;
110 }

141 Bool_t KVIDChIoCsI_camp5::Identify(KVIdentificationResult* idr, Double_t x, Double_t y)
142 {
143  //Particle identification and code setting using identification grids.
144 
145  //perform identification in ChIo(GG) - CsI(H) map
146 
147  idr->SetIDType(GetType());
148  idr->IDattempted = kTRUE;
149 
150  Double_t cigg = (y < 0. ? GetIDMapY("GG") : y);
151  Double_t lumtot = (x < 0. ? GetIDMapX() : x);
152 
153  KVIDZAGrid* theIdentifyingGrid = 0;
154 
155  fGGgrid->Identify(lumtot, cigg, idr);
156  theIdentifyingGrid = (KVIDZAGrid*)fGGgrid;
157 
158  if (idr->IDOK && idr->Z == theIdentifyingGrid->GetZmax() && TMath::Nint(GetIDMapY("GG")) == 4095) {
159  //Gestion des saturations GG
160  //on teste l identification PG
161  if (fPGgrid) {
162  Double_t cipg = (y < 0. ? GetIDMapY("PG") : y);
163  fPGgrid->Identify(lumtot, cipg, idr);
164  //on garde l identification PG si celle ci renvoie un code
165  //de 0 a 4 ou 7
166  if (idr->Zident) {
167  theIdentifyingGrid = (KVIDZAGrid*)fPGgrid;
168  }
169  }
170  }
171  if (theIdentifyingGrid == fGGgrid) {
172  if (fGGgrid->GetQualityCode() > KVIDZAGrid::kICODE6 && fPGgrid) { //we have to try PG grid (if there is one)
173  // try Z & A identification in ChIo(PG)-CsI(H) map
174  Double_t cipg = (y < 0. ? GetIDMapY("PG") : y);
175  fPGgrid->Identify(lumtot, cipg, idr);
176  theIdentifyingGrid = (KVIDZAGrid*)fPGgrid;
177  }
178  }
179  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE8) {
180  // only if the final quality code is kICODE8 do we consider that it is
181  // worthwhile looking elsewhere. In all other cases, the particle has been
182  // "identified", even if we still don't know its Z and/or A (in this case
183  // we consider that we have established that they are unknowable).
184  return kFALSE;
185  }
186 
187  if (theIdentifyingGrid->GetQualityCode() == KVIDZAGrid::kICODE7) {
188  // if the final quality code is kICODE7 (above last line in grid) then the estimated
189  // Z is only a minimum value (Zmin)
190  idr->IDcode = kIDCode5;
191  return kTRUE;
192  }
193 
194  if (theIdentifyingGrid->GetQualityCode() > KVIDZAGrid::kICODE3 &&
195  theIdentifyingGrid->GetQualityCode() < KVIDZAGrid::kICODE7) {
196  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
197  // corresponds to a point which is inbetween the lines, i.e. noise
198  idr->IDcode = kIDCode10;
199  return kTRUE;
200  }
201 
202  // set general ID code ChIo-CsI
203  idr->IDcode = kIDCode4;
204  return kTRUE;
205 }

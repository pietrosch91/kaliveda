197 void SRB::ExtractFileInfos(TString& s, DMSFile_t* f) const
198 {
199  // extract info (name, size, modification date) from listing line
200  // such as :
201  // nief 0 Lyon 230663725 2008-12-19-15.21 run788.root.2006-10-30_08:03:15
202  TObjArray* fstats = s.Tokenize(" ");
203  f->SetName(((TObjString*)(*fstats)[5])->String().Remove(TString::kBoth, ' ').Data());
204  f->SetSize((UInt_t)((TObjString*)(*fstats)[3])->String().Remove(TString::kBoth, ' ').Atoi());
205  KVDatime mt(((TObjString*)(*fstats)[4])->String().Remove(TString::kBoth, ' ').Data(), KVDatime::kSRB);
206  f->SetModTime(mt);
207  delete fstats;
208 }

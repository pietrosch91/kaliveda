122 Bool_t KVBasicVAMOSFilter::IsTrajectoryAccepted(Float_t th_vamos, Double_t delta, Double_t thetav, Double_t phiv)
123 {
124 
125  thetav -= th_vamos;
126 
127  if (!IsInPhiVrange(phiv)) {
128 // Info("IsTrajectoryAccepted","out of PhiV range");
129  return kFALSE;
130  }
131  if (!IsInThetaVrange(thetav)) {
132 // Info("IsTrajectoryAccepted","out of ThetaV range");
133  return kFALSE;
134  }
135  if (!IsInDeltaRange(delta)) {
136 // Info("IsTrajectoryAccepted","out of delta range");
137  return kFALSE;
138  }
139 
140  if (!fCutList || (fCutList->GetThetaVamos() != th_vamos))
141  FindCutListForThetaVamos(th_vamos);
142 
143  if (!fCutList) {
144  Info("IsTrajectoryAccepted", "no cutlist of theta_vamos= %.1f", th_vamos);
145  return kFALSE;
146  }
147 
148  TIter next(fCutList);
149  TCutG* cut = NULL;
150  while ((cut = (TCutG*)next())) {
151  if (cut->IsInside(thetav, delta)) return kTRUE;
152  }
153 
154 // Info("IsTrajectoryAccepted","out of all cuts");
155 
156  return kFALSE;
157 }

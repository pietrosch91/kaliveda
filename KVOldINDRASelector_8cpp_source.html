29 //KVOldINDRASelector
30 //
31 //Basic analysis class derived from TSelector for the analysis of TChains of
32 //KVINDRAReconEvent objects (ROOT files containing INDRA data).
33 //
34 //The user's analysis class, derived from KVOldINDRASelector, must define the following
35 //methods:
36 //
37 // void InitAnalysis()
38 //
39 // Called once at the beginning of the analysis.
40 // Can be used to define histograms, TTrees, etc.
41 // Also the place to define the list of global variables to be used.
42 //
43 // void InitRun();
44 //
45 // Called at the beginning of each new run.
46 // Define code masks for "good" particles here.
47 //
48 // Bool_t Analysis();
49 //
50 // The analysis of the event.
51 // Return value: kTRUE normally, kFALSE skips to the next run (?)
52 //
53 // void EndRun();
54 //
55 // Called at the end of each run.
56 //
57 // void EndAnalysis();
58 //
59 // Called at the end of the analysis.
60 // Write histograms, TTrees to TFiles, etc.
61 //
62 //Use the static method KVOldINDRASelector::Make("MySelector") in order to generate a
63 //template for your analysis class with the correct declarations for the above
64 //methods. MySelector.C and MySelector.h will be generated.
65 //
66 //Using Options
67 //=============
68 //
69 //Using Global Variables
70 //======================
71 //Many of the most frequently used global variables have been implemented as
72 //KaliVeda classes, all of which derive from the base class KVVarGlob. The KVOldINDRASelector
73 //class provides a semi-automatic way to calculate and use these variables, via
74 //an internal list of global variables (KVGVList class). In order to use a
75 //global variable in your analysis, declare it in your InitAnalysis() method,
76 //its value will automatically be calculated for each event analysed.
77 //Note that only particles with KVINDRAReconNuc::IsOK()==kTRUE are included in the calculation.
78 //
79 //For example, let us suppose you want to calculate the isotropy ratio Riso for
80 //each event. In InitAnalysis() you should add the line:
81 //
82 // AddGV( "KVRiso", "riso" );
83 //
84 //This will create a new KVRiso global variable object with the name "riso"
85 //(used to retrieve the object from the list when you want to know its value)
86 //and add it to the internal list of global variables to be calculated.
87 //
88 //In your event-by-event Analysis() method, you can retrieve the value of
89 //the global variable using e.g.
90 //
91 // Double_t rapport = GetGV("riso")->GetValue();
92 //
93 //This gives the value of the isotropy ratio for the currently analysed event.
94 //
95 //
96 //More examples can be found in the AnalyseCamp1 analysis class.
97 //
98 //Manage histograms
99 //======================
100 //Few methods are defined to make easier the management
101 //of histograms
102 //Redefine the method CreateHistos() and create the histograms
103 //like this
104 //lhisto->Add(new TH1F(...));
105 //lhisto->Add(new TH2F(...));
106 //lhisto->Add(new TProfile(...));
107 //All histograms are stored in an "owner" KVHashList "lhisto"
108 //
109 //The method FillHisto(KVString sname,Double_t one,Double_t two,Double_t three,Double_t four)
110 //allows to fill any histogram declared as shown above
111 //you only need to give the histogram name followed by the arguments you use normally in the Fill method
112 //of TH1::Fill() and derivated
113 //
114 //Finally, to write the list of histograms in a file
115 //Just call the WriteHistoToFile method indicating the name of the file (or not)
116 //
117 //
118 //Manage trees
119 //======================
120 //Few methods are defined to make easier the management
121 //of trees
122 //Redefine the method CreateTrees() and create the histograms
123 //like this
124 //TTree* tt = 0;
125 //lhisto->Add(new TTree(...));
126 //tt = (TTree* )lhisto->Last()
127 //tt->Branch(...)
128 //tt->Branch(...)
129 //All histograms are stored in an "owner" KVHashList "ltree"
130 //
131 //The method FillTree(KVString sname="")
132 //allows to fill tree(s) according to the name passed as argument
133 //if "" is given all trees are filled
134 //
135 //Finally, to write the list of trees in a file
136 //Just call the WriteTreeToFile method indicating the name of the file (or not)

607 KVVarGlob* KVOldINDRASelector::AddGV(const Char_t* class_name,
608  const Char_t* name)
609 {
610  //Add a global variable to the list of variables for the analysis.
611  //
612  //"class_name" must be the name of a valid class inheriting from KVVarGlob, e.g. any of the default global
613  //variable classes defined as part of the standard KaliVeda package (in libKVvVarGlob.so). See
614  //"Class Reference" page on website for the available classes (listed by category under "Global Variables: ...").
615  //
616  //USER-DEFINED GLOBAL VARIABLES
617  //The user may use her own global variables in an analysis class, without having to add them to the main libraries.
618  //If the given class name is not known, it is assumed to be a user-defined class and we attempt to compile and load
619  //the class from the user's source code. For this to work, the user must:
620  //
621  // (1) add to the ROOT macro path the directory where her class's source code is kept, e.g. in $HOME/.rootrc
622  // add the following line:
623  //
624  // +Unix.*.Root.MacroPath: $(HOME)/myVarGlobs
625  //
626  // (2) for each user-defined class, add a line to $HOME/.kvrootrc to define a "plugin". E.g. for a class called MyNewVarGlob,
627  //
628  // +Plugin.KVVarGlob: MyNewVarGlob MyNewVarGlob MyNewVarGlob.cpp+ "MyNewVarGlob()"
629  //
630  // It is assumed that MyNewVarGlob.h and MyNewVarGlob.cpp will be found in $HOME/myVarGlobs (in this example).
631  //
632  //"name" is a unique name for the new global variable object which will be created and added to the internal
633  //list of global variables. This name can be used to retrieve the object (see GetGV) in the user's analysis.
634  //
635  //Returns pointer to new global variable object in case more than the usual default initialisation is necessary.
636 
637  KVVarGlob* vg = 0;
638  TClass* clas = gROOT->GetClass(class_name);
639  if (!clas) {
640  //class not in dictionary - user-defined class ? Look for plugin.
641  TPluginHandler* ph = KVBase::LoadPlugin("KVVarGlob", class_name);
642  if (!ph) {
643  //not found
644  Error("AddGV(const Char_t*,const Char_t*)",
645  "Called with class_name=%s.\nClass is unknown: not in standard libraries, and plugin (user-defined class) not found",
646  class_name);
647  return 0;
648  } else {
649  vg = (KVVarGlob*) ph->ExecPlugin(0);
650  }
651  } else if (!clas->InheritsFrom("KVVarGlob")) {
652  Error("AddGV(const Char_t*,const Char_t*)",
653  "%s is not a valid class deriving from KVVarGlob.",
654  class_name);
655  return 0;
656  } else {
657  vg = (KVVarGlob*) clas->New();
658  }
659  vg->SetName(name);
660  AddGV(vg);
661  return vg;
662 }

708  // GetEvent()->AcceptIDCodes(kIDCode2|kIDCode3|kIDCode4);
709  //
710  //at the beginning of Analysis() the variables have been calculated for all particles

712  //

717  // RecalculateGlobalVariables();
718  //

721  //
722  // GetEvent()->AcceptIDCodes(kIDCode2|kIDCode3|kIDCode4);

724  //which will be used for the next event read for processing.

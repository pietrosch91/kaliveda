53  template <typename value_type>
54  void SetValueAt(const Char_t* name, value_type value, Int_t idx)
55  {
56  //associate a parameter (define by its name) and a value
57  //if the parameter is not in the list, it is inserted at
58  //location idx in the list.
59  //if it's in the list replace its value and move it at
60  //location idx in the list.
61  KVNamedParameter* par = FindParameter(name);
62  if (par) {
63  par->Set(value);
64  if (fList.IndexOf(par) != idx) {
65  fList.GetCollection()->Remove(par);
66  fList.AddAt(par, idx);
67  }
68  } else fList.AddAt(new KVNamedParameter(name, value), idx);
69  }

16 <h2>KVVAMOSTransferMatrix</h2>
17 <h4>VAMOS's transformation matrix to map the measured coordinates at the focal plane back to the target</h4>
18 <!-- */
19 // --> END_HTML
20 //
21 // For each transmitted trajectory the parameters that are to be reconstructed
22 // are expressed as a set of four independent functions:
23 //
24 // Brho = F1( Xf, Yf, Thetaf, Phif)
25 // Theta = F2( Xf, Yf, Thetaf, Phif)
26 // Phi = F3( Xf, Yf, Thetaf, Phif)
27 // path = F4( Xf, Yf, Thetaf, Phif)
28 //
29 // The four functions "F" are a polynomial of four variables (Xf, Yf, Thetaf, Phif)
30 // which have to be stored in a KVVAMOSReconTrajectory object.
31 // As a example, the expression for Brho is:
32 //Begin_Latex( fontsize=22 )
33 // B_{#rho} = #sum_{i,j,k,l=0}^{i+j+k+l=n} C_{ijkl} X_{f}^{i} Y_{f}^{j} #theta_{f}^{k} #phi_{f}^{l}
34 //
35 //End_Latex
36 // where n is the order of the polynomial and Begin_Latex( fontsize=22 ) C_{ijkl} End_Latex are
37 // the coefficients related to the properties of the experimental system.

389 void KVVAMOSTransferMatrix::RecursiveReadOfCoef(EReconParam par, ifstream& file, UChar_t order, UShort_t* nijkl, UShort_t& NtoRead, UChar_t prev_i)
390 {
391  // Method used by ReadCoefficients(...) to read coefficients in the file
392  // having a specific format which is different of the one used to store the
393  // coefficients in the array fCoef.
394  //
395  // Inputs: par - parameter associated to the coefficents to be read
396  // (kBrho, kTheta, kPhi, kPath)
397  // file - file to be read
398  // order - function order in the recursive process
399  // nijkl - array of counters [5] to count the power of each
400  // parameter (Xf, Yf, ThetaF, PhiF) i.e. to find the
401  // index i,j,k,l for the coefficient Begin_Latex( fontsize=22 ) C_{ijkl} End_Latex
402  // For the first call of this method, each element of 'n'
403  // has to be equal zero.
404  // NtoRead - number of coefficients still to be read.
405  // prev_i - indice of the parameter at the previous order in the
406  // recursive process. To be equal to 0 for the first call.
407 
408  order--;
409 
410  // loop over : 1, Xf, Yf, ThetaF, PhiF
411  for (UChar_t i = prev_i; i < 5; i++) {
412  nijkl[i]++;
413  if (order) RecursiveReadOfCoef(par, file, order, nijkl, NtoRead, i);
414  else {
415  if (!NtoRead) {
416  Error("RecursiveReadOfCoef", "Recursive processing not ended but there is not coefficient to be read anymore!");
417  return;
418  }
419  NtoRead--;
420  file >> fCoef[par][NtoRead];
421 
422  UShort_t* ijkl = nijkl + 1;
423  f_ijkl[par][NtoRead] = Code_ijkl(ijkl[0], ijkl[1], ijkl[2], ijkl[3]);
424  }
425  nijkl[i]--;
426  }
427 }

435 void KVVAMOSTransferMatrix::CompressArrays()
436 {
437  // Reduces the size of the coefficient arrays by removing the null
438  // coefficients not used in calculations.
439 
440  if (TestBit(kIsCompressed)) {
441  Warning("CompressArrays", "Compression already done");
442  return;
443  }
444 
445  // new compressed arrays
446  UShort_t Ncoef[4];
447  Double_t* coef[4];
448  UShort_t* ijkl[4];
449 
450  //loop over each parameter
451  for (UChar_t p = 0; p < 4; p++) {
452  Ncoef[p] = 0;
453 
454  //Count the number of coefficients not null
455  for (UShort_t i = 0; i < fNCoef[p]; i++) {
456  if (fCoef[p][i] != 0) Ncoef[p]++;
457  }
458 
459  // new arrays to store only not null coefficients
460  coef[p] = new Double_t[ Ncoef[p] ];
461  ijkl[p] = new UShort_t[ Ncoef[p] ];
462  Int_t j = 0;
463  for (UShort_t i = 0; i < fNCoef[p]; i++) {
464  //only fill the new arrays with not null coefficients
465  if (fCoef[p][i] != 0) {
466  coef[p][j] = fCoef [p][i];
467  ijkl[p][j] = f_ijkl[p][i];
468  j++;
469  }
470  }
471 
472  //delete and replace the old arrays
473  delete[] fCoef [p];
474  delete[] f_ijkl[p];
475 
476  fNCoef[p] = Ncoef[p];
477  fCoef [p] = coef [p];
478  f_ijkl[p] = ijkl [p];
479  }
480 
481  SetBit(kIsCompressed);
482 }

495 Bool_t KVVAMOSTransferMatrix::ReconstructFPtoLab(KVVAMOSReconTrajectory* traj)
496 {
497  // Reconstruction of the trajectory at the target point, in the reference
498  // frame of the laboratory, from the trajectory at the focal plane.
499  // Then in the object 'traj' the focal plane parameters have to be given
500  // i.e. traj->FPparamsAreReady() has to return true.
501  //
502  // The result is stored in the 'traj' object and the method returns true
503  // if the attempt is a success.
504 
505  Bool_t ok = kFALSE;
506 
507  if (!traj->FPparamsAreReady()) {
508  Error("ReconstructFPtoLab", "Focal plane position parameters are not ready to reconstruct the trajectory");
509  return ok;
510  }
511  if (!IsReady()) {
512  Error("ReconstructFPtoLab", "Not ready, the coefficients are missing");
513  return ok;
514  }
515 
516  Double_t XYTPf[4];
517  Double_t XYTPf_phi[4];
518  Double_t* xytpf[4];
519 
520  // Xf, Yf, ThetaF and PhiF for the reconstruction of Brho, Theta and Path
521  XYTPf[0] = -1.*traj->pointFP[0] / KVUnits::m; //Xf in m
522  XYTPf[1] = 0.; //Yf in m
523  XYTPf[2] = -1.*traj->GetThetaF() * TMath::DegToRad(); //ThetaF in rad
524  XYTPf[3] = 0.; //PhiF in rad
525  xytpf[kBrho] = xytpf[kTheta] = xytpf[kPath] = XYTPf;
526 
527 
528  // Xf, Yf, ThetaF and PhiF for the reconstruction of Phi
529  XYTPf_phi[0] = XYTPf[0]; //Xf in m
530  XYTPf_phi[1] = -1.*traj->pointFP[1] / KVUnits::m; //Yf in m
531  XYTPf_phi[2] = XYTPf[2]; //ThetaF in rad
532  XYTPf_phi[3] = -1.*traj->GetPhiF() * TMath::DegToRad(); //PhiF in rad
533  xytpf[kPhi] = XYTPf_phi;
534 
535 
536  Double_t res[4] = {0, 0, 0, 0};
537  UShort_t i, j, k, l;
538  for (UChar_t p = 0; p < 4; p++) { //loop over parameters (Brho, theta, phi, path)
539 
540  //loop over the coefficients for each parameter
541  for (UShort_t c = 0; c < fNCoef[p]; c++) {
542 
543  Decode_ijkl(f_ijkl[p][c], i, j, k, l);
544 
545  res[p] += fCoef[p][c] //Cijkl
546  *TMath::Power(xytpf[p][0], i) //Xf^i
547  *TMath::Power(xytpf[p][1], j) //Yf^j
548  *TMath::Power(xytpf[p][2], k) //ThetaF^k
549  *TMath::Power(xytpf[p][3], l); //PhiF^l
550  }
551  }
552 
553  // the reconstruction is OK if it satisfies the condition:
554  // Brho > 0.001 T.m
555  // -300 < Theta < 300 mrad
556  // -300 < Phi < 300 mrad
557  // 0 < path < 2000 cm
558  if (res[kBrho] > 0.001
559  &&
560  -300. < res[kTheta] && res[kTheta] < 300.
561  &&
562  -300. < res[kPhi] && res[kPhi] < 300.
563  &&
564  0 < res[kPath] && res[kPath] < 2000.
565  ) {
566  traj->path = res[kPath]; //in cm
567  traj->Brho = res[kBrho] * gVamos->GetBrhoRef(); //in Tm
568 
569  Double_t theta = - res[kTheta] / 1000; // in rad
570  Double_t phi = - res[kPhi ] / 1000; // in rad
571 
572  Double_t X = TMath::Sin(theta) * TMath::Cos(phi);
573  Double_t Y = TMath::Sin(phi);
574  Double_t Z = TMath::Cos(theta) * TMath::Cos(phi);
575 
576  traj->dirVE.SetXYZ(X, Y, Z);
577  traj->dirLab.SetXYZ(X, Y, Z);
578  traj->dirLab.RotateY(gVamos->GetAngle()*TMath::DegToRad());
579 
580  ok = kTRUE;
581  }
582 
583  traj->SetFPtoLabAttempted();
584 
585  return ok;
586 }

10 BEGIN_HTML
11 <h2>KVImpactParameter</h2>
12 <h4>Impact parameter analysis tools</h4>
13 Use a histogram containing the distribution of some observable strongly-correlated with
14 the impact parameter in order to transform distributions of said observable into
15 impact parameter distributions (using the method of C. Cavata <I>et al.</I>, <I>Phys. Rev.</I> <B>C42</B>, 1760 (1990)), and to calculate the evolution of other quantities as a function
16 of the impact parameter.<br>
17 To use, start with a pointer to a 1-D histogram of the observable, TH1* data:<br>
18 <pre>
19 <code>
20 KVImpactParameter ip(data);
21 ip.MakeScale(npoints, bmax);
22 </code>
23 </pre>
24 The <a href="#KVImpactParameter:MakeScale">MakeScale</a> method calculates the relationship between the observable and the
25 impact parameter, using
26 END_HTML
27 BEGIN_LATEX
28 \hat{b}^{2} = \frac{\int^{\infty}_{x=X} Y(x) dx}{\int_{0}^{\infty} Y(x) dx}
29 END_LATEX
30 BEGIN_HTML
31 To obtain the impact parameter distribution for some selection of events,
32 you need the distribution of the observable for the selection, TH1* obs_sel,
33 and then use:<br>
34 <pre>
35 <code>
36 TH1* ip_dist_sel = ip.GetIPDistribution(obs_sel);
37 </code>
38 </pre>
39 To obtain the impact-parameter evolution of some quantity, take a TH2* obscor, containing
40 the bidimensional plot of the quantity as a function of the observable, and then use<br>
41 <pre>
42 <code>
43 TGraph* ip_evol = ip.GetIPEvolution(obscor, "GetMean");
44 <code>
45 </pre>
46 e.g. to have the mean value of the quantity as a function of impact parameter.
47 END_HTML
48 */

150 void KVImpactParameter::MakeScale(Int_t npoints, Double_t bmax)
151 {
152  // Calculate the relationship between the impact parameter and the observable
153  // whose distribution is contained in the histogram fData.
154  // For a given value X of the observable x, the reduced impact parameter
155  // b_hat is calculated from the distribution of x, Y(x), using the following formula:
156  /*
157  BEGIN_LATEX
158  \hat{b}^{2} = \frac{\int^{\infty}_{x=X} Y(x) dx}{\int_{0}^{\infty} Y(x) dx}
159  END_LATEX
160  */
161  // npoints = number of points for which to calculate the impact parameter.
162  // The greater the number of points, the more accurate the results.
163  // Default value is 100. Maximum value is number of bins in histogram of observable, fData.
164  // bmax is the maximum reduced impact parameter for the data.
165  // To obtain absolute values of impact parameter/cross-section,
166  // use MakeAbsoluteScale.
167 
168  if (bmax > 1.0) {
169  Warning("MakeScale", "called with bmax>1.0 - calling MakeAbsoluteScale for absolute impact parameters");
170  MakeAbsoluteScale(npoints, bmax);
171  return;
172  }
173  Bmax = bmax;
174  Smax = pow(bmax, 2.); //total reduced X-section
175  make_scale(npoints);
176 }

182 void KVImpactParameter::make_scale(Int_t npoints)
183 {
184  TH1* cumul = HM.CumulatedHisto(fData, fEvol.Data(), -1, -1, "max");
185  Int_t nbins = cumul->GetNbinsX();
186  Int_t first_bin = 1;
187  Int_t last_bin = nbins;
188  npoints = TMath::Min(nbins, npoints);
189  fIPScale = new TGraph(npoints);
190  fXSecScale = new TGraph(npoints);
191  Double_t delta_bin = 1.*(last_bin - first_bin) / (npoints - 1.);
192  Int_t bin;
193  for (int i = 0; i < npoints; i++) {
194  bin = first_bin + i * delta_bin;
195  Double_t et12 = cumul->GetBinCenter(bin);
196  Double_t b = Bmax * sqrt(cumul->GetBinContent(bin));
197  Double_t sigma = Smax * cumul->GetBinContent(bin);
198  fIPScale->SetPoint(i, et12, b);
199  fXSecScale->SetPoint(i, et12, sigma);
200  }
201  delete cumul;
202 
203  fObsTransform = new TF1("fObsTransform", this, &KVImpactParameter::BTransform,
204  fData->GetXaxis()->GetXmin(), fData->GetXaxis()->GetXmax(), 0, "KVImpactParameter", "BTransform");
205  fObsTransformXSec = new TF1("fObsTransformXSec", this, &KVImpactParameter::XTransform,
206  fData->GetXaxis()->GetXmin(), fData->GetXaxis()->GetXmax(), 0, "KVImpactParameter", "XTransform");
207 }

297 TH1* KVImpactParameter::GetIPDistribution(TH1* obs, Int_t nbinx, Option_t* norm)
298 {
299  // Transform the distribution of the observable contained in the histogram 'obs'
300  // into a distribution of the impact parameter.
301  // User's responsibility to delete histo.
302  //
303  // nbinx = number of bins in I.P. histo (default = 100)
304  // norm = "" (default) : no adjustment is made for the change in bin width due to the transformation
305  // norm = "width" : bin contents are adjusted for width change, so that the integral of the histogram
306  // contents taking into account the bin width (i.e. TH1::Integral("width")) is the same.
307 
308  if (!fObsTransform) {
309  Error("GetIPDistribution", "Call MakeScale first to calculate correspondance observable<->i.p.");
310  return 0;
311  }
312  return HM.ScaleHisto(obs, fObsTransform, 0, nbinx, -1, 0., Bmax, -1, -1, norm);
313 }

326 TGraph* KVImpactParameter::GetIPEvolution(TH2* obscor, TString moment, TString axis)
327 {
328  // obscor = pointer to histogram containing bidim correlating some observable Y with
329  // the observable used to calculate the impact parameter.
330  // Return pointer to TGraph giving evolution of any given moment of Y as a function
331  // of the impact parameter, with moment = "GetMean", "GetRMS", "GetKurtosis", etc.
332  // (methods of TH1)
333  // If the impact parameter observable is on the Y-axis of obscor, use axis="X"
334  // (by default axis="Y", i.e. we assume that the I.P. observable is on the x axis).
335 
336  if (!fObsTransform) {
337  Error("GetIPEvolution", "Call MakeScale first to calculate correspondance observable<->i.p.");
338  return 0;
339  }
340  TGraphErrors* gre = HM.GetMomentEvolution(obscor, moment, "", axis);
341  TGraph* gr = HM.ScaleGraph(gre, fObsTransform, 0);
342  delete gre;
343  return gr;
344 }

433 std::vector<Double_t> KVImpactParameter::SliceXSec(Int_t nslices, Double_t totXsec)
434 {
435  // Generate vector of observable values which can be used to select nslices
436  // of constant cross-section. Each slice will correspond to a cross-section
437  // totXsec/nslices.
438  // Note that the vector will contain (nslices-1) values
439  //
440  // Example:
441  // KVImpactParameter ip(data); // histo containing observable distribution
442  // ip.MakeAbsoluteScale(100,ip.GetIPFromXSec(data->Integral()))
443  // std::vector<Double_t> slices = ip.SliceXSec(5, Xsec);
444  // if(obs > slices[0]){
445  // // most central events (observable increases when b decreases)
446  // // with cross-section Xsec/5
447  // } else if(obs > slices[1]){
448  // // next most central events, cross-section=Xsec/5
449  // }
450  // ...
451  // } else if(obs > slices[3]){
452  // // next-to-most-peripheral events
453  // } else {
454  // // most peripheral events (obs < slices[3])
455  // }
456 
457  Double_t xsecSlice = totXsec / double(nslices);
458  std::vector<Double_t> slices(nslices - 1);
459  for (int i = 0; i < nslices - 1; ++i) {
460  slices[i] = GetObservableXSec((i + 1) * xsecSlice);
461  }
462  return slices;
463 }

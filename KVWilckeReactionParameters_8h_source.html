72  static Double_t InteractionRadius(Int_t aproj, Int_t atarg)
73  {
74  // Interaction radius [fm]
75  return MatterHalfDensityRadius(atarg) + MatterHalfDensityRadius(aproj) + 4.49 -
76  (MatterHalfDensityRadius(atarg) + MatterHalfDensityRadius(aproj)) / 6.35;
77  }
78  static Double_t r0_Wilcke(Int_t aproj, Int_t atarg)
79  {
80  // R0 [fm]
81  return InteractionRadius(aproj, atarg) / (pow(aproj, THIRD) + pow(atarg, THIRD));
82  }
83  static Double_t SharpRadius(Int_t A)
84  {
85  // Equivalent sharp radius of nucleus mass A
86  return (1.28 * pow(A, THIRD) - 0.76 + 0.8 * pow(A, -THIRD));
87  }
88  static Double_t MatterHalfDensityRadius(Int_t A)
89  {
90  // Matter half-density radius for nucleus mass A
91  // with width parameter b=1fm
92  return (SharpRadius(A) * (1. - pow(SharpRadius(A), -2.)));
93  }

96  static Double_t ChargeRadius_Myers(Int_t Z, Int_t A)
97  {
98  // Charge (Coulomb) radius for nucleus charge Z mass A
99  return r0_Myers * pow(2.*Z / (1. - 3.*epsilon_bar_Myers(Z, A)) / (1. - delta_bar_Myers(Z, A)), THIRD);
100  }
101  static Double_t BSS_V0(Int_t zp, Int_t ap, Int_t zt, Int_t at)
102  {
103  // Corrected V0 formula (2.24) from Bondorf et al. Phys. Rep. C15, 83 (1974)
104  // In Wilcke et al. there is a mistake: (RCT**1/3 + RCP**1/3)**3
105  // instead of (RCT**3+RCP**3)**1/3
106  // The Wilcke formula gives negative V0 values and an incorrect form of potential.
107  // The actual values in Wilcke correspond to the correct Bondorf formula
108  Double_t v0 = pow((zp + zt), 2.) / pow(pow(ChargeRadius_Myers(zt, at), 3.) + pow(ChargeRadius_Myers(zp, ap), 3.), THIRD);
109  v0 -= (pow(zt, 2.) / ChargeRadius_Myers(zt, at) + pow(zp, 2.) / ChargeRadius_Myers(zp, ap));
110  v0 *= 3.*e2_Wilcke / 5.;
111  return v0;
112  }
113  Double_t VC(Double_t* r, Double_t*)
114  {
115  // BSS Coulomb potential for heavy ions with separation r
116  if (*r >= RCTOTAL) {
117  return (ZP * ZT * e2_Wilcke / (*r));
118  }
119  if (*r == 0) return V0;
120  return (V0 - BSS_K * pow(*r, BSS_N));
121  }
122  Double_t ProxPot(Double_t* r, Double_t*)
123  {
124  // Nuclear proximity potential for nuclei separated by r
125  // J. Blocki, J. Randrup, W.J. Swiatecki, and C.F. Tsang
126  // Ann. Phys. (N. Y.) 105, 427-462 (1977)
127  static Double_t zeta1 = 1.2511;
128  static Double_t zeta0 = 2.54;
129  static Double_t Kprox = 0.0852;
130 
131  Double_t Phi;
132  Double_t zeta = *r - CP - CT;
133  Double_t dzeta = zeta - zeta0;
134  if (zeta <= zeta1) {
135  Phi = -0.5 * pow(dzeta, 2) - Kprox * pow(dzeta, 3);
136  } else {
137  Phi = -3.437 * exp(-zeta / 0.75);
138  }
139  return PROXFACTOR * Phi;
140  }
141  Double_t Potential(Double_t* r, Double_t*)
142  {
143  // Total heavy-ion potential (nuclear proximity+coulomb) at distance r
144  return ProxPot(r, 0) + VC(r, 0);
145  }
146  Double_t CentrifugalPotential(Double_t* x, Double_t* l)
147  {
148  // Total heavy-ion potential (nuclear proximity+coulomb+centrifugal) at distance r
149  // x[0] = r
150  // l[0] = angular momentum
151 
152  Double_t R = TMath::Max(0.1, x[0]);
153  Double_t Vcent = l[0] * (l[0] + 1.) * pow(KVNucleus::hbar, 2) / (2.*MU * mu_Wilcke * R * R);
154  return ProxPot(&x[0], 0) + VC(&x[0], 0) + Vcent;
155  }

168  Double_t GetCrossSectionFromMaxAngularMomentum(Double_t e_sur_a, Double_t lmax) const
169  {
170  // Calculate cross-section [mb] for a given beam energy [MeV/nucleon] and
171  // maximum entrance channel angular momentum [hbar]
172  return 10.*(TMath::Pi() / pow(k(e_sur_a), 2)) * pow(lmax + 0.5, 2);
173  }
174  Double_t GetMaxAngularMomentumFromCrossSection(Double_t e_sur_a, Double_t sigma) const
175  {
176  // Calculate maximum entrance channel angular momentum [hbar]
177  // for a given cross-section [mb] and beam energy [MeV/nucleon]
178  // (sharp cut-off approximatin)
179  return sqrt(sigma / (10.*(TMath::Pi() / pow(k(e_sur_a), 2)))) - 0.5;
180  }
181  Double_t GetCrossSectionFromMaxImpactParameter(Double_t bmax) const
182  {
183  // Calculate cross-section [mb] for a given
184  // maximum entrance channel impact parameter [fm]
185  return 10.*TMath::Pi() * pow(bmax, 2);
186  }
187  Double_t GetMaxImpactParameterFromCrossSection(Double_t sigma) const
188  {
189  // Calculate maximum entrance channel impact parameter [fm]
190  // for a given cross-section [mb] (sharp cut-off approximation)

231  static Double_t NLDSurfaceTensionCoefficient(Int_t Z, Int_t A)
232  {
233  // Nuclear liquid-drop surface-tension coefficient
234  // J. Blocki, J. Randrup, W.J. Swiatecki, and C.F. Tsang, Ann. Phys. (N. Y.) 105, 427 (1977)
235  Double_t I = (A - 2.*Z) / (1.*A);
236  return 0.9517 * (1. - 1.7826 * I * I);
237  }
238  static Double_t SWaveFusionBarrierRadius(Int_t zp, Int_t ap, Int_t zt, Int_t at)
239  {
240  // Fusion barrier radius for s-waves
241  Double_t zpzt = zp * zt;
242  Double_t D;
243  if (zpzt < 500) {
244  D = 0.3117 * pow(zpzt, 0.2122);
245  } else {
246  D = 1.096 + 1.391e-04 * zpzt;
247  }
248  return InteractionRadius(ap, at) - D;
249  }

256  Double_t k(Double_t e_sur_a) const
257  {
258  // Asymptotic wave number in the centre-of-mass system
259  return 0.2187 * AT * AP * pow(e_sur_a, 0.5) / (1.*AC);

261  Double_t QuarterPointAngle(Double_t* x, Double_t*) const
262  {
263  // C.M. quarter point angle as a function of laboratory E/A of projectile
264  Double_t krint = k(*x) * RINT;
265  Double_t eta = Eta(*x);
266  if (krint < 2.*eta) return TMath::Pi();
267  return 2.*asin(eta / (krint - eta));
268  }
269  Double_t ProjectileLabQP(Double_t e) const
270  {
271  // Quarter-point angle in the laboratory of projectile
272  Double_t QP = QuarterPointAngle(&e, 0);
273  return atan(sin(QP) / (cos(QP) + AP / (1.*AT)));
274  }
275  Double_t ProjectileLabEQP(Double_t e) const
276  {
277  // Laboratory energy of projectile elastically scattered at quarter-point angle
278  return AP * e * (AP * AP + 2.*AP * AT * cos(QuarterPointAngle(&e, 0)) + AT * AT) / (1.*AC * AC);
279  }

307  Double_t SigmaFus(Double_t* e_sur_a, Double_t*) const
308  {
309  // Fusion cross section in mb
310  if (*e_sur_a <= 0) return 0.;
311  Double_t e = VRB / ECM(*e_sur_a);
312  Double_t S1 = (e > 1. ? 0. : GetCrossSectionFromMaxImpactParameter(RBARRIER) * (1. - e));
313  Double_t S2 = GetCrossSectionFromMaxAngularMomentum(*e_sur_a, LCRIT);
314  Double_t S = TMath::Min(S1, S2);
315  return S;
316  }

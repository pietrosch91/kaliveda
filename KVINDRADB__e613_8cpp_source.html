137 void KVINDRADB_e613::ReadChIoPressures()
138 {
139  //Read ChIo pressures for different run ranges and enter into database.
140  //Format of file is:
141  //
142  //# some comments
143  //#which start with '#'
144  //RunRange 6001 6018
145  //2_3 50.0
146  //4_5 50.0
147  //6_7 50.0
148  //8_12 30.0
149  //13_17 30.0
150  //
151  //Pressures (of C3F8) are given in mbar).
152 
153  ifstream fin;
154  if (!OpenCalibFile("Pressures", fin)) {
155  Error("ReadChIoPressures()", "Could not open file %s",
156  GetCalibFileName("Pressures"));
157  return;
158  }
159  Info("ReadChIoPressures()", "Reading ChIo pressures parameters...");
160 
161  TString sline;
162 
163  Bool_t prev_rr = kFALSE; //was the previous line a run range indication ?
164  Bool_t read_pressure = kFALSE; // have we read any pressures recently ?
165  KVNumberList nl;
166  KVDBChIoPressures* parset = 0;
167  TList* par_list = new TList();
168  TObjArray* toks = 0;
169  //any ChIo not in list is assumed absent (pressure = 0)
170 
171  Float_t pressure[5] = { 0, 0, 0, 0, 0 };
172 
173  while (fin.good()) { // parcours du fichier
174 
175  sline.ReadLine(fin);
176  if (sline.BeginsWith("#")) {
177 
178  } else if (sline.BeginsWith("RunRange")) { // run range found
179  if (!prev_rr) { // New set of run ranges to read
180 
181  //have we just finished reading some pressures ?
182  if (read_pressure) {
183  parset = new KVDBChIoPressures(pressure);
184  GetTable("ChIo Pressures")->AddRecord(parset);
185  par_list->Add(parset);
186  LinkListToRunRange(par_list, nl);
187  par_list->Clear();
188  for (int zz = 0; zz < 5; zz++) pressure[zz] = 0.;
189  read_pressure = kFALSE;
190  }
191  }
192  toks = sline.Tokenize("\t");
193 
194  if (toks->GetEntries() != 2) {
195  Error("ReadChIoPressures", "Pb de format, il faut RunRange\tRun1-Run2 ... ");
196  return;
197  }
198 
199  prev_rr = kTRUE;
200  nl.SetList(((TObjString*)toks->At(1))->GetString().Data());
201  delete toks;
202  } // Run Range found
203  else if (fin.eof()) { //fin du fichier
204  //have we just finished reading some pressures ?
205  if (read_pressure) {
206  parset = new KVDBChIoPressures(pressure);
207  GetTable("ChIo Pressures")->AddRecord(parset);
208  par_list->Add(parset);
209  LinkListToRunRange(par_list, nl);
210  par_list->Clear();
211  for (int zz = 0; zz < 5; zz++) pressure[zz] = 0.;
212  read_pressure = kFALSE;
213  }
214  } else {
215  prev_rr = kFALSE;
216 
217  toks = sline.Tokenize("\t");
218  if (toks->GetEntries() != 2) {
219  Error("ReadChIoPressures", "Pb de format, il faut numero de la chio (ex 2_3)\tpression");
220  return;
221  }
222 
223  TString chio = ((TObjString*)toks->At(0))->String();
224  TString press = ((TObjString*)toks->At(1))->String();
225 
226  printf("%s %lf\n", chio.Data(), press.Atof());
227  delete toks;
228 
229  read_pressure = kTRUE;
230 
231  if (chio == "2_3") pressure[0] = press.Atof();
232  else if (chio == "4_5") pressure[1] = press.Atof();
233  else if (chio == "6_7") pressure[2] = press.Atof();
234  else if (chio == "8_12") pressure[3] = press.Atof();
235  else if (chio == "13_17") pressure[4] = press.Atof();
236  else {
237  printf("#%s# ne correspond a rien\n", chio.Data());
238  read_pressure = kFALSE;
239  }
240  } //line with ChIo pressure data
241  } //parcours du fichier
242  delete par_list;
243  fin.close();
244 }

258  // need description of INDRA geometry

261

423 void KVINDRADB_e613::ReadChannelVolt()
424 {
425  //Read the names of pedestal files to use for each run range, found
426  //in file with name defined by the environment variable:
427  // [dataset name].INDRADB.Pedestals: ...
428 
429  //need description of INDRA geometry
430  if (!gIndra) {
431  KVMultiDetArray::MakeMultiDetector(fDataSet.Data());
432  }
433  //gIndra exists, but has it been built ?
434  if (!gIndra->IsBuilt())
435  gIndra->Build();
436 
437  KVNumberList default_run_list;
438  default_run_list.SetMinMax(kFirstRun, kLastRun);
439  Info("ReadChannelVolt", "liste des runs par defaut %s", default_run_list.AsString());
440 
441  KVFileReader flist;
442  TString fp;
443  if (!KVBase::SearchKVFile(gDataSet->GetDataSetEnv("INDRADB.ElectronicCalibration", ""), fp, gDataSet->GetName())) {
444  Error("ReadChannelVolt", "Fichier %s, inconnu au bataillon", gDataSet->GetDataSetEnv("INDRADB.ElectronicCalibration", ""));
445  return;
446  }
447 
448  if (!flist.OpenFileToRead(fp.Data())) {
449  return;
450  }
451 
452  TEnv* env = 0;
453  TEnvRec* rec = 0;
454  KVDBParameterSet* par = 0;
455  KVDBParameterSet* par_pied = 0;
456  TObjArray* toks = 0;
457  Double_t a0, a1, a2; //parametre du polynome d ordre 2
458  Double_t gain = 0; //valeur du gain de reference
459  Double_t dum2 = -2;
460  Double_t pied = 0; //valeur du piedestal
461 
462  KVINDRADBRun* dbrun = 0;
463  KVINDRADBRun* dbpied = 0;
464  KVNameValueList ring_run;
465  while (flist.IsOK()) {
466  flist.ReadLine(NULL);
467  KVString file = flist.GetCurrentLine();
468  KVNumberList nl;
469  if (file != "" && !file.BeginsWith('#')) {
470  if (KVBase::SearchKVFile(file.Data(), fp, gDataSet->GetName())) {
471  Info("ReadChannelVolt", "Lecture de %s", fp.Data());
472  TString cal_type;
473  TString sgain = "GG";
474  if (fp.Contains("PGtoVolt")) sgain = "PG";
475  cal_type = "Channel-Volt " + sgain;
476 
477  env = new TEnv();
478  env->ReadFile(fp.Data(), kEnvAll);
479  TIter it(env->GetTable());
480  while ((rec = (TEnvRec*)it.Next())) {
481  KVNumberList nring;
482  TString srec(rec->GetName());
483  //On recupere le run reference pour lequel a ete fait la calibration
484  if (srec.BeginsWith("Ring.")) {
485  srec.ReplaceAll("Ring.", "");
486  nring.SetList(srec);
487  nring.Begin();
488  while (!nring.End()) {
489  Int_t rr = nring.Next();
490  ring_run.SetValue(Form("%d", rr), TString(rec->GetValue()).Atoi());
491  Info("ReadChannelVolt", "Couronne %d, run associee %d", rr, TString(rec->GetValue()).Atoi());
492  }
493  } else if (srec.BeginsWith("Pedestal")) {
494  srec.ReplaceAll("Pedestal.", "");
495  dbpied = GetRun(TString(rec->GetValue()).Atoi());
496  } else {
497 
498  TString spar(rec->GetValue());
499  toks = spar.Tokenize(",");
500  if (toks->GetEntries() >= 3) {
501  a0 = ((TObjString*)toks->At(1))->GetString().Atof();
502  a1 = ((TObjString*)toks->At(2))->GetString().Atof();
503  a2 = ((TObjString*)toks->At(3))->GetString().Atof();
504  par_pied = ((KVDBParameterSet*)dbpied->GetLink("Pedestals", Form("%s_%s", rec->GetName(), sgain.Data())));
505  if (par_pied)
506  pied = par_pied->GetParameter();
507  //Fit Canal-Volt realise avec soustraction piedestal
508  //chgmt de variable pour passer de (Canal-piedestal) a Canal brut
509  a0 = a0 - pied * a1 + pied * pied * a2;
510  a1 = a1 - 2 * pied * a2;
511  //a2 inchange
512  //On recupere le run de ref, pour avoir le gain associe a chaque detecteur
513  KVINDRADetector* det = (KVINDRADetector*)gIndra->GetDetector(rec->GetName());
514  if (det) {
515 
516  Int_t runref = ring_run.GetIntValue(Form("%d", det->GetRingNumber()));
517  if (!dbrun) {
518  dbrun = GetRun(runref);
519  } else if (dbrun->GetNumber() != runref) {
520  dbrun = GetRun(runref);
521  }
522  if (!dbrun) {
523  Warning("ReadChannelVolt", "Pas de run reference numero %d", runref);
524  }
525  //le gain est mis comme troisieme parametre
526  KVDBParameterSet* pargain = ((KVDBParameterSet*) dbrun->GetLink("Gains", rec->GetName()));
527  if (pargain) {
528  gain = pargain->GetParameter(0);
529  } else {
530  Info("ReadChannelVolt", "pas de gain defini pour le run %d et le detecteur %s", runref, rec->GetName());
531  }
532 
533  //Si tout est dispo on enregistre la calibration pour ce detecteur
534  //
535  par = new KVDBParameterSet(Form("%s_%s", rec->GetName(), sgain.Data()), cal_type, 5);
536  par->SetParameters(a0, a1, a2, gain, dum2);
537 
538  fChanVolt->AddRecord(par);
539  LinkRecordToRunRange(par, default_run_list);
540  }
541  } else {
542  a0 = a1 = a2 = gain = 0;
543  Warning("ReadChannelVolt", "Pb de format %s", rec->GetValue());
544  }
545  delete toks;
546  }
547  }
548  delete env;
549 
550  }
551  }
552  }
553  Info("ReadChannelVolt", "End of reading");
554

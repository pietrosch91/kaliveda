62 const KVEvent* KVFissionKinematics::Generate()
63 {
64  // Generate a fission event
65 
66  fFission.Clear();
67 
68  KVNucleus* ff1, *ff2;
69  (ff1 = fFission.AddParticle())->SetZandA(fDaughter.GetZ(), fDaughter.GetA());
70  (ff2 = fFission.AddParticle())->SetZandA(fMother.GetZ() - fDaughter.GetZ(), fMother.GetA() - fDaughter.GetA());
71 
72  Double_t vrel = ff1->GetFissionVelocity(ff2);
73  Double_t mu = ff1->GetMass() * ff2->GetMass() / (ff1->GetMass() + ff2->GetMass());
74 
75  KVPosition pp(0, 180, 0, 360);
76  TVector3 fis_axis = pp.GetRandomDirection("isotropic");
77  ff1->SetVelocity((mu * vrel / ff1->GetMass())*fis_axis);
78  ff2->SetVelocity((-mu * vrel / ff2->GetMass())*fis_axis);
79 
80  if (fMother.GetVelocity().Mag2() > 0.) {
81  // boost to frame of fissioning nucleus
82  fFission.SetFrame("lab", -fMother.GetVelocity());
83  ff1->SetVelocity(ff1->GetFrame("lab")->GetVelocity());
84  ff2->SetVelocity(ff2->GetFrame("lab")->GetVelocity());
85  }
86 
87  return &fFission;
88 }

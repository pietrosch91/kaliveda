63 // GTGanilData
64 //
65 // Read GANIL formated tapes or files.
66 // Usage can be as simple as:
67 // GTGanilData myData("filename.dat");
68 //
69 // It allow connection of data parameters values by Connect() routines
70 // that can operate with the index or the parameter name.
71 //
72 // Values of parameters are stored in an array of UShort_t (0 - 65535)
73 // Before reading a new event, every parameter is set to 65535
74 // Casting back this value to a Short_t (-32768 to 32768) gives "-1" which is the signal that the parameter's coder
75 // did not fire in the event that was read.

246 {
247  // Open the data file (could be on a tape) with a few checks.
248  // Use IsOpen() to check whether the file is opened successfully.
249  // After successfully opening the file, we call ReadParameters()
250  // to fill the parameter list from the first data buffer
251  // which *has to* be a parameter buffer.

254 
255  fStatus = acq_dev_is_alloc_c(*fDevice); // Allocation test
256  if (fStatus == ACQ_OK) fAllocated = true;
257  else {
258  gan_tape_erreur(fStatus, "test d'allocation");
259  return;
260  }
261 
262  fStatus = acq_mt_alloc_c(*fDevice); // Allocation
263  if (fStatus != ACQ_OK) {
264  if (fStatus == ACQ_NOTALLOC)
265  cout << "This tape drive is already in use !" << endl;
266  gan_tape_erreur(fStatus, "allocation");
267  return;
268  }
269 
270  fStatus = acq_mt_mount_c(*fDevice, fDensity, fBufSize);
271  if (fStatus != ACQ_OK) { // There might be some imprecisions here
272  if (fStatus == ACQ_ALREADYMOUNT)
273  cout << "The tape you want to use is already mounted" << endl;
274  gan_tape_erreur(fStatus, "mount");
275  return;
276  }
277 
278  fStatus = acq_mt_open_c(fDevice, o_read, &fLun);
279  if (fStatus != ACQ_OK) { // There might be some imprecisions here
280  gan_tape_erreur(fStatus, "open");
281  return;
282  }
283  fRunNumber = 0;
284  Char_t Date[21];
285  fStatus = acq_mt_ini_run_c(fLun, fBuffer, fBufSize, &fRunNumber,
286  fStructEvent, structEvent_size, Date);
287  fDateStart = Date;
288 
289  if (fStatus != ACQ_OK) {
290  gan_tape_erreur(fStatus, "Init Run");
291  return;

293  cout << "Run " << fRunNumber << " opened" << endl;
294  ReadParameters();
295  return;

398 bool GTGanilData::Next(void)
399 {
400  // Read an event on tape/file and put in into the event array
401  // return true until read fails
402  //
403  // If scaler buffer management (fWhatScaler) has been set to
404  // kSkipScaler, kDumpScaler or kAutoWriteScaler, then every time this method
405  // returns kTRUE a new event has been read (and perhaps 1 or more scaler buffers
406  // were read and dealt with internally).
407  // In this case, a loop over all events will look like this:
408  //
409  // while( my_gtganildata->Next() ){
410  // // new event read from file
411  // my_gtganilData->GetFiredDataParameters()->ls(); // or whatever
412  // }
413  //
414  // If fWhatScaler = kReportScaler then this method also returns kTRUE after
415  // reading a scaler buffer (no new event read), so that the user can do
416  // something with the scalers.
417  // In this case, a loop over all events including treatment of scaler buffers
418  // will look like this:
419  //
420  // while( my_gtganildata->Next() ){
421  // if( my_gtganildata->IsScalerBuffer() ){
422  // // scaler buffer read from file
423  // GTScalers* scalers = my_gtganildata->GetScalers();
424  // // N.B. GTGanilData::GetScalers() also resets the IsScalerBuffer()
425  // // flag ready for next event/buffer, so even if you don't
426  // // do anything with the scalers, you should call it
427  // }
428  // else
429  // {
430  // // new event read from file
431  // my_gtganilData->GetFiredDataParameters()->ls(); // or whatever
432  // }
433  // }
434 
435  if (fEventNumber == -1) ReadBuffer();
436  if (fStatus) return (false); // Maybe more to say here
437  TString _heads(fHeader);
438  _heads.ToUpper();
439  if (_heads.Contains("ENDRUN")) {
440  //we have reached the end of the file
441  return (false);
442  }
443  if (strcmp(fHeader, SCALER_Id) == 0) {
444  switch (fWhatScaler) {
445  case kSkipScaler:
446  return (Next()); // Skip
447  case kDumpScaler:
448  fScaler->Fill((scale*) & (fBuffer->les_donnees.cas.scale));
449  fScaler->DumpScalers();
450  return (Next());
451 
452  case kReportScaler:
453  fScaler->Fill((scale*) & (fBuffer->les_donnees.cas.scale));
454  fIsScalerBuffer = true;
455  return kTRUE;
456 
457  case kAutoWriteScaler: {
458  fScaler->Fill((scale*) & (fBuffer->les_donnees.cas.scale));
459  fScaler->DumpScalers();
460  fScalerTree->Fill();
461  return (Next());
462  }
463  break;
464  default:
465  cout << "GTGanilData::Next: unexpected value for fWhatScaler" << endl;
466  }
467  }
468  if (strcmp(fHeader, EVENTDB_Id) == 0) {
469  fIsScalerBuffer = false;
470  fIsCtrl = false;
471  fEventCount++;
472  return (ReadNextEvent());
473  }
474  else if (strcmp(fHeader, EVENTCT_Id) == 0) {
475  fIsScalerBuffer = false;
476  fIsCtrl = true;
477  fEventCount++;
478  return (ReadNextEvent());
479  }
480  else if (strcmp(fHeader, EBYEDAT_Id) == 0) {
481  fIsScalerBuffer = false;
482  fIsCtrl = false;
483  fEventCount++;
484  return (ReadNextEvent_EBYEDAT());
485  }
486  cout << "Unknown header in GTGanilData::Next:" << fHeader << endl;
487  return (true);
488 }

497 void GTGanilData::MakeTree(const TString filename, UInt_t nEvents)
498 {
499  // Automatically create and fill a tree created from ganil data.
500  // Can be use to convert a ganil tape or file to a ROOT file.
501  // The number of actual converted events is set with the nEvents parameter.
502 
503  TString theFilename;
504  if (filename == "")
505  theFilename = fFileName + ".root";
506  else theFilename = filename;
507  TFile theTreeFile(theFilename, "recreate");
508  TTree theTree("AutoTree", "Automatic filled Tree");
509  for (Int_t i = 1; i <= fDataArraySize; i++) {
510  TString parName = fDataParameters->GetParName(i);
511  TString parType = parName + "/s";
512  theTree.Branch(parName, &(fDataArray[i]), parType);
513  }
514 
515  while (Next()) {
516  theTree.Fill();
517  nEvents--;
518  if (nEvents <= 0) break;
519  }
520  theTree.Write();
521 }

534 
535  if (!fIsCtrl) {
536  fStatus = get_next_event(fBuffer, fBufSize,
537  (short*)fEventBrut, fEvbsize, &fEventNumber);

633 bool GTGanilData::EventUnravelling(CTRL_EVENT* pCtrlEvent)
634 {
635  // PRIVATE
636  // If mode is variable length event, we have to reconstruct the Data buffer
637  // from the given event.
638  // WARNING: temporary the default: we dont check that it's really the case
639  // Before reading event, all parameters have their value set to -1 (65535 - fDataArray is UShort_t)
640  // Parameters which are not fired in the event will have value -1 (65535 - cast back to Short_t for real value)
641 
642  Short_t* brutData = &(pCtrlEvent->ct_par);
643 
644  Int_t eventLength = pCtrlEvent->ct_len;
645  //Info("EventUnravelling","eventLength=%d",eventLength);
646 
647  for (Int_t i = 1; i <= fDataArraySize; i++) {
648  fDataArray[i] = (Short_t) - 1;
649  }
650 
651  for (Int_t i = 0; i < eventLength; i += 2) {
652  // cout << "Param index=" << brutData[i] << " Value=" << brutData[i+1]<<endl;
653  if (brutData[i] <= fDataArraySize && brutData[i] >= 1) {
654  fDataArray[brutData[i]] = brutData[i + 1];
655  }
656  else { // More on error handling would be cool
657  /* cout << "Index overflow : Parameter index "<<i<<" is "<<brutData[i]<<
658  " but fDataArraySize is " << fDataArraySize<<endl;
659  return(false);
660  */
661  }
662  }
663  return (true);
664 }

723  // GTGanilData::kSkipScaler : Skip scaler buffers

726  // To use this, the current TFile (i.e. gFile) must be writable.

730  // flag is set to kTRUE and the data can be retrieved by

734 
735  fWhatScaler = sc;
736 
737  if (fWhatScaler == kAutoWriteScaler) {
738  if (gFile && gFile->IsWritable()) {
739  fScalerTree = new TTree("Scalers", "Automatic filled scalers");
740  fScalerTree->Branch("scalers", fScaler, 8000, 99);

744  cout << "You must open a writable TFile before calling SetScalerBuffersManagement(kAutoWriteScaler)" << endl;

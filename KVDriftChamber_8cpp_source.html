400 TH1F* KVDriftChamber::GetQrawHisto(Int_t c_num)
401 {
402  // Retruns the X position histogram with raw charges (Q) for
403  // a given chamber. c_num=[1,2] is the number of this chamber.
404  // If the histogram is empty then no acquisition parameter
405  // was fired.
406 
407  Int_t i = c_num - 1;
408  if ((i < 0) || (i > 1)) return NULL;
409  if (!fQ[0][i]) return NULL;
410  if (fQ[0][i]->GetEntries()) return fQ[0][i];
411 
412  fQ[0][i]->Reset();
413  TIter next(GetACQParamList());
414 
415  KVACQParam* par = NULL;
416  while ((par = (KVACQParam*)next())) {
417 
418  if ((par->GetLabel()[1] != Char_t('1' + i)) || !par->Fired("P")) continue;
419 
420  Int_t num = (c_num % 2 ? par->GetNumber() : _DC_N_STRIPS_ + 1 - par->GetNumber());
421  fQ[0][i]->SetBinContent(num, (Double_t)par->GetCoderData());
422  }

436  // for a given chamber. c_num=[1,2] is the number of this chamber.
437

439  if ((i < 0) || (i > 1)) return NULL;
440  if (!fQ[1][i]) return NULL;
441  if (fQ[1][i]->GetEntries()) return fQ[1][i];

509 Float_t KVDriftChamber::CalculateQThreshold(Int_t c_num)
510 {
511  // Calculate the noise in order to remove it on strips for X
512  // position of a given chamber. c_num=[1,2] is the number of this
513  // chamber.
514 
515  //Study the noise and the maximum value for the charge.
516  //At this stage the quantities noise_*** contain also the signal!
517  Float_t QTmp, noise_mean, noise_var, noise_stdev, noise_sum;
518  Float_t Q_max;
519  Q_max = noise_mean = noise_var = noise_sum = noise_stdev = 0.;
520 
521  UShort_t Q_max_Nr;
522  Q_max_Nr = 1000 ;
523 
524  TH1F* hQcal = GetQHisto(c_num);
525  if (!hQcal) return 0;
526 
527  for (Int_t j = 1; j <= hQcal->GetNbinsX(); j++) { //loop over strips
528  QTmp = hQcal->GetBinContent(j);
529  if (QTmp > 0) {
530  noise_mean += QTmp;
531  noise_var += QTmp * QTmp;
532  noise_sum += 1.;
533  }
534  if (QTmp > Q_max) {
535  Q_max = QTmp;
536  Q_max_Nr = j;
537  }
538  }
539 
540  //Study the surrounding of the maximum charge
541  Float_t max_mean, max_var, max_sum;
542  max_mean = max_var = max_sum = 0.;
543  for (Int_t j = -Border; j <= Border; j++) {
544  Int_t k = Q_max_Nr + j;
545  if ((0 < k) && (k <= hQcal->GetNbinsX())) {
546  QTmp = hQcal->GetBinContent(k);
547  if (QTmp > 0) {
548  max_mean += QTmp;
549  max_var += QTmp * QTmp;
550  max_sum += 1.;
551  }
552  }
553  }
554  //The signal of the maximum charge is removed from the quantities noise_***
555  noise_mean -= max_mean;
556  noise_var -= max_var;
557  noise_sum -= max_sum;
558  //Calculate statistical quntities related to the noise and fix the threshold
559  //as the mean of the noise + 1 time its standart deviation
560  if (noise_sum < 2. || noise_var == 0) {
561  noise_mean = noise_var = 0;
562  } else {
563  noise_mean /= noise_sum;
564  noise_var = noise_var / noise_sum - noise_mean * noise_mean;
565  noise_stdev = TMath::Sqrt(noise_var);
566  }
567  Float_t threshold = noise_mean + noise_stdev;
568 
569  return threshold;
570 }

627 {
628  // This method allows to set by hand the raw positions of the drift chamber.
629  // X1X2Yf is an array of 3 elements: X1 is the raw X position for the first row
630  // X2 is the raw X position for the second row
631  // Y is the raw Y position for the drift chamber
632  // Pad1 is the pad with the maximum Q for row 1 (used for SECHS method)
633  // Pad2 is the pad with the maximum Q for row 2 (used for SECHS method)
634  //
635  // If the raw position is set successfully then fkRawPosSet==kTRUE and the GetRawPosition()
636  // method will now return the values set by the user (in stead of calculating them
637  // using the Q histograms), and this as long as fkRawPosSet==kTRUE.
638  // To reset the raw position, the user should use the method ResetRawPosition().

642  // not be read anymore for calculating the raw positions until the next
643  // call to Initialize().
644  //
645  // It is the user responsability to verify what method is used for the raw position
646  // measurement (either SECHS method or Mean/RMS method) (see SetSECHSReconstruction()).
647 
648  if (X1_X2_Y_Pad1_Pad2) {
649  fRawPosX[0] = 0.5 * (X1_X2_Y_Pad1_Pad2[0] + X1_X2_Y_Pad1_Pad2[1]); //to be modified if we want to include statistcs...

651  fRawPosX[2] = X1_X2_Y_Pad1_Pad2[1];
652  fRawPosY = X1_X2_Y_Pad1_Pad2[2];

698 Double_t KVDriftChamber::GetRawPosition(Char_t dir, Int_t num)
699 {
700  // Returns the raw position in X/Y with respect to the center of the detector. The value is deduced from
701  // the histogram representing the calibrated charge versus strip number for X direction
702  // or returns the electron drift time for Y position.
703  // Argument 'num' is used to specified the X position to be returned:
704  // dir = 'X' and num = 0 --> return the mean value between X1 and X2
705  // dir = 'X' and num = 1 --> return X1
706  // dir = 'X' and num = 2 --> return X2
707  //
708  //
709  // Two methods can be used for X raw position: the Hyperblic Secant Squared (SECHS) method (used by default)
710  // or the Mean and RMS estimation method. The method used for the X position measurement can be changed
711  // by calling SetSECHSReconstructionX(Bool_t kSECHS)
712  //
713  // SECHS method ( see T.Roger et al. NIMA 638 (2001) and Lau et al. NIMA 366 (1995) ):
714  // We compute here the charge profile's centroid, seen on a strip of width w=StripWidth()/2.
715  // This means the avalanche centroid's position evolves in [0.,w/2.], without considering for the moment
716  // the number of the strip.
717  // To evaluate the centroid, we only consider the charge of the most significant strip and the charges of the
718  // two neighbooring ones (as the spatial resolution is expected to be degraded by the use of more than 3 strips).
719  //
720  // Mean and RMS method:
721  // We evaluate here the Mean and RMS values of the histogram representing the calibrated charge versus strip number.
722  // The two cathode plans are offset by half a strip to reduce the non linearity of the position measurement
723  // in between the strips. The center of the detector is by default at the middle of the strip 32 of the
724  // range 2 and between strips 32 and 33 of the range 1.
725  //
726  //
727  // You can apply an X offset to change the strip position with respect to
728  // the center of the detector by calling SetXOffsets.
729  //
730  //
731  // For Y raw position we evaluate the electron-drift-time canal.
732  //
733  //
734  // In any case, if the raw position in not calculable, the value -666 will be returned.
735 
736  Int_t idx = IDX(dir);
737 
738  switch (idx) {
739 
740  case 0: { // for X direction
741 
742  //If X raw position was not set by hand
743  if (!IsRawPosSetByHand()) {
744  if (num < 0 || num > 2) num = 0;
745  if (fERawPosX[ num ] > -500) return fRawPosX[ num ];
746 
747  fRawPosX [ 0 ] = 0;
748  fERawPosX[ 0 ] = 0.;
749  Double_t sum = 0.;
750 
751  for (Int_t c = 1; c < 3; c++) { //loop over both rows of the chamber
752  fRawPosX [ c ] = -666;
753  fERawPosX[ c ] = -1;
754  fPadMax[ c - 1 ] = -666;
755 
756  TH1F* hh = GetCleanQHisto(c);
757  if (!hh) continue;
758 
759  if (!hh->GetEntries()) continue;
760 
762 
763  Int_t binMax = hh->GetMaximumBin();
764  Int_t min;
765  for (min = binMax; min > 1; min--) {
766  if (hh->GetBinContent(min) <= 0.) {
767  min++;
768  break;
769  }
770  Double_t deltaQ = hh->GetBinContent(min) - hh->GetBinContent(min - 1);
771  if (deltaQ < 0) break;
772  }
773  Int_t max;
774  for (max = binMax; max < hh->GetNbinsX(); max++) {
775  if (hh->GetBinContent(max) <= 0.) {
776  max--;
777  break;
778  }
779  Double_t deltaQ = hh->GetBinContent(max) - hh->GetBinContent(max + 1);
780  if (deltaQ < 0) break;
781  }
782 
783  if ((max - min + 1) < fNstripsOK) continue;
784 
785  if (IsSECHSReconstructionX()) {
786  // Hyperbolic secant squared (SECHS) method:
787  // by convention Q0 is the charge of the most significant strip
788  // and Q+ and Q- are the charges of the left and right neighboring pads,
789  // and ww is the strip(pad) witdh.
790  fPadMax[c - 1] = hh->GetBinCenter(binMax); //the bin center is the number of the pad
791  Float_t* QQ = new Float_t[fNstripsOK];
792  QQ[0] = hh->GetBinContent(binMax - 1); //Q-
793  QQ[1] = hh->GetBinContent(binMax); //Q0
794  QQ[2] = hh->GetBinContent(binMax + 1); //Q+
795  Double_t ww = GetStripWidth(); //width of a pad in cm
796 
797  Double_t a2 = 0.5 * (TMath::Sqrt(QQ[1] / QQ[2]) + TMath::Sqrt(QQ[1] / QQ[0]));
798  Double_t a1 = (TMath::Sqrt(QQ[1] / QQ[2]) - TMath::Sqrt(QQ[1] / QQ[0])) / (2.*TMath::SinH(a2));
799  Double_t delta = ww / 2. * TMath::Log((1 + a1) / (1 - a1)) / (TMath::Log(a2 + TMath::Sqrt(a2 * a2 - 1)));
800 
801  fRawPosX [ c ] = delta;
802  fERawPosX[ c ] = 0.; //to be modified
803 
804  delete[] QQ;
805  } else { //Meand and RMS method
806  // The two cathode plans are offset by half a strip to
807  // reduce the non linearity of the position measurement
808  // in between the strips. The center of the detector
809  // is at the middle of the strip 32 of the range 2
810  // and between strips 32 and 33 of the range 1.
811  hh->GetXaxis()->SetRange(min, max);
812 
813  Double_t intgl = hh->Integral();
814  sum += intgl;
815 
816  fPadMax[ c - 1 ] = hh->GetBinCenter(binMax);
817  fRawPosX [ c ] = hh->GetMean() - (c % 2 ? 32.5 + fOffsetX[0] : 32 + fOffsetX[1]);
818  fERawPosX[ c ] = hh->GetMeanError();
819  fRawPosX [ 0 ] += intgl * fRawPosX [ c ];
820  fERawPosX[ 0 ] += intgl * fERawPosX[ c ];
821 
822  hh->GetXaxis()->SetRange();
823 
824  if (sum) {
825  fRawPosX [ 0 ] /= sum;
826  fERawPosX[ 0 ] /= sum;
827  } else {
828  fRawPosX [ 0 ] = -666;
829  fERawPosX[ 0 ] = -1;
830  }
831  }
832  }
833  return fRawPosX[ num ];
834  }
835 
836  //If X raw position set by hand
837  else return fRawPosX[ num ];
838 
839  }//end case 0
840 
841 
842  case 1: { // for Y direction returns the acq parameter of TFIL
843 
844  //If raw Y position was not set by hand
845  if (!IsRawPosSetByHand()) {
846  fRawPosY = (fTfilPar && fTfilPar->Fired("P") ? fTfilPar->GetData() : -666);
847  fERawPosY = 0.;
848 
849  return fRawPosY;
850  }
851 
852  //If raw Y position was set by hand
853  else return fRawPosY;
854 
855  }//end case 1
856  }
857 
858  return -1;
859 }

917 UChar_t KVDriftChamber::GetPosition(Double_t* XYZf, Char_t dir, Int_t num)
918 {
919  // Get calibrated and deviation-corrected positions Xf, Yf and Zf (in cm)
920  // in the focal plane reference frame from the raw positions in channel
921  // obtained with GetRawPosition(...). The argument 'XYZf' has to be an
922  // array of size 3 i.e. XYZf[3].
923  // The 3 first bits of the returned UChar_t value give an information about
924  // the coordinates well determined, calibrated and corrected. For example is the Y
925  // coordinate, with the indice 1, is good then the bit 1 is set to 1.
926  //
927  // The two cathode plans are offset by half a strip to reduce the non linearity of the position measurement
928  // in between the strips. The center of the detector is by default at the middle of the strip 32 of the
929  // range 2 and between strips 32 and 33 of the range 1.
930  //
931  // The way X position is calculated depends on the method used for raw position.
932  //
933  // If the Hyperblic Secant Squared (SECHS) method was used (by default), then
934  // the bin with the maximum charge, the strip width and the raw position
935  // (in cm = disparity with respect to the pad center) are used to fine the position.
936  // The offsets are taken into account.
937  //
938  // Else if the Mean and RMS method was used then we compute the real distance using
939  // the strip width and the found raw position.
940 
941  // Nothing is done if there is no calibrator for the position
942  if (!IsPositionCalibrated()) return 0;
943 
944  // num will contain info about dir and num
945  // X1 --> num=1
946  // X2 --> num=2
947  // else num = 0
948  num = (dir == 'X' && (num == 1 || num == 2) ? num : 0);
949 
950  UChar_t rvalue = 0; // returned value;
951 
952  Double_t Xraw = GetRawPosition('X', num);
953  Double_t Yraw = GetRawPosition('Y');
954 
955  if (Xraw >= -500) { // Calibrate X
956  if (IsSECHSReconstructionX()) {
957  Int_t pad = GetPadMax(num);
958  if (num == 1) XYZf[0] = (pad - 32.5) * GetStripWidth() + Xraw + fOffsetX[0];
959  if (num == 2) XYZf[0] = (pad - 32.) * GetStripWidth() + Xraw + fOffsetX[1];
960  } else {
961  XYZf[0] = Xraw * GetStripWidth();
962  }
963  rvalue += 1;
964  }
965  if ((Yraw >= -500) && GetDriftTimeCalibrator()) { //Calibrate Y
966  XYZf[1] = GetDriftTimeCalibrator()->Compute(Yraw) * GetDriftVelocity() / 1000.; // the velocity is in cm/um
967  rvalue += 2;
968  }
969 
970  // Set the Z offset
971  XYZf[2] = fOffsetZ[ num ];
972 
973  // Set the coordinate in the focal plane frame of reference
974  if (rvalue && ActiveVolumeToFocal(XYZf, XYZf)) {
975  if (!(rvalue & 1)) XYZf[0] = -666;
976  else if (!(rvalue & 2)) XYZf[1] = -666;
977  else rvalue += 4;
978  return rvalue;
979  }
980 
981  // No coordinates are OK
982  XYZf[0] = XYZf[1] = XYZf[2] = -666;
983  return 0;
984 }

92 void KVIVReconDataAnalyser::preAnalysis()
93 {
94  // Read and set raw data for the current reconstructed event.
95  // Initialize VAMOS spectrometer before starting the analysis
96  // of the new event.
97  //
98  // This method overwrites the same method of the mother class
99  // KVINDRAReconDataAnalyser where gIndra is replaced by gMultiDetArray
100  // in order to read and set raw data of VAMOS as well as the ones of
101  // INDRA. Indeed the ACQ parameter list of gMultiDetArray contains the
102  // parameters of both multidetectors. (for more information see the
103  // class KVINDRA_VAMOS)
104 
105  if (!theRawData) return;
106  // all recon events are numbered 1, 2, ... : therefore entry number is N-1
107  Long64_t rawEntry = GetRawEntryNumber();
108 
109  gMultiDetArray->GetACQParams()->R__FOR_EACH(KVACQParam, Clear)();
110 
111  theRawData->GetEntry(rawEntry);
112  for (int i = 0; i < NbParFired; i++) {
113  KVACQParam* par = gMultiDetArray->GetACQParam((*parList)[ParNum[i]]->GetName());
114  if (par) {
115  par->SetData(ParVal[i]);
116  }
117  }
118 
119  if (gVamos) gVamos->Initialize();
120 }

116 {
117  //Return raw PG channel number corresponding to a given detector signal in volts
118  //
119  //Any change in the coder pedestal between the current run and the effective pedestal

121  //

124

275 Double_t KVSilicon::GetVoltsFromCanalPG(Double_t chan)
276 {
277  //Return calibrated detector signal in Volts calculated from PG channel number.
278  //If "chan" is not given, the value of the "PG" acquisition parameter read from
279  //data for this detector is used to calculate the signal.
280  //If the PG parameter is not present (=-1) or no calib we return 0.
281  //Any change in the coder pedestal between the current run and the effective pedestal
282  //of the channel-volt calibration (GetCanal(V=0)) is automatically corrected for.
283 
284  if (!fChVoltPG || !fChVoltPG->GetStatus())
285  return 0;
286 
287  if (!chan) {
288  chan = GetPG();
289  }
290  if (chan < -0.5)
291  return 0.; //PG parameter absent
292  //correct for pedestal drift
293  chan = chan - (Double_t) GetPedestal("PG") + fChVoltPG->Invert(0.);
294  return (fChVoltPG->Compute(chan));
295 }

340 Double_t KVSilicon::GetVolts()
341 {
342  //Returns Volts for this detector calculated from current PG coder values.
343  //We only use PG, as the two channel-volt calibrations do not coincide and so the passage
344  //from GG to PG produces a discontinuity (unless only GG calibration is available, then
345  //we convert PG to GG and use the GG calibration)
346  //Returns 0 if no calibration available
347 
348  if (fChVoltPG && fChVoltPG->GetStatus()) {
349  return GetVoltsFromCanalPG();
350  } else if (fChVoltGG && fChVoltGG->GetStatus()) {
351  return GetVoltsFromCanalGG(GetGGfromPG());
352  }
353 
354  return 0;

407 Double_t KVSilicon::GetEnergy()
408 {
409  //Redefinition of KVDetector::GetEnergy().
410  //If energy lost in active layer is already set (e.g. by calculation of energy loss
411  //of charged particles), return its value.
412  //If not, we calculate it and set it using the values read from coders (if fired)
413  //and the channel-volts/volts-energy calibrations, if present
414  //
415  //Returns 0 if (i) no calibration present (ii) calibration present but no data in acquisition parameters
416 
417  //fELoss already set, return its value
418  Double_t ELoss = KVDetector::GetEnergy();
419  if (IsSimMode()) return ELoss; // in simulation mode, return calculated energy loss in active layer
420  if (ELoss > 0) return ELoss;
421  ELoss = GetCalibratedEnergy();
422  if (ELoss < 0) ELoss = 0;
423  SetEnergy(ELoss);
424  return ELoss;

465 void KVSilicon::SetMoultonPHDParameters(Double_t a_1, Double_t a_2, Double_t b_1, Double_t b_2)
466 {
467  //Sets parameters of Moulton formula used to calculate PHD for particles
468  //stopping in this detector. The parameters are as in the following:
469  //
470  // log_10(PHD) = b(Z) + a(Z)*log_10(E)
471  //
472  // with a(Z) = a_1*(Z**2/1000) + a_2
473  // b(Z) = b_1*(100/Z) + b_2
474  // E = energy lost by particle
475  //
476  //See class KVPulseHeightDefect
477 
478  if (fPHD) {
479  fPHD->SetParameters(a_1, a_2, b_1, b_2);
480  fPHD->SetStatus(kTRUE);
481  }
482 }

492 Short_t KVSilicon::GetCalcACQParam(KVACQParam* ACQ, Double_t ECalc) const
493 {
494  // Calculates & returns value of given acquisition parameter corresponding to
495  // given calculated energy loss in the detector

502  return -1;

519 TF1* KVSilicon::GetELossFunction(Int_t Z, Int_t A)
520 {
521  // Overrides KVDetector::GetELossFunction
522  // If the pulse height deficit (PHD) has been set for this detector,
523  // we return an energy loss function which takes into account the PHD,
524  // i.e. for an incident energy E we calculate
525  //
526  // dEphd(E,Z,A) = dE(E,Z,A) - PHD(dE,Z)
527  //
528  // If no PHD is set, we return the usual KVDetector::GetELossFunction
529  // which calculates dE(E,Z,A)
530 
531  if (fPHD && fPHD->GetStatus()) return fPHD->GetELossFunction(Z, A);
532 
533  return KVDetector::GetELossFunction(Z, A);
534 }

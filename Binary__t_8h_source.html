173 template < class T > void Binary_t < T >::Set(const Char_t* val)
174 {
175  //Set value from string containing binary number
176  //If their are more bits in string than number can hold, we only use
177  //the fNBits bits from bit 0 (LSB) to bit fNBits-1
178 
179  fVal = 0;
180  Int_t nbits = (Int_t) strlen(val);
181  Int_t ibit = 0;
182  Int_t idx_min = nbits - fNBits;
183  idx_min = idx_min * (idx_min > 0);
184  for (Int_t i = nbits - 1; i >= idx_min; i--) {
185  if (val[i] - '0') { //if a '1' is present here...
186  fVal += (T)(1 << ibit);
187  }
188  ibit++;
189  }
190 }

192 template < class T > const Char_t* Binary_t < T >::String(UChar_t nbits)
193 {
194  //Return string containing binary representation of number
195  //By default, shortest possible representation is returned.
196  //If a default number of bits has been set with SetNBits,
197  //or if the number of bits is given as argument, the number will be represented using
198  //that many bits. If this is more than are necessary, we pad with leading zeroes.
199  //If there are not enough bits to represent the whole of the number, only the bits from
200  //0 to (nbits-1) are shown.
201  //
202  //e.g. if we have Binary_t b="111011" :
203  //b.String(8) will give "00111011" (padded with 2 leading zeroes)
204  //b.String(3) will give "011" (bits 0 to 2 only, i.e. first 3 bits starting from bit 0, which is LSB)
205  UChar_t setbits = 0;
206  if (!Value()) {
207  fDumString = "0";
208  setbits = 1;
209  } else {
210  fDumString = "";
211  T cont = Value();
212  while (cont) {
213  if (cont % 2)
214  fDumString.Prepend('1');
215  else
216  fDumString.Prepend('0');
217  setbits++;
218  cont /= 2;
219  }
220  }
221  //work out number of leading zeroes to add (if any)
222  //if number is too short for required number of bits, lzos>0 and equals number of zeroes to pad with
223  //if number is too long for required number of bits, lzos<0
224  Int_t lzos =
225  (nbits ? (Int_t) nbits -
226  (Int_t) setbits : (GetNBits() ? (Int_t) GetNBits() -
227  (Int_t) setbits : 0));
228  if (lzos > 0) {
229  //Add 'lzos' leading zeroes
230  fDumString.Prepend('0', lzos);
231  } else if (lzos < 0) {
232  //cut off (-lzos) most significant bits
233  fDumString.Remove(0, -lzos);
234  }
235  return fDumString.Data();
236 }

445 template < class T > void Binary_t < T >::WriteSubvalue(const T val,
446  UChar_t msb,
447  UChar_t nbits)
448 {
449  //Write the value 'val' in the 'nbits' bits from bit 'msb' (Most Significant Bit) to
450  //bit (msb-nbits+1).
451  //Warning if the following conditions are not respected:
452  //1 <= nbits <= fNBits 0 <= msb <= (fNBits-1) (msb-nbits+1)>=0
453 
454  //transform everything to 'int' for arithmetic to work normally!
455  Int_t _fNBits, _msb, _nbits;
456  _fNBits = (Int_t) fNBits;
457  _msb = (Int_t) msb;
458  _nbits = (Int_t) nbits;
459  if (_nbits < 1 || _nbits > _fNBits || _msb > (_fNBits - 1)
460  || (_msb - _nbits + 1) < 0) {
461  std::cout << "Error in <Binary_t<T>::WriteSubvalue> : ";
462  if (_nbits < 1)
463  std::cout << "nbits<1";
464  if (_nbits > _fNBits)
465  std::cout << "nbits(" << _nbits << ")>fNBits(" << _fNBits << ")";
466  if (_msb > (_fNBits - 1))
467  std::cout << "msb(" << _msb << ") > fNBits-1(" << _fNBits - 1 << ")";
468  if ((_msb - _nbits + 1) < 0)
469  std::cout << "(msb-nbits+1) < 0 : msb=" << _msb << " nbits=" << _nbits;
470  std::cout << std::endl;
471  return;
472  }
473  //get binary representation of 'val' with nbits
474  Binary_t < T > tmp(val);
475  //get full fNBits bit representation
476  String(fNBits); //NB result is in fDumString !!
477  //replace nbits bits from bit msb to msb-nbits+1 with binary representation of 'val'
478  fDumString.Replace((_fNBits - _msb - 1), nbits, tmp.String(nbits));
479  //set new value
480  Set(fDumString.Data());
481 }

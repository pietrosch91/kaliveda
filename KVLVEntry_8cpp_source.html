26 <h3>Example of use</h3>
27 KVListView takes a list of objects (TObject or derived class) and presents them
28 as a list of items with different data for each object presented in separate columns.
29 The user specifies the base-class of the objects to display, the number of columns,
30 and the data to be presented in each column. The example in the figure shown above
31 was generated by the following:
32 <pre>
33  lvRuns = new KVListView(<a href="KVDBRun.html">KVDBRun</a>::Class(), cfRuns, 500, 250); //create list view for KVDBRun objects. 'cfRuns' is pointer to GUI frame containing the list.
34  lvRuns->SetDataColumns(4); //set number of columns to 4
35 
36  lvRuns->SetDataColumn(0, "Run", "GetNumber"); //by default, the data presented in a column with title "Toto" will be retrieved
37  lvRuns->SetDataColumn(1, "Events", "", kTextRight); //from each object by calling the "GetToto" method of the class given to the KVListView ctor.
38  lvRuns->SetDataColumn(2, "Date", "GetDatime"); //in case the "getter" method has a non-standard name, it can be given explicitly.
39 
40  lvRuns->GetDataColumn(2)-><a href="KVLVColumnData.html#KVLVColumnData:SetIsDateTime">SetIsDateTime()</a>;//some special treatment is afforded to date/time data. see method doc.
41  lvRuns->SetDataColumn(3, "Comments", "", kTextLeft);// specify text alignment for data
42 
43 
44  lvRuns->ActivateSortButtons();//when clicking a column header, the list is sorted according to that column's data.
45  ... //clicking a second time the same column sorts objects in the opposite sense.
46 </pre>
47 To display objects, put them in a TList and call
48 <pre>
49  lvRuns->Display( pointer_to_TList );
50 </pre>
51 Items in the list can be selected using either:
52  <ul>
53  <li>single left click (single object selection),</li>
54  <li>CTRL+left-click (selection of multiple non-neighbouring objects),</li>
55  <li>or SHIFT+left-click (selection of multiple & neighbouring objects)</li>
56  </ul>
57 When the selection changes, the KVListView emits the <pre>SelectionChanged()</pre> signal.<br>
58 The list of the currently selected objects can be retrieved with the method <pre>GetSelectedItems()</pre>,
59  which returns a TList of the currently selected <a href="KVLVEntry.html">KVLVEntry</a> objects,
60  each one's name corresponds to the value displayed in the first column of the list view.<br>
61 Double-left-clicking an object will execute the <pre>Browse()</pre> method of the object. This method is defined for all TObjects,
62 it can be overridden in derived classes in order to do something interesting and/or useful.<br>
63 Right-clicking an object opens the context menu of the object, allowing the usual interaction with objects as in TBrower, TCanvas, etc.

125 KVLVEntry::KVLVEntry(TObject* obj, const KVLVContainer* cnt,
126  UInt_t ncols, KVLVColumnData** coldata)
127  : TGLVEntry(cnt, TString(coldata[0]->GetDataString(obj)),
128  TString(obj->ClassName()), 0, kVerticalFrame, GetWhitePixel()), fDisconnectRefresh(kFALSE)
129 {
130  // Default constructor
131  fEditMode = kFALSE;
132  if (!fgGreyPixel) {
133  if (!fClient->GetColorByName("#f0f0f0", fgGreyPixel)) fgGreyPixel = 0;
134  fgBGColor = fgWhitePixel;
135  }
136  fBGColor = -1;
137 
138  fUserData = obj;
139  fSubnames = new TGString* [ncols];
140  fBoolean = new Bool_t [ncols];
141  for (int i = 0; i < (int)ncols - 1; i++) {
142  fSubnames[i] = new TGString(coldata[i + 1]->GetDataString(obj));
143  fBoolean[i] = coldata[i + 1]->IsBoolean();
144  }
145  fSubnames[ncols - 1] = 0;
146  fBoolean[ncols - 1] = kFALSE;
147  int j;
148  for (j = 0; fSubnames[j] != 0; ++j)
149  ;
150  fCtw = new int[j + 1];
151  fCtw[j] = 0;
152  for (int i = 0; fSubnames[i] != 0; ++i)
153  fCtw[i] = gVirtualX->TextWidth(fFontStruct, fSubnames[i]->GetString(),
154  fSubnames[i]->GetLength());
155  SetWindowName();
156 
157  // to update display of object characteristics when object is changed,
158  // the object must have a "Connect" method (for signals-slots) and a "Modified"
159  // method which emits the "Modified" signal when object is changed.
160  // N.B. the object does not have to inherit from TQObject (can use RQ_OBJECT macro)
161  // the "Modified" signal is connected to the KVLVEntry::Refresh method
162  if (obj->IsA()->GetMethodAllAny("Modified") && obj->IsA()->GetMethodAllAny("Connect")) {
163  gInterpreter->Execute(obj, obj->IsA(), "Connect",
164  Form("\"Modified()\",\"KVLVEntry\",(KVLVEntry*)%ld,\"Refresh()\"", (ULong_t)this));
165  fDisconnectRefresh = kTRUE;
166  }
167 
168  fColoured = kFALSE;
169  // objects of classes with a method "const Char_t* GetLVEntryColour()" will be displayed with
170  // the background colour returned by this method (either in hexadecimal format, i.e. "#f0f0f0"
171  // or by name, i.e. "pink")
172  if (obj->IsA()->GetMethodAllAny("GetLVEntryColour")) {
173 
174  TMethodCall mt;
175  mt.InitWithPrototype(obj->IsA(), "GetLVEntryColour", "");
176  if (mt.IsValid()) {
177  if (mt.ReturnType() == TMethodCall::kString) {
178  Char_t* ret;
179  mt.Execute(obj, "", &ret);
180  if (fClient->GetColorByName(ret, fBGColor)) {
181  fColoured = kTRUE;
182  } else {
183  Warning("KVLVEntry", "Unknown color %s requested for entry", ret);
184  }
185  } else {
186  Warning("KVLVEntry", "Object of class %s has GetLVEntryColour() method with wrong return type",
187  obj->ClassName());
188  }
189  } else {
190  Warning("KVLVEntry", "GetLVEntryColour() method not valid for class %s", obj->ClassName());
191  }
192  }
193 }

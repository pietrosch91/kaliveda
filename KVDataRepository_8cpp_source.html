39 <h2>KVDataRepository</h2>
40 <h4>Base class for managing a collection of data files</h4>
41 <p>
42 Data repositories are defined in the user's $HOME/.kvrootrc; some typical examples are given
43 in $KVROOT/KVFiles/.kvrootrc:
44 </p>
45 <pre>
46 DataRepository: home
47 home.DataRepository.RootDir: $(HOME)/Data
48 </pre>
49 <p>
50 This is the minimum requirement to define a data repository: give a name ("home") and the
51 full path to the top level directory ("...RootDir: $(HOME)/Data"). In this case the top level
52 directory must be directly accessible to the user's machine as a locally-mounted disk.
53 Note that environment variables can be used, if enclosed between "$(...)".
54 </p>
55 <pre>
56 DataRepository: ccali
57 ccali.DataRepository.Type: remote
58 ccali.DataRepository.RootDir: cchpssindra:/hpss/in2p3.fr/group/indra
59 ccali.DataRepository.ReadProtocol: root
60 ccali.DataRepository.XRDServer: ccxroot:1999
61 ccali.DataRepository.XRDRootDir: /hpss/in2p3.fr/group/indra
62 ccali.DataRepository.XRDTunnel.host: ccali.in2p3.fr
63 ccali.DataRepository.XRDTunnel.port: 10000
64 ccali.DataRepository.XRDTunnel.user:
65 ccali.DataRepository.RemoteAvailableRuns.protocol: curl
66 ccali.DataRepository.RemoteAvailableRuns.url: http://indra.in2p3.fr/KaliVedaDoc
67 ccali.DataRepository.FileTransfer.type: bbftp
68 ccali.DataRepository.FileTransfer.server: ccbbftp.in2p3.fr
69 ccali.DataRepository.FileTransfer.user:
70 </pre>
71 <p>
72 This is the definition of a "remote" data repository. It will be handled by an object of the class
73 <a href="KVRemoteDataRepository.html">KVRemoteDataRepository</a>.
74 </p>
75 <p>
76 A remote data repository is principally characterised by the fact that access to the data files is
77 via some non-local protocol: in this case it is xrootd ("...ReadProtocol: root"). The host name and port of
78 the xrootd server are given ("...XRDServer: ccxroot:1999"), as well as the root directory
79 to be used ("...XRDRootDir: /hpss/in2p3.fr/group/indra").
80 </p>
81 <p>
82 In order to use a remote repository, some way to access the database of available
83 data files for each run type must be defined. Here it is the curl programme which is used to read
84 them via a website ("...RemoteAvailableRuns.protocol: curl"; "...RemoteAvailableRuns.url: http://indra.in2p3.fr/KaliVedaDoc").
85 </p>
86 <p>
87 Another particularity of remote data repositories is that some additional stuff may be needed
88 in order to be able to access the data.
89 This is handled by <a href="KVRemoteDataRepository.html#KVRemoteDataRepository:IsConnected">KVRemoteDataRepository::IsConnected()</a>.
90 In the present example, an SSH tunnel is set up in order to allow secure connection to the
91 xrootd server ("...XRDTunnel.*:").
92 </p>
93 <p>
94 Finally, transfer of data between data repositories is handled by <a href="KVDataTransfer.html">KVDataTransfer</a> and child classes.
95 In the present example, this is configured to use bbftp ("...FileTransfer.type: bbftp"), for which the required details are given.
96 </p>
97 <h4>Default data repository</h4>
98 <p>If more than one repository is defined, which one will be "active" (gDataRepository)
99 after initialisation of the data repository manager ? You can define the default repository
100 by setting the variable
101 </p>
102 <pre>
103 DataRepository.Default: [name]
104 </pre>
105 <p>in your .kvrootrc file. If you don't, the default repository "by default" will be either:
106 the data repository named "default" if there is one; or the last one defined in your .kvrootrc.
107 </p>

221 Bool_t KVDataRepository::Init()
222 {
223  //Initialises data repository based on information in .kvrootrc file.
224  //Each line beginning "name.DataRepository" is used for this data repository,
225  //where "name" is the name of this repository.
226  //
227  //Then we create and initialise the data set manager for this repository.
228  //
229  //returns kTRUE if all goes well.
230  //returns kFALSE if not (i.e. data set manager cannot be initialised, no available datasets)
231 
232  fLocalrootdir =
233  gEnv->GetValue(Form("%s.DataRepository.RootDir", GetName()), "");
234  if (fLocalrootdir == "") {
235  Error("Init", "Top-level directory for repository %s is not defined. Set %s.DataRepository.RootDir variable in .kvrootrc",
236  GetName(), GetName());
237  return kFALSE;
238  }
239  fAccessprotocol =
240  gEnv->GetValue(Form("%s.DataRepository.AccessProtocol", GetName()),
241  "local");
242  fReadprotocol =
243  gEnv->GetValue(Form("%s.DataRepository.ReadProtocol", GetName()),
244  "local");
245  //can we create and write directly new files in the repository ?
246  //for local repositories, default is yes, unless 'DataRepository.CanWrite' = 'NO'
247  //for remote repositories, default is no, unless 'DataRepository.CanWrite' = 'YES'
248  fCanWrite = (Bool_t)gEnv->GetValue(Form("%s.DataRepository.CanWrite", GetName()), (Int_t)(!IsRemote()));
249  //xrootd protocol
250  fXrootdserver =
251  gEnv->GetValue(Form("%s.DataRepository.XRDServer", GetName()), "");
252  fXrootdrootdir =
253  gEnv->GetValue(Form("%s.DataRepository.XRDRootDir", GetName()), "");
254  //access to xrootd via ssh tunnel ?
255  fXRDtunnel = (fXRDtunPort = (Int_t)gEnv->GetValue(Form("%s.DataRepository.XRDTunnel.port", GetName()), 0));
256  if (fXRDtunnel) {
257  PrepareXRDTunnel();
258  }
259  //rfio protocol
260  fRfioserver =
261  gEnv->GetValue(Form("%s.DataRepository.RFIOServer", GetName()), "");
262  fRfiorootdir =
263  gEnv->GetValue(Form("%s.DataRepository.RFIORootDir", GetName()),
264  "");
265  //file transfer protocol
266  fTransfertype =
267  gEnv->
268  GetValue(Form("%s.DataRepository.FileTransfer.type", GetName()),
269  "sftp");
270  if (fTransfertype.Contains("bbftp")) {
271  fTransferExec = fTransfertype;
272  fTransfertype = "bbftp";
273  if (!KVBase::FindExecutable(fTransferExec)) {
274  Error("Init", "Executable for bbftp client not found. Check %s.DataRepository.FileTransfer.type",
275  GetName());
276  }
277  } else if (fTransfertype.Contains("sftp")) {
278  fTransferExec = fTransfertype;
279  fTransfertype = "sftp";
280  if (!KVBase::FindExecutable(fTransferExec)) {
281  fTransferExec = "";
282  }
283  } else if (fTransfertype.Contains("root")) {
284  fTransferExec = "root";
285  fTransfertype = "xrd";
286  if (!fXRDtunnel) fTransferExec = ""; // must have viable SSH tunnel description
287  }
288  fTransferserver =
289  gEnv->
290  GetValue(Form("%s.DataRepository.FileTransfer.server", GetName()),
291  "");
292  fTransferuser = gEnv->GetValue(Form("%s.DataRepository.FileTransfer.user", GetName()), ""); //if not given, we use current username
293  if (fTransferuser == "") {
294  UserGroup_t* user = gSystem->GetUserInfo();
295  fTransferuser = user->fUser;
296  }
297  //set full paths to be used for checking existence of files and for opening files
298  SetFullPath(fAccessroot, fAccessprotocol.Data());
299  SetFullPath(fReadroot, fReadprotocol.Data());
300  //data set manager
301  if (fDSM) {
302  delete fDSM;
303  }
304  fDSM = NewDataSetManager();
305 
306  //return status of dataset manager
307  return fDSM->Init(this);
308 }

412 void KVDataRepository::SetFullPath(TString& path, const Char_t* protocol)
413 {
414  //Will replace contents of 'path' with the full base path needed for the given protocol.
415  //
416  //protocol = "local" : path = fLocalrootdir
417  //protocol = "root" : path = "root://" + fXrootdserver + "/" + fXrootdrootdir
418  //protocol = "rfio" : path = "rfio:" + fRfioserver + ":" + fRfiorootdir
419 
420  TString tmp(protocol);
421  if (tmp == "local")
422  path = fLocalrootdir;
423  else if (tmp == "root")
424  path.Form("root://%s/%s", fXrootdserver.Data(),
425  fXrootdrootdir.Data());
426  else if (tmp == "rfio")
427  path.Form("rfio:%s:%s", fRfioserver.Data(), fRfiorootdir.Data());
428  else
429  Warning("SetFullPath",
430  "Unknown protocol in call to SetFullPath : %s", protocol);
431 }

515 const Char_t* KVDataRepository::GetFullPathToOpenFile(const KVDataSet* dataset,
516  const Char_t*
517  datatype,
518  const Char_t*
519  runfile)
520 {
521  //Returns the full path (including protocol specification) needed by TFile::Open or TChain::Add
522  //to open a runfile belonging to the given dataset.
523  //The protocol depends on the value of fReadprotocol.
524  //If a special protocol has been defined for the given dataset and/or datatype, as in this example
525  //for "raw" data:
526  //
527  //ccali.DataRepository.ReadProtocol.raw: rfio
528  //
529  //it will be used instead of the default (fReadprotocol).
530  //If a given data type is not accessible for a repository, use "none":
531  //
532  //ccali.DataRepository.ReadProtocol.raw: none
533  //
534  //If this is the case, an error message will be printed and an empty string returned.
535 
536  static TString path;
537 
538  TString read_proto = GetReadProtocol(dataset->GetName(), datatype);
539  if (read_proto == "none") {
540  Error("GetFullPathToOpenFile", "Datatype \"%s\" can not be read from the repository \"%s\"",
541  datatype, GetName());
542  path = "";
543  return path.Data();
544  }
545  TString read_root;
546  if (read_proto != fReadprotocol) {
547  SetFullPath(read_root, read_proto.Data());
548  } else {
549  read_root = fReadroot;
550  }
551  TString tmp, datasetdir = dataset->GetDataPathSubdir();
552  AssignAndDelete(path,
553  gSystem->ConcatFileName(read_root.Data(),
554  datasetdir.Data()));
555  AssignAndDelete(tmp, gSystem->ConcatFileName(path.Data(), dataset->GetDataTypeSubdir(datatype)));
556  AssignAndDelete(path, gSystem->ConcatFileName(tmp.Data(), runfile));
557  return path.Data();
558 }

607 const Char_t* KVDataRepository::GetReadProtocol(const Char_t* dataset,
608  const Char_t* datatype)
609 {
610  //Returns string containing protocol for reading files of the given datatype
611  //belonging to the dataset whose name is given.
612  //Protocol = "local", "root", "rfio" , "none"
613  //(if "none", the data can not be read)
614  //If a specific protocol for the dataset and/or datatype is not defined in $KVROOT/KVFiles/.kvrootrc,
615  //the default protocol (=fReadprotocol) is returned.
616  //Format for defining specific protocols:
617  // name_of_repository.DataRepository.ReadProtocol.dataset_name.data_type: protocol
618  // name_of_repository.DataRepository.ReadProtocol.dataset_name: protocol
619  // name_of_repository.DataRepository.ReadProtocol.data_type: protocol
620 
621  static TString prot;
622  prot =
623  gEnv->
624  GetValue(Form
625  ("%s.DataRepository.ReadProtocol.%s.%s", GetName(),
626  dataset, datatype), "");
627  if (prot != "")
628  return prot.Data();
629  prot =
630  gEnv->
631  GetValue(Form
632  ("%s.DataRepository.ReadProtocol.%s", GetName(), dataset),
633  "");
634  if (prot != "")
635  return prot.Data();
636  prot =
637  gEnv->
638  GetValue(Form
639  ("%s.DataRepository.ReadProtocol.%s", GetName(), datatype),
640  "");
641  if (prot != "")
642  return prot.Data();
643  return fReadprotocol;
644 }

746 KVUniqueNameList* KVDataRepository::GetDirectoryListing(const KVDataSet* dataset,
747  const Char_t* datatype)
748 {
749  //Use the access protocol defined by DataRepository.AccessProtocol (=local by default)
750  //in order to open the directory
751  //
752  // /root_of_data_repository/[datasetdir]/[datatype]
753  // /root_of_data_repository/[datasetdir] (if datatype="", default value)
754  //
755  //and fill a TList with one KVBase object for each entry in the directory,
756  //excluding "." and ".."
757  //User must delete the TList after use (list will delete its members)
758 
759  TString path, tmp;
760  AssignAndDelete(path,
761  gSystem->ConcatFileName(fAccessroot.Data(),
762  dataset->GetDataPathSubdir()));
763  if (strcmp(datatype, "")) {
764  AssignAndDelete(tmp, gSystem->ConcatFileName(path.Data(), dataset->GetDataTypeSubdir(datatype)));
765  path = tmp;
766  }
767  //open directory
768  void* dirp = gSystem->OpenDirectory(path.Data());
769  if (!dirp) {
770  Error("KVDataRepository::GetDirectoryListing", "Cannot open %s",
771  path.Data());
772  return 0;
773  }
774 
775  KVUniqueNameList* dirlist = new KVUniqueNameList(kTRUE);
776  dirlist->SetOwner(kTRUE);
777 
778  TObjString* direntry = new TObjString(gSystem->GetDirEntry(dirp));
779  while (direntry->GetString() != "") { //loop over all entries in directory
780  if (direntry->GetString() == "." || direntry->GetString() == "..") {
781  //skip "." and ".."
782  delete direntry;
783  } else {
784  dirlist->Add(new KVBase(direntry->GetString().Data()));
785  delete direntry;
786  }
787  //get next entry
788  direntry = new TObjString(gSystem->GetDirEntry(dirp));
789  }
790  //delete last TObjString, which should contain ""
791  delete direntry;
792  //close directory
793  gSystem->FreeDirectory(dirp);
794  return dirlist;
795 }

49 KVVirtualIDFitter* KVVirtualIDFitter::GetDefaultFitter()
50 {
51  // Static method. Creates (if necessary) and returns pointer to
52  // static instanciation of default ID fitter object. The class of this
53  // object is defined by the variable
54  //
55  // KVVirtualIDFitter.DefaultFitter: xxxxxxxxxxx
56  //
57  // which can be customised in the user's .kvrootrc. The class must be defined
58  // as a KVVirtualIDFitter plugin and have a default ctor with no arguments:
59  //
60  // +Plugin.KVVirtualIDFitter: MyFitter MyFitter MyFitter.cpp+ "MyFitter()"
61 
62  if (!gDefaultFitter) {
63  TString df = gEnv->GetValue("KVVirtualIDFitter.DefaultFitter", "");
64  if (df == "") {
65  ::Error("KVVirtualIDFitter::GetDefaultFitter",
66  "No default fitter defined. Use variable KVVirtualIDFitter.DefaultFitter");
67  return 0;
68  }
69  TPluginHandler* ph;
70  if (!(ph = LoadPlugin("KVVirtualIDFitter", df.Data()))) return 0;
71  gDefaultFitter = (KVVirtualIDFitter*) ph->ExecPlugin(0);
72  }
73  return gDefaultFitter;
74 }

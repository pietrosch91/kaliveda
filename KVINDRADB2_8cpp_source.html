111 void KVINDRADB2::GoodRunLine()
112 {
113  //For each "good run line" in the run list file, we:
114  // add a KVINDRADBRun to the database if it doesn't already exist
115  // add a KVDBTape to the database if the "tape" field is active and if it doesn't already exist
116  // set properties of run and tape objects
117  //kFirstRun & kLastRun are set
118 
119  KVRunListLine* csv_line = GetLineReader();
120 
121  //run number
122  Int_t run_n = csv_line->GetIntField("Run");
123 
124  if (!run_n) {
125  cout << "run_n = 0 ????????? line number =" << GetRLLineNumber() <<
126  endl;
127  GetLineReader()->Print();
128  return;
129  }
130  kLastRun = TMath::Max(kLastRun, run_n);
131  kFirstRun = TMath::Min(kFirstRun, run_n);
132 
133  /*********************************************
134  IF LINE HAS A TAPE NUMBER WE
135  LOOK FOR THE TAPE IN THE DATA
136  BASE. IF IT DOESN'T EXIST WE
137  CREATE IT.
138  *********************************************/
139  KVDBTape* tape = 0;
140  //tape number (if tape field is filled)
141  if (csv_line->HasFieldValue("Tape")) {
142  Int_t tape_n = csv_line->GetIntField("Tape");
143  //already exists ?
144  tape = GetTape(tape_n);
145  if (!tape) {
146  tape = new KVDBTape(tape_n);
147  AddTape(tape);
148  }
149  }
150 
151  /*********************************************
152  WE CREATE A NEW RUN AND ADD
153  IT TO THE DATABASE. WE SET ALL
154  AVAILABLE INFORMATIONS ON
155  RUN FROM THE FILE. ERROR IF
156  DBASE RUN ALREADY EXISTS =>
157  SAME RUN APPEARS TWICE
158  *********************************************/
159  KVINDRADBRun* run = (KVINDRADBRun*) GetRun(run_n);
160  if (!run) {
161 
162  run = new KVINDRADBRun(run_n);
163  AddRun(run);
164 
165  //add run to tape ?
166  if (tape)
167  tape->AddRun(run);
168 
169  if (csv_line->HasFieldValue("Events"))
170  run->SetEvents(csv_line->GetIntField("Events"));
171  else
172  Error("GoodRunLine", "No Events field ? run=%d", run_n);
173  if (csv_line->HasFieldValue("FAR"))
174  run->SetScaler("Faraday 1", csv_line->GetIntField("FAR"));
175  else
176  Error("GoodRunLine", "No FAR field ? run=%d", run_n);
177  if (csv_line->HasFieldValue("MFI"))
178  run->SetScaler("MFI", csv_line->GetIntField("MFI"));
179  if (csv_line->HasFieldValue("Blocs"))
180  run->SetScaler("Buffers", csv_line->GetIntField("Blocs"));
181  else
182  Error("GoodRunLine", "No Blocs field ? run=%d", run_n);
183  if (csv_line->HasFieldValue("G. dir"))
184  run->SetScaler("Gene DIRECT", csv_line->GetIntField("G. dir"));
185  else
186  Error("GoodRunLine", "No G. dir field ? run=%d", run_n);
187  if (csv_line->HasFieldValue("G. marq"))
188  run->SetScaler("Gene MARQ", csv_line->GetIntField("G. marq"));
189  else
190  Error("GoodRunLine", "No G. marq field ? run=%d", run_n);
191  if (csv_line->HasFieldValue("G. TM"))
192  run->SetScaler("Gene TM", csv_line->GetIntField("G. TM"));
193  else
194  Error("GoodRunLine", "No G. TM field ? run=%d", run_n);
195  if (csv_line->HasFieldValue("TM %"))
196  run->SetTMpercent(csv_line->GetFloatField("TM %"));
197  else
198  Error("GoodRunLine", "No TM %% field ? run=%d", run_n);
199  if (csv_line->HasField("Trigger"))
200  run->SetTrigger(GetRunListTrigger("Trigger", "M>=%d"));
201  else
202  Error("GoodRunLine", "No Trigger field ? run=%d", run_n);
203  if (csv_line->HasFieldValue("NB"))
204  run->SetComments(csv_line->GetField("NB"));
205 
206  } else {
207  Error("GoodRunLine", "Run %d already exists", run_n);
208  }
209 }

123 KVIDQALine* KVIDQAGrid::GetQLine(Int_t q, Int_t& index) const
124 {
125  //Returns ID line for which GetQ() returns 'q'. index=index of line found in
126  //fIDLines list (-1 if not found).
127  //To increase speed, this is done by dichotomy, rather than by looping over
128  //all the lines in the list.
129 
130  index = -1;
131  Int_t nlines = GetNumberOfIdentifiers();
132  UInt_t idx_min = 0; //minimum index
133  UInt_t idx_max = nlines - 1; // maximum index
134  //as a first guess, we suppose that the lines are numbered sequentially from
135  //Q=1 (index 0) to Q=qmax=nlines (index nlines-1) with no gaps
136  UInt_t idx = ((UInt_t) q - 1 > idx_max ? idx_max : (UInt_t) q - 1);
137 
138  while (idx_max > idx_min + 1) {
139  KVIDQALine* line = (KVIDQALine*) GetIdentifierAt(idx);
140  Int_t qline = line->GetQ();
141  //Found it ?
142  if (qline == q) {
143  index = idx;
144  return line;
145  }
146 
147  if (q > qline) {
148  //increase index
149  idx_min = idx;
150  idx += (Int_t)((idx_max - idx) / 2 + 0.5);
151  } else {
152  //decrease index
153  idx_max = idx;
154  idx -= (Int_t)((idx - idx_min) / 2 + 0.5);
155  }
156  }
157  //if one of these two lines has the right Q, return its pointer
158  KVIDQALine* line = (KVIDQALine*) GetIdentifierAt(idx_min);
159  if (line->GetQ() == q) {
160  index = idx_min;
161  return line;
162  }
163  line = (KVIDQALine*) GetIdentifierAt(idx_max);
164  if (line->GetQ() == q) {
165  index = idx_max;
166  return line;
167  }
168  //if not, there is no line in the grid with the right Q
169  return 0;
170 }

188 void KVIDQAGrid::CalculateLineWidths()
189 {
190  //Calculate natural "width" of each line in the grid.
191  //The lines in the grid are first sorted so that they are in order of ascending 'Y'
192  //i.e. first line is Qmin, last line is Qmax (highest line).
193  //
194  //Then, for a given line we use the separation between each pair of lines.
195  //
196  //In each case we find D_L (the separation between the two lines at their extreme left)
197  //and D_R (their separation at extreme right). The width of the line is then calculated
198  //from these two using the method KVIDQALine::SetAsymWidth (which may be overridden in child
199  //classes).
200 
201 
202 // Info("CalculateLineWidths",
203 // "For grid %s (%s vs. %s, runs %s).", GetName(), GetVarY(), GetVarX(), GetRunList());
204 
205  for (Int_t i = 0; i < (Int_t) GetNumberOfIdentifiers(); i++) {
206 
207  KVIDQALine* _line = (KVIDQALine*) GetIdentifierAt(i);
208 
209  //Q of lines above and below this line - Qxx=-1 if there is no line above or below
210  Int_t Qhi =
211  ((i < (Int_t) GetNumberOfIdentifiers() - 1) ? ((KVIDQALine*) GetIdentifierAt(i + 1))->GetQ() : -1);
212  Int_t Qlo = (i > 0 ? ((KVIDQALine*) GetIdentifierAt(i - 1))->GetQ() : -1);
213 
214  // find line for comparison.
215  // It should have only one line per Q, so we calculate
216  // widths by comparing neighbouring Q.
217 
218  Int_t i_other;// index of line used to calculate width
219  i_other = (Qhi > -1 ? i + 1 : (Qlo > -1 ? i - 1 : -1));
220 
221  //default width of 16000 in case of "orphan" line
222  if (i_other < 0) {
223  Info("CalculateLineWidths", "No line found for comparison with %s. Width set to 16000", _line->GetName());
224  _line->SetWidth(16000.);
225  continue; // skip to next line
226  }
227 
228  KVIDQALine* _otherline = (KVIDQALine*) GetIdentifierAt(i_other);
229 
230  //calculate asymptotic distances between lines at left and right.
231  //do this by finding which line's endpoint is between both endpoints of the other line
232  Double_t D_L, D_R;
233 
234  //*************** LEFT SIDE ********************************
235 
236  //get coordinates of first point of our line
237  Double_t x1, y1;
238  _line->GetStartPoint(x1, y1);
239  KVIDQALine* _aline, *_bline;
240 
241  if (_otherline->IsBetweenEndPoints(x1, y1, "x")) {
242 
243  //first point of our line is "inside" the X-coords of the other line:
244  //asymptotic distance LEFT is distance from our line's starting point (x1,Y1) to the other line
245  _aline = _otherline;
246  _bline = _line;
247 
248  } else {
249 
250  //first point of other line is "inside" the X-coords of the our line:
251  //asymptotic distance LEFT is distance from other line's starting point (x1,Y1) to our line
252  _aline = _line;
253  _bline = _otherline;
254  _otherline->GetStartPoint(x1, y1);
255 
256  }
257 
258  //calculate D_L
259  Int_t dummy = 0;
260  D_L = _aline->DistanceToLine(x1, y1, dummy);
261 
262  //make sure that D_L is positive : if not, we try to calculate other way (inverse line and starting point)
263  if (D_L < 0.) {
264 
265  Double_t oldD_L = D_L;
266 
267  _aline->GetStartPoint(x1, y1);
268  D_L = _bline->DistanceToLine(x1, y1, dummy);
269 
270  //still no good ? then we keep the smallest absolute value
271  if (D_L < 0.) {
272  D_L = TMath::Abs(TMath::Max(D_L, oldD_L));
273  }
274  }
275  //*************** RIGHT SIDE ********************************
276 
277  //get coordinates of last point of our line
278  _line->GetEndPoint(x1, y1);
279 
280  if (_otherline->IsBetweenEndPoints(x1, y1, "x")) {
281 
282  //last point of our line is "inside" the X-coords of the other line:
283  //asymptotic distance RIGHT is distance from our line's end point (x1,Y1) to the other line
284  _aline = _otherline;
285  _bline = _line;
286 
287  } else {
288 
289  //last point of other line is "inside" the X-coords of the our line:
290  //asymptotic distance RIGHT is distance from other line's end point (x1,Y1) to our line
291  _aline = _line;
292  _bline = _otherline;
293  _otherline->GetEndPoint(x1, y1);
294 
295  }
296 
297  //calculate D_R
298  D_R = _aline->DistanceToLine(x1, y1, dummy);
299 
300  //make sure that D_R is positive : if not, we try to calculate other way (inverse line and end point)
301  if (D_R < 0.) {
302 
303  Double_t oldD_R = D_R;
304 
305  _aline->GetEndPoint(x1, y1);
306  D_R = _bline->DistanceToLine(x1, y1, dummy);
307 
308  //still no good ? then we keep the smallest absolute value
309  if (D_R < 0.) {
310  D_R = TMath::Abs(TMath::Max(D_R, oldD_R));
311  }
312  }
313  //*************** SET NATURAL WIDTH OF LINE ********************************
314 
315  _line->SetAsymWidth(D_L, D_R);
316 
317 
318  //Info("CalculateLineWidths", "...width for line %s set to : %f (D_L=%f,D_R=%f)", _line->GetName(), _line->GetWidth(), D_L, D_R);
319  }
320 }

380 void KVIDQAGrid::Identify(Double_t x, Double_t y, KVIdentificationResult* idr) const
381 {
382  // Fill the KVIdentificationResult object with the results of identification for point (x,y)
383  // corresponding to some physically measured quantities related to a reconstructed nucleus.
384  //
385  // By default (OnlyQId()=kFALSE) this means identifying the Q & A of the nucleus.
386  // In this case, we consider that the nucleus' Q & A have been correctly measured
387  //
388  //
389  // if the returned 'quality code' is < kICODE4:
390  // we set idr->Zident (Z<->Q) and idr->Aident is determined if
391  // OnlyQId()=kFALSE.
392  // The mass idr->A is set to the mass of the nearest A marker of the KVIDQALine if the A identification went well .
393  //
394  // If OnlyQId()=kTRUE, only the Q of the nucleus is established.
395  // In this case, we consider that only the nucleus' charge state Q has been correctly measured.
396  //
397  // Real & integer masses for A-identified nuclei
398  // ========================================================
399  // For points anddlying between two A markers of a same Q line (fICode<kIDCode4)
400  // the "real" mass is given by interpolation between the two masses.
401  // The integer mass is the A of the marker closest to the point.
402 
403 
404  idr->IDOK = kFALSE;
405  const_cast < KVIDQAGrid* >(this)->fRealQ = -1.0;
406 
407  Int_t idx, idx_inf, idx_sup;
408  Double_t dist, dist_inf, dist_sup;
409  KVIDLine* line = const_cast<KVIDQAGrid*>(this)->FindNearestIDLineFast(x, y, "above", idx, idx_inf, idx_sup, dist, dist_inf, dist_sup);
410  KVIDQALine* closest_line = NULL;
411 
412  if (!line || !line->InheritsFrom(KVIDQALine::Class())) {
413  //no line corresponding to point was found
414  const_cast < KVIDQAGrid* >(this)->fICode = kICODE8; // Q indetermine ou (x,y) hors limites
415  } else {
416  //the closest_line is found
417  closest_line = (KVIDQALine*)line;
418 
419  // line below the point
420  line = (KVIDLine*)GetIdentifierAt(idx_inf);
421  KVIDQALine* inf_line = (line && line->InheritsFrom(KVIDQALine::Class()) ? (KVIDQALine*)line : NULL);
422 
423  // line above the point
424  line = (KVIDLine*)GetIdentifierAt(idx_sup);
425  KVIDQALine* sup_line = (line && line->InheritsFrom(KVIDQALine::Class()) ? (KVIDQALine*)line : NULL);
426 
427  // WARNING: the method KVIDQAGrid::FindNearestIDLineFast always find
428  // 2 nearest lines even for point below (above) the first (last) line.
429  // we have to remove the second found line for these points
430 
431  if ((closest_line == GetIdentifiers()->First()) && closest_line->WhereAmI(x, y, "below"))
432  sup_line = NULL;
433  else if (closest_line == GetIdentifiers()->Last() && closest_line->WhereAmI(x, y, "above"))
434  inf_line = NULL;
435 
436 
437  Double_t Q = 0.;
438  Double_t deltaQ = 0.;
439 
440  // case where 2 embracing lines are found
441  if (inf_line && sup_line && (inf_line != sup_line)) {
442  Int_t dQ = sup_line->GetQ() - inf_line->GetQ();
443  Double_t tot_dist = (dist_inf + dist_sup) / (1.0 * dQ);
444  deltaQ = dist / tot_dist;
445 
446  // if the point is above the closest_line (i.e. the
447  // closest_line is the inf_line) we must add deltaQ
448  // to Q, otherwise, we must take deltaQ away from Q
449  // (i.e. closest_line is the sup_line)
450  if (closest_line == sup_line) deltaQ *= -1.;
451 
452  if (dist > closest_line->GetWidth() / 2.) {
453  if (deltaQ > 0)
454  const_cast < KVIDQAGrid* >(this)->fICode = kICODE1; // "slight ambiguity of Q, which could be larger"
455  else
456  const_cast < KVIDQAGrid* >(this)->fICode = kICODE2; // "slight ambiguity of Q, which could be smaller"
457 
458  } else const_cast < KVIDQAGrid* >(this)->fICode = kICODE0; // ok
459 
460  if (deltaQ > 0.5) Info("Identify", "deltaQ= %f, Qclosest= %d, Qinf= %d, Qsup= %d, icode= %d, X= %f, Y= %f", deltaQ, closest_line->GetQ(), inf_line->GetQ(), sup_line->GetQ(), fICode, x, y);
461  }
462  // case where only 1 embracing line is found.
463  // in this case, the distance between the point and the line
464  // has to be lower than the line width
465  else {
466  if (dist > closest_line->GetWidth() / 2.) {
467  // the distance from the closest line is to high
468 
469  if (closest_line == inf_line)
470  const_cast < KVIDQAGrid* >(this)->fICode = kICODE6; // "(x,y) is below first line in grid"
471  else
472  const_cast < KVIDQAGrid* >(this)->fICode = kICODE7; // "(x,y) is above last line in grid"
473  } else {
474  deltaQ = dist / closest_line->GetWidth();
475  if (closest_line == sup_line) deltaQ *= -1.;
476  const_cast < KVIDQAGrid* >(this)->fICode = kICODE3; // "slight ambiguity of Q, which could be larger or smaller"
477  }
478  if (deltaQ > 0.5) Info("Identify", "deltaQ= %f, Qclosest= %d, icode= %d, X= %f, Y= %f", deltaQ, closest_line->GetQ(), fICode, x, y);
479  }
480 
481  Q = closest_line->GetQ() + deltaQ;
482  const_cast < KVIDQAGrid* >(this)->fRealQ = Q;
483  idr->Z = closest_line->GetQ();
484  idr->PID = Q;
485  }
486 
487 
488  switch (fICode) {
489  case kICODE0:
490  idr->SetComment("ok");
491  break;
492  case kICODE1:
493  idr->SetComment("slight ambiguity of Q, which could be larger");
494  break;
495  case kICODE2:
496  idr->SetComment("slight ambiguity of Q, which could be smaller");
497  break;
498  case kICODE3:
499  idr->SetComment("slight ambiguity of Q, which could be larger or smaller");
500  break;
501  case kICODE4:
502  idr->SetComment("point is in between two lines of different Q, too far from either to be considered well-identified");
503  break;
504  case kICODE5:
505  idr->SetComment("point is in between two lines of different Q, too far from either to be considered well-identified");
506  break;
507  case kICODE6:
508  idr->SetComment("(x,y) is below first line in grid");
509  break;
510  case kICODE7:
511  idr->SetComment("(x,y) is above last line in grid");
512  break;
513  default:
514  idr->SetComment("no identification: (x,y) out of range covered by grid");
515  }
516 
517  idr->IDquality = fICode;
518  if (fICode < kICODE4) {
519  idr->Zident = kTRUE;
520  idr->IDOK = kTRUE;
521 
522  // if the grid is also used for A identification then start it
523  if (!OnlyQId()) {
524  //idr->IDOK = kFALSE;
525  Int_t Ai;
526  Double_t Ar;
527  Int_t code;
528  closest_line->IdentA(x, y, Ai, Ar, code);
529  idr->A = Ai;
530 
531  switch (code) {
532  case kICODE0:
533  idr->SetComment("ok");
534  break;
535  case kICODE1:
536  idr->SetComment("slight ambiguity of Q or A, which could be larger");
537  break;
538  case kICODE2:
539  idr->SetComment("slight ambiguity of Q or A, which could be smaller");
540  break;
541  case kICODE3:
542  idr->SetComment("slight ambiguity of Q or A, which could be larger or smaller");
543  break;
544  case kICODE4:
545  idr->SetComment("point is in between two lines (markers) of different Q (A), too far from either to be considered well-identified");
546  break;
547  case kICODE5:
548  idr->SetComment("point is in between two lines (markers) of different Q (A), too far from either to be considered well-identified");
549  break;
550  case kICODE6:
551  idr->SetComment("(x,y) is below (on the left of) first line (marker) in grid (line)");
552  break;
553  case kICODE7:
554  idr->SetComment("(x,y) is above (on the right of) last line (marker) in grid (line)");
555  break;
556  default:
557  idr->SetComment("no identification: (x,y) out of range covered by grid (line)");
558  }
559 
560  if (code < kICODE4) {
561  if (code > kICODE0) {
562  const_cast < KVIDQAGrid* >(this)->fICode = code;
563  idr->IDquality = fICode;
564  }
565  idr->PID = Ar;
566  idr->Aident = kTRUE;
567  idr->IDOK = kTRUE;
568  }
569  }
570  }

706 KVIDGraph* KVIDQAGrid::MakeSubsetGraph(TList* lines, TClass* graph_class)
707 {
708  // Create a new graph/grid using the subset of lines of this grid contained in TList 'lines'.
709  // By default the new graph/grid will be of the same class as this one, unless graph_class !=0,
710  // in which case it must contain the address of a TClass object representing a class which
711  // derives from KVIDGraph.
712  // A clone of each line will be made and added to the new graph, which will have the same
713  // name and be associated with the same ID telescopes as this one.
714 
715  if (!graph_class) graph_class = IsA();
716  if (!graph_class->InheritsFrom("KVIDGraph")) {

949 void KVIDQAGrid::TestIdentification(TH2F* data, TH1F* h1_q,
950  TH2F* h2_q_qxaoq, TH2F* h2_q_a)
951 {
952  //This method allows to test the identification capabilities of the grid using data in a TH2F.
953  //We assume that 'data' contains an identification map, whose 'x' and 'y' coordinates correspond
954  //to this grid. Then we loop over every bin of the histogram, perform the identification (if
955  //IsIdentifiable() returns kTRUE) and fill the two histograms with the resulting identification
956  //and its dependence on the 'x'-coordinate of the 'data' histogram,
957  //each identification weighted by the contents of the original data bin.
958  //
959  // Returned histograms:
960  // h1_q -- 1D histogram with distribution of identified charge states (Qreal).
961  // h2_q_qxaoq -- 2D map showing the distibrution on Qreal vs Qint*X plot.
962  // where Qint is the integer value of Qreal and X the
963  // X-coordinate of the initial data plot.
964  // h2_q_a -- 2D map showing the distribution on Qreal vs Areal plot
965  // where Areal is the mass identified with the markers.
966  // This map is filled only if the grid has OnlyQId()=false.
967 
968 
969  //Initialize the grid: calculate line widths etc.
970  Initialize();
971 
972  KVIdentificationResult* idr = new KVIdentificationResult;
973 
974  h1_q->Reset();
975  h2_q_qxaoq->Reset();
976  Int_t tot_events = (Int_t) data->GetSum();
977  Int_t events_read = 0;
978  Float_t percent = 0., cumul = 10.;
979  Bool_t qaMap = (!IsOnlyQId()) && (h2_q_a);
980 
981  Int_t Amin, Amax, Qmin, Qmax;
982  Double_t AoQmin, AoQmax;
983  GetLimitsOf_A_Q_AoQ(Amin, Amax, Qmin, Qmax, AoQmin, AoQmax);
984  Amin = AoQmin * Qmin;
985  Amax = AoQmax * Qmax;
986  Info("TestIdentification", "Amin %d, Amax %d, Qmin %d, Qmax %d, AoQmin %f, AoQmax %f", Amin, Amax, Qmin, Qmax, AoQmin, AoQmax);
987 
988  // change ranges and titles of histograms
989  TString title;
990  title = "Q_{real} distribution;Q_{real}";
991  h1_q->SetTitle(title.Data());
992  h1_q->SetBins(data->GetNbinsY(), Float_t(Qmin - 1), Float_t(Qmax + 1));
993 
994  title.Form("A vs. Q_{real};Q_{int}#times(%s);Q_{real}", data->GetXaxis()->GetTitle());
995  h2_q_qxaoq->SetTitle(title.Data());
996  h2_q_qxaoq->SetBins(data->GetNbinsX(), Float_t(Amin - 1), Float_t(Amax + 1), data->GetNbinsY(), Float_t(Qmin - 1), Float_t(Qmax + 1));
997 
998  if (qaMap) {
999  title = "A vs. Q_{real};A_{real};Q_{real}";
1000  h2_q_a->SetTitle(title.Data());
1001  h2_q_a->SetBins(data->GetNbinsX(), Float_t(Amin - 1), Float_t(Amax + 1), data->GetNbinsY(), Float_t(Qmin - 1), Float_t(Qmax + 1));
1002  }
1003 
1004  //loop over data in histo
1005  for (int i = 1; i <= data->GetNbinsX(); i++) {
1006  for (int j = 1; j <= data->GetNbinsY(); j++) {
1007 
1008  Stat_t poids = data->GetBinContent(i, j);
1009  if (poids == 0)
1010  continue;
1011 
1012  Axis_t x0 = data->GetXaxis()->GetBinCenter(i);
1013  Axis_t y0 = data->GetYaxis()->GetBinCenter(j);
1014  Axis_t wx = data->GetXaxis()->GetBinWidth(i);
1015  Axis_t wy = data->GetYaxis()->GetBinWidth(j);
1016  //If bin content ('poids') is <=20, we perform the identification 'poids' times, each time with
1017  //randomly-drawn x and y coordinates inside this bin
1018  //If 'poids'>20, we perform the identification 20 times and we fill the histograms with
1019  //a weight poids/20
1020  Double_t x, y;
1021  Int_t kmax = (Int_t) TMath::Min(20., poids);
1022  Double_t weight = (kmax == 20 ? poids / 20. : 1.);
1023  for (int k = 0; k < kmax; k++) {
1024 
1025  x = gRandom->Uniform(x0 - .5 * wx, x0 + .5 * wx);
1026  y = gRandom->Uniform(y0 - .5 * wy, y0 + .5 * wy);
1027  if (IsIdentifiable(x, y)) {
1028  Identify(x, y, idr);
1029  if (AcceptIDForTest()) {
1030  Double_t realQ = 0.;
1031  Double_t realA = 0.;
1032 
1033  if (idr->Aident) {
1034  realQ = fRealQ;
1035  realA = idr->PID;
1036  } else if (idr->Zident) realQ = idr->PID;
1037  h1_q->Fill(realQ, weight);
1038  h2_q_qxaoq->Fill(x * idr->Z, realQ, weight);
1039  if (qaMap) h2_q_a->Fill(realA, realQ, weight);
1040  }
1041  }
1042  }
1043  events_read += (Int_t) poids;
1044  percent = (1. * events_read / tot_events) * 100.;
1045  if (events_read % (tot_events / 10) == 0) Increment((Float_t) events_read); //sends signal to GUI progress bar
1046  if (percent >= cumul) {
1047  //std::cout << (Int_t) percent << "\% processed" << std::endl;
1048  cumul += 10;
1049  }
1050  gSystem->ProcessEvents();
1051  }
1052  }
1053 
1054  delete idr;
1055 }

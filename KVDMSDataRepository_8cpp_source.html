280 Bool_t KVDMSDataRepository::GetFileInfo(const KVDataSet* ds,
281  const Char_t* datatype,
282  const Char_t* runfile,
283  FileStat_t& fs)
284 {
285  //Checks if the run file of given type is physically present in dataset subdirectory,
286  //i.e. (schematically), if
287  //
288  // /root_of_data_repository/[datasetdir]/[datatypedir]/[runfile]
289  //
290  //exists. If it does, the returned value is kTRUE (=1), in which case the FileStat_t object
291  //contains information about the file:
292  // WARNING:
293  // only fs.fSize and fs.fMtime are used
294  // in addition, the modification time corresponds to the last time that
295  // the DMS declaration of the file was changed, not the last physical
296  // modification of the file, i.e. it will be the date at which the file
297  // was imported into the DMS catalogue, the file may be much older.
298 
299  TString path, tmp;
300  AssignAndDelete(path,
301  gSystem->ConcatFileName(fAccessroot.Data(), ds->GetDataPathSubdir()));
302  AssignAndDelete(tmp, gSystem->ConcatFileName(path.Data(), ds->GetDataTypeSubdir(datatype)));
303  AssignAndDelete(path, gSystem->ConcatFileName(tmp.Data(), runfile));
304 
305  DMSFile_t DMSfile;
306  if (fDMS->GetPathInfo(path.Data(), DMSfile)) {
307  fs.fSize = DMSfile.GetSize();
308  fs.fMtime = DMSfile.GetModTime().Convert();
309  return kTRUE;
310  }
311  return kFALSE;
312 }

356 TObject* KVDMSDataRepository::OpenDataSetRunFile(const KVDataSet* ds, const Char_t* type, Int_t run, Option_t* opt)
357 {
358  // Overrides KVDataRepository method
359  //
360  // If ReadProtocol == "root" we use KVDataRepository method to open file via xrootd server
361  //
362  // If ReadProtocol != "root" (default if user does not specify [name].DataRepository.ReadProtocol:root):
363  //
364  // 1) we look for a copy of the file in the user's temporary directory (gSystem->TempDirectory());
365  // 2) if no copy exists, we download the file to the user's temporary directory
366  // 3) we open the file in the user's temporary directory
367 
368  // get read protocol for dataset and datatype
369  TString RP = GetReadProtocol(ds->GetName(), type);
370  if (RP == "root") return KVDataRepository::OpenDataSetRunFile(ds, type, run, opt);
371 
372  // get name of file
373  TString filename = ds->GetRunfileName(type, run);
374  if (filename == "") {
375  Error("OpenDataSetRunFile", "No file found for run %d of data-type %s", run, type);
376  return nullptr;
377  }
378 
379  // look for file in temp dir
380  TString tmpdir_filepath;
381  AssignAndDelete(tmpdir_filepath, gSystem->ConcatFileName(gSystem->TempDirectory(), filename.Data()));
382 
383  if (gSystem->AccessPathName(tmpdir_filepath)) {
384  // copy file to temp dir
385  CopyFileFromRepository(ds, type, filename, tmpdir_filepath);
386  }
387 
388  return OpenDataSetFile(ds, type, tmpdir_filepath, opt);
389 }

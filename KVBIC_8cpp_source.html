25 //KVBIC
26 //Blocking Ionisation Chambers (BIC) for E416/E416a
27 //
28 //Here is the structure used to describe the Blocking Ionisation Chambers:
29 //Begin_Html
30 //<img src="http://indra.in2p3.fr/KaliVedaDoc/images/bic.gif">
31 //End_Html
32 //
33 //structure as absorbers:
34 //GetAbsorber(0) - 1.5µm mylar entrance window
35 //GetAbsorber(1) - CF4 at x Torr with thickness given by GetEffectiveEntryThickness()
36 //GetAbsorber(2) - 1 µm mylar internal window
37 //GetAbsorber(3) - 60 mm of CF4 at x Torr
38 //GetAbsorber(4) - 1 µm mylar internal window
39 //GetAbsorber(5) - CF4 at x Torr with thickness given by GetEffectiveEntryThickness()
40 //GetAbsorber(6) - 1.5µm mylar exit window
41 //
42 //Bombage:
43 // La valeur du bombage des fenetres externes a ete "estimee" par C. Mazur
44 // pour des fenetres neuves. Il est peut-etre prudent de la passer a 5mm,
45 // ce qui conduirait a une epaisseur moyenne de 4mm environ de CF4 a
46 // l'entree.
47 //
48 //The bombage can be set/modified using SetBombage(). The default value is 5mm.
49 //The effective thickness of CF4 between the two mylar windows, "averaged" over the whole surface of the
50 //entry window, is then "estimated" according to the formula (see GetEffectiveEntryThickness() ):
51 // e = bombage/2 + 1.5 [mm]
52 //With the default bombage of 5mm, this gives e=4mm.

117 KVBIC::KVBIC(Float_t pressure, Float_t bomb): KVChIo()
118 {
119  //BIC detector.
120  //Give pressure in Torr
121  //
122  //Segmentation index = 1
123  //because particles passing through BIC can only hit 1 detector - SIB - behind it
124  //this is analogous to the Si-CsI telescopes of ring 1.
125  //The type of these detectors is "BIC"
126  //Pressure units are Torr
127 
128  SetType("BIC");
129  SetSegment(1);
130  fLinCal = 0;
131 
132  fBomb = bomb;
133  //build detector
134  AddAbsorber(new KVMaterial("Myl", 1.5 * KVUnits::um)); //mylar entry window
135  Float_t e = GetEffectiveEntryThickness();
136  KVMaterial* gas = new KVMaterial("CF4", e * KVUnits::mm, pressure);
137  AddAbsorber(gas); //gas between two windows
138  AddAbsorber(new KVMaterial("Myl", 1.0 * KVUnits::um)); //interior window
139  gas = new KVMaterial("CF4", 60 * KVUnits::mm, pressure);
140  AddAbsorber(gas); //main body of gas
141  SetActiveLayer(gas); //active layer - energy loss is measured
142  AddAbsorber(new KVMaterial("Myl", 1.0 * KVUnits::um)); //2nd interor window
143  gas = new KVMaterial("CF4", e * KVUnits::mm, pressure);
144  AddAbsorber(gas); //gas between two exit windows
145  AddAbsorber(new KVMaterial("Myl", 1.5 * KVUnits::um)); //exit window
146 
147 }

265 Double_t KVBIC::GetCorrectedEnergy(KVNucleus* nuc, Double_t e, Bool_t transmit)
266 {
267  //For a particle which passes through the BIC (i.e. is stopped in SIB behind it)
268  //this will give the total energy loss of the particle in the BIC, i.e. including all
269  //the dead zones.
270  //If the particle stops in the BIC, this will not be correct, but in this case no
271  //identification of the particle is possible, Z and A will not be known, so
272  //this function should not be called.
273  //If argument 'egas' not given, KVBIC::GetEnergy() is used
274 
275  if (!transmit) return 0.;
276 
277  e = ((e < 0.) ? GetEnergy() : e);
278 
279  if (e <= 0.)
280  return 0.0; //check measured (calibrated) energy in gas is reasonable (>0)
281 
282  Bool_t maxDE = kFALSE;
283 
284  //egas > max possible ?
285  if (e > GetMaxDeltaE(nuc->GetZ(), nuc->GetA())) {
286  e = GetMaxDeltaE(nuc->GetZ(), nuc->GetA());
287  maxDE = kTRUE;
288  }
289  //calculate incident energy
290  Double_t e_inc = GetIncidentEnergy(nuc->GetZ(), nuc->GetA(), e);
291 
292  //calculate residual energy
293  Double_t e_res = GetERes(nuc->GetZ(), nuc->GetA(), e_inc);
294 
295  Double_t eloss = e_inc - e_res;
296  if (maxDE)
297  return (-eloss);
298  return eloss;
299 }

396 Double_t KVBIC::GetEnergy()
397 {
398  //Redefinition of KVChIo::GetEnergy().
399  //If energy lost in active layer is already set (e.g. by calculation of energy loss
400  //of charged particles), return its value.
401  //If not, we calculate it and set it using the values read from coders
402  //and the linear channel-energy calibration if initialised
403  //
404  //Returns -99 if (i) no calibration present (ii) calibration present but no data in acquisition parameters
405 
406  //fELoss already set, return its value
407  if (KVDetector::GetEnergy())
408  return KVDetector::GetEnergy();
409  Double_t ELoss = -99;
410  //is detector calibration available and ready ?
411  if (IsCalibrated()) {
412  ELoss = fLinCal->Compute(-1);
413  if (ELoss > -99.) {
414  SetEnergy(ELoss);
415  }
416  }
417  return ELoss;
418 }

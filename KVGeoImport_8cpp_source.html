22 <h2>KVGeoImport</h2>
23 <h4>Import a ROOT geometry into a KVMultiDetArray object</h4>
24 <!-- */
25 // --> END_HTML
26 //
27 // To use:
28 // KVGeoImport geo(gGeoManager, new KVedaLoss, new KVMultiDetArray);
29 // gImp.ImportGeometry();
30 //
31 // RULES FOR VALID DETECTOR GEOMETRY
32 //
33 // 1.) All detector volumes & nodes must have names which begin with "DET_"
34 //
35 // 2.) They must be made of materials which are known by the range table fRangeTable.
36 //
37 // 3.) For multi-layer detectors, the "active" layer volume/node must have a name beginning with "ACTIVE_"
38 //
39 // 4.) The "thickness" of the detector or any layer inside a multilayer detector
40 // will be taken as the size of the volume's shape along its Z-axis
41 // (so make sure that you define your detector volumes in this way).
42 //
43 // 5.) It is assumed that the natural length units of the geometry are centimetres.
44 //
45 // 6.) The name of the KVDetector object created and added to the array will be taken
46 // from the unique full path of the node corresponding to the geometrical positioning
47 // of the detector, see KVGeoNavigator::ExtractDetectorNameFromPath
48 //
49 // STRUCTURE ELEMENTS
50 // A structure element is any association of detectors which may occur several times
51 // in a geometry in different places, with the same internal structure.
52 // For example: a wall of detectors in a rectangular array.
53 // The detectors in the wall may all be represented by a single volume (if they
54 // are identical) called "DET_SI", for example, which is placed N times in N
55 // different positions on nodes called, by default, "DET_SI_1", "DET_SI_2", etc.
56 // Once you have defined the wall volume, call it "STRUCT_SIWALL" (or whatever you
57 // want: only the "STRUCT_" part is mandatory).
58 // Now you can place the same wall structure in different positions of your
59 // geometry on nodes called "STRUCT_SIWALL_1", "STRUCT_SIWALL_2", etc.
60 // This will give a full path to the node of DET_SI_1 in SIWALL_2 like this:
61 // "/TOP_1/STRUCT_SIWALL_2/DET_SI_1"
62 // which will be interpreted by KVGeoNavigator::ExtractDetectorNameFromPath as a
63 // detector with unique name:
64 // "SIWALL_2_SI_1"
65 // (the "STRUCT_" and "DET_" parts are stripped off).

151 void KVGeoImport::ParticleEntersNewVolume(KVNucleus*)
152 {
153  // All detectors crossed by the particle's trajectory are added to the multidetector
154  // and the groups (KVGroup) of aligned detectors are set up
155 
156  KVDetector* detector = GetCurrentDetector();
157  if (!detector) return;
158  Bool_t group_inconsistency = kFALSE;
159  if (fCreateArray) {
160  if (!fCurrentGroup) {
161  if (detector->GetGroup()) {
162  fCurrentGroup = detector->GetGroup();
163  } else {
164  fCurrentGroup = new KVGroup;
165  fCurrentGroup->SetNumber(++fGroupNumber);
166  fCurrentGroup->Add(detector);
167  fArray->Add(fCurrentGroup);
168  }
169  } else {
170  KVGroup* det_group = detector->GetGroup();
171  if (!det_group) {
172  fCurrentGroup->Add(detector);
173  } else {
174  if (det_group != fCurrentGroup) {
175 // Warning("ParticleEntersNewVolume",
176 // "Detector %s : already belongs to %s, now seems to be in %s",
177 // detector->GetName(), det_group->GetName(),
178 // fCurrentGroup->GetName());
179  group_inconsistency = kTRUE;
180  }
181  }
182  }
183  }
184  detector->GetNode()->SetName(detector->GetName());
185  if (fLastDetector && detector != fLastDetector && !group_inconsistency) {
186  fLastDetector->GetNode()->AddBehind(detector);
187  detector->GetNode()->AddInFront(fLastDetector);
188  fCurrentTrajectory.AddLast(fLastDetector->GetNode());
189  }
190  fLastDetector = detector;
191 }

203 void KVGeoImport::ImportGeometry(Double_t dTheta, Double_t dPhi,
204  Double_t ThetaMin, Double_t PhiMin, Double_t ThetaMax, Double_t PhiMax)
205 {
206  // Scan the geometry in order to find all detectors and detector alignments.
207  // This is done by sending out "particles" from (0,0,0) in all directions between
208  // (ThetaMin,ThetaMax) - with respect to Z-axis - and (PhiMin,PhiMax) - cylindrical
209  // angle in the (X,Y)-plane, over a grid of step dTheta in Theta and dPhi in Phi.
210 
211  // note that ImportGeometry can be called for a KVMultiDetArray
212  // which already contains detectors, groups and id telescopes
213  fGroupNumber = unique_ptr<KVSeqCollection>(fArray->GetStructureTypeList("GROUP"))->GetEntries();
214  Int_t ndets0 = fArray->GetDetectors()->GetEntries();
215 
216  unique_ptr<KVEvent> evt(new KVEvent());
217  KVNucleus* nuc = evt->AddParticle();
218  nuc->SetZAandE(1, 1, 1);
219 
220  Info("ImportGeometry",
221  "Importing geometry in angular ranges : Theta=[%f,%f:%f] Phi=[%f,%f:%f]", ThetaMin, ThetaMax, dTheta, PhiMin, PhiMax, dPhi);
222  Int_t count = 0;
223  if (!KVDataAnalyser::IsRunningBatchAnalysis())
224  std::cout << "\xd" << "Info in <KVGeoImport::ImportGeometry>: tested " << count << " directions" << std::flush;
225  for (Double_t theta = ThetaMin; theta <= ThetaMax; theta += dTheta) {
226  for (Double_t phi = PhiMin; phi <= PhiMax; phi += dPhi) {
227  nuc->SetTheta(theta);
228  nuc->SetPhi(phi);
229  fCurrentGroup = nullptr;
230  fLastDetector = nullptr;
231  PropagateEvent(evt.get());
232  count++;
233  if (!KVDataAnalyser::IsRunningBatchAnalysis())
234  std::cout << "\xd" << "Info in <KVGeoImport::ImportGeometry>: tested " << count << " directions" << std::flush;
235  }
236  }
237  if (KVDataAnalyser::IsRunningBatchAnalysis())
238  std::cout << "Info in <KVGeoImport::ImportGeometry>: tested " << count << " directions" << std::endl;
239  else
240  std::cout << std::endl;
241 
242  Info("ImportGeometry",
243  "Imported %d detectors into array", fArray->GetDetectors()->GetEntries() - ndets0);
244 
245  // make sure detector nodes are correct
246  TIter next(fArray->GetDetectors());
247  KVDetector* d;
248  while ((d = (KVDetector*)next())) {
249  d->GetNode()->RehashLists();
250  d->SetNameOfArray(fArray->GetName());
251  }
252  // set up all detector node trajectories
253  fArray->AssociateTrajectoriesAndNodes();
254 
255  if (fCreateArray) {
256  fArray->SetGeometry(GetGeometry());
257  // Set up internal navigator of array with all informations on detector/
258  // structure name formatting, correspondance lists, etc.
259  KVGeoNavigator* nav = fArray->GetNavigator();
260  nav->SetDetectorNameFormat(fDetNameFmt);
261  for (int i = 0; i < fStrucNameFmt.GetEntries(); i++) {
262  KVNamedParameter* fmt = fStrucNameFmt.GetParameter(i);
263  nav->SetStructureNameFormat(fmt->GetName(), fmt->GetString());
264  }
265  nav->SetNameCorrespondanceList(fDetStrucNameCorrespList);
266  nav->AbsorbDetectorPaths(this);
267  fArray->CalculateDetectorSegmentationIndex();
268  fArray->DeduceIdentificationTelescopesFromGeometry();
269  fArray->CalculateReconstructionTrajectories();
270  }
271 }

320 KVDetector* KVGeoImport::GetCurrentDetector()
321 {
322  // Returns pointer to KVDetector corresponding to current location
323  // in geometry. Detector is created and added to array if needed.
324  // We also set up any geometry structure elements (from nodes beginning with "STRUCT_")
325 
326  KVString detector_name;
327  Bool_t multilay;
328  TGeoVolume* detector_volume = GetCurrentDetectorNameAndVolume(detector_name, multilay);
329  // failed to identify current volume as part of a detector
330  if (!detector_volume) return 0;
331 
332  // has detector already been built ? if not, do it now
333  KVDetector* det = fArray->GetDetector(detector_name);
334  if (!fCreateArray) {
335  if (det) {
336  // set matrix & shape for entrance window if not done yet
337  if (!det->GetEntranceWindowMatrix()) {
338  det->SetEntranceWindowMatrix(GetCurrentMatrix());
339  det->SetEntranceWindowShape((TGeoBBox*)GetCurrentVolume()->GetShape());
340  }
341  TString vol_name(GetCurrentVolume()->GetName());
342  if (!multilay || vol_name.BeginsWith("ACTIVE_")) {
343  // set matrix & shape for active layer
344  det->SetActiveLayerMatrix(GetCurrentMatrix());
345  det->SetActiveLayerShape((TGeoBBox*)GetCurrentVolume()->GetShape());
346  // set full path to physical node as title of detector's node (KVGeoDetectorNode)
347  det->GetNode()->SetTitle(GetCurrentPath());
348  }
349  // add entry to correspondance list between physical nodes and detectors (all layers)
350  fDetectorPaths.Add(new KVGeoDetectorPath(GetCurrentPath(), det));
351  }
352  } else {
353  if (!det) {
354  det = BuildDetector(detector_name, detector_volume);
355  if (det) {
356  // Setting the entrance window shape and matrix
357  // ============================================
358  // for consistency, the matrix and shape MUST correspond
359  // i.e. we cannot have the matrix corresponding to the entrance window
360  // of a multilayer detector and the shape corresponding to the
361  // whole detector (all layers) - otherwise, calculation of points
362  // on detector entrance window will be false!
363 // Info("GetCurrentDetector","Setting EW matrix to current matrix:");
364 // GetCurrentMatrix()->Print();
365  det->SetEntranceWindowMatrix(GetCurrentMatrix());
366  det->SetEntranceWindowShape((TGeoBBox*)GetCurrentVolume()->GetShape());
367  TString vol_name(GetCurrentVolume()->GetName());
368  if (!multilay || vol_name.BeginsWith("ACTIVE_")) {
369  // first layer of detector (or only layer) is also active layer
370 // Info("GetCurrentDetector","and also setting active layer matrix to current matrix:");
371 // GetCurrentMatrix()->Print();
372  det->SetActiveLayerMatrix(GetCurrentMatrix());
373  det->SetActiveLayerShape((TGeoBBox*)GetCurrentVolume()->GetShape());
374  // set full path to physical node as title of detector's node (KVGeoDetectorNode)
375  det->GetNode()->SetTitle(GetCurrentPath());
376  }
377  // add entry to correspondance list between physical nodes and detectors (all layers)
378  fDetectorPaths.Add(new KVGeoDetectorPath(GetCurrentPath(), det));
379  fArray->Add(det);
380  Int_t nstruc = CurrentStructures().GetEntries();
381  if (nstruc) {
382  // Build and add geometry structure elements
383  KVGeoStrucElement* ELEM = fArray;
384  for (int i = 0; i < nstruc; i++) {
385  KVGeoStrucElement* elem = (KVGeoStrucElement*)CurrentStructures()[i];
386  KVGeoStrucElement* nextELEM = ELEM->GetStructure(elem->GetName());
387  if (!nextELEM) {
388  // make new structure
389  nextELEM = new KVGeoStrucElement(elem->GetName(), elem->GetType());
390  nextELEM->SetNumber(elem->GetNumber());
391  ELEM->Add(nextELEM);
392  }
393  ELEM = nextELEM;
394  }
395  // add detector to last structure
396  ELEM->Add(det);
397  }
398  }
399  } else {
400  // Detector already built, are we now in its active layer ?
401  TString vol_name(GetCurrentVolume()->GetName());
402  if (!multilay || vol_name.BeginsWith("ACTIVE_")) {
403 // Info("GetCurrentDetector","Setting active layer matrix to current matrix:");
404 // GetCurrentMatrix()->Print();
405  det->SetActiveLayerMatrix(GetCurrentMatrix());
406  det->SetActiveLayerShape((TGeoBBox*)GetCurrentVolume()->GetShape());
407  // set full path to physical node as title of detector's node (KVGeoDetectorNode)
408  det->GetNode()->SetTitle(GetCurrentPath());
409  }
410  // add entry to correspondance list between physical nodes and detectors (all layers)
411  fDetectorPaths.Add(new KVGeoDetectorPath(GetCurrentPath(), det));
412  }
413  }
414  return det;
415 }

450 KVDetector* KVGeoImport::BuildDetector(TString det_name, TGeoVolume* det_vol)
451 {
452  // Create a KVDetector with given name for the given volume
453  //
454  // Detector definition in geometry
455  // ===============================
456  // 1.) All detector volumes & nodes must have names which begin with "DET_"
457  //
458  // 2.) They must be made of materials which are known by the range table fRangeTable.
459  //
460  // 3.) For multi-layer detectors, the "active" layer volume/node must have a name beginning with "ACTIVE_"
461  //
462  // 4.) The "thickness" of the detector or any layer inside a multilayer detector
463  // will be taken as the size of the volume's shape along its Z-axis
464  // (so make sure that you define your detector volumes in this way).
465  //
466  // 5.) It is assumed that the natural length units of the geometry are centimetres.
467  //
468  // 6.) The name of the KVDetector object created and added to the array will be taken
469  // from the unique full path of the node corresponding to the geometrical positioning
470  // of the detector, see KVGeoNavigator::ExtractDetectorNameFromPath
471  //
472  // 7.) The 'type' of the detector will be set to the name of the material
473  // in the detector's active layer i.e. if active layer material name is "Si",
474  // detector type will be 'Si'
475  //
476  // 8.) Default class for all detectors is KVDetector
477  // if you want to use an other class
478  // you need to defined it using SetDetectorPlugin method and put the
479  // associated line in your .kvrootrc configuration file. This plugin
480  // has to be loaded by your KVMultiDetArray object
481 
482 
483  KVDetector* d = 0;
484  TPluginHandler* ph = NULL;
485  if (fDetectorPlugin == "" || !(ph = LoadPlugin("KVDetector", fDetectorPlugin))) {
486  d = new KVDetector;
487  } else {
488  d = (KVDetector*)ph->ExecPlugin(0);
489  }
490 
491  d->SetName(det_name);
492 
493  Int_t nlayer = det_vol->GetNdaughters();
494  if (nlayer) {
495  for (int i = 0; i < nlayer; i++) {
496  AddLayer(d, det_vol->GetNode(i)->GetVolume());
497  }
498  } else
499  AddLayer(d, det_vol);
500  TString type = d->GetActiveLayer()->GetName();
501  //type.ToUpper();
502  d->SetType(type);
503  return d;
504 }

513 void KVGeoImport::AddLayer(KVDetector* det, TGeoVolume* vol)
514 {
515  // Add an absorber layer to the detector
516  // Volumes representing 'active' layers in detectors must have names
517  // which begin with "ACTIVE_"
518 
519  TString vnom = vol->GetName();
520  // exclude dead zone layers
521  if (vnom.BeginsWith("DEADZONE")) return;
522  TGeoMaterial* material = vol->GetMaterial();
523  KVIonRangeTableMaterial* irmat = fRangeTable->GetMaterial(material);
524  if (!irmat) {
525  Warning("AddLayer", "Unknown material %s/%s used in layer %s of detector %s",
526  material->GetName(), material->GetTitle(), vol->GetName(), det->GetName());
527  return;
528  }
529  TGeoBBox* sh = dynamic_cast<TGeoBBox*>(vol->GetShape());
530  if (!sh) {
531  Warning("AddLayer", "Unknown shape class %s used in layer %s of detector %s",
532  vol->GetShape()->ClassName(), vol->GetName(), det->GetName());
533  return; // just in case - for now, all shapes derive from TGeoBBox...
534  }
535  Double_t width = 2.*sh->GetDZ(); // thickness in centimetres
536  KVMaterial* absorber;
537  if (irmat->IsGas()) {
538  Double_t p = material->GetPressure();
539  Double_t T = material->GetTemperature();
540  absorber = new KVMaterial(irmat->GetType(), width, p, T);
541  } else
542  absorber = new KVMaterial(irmat->GetType(), width);
543  det->AddAbsorber(absorber);
544  if (vnom.BeginsWith("ACTIVE_")) det->SetActiveLayer(absorber);
545 }

87 void KVTextEntry::Insert(const char* newText)
88 {
89  // Removes any currently selected text, inserts newText,
90  // sets it as the new contents of the text entry.
91 
92  TString old(GetText());
93  TString t(newText);
94 
95  if (t.IsNull()) return;
96 
97  for (int i = 0; i < t.Length(); i++) {
98  if (t[i] < ' ') t[i] = ' '; // unprintable/linefeed becomes space
99  }
100 
101  Int_t minP = MinMark();
102  Int_t maxP = MaxMark();
103  Int_t cp = fCursorIX;
104 
105  if (HasMarkedText()) {
106  fText->RemoveText(minP, maxP - minP);
107  cp = minP;
108  }
109 
110  if (fInsertMode == kReplace) fText->RemoveText(cp, t.Length());
111  Int_t ncp = TMath::Min(cp + t.Length(), GetMaxLength());
112  fText->AddText(cp, t.Data());
113

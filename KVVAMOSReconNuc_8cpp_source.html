224 void KVVAMOSReconNuc::Calibrate()
225 {
226  //Calculate and set the energy of a (previously identified) reconstructed nucleus,
227  //including an estimate of the energy loss in the stripping foil and in the target.
228  //
229  //Starting from the detector in which the nucleus stopped, we add up the
230  //'corrected' energy losses in all of the detectors through which it passed.
231  //Whenever possible, for detectors which are not calibrated or not working,
232  //we calculate the energy loss.
233  //
234  //For nuclei whose energy before hitting the first detector in their path has been
235  //calculated after this step we then add the calculated energy loss in stripping foil and in the target,
236  //using:
237  // gVamos->GetStripFoilEnergyLossCorrection();
238  // gMultiDetArray->GetTargetEnergyLossCorrection().
239 
240  if (gVamos->Calibrate(this)) {
241  // if the nucleus is calibrated by this way then the energy losses
242  // in each detector are not calculated then they are set to -1
243  if (fdebug) Info("Calibrate", "... gVamos->Calibrate() was success, CalibrateFromDetList() won't be called ...");
244  if (fDetE.empty()) {
245  Int_t N = GetDetectorList()->GetEntries();
246  fDetE.resize(N, -1.);
247  }
248  } else {
249  if (fdebug) Info("Calibrate", "... gVamos->Calibrate() wasn't success, CalibrateFromDetList() will be called ...");
250  CalibrateFromDetList();
251  }
252 
253 
254  if (IsCalibrated() && GetEnergy() > 0) {
255 
256  // set angles of momentum from trajectory reconstruction
257  if (fRT.FPtoLabWasAttempted()) {
258  SetTheta(GetThetaI());
259  SetPhi(GetPhiI());
260  }
261 
262  if (GetZ()) {
263 
264  Double_t E_tot = GetEnergy();
265  Double_t E_sfoil = 0.;
266  Double_t E_targ = 0.;
267 
268  //add correction for stripping foil energy loss - moving charged particles only
269  E_sfoil = gVamos->GetStripFoilEnergyLossCorrection(this);
270  SetStripFoilEnergyLoss(E_sfoil);
271  SetEnergy(E_tot += E_sfoil);
272 
273  //add correction for target energy loss - moving charged particles only
274  if (gMultiDetArray->GetTarget()) {
275  E_targ = gMultiDetArray->GetTargetEnergyLossCorrection(this);
276  SetTargetEnergyLoss(E_targ);
277  SetEnergy(E_tot += E_targ);
278  }
279  }
280  }

306 void KVVAMOSReconNuc::CalibrateFromDetList()
307 {
308  // The energy of each particle is calculated as follows:
309  //
310  // E = dE_1 + dE_2 + ... + dE_N
311  //
312  // dE_1, dE_2, ... = energy losses measured in each detector through which
313  // the particle has passed (or stopped, in the case of dE_N).
314  // These energy losses are corrected for (Z,A)-dependent effects
315  // such as pulse-heigth defect in silicon detectors, losses in
316  // windows of gas detectors, etc.
317  //
318  // If none of the detectors of the detector list is calibrated, the particle's energy cannot be calculated &
319  // the code will be kECode0.
320  // Whenever possible, the energy loss for fired detectors which are uncalibrated
321  // or not functioning is calculated.
322  // If the detectors used to measure energy are not calibrated or not fired, or multihit, the
323  // energy code will be kECode2.
324  // Otherwise, the code will be kECode1.
325  // The flag returned by IsCalibrated will be true is the energy code is different from kECode0.
326 
327  //status code
328  SetECode(kECode0);
329 
330  //uncharged particles
331  if (GetZ() == 0) return;
332 
333  SetECode(kECode1);
334  SetIsCalibrated();
335 
336  Double_t Etot = 0;
337  Bool_t stopdetOK = kTRUE;
338  KVVAMOSDetector* stopdet = (KVVAMOSDetector*)GetStoppingDetector();
339  KVVAMOSDetector* det = NULL;
340  TIter next_det(GetDetectorList());
341  if (fDetE.empty()) fDetE.resize(GetDetectorList()->GetEntries(), 0.);
342  Int_t idx = -1;
343  while ((det = (KVVAMOSDetector*)next_det())) {
344  fDetE[++idx] = 0.;
345  if (!stopdetOK) continue;
346  // transmission=kFALSE if particle stop in det
347  Bool_t transmission = (det != stopdet);
348 
349  // The stopping detector has to be fired, calibrated, hit by only one
350  // particle
351  if (!transmission && !((det->GetNHits() == 1) && det->Fired() && det->IsECalibrated())) {
352  SetECode(kECode0);
353  SetIsUncalibrated();
354  stopdetOK = kFALSE;
355  continue;
356  }
357 
358  // modify thicknesses of all layers according to orientation,
359  // and store original thicknesses in array. Effective thickness
360  // is previously calculated in the method 'Propagate' and stored
361  // in the list 'fParameters'
362  Double_t* thickness;
363  thickness = new Double_t[det->GetListOfAbsorbers()->GetEntries()];
364  KVMaterial* abs;
365  Int_t i = 0;
366  TIter next_abs(det->GetListOfAbsorbers());
367  TIter next_par(GetParameters()->GetList());
368  TString parname, absname;
369  while ((abs = (KVMaterial*) next_abs())) {
370  thickness[i++] = abs->GetThickness();
371  absname = abs->GetAbsGeoVolume()->GetName();
372  if (absname.BeginsWith("DET_")) parname.Form("STEP:%s", det->GetName());
373  else parname.Form("STEP:%s/%s", det->GetName(), absname.Data());
374  KVNamedParameter* par = NULL;
375  while ((par = (KVNamedParameter*)next_par())) {
376  if (!strncmp(parname.Data(), par->GetName(), parname.Sizeof() - 1)) {
377  abs->SetThickness(par->GetDouble());
378 // Info("CalibrateFromDetList","thickness of absorber %s has been changed: %f --> %f", parname.Data(), thickness[i-1], par->GetDouble());
379  break;
380  }
381  }
382  }
383 
384  det->SetEResAfterDetector(Etot);
385  Double_t Edet = det->GetCorrectedEnergy(this, -1, transmission);
386 
387  // Detector is calibrated and hit by only one particle
388  if (Edet > 0 && det->GetNHits() == 1) {
389  det->SetEResAfterDetector(Etot);
390 
391  // if Edet is greater than the maximal possible theoretical value
392  // this may be because Z or A are underestimated and one of both
393  // has to be increased (see KVDetector::GetCorrectedEnergy)
394  if (GetParameters()->GetIntValue("GetCorrectedEnergy.Warning") == 1) {
395  Int_t z0 = GetZ();
396  Int_t a0 = GetA();
397  Int_t z = z0, a = a0;
398  Bool_t isOK = kFALSE;
399  Edet = det->GetEnergy();
400 
401  //Warning("CalibrateFromDetList", "MeasuredDE>MaxDE in %s, Z= %d, A=%d, MeasuredDE= %f MeV, MaxDE %f MeV", det->GetName(), z0, a0, Edet, det->GetMaxDeltaE(z0, a0));
402 
403  while (!isOK && (z - GetZ() <= 10)) {
404  //increase Z
405  ++z;
406  //modify A only if it is not measured
407  if (!IsAMeasured()) a = GetAFromZ(z, (Char_t)GetMassFormula());
408  isOK = (Edet <= det->GetMaxDeltaE(z, a));
409  //Info("CalibrateFromDetList", "changing Z= %d, A=%d, MeasuredDE= %f MeV, MaxDE %f MeV", z, a, Edet, det->GetMaxDeltaE(z, a));
410  }
411  //Info("CalibrateFromDetList", "MeasuredDE<MaxDE in %s by changing Z: %d->%d, A: %d->%d \n", det->GetName(), z0 , z, a0 , a);
412  SetZandA(z, a);
413  det->SetEResAfterDetector(Etot);
414  Edet = det->GetCorrectedEnergy(this, -1, transmission);
415  // we back to initial values of Z and A
416  SetZandA(z0, a0);
417  }
418 
419  Etot += Edet;
420  fDetE[idx] = Edet;
421 // Info("CalibrateFromDetList","Corrected DeltaE= %f in %s, idx= %d", Edet, det->GetName(), idx);
422  }
423  // Detector is uncalibrated/unfired/multihit. Calculate energy loss.
424  // calculate energy of particle before detector from energy after
425  // detector
426  else {
427 
428  Edet = det->GetDeltaEFromERes(GetZ(), GetA(), Etot);
429  if (Edet < 0.0) Edet = 0.0;
430 
431  if (det->GetNHits() > 1) {
432  if (!(det->Fired() && det->IsECalibrated()))
433  det->SetEnergyLoss(Edet + det->GetEnergy());// sum up calculated energy losses in uncalibrated detector
434  //Info("CalibrateFromDetList", "MultiHit in %s", det->GetName());
435  } else if (!det->Fired() || !det->IsECalibrated())
436  det->SetEnergyLoss(Edet);
437 
438  det->SetEResAfterDetector(Etot);
439  Edet = det->GetCorrectedEnergy(this, Edet, transmission);
440  Etot += Edet;
441  fDetE[idx] = Edet;
442  if (det->IsUsedToMeasure("E")) SetECode(kECode2);
443  //Info("CalibrateFromDetList", "Calculated DeltaE= %f in %s, idx= %d", Edet, det->GetName(), idx);
444  }
445 
446  // reset thicknesses of absorbers
447  i = 0;
448  next_abs.Reset();
449  while ((abs = (KVMaterial*) next_abs())) {
450  abs->SetThickness(thickness[i++]);
451  }
452  delete [] thickness;
453  }
454  SetEnergy(Etot);
455 }

744 {
745  // VAMOS-specific Q and A identification.
746  //
747  // +e494s: First we loop over each Q-A ID telescope placed at the front
748  // of the stopping detector. For each ID telescope we loop
749  // over each time of flight listed in the environment variable
750  // KVVAMOSCodes.ACQParamListForToF to look for the corresponding
751  // ID grid before to perform identification. This process is stopped
752  // once an identification is OK. Then the corresponding TCode and
753  // the identification result are set to this nucleus
754  //
755  // +e503:
756 
757  //Raw identification:
758  //Apply VAMOS routine identification
759  //without any application of data corrections
760  //and save the associated informations.
761  Bool_t ok = kFALSE;
762  KVSeqCollection* idt_list = GetIDTelescopes();
763  if (idt_list && idt_list->GetSize() > 0) {
764 
765  KVIDTelescope* idt;
766  TIter next(idt_list);
767  while ((idt = (KVIDTelescope*) next())) {
768 
769  // is telescope able to identify for this run ?
770  if (!idt->IsReadyForID()) continue;
771

904 void KVVAMOSReconNuc::ReconstructFPtraj()
905 {
906  // Reconstruct the Focal-Plane trajectory and set the appropriate
907  // FPCode. If the reconstruction is well carried out (i.e. FPCode>kFPCode0)
908  // then the tracking along this trajectory is runned.
909 
910  UChar_t res = 0;
911  Double_t xyzf[3]; // [3] => [X, Y, Z]
912  Double_t XYZf[4][3]; // [4] => [Complete Det1, Complete Det2, Incomplete Det1, Incomplete Det2]
913  // [3] => [X, Y, Z]
914  Short_t Idx[4] = { -1, -1, -1, -1}; // [4] => [Complete Det1, Complete Det2, Incomplete Det1, Incomplete Det2]
915  UChar_t IncDetBitmask = 3; // bits 11 => 3, first bit means missing X, second bit means missing Y
916  Bool_t inc1IsX = kTRUE; // true if the coordinate given by the first incomplete position is X ( false if Y)
917  UShort_t Ncomp = 0, Ninc = 0; // Number of Complete and Incomplete
918  // position measurments
919 
920  SetFPCode(kFPCode0); // Initialize FP codes to code 0 "no FP position recon."
921  fRT.dirFP.SetXYZ(0, 0, 1);
922 
923  const Char_t* FPdetName = NULL;
924  KVVAMOSDetector* d = NULL;
925  // Loop over detector name to be used for the Focal plane Position
926  // reconstruction
927  for (Short_t i = 0; (FPdetName = GetCodes().GetFPdetName(i)); i++) {
928 
929  // Look at only the detectors in 'fDetList' with the name or
930  // type 'FPdetName' measuring a position.
931  if (
932  ((d = (KVVAMOSDetector*)GetDetectorList()->FindObject(FPdetName)) ||
933  (d = (KVVAMOSDetector*)GetDetectorList()->FindObjectByType(FPdetName)))
934  && (res = d->GetPosition(xyzf)) > 4
935  ) {
936 
937 
938  if (res == 7) {
939  // X, Y and Z measured bits 111 => 7
940  // Complete position (XYZ)
941 
942  Idx[ Ncomp ] = i;
943  for (Int_t j = 0; j < 3; j++) XYZf[ Ncomp ][ j ] = xyzf[ j ];
944  Ncomp++;
945  } else if (IncDetBitmask & res) {
946  // look at only incomplete position measurment (XZ or YZ)
947  // not found yet
948 
949  Idx[ 2 + Ninc ] = i;
950  for (Int_t j = 0; j < 3; j++) XYZf[ 2 + Ninc ][ j ] = xyzf[ j ];
951  if (Ninc == 0) inc1IsX = res & 1;
952  Ninc++;
953  IncDetBitmask = (IncDetBitmask | 4)^res;
954  }
955 
956 
957  if (Ncomp == 2) {
958  //Case where the Focal plane Position is reconstructed from 2 complete position measurment.
959 
960  fRT.dirFP.SetX((XYZf[1][0] - XYZf[0][0]) / (XYZf[1][2] - XYZf[0][2]));
961  fRT.dirFP.SetY((XYZf[1][1] - XYZf[0][1]) / (XYZf[1][2] - XYZf[0][2]));
962  GetCodes().SetFPCode(Idx[0], Idx[1], Idx[2], Idx[3], inc1IsX);
963  break;
964  } else if ((Ncomp == 1) && !IncDetBitmask) {
965  //Case where the Focal plane Position is reconstructed from 1 complete position measurment and 2 incomplete position measurment.
966 
967  fRT.dirFP.SetX((XYZf[3 - inc1IsX][0] - XYZf[0][0]) / (XYZf[3 - inc1IsX][2] - XYZf[0][2]));
968  fRT.dirFP.SetY((XYZf[2 + inc1IsX][1] - XYZf[0][1]) / (XYZf[2 + inc1IsX][2] - XYZf[0][2]));
969  GetCodes().SetFPCode(Idx[0], Idx[1], Idx[2], Idx[3], inc1IsX);
970  break;
971  }
972 
973  }
974  }
975 
976  if (GetCodes().TestFPCode(kFPCode0)) return;
977 
978  // normalize the direction vector dirFP
979  fRT.dirFP *= 1. / fRT.dirFP.Mag();
980 
981  // Xf = Xc1 - Zc1*tan( Thetaf )
982  fRT.pointFP[0] = XYZf[0][0] - XYZf[0][2] * fRT.dirFP.X() / fRT.dirFP.Z();
983 
984  // Yf = Yc1 - Zc1*tan( Phif )
985  fRT.pointFP[1] = XYZf[0][1] - XYZf[0][2] * fRT.dirFP.Y() / fRT.dirFP.Z();
986 
987  fRT.SetFPparamsReady();
988 
989  // Info("ReconstructFPtraj","\n Xf= %f, Yf= %f, Thetaf= %f, Phif= %f\n FPCode%d %s", GetXf(), GetYf(),GetThetaF(),GetPhiF(),GetCodes().GetFPCodeIndex(), GetCodes().GetFPStatus());
990 }

1002 Bool_t KVVAMOSReconNuc::ReconstructFPtrajByFitting()
1003 {
1004  // Reconstruct the Focal-Plane trajectory and set the appropriate
1005  // FPCode. If the reconstruction is well carried out (i.e. FPCode>kFPCode0)
1006  // then the tracking along this trajectory is runned.
1007  // In this method, the reconstruction of the FP trajectory is done by
1008  // fitting a first-order polynomial on the positions measured by the
1009  // detector listed in the the environment variable: KVVAMOSCodes.FocalPlanReconDetList
1010 
1011  UChar_t res = 0;
1012  Double_t xyzf [3]; // [3] => [X, Y, Z]
1013  Double_t dxyzf[3]; // [3] => [X, Y, Z]
1014 
1015  Short_t Idx[4] = { -1, -1, -1, -1}; // [4] => [Complete Det1, Complete Det2, Incomplete Det1, Incomplete Det2]
1016  UChar_t IncDetBitmask = 3; // bits 11 => 3, first bit means missing X, second bit means missing Y
1017  Bool_t inc1IsX = kTRUE; // true if the coordinate given by the first incomplete position is X ( false if Y)
1018  UShort_t Ncomp = 0, Ninc = 0; // Number of Complete and Incomplete
1019 
1020  SetFPCode(kFPCode0); // Initialize FP codes to code 0 "no FP position recon."
1021 
1022  static TGraphErrors graphX(4); // 4 is the max number of FPdetectors
1023  static TGraphErrors graphY(4); // 4 is the max number of FPdetectors
1024  Int_t NptX = 0, NptY = 0; // Number of Points in X-Z and Y-Z plans
1025  const Char_t* FPdetName = NULL;
1026  KVVAMOSDetector* d = NULL;
1027  // Loop over the detector names to be used for the Focal plane Position
1028  // reconstruction
1029  for (Short_t i = 0; (FPdetName = GetCodes().GetFPdetName(i)) && i < 4; i++) {
1030 
1031  // Look at only the detectors in 'fDetList' with the name or
1032  // type 'FPdetName' measuring a position.
1033  if (
1034  ((d = (KVVAMOSDetector*)GetDetectorList()->FindObject(FPdetName)) ||
1035  (d = (KVVAMOSDetector*)GetDetectorList()->FindObjectByType(FPdetName)))
1036  ) {
1037 
1038  // initializing the result
1039  res = 0;
1040 
1041  // X position
1042  if (d->GetNMeasuredX() < 2) { // case where only one X position is measurable
1043  if ((d->GetPosition(xyzf, 'X') & 5) == 5) {
1044  d->GetDeltaXYZf(dxyzf, 'X');
1045  graphX.SetPoint(NptX, xyzf[2], xyzf [0]);
1046  graphX.SetPointError(NptX, 0, dxyzf[0] > 0.01 ? dxyzf[0] : 0.01);
1047  NptX++;
1048  res |= 5;
1049  }
1050  } else { // case where several X positions are measurable
1051  for (Int_t j = 1; j <= d->GetNMeasuredX(); j++) {
1052  // look at only well measured X position with its Z
1053  if ((d->GetPosition(xyzf, 'X', j) & 5) == 5) {
1054  d->GetDeltaXYZf(dxyzf, 'X', j);
1055  graphX.SetPoint(NptX, xyzf[2], xyzf [0]);
1056  graphX.SetPointError(NptX, 0, dxyzf[0] > 0.01 ? dxyzf[0] : 0.01);
1057  NptX++;
1058  res |= 5;
1059  }
1060  }
1061  }
1062 
1063  // Y position
1064  if (d->GetNMeasuredY() < 2) { // case where only one Y position is measurable
1065  if ((d->GetPosition(xyzf, 'Y') & 6) == 6) {
1066  d->GetDeltaXYZf(dxyzf, 'Y');
1067  graphY.SetPoint(NptY, xyzf[2], xyzf [1]);
1068  graphY.SetPointError(NptY, 0, dxyzf[1] > 0.01 ? dxyzf[1] : 0.01);
1069  NptY++;
1070  res |= 6;
1071  }
1072  } else { // case where several Y positions are measurable
1073  for (Int_t j = 1; j <= d->GetNMeasuredY(); j++) {
1074  // look at only well measured Y position with its Z
1075  if ((d->GetPosition(xyzf, 'Y', j) & 6) == 6) {
1076  d->GetDeltaXYZf(dxyzf, 'Y', j);
1077  graphY.SetPoint(NptY, xyzf[2], xyzf [1]);
1078  graphY.SetPointError(NptY, 0, dxyzf[1] > 0.01 ? dxyzf[1] : 0.01);
1079  NptY++;
1080  res |= 6;
1081  }
1082  }
1083  }
1084 
1085 
1086  if (res <= 4) continue; // look at only case with Z well determined
1087  else if ((res & 7) == 7) {
1088  // if X, Y and Z are well measured, bits 111 => 7
1089  // Complete position (XYZ)
1090 
1091  Idx[ Ncomp ] = i;
1092  Ncomp++;
1093  } else if (IncDetBitmask & res) {
1094  // look at only incomplete position measurment (XZ or YZ)
1095  // if it is not found yet
1096 
1097  Idx[ 2 + Ninc ] = i;
1098  if (Ninc == 0) inc1IsX = res & 1;
1099  Ninc++;
1100  IncDetBitmask = (IncDetBitmask | 4)^res;
1101  }
1102 
1103  if ((Ncomp == 2) || ((Ncomp == 1) && !IncDetBitmask)) {
1104  // we break the look if:
1105  // - the Focal plane Position is reconstructed from 2 complete position measurment
1106  // or if:
1107  // - the Focal plane Position is reconstructed from 1 complete position measurment and 2 incomplete position measurment.
1108 
1109  graphX.Set(NptX);
1110  graphY.Set(NptY);
1111 
1112  fRT.dirFP.SetXYZ(0, 0, 1);
1113  TF1* ff = (TF1*)gROOT->GetFunction("pol1");
1114 
1115  graphX.Fit(ff, "QNC");
1116  fRT.dirFP.SetX(ff->GetParameter(1));
1117  fRT.pointFP[0] = ff->GetParameter(0);
1118 
1119  graphY.Fit(ff, "QNC");
1120  fRT.dirFP.SetY(ff->GetParameter(1));
1121  fRT.pointFP[1] = ff->GetParameter(0);
1122 
1123  // normalize the direction vector dirFP
1124  fRT.dirFP *= 1. / fRT.dirFP.Mag();
1125 
1126  GetCodes().SetFPCode(Idx[0], Idx[1], Idx[2], Idx[3], inc1IsX);
1127 
1128  fRT.SetFPparamsReady();
1129 
1130  //Info("ReconstructFPtrajByFitting","\n Xf= %f, Yf= %f, Thetaf= %f, Phif= %f\n FPCode%d %s", GetXf(), GetYf(),GetThetaF(),GetPhiF(),GetCodes().GetFPCodeIndex(), GetCodes().GetFPStatus());
1131  break;
1132  }
1133  }
1134  }
1135  return !GetCodes().TestFPCode(kFPCode0);
1136 }

1583 Double_t KVVAMOSReconNuc::CalculateCorrectedT_HF(Double_t tof, Double_t dist, Int_t& nHF) const
1584 {
1585  // Returns the corrected time of flight obtained from beam pulse HF, by
1586  // removing or adding N times the beam pulse period. N is fitted by
1587  // minimizing the difference between the measured energy (Emeas) and the energy (E) deduced
1588  // from the velocity. The procedure is as follows:
1589  //
1590  // - let Begin_Latex T_{f} = T_{i} + #DeltaT End_Latex be the corrected time of flight
1591  // where Begin_Latex T_{i} End_Latex is the initial time of flight set as an argument;
1592  // - Begin_Latex #Delta T End_Latex is deduced by minimizing the quantity
1593  // Begin_Latex #chi = (E-E_{meas})^{2} End_Latex
1594  // i.e. we resolve the equation
1595  // Begin_Latex #frac{d#chi}{d#DeltaT} = 0 End_Latex.
1596  // The solution is
1597  // Begin_Latex #DeltaT = #frac{D}{c#sqrt{1-#gamma^{2}}} - T_{i} End_Latex where Begin_Latex #gamma = #frac{1}{#frac{E_{cal}}{mc^{2}}+1} End_Latex,
1598  // D is the flight distance, m is the mass returned by GetMass() and c is the speed of light in the vacuum;
1599  // - N if given by the rounded value of Begin_Latex #DeltaT/period_{HF} End_Latex
1600  //
1601  // This procedure can be applied only if the nucleus is identified and calibrated (i.e. IsCalibrated = true ).
1602  // Otherwise this method returns the initial tof.
1603  //
1604  // Arguments :
1605  // tof - initial time of flight in ns
1606  // dist - flight distance in cm
1607 
1608  if (!IsCalibrated() || (GetEnergy() <= 0)) return tof;
1609  if (gVamos->GetBeamPeriod() < 0) {

1612  }
1613

1615

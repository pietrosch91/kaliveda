132 Double_t KVI1::ComputeBaseLine()
133 {
134  // special case for current signal
135  // in case the pulse start to early,
136  // base line is calculated at the end of the signal
137 
138  KVSignal::ComputeBaseLine();
139  KVSignal::ComputeEndLine();
140 
141  if (fBaseLine <= fEndLine) {
142  //do nothing baseline is kept as calculated by default KVSignal method

144  //compute the base line at the end of the signal
145  //baseline length is taken as for the standard base line calculation
146  //
147  ComputeEndLine();
148  fBaseLine = fEndLine;
149  fSigmaBase = fSigmaEnd;
150  }
151 
152  return fBaseLine;

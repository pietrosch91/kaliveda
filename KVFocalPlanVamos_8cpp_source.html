229 void KVFocalPlanVamos::BuildGeometry()
230 {
231 
232  // Construction of E503 Focal Plan of Vamos.
233  // the focal plan is made of 18 Si detector (2 lines * 9; size 5*7cm2); each Si is followed by 6 CsI 2.5*2.5cm
234  // In order to simplify the implementation of the FP in the KaliVeda we decided to give a structure like INDRA
235  // ring and module. They are 18 rings of CsI, 6 module per ring; for Si they are 9 rings numbred 1,3,..,17
236  // containing 2 Si per ring.
237  // The polar angle ofthe entrance of Vamos relative to the beam was set at 2. with an opening angle of 5, it was
238  // covering the angular range 2-7 deg. We tried to make a kind of an image of this angular opening to the FP.
239 
240 //Build ChIo layer
241 //1st Layer - ChIo
242  /*
243  Float_t step = 9./7.; // I assume : 9 deg is the opening polar angle of Vamos that I divide by 9 det in horizontal line of vamos
244  KVLayer *kvl = new KVLayer;
245  for(Int_t ii=0;ii<7;ii+=1){
246  Char_t type[50];
247  sprintf(type,"ChIo Ring%i",ii+1);
248  Float_t theta_min=85.5+ii*step;
249  Float_t theta_max=85.5+(ii+1)*step;
250  Int_t ringnum = ii+1;
251  kvl-> // verifier phi et distance
252  AddRing(new
253  KVRing(ringnum, theta_min+.01, theta_max, 90, 1, 5056.,
254  (KVTelescope *) fTelescopes->
255  FindObjectByType(type), 1,1,6.) //60
256  );
257 
258  }
259 
260  kvl->SetName("CI");
261  AddLayer(kvl);
262  cout<<"Build CI Ring"<<endl;
263  */
264 
265  //Build Si layer
266  //2nd Layer - Si
267  Float_t step = 9. / 9.; // I assume : 9 deg is the opening polar angle of Vamos that I divid by 9 det in horizontal line of vamos
268  KVLayer* kvl = new KVLayer;
269  for (Int_t ii = 0; ii < 9; ii += 1) {
270  Char_t type[50];
271  sprintf(type, "Si Ring%i", 2 * ii + 1);
272  Float_t theta_min = 85.5 + ii * step;
273  Float_t theta_max = 85.5 + (ii + 1) * step;
274  Int_t ringnum = 2 * ii + 1;
275  kvl->
276  AddRing(new
277  KVRing(ringnum, theta_min + .01, theta_max, 88.5, 2, 5156.,
278  (KVTelescope*) fTelescopes->
279  FindObjectByType(type), 1, 1, 3.) //30
280  );
281 
282  }
283 
284  kvl->SetName("SIE");

286  fSiLayer = kvl;
287  cout << "Build SI Ring" << endl;
288 
289  //Build CsI layer
290 
291  kvl = new KVLayer;
292  step = 9. / 18.; // horizontaly they are 2 CsI for one Si (total of 18)

301  AddRing(new

303  (KVTelescope*) fTelescopes->
304  FindObjectByType(type), 1, 1, 1.) //10.

306  );

94 void KVNameValueListGUI::ReadData()
95 {
96  for (int i = 0; i < theList->GetNpar(); ++i) {
97  KVNamedParameter* par = theList->GetParameter(i);
98  if (par->IsDouble()) {
99  par->Set(dynamic_cast<TGNumberEntry*>(fData[i])->GetNumber());
100  } else if (par->IsInt()) {
101  par->Set((Int_t)dynamic_cast<TGNumberEntry*>(fData[i])->GetNumber());
102  } else if (par->IsString()) {
103  par->Set(dynamic_cast<TGTextEntry*>(fData[i])->GetText());
104  } else if (par->IsBool()) {
105  par->Set(dynamic_cast<TGCheckButton*>(fData[i])->IsDown());
106  }
107  }
108 }

127 KVNameValueListGUI::KVNameValueListGUI(const TGWindow* main, KVNameValueList* params, Bool_t* cancel_pressed, Bool_t wait_for_main)
128  : fData(params->GetNpar())
129 {
130  // Constructor
131  fMain = new TGTransientFrame(gClient->GetDefaultRoot(), main, 1, 1, kVerticalFrame);
132  fMain->CenterOnParent();
133  fMain->Connect("CloseWindow()", "KVNameValueListGUI", this, "DoClose()");
134  fMain->DontCallClose(); // to avoid double deletions.
135  // use hierarchical cleaning

137 
138  fCancel = cancel_pressed;
139  *fCancel = kFALSE;
140 
141  // SET UP GUI
142  theList = params;
143  theList->Copy(fOriginal); // copy original values in case "Cancel" is pressed
144  // for each name-value pair, add a horizontal frame with a label (name)
145  // and a text entry box (value)
146  Int_t nentries = theList->GetEntries();
147  max_width = 1;
148  for (Int_t i = 0; i < nentries; ++i) {
149  TGHorizontalFrame* hf = new TGHorizontalFrame(fMain);
150  TGLabel* lab = new TGLabel(hf, theList->GetNameAt(i));
151  max_width = TMath::Max(max_width, lab->GetDefaultWidth() + 10);
152  lab->Resize(lab->GetDefaultWidth(), lab->GetDefaultHeight());
153  hf->AddFrame(lab, new TGLayoutHints(kLHintsCenterY | kLHintsLeft | kLHintsExpandX, 2, 3, 1, 1));
154 
155  if (theList->GetParameter(i)->IsInt()) {
156  fData[i] = AddAInt(i, hf);
157  } else if (theList->GetParameter(i)->IsDouble()) {
158  fData[i] = AddADouble(i, hf);
159  } else if (theList->GetParameter(i)->IsBool()) {
160  fData[i] = AddABool(i, hf);
161  } else {
162  fData[i] = AddAString(i, hf);
163  }
164 
165  fMain->AddFrame(hf, new TGLayoutHints(kLHintsTop | kLHintsExpandX, 1, 1, 1, 1));
166  }
167 
168  //--- create the OK, Apply and Cancel buttons
169 
170  UInt_t width = max_width * 2, height = 0;
171 
172  TGHorizontalFrame* hf = new TGHorizontalFrame(fMain);
173 
174  fOKBut = new TGTextButton(hf, "&OK", 1);
175  hf->AddFrame(fOKBut, new TGLayoutHints(kLHintsCenterY | kLHintsExpandX, 2, 3, 0, 0));
176  height = fOKBut->GetDefaultHeight();
177  fOKBut->SetToolTipText("Save changes and close");
178  fOKBut->Resize(max_width, height);
179  fOKBut->Connect("Clicked()", "KVNameValueListGUI", this, "ReadData()");
180  fOKBut->Connect("Clicked()", "KVNameValueListGUI", this, "DoClose()");
181 
182  fCancelBut = new TGTextButton(hf, "&Cancel", 3);
183  hf->AddFrame(fCancelBut, new TGLayoutHints(kLHintsCenterY | kLHintsExpandX, 3, 2, 0, 0));
184  fCancelBut->Resize(max_width, height);
185  fCancelBut->SetToolTipText("Discard changes and close");
186  fCancelBut->Connect("Clicked()", "KVNameValueListGUI", this, "RestoreData()");
187  fCancelBut->Connect("Clicked()", "KVNameValueListGUI", this, "DoClose()");
188 
189  fMain->AddFrame(hf, new TGLayoutHints(kLHintsBottom | kLHintsCenterX | kLHintsExpandX, 0, 0, 10, 5));
190 
191  // map all widgets and calculate size of dialog
192  fMain->MapSubwindows();

194  height = fMain->GetDefaultHeight();
195  fMain->Resize(width, height);
196  fMain->SetWMSize(width, height);

199  kMWMDecorMinimize | kMWMDecorMenu,
200  kMWMFuncAll | kMWMFuncResize | kMWMFuncMaximize |

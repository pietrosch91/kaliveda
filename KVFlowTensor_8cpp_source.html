21 // This global variable class implements the kinetic energy flow tensor of
22 // M. Gyulassy et al., Phys. Lett. 110B(1982)185:
23 // BEGIN_LATEX
24 // $F_{ij}=\sum_{\nu=1}^{N}w_{\nu}p_{i}(\nu)p_{j}(\nu)$
25 // END_LATEX
26 // which is built from the components of the momenta of all (or a selection of)
27 // reaction products in each event.
28 // With the weight w=1, this expression gives the sphericity tensor. Such a
29 // tensor is not appropriate for analysing collisions in which composite
30 // fragments are produced. To correctly weight composite fragments, Gyulassy
31 // et al. proposed to use the weight
32 // BEGIN_LATEX
33 // $w_{\nu}=(2m_{\nu})^{-1}$
34 // END_LATEX
35 // in which case F becomes the (non-relativistic) kinetic energy flow tensor
36 // (the trace of F is the total non-relativistic kinetic energy of the products).
37 // This is the default weight used.
38 // To change the weight, call method SetOption("weight","[your choice]")
39 // with one of the following choices:
40 //
41 // "RKE" : relativistic kinetic energy tensor
42 // i.e. with
43 // BEGIN_LATEX
44 // $w_{\nu}={1}\over{m_{\nu}(\gamma_{\nu}+1)}$
45 // END_LATEX
46 //
47 // "ONE","1", or "" : weight w=1
48 //
49 // "NRKE" : non-relativistic KE tensor (default)
50 //
51 // Gyulassy et al.: "We emphasize that event shape analysis makes sense only in the
52 // nucleus-nucleus center of mass system since the eigenvalues of F specify
53 // an ellipsoid centered at the origin."
54 // => default frame is "CM"

390 void KVFlowTensor::Calculate()
391 {
392  // Calculate eigenvalues & eigenvectors of tensor
393 
394  TMatrixT<double> evectors = fTensor.EigenVectors(fEVal);
395 
396  for (int i = 0; i < 3; i++) {
397  TVectorD col = TMatrixDColumn(evectors, i);
398  fEVec[i].SetXYZ(col[0], col[1], col[2]);
399  }
400 
401  // check orientation of flow axis
402  // by symmetry/convention, we allow FlowAngle between 0 & 90° i.e. flow vector always
403  // points in the forward beam direction.
404  if (fEVec[0].Theta() > TMath::PiOver2()) {
405  fEVec[0] = -fEVec[0];
406  }
407 
408  // check we have an orthonormal basis
409  // we choose evec[0]=>'z' evec[1]=>'y' evec[2]=>'x'
410  // therefore we should have evec[2].Cross(evec[1])=evec[0]
411  // if not we change the sign of evec[2]
412  if (fEVec[2].Cross(fEVec[1]) != fEVec[0]) fEVec[2] = -fEVec[2];
413 
414  // set up azimuthal rotation of CM axes around beam axis in order to put new 'X'-axis in reaction plane
415  // we use the phi of the major axis (largest eigenvector)
416  fAziReacPlane.SetToIdentity();
417  fAziReacPlane.RotateZ(e(1).Phi());
418 
419  fFlowReacPlane.SetToIdentity();
420  fFlowReacPlane.RotateY(e(1).Theta());
421  fFlowReacPlane.RotateZ(e(1).Phi());
422 
423  // calculate Gutbrod squeeze angle (PRC42(1990)640
424  // defined as the angle by which the middle eigenvector e2
425  // needs to be rotated around the e1 flow axis
426  // in order to be brought in to the reaction plane
427  TVector3 normReac = e(1).Cross(TVector3(0, 0, 1)); //normal to reaction plane: e1 x z
428  Double_t angle = TMath::RadToDeg() * e(2).Angle(normReac);
429  // on the contrary to Gutbrod et al, we define this angle between 0 and 90 degrees.
430  // see Fig. 3 of paper: only 0-90 angles can be defined
431  fSqueezeAngle = (angle < 90 ? 90 - angle : angle - 90);
432 
433  // now calculate the in-plane and out-of-plane flow according to Fig. 5 of Gutbrod et al
434  Double_t a = pow(f(2), 0.5); //semi-axes of ellipsoid given by square roots of eigenvalues
435  Double_t b = pow(f(3), 0.5);
436  Double_t tan_t = -a / b * TMath::Tan(fSqueezeAngle * TMath::DegToRad());
437  Double_t t0 = TMath::ATan(tan_t);
438  Double_t inPlane = a * cos(t0) * cos(fSqueezeAngle * TMath::DegToRad()) - b * sin(t0) * sin(fSqueezeAngle * TMath::DegToRad());
439  tan_t = a / (b * tan(fSqueezeAngle * TMath::DegToRad()));
440  t0 = atan(tan_t);
441  Double_t outOfPlane = a * cos(t0) * sin(fSqueezeAngle * TMath::DegToRad()) + b * sin(t0) * cos(fSqueezeAngle * TMath::DegToRad());
442  if (inPlane <= 0.) fSqOutRatio = -1.;
443  else fSqOutRatio = TMath::Min(1.e+03, outOfPlane / inPlane);
444 
445  // calculate sphericity
446  Double_t sum_val_prop = f(1) + f(2) + f(3);
447  if (sum_val_prop > .1) {
448  fSphericity = 3. * (1. - f(1) / sum_val_prop) / 2.;
449  } else
450  fSphericity = -1;
451 
452  fCalculated = kTRUE;
453 }

509 void KVFlowTensor::Fill(KVNucleus* n)
510 {
511  // Fill tensor components with nucleus' momentum components in required
512  // frame, using the required weight
513  // If option "DOUBLE" is set, the chosen weight will be doubled
514  // (this is for the case where e.g. only forward hemisphere particles are included)
515 
516  Double_t W;
517  const KVParticle* frame = n->GetFrame(GetFrame(), kFALSE);
518  switch (weight) {
519  case kONE:
520  W = 1;
521  break;
522  case kRKE:
523  W = 1. / (n->GetMass() * (1 + frame->Gamma()));
524  break;
525  default:
526  case kNRKE:
527  W = 1. / (2.*n->GetMass());
528  break;
529  }
530  if (IsOptionGiven("DOUBLE")) W *= 2.;
531  for (int i = 0; i < 3; i++) {
532  for (int j = i; j < 3; j++) {
533  Double_t xx = W * frame->GetMomentum()[i] * frame->GetMomentum()[j];
534  fTensor(i, j) += xx;
535  if (i != j) fTensor(j, i) += xx;
536  }
537  }
538  ++fNParts;
539 }

36  Double_t mdweight::edist(Double_t* arg, Double_t* par)
37  {
38  // energy distribution of particle in gas
39  // arg[0] = energy/available energy
40  // par[0] = number of particles in gas
41  // par[1] = mass ratio = massTot/(massTot-mPart)
42  Double_t e = arg[0];
43  Double_t N = par[0];
44  Double_t massRat = par[1];
45 
46  Double_t val = 0.;
47  if ((e > 0) && (e * massRat < 1) && (N > 2)) {
48  val = TMath::Sqrt(e) * TMath::Power(1. - massRat * e, (3.*N - 8.) / 2.);
49  }
50  return val;
51  }

110  void mdweight::SetWeight(KVEvent* e, Double_t E)
111  {
112  // Set available energy, E, and calculate statistical weight
113  // for this event
114 
115  if (E <= 0) {
116  setAvailableEnergy(0.);
117  setWeight(0.);
118  return;
119  }
120  setAvailableEnergy(E);
121  Double_t N = e->GetMult();
122  Double_t logmass_sum, mass_sum;
123  logmass_sum = mass_sum = 0.;
124  KVNucleus* n;
125  while ((n = e->GetNextParticle())) {
126  Double_t m = n->GetMass();
127  logmass_sum += TMath::Log(m);
128  mass_sum += m;
129  }
130  A = (3 * (N - 1) / 2.) * log2pi - TMath::LnGamma(3 * (N - 1) / 2.)
131  + 1.5 * (logmass_sum - TMath::Log(mass_sum));
132  B = (3 * N - 5) / 2.;
133 
134  Double_t w = TMath::Exp((A + B * TMath::Log(E)) - log10twelve);
135  setWeight(w);
136  }

176  void mdweight::nextparticleGenerateEvent(Int_t N, KVNucleus* part)
177  {
178  // Called by StatWeight::GenerateEvent when adding a particle to the event
179  // N is the number of particles still to add including this one
180  //
181  // The algorithm was written by Daniel Cussol (LPC Caen, France).
182 
183  Double_t mPart = part->GetMass();
184  Double_t rR = mPart / massTot;
185  Double_t ratio;
186  if (rR < 1.) ratio = 1. / (1. - rR);
187  else ratio = 1.;
188  Double_t ec = 0.; // kinetic energy of particle
189  Double_t ppx, ppy, ppz; // components of particle momentum
190  ppx = ppy = ppz = 0;
191  if (N >= 2) {
192  Double_t p = 0.; //momentum to give particle
193  if (N > 2) {
194  // draw random KE from 1-particle distribution for given N & ratio
195  ec = eDisp * getKEdist(N, ratio)->GetRandom();
196  p = sqrt(2.*mPart * ec);
197  } else {
198  // last 2 particles: share remaining available energy
199  p = sqrt(2.*(massTot - mPart) * mPart * eDisp / massTot);
200  ec = p * p / 2. / mPart;
201  }
202  Double_t ct = 1. - 2.*gRandom->Rndm();
203  Double_t st = TMath::Sqrt(1. - ct * ct);
204  Double_t phi = gRandom->Rndm() * 2.*TMath::Pi();
205  ppz = ct * p;
206  ppx = st * TMath::Cos(phi) * p;
207  ppy = st * TMath::Sin(phi) * p;
208  ppx += px * rR;
209  ppy += py * rR;
210  ppz += pz * rR;
211  } else if (N == 1) {
212  ppx = px;
213  ppy = py;
214  ppz = pz;
215  ec = 0;
216  }
217  part->SetMomentum(ppx, ppy, ppz);
218 
219  eDisp -= ec * ratio;
220  massTot -= mPart;
221  px -= ppx;
222  py -= ppy;
223  pz -= ppz;
224  }

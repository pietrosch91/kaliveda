68 void KVINDRADetector::AddACQParamType(const Char_t* type)
69 {
70  // Add an acquisition parameter of given type to this detector
71  // The parameter will be named according to INDRA/GANIL acquisition
72  // convention, i.e. name_of_detector_type e.g.
73  // CI_0201_PG (type="PG")
74  // CSI_0913_R (type="R")
75  //
76  // If you want to use a different extension for a given type for certain
77  // detectors, you can define them in the .kvrootrc file, e.g.:
78  //
79  // e613.KVACQParam.SI_0101.T: TOF
80  //
81  // Then for the dataset 'e613', calling GetACQParam("T") for the detector
82  // SI_0101 will return the parameter with name "SI_0101_TOF"
83 
84  KVACQParam* par = new KVACQParam();
85  TString extension = gDataSet->GetDataSetEnv(Form("KVACQParam.%s.%s", GetName(), type), type);
86  TString name;
87  name = this->GetName();
88  name.Append("_");
89  name.Append(extension.Data());
90  par->SetName(name);
91  par->SetType(type);
92  AddACQParam(par);
93 }

119 KVACQParam* KVINDRADetector::GetACQParam(const Char_t* type)
120 {
121  // Look for acquisition parameter with given type in list
122  // of parameters associated with this detector.
123  //
124  // *** WARNING ***
125  // This method overrides the KVDetector::GetACQParam(const Char_t* name)
126  // method, which interprets the argument as the FULL name of the parameter.
127  // In this version, only the 'type' needs to be given,
128  // e.g. GetACQParam("PG") to retrieve e.g. the "CI_0201_PG" parameter
129  // associated to detector "CI_0201".
130  // If using this method in a loop over a list of INDRA- and non-INDRA
131  // detectors, the behaviour of the method will be different for the two
132  // types of detectors, and results may not be what you wanted.
133  //
134  // If you want to use a different extension for a given type for certain
135  // detectors, you can define them in the .kvrootrc file, e.g.:
136  //
137  // e613.KVACQParam.SI_0101.T: TOF
138  //
139  // Then for the dataset 'e613', calling GetACQParam("T") for the detector
140  // SI_0101 will return the parameter with name "SI_0101_TOF"
141 
142  TString extension = gDataSet->GetDataSetEnv(Form("KVACQParam.%s.%s", GetName(), type), type);
143  extension.Prepend("_");
144  // search just using the extension
145  // (in case e.g. of a mismatch between detector & parameter names)
146  TIter next(GetACQParamList());
147  KVACQParam* p;
148  while ((p = (KVACQParam*)next())) {
149  TString name = p->GetName();
150  if (name.EndsWith(extension)) return p;
151  }
152  return 0;
153 }

165 Double_t KVINDRADetector::GetPGfromGG(Double_t GG)
166 {
167  //Linear conversion from GG to PG coder values (silicon and ionisation chamber detectors)
168  //If GG is not given as argument, the current value of the detector's GG ACQParam is read
169  //The PG value returned includes the current pedestal:
170  // PG = pied_PG + fGGtoPG_0 + fGGtoPG_1 * (GG - pied_GG)
171  //alpha, beta coefficients are obtained by fitting (PG-pied) vs. (GG-pied) for data.
172  if (GG < 0)
173  GG = (Double_t) GetGG();
174  Double_t PG =
175  GetPedestal("PG") + fGGtoPG_0 + fGGtoPG_1 * (GG - GetPedestal("GG"));
176  return PG;
177 }

189 Double_t KVINDRADetector::GetGGfromPG(Double_t PG)
190 {
191  //Linear conversion from PG to GG coder values (silicon and ionisation chamber detectors)
192  //If PG is not given as argument, the current value of the detector's PG ACQParam is read
193  //The GG value returned includes the current pedestal:
194  // GG = pied_GG - (fGGtoPG_0/fGGtoPG_1) + (PG - pied_PG)/fGGtoPG_1
195  //alpha, beta coefficients are obtained by fitting (PG-pied) vs. (GG-pied) for data.
196  if (PG < 0)
197  PG = (Double_t) GetPG();
198  Double_t GG =
199  GetPedestal("GG") - (fGGtoPG_0 / fGGtoPG_1) + (PG - GetPedestal("PG")) / fGGtoPG_1;
200  return GG;
201 }

210 KVINDRADetector* KVINDRADetector::FindChIo()
211 {
212  //PRIVATE METHOD
213  //Used when GetChIo is called the first time to retrieve the
214  //pointer to the ChIo of the group associated to this detector
215  if (GetTelescope()) {
216  KVGroup* kvgr = GetTelescope()->GetGroup();
217  if (kvgr) {
218  const KVSeqCollection* dets = kvgr->GetDetectors();
219  TIter next_det(dets);
220  KVINDRADetector* dd;
221  while ((dd = (KVINDRADetector*) next_det())) {
222  if (dd->InheritsFrom("KVChIo"))
223  fChIo = dd;
224  }
225  }
226  } else
227  fChIo = 0;
228  return fChIo;
229 }

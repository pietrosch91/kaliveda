88 void KVLevelScheme::ReadLevels(const char* file)
89 {
90  KVFileReader fr;
91  fr.OpenFileToRead(file);
92 
93  while (fr.IsOK()) {
94  fr.ReadLine("|");
95  KVString tmp = "";
96 
97  Int_t npars = fr.GetNparRead();
98  if (npars == 0) break;
99 
100  Int_t aa, zz;
101  KVString jpi, gamma;
102  Double_t ee, de; //, gam, dgam;
103  ee = de = 0.;
104 
105  aa = fr.GetIntReadPar(0);
106  zz = fr.GetIntReadPar(1);
107 
108  if (aa != fCompNuc->GetA() || zz != fCompNuc->GetZ()) continue;
109 
110  ee = fr.GetDoubleReadPar(2);
111 
112  tmp = fr.GetReadPar(3);
113  tmp.RemoveAllExtraWhiteSpace();
114  if (!tmp.EqualTo("")) de = tmp.Atof();
115  else de = 0.;
116 
117  tmp = fr.GetReadPar(4);
118  tmp.RemoveAllExtraWhiteSpace();
119  if (!tmp.EqualTo("")) {
120  jpi = tmp.Data();
121  jpi.ReplaceAll("GE", "");
122  jpi.ReplaceAll(" &amp; ", ",");
123  jpi.ReplaceAll("LE", "");
124  jpi.ReplaceAll("LT", "");
125  jpi.ReplaceAll("AP", "");
126  jpi.ReplaceAll(" OR ", "or");
127  jpi.ReplaceAll(" TO ", "to");
128  } else jpi = "";
129 
130  gamma = "";
131  if (npars == 6) {
132  gamma = fr.GetReadPar(5);
133  }
134 
135  AddResonance(ee, jpi.Data(), gamma.Data());
136  }
137  fr.CloseFile();
138 }

174 double KVLevelScheme::func(double xx, double tt)
175 {
176  double yy = 0.;
177  for (Int_t ii = 0; ii < fLevels.GetSize(); ii++) {
178  KVExcitedState* ll = (KVExcitedState*)fLevels.At(ii);
179  yy += ll->eval(xx);
180  }
181  yy *= TMath::Exp(-xx / tt);

336 void KVLevelScheme::GetParticlesFromErel(KVNucleus* n1, KVNucleus* n2, double erel, bool randAngle, TVector3* vsrc)
337 {
338  KVNucleus nuc1(n1->GetZ(), n1->GetA());
339  KVNucleus nuc2(n2->GetZ(), n2->GetA());
340 
341  double m1 = nuc1.GetMass();
342  double m2 = nuc2.GetMass();
343  double e1 = (m2 / (m1 + m2)) * erel;
344  double e2 = (m1 / (m1 + m2)) * erel;
345 
346  nuc1.SetKE(e1);
347  nuc1.SetTheta(0);
348  nuc2.SetKE(e2);
349  nuc2.SetTheta(180);
350 
351  if (randAngle) {
352  TVector3 v1r = nuc1.GetVelocity();
353  TVector3 v2r = nuc2.GetVelocity();
354  TRotation rr;
355  rr.SetXEulerAngles(gRandom->Rndm() * 2.*TMath::Pi(), TMath::ACos(gRandom->Rndm() * 2. - 1.), gRandom->Rndm() * 2.*TMath::Pi());
356  v1r *= rr;
357  nuc1.SetVelocity(v1r);
358  v2r *= rr;
359  nuc2.SetVelocity(v2r);
360  }
361 
362  if (vsrc) {
363  *vsrc *= -1;
364  nuc1.SetFrame("src", *vsrc);
365  nuc2.SetFrame("src", *vsrc);
366  n1->SetVelocity(nuc1.GetFrame("src")->GetVelocity());
367  n2->SetVelocity(nuc2.GetFrame("src")->GetVelocity());
368  }
369 
370 }

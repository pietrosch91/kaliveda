181 void KVIDQALine::IdentA(Double_t x, Double_t y, Int_t& A, Double_t& realA, Int_t& code) const
182 {
183 
184  // Default values
185  A = -1;
186  realA = -1.;
187 
188  Double_t dist, dist_r, dist_l;
189  Int_t idx, idx_r, idx_l;
190  KVIDQAMarker* closest_mk = const_cast<KVIDQALine*>(this)->FindNearestIDMarker(x, y, idx, idx_l, idx_r, dist, dist_l, dist_r);
191 
192  if (!closest_mk) {
193  //no marker corresponding to point was found
194  //we assume that realA = Q * A/Q and set kICODE0
195  realA = x * GetQ();
196  code = KVIDQAGrid::kICODE0;
197  } else {
198  //the closest marker is found
199  //Info("IdentA","ID marker found: %s",closest_mk->GetName());
200 
201  //marker at the left of the point
202  KVIDQAMarker* l_mk = GetMarkerAt(idx_l);
203 
204  //marker at the right of the point
205  KVIDQAMarker* r_mk = GetMarkerAt(idx_r);
206 
207  // WARNING: the method KVIDQALine::FindNearestIDMarker always find
208  // 2 nearest markers even for point left (right) the first (last) marker.
209  // we have to remove the second found marker for these points
210 
211  if ((closest_mk == GetMarkers()->First()) && !ProjIsBetween(x, y, l_mk, r_mk))
212  r_mk = NULL;
213  else if (closest_mk == GetMarkers()->Last() && !ProjIsBetween(x, y, l_mk, r_mk))
214  l_mk = NULL;
215 
216  Double_t deltaA = 0.;
217 
218  // case where 2 embracing markers are found
219  if (l_mk && r_mk && (l_mk != r_mk)) {
220  if (dist > closest_mk->GetWidth() / 2.) {
221  if (closest_mk == l_mk) code = KVIDQAGrid::kICODE1; // "slight ambiguity of A, which could be larger"
222  else code = KVIDQAGrid::kICODE2; // "slight ambiguity of A, which could be smaller"
223  } else code = KVIDQAGrid::kICODE0; // ok
224 
225  }
226  // case where only 1 embracing marker is found.
227  // in this case, the distance between the point and the marker
228  // has to be lower than the marker width
229  else {
230  if (dist > closest_mk->GetWidth() / 2.) {
231  // the distance from the closest marker is to high
232  if (closest_mk == l_mk) code = KVIDQAGrid::kICODE6; // "(x,y) is below first marker in line"
233  else code = KVIDQAGrid::kICODE7; // "(x,y) is above last marker in line"
234  } else {
235  code = KVIDQAGrid::kICODE3; // "slight ambiguity of A, which could be larger or smaller"
236  }
237  }
238 
239  deltaA = (x - closest_mk->GetX()) * GetQ();
240  realA = closest_mk->GetA() + deltaA;
241  }
242  A = TMath::Nint(realA); // since some A-markers are missing in the line
243 }

254 
255  Int_t n_mk = GetNumberOfMasses();
256  if (n_mk == 0) return NULL;

274  KVIDQAMarker* mk_up = GetMarkerAt(idx_up);

428 void KVIDQALine::IncrementPtIdxOfMarkers(Int_t idx, Int_t ival)
429 {
430  TIter next(fMarkers);
431  KVIDQAMarker* m = NULL;
432  while ((m = (KVIDQAMarker*)next())) {
433  Int_t idx_m;
434  Double_t delta_m;
435  m->GetPointIndexAndDelta(idx_m, delta_m);
436 
437  if ((idx - 1) <= idx_m) {
438 
439  Double_t delta = 0;
440  if ((idx_m == (idx - 1)) && (delta_m)) {
441  //vector givent by the initial low and up neighbour points
442  Double_t xi = fX[idx - 1] - fX[idx + 1];
443  Double_t yi = fY[idx - 1] - fY[idx + 1];
444  Double_t li = TMath::Sqrt(TMath::Power(xi, 2) + TMath::Power(yi, 2));
445 
446  //vector givent by the final low and up points
447  Double_t xf = fX[idx - 1] - fX[idx];
448  Double_t yf = fY[idx - 1] - fY[idx];
449  Double_t lf = TMath::Sqrt(TMath::Power(xf, 2) + TMath::Power(yf, 2));
450 
451  if (delta_m * li < lf) {
452  delta = delta_m * li / lf;
453  } else {
454  delta = (delta_m * li - lf) / (li - lf);
455  idx_m += ival;
456  }
457  } else if (idx <= idx_m) {
458  delta = delta_m;
459  idx_m += ival;
460  }
461 
462  if (idx_m < 0) {
463  idx_m = 0;
464  delta = 0.;
465  }
466  if (idx_m > fNpoints - 2) {
467  idx_m = fNpoints - 2;
468  delta = 1.;
469  }
470 
471  m->SetPointIndex(idx_m, delta);
472  }
473  }

503 void KVIDQALine::FindAMarkers(TH1* h)
504 {
505 
506  //find mass peaks in histogram h.
507  //Use TSpectrum to find the peak candidates.
508  //Build KVIDQAMarker for each peak found.
509 
510  TSpectrum s(60);
511  Int_t nfound = s.Search(h, 2, "goff");
512 
513  Info("FindAMarkers", "%d masses found in histogram %s for %s line", nfound, h->GetName(), GetName());
514 
515  // Clear list of IDQAMarkers to build a new one
516  fMarkers->Clear();
517 
518  // Build new IDQAMarkers
519 #if ROOT_VERSION_CODE >= ROOT_VERSION(6,0,0)
520  Double_t* xpeaks = s.GetPositionX();
521 #else
522  Float_t* xpeaks = s.GetPositionX();
523 #endif
524  for (int p = 0; p < nfound; p++) {
525  Double_t x = xpeaks[p];
526  KVIDQAMarker* m = new KVIDQAMarker;
527  m->SetParent(this);
528  m->SetX(x);
529  AddMarker(m);
530  }
531 
532  // sort fMarkers list from X coordinate
533  SortMarkers();
534 
535  // find neighbours simply looping all points
536  // and find also the 2 adjacent points: (low2 < low < x < up < up2 )
537  // needed in case x is outside the graph ascissa interval
538  Int_t low = -1;
539  TIter next(fMarkers);
540  KVIDQAMarker* m = NULL;
541 
542  Int_t a = 0; // we assum that xpeaks gives A/Q
543  Bool_t stop = kFALSE;
544  KVList del_list;
545  while ((m = (KVIDQAMarker*)next())) {
546 
547  Double_t x = m->GetX();
548  Double_t tmp = TMath::Nint(x * GetQ());
549  if (a <= tmp) a = tmp;
550 
551  m->SetA(a++);
552  Int_t up = -1;
553 
554  if (!stop) {
555  for (Int_t i = (low > -1 ? low : 0); i < fNpoints; ++i) {
556  if (fX[i] < x) {
557  if ((low == -1) || (fX[i] > fX[low])) low = i;
558  } else if (fX[i] > x) {
559  if ((up == -1) || (fX[i] < fX[up])) up = i;
560  } else { // case x == fX[i]
561  low = up = i;
562  break;
563  }
564  }
565  }
566 
567  // treat cases when x is outside line min max abscissa
568  if ((up == -1) || (low == -1)) {
569  // list the markers which are outside line to be deleted
570  del_list.Add(m);
571  if (up == -1) stop = kTRUE;
572  continue;
573  }
574 
575  // set index and the X coordinate to the marker
576  if (low == up) m->SetPointIndex(low);
577  else m->SetPointIndexAndX(low, x);
578 
579  // draw the new marker if its parent Q-line is already drawn in the
580  // current pad
581  if (gPad && gPad->GetListOfPrimitives()->FindObject(this)) m->Draw();
582  }
583 
584  if (del_list.GetEntries()) {
585  Warning("FindAMarkers", "%d markers will be deleted (outside of %s-line)", del_list.GetEntries(), GetName());
586  del_list.ls();
587  }
588 }

687 void KVIDQALine::CalculateMarkerWidths()
688 {
689 //Calculate natural "width" of each ID markers in the line.
690  //The markers in the line have first to sorted so that they are in order of ascending 'X'
691  //i.e. first markers has lower A, last line is the heaviest mass .
692  //
693  // The width is supposed to be the distance between the first and the
694  // last marker divided by the difference of A of both markers.
695 
696  KVIDQAMarker* first = (KVIDQAMarker*)fMarkers->First();
697  KVIDQAMarker* last = (KVIDQAMarker*)fMarkers->Last();
698  if ((!last) || (!first)) return;
699  Double_t width = first->DistanceToMarker(last->GetX(), last->GetY());
700  Double_t dA = last->GetA() - first->GetA();
701  width = TMath::Abs(width / dA);
702  fMarkers->R__FOR_EACH(KVIDQAMarker, SetWidth)(width);
703 }

721 Bool_t KVIDQALine::ProjIsBetween(Double_t x, Double_t y, KVIDQAMarker* m1, KVIDQAMarker* m2) const
722 {
723  //Given a line segment defined by 2 markers m1 and m2 test if the
724  // projection of the point P(x,y) is between both markers.
725  //
726  // m1 x---------------------------------------x m2
727  // |
728  // |
729  // |
730  // x
731  // P (x,y)
732  //
733  // returns kFALSE if the projection is outside the segment given
734  // by both markers.
735 
736  if (!m1 || !m2) return kFALSE;
737 
738  Double_t x1 = m1->GetX();
739  Double_t x2 = m2->GetX();
740  Double_t y1 = m1->GetY();
741  Double_t y2 = m2->GetY();
742  // scale Y coordinate to A/Q
743 
744  if (IsAvsAoQ()) {
745  if (GetQ()) {
746  y /= GetQ();
747  y1 /= GetQ();
748  y2 /= GetQ();
749  } else return kFALSE;
750  } else {
751  if (y > 0 && y1 > 0 && y2 > 0) {
752  y = GetA() / y;
753  y1 = GetA() / y1;
754  y2 = GetA() / y2;
755  } else return kFALSE;
756  }
757  TVector2 P1P2(x2 - x1, y2 - y1);
758  if (P1P2.Mod2() == 0) return kFALSE;
759  TVector2 P1M(x - x1, y - y1);
760 
761  Double_t proj = (P1M * P1P2) / P1P2.Mod();
762  //Info("ProjIsBetween","P1M %f, P1P2 %f, proj %f",P1M.Mod(), P1P2.Mod(), proj);

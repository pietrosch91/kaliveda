81 KVHarpeeIC::KVHarpeeIC(UInt_t number, Float_t pressure, Float_t temp, Float_t thick) : KVVAMOSDetector(number, "C4H10")
82 {
83  // Make the ionisation chamber of Harpee composed by:
84  // - one 1.5 mylar window;
85  // - 3 volumes of C4H10 with thicknesses (2*dz): 2.068 cm, thick, 1.278 cm.
86  // By default thick=10.457 cm;
87  // Only the 2nd gaz volume is "active".
88  //
89  // The width (2*dx) of this detector is ??? and the height (2*dy) is ???.
90  // The ionisation chamber is closed with the silicon wall in Harpee.
91  //
92  // Input: pressure - pressure of the gaz in mbar
93  // temp - temperature of the gaz in degrees celsius.
94  // Default temperature = 19°C
95  // thick - the thickness in cm of the 2nd volume of gaz.
96  Warning("KVHarpeeIC", "Check if the width, the height and the thickness are correct in this constructor");
97 
98  init();
99  SetType("CHI");
100  SetLabel("CHI");
101  SetName(GetArrayName());
102 
103  // number of absorber
104  Int_t Nabs = 4;
105 
106  // material of each absorber
107  const Char_t* mat[] = {"Myl", "C4H10", "C4H10", "C4H10"};
108 
109  // thickness of each absorber
110  Float_t th[] = {
111  static_cast<Float_t>(1.5 * KVUnits::um),
112  static_cast<Float_t>(2.068 * KVUnits::cm),
113  static_cast<Float_t>(thick * KVUnits::cm),
114  static_cast<Float_t>(1.278 * KVUnits::cm)
115  };
116 
117  // active absorber flag
118  Bool_t active[] = { kFALSE, kFALSE, kTRUE, kFALSE};
119 
120  // width and height of the detector. TO BE VERIFIED
121  Float_t w = 40 * KVUnits::cm;
122  Float_t h = 12 * KVUnits::cm;
123 
124  // total thickness of the detector
125  Float_t wtot = 0;
126  for (Int_t i = 0; i < Nabs; i++) wtot += th[i];
127 
128  // pressure and temperature of the detector
129  SetPressure(pressure * KVUnits::mbar);
130  SetTemperature(temp);
131 
132  // Adding the absorbers
133  TGeoShape* shape = NULL;
134  for (Int_t i = 0; i < Nabs; i++) {
135 
136  Double_t dx = w / 2;
137  Double_t dy = h / 2;
138  Double_t dz = th[i] / 2;
139 
140  fTotThick += th[i];
141 
142  // box shape of the absorber
143  shape = new TGeoBBox(dx, dy, dz);
144 
145  // build and position absorber in mother volume.
146  // Reference is the center of absorber
147  Double_t ztrans = dz - wtot / 2;
148  for (Int_t j = 0; j < Nabs - 1; j++) ztrans += (j < i) * th[j];
149  TGeoTranslation* tr = new TGeoTranslation(0., 0., ztrans);
150  AddAbsorber(mat[i], shape, tr, active[i]);
151  }
152 }

256 Double_t KVHarpeeIC::GetEnergy()
257 {
258  //Redefinition of KVDetector::GetEnergy().
259  //If energy lost in active (gas) volume is already set (e.g. by calculation of energy loss
260  //of charged particles), return its value.
261  //If not, we calculate it and set it using the values read from coders (if fired)
262  //and the calibrations, if present
263  //
264  //Returns 0 if (i) no calibration present (ii) calibration present but no data in acquisition parameters
265 
266  //fELoss already set, return its value
267  Double_t ELoss = KVDetector::GetEnergy();
268  if (IsSimMode()) return ELoss; // in simulation mode, return calculated energy loss in active layer
269  if (ELoss > 0) return ELoss;
270  ELoss = GetCalibE();
271  if (ELoss < 0) ELoss = 0;
272  SetEnergy(ELoss);
273  return ELoss;
274 }

286 UInt_t KVHarpeeIC::GetFiredSegNumber(Option_t* opt)
287 {
288  // return the number of the fired segment of this Harpee ionisation chamber
289  // ( number between 1 to 7 ). Returns 0 if no segment or several segments
290  // are fired. A segment is considered as fired if at least one of its 3
291  // acquisition parameters (A, B and C) is fired (KVACQParam::Fired( opt )).
292  // Set the option opt = "P" to accept only the acquisition parameters with
293  // their value above the pedestal.
294 
295  TIter next(GetACQParamList());
296  KVACQParam* par = NULL;
297  UInt_t num = 0;
298 
299  while ((par = (KVACQParam*)next())) {
300  if (par->Fired(opt)) {
301  if (num && (num != par->GetNumber())) return 0;
302  num = par->GetNumber();
303  }
304  }
305  return num;
306 }

318 Double_t KVHarpeeIC::GetCalibE()
319 {
320  // Calculate energy in MeV from coder values. Only the 7 segments B
321  // are used to calculate calibrated energy.
322  // Returns 0 if calibration not ready for fired acquisition parameters
323  // or if no acquisition parameter is fired.
324  // (we require that at least one acquisition parameter has a value
325  // greater than the current pedestal value).
326 
327 
328  Bool_t ok = kTRUE;
329  Double_t E = 0;
330 
331  TIter next(GetListOfCalibrators());
332  KVFunctionCal* cal = NULL;
333  while (ok && (cal = (KVFunctionCal*)next())) {
334 
335  //Energy only coded in ACQParam with label 'A'
336  if (cal->GetLabel()[0] != 'A') continue;
337 
338 // Int_t num = cal->GetUniqueID();
339 // if( ((num%100)/10) != 1 ) continue;
340 
341  if (cal->GetACQParam()->Fired("P")) {
342  if (cal->GetStatus()) {
343  E += cal->Compute();
344  } else ok = kFALSE;
345  }
346  }
347  return (ok ? E : 0.);
348 }

427 void KVHarpeeIC::SetACQParams()
428 {
429  // Setup acquisition parameters of this ionisation chamber.
430  // ACQ parameters with type 'E':
431  // E[detector type]_[A,B,C]_[detector number]
432  //
433  // WARNING: actually only the acquisition parameters with indice 'A'
434  // are defined ('B' and 'C' were not coded in e494s and e503 experiment ).
435 
436  TString name;
437  Char_t idx[] = {'A', 'B', 'C'};
438 
439  for (Int_t num = 1; num <= ARPEEIC_NSEG; num++) {
440 // for(Int_t i = 0; i<3; i++){
441  for (Int_t i = 0; i < 1; i++) {
442  KVACQParam* par = new KVACQParam;
443  name.Form("E%s_%c_%d", GetType(), idx[i], num);
444  par->SetName(name);
445  par->SetType("E");
446  par->SetLabel(Form("%c", idx[i]));
447  par->SetNumber(num);
448  par->SetUniqueID(CalculateUniqueID(par));
449  AddACQParam(par);
450  }
451  }

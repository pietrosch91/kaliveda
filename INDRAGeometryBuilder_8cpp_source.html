53 void INDRAGeometryBuilder:: CorrectCoordinates(Double_t* coo, Double_t& offX, Double_t& offY)
54 {

56  // correct coordinates for offset

148 void INDRAGeometryBuilder::CalculateBackPlaneCoordinates(TVector3* frontcoords, TVector3 normal, Double_t depth, TVector3* backcoords)
149 {
150  // Given an array of ncoords points frontcoords[ncoords] which lie in a plane
151  // calculate the corresponding points on a parallel plane which is
152  // at a distance from the origin which is greater by depth
153  // (distance plane-origin = length of normal to plane passing through origin)
154  // 'normal' is vector point of intersection plane-normal to origin
155 
156  Double_t planeDist = normal.Mag();//distance original plane-origin
157  for (int i = 0; i < 4; i++) {
158  Double_t rap = (planeDist + depth) / planeDist;
159  backcoords[i] = rap * frontcoords[i];
160  }
161 }

215  // by -150 degrees around the Z-axis
216  for (int i = 0; i <= 3; i++) {

279 void INDRAGeometryBuilder::MakeFrame(TString det_type, Int_t ring_num)
280 {
281  // Create the TGeoVolume corresponding to the outer casing
282  // (dead zone) of the detector
283 
284  TVector3 fFrameBack[4];
285  TVector3 fFrameBackCentre;
286  //CalculateBackPlaneCoordinates(fFrameFront, fFrameCentre, fTotalThickness, fFrameBack);
287  // make deadzone be 1um thick
288  Double_t dz = 5.e-5;
289  CalculateBackPlaneCoordinates(fFrameFront, fFrameCentre, 2 * dz, fFrameBack);
290  CalculateCentre(fFrameBack, fFrameBackCentre);
291 
292  TVector3 corners[8]; // 8 vertices of the volume
293  TransformToOwnFrame(fFrameFront, fFrameCentre, corners);
294  TransformToOwnFrame(fFrameBack, fFrameBackCentre, &corners[4]);
295 
296  Double_t vertices[16];
297  for (int i = 0; i < 8; i++) {
298  vertices[2 * i] = corners[i].X();
299  vertices[2 * i + 1] = corners[i].Y();
300  }
301 
302  //Double_t dz = 0.99 * fTotalThickness / 2.;
303  TString vol_name;
304  vol_name.Form("STRUCT_%s_%02d", det_type.Data(), ring_num);
305  fFrameVolume = gGeoManager->MakeVolumeAssembly(vol_name);
306  fFrameVolume->SetMedium(gGeoManager->GetMedium("Vacuum"));
307  vol_name.Form("DEADZONE_%s_%02d", det_type.Data(), ring_num);
308  TGeoMedium* med = fFrameMat.GetGeoMedium();
309  Double_t offX, offY;
310  CorrectCoordinates(vertices, offX, offY);
311  // place the deadzone 1um (=1.e-4 cm) behind the detector entrance window
312  TGeoTranslation* offset = new TGeoTranslation(offX, offY, -(0.5 * fTotalThickness - (1.e-4 + dz)));
313  TGeoVolume* frame = gGeoManager->MakeArb8(vol_name.Data(), med, dz, vertices);
314  frame->SetLineColor(med->GetMaterial()->GetDefaultColor());
315  fFrameVolume->AddNode(frame, 1, offset);
316 }

325 void INDRAGeometryBuilder::TransformToOwnFrame(TVector3* orig, TVector3& centre, TVector3* ownframe)
326 {
327  // takes the four points (TVector3) in the array and transforms them
328  // to a reference frame in which the +ve z-axis goes through their centre
329 
330  TRotation rot_to_frame;
331  rot_to_frame.SetYEulerAngles(-centre.Phi(), -centre.Theta(), 0.);
332  TVector3 displZ(0, 0, centre.Mag());
333  for (int i = 0; i < 4; i++) {
334  ownframe[i] = rot_to_frame * orig[i] - displZ;
335  }

345 void INDRAGeometryBuilder::PlaceFrame(Double_t phi, Int_t copy_no)
346 {

348 
349  Double_t theta = fFrameCentre.Theta() * TMath::RadToDeg();
350  TGeoRotation rot1, rot2;
351  rot2.SetAngles(phi + 90., theta, 0.);
352  rot1.SetAngles(-90., 0., 0.);
353  // distance to frame centre
354  Double_t trans_z = fFrameCentre.Mag() + fTotalThickness / 2.;
355

624  // i.e. make sure all "layers" have non-zero thickness
625  Bool_t multi_layer = fLayers->GetSize() > 1;
626  if (multi_layer) {
627  TIter next(fLayers);
628  KVMaterial* abs;
629  Int_t no_abs = 0;
630  while ((abs = (KVMaterial*)next())) {
631  Double_t thick = abs->GetThickness();
632  if (thick > 0.0) no_abs++;
633  }
634  multi_layer = (no_abs > 1);
635  }
636  if (multi_layer) {
637  fDetVolume = gGeoManager->MakeVolumeAssembly(Form("DET_%s", det));
638  fDetVolume->SetMedium(gGeoManager->GetMedium("Vacuum"));
639  }
640  TVector3 frontPlane[4], backPlane[4], frontCentre, backCentre;
641  // front plane of first absorber is front plane of detector
642  for (int i = 0; i < 4; i++) frontPlane[i] = som[i];
643  frontCentre = cen;
644 
645  /**** BUILD & ADD LAYERS ****/
646  TIter next(fLayers);
647  KVMaterial* abs;
648  Double_t depth_in_det = 0.;
649  Int_t no_abs = 1;
650 
651  Double_t offX, offY; //offset of each layer
652  fDetectorPosition = 0;
653 
654  while ((abs = (KVMaterial*)next())) {
655  // get medium for absorber
656  TGeoMedium* med = abs->GetGeoMedium();
657  Double_t thick = abs->GetThickness();
658 
659  if (thick == 0.0) {
660  no_abs++; // ignore zero thickness layers
661  continue;
662  }
663 
664  // calculate coordinates of back plane
665  CalculateBackPlaneCoordinates(frontPlane, frontCentre, thick, backPlane);
666  // calculate centre of back plane
667  CalculateCentre(backPlane, backCentre);
668  // calculate corners of volume
669  TransformToOwnFrame(frontPlane, fFrameCentre, corners);
670  TransformToOwnFrame(backPlane, fFrameCentre, &corners[4]);
671  for (int i = 0; i < 8; i++) {
672  vertices[2 * i] = corners[i].X();
673  vertices[2 * i + 1] = corners[i].Y();
674  }
675 
676  Double_t dz = thick / 2.;
677  TString vol_name;
678  if (multi_layer) {
679  if (no_abs == fActiveLayer) vol_name = Form("ACTIVE_%s", det);
680  else vol_name = Form("%s_%d_%s", det, no_abs, abs->GetName());
681  } else
682  vol_name = Form("DET_%s", det);
683  CorrectCoordinates(vertices, offX, offY);
684  TGeoVolume* vol =
685  gGeoManager->MakeArb8(vol_name.Data(), med, dz, vertices);
686  vol->SetLineColor(med->GetMaterial()->GetDefaultColor());
687  if (multi_layer) {
688  /*** position absorber in mother ***/
689  Double_t trans_z = -fTotalThickness / 2. + depth_in_det + dz; // (note: reference is CENTRE of absorber)
690  TGeoTranslation* tr = new TGeoTranslation(offX, offY, trans_z);
691  fDetVolume->AddNode(vol, no_abs, tr);
692  } else {
693  // single absorber: mother is absorber is detector is mother is ...
694  fDetVolume = vol;
695  fDetectorPosition = new TGeoTranslation(offX, offY, 0);
696  }
697 
698  depth_in_det += thick;
699  no_abs++;
700 
701  // front of next absorber is back of current absorber
702  for (int i = 0; i < 4; i++) frontPlane[i] = backPlane[i];
703  frontCentre = backCentre;
704  }
705 }

718 void INDRAGeometryBuilder::CalculateCentre(TVector3* corners, TVector3& inter)
719 {
720  // TVector3 corners[4] array of corner coordinates in usual order:
721  // corners[3] : theta-min, phi-min
722  // corners[2] : theta-max, phi-min
723  // corners[1] : theta-max, phi-max
724  // corners[0] : theta-min, phi-max
725  //
726  // Calculate the intersection of the normal to the plane passing through the origin.
727 
728  TVector3 a10 = corners[1] - corners[0];
729  TVector3 a21 = corners[2] - corners[1];
730  TVector3 normal = a10.Cross(a21);
731  Double_t d = (corners[3] * normal) / normal.Mag2();
732  inter = normal * d;
733 }

747 void INDRAGeometryBuilder::CalculateCornersInPlane(TVector3* plane, Double_t thetamin,
748  Double_t thetamax, Double_t phimin, Double_t phimax, TVector3* corners)
749 {
750  // fill TVector3 corners[4] array in usual order:
751  // corners[3] : theta-min, phi-min
752  // corners[2] : theta-max, phi-min
753  // corners[1] : theta-max, phi-max
754  // corners[0] : theta-min, phi-max
755  // theta, phi in degrees
756  // These points lie in the plane defined by the equivalent TVEctor3 plane[4] corners
757 
758  TVector3 a10 = plane[1] - plane[0];
759  TVector3 a21 = plane[2] - plane[1];
760  TVector3 normal = a10.Cross(a21);
761  Double_t d0 = plane[3] * normal;
762 
763  TVector3 l;
764  l.SetMagThetaPhi(1., thetamin * TMath::DegToRad(), phimin * TMath::DegToRad());
765  corners[3] = l * (d0 / (l * normal));
766  l.SetMagThetaPhi(1., thetamax * TMath::DegToRad(), phimin * TMath::DegToRad());
767  corners[2] = l * (d0 / (l * normal));
768  l.SetMagThetaPhi(1., thetamax * TMath::DegToRad(), phimax * TMath::DegToRad());
769  corners[1] = l * (d0 / (l * normal));
770  l.SetMagThetaPhi(1., thetamin * TMath::DegToRad(), phimax * TMath::DegToRad());
771  corners[0] = l * (d0 / (l * normal));
772 }

887 void INDRAGeometryBuilder::Build(const KVNumberList& rings, const KVNameValueList& detectors)
888 {

890  // only the detectors which are in the KVNameValueList:
891  //
892  // KVNameValueList dets;

901  // Include "ring+100" in number list to build etalon telescope for "ring"
902  //
903  // This method will not destroy any pre-existing geometry (gGeoManager),
904  // but will create one if none exists.

914  Error("Build", "You must defined gGeoManager with KVMultiDetArray::CreateGeoManager before calling this method");
915  return;

917 
918  if (rings.Contains(1)) {
919  if (detectors.HasParameter("PHOS")) MakeRing("PHOS", 1);
920  if (detectors.HasParameter("SI"))MakeRing("SI", 1);
921  if (detectors.HasParameter("CSI"))MakeRing("CSI", 1);
922  }
923  for (int ring = 2; ring <= 16; ring += 2) {
924  if (rings.Contains(ring) && detectors.HasParameter("CI")) MakeRing("CI", ring);

939  BuildTarget();

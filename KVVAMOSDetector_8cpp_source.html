156 Bool_t KVVAMOSDetector::Fired(Option_t* opt, Option_t* optP)
157 {
158  //Returns kTRUE if detector was hit (fired) in an event
159  //
160  //The actual meaning of hit/fired depends on the context and the option string opt.
161  //
162  //opt="any" (default) and optP="":
163  //Returns true if at least one working acquisition parameter
164  //associated with the detector was fired in an event, for ANY of the types
165  //in the list*.
166  //
167  //opt="all" and optP="" :
168  //Returns true if at least one working acquisition parameter
169  //associated with the detector was fired in an event, for ALL of the types
170  //in the list*.
171  //
172  //opt="any" and optP="P" :
173  //Returns true if at least one working acquisition parameter
174  //associated with the detector was fired in an event and have a value
175  //greater than their pedestal value, for ANY of the types in the list*.
176  //
177  //opt="all" and optP="P":
178  //Returns true if at least one working acquisition parameter
179  //associated with the detector was fired in an event and have a value
180  //greater than their pedestal value, for ALL of the types in the list*.
181  //
182  // *the actual parameters taken into account can be fine tuned using environment variables such as
183  // KVVAMOSDetector.Fired.ACQParameterList.[type]: Q,E,T,T_HF,X,Y
184  // See KVAMOSDetector::SetFiredBitmask() for more details.
185 
186 
187  if (!IsDetecting()) return kFALSE; //detector not working, no answer at all
188 
189  Bool_t opt_all = !strcmp(opt, "all");
190  Binary8_t event; // bitmask for event
191 
192  // Look at the three first bits for XYZ positions
193  UChar_t xyz_mask = fFiredMask.Subvalue(2, 3);
194 // Info("Fired","Option %s, FiredBitmask %s, xyz_mask (%d)",opt,fFiredMask.String(), xyz_mask);
195  if (xyz_mask) {
196  Double_t xyz[3];
197  UChar_t xyz_res = GetRawPosition(xyz);
198 // cout<<Form(" xyz_res (%d)",xyz_res)<<endl;
199  if (opt_all && (xyz_mask != xyz_res)) return kFALSE;
200  event.Set(xyz_res);
201  }
202 
203  // Look at the other bits for ACQ Parameters
204  UChar_t Nbits = fFiredMask.GetNBits();
205  Binary8_t keep_up(fFiredMask.Subvalue(Nbits - 1, Nbits - 3));
206 // Info("Fired","keep_up %s",keep_up.String());
207  UChar_t id;
208  TIter next(GetACQParamList());
209  TIter next_t(GetTACQParamList());
210  KVACQParam* par;
211  while (keep_up.Value() && ((par = (KVACQParam*)next()) || (par = (KVACQParam*)next_t()))) {
212  if (par->IsWorking() && par->Fired(optP)) {
213  id = GetACQParamTypeIdxFromID(par->GetUniqueID());
214  event.SetBit(id + 3);
215  keep_up.ResetBit(id);
216  }
217 // Info("Fired","%s is %s fired, keep_up %s",par->GetName(), par->Fired( optP ) ? "" : "NOT" ,keep_up.String());
218  }
219 
220  Binary8_t ok = fFiredMask & event;
221 // Info("Fired","ok %s", ok.String());
222 
223  // "all" considered parameters fired if ok == mask
224  // "any" considered parameters fired if ok != 0
225  if (opt_all) return (ok == fFiredMask);
226  return (ok != "0");
227 }

251 void KVVAMOSDetector::SetCalibrators()
252 {
253  // Setup the calibrators for this detector. Call once name
254  // has been set.
255  // The calibrators are KVFunctionCal.
256  // By default the all the calibration functions are first-degree
257  // polynomial function and the range [Xmin,Xmax]=[0,4096].
258  // Here the calibrator are not ready (KVFunctionCal::GetStatus()).
259  // You have to give the parameters and change the status
260  // (see KVFunctionCal::SetParameter(...) and KVFunctionCal::SetStatus(...))
261 
262 
263  TIter nextpar(GetACQParamList());
264 
265  KVACQParam* par = NULL;
266  Double_t maxch = 16384.; // 14 bits
267 
268  TString calibtype("ERROR");
269 
270  while ((par = (KVACQParam*)nextpar())) {
271  Bool_t isTparam = kFALSE;
272 
273  if (par->IsType("E")) {
274  calibtype = "channel->MeV";
275  } else if (par->IsType("Q")) {
276  calibtype = "channel->Volt";
277  maxch = 4096.; // 12 bits
278  } else if (par->GetType()[0] == 'T') {
279  isTparam = kTRUE;
280  calibtype = "channel->ns";
281  } else continue;
282 
283  calibtype.Append(" ");
284  calibtype.Append(par->GetName());
285 
286  TF1* func = new TF1(calibtype.Data(), "pol1", 0., maxch);
287  KVFunctionCal* c = new KVFunctionCal(this, func);
288  c->SetType(calibtype.Data());
289  c->SetLabel(par->GetLabel());
290  c->SetNumber(par->GetNumber());
291  c->SetUniqueID(par->GetUniqueID());
292  c->SetACQParam(par);
293  c->SetStatus(kFALSE);
294  if (!AddCalibrator(c)) delete c;
295  else if (isTparam) {
296  if (!fTlist) fTlist = new TList;
297  fTlist->Add(par);
298  if (!fT0list) fT0list = new TList;
299  fT0list->Add(new KVNamedParameter(par->GetName(), 0.));
300  }
301  }
302 
303  // Define and set to zero the T0 values for time of flight measurment
304  // from this detector. The time of flight acq parameters are associated
305  // to gVamos
306  if (gVamos) {
307  TIter next_vacq(gVamos->GetVACQParams());
308  while ((par = (KVACQParam*)next_vacq())) {
309  if ((par->GetType()[0] == 'T') && IsStartForT(par->GetName() + 1)) {
310  if (!fTlist) fTlist = new TList;
311  fTlist->Add(par);
312  if (!fT0list) fT0list = new TList;
313  fT0list->Add(new KVNamedParameter(par->GetName(), 0.));
314  }
315  }
316  }

662 void KVVAMOSDetector::SetFiredBitmask(KVString& lpar_dummy)
663 {
664  // Set bitmask used to determine which acquisition parameters are
665  // taken into account by KVVAMOSDetector::Fired based on the environment variables
666  // [dataset].KVACQParam.[par name].Working: NO
667  // [classname].Fired.ACQParameterList.[type]: PG,GG,T
668  // where [classname]=KVDetector by default, or the name of some class
669  // derived from KVDetector which calls the method KVDetector::SetKVDetectorFiredACQParameterListFormatString()
670  // in its constructor.
671  // The first allows to define certain acquisition parameters as not functioning;
672  // they will not be taken into account.
673  // The second allows to "fine-tune" what is meant by "all" or "any" acquisition parameters
674  // (i.e. when using Fired("all"), Fired("any"), Fired("Pall", etc.).
675  // For each detector type, give a comma-separated list of the acquisition
676  // parameter types to be taken into account in the KVDetector::Fired method.
677  // Only those parameters which appear in the list will be considered:
678  // then "all" means => for each type in the list at least one acquisition
679  // parameter has to be fired
680  // and "any" means => at least for one type of the list an acquisition parameter
681  // has to be fired
682  // These lists are read during construction of VAMOS (KVVAMOS::Build),
683  // the method KVVAMOS::SetACQParams uses them to define a mask for each detector
684  // of the spectrometer.
685  // X, Y and Z types that we can set in a list are not associated to acquisition
686  // parameters. In the bitmask, the 3 first bits are kept for these coordinates.
687  // If one of these 3 bits is 1 then the methode GetRawPosition(Double_t *)
688  // is called and the returned values is compared to these 3 bits.
689  // If no variable [dataset].[classname].Fired.ACQParameterList.[type] exists,
690  // we set a bitmask authorizing all acquisition parameters of the detector, e.g.
691  // if the detector has 3 types of acquisition parameters which are fired byt
692  // no position type then the bitmask will be "111000"
693 
694  IGNORE_UNUSED(lpar_dummy); // lpar is determined below
695 
696  fFiredMask.Set("");
697 
698  KVString inst;
699  inst.Form(GetFiredACQParameterListFormatString(), GetType());
700  KVString lpar = gDataSet->GetDataSetEnv(inst);
701  TObjArray* toks = lpar.Tokenize(",");
702  if (!toks->GetEntries()) {
703  fFiredMask.Set("11111111");
704  delete toks;
705  return;
706  }
707 
708  // 3 first bits for XYZ positions
709  UChar_t idx;
710  const Char_t* pos[3] = {"X", "Y", "Z"};
711  for (Int_t i = 0; i < 3; i++) {
712  idx = GetPositionTypeIdx(pos[i]) ;
713  if ((idx < 9) && (toks->FindObject(pos[i]))) fFiredMask.SetBit(i);
714  }
715 
716  UChar_t Nbits = 3;
717 
718  // other bits for Acquisition parameters
719  UChar_t idx_max = 0;
720  Bool_t found = kFALSE;
721  TIter next(toks);
722  TObject* obj = NULL;
723  while ((obj = next())) {
724  idx = GetACQParamTypeIdx(obj->GetName());
725  if (idx < 9) {
726  found = kTRUE;
727  fFiredMask.SetBit(idx + Nbits);
728  if (idx > idx_max) idx_max = idx;
729  }
730  }
731  delete toks;
732 
733  if (found) Nbits += idx_max + 1;
734 
735  fFiredMask.SetNBits(Nbits);
736 }

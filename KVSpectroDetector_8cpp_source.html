151 void KVSpectroDetector::AddAbsorber(const Char_t* material, TGeoShape* shape, TGeoMatrix* matrix, Bool_t active)
152 {
153  // Add an absorber material defined as a daughter volume TGeoVolume.
154  // This volume is defined by its material and its shape.
155  // It is positioned by supplying a geometrical transformation, with
156  // respect to the local reference frame of the detector.
157  // By default the volume is referenced as "inactive".
158  // If the volume is not the first absorber or if the matrix is not null
159  // the volume is put in an assembly of volumes TGeoVolumeAssembly.
160  //
161  // Input: material - material of the absorber. The list of available
162  // materials can be found with
163  // det->GetRangeTable()->GetListOfMaterials()->ls()
164  // where det is a KVSpectroDetector or another
165  // object inheriting from KVMaterial.
166  // If material is empty, the name of the detector
167  // is used.
168  // shape - shape of the volume.
169  // matrix - matrix of the geometrical transformation.
170  // active - kTRUE if the absorber is active, kFALSE othewise.
171 
172 
173  TString vol_name;
174  vol_name.Form("%s_%d_%s", GetName(), fNumVol++, material);
175  TGeoVolume* vol = GetGeoVolume(vol_name.Data(), material, shape);
176  if (!vol) {
177  Error("AddAbsorber", "Impossible to build the volume");
178  return;
179  }
180  AddAbsorber(vol, matrix, active);
181 }

193 void KVSpectroDetector::AddAbsorber(TGeoVolume* vol, TGeoMatrix* matrix, Bool_t active)
194 {
195  // Add an absorber material defined as a daughter volume TGeoVolume
196  // and positioned by supplying a geometrical transformation matrix, with
197  // respect to the local reference frame of the detector.
198  // By default the volume is referenced as "inactive".
199  // If the volume is not the first absorber or if the matrix is not null
200  // the volume is put in an assembly of volumes TGeoVolumeAssembly.
201 
202 
203  if (active) SetActiveVolume(vol);
204 
205  AddAbsorberLayer(vol, active);
206 
207  TGeoVolume* prev_vol = GetAbsGeoVolume();
208 
209  if (!prev_vol && !matrix) {
210  // If there is not any absorber and any matrix
211  // the volume is considered as the volume of the detector
212  SetAbsGeoVolume(vol);
213  return;
214  }
215 
216  TGeoVolume* vol_as = NULL;
217 
218  // Case where we want to position with the matrix the absorber which
219  // is the first one of the detector
220  if (!prev_vol) vol_as = gGeoManager->MakeVolumeAssembly(Form("%s_%d", GetName(), fNumVol++));
221  else if (!prev_vol->InheritsFrom("TGeoVolumeAssembly")) {
222 
223  // build an assembly of volumes if the volume of the
224  // detector is not and put the absorbers inside
225  vol_as = gGeoManager->MakeVolumeAssembly(Form("%s_%d", GetName(), fNumVol++));
226  vol_as->AddNode(prev_vol, 1);
227  } else vol_as = prev_vol;
228 
229  // Add the volume in the assembly of volumes
230  vol_as->AddNode(vol, 1, matrix);
231  SetAbsGeoVolume(vol_as);
232 }

241 void KVSpectroDetector::AddAbsorberLayer(TGeoVolume* vol, Bool_t active)
242 {
243  // Add an absorber layer to the detector made from the shape of the
244  // volume.
245  // If active = kTRUE the layer is set active.
246 
247  TGeoMaterial* material = vol->GetMaterial();
248  KVIonRangeTableMaterial* irmat = KVMaterial::GetRangeTable()->GetMaterial(material);
249  if (!irmat) {
250  Warning("AddAbsorberLayer", "Unknown material %s/%s used in layer %s of detector %s",
251  material->GetName(), material->GetTitle(), vol->GetName(), GetName());
252  return;
253  }
254  TGeoBBox* sh = dynamic_cast<TGeoBBox*>(vol->GetShape());
255  if (!sh) {
256  Warning("AddAbsorberLayer", "Unknown shape class %s used in layer %s of detector %s",
257  vol->GetShape()->ClassName(), vol->GetName(), GetName());
258  return; // just in case - for now, all shapes derive from TGeoBBox...
259  }
260  Double_t width = 2.*sh->GetDZ(); // thickness in centimetres
261  KVMaterial* absorber;
262  if (irmat->IsGas()) {
263  Double_t p = material->GetPressure();
264  Double_t T = material->GetTemperature();
265  absorber = new KVMaterial(irmat->GetType(), width, p, T);
266  } else
267  absorber = new KVMaterial(irmat->GetType(), width);
268  KVDetector::AddAbsorber(absorber);
269  absorber->SetAbsGeoVolume(vol);
270  ClearHits();
271  if (active) SetActiveLayer(absorber);
272 }

284 Bool_t KVSpectroDetector::BuildGeoVolume(TEnv* infos, TGeoVolume* ref_vol)
285 {
286  // Build the detector geometry (i.e. the volumes) from informations loaded
287  // in a TEnv object.
288  //
289  // If the detector has already a volume, i.e. the detector is
290  // already built then this method does nothing.
291  // The volumes will be placed inside the ref_vol if it is given in input.
292 
293  if (GetAbsGeoVolume()) {
294  Error("BuildGeoFromFile", "Volume already existing");
295  return kFALSE;
296  }
297 
298  const Char_t* infotype[] = {"NABS", "WIDTH", "HEIGHT", "X", "Y", "Z"};
299  Binary8_t errorbit;
300  Int_t Nbit = 0;
301  TString miss;
302 
303  Int_t Nabs = GetDetectorEnv(infotype[Nbit++], -1, infos);
304  if (Nabs < 0) errorbit.SetBit(Nbit - 1);
305 
306  Double_t width = GetDetectorEnv(infotype[Nbit++], -1., infos);
307  if (width < 0) errorbit.SetBit(Nbit - 1);
308 
309  Double_t height = GetDetectorEnv(infotype[Nbit++], -1., infos);
310  if (height < 0) errorbit.SetBit(Nbit - 1);
311 
312  Double_t pos[3]; // X, Y, Z of the detector origine in the reference volume
313  // This positions have to be present if ref_vol is defined
314  if (ref_vol) {
315  for (Int_t i = 0; i < 3; i++) {
316  pos[i] = GetDetectorEnv(infotype[Nbit++], -666., infos);
317  if (pos[i] <= -666) errorbit.SetBit(Nbit - 1);
318  }
319  }
320 
321  // check if main informations are not missing
322  if (errorbit.Value()) {
323  for (Int_t i = 0; i < Nbit; i++) {
324  if (!errorbit.TestBit(i)) continue;
325  miss += (miss.IsNull() ? "" : "; ");
326  miss += infotype[i];
327  }
328  Error("BuildGeoVolume", "Missing geometry informations (%s) for detector %s", miss.Data(), GetName());
329  return kFALSE;
330  }
331 
332 
333  KVNumberList activeabs(GetDetectorEnv("ACTIVEABS", "", infos));
334  Double_t thick_tot = 0;
335 
336  std::vector<Double_t> thick;
337  thick.resize(Nabs);
338 
339  std::vector<TString> mat;
340  mat.resize(Nabs, "");
341  TString type;
342 
343  miss = "";
344  // look at information concerning layers (thickness, material, ...)
345  for (Int_t i = 0; i < Nabs; i++) {
346  type.Form("ABS.%d.MATERIAL", i + 1);
347  mat[i] = GetDetectorEnv(type.Data(), "", infos);
348  if (mat[i].IsNull()) {
349  miss += (miss.IsNull() ? "" : "; ");
350  miss += type;
351  }
352 
353  type.Form("ABS.%d.THICK", i + 1);
354  thick[i] = GetDetectorEnv(type.Data(), -1., infos);
355  if (thick[i] < 0) {
356  miss += (miss.IsNull() ? "" : "; ");
357  miss += type;
358  }
359 
360  thick_tot += thick[i];
361  }
362  if (!miss.IsNull()) {
363  Error("BuildGeoVolume", "Missing geometry informations (%s) for absorbers of detector %s", miss.Data(), GetName());
364  return kFALSE;
365  }
366 
367  // build volumes
368  TGeoShape* shape = NULL;
369  TString tmp;
370  for (Int_t i = 0; i < Nabs; i++) {
371 
372  fTotThick += thick[i];
373 
374  // box shape of the absorber
375  shape = new TGeoBBox(width / 2, height / 2, thick[i] / 2);
376 
377  // build and position absorber in mother volume.
378  // Reference is the center of absorber
379  Double_t ztrans = (thick[i] - thick_tot) / 2;
380  for (Int_t j = 0; j < Nabs - 1; j++) ztrans += (j < i) * thick[j];
381  tmp.Form("%s_mat%d_tr", GetName(), i + 1);
382  TGeoTranslation* tr = (ztrans ? new TGeoTranslation(tmp.Data(), 0., 0., ztrans) : NULL);
383  AddAbsorber(mat[i].Data(), shape, tr, activeabs.Contains(i + 1));
384  }
385 
386  // incline of the detector around X axis
387  Double_t Xincline = GetDetectorEnv("INCLINE.X", 0., infos);
388  if (Xincline) {
389  TGeoVolume* vol_as = gGeoManager->MakeVolumeAssembly(Form("%s_%d", GetName(), fNumVol++));
390  tmp.Form("%s_rot_x", GetName());
391  TGeoRotation* rot = new TGeoRotation(tmp.Data(), 0., Xincline, 0.);
392  vol_as->AddNode(GetAbsGeoVolume(), 1, rot);
393  SetAbsGeoVolume(vol_as);
394  //update thickness of absorbers;
395  TIter next(GetListOfAbsorbers());
396  KVMaterial* abs = NULL;
397  while ((abs = (KVMaterial*)next())) {
398  Double_t tmptick = abs->GetThickness();
399  tmptick /= TMath::Cos(TMath::DegToRad() * Xincline);
400  abs->SetThickness(tmptick);
401  }
402  }
403 
404  UpdateVolumeAndNodeNames();
405 
406  if (!ref_vol) return kTRUE;
407 
408  Double_t ref_pos[3]; // Xref, Yref, Zref
409  for (Int_t i = 0; i < 3; i++) {
410  type.Form("REF.%c", 'X' + i);
411  ref_pos[i] = GetDetectorEnv(type.Data(), 0., infos);
412  pos[i] -= ref_pos[i];
413  }
414 
415  // place the detector in the reference volume 'ref_vol'
416  tmp.Form("%s_tr", GetName());
417  TGeoTranslation* ref_tr = new TGeoTranslation(tmp.Data(), pos[0], pos[1], pos[2]);
418  ref_vol->AddNode(GetAbsGeoVolume(), 1, ref_tr);
419  // Set at this last node the name of the volume
420  // otherwise the detector name returned by a KVGeoNavigator will have an
421  // extra number at the end (corresponding to the number of the node)
422  ((TGeoNodeMatrix*)ref_vol->GetNodes()->Last())->SetName(GetAbsGeoVolume()->GetName());
423 
424  return kTRUE;
425 }

434 Int_t KVSpectroDetector::Compare(const TObject* obj) const
435 {
436  //Compare two KVSpectroDetector objects from the Z coordinate of their
437  //first active volume in the Focal Plan reference frame. Returns 0 when Z
438  //is equal, -1 if this is smaller and +1 when bigger.
439 
440  Double_t xyz[] = {0, 0, 0, 0, 0, 0};
441 
442  Bool_t ok = kTRUE;
443  ok &= ((KVSpectroDetector*)obj)->ActiveVolumeToFocal(xyz, xyz);
444  ok &= ((KVSpectroDetector*)this)->ActiveVolumeToFocal(xyz + 3, xyz + 3);
445  if (!ok) {
446  Warning("Compare", "Impossible to compare both %s objects %s and %s", ClassName(), GetName(), obj->GetName());
447  return 0;
448  }
449  Double_t Delta = xyz[5] - xyz[2];
450  if (Delta == 0) return 0;
451  if (Delta < 0) return -1;
452  return 1;
453 }

485 TGeoHMatrix& KVSpectroDetector::GetActiveVolToFocalMatrix(Int_t i) const
486 {
487  // Returns the matrix which transforms coordinates form the reference
488  // frame of the active volume 'i' to the reference frame of the focal
489  // plan.
490 
491  static TGeoHMatrix mm;
492  mm.Clear();
493 
494  TGeoVolume* vol = (TGeoVolume*)fActiveVolumes->At(i);
495 
496  if (vol) {
497 
498  Int_t prev_id = gGeoManager->GetCurrentNodeId();
499  // To be sure that the matrices will be calculated again
500  gGeoManager->CdTop();
501 
502  gGeoManager->CdNode(vol->GetUniqueID());
503 
504  // matrix 'volume to target'
505  TGeoMatrix* vol_to_tgt = gGeoManager->GetCurrentMatrix();
506 
507  if (fFocalToTarget) mm = fFocalToTarget->Inverse() * (*vol_to_tgt);
508  else mm = *vol_to_tgt;
509 
510  mm.SetName(Form("%s_to_focal", vol->GetName()));
511 
512  // just to not create problems if this method is called during a tracking
513  gGeoManager->CdNode(prev_id);
514  } else mm.SetName("Identity_matrix");
515 
516  return mm;
517 }

545 TGeoMedium* KVSpectroDetector::GetGeoMedium(const Char_t* mat_name)
546 {
547  // By default, return pointer to TGeoMedium corresponding to this KVMaterial.
548  // If argument "mat_name" is given, a pointer to a medium is return for this material.
549  // mat_name = "Vacuum" is a special case: if the "Vacuum" does not exist, we create it.
550  //
551  // Instance of geometry manager class TGeoManager must be created before calling this
552  // method, otherwise 0x0 will be returned.
553  // If the required TGeoMedium is not already available in the TGeoManager, we create
554  // a new TGeoMedium corresponding to the material given in argument.
555 
556  if (!gGeoManager) return NULL;
557 
558 
559  TString medName, matName;
560  if (!strcmp(mat_name, "")) {
561  // for gaseous materials, the TGeoMedium/Material name is of the form
562  // gasname_pressure
563  // e.g. C3F8_37.5 for C3F8 gas at 37.5 torr
564  // each gas with different pressure has to have a separate TGeoMaterial/Medium
565  matName = GetName();
566  KVIonRangeTableMaterial* irmat = KVMaterial::GetRangeTable()->GetMaterial(matName.Data());
567  if (irmat->IsGas()) medName.Form("%s_%f", matName.Data(), GetPressure());
568  else medName = GetName();
569  } else {
570  matName = mat_name;
571  medName = mat_name;
572  }
573 
574  TGeoMedium* gmed = gGeoManager->GetMedium(medName);
575  if (gmed) return gmed;
576 
577  TGeoMaterial* gmat = gGeoManager->GetMaterial(medName);
578  if (!gmat) {
579  if (!strcmp(matName.Data(), "Vacuum")) {
580  // create material
581  gmat = new TGeoMaterial("Vacuum", 0, 0, 0);
582  } else {
583  // create material
584  gmat = GetRangeTable()->GetTGeoMaterial(matName.Data());
585  if (!gmat) {
586  Error("GetGeoMedium", "Material %s is nowhere to be found in %s"
587  , matName.Data(), GetRangeTable()->GetName());
588  return NULL;
589  }
590  gmat->SetPressure(GetPressure());
591  gmat->SetTemperature(GetTemperature());
592  gmat->SetTransparency(0);
593  }
594  }
595 
596  // For the moment the names of material and medium do not
597  // depend on the temperature of the material.
598  gmat->SetName(medName);
599  gmat->SetTitle(matName);
600 
601  // create medium
602  TGeoMedium* lastmed = (TGeoMedium*)gGeoManager->GetListOfMedia()->Last();
603  Int_t numed = (lastmed ? lastmed->GetId() + 1 : 0); // static counter variable used to number media
604  gmed = new TGeoMedium(medName, numed, gmat);
605  numed += 1;
606 
607  return gmed;
608 }

830 UChar_t KVSpectroDetector::GetPosition(Double_t* XYZf, Char_t /* dir */, Int_t idx)
831 {
832  // Returns in the array 'XYZf', the coordinates (in cm) of a point randomly drawn in the
833  // active volume with index 'idx'. We assume that the shape of this
834  // volume is a TGeoBBox. These coordinates are given in the focal-plane
835  // of reference.
836  // The function returns a binary code where:
837  // - bit 0 = 1 if X is OK (001);
838  // - bit 1 = 1 if Y is OK (010);
839  // - bit 2 = 1 if Z is OK (100);
840  //
841  // If no coordinates are OK, the returned value is null (000) and if
842  // X, Y and Z are OK then the returned value is equal 7 (111).
843  //
844  // Argument 'dir' is used in child classes.
845 
846  TGeoVolume* vol = GetActiveVolume(idx);
847  if (!vol || !PositionIsOK()) return 0;
848 
849  const TGeoBBox* box = (TGeoBBox*)vol->GetShape();
850  Double_t dx = box->GetDX();
851  Double_t dy = box->GetDY();
852  Double_t dz = box->GetDZ();
853  Double_t ox = (box->GetOrigin())[0];
854  Double_t oy = (box->GetOrigin())[1];
855  Double_t oz = (box->GetOrigin())[2];
856  Double_t xyz[3];
857  xyz[0] = ox + (TestBit(kRdmPos) ? dx * (2 * gRandom->Rndm() - 1) : 0.);
858  xyz[1] = oy + (TestBit(kRdmPos) ? dy * (2 * gRandom->Rndm() - 1) : 0.);
859  xyz[2] = oz + (TestBit(kRdmPos) ? dz * (2 * gRandom->Rndm() - 1) : 0.);
860  if (ActiveVolumeToFocal(xyz, XYZf, idx)) return 7;
861  return 0;
862 }

934 void KVSpectroDetector::UpdateVolumeAndNodeNames()
935 {
936  // Update the names of Volumes and the names of the Nodes of this
937  // detector.
938  // The name of the volume representing the detector (returned
939  // by GetAbsGeoVolume()) is DET_<detector name>.
940  // The name of the active volumes is ACTIVE_<detector name>_<material name>.
941  // The name of the other volumes is <detector name>_<material name>.
942 
943  GetAbsGeoVolume()->SetName(Form("DET_%s", GetName()));
944  TObjArray* nodes = GetAbsGeoVolume()->GetNodes();
945  TGeoNode* node = NULL;
946  TGeoVolume* vol = NULL;
947 
948  TIter next(nodes);
949  while ((node = (TGeoNode*)next())) {
950  TString name, nname;
951  vol = node->GetVolume();
952  name.Form("%s_%s", GetName(), vol->GetMaterial()->GetName());
953  if (GetActiveVolumes()->Contains(vol))
954  name.Prepend("ACTIVE_");
955  vol->SetName(name.Data());
956  nname = name;
957  Int_t i = 0;
958  while (nodes->FindObject(nname.Data()))
959  nname.Form("%s_%d", name.Data(), ++i);
960  node->SetName(nname.Data());
961  node->SetNumber(i);
962  }

40 Double_t CalculLumiereVamos(Double_t* x, Double_t* par)
41 {
42  //Calcul de la lumiere totale a partir de Z, A d'une particule et son energie
43  //
44  // x[0] = energie (MeV)
45  // par[0] = a1
46  // par[1] = a2
47  // par[2] = a3 =0
48  // par[3] = Z
49  // par[4] = A
50 
51  Double_t Z = par[3];
52  Double_t A = par[4];
53  Double_t w = A * Z * Z;
54  Double_t energie = x[0] / w;
55 
56  Double_t lumcalc = par[0] * (energie - par[1] * TMath::Log(1. + energie / par[2]));
57 
58  return lumcalc * w;
59 }

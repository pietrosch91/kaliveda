16 <h2>KVIDZAFromZGrid</h2>
17 <h4>Hybrid identification grid</h4>
18 Such a grid can identify simultaneously both the mass and charge of detected
19 particles (if contains a convertion table for the given element), or solely the charge
20 (if no convertion table).
21 
22 The convertion table has to be included in the parameter list of the grid as follow:
23 
24 <PARAMETER> PIDRANGE=[zmax]
25 <PARAMETER> PIDRANGE[z]=[a1]:[PID min],[PID mean],[PID max]|[a2]:[PID min],[PID mean],[PID max]|[a3]:[PID min],[PID mean],[PID max]
26 (...)
27 
28 with:
29 <ul>
30  <li> [zmax] highest Z for witch A will be extracted</li>
31  <li> [z] charge corresponding to the following line</li>
32  <li> [ai] mass corresponding to the following interval</li>
33  <li> [PID mean] PID value of the peak of corresponding to mass [ai]</li>
34  <li> [PID min/max] PID interval corresponding to mass [ai] used to attribute the ICODE (optional)</li>
35 </ul>
36 
37 <h3>Identification quality codes</h3>
38 After each identification attempt, the value returned by GetQualityCode() indicates whether the
39 identification was successful or not. The meaning of the different codes depends on the type
40 of identification.
41 
42 <h4>Z & A (mass & charge) isotopic identification grid</h4>
43 <ul>
44  <li> KVIDZAGrid::kICODE0, ok</li>
45  <li> KVIDZAGrid::kICODE1, Z ok, slight ambiguity of A, which could be larger</li>
46  <li> KVIDZAGrid::kICODE2, Z ok, slight ambiguity of A, which could be smaller</li>
47  <li> KVIDZAGrid::kICODE3, Z ok, slight ambiguity of A, which could be larger or smaller</li>
48  <li> KVIDZAGrid::kICODE4, point out of mass identification intervals, strong ambiguity of A</li>
49  <li> KVIDZAGrid::kICODE5, point is in between two isotopes of different Z, too far from either to be considered well-identified</li>
50  <li> KVIDZAGrid::kICODE6, (x,y) is below first line in grid</li>
51  <li> KVIDZAGrid::kICODE7, (x,y) is above last line in grid</li>
52  <li> KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid</li>
53 </ul>
54 
55 <h4>Z-only charge identification grid</h4>
56 <ul>
57  <li> KVIDZAGrid::kICODE0, ok</li>
58  <li> KVIDZAGrid::kICODE1, slight ambiguity of Z, which could be larger</li>
59  <li> KVIDZAGrid::kICODE2, slight ambiguity of Z, which could be smaller</li>
60  <li> KVIDZAGrid::kICODE3, slight ambiguity of Z, which could be larger or smaller</li>
61  <li> KVIDZAGrid::kICODE4, point is in between two lines of different Z, too far from either to be considered well-identified</li>
62  <li> KVIDZAGrid::kICODE5, point is in between two lines of different Z, too far from either to be considered well-identified</li>
63  <li> KVIDZAGrid::kICODE6, (x,y) is below first line in grid</li>
64  <li> KVIDZAGrid::kICODE7, (x,y) is above last line in grid</li>
65  <li> KVIDZAGrid::kICODE8, no identification: (x,y) out of range covered by grid</li>
66 </ul>
67 
68 In both cases, an acceptable identification is achieved if the quality code is kICODE0, kICODE1, kICODE2, or kICODE3.<br>
69 Points with codes kICODE4 or kICODE5 are normally considered as "noise" and should be rejected.<br>
70 Points which are (vertically) out of range for this grid have code kICODE6 (point too far below) or kICODE7 (point too far above).<br>
71 Points with code kICODE8 are totally out of range.

270  pid = val.Next().Atof();
271  pidmax = val.Next().Atof();
272  itv->add(aa, pid, pidmin, pidmax);
273 // itv->add(aa, pid, pid-0.02, pid+0.02);

276  if (zz < fZminInt) fZminInt = zz;

420 void KVIDZAFromZGrid::Identify(Double_t x, Double_t y, KVIdentificationResult* idr) const
421 {
422  // Fill the KVIdentificationResult object with the results of identification for point (x,y)
423  // corresponding to some physically measured quantities related to a reconstructed nucleus.
424  //
425  // By default (OnlyZId()=kFALSE) this means identifying the Z & A of the nucleus.
426  // In this case, we consider that the nucleus' Z & A have been correctly measured
427  // if the 'quality code' returned by IdentZA() is < kICODE4:
428  // we set idr->Zident and idr->Aident to kTRUE if fICode<kICODE4
429  //
430  // If OnlyZId()=kTRUE, only the Z of the nucleus is established.
431  // In this case, we consider that the nucleus' Z has been correctly measured
432  // if the 'quality code' returned by IdentZ() is < kICODE4, thus:
433  // we set idr->Zident to kTRUE if fICode<kICODE4
434  // The mass idr->A is set to the mass of the nearest line.
435  //
436  // Real & integer masses for isotopically identified particles
437  // ===================================================
438  // For points lying between two lines of same Z and different A (fICode<kIDCode4)
439  // the "real" mass is given by interpolation between the two masses.
440  // The integer mass is the A of the line closest to the point.
441  // This means that the integer A is not always = nint("real" A),
442  // as for example if a grid is drawn with lines for 7Be & 9Be but not 8Be
443  // (usual case), then particles between the two lines can have "real" masses
444  // between 7.5 and 8.5, but their integer A will be =7 or =9, never 8.
445  //
446 
447  idr->IDOK = kFALSE;
448  if (!const_cast<KVIDZAFromZGrid*>(this)->FindFourEmbracingLines(x, y, "above")) {
449  const_cast < KVIDZAFromZGrid* >(this)->fICode = kICODE8; // Z indetermine ou (x,y) hors limites
450  idr->IDquality = kICODE8;
451  idr->SetComment("no identification: (x,y) out of range covered by grid");
452  return;
453  }
454 
455  Double_t Z;
456  const_cast < KVIDZAFromZGrid* >(this)->IdentZ(x, y, Z);
457  idr->IDquality = fICode;
458  if (fICode < kICODE4 || fICode == kICODE7) {
459  idr->Zident = kTRUE;
460  }
461  if (fICode < kICODE4) {
462  idr->IDOK = kTRUE;
463  }
464  idr->Z = Zint;
465  idr->PID = Z;
466  idr->Aident = kFALSE;
467 
468  if ((fPIDRange && (idr->IDOK) && (idr->Z <= fZmaxInt) && (idr->Z > fZminInt - 1) && (const_cast < KVIDZAFromZGrid* >(this)->is_inside(Z)))
469  && ((!fHasMassCut) || (fHasMassCut && GetIdentifier("MassID")->IsInside(x, y)))) {
470 // Info("Identify","try mass ID..");
471  const_cast < KVIDZAFromZGrid* >(this)->DeduceAfromPID(idr); // IDQuality and comments assigned here
472  if (idr->IDquality < kICODE4) { // should always be true: to be verified...
473  if (const_cast < KVIDZAFromZGrid* >(this)->GetIntervalSet(idr->Z)->GetNPID() > 1) idr->Aident = kTRUE;
474  idr->IDOK = kTRUE;
475  } else if (idr->IDquality == kICODE4) idr->IDquality = kICODE3;
476  } else {
477  switch (fICode) {
478  case kICODE0:
479  idr->SetComment("ok");
480  break;
481  case kICODE1:
482  idr->SetComment("slight ambiguity of Z, which could be larger");
483  break;
484  case kICODE2:
485  idr->SetComment("slight ambiguity of Z, which could be smaller");
486  break;
487  case kICODE3:
488  idr->SetComment("slight ambiguity of Z, which could be larger or smaller");
489  break;
490  case kICODE4:
491  idr->SetComment("point is in between two lines of different Z, too far from either to be considered well-identified");
492  break;
493  case kICODE5:
494  idr->SetComment("point is in between two lines of different Z, too far from either to be considered well-identified");
495  break;
496  case kICODE6:
497  idr->SetComment("(x,y) is below first line in grid");
498  break;
499  case kICODE7:
500  idr->SetComment("(x,y) is above last line in grid");
501  break;
502  default:
503  idr->SetComment("no identification: (x,y) out of range covered by grid");
504  }
505  }
506 }

530 double interval_set::eval(KVIdentificationResult* idr)
531 {
532  double pid = idr->PID;
533  if (pid < 0.5) return 0.;
534  double res = fPIDs.Eval(pid);
535  int ares = 0;
536 
537  if (fType == KVIDZAFromZGrid::kIntType) {
538  for (int ii = 0; ii < fNPIDs; ii++) {
539  if (((interval*)fIntervals.At(ii))->is_inside(pid)) {
540  ares = ((interval*)fIntervals.At(ii))->GetA();
541  break;
542  }
543  }
544  if (ares != 0) {
545  idr->A = ares;
546  idr->PID = res;
547  idr->IDquality = KVIDZAGrid::kICODE0;
548  idr->SetComment("ok");
549  } else {
550  ares = TMath::Nint(res);
551  idr->A = ares;
552  idr->PID = res;
553  if (pid > ((interval*)fIntervals.At(0))->GetPIDmin() && pid < ((interval*)fIntervals.At(fNPIDs - 1))->GetPIDmax()) {
554  idr->IDquality = KVIDZAGrid::kICODE3;
555  idr->SetComment("slight ambiguity of A, which could be larger or smaller");
556  } else {
557  idr->IDquality = KVIDZAGrid::kICODE4;
558  idr->SetComment("point out of mass identification intervals, strong ambiguity of A");
559  }
560  }
561  } else {
562  ares = TMath::Nint(res);
563  idr->A = ares;
564  idr->PID = res;
565  if (ares > fPIDs.GetX()[0] && ares < fPIDs.GetX()[fNPIDs - 1]) {
566  idr->IDquality = KVIDZAGrid::kICODE0;
567  idr->SetComment("ok");
568  } else {
569  idr->IDquality = KVIDZAGrid::kICODE4;
570  idr->SetComment("point out of mass identification intervals, strong ambiguity of A");
571  }
572  }
573  return res;
574 }

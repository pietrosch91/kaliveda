8 // Description
9 //
10 // A simple Silicon-Isobutane-CsI detector stack, used in the A value estimation
11 // algorithms - see SiliconEnergyMinimiser. Allows one to calculate the "delta"
12 // between the measured and simulated silicon detector energies in order to
13 // determine the most likely candidate for the A value (ideally "delta" would be
14 // zero so we look for the minimum value of "delta").

132 Bool_t MEDetectorStack::Simulate(
133  const struct me::SimulationParameters* const parameters,
134  struct me::SimulationResult* const result
135 )
136 {
137  assert(kInitialised_);
138 
139  assert(parameters);
140  assert(result);
141 
142  // Check for sensible input parameters
143  assert((parameters->z > 0) && (parameters->z < 120));
144  assert((parameters->a > 0) && (parameters->a < 300));
145  assert((parameters->si_energy > 0.) && (parameters->si_energy < 10000.));
146 
147  // We now need to calculate the incident energy of the nucleus (so that we
148  // can simulate its passage through the idtelescope).
149 
150  // We have the csi_light, so let's work out the csi_energy first using the
151  // light->energy calibration function defined for this caesium iodide
152  // detector.
153 
154  if (!calibrator_) return kFALSE;
155  if (!calibrator_->GetStatus()) return kFALSE;
156 
157  Double_t csi_energy(
158  calibrator_->Compute(
159  parameters->z,
160  parameters->a,
161  parameters->csi_light
162  )
163  );
164 
165  if ((csi_energy <= 0.) || (csi_energy > 10000.)) {
166  // Out of bounds csi energy, discard this event. For some reason there
167  // are occasionally energies of 0. MeV returned for non-zero csi light,
168  // I'm not sure whether that is a fault in the code or if it is the
169  // default return value when TF1::GetX() fails. To see these events
170  // enable the assertions and uncomment the above assertion.
171 
172  return kFALSE;
173  }
174 
175  // We then calculate the incident energy for the isobutane layer, which
176  // resides behind the silicon layer and in front of the caesium iodide. We
177  // calculate this by finding the residual energy just after the silicon
178  // layer (using the energy measured in the silicon).
179 
180  Double_t isobutane_incident_energy(
181  si_->GetEResFromDeltaE(
182  parameters->z,
183  parameters->a,
184  parameters->si_energy
185  )
186  );
187 
188  if (isobutane_incident_energy <= 0.) {
189  // Particle does not punch through the silicon material and so we can
190  // not identify it in the SI-CSI telescope.
191 
192  return kFALSE;
193  }
194 
195  if (isobutane_incident_energy > 10000.) {
196  // This generally happens when the silicon energy (from the silicon
197  // calibration) is too low, this results in an over-estimation of the
198  // isobutane incident energy.
199 
200  return kFALSE;
201  }
202 
203  Double_t isobutane_energy(
204  isobutane_->GetDeltaE(
205  parameters->z,
206  parameters->a,
207  isobutane_incident_energy
208  )
209  );
210 
211  if (isobutane_energy <= 0.) {
212  // Problem finding the energy solution for this point, can not do
213  // anything with it.
214  return kFALSE;
215  }
216 
217  assert((isobutane_energy > 0.) && (isobutane_energy < 10000.));
218 
219  // We now have all the data we need to calculate the incident energy of the
220  // nucleus prior to entering the silicon layer
221 
222  Double_t incident_energy(
223  parameters->si_energy +
224  isobutane_energy +
225  csi_energy
226  );
227 
228  // Now we can simulate the passage of the nucleus through the idtelescope
229  // and calculate 'delta' (the absolute difference between the measured
230  // silicon energy and that simulated for this nucleus)
231 
232  assert((incident_energy > 0.) && (incident_energy < 10000.));
233 
234  sim_nucleus_->Clear();
235  sim_nucleus_->SetZAandE(parameters->z, parameters->a, incident_energy);
236 
237 #if __cplusplus < 201103L
238  si_->DetectParticle(sim_nucleus_);
239  isobutane_->DetectParticle(sim_nucleus_);
240  csi_->DetectParticle(sim_nucleus_);
241 #else
242  si_->DetectParticle(sim_nucleus_.get());
243  isobutane_->DetectParticle(sim_nucleus_.get());
244  csi_->DetectParticle(sim_nucleus_.get());
245 #endif
246 
247  Double_t delta(0.);
248 
249  if (absolute_delta_) {
250  // We've specified absolute delta (default)
251  delta = TMath::Abs(parameters->si_energy - si_->GetEnergyLoss());
252  } else {
253  // Otherwise simply return the difference
254  delta = parameters->si_energy - si_->GetEnergyLoss();
255  }
256 
257  // Pack the result struct
258 
259  result->z = parameters->z;
260  result->a = parameters->a;
261  result->si_energy = si_->GetEnergyLoss();
262  result->gap_energy = isobutane_energy;
263  result->csi_energy = csi_energy;
264  result->delta = delta;
265 
266  // Clear the materials for the next simulation
267  si_->Clear();
268  isobutane_->Clear();
269  csi_->Clear();
270 
271  return kTRUE;
272 }

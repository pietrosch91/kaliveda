141 Bool_t KVGRUTapeGanilReader::GetNextEvent()
142 {
143  // Get next event from buffer
144  // Returns kTRUE if an event is read.
145  // Returns kFALSE if no acquisition running (check status)
146  // Returns kFALSE if no buffer/event read after fTimeout seconds (check status)
147 
148  fStatus = kSTATUS_OK;
149  if (fReadFirstBuffer) {
150  // first time we call, we need to read a buffer
151  fReadFirstBuffer = kFALSE;
152  fGRUAcq->GetDeviceIn()->ReadBuffer();
153  fGRUAcq->GetEvent()->RazEvent();
154  Info("GetNextEvent", "ReadFirstBuffer: looking for event buffer");
155  while (!fGRUAcq->GetDeviceIn()->GetCurrentBuffer()->IsAeventBuffer()) {
156  fGRUAcq->GetDeviceIn()->ReadBuffer();
157  }
158  Info("GetNextEvent", "ReadFirstBuffer: got event buffer");
159  }
160  int Status = fGRUAcq->GetEvent()->NextEvent(fGRUAcq->GetDeviceIn()->GetCurrentBuffer());
161  Info("GetNextEvent", "Status = %0#x", Status);
162  if (Status != ACQ_OK) {
163  // we have read all events in buffer. get a new buffer.
164  Info("GetNextEvent", "we have read all events in buffer. get a new buffer.");
165  fGRUAcq->GetDeviceIn()->ReadBuffer();
166  fGRUAcq->GetEvent()->RazEvent();
167  Info("GetNextEvent", "looking for event buffer");
168  while (!fGRUAcq->GetDeviceIn()->GetCurrentBuffer()->IsAeventBuffer()) {
169  fGRUAcq->GetDeviceIn()->ReadBuffer();
170  }
171  Info("GetNextEvent", "got buffer, Status = %0#x", Status);
172  while (Status != ACQ_OK) {
173  Status = fGRUAcq->GetEvent()->NextEvent(fGRUAcq->GetDeviceIn()->GetCurrentBuffer());
174  Info("GetNextEvent", "Status = %0#x", Status);
175  }
176  }
177  FillFiredParameterList();
178  return kTRUE;
179 }

26 // Given a valid ROOT geometry, we propagate the particles of an event
27 // and, every time a particle traverses a volume made of a TGeoMaterial
28 // with a name corresponding to a material known by this range table,
29 // we calculate the energy loss of the particle,
30 // and store it in the particle's list KVParticle::fParameters in the form
31 //
32 // "DE:[detector name]" = [energy lost in volume]
33 //
34 // We also store the coordinates (world-frame) of the particle's entry and exit
35 // in each volume:
36 //
37 // "Xin:[detector name]" = [entry coordinates for volume]
38 // "Yin:[detector name]"
39 // "Zin:[detector name]"
40 // "Xout:[detector name]" = [exit coordinates for volume]
41 // "Yout:[detector name]"
42 // "Zout:[detector name]"
43 //
44 // If the particle stops in the volume, the 'out' coordinates will correspond to
45 // the calculated range of the particle in the volume.
46 // In the case of multilayer detectors, "[detector name]" in the previous
47 // examples is replaced by "[detector name]/[layer name]"
48 //
49 // EXAMPLE OF USE
50 // ==============
51 //
52 // In this case the geometry is made up of 3-layer Ionisation Chamber detectors
53 // (window + gas + window), with an "ACTIVE" gas layer, and single-layer
54 // Silicon detectors placed behind the IC.

201 void KVRangeTableGeoNavigator::ParticleEntersNewVolume(KVNucleus* part)
202 {
203  // Overrides method in KVGeoNavigator base class.
204  // Every time a particle enters a new volume, we check the material to see
205  // if it is known (i.e. contained in the range table fRangeTable).
206  // If so, then we calculate the energy loss of the particle in this volume,
207  // reduce the particle's energy accordingly, and, if the energy falls below
208  // a certain cut-off (default 1.e-3 MeV; can be modified with
209  // SetCutOffKEForPropagation(Double_t) ), we stop the propagation.
210  //
211  // The (cumulated) energy losses in the active layers of all hit detectors
212  // are updated with the energy lost by this particle
213 
214  Double_t de = 0;
215  Double_t e = part->GetEnergy();
216  if (e <= fCutOffEnergy) {
217  e = 0.;
218  SetStopPropagation();//propagation will stop after this step
219  if (IsTracking()) {
220  AddPointToCurrentTrack(GetEntryPoint().X(), GetEntryPoint().Y(), GetEntryPoint().Z());
221  AddPointToCurrentTrack(GetExitPoint().X(), GetExitPoint().Y(), GetExitPoint().Z());
222  }
223  return;
224  }
225 
226  // calculate energy losses in known materials for charged particles
227  TGeoMaterial* material = GetCurrentVolume()->GetMaterial();
228  KVIonRangeTableMaterial* irmat = 0;
229  if ((irmat = fRangeTable->GetMaterial(material))) {
230  de = irmat->GetLinearDeltaEOfIon(
231  part->GetZ(), part->GetA(), e, GetStepSize(), 0.,
232  material->GetTemperature(),
233  material->GetPressure());
234  e -= de;
235  if (e <= fCutOffEnergy) {
236  e = 0.;
237  SetStopPropagation();//propagation will stop after this step
238  }
239  //set flag to say that particle has been slowed down
240  part->SetIsDetected();
241  //If this is the first absorber that the particle crosses, we set a "reminder" of its
242  //initial energy
243  if (!part->GetPInitial()) part->SetE0();
244 
245  TString absorber_name;
246  KVDetector* theDet = GetDetectorFromPath(GetCurrentPath());
247  Bool_t active_layer = kFALSE;
248  if (theDet) {
249  if (!theDet->IsSingleLayer()) {
250  absorber_name.Form("%s/%s", theDet->GetName(), GetCurrentNode()->GetName());
251  if (strncmp(GetCurrentNode()->GetName(), "ACTIVE", 6) == 0) active_layer = kTRUE;
252  } else {
253  absorber_name = theDet->GetName();
254  active_layer = kTRUE;
255  }
256  } else
257  absorber_name = irmat->GetName();
258 
259  if (part->GetZ()) {
260  part->GetParameters()->SetValue(Form("DE:%s", absorber_name.Data()), de);
261  if (active_layer) {
262  // update energy loss in active layer of detector
263  Double_t E = theDet->GetEnergyLoss() + de;
264  theDet->SetEnergyLoss(E);
265  theDet->AddHit(part);
266  }
267  }
268  part->GetParameters()->SetValue(Form("Xin:%s", absorber_name.Data()), GetEntryPoint().X());
269  part->GetParameters()->SetValue(Form("Yin:%s", absorber_name.Data()), GetEntryPoint().Y());
270  part->GetParameters()->SetValue(Form("Zin:%s", absorber_name.Data()), GetEntryPoint().Z());
271  if (StopPropagation()) {
272  // If particle stops in this volume, we use as 'exit point' the point corresponding to
273  // the calculated range of the particle
274  Double_t r = irmat->GetRangeOfLastDE() / irmat->GetDensity();
275  TVector3 path = GetExitPoint() - GetEntryPoint();
276  TVector3 midVol = GetEntryPoint() + (r / path.Mag()) * path;
277  part->GetParameters()->SetValue(Form("Xout:%s", absorber_name.Data()), midVol.X());
278  part->GetParameters()->SetValue(Form("Yout:%s", absorber_name.Data()), midVol.Y());
279  part->GetParameters()->SetValue(Form("Zout:%s", absorber_name.Data()), midVol.Z());
280  if (IsTracking()) {
281  AddPointToCurrentTrack(GetEntryPoint().X(), GetEntryPoint().Y(), GetEntryPoint().Z());
282  AddPointToCurrentTrack(midVol.X(), midVol.Y(), midVol.Z());
283  }
284  } else {
285  part->GetParameters()->SetValue(Form("Xout:%s", absorber_name.Data()), GetExitPoint().X());
286  part->GetParameters()->SetValue(Form("Yout:%s", absorber_name.Data()), GetExitPoint().Y());
287  part->GetParameters()->SetValue(Form("Zout:%s", absorber_name.Data()), GetExitPoint().Z());
288  if (IsTracking()) {
289  AddPointToCurrentTrack(GetEntryPoint().X(), GetEntryPoint().Y(), GetEntryPoint().Z());
290  AddPointToCurrentTrack(GetExitPoint().X(), GetExitPoint().Y(), GetExitPoint().Z());
291  }
292  }
293  part->SetEnergy(e);
294  }
295 }

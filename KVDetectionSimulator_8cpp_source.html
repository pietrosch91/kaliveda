83 void KVDetectionSimulator::DetectEvent(KVEvent* event, const Char_t* detection_frame)
84 {
85  //Simulate detection of event by multidetector array.
86  //
87  // optional argument detection_frame(="" by default) can be used to give name of
88  // inertial reference frame (defined for all particles of 'event') to be used.
89  // e.g. if the simulated event's default reference frame is the centre of mass frame, before calling this method
90  // you should create the 'laboratory' or 'detector' frame with KVEvent::SetFrame(...), and then give the
91  // name of the 'LAB' or 'DET' frame as 3rd argument here.
92  //
93  //For each particle in the event we calculate first its energy loss in the target (if the target has been defined, see KVMultiDetArray::SetTarget).
94  //By default these energy losses are calculated from a point half-way along the beam-direction through the target (taking into account the orientation
95  //of the target), if you want random depths for each event call GetTarget()->SetRandomized() before using DetectEvent().
96  //
97  //If the particle escapes the target then we look for the group in the array that it will hit. If there is one, then the detection of this particle by the
98  //different members of the group is simulated.
99  //
100  //The detectors concerned have their fEloss members set to the energy lost by the particle when it crosses them.
101  //
102  //Give tags to the simulated particles via KVNucleus::AddGroup() method
103  //Two general tags :
104  // - DETECTED : cross at least one active layer of one detector
105  // - UNDETECTED : go through dead zone or stopped in target
106  //We add also different sub group :
107  // - For UNDETECTED particles : "DEAD ZONE", "STOPPED IN TARGET" and "THRESHOLD", the last one concerned particle
108  //go through the first detection stage of the multidetector array but stopped in an absorber (ie an inactive layer)
109  // - For DETECTED particles :
110  // "PUNCH THROUGH" corrresponds to particle which cross all the materials in front of it
111  // (high energy particle punh through), or which miss some detectors due to a non perfect
112  // overlap between defined telescope,
113  // "INCOMPLETE" corresponds to particles which stopped in the first detection stage of the multidetector
114  // in a detector which can not give alone a clear identification,
115  // this correponds to status=3 or idcode=5 in INDRA data
116  //
117 
118  // Reset detectors in array hit by any previous events
119  fHitGroups.Clear();
120 
121  event->ResetGetNextParticle();
122  KVNucleus* part;
123  while ((part = event->GetNextParticle())) { // loop over particles
124 
125  KVNucleus* _part = (KVNucleus*)part->GetFrame(detection_frame, kFALSE);
126 
127  KVNameValueList det_stat, nvl;
128  Double_t eLostInTarget = 0;
129 
130  TVector3 initial_momentum = _part->GetMomentum();
131 
132  if (part->GetZ() == 0) {
133  det_stat.SetValue("UNDETECTED", "NEUTRON");
134 
135  part->AddGroup("UNDETECTED");
136  part->AddGroup("NEUTRON");
137  } else if (_part->GetKE() < GetMinKECutOff()) {
138  det_stat.SetValue("UNDETECTED", "NO ENERGY");
139 
140  part->AddGroup("UNDETECTED");
141  part->AddGroup("NO ENERGY");
142  } else {
143  if (IncludeTargetEnergyLoss() && GetTarget()) {
144 
145  GetTarget()->SetOutgoing(kTRUE);
146  //simulate passage through target material
147  Double_t ebef = _part->GetKE();
148  GetTarget()->DetectParticle(_part);
149  eLostInTarget = ebef - _part->GetKE();
150  if (_part->GetKE() < GetMinKECutOff()) {
151  det_stat.SetValue("UNDETECTED", "STOPPED IN TARGET");
152 
153  part->AddGroup("UNDETECTED");
154  part->AddGroup("STOPPED IN TARGET");
155  }
156  GetTarget()->SetOutgoing(kFALSE);
157  }
158 
159  if (_part->GetKE() > GetMinKECutOff()) {
160 
161  nvl = DetectParticle(_part);
162 
163  if (nvl.IsEmpty()) {
164 
165  det_stat.SetValue("UNDETECTED", "DEAD ZONE");
166 
167  part->AddGroup("UNDETECTED");
168  part->AddGroup("DEAD ZONE");
169 
170  } else {
171  part->AddGroup("DETECTED");
172  det_stat.SetValue("DETECTED", "OK");
173  }
174  }
175  }
176 
177  if (IncludeTargetEnergyLoss() && GetTarget()) part->SetParameter("TARGET Out", eLostInTarget);
178  if (!nvl.IsEmpty()) {
179  Int_t nbre_nvl = nvl.GetNpar();
180  KVString LastDet(nvl.GetNameAt(nbre_nvl - 1));
181  part->SetParameter("STOPPING DETECTOR", LastDet.Data());
182  for (Int_t ii = 0; ii < nvl.GetNpar(); ++ii) {
183  part->SetParameter(nvl.GetNameAt(ii), nvl.GetDoubleValue(ii));
184  }
185  }
186  for (Int_t ii = 0; ii < det_stat.GetNpar(); ii += 1) {
187  part->SetParameter(det_stat.GetNameAt(ii), det_stat.GetStringValue(ii));
188  }
189 
190  _part->SetMomentum(initial_momentum);
191 
192  }
193 
194 }

210 KVNameValueList KVDetectionSimulator::DetectParticle(KVNucleus* part)
211 {
212  // Simulate detection of a single particle
213  //
214  // Propagate particle through the array,
215  // calculating its energy losses in all absorbers, and setting the
216  // energy loss members of the active detectors on the way.
217  //
218  // Returns a list containing the name and energy loss of each
219  // detector hit in array (list is empty if none i.e. particle
220  // in beam pipe or dead zone of the multidetector)
221 
222  fArray->GetNavigator()->PropagateParticle(part);
223 
224  // particle missed all detectors
225  if (part->GetParameters()->IsEmpty()) return KVNameValueList();
226 
227  // list of energy losses in active layers of detectors
228  KVNameValueList NVL;
229 
230  // find detectors in array hit by particle
231  KVDetector* last_detector = nullptr;
232  TIter next(part->GetParameters()->GetList());
233  KVNamedParameter* param;
234  while ((param = (KVNamedParameter*)next())) {
235  KVString pname(param->GetName());
236  pname.Begin(":");
237  KVString pn2 = pname.Next();
238  KVString pn3 = pname.Next();
239  if (pn2 == "DE") {
240  pn3.Begin("/");
241  KVString det_name = pn3.Next();
242  if (pn3.End() || pn3.Next().BeginsWith("ACTIVE")) {
243  // energy loss in active layer of detector
244  KVDetector* curDet = last_detector = fArray->GetDetector(det_name);
245  if (!curDet) {
246  Error("DetectParticle",
247  "Cannot find detector %s corresponding to particle energy loss %s",
248  det_name.Data(), pname.Data());
249  } else {
250  Double_t de = param->GetDouble();
251  NVL.SetValue(curDet->GetName(), de);
252  }
253  }
254  }
255  }
256 
257  // add hit group to list if not already in it
258  if (last_detector) fHitGroups.AddGroup(last_detector->GetGroup());
259 
260  return NVL;
261 }

273 KVNameValueList KVDetectionSimulator::DetectParticleIn(const Char_t* detname, KVNucleus* kvp)
274 {
275  // Given the name of a detector, simulate detection of a given particle
276  // by the complete corresponding group. The particle's theta and phi are set
277  // at random within the limits of detector entrance window
278  //
279  // Returns a list containing the name and energy loss of each
280  // detector hit in array
281 
282  KVDetector* kvd = GetArray()->GetDetector(detname);
283  if (kvd) {
284  kvp->SetMomentum(kvp->GetEnergy(), kvd->GetRandomDirection("random"));
285  return DetectParticle(kvp);
286  } else {
287  Error("DetectParticleIn", "Detector %s not found", detname);
288  }
289  return KVNameValueList();
290 }

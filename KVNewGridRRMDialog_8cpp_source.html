36 KVNewGridRRMDialog::KVNewGridRRMDialog(const TGWindow* p,
37  const TGWindow* main, UInt_t w,
38  UInt_t h, Option_t* dialog_type,
39  KVIDGraph* g)
40  : KVNewGridDialog()
41 {
42  //Dialog box for grid derived from KVINDRARRMValidator (with lists of runs, rings and modules).
43  //We use text entry widgets to handle these lists, instead of the "min" and "max" number entry
44  //fields used in the default version
45  if (!strcmp(dialog_type, "Edit") || !strcmp(dialog_type, "Copy")) {
46  //in "Edit" and "Copy" modes a pointer to the selected grid is passed to the ctor.
47  set_selected_grid(g);
48  //if fSelectedGrid is still = 0 after this call, there is something wrong with the pointer 'g' given to the ctor,
49  //and we should exit without setting up the interface
50  if (!fSelectedGrid)
51  return;
52  }
53  //initialise dialog
54  init_interface(p, main, w, h, dialog_type);
55 
56  //for "New grid" dialog, add class list drop-down menu
57  if (!strcmp(dialog_type, "New"))
58  new_grid_class_list();
59 
60  //for "edit" and "copy" dialog boxes, we add controls for setting scale factors, detector numbers, etc.
61  if (!strcmp(dialog_type, "Edit") || !strcmp(dialog_type, "Copy"))
62  layout_parameter_control_widgets();
63 
64  //add buttons for "OK", "Cancel" etc.
65  layout_ok_apply_cancel_buttons(dialog_type);
66 
67  //set up grid editor if in "Edit" mode
68  if (!strcmp(dialog_type, "Edit"))
69  layout_grid_editor();
70 
71  //finish set up and draw dialogue box
72  map_interface(dialog_type);
73 }

80 void KVNewGridRRMDialog::set_selected_grid(KVIDGraph* g)
81 {
82  //Make sure selected grid is of right type i.e. inherits from KVINDRARRMValidator, otherwise return with error message
83  fSelectedGrid = g;
84  if (!g->InheritsFrom("KVINDRARRMValidator")) {
85  cout << "Error in KVNewGridRRMDialog ctor:" << endl;
86  cout <<
87  "This dialogue box can only be used with grids derived from KVINDRARRMValidator"
88  << endl;
89  fSelectedGrid = 0; // will force exit from base class ctor, dialogue box will not be built
90  return;
91  }
92  //get pointer to RRMValidator part of grid (for access to GetRunList(), GetRingList(), GetModuleList() methods)
93  fRRMGrid = dynamic_cast < KVINDRARRMValidator* >(g);
94 }

306 void KVNewGridRRMDialog::ReadAndSetSelectedGridProperties(KVIDGraph* grid)
307 {
308  KVINDRARRMValidator* RRMGrid =
309  dynamic_cast < KVINDRARRMValidator* >(grid);
310  if (_choose_rings()) {
311  RRMGrid->GetRingList().SetList(fRingListString);
312  } else {
313  RRMGrid->GetRingList().Clear();
314  }
315  if (_choose_mods()) {
316  RRMGrid->GetModuleList().SetList(fModListString);
317  } else {
318  RRMGrid->GetModuleList().Clear();
319  }
320  if (_choose_runs()) {
321  RRMGrid->GetRunList().SetList(fRunListString);
322  } else {
323  RRMGrid->GetRunList().Clear();
324  }
325  if (_choose_scales()) {
326  if (_scale_x() != 1)
327  grid->SetXScaleFactor(_scale_x());
328  else
329  grid->SetXScaleFactor();
330  if (_scale_y() != 1)
331  grid->SetYScaleFactor(_scale_y());
332  else
333  grid->SetYScaleFactor();
334  } else {
335  grid->SetXScaleFactor();
336  grid->SetYScaleFactor();
337  }
338 }

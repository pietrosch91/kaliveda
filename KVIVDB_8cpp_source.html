113 void KVIVDB::ReadDeltaPedestal(ifstream& ifile)
114 {
115  // Reading the pedestal correction (DeltaPed). Method called by
116  // KVINDRADB_e503::ReadPedestalCorrection(). DeltaPed is equal to
117  // DeltaNoise for high gain coder data. For anyother coder data,
118  // DetaPed is given by DeltaGene if DeltaGene is less than 20 channels,
119  // otherwise it is given by DeltaNoise. DeltaGene (DeltaNoise) is the
120  // difference between pedestal positions from the generator (noise)
121  // of the current run and of the reference run.
122 
123 
124  KVString sline, signal, parname;
125  KVDBParameterSet* parset = NULL;
126  Int_t numQDC = -1;
127  KVNumberList runs;
128  Double_t Dped = -1., Dgene = -1., Dbruit = -1.;
129 
130  while (ifile.good()) { //reading the file
131  sline.ReadLine(ifile);
132  // Skip comment line
133  if (sline.BeginsWith("#")) continue;
134  // End character
135  if (sline.BeginsWith("!")) return;
136 
137  sline.ReplaceAll("\t", " ");
138  // QDC number
139  if (sline.BeginsWith("QDC=")) {
140  sline.Remove(0, 4);
141  numQDC = sline.Atof();
142  continue;
143  }
144  // DeltaPed for each signal
145  Ssiz_t idx = sline.Index(":");
146  runs.SetList(TString(sline).Remove(idx));
147  sline.Remove(0, idx + 1);
148  sline.Begin(" ");
149  while (!sline.End()) {
150  signal = sline.Next(kTRUE);
151  Dgene = sline.Next(kTRUE).Atof();
152  Dbruit = sline.Next(kTRUE).Atof();
153 
154  // Marie-France's selection
155  Dped = ((Dgene > 20) || !strcmp(signal.Data(), "GG") ? Dbruit : Dgene);
156  if (Dped > 20) Warning("ReadDeltaPedestal", "DeltaPed>20 for runs %s, signal %s", runs.GetList(), signal.Data());
157 
158  parname.Form("QDC%.2d", numQDC);
159  parset = new KVDBParameterSet(parname.Data(), signal.Data(), 1);
160  parset->SetParameter(Dped);
161  fDeltaPed->AddRecord(parset);
162  LinkRecordToRunRange(parset, runs);
163  }
164  }
165 }

176 {

194  sline.ReadLine(fin);
195  // Skip comment line
196  if (sline.BeginsWith("#")) continue;

351 Bool_t KVIVDB::ReadVamosCalibFile(ifstream& ifile)
352 {
353  // Reads the calibration file loaded in 'infile'.
354  // Retruns kTRUE if the mimimum of information is
355  // given inside the file (Run list, type of calibration,
356  // function of calibration, acquisition parameter
357  // associated to the calibration parameters... ).
358  // The format is the following:
359  //
360  // RUNS: 1-30 34 50-200
361  // TYPE: channel->Volt (or channel->ns, Volt->MeV)
362  // FUNCTION: pol1 (or formula format of TF1)
363  //
364  // these tree infromations are the minimum to give in the file.
365  //
366  // If the FUNCTION is 'no'( 'NO' or empty ) then it is assume that
367  // the calibrator is defined by a class inheriting from KVCalibrator
368  // different from KVFunctionCal.
369  //
370  // The following depends of the calibration type:
371  // - If the type contains 'channel' then the calibration
372  // parameters have to be associated to an acquisition
373  // parameter. In this case the name of the acquisition
374  // parameter has to be given first followed by ':' and the parameters.
375  // - else the calibration parameters have to be
376  // associated to the corresponding detector.
377  //
378  // The parameters can be add on a new line if the previous line end
379  // with the character '\'
380  //
381  // The comment line begins with '#'.
382  //
383  // You can change when you want the run list, or the
384  // function. The example below illustrates these
385  // possibilities:
386  //
387  // RUNS: 1-30 34 50-200
388  // TYPE: channel->Volt
389  // FUNCTION: pol1
390  // SED1_X_001: par0 par1
391  // SED1_X_002: par0 par1
392  // #
393  // FUNCTION: [0]+[1]*x+[2]/x
394  // SED1_X_003: par0 par1 par2
395  // ...
396  // ...
397  // RUNS: 201 203 207-266
398  // SED1_X_001: par0 par1 par2
399  // SED1_X_002: par0 par1 par3
400  // ...
401  // ...
402  // RUNS: 1-266
403  // FUNCTION: pol6
404  // TYPE: channel->ns
405  // SED1_HF: par0 par1 par2 par3 par4 par5 par6
406  //
407  // TYPE: position->cm
408  // FUNCTION: no
409  // SED1: par1 par2 par3 par4 par5 \.
410  // par6 par7 par8 par8 par10
411  //
412  //
413  // It is possible to set a parameter for a detector (or for VAMOS) if
414  // a Setter method exists for this parameter in the classe describing
415  // the detector (or VAMOS). In this case the name of the parameter has
416  // to be given in TYPE and FUNCTION has to be set to 'no' or null. For
417  // example to set the reference Brho 0.2 T.m to VAMOS for run 1 to 100
418  // write:
419  //
420  // RUNS: 1-100
421  // TYPE: BrhoRef
422  // FUNCTION: no
423  // VAMOS: 0.2
424 
425 
426  KVString sline, name;
427  Int_t idx;
428  Int_t OKbit = 0;
429  KVDBParameterSet* parset = NULL;
430  KVNumberList runs;
431  TObjArray* tok = NULL;
432  KVString infoNames[3] = {"RUNS", "TYPE", "FUNCTION"};
433  KVString infos[3];
434 
435  while (ifile.good()) { //reading the file
436  sline.ReadLine(ifile);
437 
438  if (sline.IsNull()) continue;
439  // Skip comment line
440  if (sline.BeginsWith("#")) continue;
441 
442  sline.ReplaceAll("\t", " ");
443  // Mandatory character ':'
444  if ((idx = sline.Index(":")) < 0) continue;
445 
446  name = sline;
447  name.Remove(idx);
448  name.Remove(KVString::kBoth, ' ');
449  sline.Remove(0, idx + 1);
450  sline.Remove(KVString::kBoth, ' ');
451 
452  Int_t i;
453  for (i = 0; i < 3; i++)
454  if (name == infoNames[i]) break;
455 
456  // if line contains an info ( RUNS, TYPE or FUNCTION )
457  if (i < 3) {
458  infos[i] = sline;
459  OKbit = OKbit | (1 << i);
460  if (i == 0) runs.SetList(sline);
461  } else if (OKbit != 7) {
462  // OKbit different from 7 means RUNS, TYPE or FUNCTION are missing
463  for (Int_t j = 0; j < 3; j++) {
464  if (!(OKbit & (1 << j)))
465  Error("ReadVamosCalibFile", "Missing %s for the calibration of %s", infoNames[j].Data(), name.Data());
466 
467  }
468  return kFALSE;
469  } else {
470 
471  if (sline.EndsWith("\\")) {
472  sline.Remove(KVString::kTrailing, '\\');
473  sline.Remove(KVString::kTrailing, ' ');
474  KVString extra;
475  Bool_t next = kTRUE;
476  while (next) {
477  extra.ReadLine(ifile);
478  if (extra.BeginsWith("#")) continue;
479  if (extra.EndsWith("\\")) extra.Remove(KVString::kTrailing, '\\');
480  else next = kFALSE;
481  extra.Remove(KVString::kBoth, ' ');
482  sline.Append(" ");
483  sline.Append(extra);
484  }
485  }
486 
487  tok = sline.Tokenize(" ");
488 
489  if (infos[2].CompareTo("no") && infos[2].CompareTo("NO") && !(infos[2].IsNull())) {
490  // for calibrator with formula expression
491  sline.Form("f%s", name.Data());
492  TF1 f(sline.Data(), infos[2].Data());
493  if (tok->GetEntries() != f.GetNpar()) {
494  cout << "info 2: " << infos[2].Data() << endl;
495  Error("ReadVamosCalibFile", "Different numbers of parameters:");
496  cout << "- the function " << f.GetExpFormula().Data() << " ( " << f.GetNpar() << " parameters )" << endl;
497  cout << "- the calibrator of " << name.Data() << " ( " << tok->GetEntries() << " parameters )" << endl;
498  delete tok;
499  return kFALSE;
500  }
501  TString pname(infos[1]);
502  pname.Append(" f(x)=");
503  pname.Append(f.GetExpFormula());
504  parset = new KVDBParameterSet(name.Data(), pname.Data(), f.GetNpar() + 1);
505  parset->SetParamName(0, "Npar");
506  parset->SetParameter(0, f.GetNpar());
507  for (Int_t j = 0; j < f.GetNpar(); j++) {
508 // parset->SetParamName( j+1, f.GetParName(j));
509 // parset->SetParameter( j+1, ((TObjString *)tok->At(j))->GetString().Atof() );
510  parset->SetParamName(j + 1, ((TObjString*)tok->At(j))->GetString().Data());
511  }
512  } else {
513  // for calibrator without formula expression
514  parset = new KVDBParameterSet(name.Data(), infos[1].Data(), tok->GetEntries() + 1);
515  parset->SetParamName(0, "Npar");
516  parset->SetParameter(0, tok->GetEntries());
517  for (Int_t j = 0; j < tok->GetEntries() ; j++) {
518 // parset->SetParamName( j+1, Form("par%d",j) );
519 // parset->SetParameter( j+1, ((TObjString *)tok->At(j))->GetString().Atof() );
520  parset->SetParamName(j + 1, ((TObjString*)tok->At(j))->GetString().Data());
521  }
522  }
523  delete tok;
524  fVAMOSCalConf->AddRecord(parset);
525  LinkRecordToRunRange(parset, runs);
526 
527 // cout<<parset->GetName()<<" "<<parset->GetTitle()<<" "<<runs.GetList()<<" "<<parset->GetParamName(0);
528 // for( Int_t j=0; j< parset->GetParamNumber(); j++ )
529 // cout<<" "<<parset->GetParameter(j);
530 // cout<<endl;
531  }
532  }
533 
534  return kTRUE;
535 }

548 {

558  Error("ReadVamosCalibrations", "Could not open file %s", GetCalibFileName("VamosCalibFile"));
559  return;
560  }
561 
562  Info("ReadVamosCalibrations", "Reading calibration parameters of VAMOS...");
563 
564  TString sline;
565  while (!inf.eof()) {
566  sline.ReadLine(inf);
567  if (sline.IsNull()) continue;
568 
569  // Skip comment line
570  if (sline.BeginsWith("#")) continue;
571 
572  ifstream inf2;
573  if (!gDataSet->OpenDataSetFile(sline.Data(), inf2)) {
574  Error("ReadVamosCalibrations", "Could not open file %s", sline.Data());

577  }
578 
579  Info("ReadVamosCalibrations", "in file %s", sline.Data());
580  if (!ReadVamosCalibFile(inf2)) Error("ReadVamosCalibrations", "Bad structure inside the file %s", sline.Data());
581 
582  inf2.close();
583  }

607 void KVIVDB::ReadAndCorrectVamosScalers(const Char_t* runlist, Option_t* opt)
608 {
609  // This method read the VAMOS scaler buffers in the raw data files for the given
610  // runlist. The true value of each scaler is calculated. This calculation is
611  // based on program "scaler.cpp" developped by Abdou Chbihi (GANIL).
612  // Then the new value of scalers are stored in this database.
613  // Moreover the Vamos Dead-Time (DT in percent) is calculated:
614  // DT = 1 - FTA/ORSI
615  //
616  // If the option "save" is set, then the new database is saved in the file Runlist.csv
617  // (see method WriteRunListFile ).
618  //
619  // HOW USE THIS METHODE?
620  //
621  // gDataRepositoryManager->GetRepository("ccali")->cd();
622  // KVDataSet *ds = gDataRepository->GetDataSetManager()->GetDataSet("INDRA_e494s");
623  // ds->cd();
624  // ((KVIVDB *)gIndraDB)->ReadAndCorrectVamosScalers("527-532","save");
625 
626  KVDataSet* ds = gDataSet;
627 
628  gDataSet = NULL; // faster computing if gDataSet is null (no KVMultiDetArray built and loaded)
629 
630  // By default, scaler buffers are ignored during reading of raw data file.
631  // Change the appropriate environment variable to be able to read the scaler buffers.
632  TString sbm = gEnv->GetValue("KVGANILDataReader.ScalerBuffersManagement", "kReportScaler");
633  gEnv->SetValue("KVGANILDataReader.ScalerBuffersManagement", "kReportScaler");
634 
635  // list of runs to be processed
636  KVNumberList rl(runlist);
637  rl.Begin();
638  Int_t r_counter = 0; // counter of runs
639  while (!rl.End()) { // loop over the runs
640 
641  Int_t run = rl.Next();
642  r_counter ++;
643 
644  // Open raw data file for current run
645  TString fname = ds->GetFullPathToRunfile("raw", run);
646  if (fname.IsNull()) {
647  Error("ReadAndCorrectVamosScalers", "Error in <ReadScalersWithKaliVeda>: no raw data file found for run %d (dataset: %s, repository: %s)", run, ds->GetLabel(), ds->GetRepository()->GetName());
648  continue;
649  }
650  KVGANILDataReader runfile(fname.Data());
651 
652 
653  UInt_t n_event = 0; // number of entries in the current data file
654  Int_t n_buffers = 0; // number of scaler buffer in the current data file
655  Int_t n_buffers_ok = 0; // number of scaler buffer in the current data file with status == 1 (ok)
656 
657  UInt_t first_scaler_count[40];// first buffer of scalers
658  UInt_t scaler_count[40]; // last buffer of scalers
659  Int_t first_event_count = 0;
660  Int_t event_count = 0;
661 
662  while (runfile.GetNextEvent()) { // loop over the events
663  if (runfile.HasScalerBuffer()) {
664  n_buffers++;
665  Int_t n_scalers = runfile.GetNumberOfScalers();
666  if (n_scalers != 40) {
667  // VAMOS has 40 scalers
668  Error("ReadAndCorrectVamosScalers", "number of scalers != 40 for run %d at event %d", run, n_event);
669  continue;
670  }
671 
672  Int_t status = runfile.GetScalerStatus(0);
673 
674  if (status == 1) { // acquisition on
675  n_buffers_ok++;
676  for (Int_t i = 0; i < n_scalers; i++) { // loop over VAMOS scalers
677  // store scalers of the first buffer
678  if (n_buffers_ok == 1) first_scaler_count[i] = runfile.GetScalerCount(i);
679  // store scalers of the last buffer
680  else scaler_count[i] = runfile.GetScalerCount(i);
681  } // loop over VAMOS scalers
682 
683  // store number of event of the first buffer
684  if (n_buffers_ok == 1) first_event_count = n_event;
685  // store number of event of the last buffer
686  else event_count = n_event;
687 
688  } // acquistion on
689  } // has scaler buffer
690  else n_event++;
691  } // loop over the events
692 
693  UInt_t total_event_count = n_event;
694  //UInt_t total_event_count = runfile.GetEventCount();
695 
696  //--------------------------------------------------
697  // Correction of scalers: last_buffer - first_buffer
698  // and load scalers in the database
699  //--------------------------------------------------
700 
701  KVINDRADBRun* db_run = GetRun(run);
702 
703  if (n_buffers_ok > 0) {
704 
705  if (n_buffers_ok > 1) {
706  // the new scaler is the difference between the last and the first buffer
707  // having status equal 1.
708  for (Int_t i = 0; i < 40; i++) scaler_count[i] -= first_scaler_count[i];
709  event_count -= first_event_count;
710  }
711  if ((n_buffers_ok == 1) || (scaler_count[27] <= 0 /*FTA==0*/)) {
712  // if there is only 1 buffer then we assume that once it will be
713  // normalized to the total_event_count then the new value will be
714  // a good approximation
715  for (Int_t i = 0; i < 40; i++) scaler_count[i] = first_scaler_count[i];
716  event_count = scaler_count[27];
717  }
718 
719  // correction factor to normalize the scalers to the total_event_count
720  Double_t corrFactor = (event_count ? (1.*total_event_count) / event_count : 1.);
721  //Double_t corrFactor = (scaler_count[27] ? (1.*total_event_count)/scaler_count[27] : 1. ); // total_event/FTA
722 
723  for (Int_t i = 0; i < 40; i++) {
724  scaler_count[i] *= corrFactor;
725  // set the scaler in the database
726  db_run->SetScaler(GetVamosScalerName(i), scaler_count[i]);
727  }
728 
729  // set VAMOS dead time in the database DT = 1-FTA/ORSI
730  Double_t DT = 100.*(1. - (1.*scaler_count[27]) / scaler_count[37]);
731  db_run->Set("DT", DT);
732 
733 // db_run->SetScaler("totalEvent",total_event_count);
734 // db_run->SetScaler("EventCount",event_count);
735 // db_run->SetScaler("NbuffersOK",n_buffers_ok);
736  }
737 
738  db_run->Print();
739 
740  cout << endl;
741  Info("ReadAndCorrectVamosScalers", "processing ------> %.0f %s \n", (100.*r_counter) / rl.GetNValues(), "%");
742 
743  }
744 
745  // go back the the initial configuration
746  ds->cd();
747  gEnv->SetValue("KVGANILDataReader.ScalerBuffersManagement", sbm.Data());
748 
749  // write the run list file if the option is set
750  TString option = opt;
751  option.ToLower();
752  if (option == "save") WriteRunListFile();
753 }

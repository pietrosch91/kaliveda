84 void KVExpSetUp::Build(Int_t run)
85 {
86  // Build the combined arrays
87 
88  // default ROOT geometry for all arrays
89  gEnv->SetValue("KVMultiDetArray.ROOTGeometry", "yes");
90 
91  CreateGeoManager();
92 
93  Info("Build", "navigator=%p", GetNavigator());
94 
95  KVRangeTableGeoNavigator* gnl = new KVRangeTableGeoNavigator(gGeoManager, KVMaterial::GetRangeTable());
96  SetNavigator(gnl);
97 
98  KVMultiDetArray* tmp(nullptr);
99  Int_t group_offset = 0;// for renumbering groups
100  lmultidetarrayclasses =
101  gDataSetManager->GetDataSet(fDataSet)->GetDataSetEnv("DataSet.ExpSetUp.ClassList", IsA()->GetName());
102  lmultidetarrayclasses.Begin(" ");
103  while (!lmultidetarrayclasses.End()) {
104  KVString sname = lmultidetarrayclasses.Next();
105  Info("Build", "Build %s %s\n", gDataSet->GetName(), sname.Data());
106  gMultiDetArray = nullptr; //otherwise MakeMultiDetector will delete any previously built array
107  tmp = MakeMultiDetector(gDataSet->GetName(), run, sname.Data());
108  if (tmp) {
109 
110  unique_ptr<KVSeqCollection> groups(tmp->GetStructureTypeList("GROUP"));
111  if (group_offset) {
112  // renumber all groups to keep unique names/numbers
113  TIter next(groups.get());
114  KVGroup* group;
115  while ((group = (KVGroup*)next())) {
116  Int_t group_number = group->GetNumber();
117  group->SetNumber(group_number + group_offset);
118  }
119  }
120  else
121  group_offset += groups->GetEntries();
122 
123  fMDAList.Add(tmp);
124  fDetectors.AddAll((KVUniqueNameList*)tmp->GetDetectors());
125  fStructures.AddAll((KVUniqueNameList*)tmp->GetStructures());
126  fIDTelescopes->AddAll(tmp->GetListOfIDTelescopes());
127 
128  // retrieve correspondance list node path<->detector
129  gnl->AbsorbDetectorPaths(tmp->GetNavigator());
130 
131  }
132  else {
133  Error("Build", "NULL pointer returned by MakeMultiDetector");
134  }
135  }
136 
137  gGeoManager->DefaultColors();
138  gGeoManager->CloseGeometry();
139 
140  SetGeometry(gGeoManager);
141 
142  gMultiDetArray = this;
143  fCurrentRun = (run > 0 ? run : 0);
144  SetBit(kIsBuilt);
145 }

165 void KVExpSetUp::FillDetectorList(KVReconstructedNucleus* rnuc, KVHashList* DetList, const KVString& DetNames)
166 {

185 void KVExpSetUp::AcceptParticleForAnalysis(KVReconstructedNucleus* NUC) const
186 {
187  // Overrides KVMultiDetArray method
188  // We use the "ARRAY" parameter of the reconstructed particle (if set)
189  // to know which array of the setup detected it.
190 
191  if (NUC->GetParameters()->HasStringParameter("ARRAY")) {
192  KVMultiDetArray* whichArray = GetArray(NUC->GetParameters()->GetStringValue("ARRAY"));
193  if (whichArray) {
194  whichArray->AcceptParticleForAnalysis(NUC);
195  }
196  else
197  Warning("AcceptParticleForAnalysis", "ReconstructedNucleus has ARRAY=%s but no KVMultiDetArray with this name in set-up",
198  NUC->GetParameters()->GetStringValue("ARRAY"));
199  }
200  else {
201  Warning("AcceptParticleForAnalysis", "ReconstructedNucleus has no ARRAY parameter: cannot determine correct KVMultiDetArray");
202  }
203 }

214 Bool_t KVExpSetUp::handle_raw_data_event_mfmframe(const MFMCommonFrame& mfmframe)
215 {
216  // Handle single (not merged) MFM frames of raw data. It is assumed
217  // that each frame type corresponds to a different detector array.
218  // Therefore as soon as one of them treats the data in the frame,
219  // we return kTRUE.
220 
221  TIter next_array(&fMDAList);
222  KVMultiDetArray* mda;
223  while ((mda = (KVMultiDetArray*)next_array())) {
224  if (mda->handle_raw_data_event_mfmframe(mfmframe)) return kTRUE;
225  }
226  return kFALSE;
227 }

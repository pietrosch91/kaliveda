57 KVedaLossInverseRangeFunction::KVedaLossInverseRangeFunction(TF1* range_func,
58  Int_t A,
59  Double_t riso,
60  Int_t ninter)
61 {
62  // Initialise from VEDALOSS range function object
63  // A is mass of ion used in call to KVedaLossMaterial::GetRangeFunction
64  // riso is value returned by KVedaLossMaterial::get_riso corresponding to
65  // previous call to KVedaLossMaterial::GetRangeFunction
66  // Default number of points in graph used for interpolation ninter=50
67  // We make a graph from 0.01 MeV/nucleon to EMAX
68 
69  Double_t emin, emax, estep;
70  range_func->GetRange(emin, emax);
71 
72  TGraph e_vs_range;
73  emin = 0.01;
74  emax /= A;
75 
76  Double_t logriso = LOG(riso);
77 
78  estep = (LOG(emax) - LOG(emin)) / (ninter - 1.);
79 
80  for (int i = 0; i < ninter; ++i) {
81  double loge = LOG(emin) + i * estep;
82  // we divide the calculated range by the current value of riso,
83  // thereby effectively ensuring that we always calculate range
84  // values corresponding to riso=1
85  double R = LOG(range_func->Eval(EXP(loge) * A)) - logriso;
86  e_vs_range.SetPoint(i, R, loge);
87  }
88 
89  fInterpol = new TSpline3(range_func->GetTitle(), &e_vs_range);
90 }

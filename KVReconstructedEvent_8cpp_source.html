121 void KVReconstructedEvent::Streamer(TBuffer& R__b)
122 {
123  //Stream an object of class KVReconstructedEvent.
124  //We set the particles' angles depending on whether mean or random angles
125  //are wanted (fMeanAngles = kTRUE or kFALSE)
126 
127  if (R__b.IsReading()) {
128  R__b.ReadClassBuffer(KVReconstructedEvent::Class(), this);
129  // if the multidetector object exists, update some informations
130  // concerning the detectors etc. hit by this particle
131  if (gMultiDetArray) {
132  //set angles
133  KVReconstructedNucleus* par;
134  while ((par = GetNextParticle())) {
135  if (HasMeanAngles())
136  par->GetAnglesFromStoppingDetector("mean");
137  else
138  par->GetAnglesFromStoppingDetector("random");
139  //reconstruct fAnalStatus information for KVReconstructedNucleus
140  if (par->GetStatus() == 99) //AnalStatus has not been set for particles in group
141  if (par->GetGroup())
142  KVReconstructedNucleus::AnalyseParticlesInGroup(par->GetGroup());
143  // apply identification & calibration code selection
144  gMultiDetArray->AcceptParticleForAnalysis(par);
145  }
146  }
147  } else {
148  R__b.WriteClassBuffer(KVReconstructedEvent::Class(), this);
149  }
150 }

167 Bool_t KVReconstructedEvent::AnalyseDetectors(TList* kvtl)
168 {
169  // Loop over detectors in list
170  // if any detector has fired, start construction of new detected particle
171  // More precisely: If detector has fired,
172  // making sure fired detector hasn't already been used to reconstruct
173  // a particle, then we create and fill a new detected particle.
174  // In order to avoid creating spurious particles when reading data,
175  // by default we ask that ALL coder values be non-zero here i.e. data and time-marker.
176  // This can be changed by calling SetPartSeedCond("any"): in this case,
177  // particles will be reconstructed starting from detectors with at least 1 fired parameter.
178 
179  KVDetector* d;
180  TIter next(kvtl);
181  while ((d = (KVDetector*)next())) {
182  /*
183  If detector has fired,
184  making sure fired detector hasn't already been used to reconstruct
185  a particle, then we create and fill a new detected particle.
186  */
187  if ((d->Fired(fPartSeedCond.Data()) && !d->IsAnalysed())) {
188 
189  KVReconstructedNucleus* kvdp = AddParticle();
190  //add all active detector layers in front of this one
191  //to the detected particle's list
192  kvdp->Reconstruct(d);
193 
194  //set detector state so it will not be used again
195  d->SetAnalysed(kTRUE);
196  }
197  }
198 
199  return kTRUE;
200 }

241 void KVReconstructedEvent::IdentifyEvent()
242 {
243  //All particles which have not been previously identified (IsIdentified=kFALSE), and which
244  //may be identified independently of all other particles in their group according to the 1st
245  //order coherency analysis (KVReconstructedNucleus::GetStatus=0), will be identified.
246  //Particles stopping in first member of a telescope (KVReconstructedNucleus::GetStatus=3) will
247  //have their Z estimated from the energy loss in the detector (if calibrated).
248 
249  KVReconstructedNucleus* d;
250  while ((d = GetNextParticle())) {
251  if (!d->IsIdentified()) {
252  if (d->GetStatus() == KVReconstructedNucleus::kStatusOK) {
253  // identifiable particles
254  d->Identify();
255  } else if (d->GetStatus() == KVReconstructedNucleus::kStatusStopFirstStage) {
256  // particles stopped in first member of a telescope
257  // estimation of Z (minimum) from energy loss (if detector is calibrated)
258  UInt_t zmin = d->GetStoppingDetector()->FindZmin(-1., d->GetMassFormula());
259  if (zmin) {
260  d->SetZ(zmin);
261  d->SetIsIdentified();
262  // "Identifying" telescope is taken from list of ID telescopes
263  // to which stopping detector belongs
264  d->SetIdentifyingTelescope((KVIDTelescope*)d->GetStoppingDetector()->GetIDTelescopes()->At(0));
265  }
266  }
267  }
268  }
269 }

284 void KVReconstructedEvent::CalibrateEvent()
285 {
286  // Calculate and set energies of all identified particles in event.
287  //
288  // This will call the KVReconstructedNucleus::Calibrate() method of each
289  // uncalibrated particle (those for which KVReconstructedNucleus::IsCalibrated()
290  // returns kFALSE).
291  //
292  // In order to make sure that target energy loss corrections are correctly
293  // calculated, we first set the state of the target in the current multidetector
294 
295  KVTarget* t = gMultiDetArray->GetTarget();
296  if (t) {
297  t->SetIncoming(kFALSE);
298  t->SetOutgoing(kTRUE);
299  }
300 
301  KVReconstructedNucleus* d;
302 
303  while ((d = GetNextParticle())) {
304 
305  if (d->IsIdentified() && !d->IsCalibrated()) {
306  d->Calibrate();
307  }
308 
309  }

58 KVPhoswich::KVPhoswich(Float_t): KVINDRADetector("NE102", 0.05)
59 {
60  //Build phoswich telescope from two layers of plastic - (NE102) and (NE115).
61  // 'thick' is a dummy argument, it is not used
62  // thickness of NE102 = 0.05cm for standard INDRA phoswich
63  // thickness of NE115 = 25cm for standard INDRA phoswich
64  //Both layers are "active" - this requires redefinition of GetEnergy etc. methods
65  //Type of detector is "PHOS"
66 
67  KVMaterial* ne115 = new KVMaterial("NE102", 25);
68  AddAbsorber(ne115);
69 
70  fSegment = 2;
71  SetType("PHOS");

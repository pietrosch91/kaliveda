15 // This class is for reading primary events before secondary decay,
16 // i.e. output files with names like: elie_58_197_32_1000_primary.output
17 // The events will be stored in a ROOT file with a name like:
18 // ELIE_[PROJ]_[TARG]_[EBEAM]AMeV_PRIM.root
19 // This file will contain a TTree called "ELIE" with a branch "Simulated_evts"
20 // All the informations in the header of the ELIE output file are stored in
21 // a KVNameValueList in the TTree:
22 // ELIE->GetUserInfo()->FindObject("ELIE Parameters")->ls()
23 //KVNameValueList::ELIE Parameters : (0x55b2067172f0)
24 // <ievt=50>
25 // <projectile=Ni>
26 // <target=Au>
27 // <a_proj=58>
28 // <z_proj=28>
29 // <a_targ=197>
30 // <z_targ=79>
31 // <ebeam=32>
32 // <ilevel=0>
33 // <medium_effect=0>
34 // <g_al=1>
35 // <fraction_ex=0.5>
36 // <tempe_limite=5.5>
37 // <decay_in_flight=1>
38 // <lab_frame=0>
39 // <bminimum=0>
40 // <bmaximum=1>
41 // <choix_ex=0>
42 // <fermi=1>
43 // <version=1000>
44 // <periph_trick=0>
45 // <b_limite=0.75>
46 // <freeze_out=3>
47 // <proba_free_nucleon=0.001>
48 // <tole_pourcent=0.01>
49 // <discrete=1>
50 // <minimisation=0>
51 // <geometry=0>
52 // <free_nucleon=0>
53 // <lambda_ex=1.3>
54 //
55 // If parameter lab_frame=1, we transform the events to the CM frame.
56 // In this way, for further analysis/filtering, ELIE events are always in the CM frame.

223 Bool_t KVSimReader_ELIE::ReadHeader()
224 {
225  // File header contains info on simulation
226  //
227  // 54 129 50 119 32
228  // 20000
229  // ievt=50
230  // projectile=Ni
231  // target=Au
232  // a_proj=58
233  // z_proj=28
234  // a_targ=197
235  // z_targ=79
236  // [...]
237  // geometry=0
238  // free_nucleon=0
239  // lambda_ex=1.3
240  //
241  // The simulation parameters are stored in a KVNameValueList
242 
243  Int_t res = ReadLineAndCheck(5, " ");
244  switch (res) {
245  case 0:
246  Info("ReadHeader", "Can't read system");
247  return kFALSE;
248  case 1:
249  AddInfo("Aproj", GetReadPar(1));
250  AddInfo("Zproj", GetReadPar(0));
251  AddInfo("Atarg", GetReadPar(3));
252  AddInfo("Ztarg", GetReadPar(2));
253  AddInfo("Ebeam", GetReadPar(4));
254  proj.SetZAandE(GetIntReadPar(0), GetIntReadPar(1), GetIntReadPar(1)*GetDoubleReadPar(4));
255  targ.SetZandA(GetIntReadPar(2), GetIntReadPar(3));
256  ebeam = GetDoubleReadPar(4);
257  break;
258  default:
259  return kFALSE;
260  }
261 
262  res = ReadLineAndCheck(1, " ");
263  switch (res) {
264  case 0:
265  Info("ReadHeader", "Can't read events");
266  return kFALSE;
267  case 1:
268  AddInfo("Nevents", GetReadPar(0));
269  break;
270 
271  default:
272  return kFALSE;
273  }
274  elie_params->Clear();
275  elie_params->SetName("ELIE Parameters");
276  res = ReadLineAndCheck(2, "=");
277  while (res < 2) {
278  if (res > 0) {
279  if (GetReadPar(1).IsDigit()) elie_params->SetValue(GetReadPar(0), GetIntReadPar(1));
280  else if (GetReadPar(1).IsFloat()) elie_params->SetValue(GetReadPar(0), GetDoubleReadPar(1));
281  else elie_params->SetValue(GetReadPar(0), GetReadPar(1));
282  }
283  res = ReadLineAndCheck(2, "=");
284  }
285  if (elie_params->GetEntries()) AddObject(elie_params);
286 
287  return kTRUE;
288 }

301 Bool_t KVSimReader_ELIE::ReadEvent()
302 {
303  // Event structure: (primary events)
304  // nevt multiplicite b reduit
305  // 0 2 0.998654688551
306  // particules dans l'evt
307  // indice_particule charge_particule masse_particule teta_particule(degres) phi_particule(degres) indice_particule energie d'excitation totale (MeV)
308  // 0 27 57 3.27897003986 230.52425244 1109.37002505 0 0.00499304975261
309  // 1 80 198 176.726338776 129.481056376 319.364098122 1 0.017344278088
310 
311  evt->Clear();
312 
313  // first line of first event will have been read already by ReadHeader,
314  // therefore we do not read a new line in this case (when nevt=0)
315  Int_t res = (nevt ? ReadLineAndCheck(3, " ") : ReuseLineAndCheck(3, " "));
316  Int_t mult = 0;
317  switch (res) {
318  case 0:
319  return kFALSE;
320  case 1:
321  evt->SetNumber(GetIntReadPar(0));
322  mult = GetIntReadPar(1);
323  evt->GetParameters()->SetValue("bred", GetDoubleReadPar(2));
324  evt->GetParameters()->SetValue("mult", mult);
325  break;
326  default:
327  return kFALSE;
328  }
329  for (Int_t mm = 0; mm < mult; mm++) {
330  nuc = (KVSimNucleus*)evt->AddParticle();
331  if (!ReadNucleus()) return kFALSE;
332  }
333  nevt++;
334 
335  // if in lab, transform to CM frame
336  if (elie_params->GetIntValue("lab_frame") > 0) transform_to_cm();
337  else evt->SetFrameName("CM");
338 
339  return kTRUE;
340 }

353 Bool_t KVSimReader_ELIE::ReadNucleus()
354 {
355  // Event structure: (primary events)
356  // nevt multiplicite b reduit
357  // 0 2 0.998654688551
358  // particules dans l'evt
359  // indice_particule charge_particule masse_particule teta_particule(degres) phi_particule(degres) indice_particule energie d'excitation totale (MeV)
360  // 0 27 57 3.27897003986 230.52425244 1109.37002505 0 0.00499304975261
361  // 1 80 198 176.726338776 129.481056376 319.364098122 1 0.017344278088
362 
363  Int_t res = ReadLineAndCheck(8, " ");
364  switch (res) {
365  case 0:
366  Info("ReadNucleus", "case 0 line est vide");
367  return kFALSE;
368 
369  case 1:
370  nuc->SetZ(GetIntReadPar(1));
371  nuc->SetA(GetIntReadPar(2));
372  nuc->SetExcitEnergy(GetDoubleReadPar(7));
373  nuc->SetEnergy(GetDoubleReadPar(5));
374  nuc->SetTheta(GetDoubleReadPar(3));
375  nuc->SetPhi(GetDoubleReadPar(4));
376  return kTRUE;
377  break;
378 
379  default:
380  return kFALSE;
381  }
382 
383  return kFALSE;
384 }

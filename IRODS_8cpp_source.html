219 void IRODS::ExtractFileInfos(TString& s, DMSFile_t* f) const
220 {
221  // extract info (name, size, modification date) from listing line
222  // such as :
223  // indramgr 0 diskcache2 208207872 2009-10-06.16:56 & run_0002.dat.15-Feb-07.18:00:58
224  // indramgr 1 HPSS2 208207872 2009-10-06.16:56 & run_0002.dat.15-Feb-07.18:00:58
225  // indramgr 0 diskcache2 208224256 2009-10-06.16:56 & run_0003.dat.15-Feb-07.20:21:50
226  // indramgr 1 HPSS2 208224256 2009-10-06.16:56 & run_0003.dat.15-Feb-07.20:21:50
227  //
228  // note that when used outside of CCIN2P3, only 'ils' works, not 'ils -l', so the listing looks like:
229  // run_0002.dat.15-Feb-07.18:00:58
230  // run_0003.dat.15-Feb-07.20:21:50
231 
232 #ifdef CCIN2P3_BUILD
233  TObjArray* fstats = s.Tokenize(" ");
234  f->SetName(((TObjString*)(*fstats)[fstats->GetEntries() - 1])->String().Remove(TString::kBoth, ' ').Data());
235  f->SetSize((UInt_t)((TObjString*)(*fstats)[3])->String().Remove(TString::kBoth, ' ').Atoi());
236  KVDatime mt(((TObjString*)(*fstats)[4])->String().Remove(TString::kBoth, ' ').Data(), KVDatime::kIRODS);
237  f->SetModTime(mt);
238  delete fstats;
239 #else
240  f->SetName(s.Remove(TString::kBoth, ' '));
241 #endif
242 }

106  UInt_t last_run)
107 {
108  //If the KVDBRecord 'rec' (i.e. set of calibration parameters, reaction system, etc.) is
109  //associated to, or valid for, a range of runs, we use this method in order to link the record

162 void KVExpDB::LinkRecordToRunRanges(KVDBRecord* rec, UInt_t rr_number,

164 {
165  //Call LinkRecordToRunRange for a set of run ranges stored in the two-dimensional array
166  //in the following way:
167  // run_ranges[0][0] = first run of first run range
168  // run_ranges[0][1] = last run of first run range
169  // run_ranges[1][0] = first run of second run range etc. etc.
170  //rr_number is the number of run ranges in the array
171 
172  for (UInt_t i = 0; i < rr_number; i++) {
173  LinkRecordToRunRange(rec, run_ranges[i][0], run_ranges[i][1]);
174  }

182 void KVExpDB::LinkListToRunRanges(TList* list, UInt_t rr_number,
183  UInt_t run_ranges[][2])
184 {
185  //Link the records contained in the list to the set of runs (see LinkRecordToRunRanges).
186 
187  if (!list) {
188  Error("LinkListToRunRanges",
189  "NULL pointer passed for parameter TList");
190  return;
191  }
192  if (list->GetSize() == 0) {
193  Error("LinkListToRunRanges(TList*,UInt_t,UInt_t*)",
194  "The list is empty");
195  return;
196  }
197  TIter next(list);
198  KVDBRecord* rec;
199 
200  for (UInt_t ru_ra = 0; ru_ra < rr_number; ru_ra++) {
201  UInt_t first_run = run_ranges[ru_ra][0];
202  UInt_t last_run = run_ranges[ru_ra][1];
203  for (UInt_t i = first_run; i <= last_run; i++) {
204  KVDBRecord* run = GetDBRun(i);
205  while ((rec = (KVDBRecord*) next())) {
206  if (run)
207  rec->AddLink("Runs", run);
208  }
209  next.Reset();
210  }
211  }
212 }

267 void KVExpDB::ReadSystemList()
268 {
269  //Reads list of systems with associated run ranges, creates KVDBSystem
270  //records for these systems, and links them to the appropriate KVDBRun
271  //records using LinkListToRunRanges.
272  //
273  //There are 2 formats for the description of systems:
274  //
275  //+129Xe + natSn 25 MeV/A '+' indicates beginning of system description
276  //129 54 119 50 0.1 25.0 A,Z of projectile and target, target thickness (mg/cm2), beam energy (MeV/A)
277  //Run Range : 614 636 runs associated with system
278  //Run Range : 638 647 runs associated with system
279  //
280  //This is sufficient in the simple case where the experiment has a single
281  //layer target oriented perpendicular to the beam. However, for more
282  //complicated targets we can specify as follows :
283  //
284  //+155Gd + 238U 36 MeV/A
285  //155 64 238 92 0.1 36.0
286  //Target : 3 0.0 target with 3 layers, angle 0 degrees
287  //C 0.02 1st layer : carbon, 20 g/cm2
288  //238U 0.1 2nd layer : uranium-238, 100 g/cm2
289  //C 0.023 3rd layer : carbon, 23 g/cm2
290  //Run Range : 770 804
291  //
292  //Lines beginning '#' are comments.
293 
294 
295  std::ifstream fin;
296  if (OpenCalibFile("Systems", fin)) {
297  Info("ReadSystemList()", "Reading Systems parameters ...");
298 
299  TString line;
300 
301  char next_char = fin.peek();
302  while (next_char != '+' && fin.good()) {
303  line.ReadLine(fin, kFALSE);
304  next_char = fin.peek();
305  }
306 
307  while (fin.good() && !fin.eof() && next_char == '+') {
308  KVDBSystem* sys = new KVDBSystem("NEW SYSTEM");
309  AddSystem(sys);
310  sys->Load(fin);
311  next_char = fin.peek();
312  }
313  fin.close();
314  } else {
315  Error("ReadSystemList()", "Could not open file %s",
316  GetCalibFileName("Systems"));
317  }
318  // if any runs are not associated with any system
319  // we create an 'unknown' system and associate it to all runs
320  KVDBSystem* sys = 0;
321  TIter nextRun(GetRuns());
322  KVDBRun* run;
323  while ((run = (KVDBRun*)nextRun())) {
324  if (!run->GetSystem()) {
325  if (!sys) {
326  sys = new KVDBSystem("[unknown]");
327  AddSystem(sys);
328  }
329  sys->AddRun(run);
330  }
331  }
332 
333  // rehash the record table now that all names are set
334  fSystems->Rehash();
335 }

399 void KVExpDB::WriteRunListFile() const
400 {
401  //Write a file containing a line describing each run in the database.
402  //The delimiter symbol used in each line is '|' by default.
403  //The first line of the file will be a header description, given by calling
404  //KVDBRun::WriteRunListHeader() for the first run in the database.
405  //Then we call KVDBRun::WriteRunListLine() for each run.
406  //These are virtual methods redefined by child classes of KVDBRun.
407 
408  ofstream rlistf;
409  KVBase::SearchAndOpenKVFile(GetDBEnv("Runlist"), rlistf, fDataSet.Data());
410  TDatime now;
411  rlistf << "# " << GetDBEnv("Runlist") << " file written by "
412  << ClassName() << "::WriteRunListFile on " << now.AsString() << endl;
413  cout << GetDBEnv("Runlist") << " file written by "
414  << ClassName() << "::WriteRunListFile on " << now.AsString() << endl;
415  if (GetRuns() && GetRuns()->GetEntries() > 0) {
416  TIter next_run(GetRuns());
417  //write header in file
418  ((KVDBRun*) GetRuns()->At(0))->WriteRunListHeader(rlistf, GetDBEnv("Runlist.Separator")[0]);
419  KVDBRun* run;
420  while ((run = (KVDBRun*) next_run())) {
421 
422  run->WriteRunListLine(rlistf, GetDBEnv("Runlist.Separator")[0]);
423 
424  }
425  } else {
426  Warning("WriteRunListFile()", "run list is empty !!!");
427  }
428  rlistf.close();
429 }

451 Bool_t KVExpDB::OpenCalibFile(const Char_t* type, ifstream& fs) const
452 {
453  //Find and open calibration parameter file of given type. Return kTRUE if all OK.
454  //types are defined in $KVROOT/KVFiles/.kvrootrc by lines such as (use INDRA as example)
455  //
456  //# Default name for file describing systems for each dataset.
457  //INDRADB.Systems: Systems.dat
458  //
459  //A file with the given name will be looked for in $KVROOT/KVFiles/name_of_dataset
460  //where name_of_dataset is given by KVDataBase::fDataSet.Data()
461  //i.e. the name of the associated dataset.
462  //
463  //Filenames specific to a given dataset may also be defined:
464  //
465  //INDRA_camp5.INDRADB.Pedestals: Pedestals5.dat
466  //
467  //where 'INDRA_camp5' is the name of the dataset in question.
468  //GetDBEnv() always returns the correct filename for the currently active dataset.
469 
470  return KVBase::SearchAndOpenKVFile(GetDBEnv(type), fs, fDataSet.Data());
471 }

503  //
504  // root [9] gIndraDB->PrintRuns("8100-8120")

506  // ------------------------------------------------------------------------------------------------------------------
507  // 8100 129Xe + 58Ni 8 MeV/A M>=2 968673

513

516  nl.Begin();
517  while (!nl.End()) {
518  KVDBRun* run = GetDBRun(nl.Next());
519  if (!run) continue;
520  printf("%4d\t%-30s\t%s\t\t%d\t\t%s\n",
521  run->GetNumber(), (run->GetSystem() ? run->GetSystem()->GetName() : " "), run->GetTriggerString(),
522  run->GetEvents(), run->GetComments());

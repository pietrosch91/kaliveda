12 <h2>KVGumbelDistribution</h2>
13 <h4>Gumbel distributions for rank-ordered extremal variables</h4>
14 Consider a probability distribution f(Z) which is used to generate
15 m values of the random variable Z, which can then be ordered
16 in terms of decreasing size
17 <!-- */
18 // --> END_HTML
19 /*
20 BEGIN_LATEX
21 Z_{1}>Z_{2}> ... >Z_{m}
22 END_LATEX
23 */
24 // BEGIN_HTML <!--
25 /* -->
26 If such
27 a procedure is repeated a sufficiently large number of times, it can
28 be shown that the resulting probability distributions of the Z_k
29 are given by the Gumbel distributions:
30 <!-- */
31 // --> END_HTML
32 /*
33 BEGIN_LATEX
34 \phi_{k}(s_{k}) = \frac{k^{k}}{(k-1)!}\frac{1}{b_{m}}e^{-ks_{k}-ke^{-s_{k}}}
35 END_LATEX
36 */
37 // BEGIN_HTML <!--
38 /* -->
39 where
40 <!-- */
41 // --> END_HTML
42 /*
43 BEGIN_LATEX
44 s_{k} = \frac{Z_{k}-a_{k}}{b_{k}}
45 END_LATEX
46 */

109  Double_t xmin, Double_t xmax)
110  : TF1(name, this, &KVGumbelDistribution::GDk, xmin, xmax, 3 - (int)norm,
111  "KVGumbelDistribution", "GDk")
112 {
113  // Gumbel distribution of k-th rank
114  // if norm=kTRUE: normalised PDF, 2 free parameters (a,b)
115  // if norm=kFALSE: 3 parameters (a,b,Integral)

117  fkFac = TMath::Power(k, k) / TMath::Factorial(k - 1);

195 Double_t KVGumbelDistribution::GDk(Double_t* x, Double_t* p)
196 {
197  // Evaluate Gumbel distribution of rank fRank for x
198  // with parameters
199  // par[0] = a
200  // par[1] = b
201  // If distribution is not normalised, then we have
202  // par[2] = normalisation
203 
204  if (p[1] == 0.) return 0.;
205  Double_t s = (*x - p[0]) / p[1];
206  Double_t gum = (fkFac / p[1]);
207  Double_t es = -fRank * (s + TMath::Exp(-s));
208  gum *= TMath::Exp(es);
209  if (!fNormalised) gum *= p[2];
210  return gum;
211 }

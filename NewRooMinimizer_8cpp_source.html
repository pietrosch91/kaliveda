97 NewRooMinimizer::NewRooMinimizer(RooAbsReal& function)
98 {
99  // Construct MINUIT interface to given function. Function can be anything,
100  // but is typically a -log(likelihood) implemented by RooNLLVar or a chi^2
101  // (implemented by RooChi2Var). Other frequent use cases are a RooAddition
102  // of a RooNLLVar plus a penalty or constraint term. This class propagates
103  // all RooFit information (floating parameters, their values and errors)
104  // to MINUIT before each MINUIT call and propagates all MINUIT information
105  // back to the RooFit object at the end of each call (updated parameter
106  // values, their (asymmetric errors) etc. The default MINUIT error level
107  // for HESSE and MINOS error analysis is taken from the defaultErrorLevel()
108  // value of the input function.
109 
110  RooSentinel::activate() ;
111 
112  // Store function reference
113  _extV = 0 ;
114  _func = &function ;
115  _optConst = kFALSE ;
116  _verbose = kFALSE ;
117  _profile = kFALSE ;
118  _profileStart = kFALSE ;
119  _printLevel = 1 ;
120  _minimizerType = "Minuit"; // default minimizer
121 
122  if (_theFitter) delete _theFitter ;
123  _theFitter = new ROOT::Fit::Fitter;
124  _fcn = new NewRooMinimizerFcn(_func, this, _verbose);
125  _theFitter->Config().SetMinimizer(_minimizerType.c_str());
126  setEps(1.0); // default tolerance
127  // default max number of calls
128  _theFitter->Config().MinimizerOptions().SetMaxIterations(500 * _fcn->NDim());
129  _theFitter->Config().MinimizerOptions().SetMaxFunctionCalls(500 * _fcn->NDim());
130 
131  // Shut up for now
132  setPrintLevel(-1) ;
133 
134  // Use +0.5 for 1-sigma errors
135  setErrorLevel(_func->defaultErrorLevel()) ;
136 
137  // Declare our parameters to MINUIT
138  _fcn->Synchronize(_theFitter->Config().ParamsSettings(),
139  _optConst, _verbose) ;
140 
141  // Now set default verbosity
142  if (RooMsgService::instance().silentMode()) {
143  setPrintLevel(-1) ;
144  } else {
145  setPrintLevel(1) ;
146  }
147 }

326 NewRooFitResult* NewRooMinimizer::fit(const char* options)
327 {
328  // Parse traditional RooAbsPdf::fitTo driver options
329  //
330  // m - Run Migrad only
331  // h - Run Hesse to estimate errors
332  // v - Verbose mode
333  // l - Log parameters after each Minuit steps to file
334  // t - Activate profile timer
335  // r - Save fit result
336  // 0 - Run Migrad with strategy 0
337 
338  TString opts(options) ;
339  opts.ToLower() ;
340 
341  // Initial configuration
342  if (opts.Contains("v")) setVerbose(1) ;
343  if (opts.Contains("t")) setProfile(1) ;
344  if (opts.Contains("l")) setLogFile(Form("%s.log", _func->GetName())) ;
345  if (opts.Contains("c")) optimizeConst(1) ;
346 
347  // Fitting steps
348  if (opts.Contains("0")) setStrategy(0) ;
349  migrad() ;
350  if (opts.Contains("0")) setStrategy(1) ;
351  if (opts.Contains("h") || !opts.Contains("m")) hesse() ;
352  if (!opts.Contains("m")) minos() ;
353 
354  return (opts.Contains("r")) ? save() : 0 ;
355 }

520 Int_t NewRooMinimizer::minos(const RooArgSet& minosParamList)
521 {
522  // Execute MINOS for given list of parameters. Changes in parameter values
523  // and calculated errors are automatically
524  // propagated back the RooRealVars representing
525  // the floating parameters in the MINUIT operation
526 
527  if (_theFitter->GetMinimizer() == 0) {
528  coutW(Minimization) << "NewRooMinimizer::minos: Error, run Migrad before Minos!"
529  << endl ;
530  _status = -1;
531  } else if (minosParamList.getSize() > 0) {
532 
533  _fcn->Synchronize(_theFitter->Config().ParamsSettings(),
534  _optConst, _verbose) ;
535  profileStart() ;
536  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::CollectErrors) ;
537  RooAbsReal::clearEvalErrorLog() ;
538 
539  // get list of parameters for Minos
540  TIterator* aIter = minosParamList.createIterator() ;
541  RooAbsArg* arg ;
542  std::vector<unsigned int> paramInd;
543  while ((arg = (RooAbsArg*)aIter->Next())) {
544  RooAbsArg* par = _fcn->GetFloatParamList()->find(arg->GetName());
545  if (par && !par->isConstant()) {
546  Int_t index = _fcn->GetFloatParamList()->index(par);
547  paramInd.push_back(index);
548  }
549  }
550  delete aIter ;
551 
552  if (paramInd.size()) {
553  // set the parameter indeces
554  _theFitter->Config().SetMinosErrors(paramInd);
555 
556  _theFitter->Config().SetMinimizer(_minimizerType.c_str());
557  bool ret = _theFitter->CalculateMinosErrors();
558  _status = ((ret) ? _theFitter->Result().Status() : -1);
559  // to avoid that following minimization computes automatically the Minos errors
560  _theFitter->Config().SetMinosErrors(kFALSE);
561 
562  }
563 
564  RooAbsReal::setEvalErrorLoggingMode(RooAbsReal::PrintErrors) ;
565  profileStop() ;
566  _fcn->BackProp(_theFitter->Result());
567 
568  saveStatus("MINOS", _status) ;
569 
570  }
571 
572  return _status ;
573 }

743 NewRooFitResult* NewRooMinimizer::save(const char* userName, const char* userTitle)
744 {
745  // Save and return a NewRooFitResult snaphot of current minimizer status.
746  // This snapshot contains the values of all constant parameters,
747  // the value of all floating parameters at NewRooMinimizer construction and
748  // after the last MINUIT operation, the MINUIT status, variance quality,
749  // EDM setting, number of calls with evaluation problems, the minimized
750  // function value and the full correlation matrix
751 
752  if (_theFitter->GetMinimizer() == 0) {
753  coutW(Minimization) << "NewRooMinimizer::save: Error, run minimization before!"
754  << endl ;
755  return 0;
756  }
757 
758  TString name, title ;
759  name = userName ? userName : Form("%s", _func->GetName()) ;
760  title = userTitle ? userTitle : Form("%s", _func->GetTitle()) ;
761  NewRooFitResult* fitRes = new NewRooFitResult(name, title) ;
762 
763  // Move eventual fixed parameters in floatList to constList
764  Int_t i ;
765  RooArgList saveConstList(*(_fcn->GetConstParamList())) ;
766  RooArgList saveFloatInitList(*(_fcn->GetInitFloatParamList())) ;
767  RooArgList saveFloatFinalList(*(_fcn->GetFloatParamList())) ;
768  for (i = 0 ; i < _fcn->GetFloatParamList()->getSize() ; i++) {
769  RooAbsArg* par = _fcn->GetFloatParamList()->at(i) ;
770  if (par->isConstant()) {
771  saveFloatInitList.remove(*saveFloatInitList.find(par->GetName()), kTRUE) ;
772  saveFloatFinalList.remove(*par) ;
773  saveConstList.add(*par) ;
774  }
775  }
776  saveConstList.sort() ;
777 
778  fitRes->setConstParList(saveConstList) ;
779  fitRes->setInitParList(saveFloatInitList) ;
780 
781  fitRes->setStatus(_status) ;
782  fitRes->setCovQual(_theFitter->GetMinimizer()->CovMatrixStatus()) ;
783  fitRes->setMinNLL(_theFitter->Result().MinFcnValue()) ;
784  fitRes->setNumInvalidNLL(_fcn->GetNumInvalidNLL()) ;
785  fitRes->setEDM(_theFitter->Result().Edm()) ;
786  fitRes->setFinalParList(saveFloatFinalList) ;
787  if (!_extV) {
788  std::vector<double> globalCC;
789  TMatrixDSym corrs(_theFitter->Result().Parameters().size()) ;
790  TMatrixDSym covs(_theFitter->Result().Parameters().size()) ;
791  for (UInt_t ic = 0; ic < _theFitter->Result().Parameters().size(); ic++) {
792  globalCC.push_back(_theFitter->Result().GlobalCC(ic));
793  for (UInt_t ii = 0; ii < _theFitter->Result().Parameters().size(); ii++) {
794  corrs(ic, ii) = _theFitter->Result().Correlation(ic, ii);
795  covs(ic, ii) = _theFitter->Result().CovMatrix(ic, ii);
796  }
797  }
798  fitRes->fillCorrMatrix(globalCC, corrs, covs) ;
799  } else {
800  fitRes->setCovarianceMatrix(*_extV) ;
801  }
802 
803  fitRes->setStatusHistory(_statusHistory) ;
804 
805  return fitRes ;
806 
807 }

815 RooPlot* NewRooMinimizer::contour(RooRealVar& var1, RooRealVar& var2,
816  Double_t n1, Double_t n2, Double_t n3,
817  Double_t n4, Double_t n5, Double_t n6)

1031  Double_t xhi = _theFitter->Config().ParSettings(i).UpperLimit();

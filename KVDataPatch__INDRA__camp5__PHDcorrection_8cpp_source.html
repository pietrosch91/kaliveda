83 void KVDataPatch_INDRA_camp5_PHDcorrection::ApplyToParticle(KVNucleus* n)
84 {
85  // Patch is applied to all identified & calibrated nuclei with Z>10 on rings 1-9.
86  // The particle energy is recalibrated.
87 
88  KVINDRAReconNuc* N = (KVINDRAReconNuc*)n;
89  if (N->IsIdentified() && N->IsCalibrated() && N->GetZ() > 10 && N->GetRingNumber() < 10) {
90  KVTarget* t = gMultiDetArray->GetTarget();
91  if (t) {
92  // make sure target is in correct state to calculate
93  // target energy losses of fragments
94  if (t->IsIncoming()) t->SetIncoming(kFALSE);
95  if (!t->IsOutgoing()) t->SetOutgoing(kTRUE);
96  }
97  N->Calibrate();//recalibrate particle
98  }
99 }

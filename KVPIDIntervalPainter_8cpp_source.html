176 void KVPIDIntervalPainter::Paint(Option_t*)
177 {

191  fMarker->SetY(fLinearHisto->GetBinContent(fLinearHisto->FindBin(fMarker->GetX())));
192  fInterval->SetPID(fMarker->GetX());
193  pid = fMarker->GetX();
194  if (fDisplayLabel) {
195  fLabel->SetX(fMarker->GetX());
196  fLabel->SetY(fMarker->GetY());
197  }
198  IntMod();
199  }
200  if (fLine1->GetX1() != min) {
201  fInterval->SetPIDmin(fLine1->GetX1());
202  min = fLine1->GetX1();
203  IntMod();
204  }

206  fInterval->SetPIDmax(fLine2->GetX1());
207  max = fLine2->GetX1();

209  }

106 void KVTarArchive::CheckDirectory(const Char_t* dirname, const Char_t* path)
107 {
108  // We look for the directory 'dirname' in the directory 'path', i.e. we look for
109  // path/dirname
110  //
111  // If not found, we look for the following compressed archive:
112  //
113  // path/dirname.tgz
114  //
115  // and if found, uncompress it (with 'tar -zxf').
116  //
117  // Sets fOK to kTRUE if directory/archive file exists.Full path to directory returned in 'fullpath'.
118  //
119  // Sets fOK to kFALSE if directory does not exist and no archive file found.
120 
121  fTGZ = kFALSE;
122  fOK = kFALSE;
123  KVString search = dirname;
124  if (!SearchFile(search.Data(), fFullpath, 1, path)) {
125  // directory not found; look for .tgz
126  search += ".tgz";
127  if (SearchFile(search.Data(), fFullpath, 1, path)) {
128  // uncompress in directory where tgz is found
129  TString pwd = gSystem->pwd();
130  gSystem->cd(gSystem->DirName(fFullpath.Data()));
131  TString cmd;
132  cmd.Form("tar -zxf %s", gSystem->BaseName(fFullpath.Data()));
133  Info("CheckDirectory", "Opening archive file %s", fFullpath.Data());
134  if (gSystem->Exec(cmd.Data())) {
135  Error("CheckDirectory", "Problem executing %s", cmd.Data());
136  fFullpath = "";
137  } else {
138  fOK = kTRUE;
139  // strip '.tgz' to leave just full path to directory
140  fFullpath.Remove(fFullpath.Length() - 4);
141  fTGZ = kTRUE;
142  }
143  gSystem->cd(pwd.Data());
144  }
145  } else {
146  fOK = kTRUE;
147  Info("CheckDirectory", "Found directory %s", fFullpath.Data());
148  }
149 }

158 void KVTarArchive::DeleteDirectory(const Char_t* dirpath)
159 {
160  // Delete all files in directory 'dirpath' (give full path) then delete the directory itself.
161  // This is to get round the fact that gSystem->Unlink(dirname) does not work if the
162  // directory contains any files.
163 
164  Info("DeleteDirectory", "Deleting %s", dirpath);
165  TString pwd = gSystem->WorkingDirectory();//keep working directory
166  TSystemDirectory dir("dir", dirpath);
167  // get list of files (and directories)
168  TList* files = dir.GetListOfFiles();
169  TIter nxtFile(files);
170  TSystemFile* fil = 0;
171  // delete all files (and directories)
172  while ((fil = (TSystemFile*)nxtFile())) {
173  if (fil->IsDirectory()) {
174  // recursively remove subdirectories if necessary
175  if (strcmp(fil->GetName(), "..") && strcmp(fil->GetName(), ".")) {
176  DeleteDirectory(fil->GetTitle());
177  }
178  } else {
179  if (gSystem->Unlink(Form("%s/%s", fil->GetTitle(), fil->GetName())) == -1) {
180  Info("DeleteDirectory", "Cannot delete file %s in %s", fil->GetName(), fil->GetTitle());
181  }
182  }
183  }
184  delete files;
185  gSystem->cd(pwd.Data());//change back to working directory
186  // delete directory
187  if (gSystem->Unlink(dirpath) == 0)
188  Info("DeleteDirectory", "Directory %s deleted", dirpath);
189  else
190  Info("DeleteDirectory", "Cannot delete directory %s", dirpath);
191 }

730 void KVSimDirGUI::RunAnalysis(const TString& type)
731 {
732  unique_ptr<TList> selected_sim_runs(fLVsimData->GetSelectedObjects());
733  unique_ptr<TList> selected_filt_runs(fLVfiltData->GetSelectedObjects());
734  TList* runs_to_analyse(nullptr);
735  TString analysis_task;
736  if (type == "tree") { // analysis of simulated or filtered events
737  if (!selected_sim_runs->GetEntries() && !selected_filt_runs->GetEntries()) {
738  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunAnalysis", "Choose one or more simulated or filtered data files!", kMBIconExclamation);
739  return;
740  }
741  if (selected_sim_runs->GetEntries() && selected_filt_runs->GetEntries()) {
742  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunAnalysis", "Choose EITHER simulated or filtered data files!", kMBIconExclamation);
743  return;
744  }
745  if (fAnalClassHeader == "" || fAnalClassImp == "") {
746  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunAnalysis", "Choose a valid analysis class!", kMBIconExclamation);
747  return;
748  }
749  if (selected_sim_runs->GetEntries()) {
750  runs_to_analyse = selected_sim_runs.get();
751  analysis_task = "analysis simulated";
752  }
753  else {
754  runs_to_analyse = selected_filt_runs.get();
755  analysis_task = "analysis filtered";
756  }
757  }
758  else {
759  // filtering
760  if (selected_filt_runs->GetEntries()) {
761  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunFilter", "Only simulated events can be filtered!", kMBIconExclamation);
762  return;
763  }
764  if (!selected_sim_runs->GetEntries()) {
765  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunFilter", "Choose a simulated data file to filter!", kMBIconExclamation);
766  return;
767  }
768 // if (selected_sim_runs->GetEntries() > 1) {
769 // new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunFilter", "Choose ONE simulated data file to filter!", kMBIconExclamation);
770 // return;
771 // }
772  runs_to_analyse = selected_sim_runs.get();
773  analysis_task = "filter simulated";
774  }
775  KVDataAnalysisTask* anTask = gDataSetManager->GetAnalysisTaskAny(analysis_task);
776  gDataAnalyser = KVDataAnalyser::GetAnalyser(anTask->GetDataAnalyser());
777  gDataAnalyser->SetAnalysisTask(anTask);
778  gDataAnalyser->SetFileList(runs_to_analyse);
779  if (anTask->WithUserClass()) {
780  gDataAnalyser->SetUserClass(fAnalClassName);
781  if (!gDataAnalyser->IsUserClassValid()) {
782  new TGMsgBox(gClient->GetRoot(), MainFrame, "KVSimDirGUI::RunAnalysis", "Compilation failed. Correct mistakes and try again!", kMBIconExclamation);
783  }
784  }
785  else if (strcmp(anTask->GetUserBaseClass(), ""))
786  gDataAnalyser->SetUserClass(anTask->GetUserBaseClass(), kFALSE);//task with default "user" class
787  Bool_t all_events = fCBAllEvents->IsDown();
788  if (!all_events)
789  gDataAnalyser->SetNbEventToRead((Long64_t)fNENumberEvents->GetNumber());
790  else
791  gDataAnalyser->SetNbEventToRead(0);
792  Bool_t cancel_batch_job = kFALSE;
793  if (fWithPROOF) {
794  gBatchSystem->Clear();
795  KVNameValueList batchParams;
796  gBatchSystem->GetBatchSystemParameterList(batchParams);
797  new KVBatchSystemParametersGUI(MainFrame, &batchParams, gDataAnalyser, &cancel_batch_job);
798  if (!cancel_batch_job) {
799  gBatchSystem->SetBatchSystemParameters(batchParams);
800  gDataAnalyser->SetBatchSystem(gBatchSystem);
801  }
802  }
803  if (type == "filter") SetFilterOptions();
804  if (!cancel_batch_job) gDataAnalyser->Run();
805  selected_filt_runs.reset(nullptr);
806  selected_sim_runs.reset(nullptr);
807  RefreshSimDir();
808 }

839  cd.CalculateKinematics();

867 void KVSimDirGUI::ImportSimulation()
868 {
869  // Import simulation data from currently selected directory
870 
871  const char* filetypes[] = {
872  "All files", "*.*",
873  0, 0
874  };
875  TGFileInfo fi;
876  fi.fIniDir = StrDup(fSelectedSimDir->GetDirectory());
877  fi.fFileTypes = filetypes;
878  fi.SetMultipleSelection(true);
879 
880  new KVFileDialog(gClient->GetDefaultRoot(), MainFrame, kKVFDOpen, &fi);
881 
882  bool one_file = (!fi.fMultipleSelection && fi.fFilename);
883  bool many_files = (fi.fMultipleSelection && fi.fFileNamesList);//may only be one
884  if (one_file || many_files) {
885  // set up list of KVSimReader plugins
886  KVString plugins = KVBase::GetListOfPlugins("KVSimReader");
887  KVString choices;
888  plugins.Begin(" ");
889  while (!plugins.End()) {
890  // get URI corresponding to plugin
891  KVString uri = KVBase::GetPluginURI("KVSimReader", plugins.Next());
892  if (choices.Length()) choices += " ";
893  choices += uri;
894  }
895  // get model from user
896  TString model;
897  Bool_t ok;
898  new KVDropDownDialog(MainFrame, "Choose the simulation model and type of events", choices, 0, &model, &ok);
899  if (ok) {
900 
901  if (one_file) {
902  unique_ptr<KVSimReader> SR(KVSimReader::MakeSimReader(model));
903  SR->SetOutputDirectory(fSelectedSimDir->GetDirectory());
904  SR->ConvertAndSaveEventsInFile(fi.fFilename);
905  }
906  else {
907  if (fi.fFileNamesList->GetEntries() == 1) {
908  unique_ptr<KVSimReader> SR(KVSimReader::MakeSimReader(model));
909  SR->SetOutputDirectory(fSelectedSimDir->GetDirectory());
910  SR->ConvertAndSaveEventsInFile(fi.fFileNamesList->First()->GetName());
911  }
912  else {
913  TIter it(fi.fFileNamesList);
914  TObject* o;
915  int i = 1;
916  while ((o = it())) {
917  unique_ptr<KVSimReader> SR(KVSimReader::MakeSimReader(model));
918  SR->SetMultiFiles();
919  SR->SetFileIndex(i++);
920  SR->SetOutputDirectory(fSelectedSimDir->GetDirectory());
921  SR->ConvertAndSaveEventsInFile(o->GetName());
922  }
923  }
924  }
925 
926  RefreshSimDir();
927  }
928  }
929  else if (fi.fMultipleSelection && fi.fFileNamesList) {
930  fi.fFileNamesList->ls();
931  }

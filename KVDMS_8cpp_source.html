67 Bool_t KVDMS::buildCommand(const Char_t* scmd, const Char_t* args, Option_t* opts)
68 {
69  // scmd = one of the DMS commands (init, ls, get, etc.)
70  // args = (optional) list of arguments for command (filename, etc.)
71  // opts = (optional) list of options for command
72  // If the command give in 'scmd' is found in the path specified by the
73  // user's PATH environment variables, this method returns kTRUE and
74  // private member variables fexec and fcmd will hold the full path to
75  // the executable and the full command line to pass to gSystem->Exec.
76  // If executable is not found, returns kFALSE.
77 
78  fexec = scmd;
79  if (!FindExecutable(fexec)) {
80  fexec = "";
81  fcmd = "";
82  return kFALSE;
83  }
84  fcmd = fexec + " ";
85  fcmd += opts;
86  fcmd += " ";
87  fcmd += args;
88  return kTRUE;
89 }

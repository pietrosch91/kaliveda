30 // Any KVEvent object can be represented by using KVEventViewer::DrawEvent(KVEvent*, const Char_t* frame="").
31 // The second (optional) argument allows to choose which reference frame to use (frame must already
32 // have been defined for the KVEvent using KVEvent::SetFrame).
33 //
34 // For each nucleus of the event, a spherical cluster of protons and neutrons will be displayed,
35 // using the default colors for these particles (see constructor and related methods).
36 // One or more nuclei of each event can be "highlighted" (i.e. given different colors for their
37 // neutrons and protons) according to the value given to method SetHighlightMode.
38 //
39 // You can use KVEventViewer::DrawNextEvent() to display, one after the other, a set of events stored in a TTree or in a text file.
40 // With a TTree:
41 //
42 // KVEventViewer view;
43 // view.SetInput( tree->GetBranch("whereTheEventsAre") );
44 // view.DrawNextEvent();
45 // view.DrawNextEvent(); etc. etc.
46 //
47 // With a text file (containing Z and velocity components of each nucleus):
48 //
49 // KVEventViewer view;
50 // view.SetInput( "myEvents.dat" );
51 // view.DrawNextEvent();
52 // view.DrawNextEvent(); etc. etc.

151 void KVEventViewer::DrawNucleus(KVNucleus* nucleus, const Char_t* frame)
152 {
153  // Draw nucleus
154 
155  Int_t N = nucleus->GetN();
156  Int_t Z = nucleus->GetZ();
157 
158  TVector3 V = (fMomentumSpace ? nucleus->GetFrame(frame, kFALSE)->GetMomentum() : nucleus->GetFrame(frame, kFALSE)->GetV());
159 
160  Bool_t Highlight = SetHighlight(nucleus);
161 
162  if (nucleus->GetA() == 1) {
163  TGeoVolume* ball = 0;
164  if (Z == 0) ball = geom->MakeSphere("n", Nuc, 0., free_nucleon_radius);
165  else if (Z == 1) ball = geom->MakeSphere("p", Nuc, 0., free_nucleon_radius);
166  int color = 0;
167  if (Z == 0) color = (Highlight ? fNeutron_color : fneutron_color);
168  else if (Z == 1) color = (Highlight ? fProton_color : fproton_color);
169  ball->SetLineColor(color);
170  top->AddNode(ball, ivol++, new TGeoTranslation(fScaleFactor * V.X(), fScaleFactor * V.Y(), fScaleFactor * V.Z()));
171  return;
172  }
173 
174  double sph_rad = pow(nuclear_radius_parameter, 3.) * (N + Z);
175 
176  for (int i = 0; i < N; i++) {
177  TGeoVolume* ball = geom->MakeSphere("n", Nuc, 0., nucleon_radius);
178  ball->SetLineColor(Highlight ? fNeutron_color : fneutron_color);
179  double rvec = pow(gRandom->Uniform(0., sph_rad), .33333);
180  double x, y, z;
181  gRandom->Sphere(x, y, z, rvec);
182  top->AddNode(ball, ivol++, new TGeoTranslation(fScaleFactor * V.X() + x, fScaleFactor * V.Y() + y, fScaleFactor * V.Z() + z));
183  }
184  for (int i = 0; i < Z; i++) {
185  TGeoVolume* ball = geom->MakeSphere("p", Nuc, 0., nucleon_radius);
186  ball->SetLineColor(Highlight ? fProton_color : fproton_color);
187  double rvec = pow(gRandom->Uniform(0., sph_rad), .33333);
188  double x, y, z;
189  gRandom->Sphere(x, y, z, rvec);
190  top->AddNode(ball, ivol++, new TGeoTranslation(fScaleFactor * V.X() + x, fScaleFactor * V.Y() + y, fScaleFactor * V.Z() + z));
191  }
192 }

199 void KVEventViewer::DrawEvent(KVEvent* event, const Char_t* frame)
200 {
201  // Draw all particles in event which are "ok"
202 
203  if (geom) delete geom;
204  geom = new TGeoManager(Form("Event%d", event->GetNumber()), "Event view");
205  geom->SetNsegments(500);
206  matEmptySpace = new TGeoMaterial("EmptySpace", 0, 0, 0);
207  matNuc = new TGeoMaterial("Nuc", .938, 1., 10000.);
208  matBox = new TGeoMaterial("Box", .938, 1., 10000.);
209 
210  EmptySpace = new TGeoMedium("Empty", 1, matEmptySpace);
211  Nuc = new TGeoMedium("Nuc", 1, matNuc);
212  Box = new TGeoMedium("Box", 1, matBox);
213 
214  top = geom->MakeBox("WORLD", EmptySpace, 30, 30, 30);
215  geom->SetTopVolume(top);
216 
217  // Find biggest velocity/momentum & biggest fragment
218  maxV = 0;
219  maxZ = 0;
220  KVNucleus* nuc;
221  if (fMomentumSpace) fScaleFactor = 1.e-2;
222  else fScaleFactor = 1.;
223  while ((nuc = event->GetNextParticle("ok"))) {
224  Double_t v = fScaleFactor * (fMomentumSpace ? nuc->GetFrame(frame, kFALSE)->GetMomentum().Mag() :
225  nuc->GetFrame(frame, kFALSE)->GetV().Mag());
226  if (v > maxV) maxV = v;
227  if (nuc->GetZ() > maxZ) maxZ = nuc->GetZ();
228  }
229  // scale down
230  maxV *= 0.5;
231  if (fFixPerspective) {
232  // fixed perspective mode
233  // calculate maxV once (first event), store it in fMaxVelocity, then use
234  // for all other events
235  if (fMaxVelocity > 0) maxV = fMaxVelocity;
236  else fMaxVelocity = maxV;
237  } else if (fMaxVelocity > 0) maxV = fMaxVelocity;
238 
239  TGeoVolume* box = geom->MakeSphere("box", Box, 0.99 * maxV, maxV);
240  box->SetLineColor(kBlack);
241  box->SetTransparency(100);
242  top->AddNode(box, 1000);
243 
244  // set volume (nucleon) counter to 0
245  ivol = 0;
246 
247  if (fFixSeed) {
248  if (!(event->GetNumber() % fRefresh)) fSeed += 7;
249  gRandom->SetSeed(fSeed);
250  }
251 
252  while ((nuc = event->GetNextParticle("ok"))) DrawNucleus(nuc, frame);
253 
254  geom->CloseGeometry();
255 
256  top->SetVisContainers(kTRUE);
257  top->Draw("ogl");
258  TGLViewer* view = (TGLViewer*)gPad->GetViewer3D();
259  if (fAxesMode) view->SetGuideState(2, 1, 0, 0);
260  view->SetClearColor(kWhite);
261  view->SetSmoothLines(1);
262  view->SetSmoothPoints(1);
263  view->GetLightSet()->SetLight(TGLLightSet::kLightTop, 1);
264  view->GetLightSet()->SetLight(TGLLightSet::kLightBottom, 0);
265  view->GetLightSet()->SetLight(TGLLightSet::kLightLeft, 1);
266  view->GetLightSet()->SetLight(TGLLightSet::kLightRight, 0);
267  view->GetLightSet()->SetLight(TGLLightSet::kLightFront, 1);
268  view->GetLightSet()->SetLight(TGLLightSet::kLightSpecular, 1);
269  view->SetCurrentCamera(TGLViewer::kCameraOrthoZOY);
270  ((TGLOrthoCamera&)view->CurrentCamera()).SetEnableRotate(kTRUE);
271  if (fXYMode) view->SetOrthoCamera(TGLViewer::kCameraOrthoZOY, 0.9, 0, 0, 0, TMath::Pi() / 2.); //TMath::Pi()/8.,TMath::Pi()/8.);
272  else view->SetOrthoCamera(TGLViewer::kCameraOrthoZOY, 0.9, 0, 0, TMath::Pi() / 8., TMath::Pi() / 8.);
273 // if(fSavePicture) view->SavePicture(Form("Event%d.gif",event->GetNumber()));
274  if (fSavePicture) view->SavePicture(Form("Event-%d.png", event->GetNumber()));
275 }

348 void KVEventViewer::ReadTextEvent()
349 {
350  // Read event from text input
351  // We assume a file containing, for each event,
352  // the multiplicity, N,
353  // followed by N lines containing Z Vx Vy Vz
354  // the atomic number and velocity components (in cm/ns)
355 
356  theEvent->Clear();
357 
358  Double_t V[3];
359  Int_t Z;
360 
361  Int_t nballs;
362  eventFile >> nballs;
363  eventCounter++;
364 
365  theEvent->SetNumber(eventCounter);
366 
367  for (int i = 0; i < nballs; i++) {
368 
369  eventFile >> Z >> V[0] >> V[1] >> V[2];
370 
371  KVNucleus* nuc = theEvent->AddParticle();
372  nuc->SetZ(Z);
373  TVector3 v(V);
374  nuc->SetVelocity(v);
375 
376  }
377 }

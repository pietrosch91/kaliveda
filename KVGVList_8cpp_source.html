15 This class allows to process in a single call many KVVargGlob instances.
16 The methods used to initialize and fill variables in such a list are:
17

19  void Init(void); // initializes the global variables (call once)
20  void CalculateGlobalVariables(KVEvent *e); // compute all global variables for the event

22 
23 By default the KVGVList does not own the objects it contains (they may be on the stack).
24 User's responsibility in this case to delete heap-based objects after use.
25 
26 ~~~~~~~~~~~~

29  KVZmean zmean;
30  KVZmax zmax;

34  gvlist.Add(&zmax);

36

39 
40  cout << "Total kinetic energy : " << Sekin->GetValue() << endl;

42  cout << "Standard deviation : " << zmean("RMS") << endl;

207 void KVGVList::CalculateGlobalVariables(KVEvent* e)
208 {
209  // This method will calculate all global variables defined in the list for the event 'e'.
210  // - all 1-body variables will be calculated in a single loop over the particles;
211  // - all 2-body variables will be calculated in a single loop over particle pairs;
212  // - all N-body variables will be calculated
213 
214  Reset();
215  if (Has1BodyVariables() || Has2BodyVariables()) {
216 
217 #ifdef __WITH_TITER_BUG
218  Int_t mult = e->GetMult();
219  for (int it1 = 1; it1 <= mult; ++it1) {
220  KVNucleus* n1 = e->GetParticle(it1);
221  if (!n1->IsOK()) continue;
222  if (Has1BodyVariables()) Fill(n1); // calculate 1-body variables
223 
224  if (Has2BodyVariables()) {
225  for (int it2 = it1; it2 <= mult; ++it2) {
226  KVNucleus* n2 = e->GetParticle(it2);
227  if (!n2->IsOK()) continue;
228  // calculate 2-body variables
229  // we use every pair of particles (including identical pairs) in the event
230  Fill2(n1, n2);
231  }
232  }
233  }
234 #else
235 #ifdef WITH_CPP11
236  for (KVEvent::Iterator it1(e, KVEvent::Iterator::Type::OK); it1 != KVEvent::Iterator::End(); ++it1) {
237 #else
238  for (KVEvent::Iterator it1(e, KVEvent::Iterator::OK); it1 != KVEvent::Iterator::End(); ++it1) {
239 #endif
240  if (Has1BodyVariables()) Fill(&(*it1));// calculate 1-body variables
241  if (Has2BodyVariables()) {
242  for (KVEvent::Iterator it2(it1); it2 != KVEvent::Iterator::End(); ++it2) {
243  // calculate 2-body variables
244  // we use every pair of particles (including identical pairs) in the event
245  Fill2(&(*it1), &(*it2));
246  }
247  }
248  }
249 #endif
250  }
251 
252  // calculate N-body variables
253  if (HasNBodyVariables()) FillN(e);
254 }

277 void KVGVList::Add(TObject* obj)
278 {
279  // Overrides KVUniqueNameList::Add(TObject*) so that global variable pointers are sorted
280  // between the 3 lists used for 1-body, 2-body & N-body variables.
281  //
282  // We also print a warning in case the user tries to add a global variable with the
283  // same name as one already in the list. In the case we retain only the first global variable,
284  // any others with the same name are ignored
285 
286  KVUniqueNameList::Add(obj); // add object to main list, check duplicates
287  if (!ObjectAdded()) {
288  Warning("Add", "You tried to add a global variable with the same name as one already in the list");
289  Warning("Add", "Only the first variable added to the list will be used: name=%s class=%s",
290  GetGV(obj->GetName())->GetName(), GetGV(obj->GetName())->ClassName());
291  Warning("Add", "The following global variable (the one you tried to add) will be ignored:");
292  printf("\n");
293  obj->Print();
294  return;
295  }
296  if (obj->InheritsFrom("KVVarGlob")) {
297  // put global variable pointer in appropriate list
298  KVVarGlob* vg = (KVVarGlob*)obj;
299  if (vg->IsOneBody()) fVG1.Add(vg);
300  else if (vg->IsTwoBody()) fVG2.Add(vg);
301  else if (vg->IsNBody()) fVGN.Add(vg);
302  }
303 }

319 void KVGVList::MakeBranches(TTree* tree)
320 {
321  // Create a branch in the TTree for each global variable in the list.
322  // A leaf with the name of each global variable will be created to hold the
323  // value of the variable (result of KVVarGlob::GetValue() method).
324  // For multi-valued global variables we add a branch for each value with name
325  //
326  // `GVname.ValueName`
327  //
328  // Any variable for which KVVarGlob::SetMaxNumBranches() was called with argument `0`
329  // will not be added to the TTree.
330 
331  if (!tree) return;
332  if (fNbIBranch >= MAX_CAP_BRANCHES && fNbBranch >= MAX_CAP_BRANCHES) return;
333 
334  // Make sure all variables are initialised before proceeding
335  Init();
336 
337  TIter next(this);
338  KVVarGlob* ob;
339  while ((ob = (KVVarGlob*)next())) {
340  if (ob->GetNumberOfBranches()) { //skip variables for which KVVarGlob::SetMaxNumBranches(0) was called
341  if (ob->GetNumberOfValues() > 1) {
342  // multi-valued variable
343  for (int i = 0; i < ob->GetNumberOfBranches(); i++) {
344  // replace any nasty mathematical symbols which could pose problems
345  // in names of TTree leaves/branches
346  TString sane_name(ob->GetValueName(i));
347  sane_name.ReplaceAll("*", "star");
348  if (ob->GetValueType(i) == 'I') {
349  if (fNbIBranch < MAX_CAP_BRANCHES) tree->Branch(Form("%s.%s", ob->GetName(), sane_name.Data()), &fIBranchVar[ fNbIBranch++ ], Form("%s.%s/I", ob->GetName(), sane_name.Data()));
350  } else {
351  if (fNbBranch < MAX_CAP_BRANCHES) tree->Branch(Form("%s.%s", ob->GetName(), sane_name.Data()), &fBranchVar[ fNbBranch++ ], Form("%s.%s/D", ob->GetName(), sane_name.Data()));
352  }
353  if (fNbIBranch == MAX_CAP_BRANCHES) break;
354  if (fNbBranch == MAX_CAP_BRANCHES) break;
355  }
356  } else {
357  if (ob->GetValueType(0) == 'I') {
358  if (fNbIBranch < MAX_CAP_BRANCHES) tree->Branch(ob->GetName(), &fIBranchVar[ fNbIBranch++ ], Form("%s/I", ob->GetName()));
359  } else {
360  if (fNbBranch < MAX_CAP_BRANCHES) tree->Branch(ob->GetName(), &fBranchVar[ fNbBranch++ ], Form("%s/D", ob->GetName()));
361  }
362  }
363  }
364  }
365 }

377 void KVGVList::FillBranches()
378 {
379  // Use this method ONLY if you first use MakeBranches(TTree*) in order to
380  // automatically create branches for your global variables.
381  // Call this method for each event in order to put the values of the variables
382  // in the branches ready for TTree::Fill to be called (note that TTree::Fill is not
383  // called in this method: you should do it after this).
384 
385  if (!fNbBranch && !fNbIBranch) return; // MakeBranches has not been called
386 
387  int INT_index = 0;
388  int FLT_index = 0;
389  TIter next(this);
390  KVVarGlob* ob;
391  while ((ob = (KVVarGlob*)next())) {
392  if (ob->GetNumberOfBranches()) { //skip variables for which KVVarGlob::SetMaxNumBranches(0) was called
393 
394  if (ob->GetNumberOfValues() > 1) {
395  // multi-valued variable
396  for (int j = 0; j < ob->GetNumberOfBranches(); j++) {
397  if (ob->GetValueType(j) == 'I') fIBranchVar[ INT_index++ ] = (Int_t)ob->GetValue(j);
398  else fBranchVar[ FLT_index++ ] = ob->GetValue(j);
399  }
400  } else {
401  if (ob->GetValueType(0) == 'I') fIBranchVar[ INT_index++ ] = (Int_t)ob->GetValue();
402  else fBranchVar[ FLT_index++ ] = ob->GetValue();
403  }
404 
405  }
406  }
407 }

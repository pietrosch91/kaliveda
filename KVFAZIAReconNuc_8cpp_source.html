300 void KVFAZIAReconNuc::Identify()
301 {
302  // Try to identify this nucleus by calling the Identify() function of each
303  // ID telescope crossed by it, starting with the telescope where the particle stopped, in order
304  // - attempt identification only in ID telescopes containing the stopping detector
305  // - only telescopes which have been correctly initialised for the current run are used,
306  // i.e. those for which KVIDTelescope::IsReadyForID() returns kTRUE.
307  // This continues until a successful identification is achieved or there are no more ID telescopes to try.
308  // The identification code corresponding to the identifying telescope is set as the identification code of the particle.
309 
310 
311  KVList* idt_list = GetStoppingDetector()->GetIDTelescopes();
312  KVIdentificationResult* IDR = 0;
313  Int_t idnumber = 1;
314  if (idt_list && idt_list->GetSize() > 0) {
315 
316  KVIDTelescope* idt;
317  TIter next(idt_list);
318  while ((idt = (KVIDTelescope*) next())) { // && !IsIdentified()) {
319  if (StoppedInSI1() && !strcmp(idt->GetType(), "Si-Si")) continue; // why ?
320  if (StoppedInSI2() && !strcmp(idt->GetType(), "Si-CsI")) continue; // why ?
321  IDR = GetIdentificationResult(idnumber);
322  IDR->SetName(idt->GetName());
323  IDR->SetType(idt->GetType());
324  if (idt->IsReadyForID()) { // is telescope able to identify for this run ?
325 
326  IDR->IDattempted = kTRUE;
327  IDR->IDOK = kFALSE;
328  idt->Identify(IDR);
329  if (IDR->IDOK) { //Correspond to Quality code <=3
330  SetIdentification(IDR);
331  SetIdentifyingTelescope(idt);
332  SetIsIdentified();
333  //return;
334  } else {
335  SetIdentification(IDR);
336  SetIdentifyingTelescope(idt);
337  //SetIDCode(0);
338  //SetZandA(0, 0);
339  //SetIsIdentified();
340  }
341  } else {
342  IDR->IDattempted = kFALSE;
343  }
344  idnumber += 1;
345  }
346 
347  KVIdentificationResult partID;
348  Bool_t ok = kFALSE;
349  if (StoppedInSI1()) {
350  ok = CoherencySi(partID);
351  } else if (StoppedInSI2()) {
352  ok = CoherencySiSi(partID);
353  } else if (StoppedInCSI()) {
354  ok = CoherencySiCsI(partID);
355  }
356 
357  if (ok) {
358  SetIsIdentified();
359  KVIDTelescope* idt = (KVIDTelescope*)GetIDTelescopes()->FindObjectByType(partID.GetType());
360  if (!idt) {
361  Warning("Identify", "cannot find ID telescope with type %s", partID.GetType());
362  GetIDTelescopes()->ls();
363  partID.Print();
364  }
365  SetIdentifyingTelescope(idt);
366  SetIdentification(&partID);
367  }
368 
369  }

538 void KVFAZIAReconNuc::Calibrate()
539 {
540  // Perform energy calibration of (previously identified) particle
541 
542  KVNucleus avatar;
543  //printf("start Calibrate\n");
544  Int_t ntot = GetDetectorList()->GetEntries();
545  if (ntot < 1)
546  return;
547  Bool_t punch_through = kFALSE;
548  Bool_t incoherency = kFALSE;
549  Bool_t pileup = kFALSE;
550  Bool_t check_error = kFALSE;
551 
552  double error_si1 = 0, error_si2 = 0; // error_csi=0;
553  Double_t* eloss = new Double_t[ntot];
554  for (Int_t ii = 0; ii < ntot; ii += 1) eloss[ii] = 0;
555  TIter next(GetDetectorList());
556  KVFAZIADetector* det = 0;
557  Int_t ndet = 0;
558  Int_t ndet_calib = 0;
559  Double_t etot = 0;
560  while ((det = (KVFAZIADetector*)next())) {
561 
562  if (det->IsCalibrated()) {
563  if (det->GetIdentifier() == KVFAZIADetector::kCSI) {
564  if (det->GetCalibrator("Channel-Energy")->InheritsFrom("KVLightEnergyCsIFull")) {
565  KVLightEnergyCsIFull* calib = (KVLightEnergyCsIFull*)det->GetCalibrator("Channel-Energy");
566  calib->SetZ(GetZ());
567  calib->SetA(GetA());
568  eloss[ntot - ndet - 1] = calib->Compute(det->GetQ3Amplitude());
569  } else if (det->GetCalibrator("Channel-Energy")->InheritsFrom("KVLightEnergyCsI")) {
570  KVLightEnergyCsI* calib = (KVLightEnergyCsI*)det->GetCalibrator("Channel-Energy");
571  calib->SetZ(GetZ());
572  calib->SetA(GetA());
573  eloss[ntot - ndet - 1] = calib->Compute(det->GetQ3Amplitude());
574  }
575  } else eloss[ntot - ndet - 1] = det->GetEnergy();
576 
577  if (det->GetIdentifier() == KVFAZIADetector::kSI1) fESI1 = eloss[ntot - ndet - 1];
578  else if (det->GetIdentifier() == KVFAZIADetector::kSI2) fESI2 = eloss[ntot - ndet - 1];
579  else if (det->GetIdentifier() == KVFAZIADetector::kCSI) fECSI = eloss[ntot - ndet - 1];
580  etot += eloss[ntot - ndet - 1];
581  ndet_calib += 1;
582  }
583  ndet += 1;
584  }
585  if (ndet == ndet_calib) {
586  Double_t E_targ = 0;
587  SetEnergy(etot);
588 
589  if (IsAMeasured()) {
590  Double_t etot_avatar = 0;
591  Double_t chi2 = 0;
592  avatar.SetZAandE(GetZ(), GetA(), GetKE());
593  for (Int_t nn = ntot - 1; nn >= 0; nn -= 1) {
594  det = (KVFAZIADetector*)GetDetector(nn);
595  Double_t temp = det->GetELostByParticle(&avatar);
596  etot_avatar += temp;
597  chi2 += TMath::Power((eloss[ntot - 1 - nn] - temp) / eloss[ntot - 1 - nn], 2.);
598  avatar.SetKE(avatar.GetKE() - temp);
599  if (det->GetIdentifier() == KVFAZIADetector::kSI1) error_si1 = (fESI1 - temp) / fESI1;
600  else if (det->GetIdentifier() == KVFAZIADetector::kSI2) error_si2 = (fESI2 - temp) / fESI2;
601 // else if (det->GetIdentifier() == KVFAZIADetector::kCSI) error_csi = (fECSI-temp)/fECSI;
602  }
603 
604  chi2 /= ndet;
605  if (avatar.GetKE() > 0) {
606  //Warning("Calibrate", "Incoherence energie residuelle %lf (PUNCH THROUGH) %s", avatar.GetKE(),GetStoppingDetector()->GetName());
607  punch_through = kTRUE;
608  // } else if (TMath::Abs(etot - etot_avatar) > 1e-3) {
609  } else if (chi2 > 10.) {
610  //Warning("Calibrate", "Incoherence %lf != %lf", etot, etot_avatar);
611  incoherency = kTRUE;
612  } else if (TMath::Abs(error_si1) > 0.15 || TMath::Abs(error_si1) + TMath::Abs(error_si2) > 0.15) {
613  if (StoppedInCSI() && (fECSI / etot) < 0.03) pileup = kTRUE;
614  else check_error = kTRUE;
615  } else {
616  // chi2 /= ndet;
617  }
618  }
619 
620  if (GetZ() && GetEnergy() > 0) {
621  E_targ = gMultiDetArray->GetTargetEnergyLossCorrection(this);
622  SetTargetEnergyLoss(E_targ);
623  }
624 
625  Double_t E_tot = GetEnergy() + E_targ;
626  SetEnergy(E_tot);
627  // set particle momentum from telescope dimensions (random)
628  GetAnglesFromStoppingDetector();
629  SetECode(0);
630  if (punch_through) SetECode(2);
631  if (incoherency) SetECode(3);
632  if (check_error) SetECode(5); //
633  if (pileup) SetECode(4); //
634 
635  SetIsCalibrated();
636  } else {
637  if (StoppedInCSI() && !(GetCSI()->IsCalibrated()) && ndet_calib == 2) {
638  if (GetZ() > 0) {
639  if (!IsAMeasured()) {
640 
641  if (GetZ() == 1) SetA(1);
642  else if (GetZ() == 2) SetA(4);
643  else if (GetZ() == 20) SetA(48);
644  else {
645  SetA(1.04735 + 1.99941 * GetZ() + 0.00683224 * TMath::Power(GetZ(), 2.));
646  }
647  }
648 
649  Double_t E_targ = 0;
650  fECSI = GetSI2()->GetEResFromDeltaE(GetZ(), GetA(), fESI2);
651  SetEnergy(fECSI + fESI1 + fESI2);
652 
653  E_targ = gMultiDetArray->GetTargetEnergyLossCorrection(this);
654  Double_t E_tot = GetEnergy() + E_targ;
655  SetECode(1);
656  SetIsCalibrated();
657  SetEnergy(E_tot);
658  GetAnglesFromStoppingDetector();
659  }
660  }
661  }
662 
663  delete [] eloss;

206 Double_t KVHarpeeSi::GetCalibT(const Char_t* type)
207 {
208  // Calculate calibrated time in ns of type 'type' (SI_HF, SI_SED1,
209  // SI_INDRA, SI_MCP, ...) for coder values.
210  // Returns 0 if calibration not ready or time ACQ parameter not fired.
211  // (we require that the acquisition parameter has a value
212  // greater than the current pedestal value).
213  // The reference time T0 used for the calibration is the one of this
214  // detector if it is fired otherwise T0 is the one of the
215  // fired silicon detector of Harpee which stopped the time.
216 
217  KVHarpeeSi* firedSi = (Fired("Pany") ? this : GetFiredHarpeeSi());
218  if (!firedSi) return 0;
219 
220  KVFunctionCal* cal = GetTCalibrator(type);
221  if (cal && cal->GetStatus() && cal->GetACQParam()->Fired("P"))
222  return cal->Compute() + firedSi->GetT0(type);
223  return 0;
224 }

275 TF1* KVHarpeeSi::GetELossFunction(Int_t Z, Int_t A)
276 {
277  // Overrides KVDetector::GetELossFunction
278  // If the pulse height deficit (PHD) has been set for this detector,
279  // we return an energy loss function which takes into account the PHD,
280  // i.e. for an incident energy E we calculate
281  //
282  // dEphd(E,Z,A) = dE(E,Z,A) - PHD(dE,Z)
283  //
284  // If no PHD is set, we return the usual KVDetector::GetELossFunction
285  // which calculates dE(E,Z,A)
286 
287  if (fPHD && fPHD->GetStatus()) return fPHD->GetELossFunction(Z, A);
288 
289  return KVDetector::GetELossFunction(Z, A);
290 }

304 Double_t KVHarpeeSi::GetEnergy()
305 {
306  //Redefinition of KVDetector::GetEnergy().
307  //If energy lost in active volume is already set (e.g. by calculation of energy loss

441 Bool_t KVHarpeeSi::PositionIsOK()
442 {
443  // Returns true if all the conditions to access to the particle position
444  // are verified. In this case the position is given by the method
445  // GetPosition(...).
446  // The conditions are:
447  // -the multiplicity of fired ( "Pany" option of Fired() ) Harpee Si
448  // detectors must be equal to one;
449  // -this detectector must be the fired detector.
450 
451  if (!TestBit(kPosIsOK)) {
452  Int_t mult = 0;
453  TIter next(fHarpeeSiList);
454  KVHarpeeSi* si = NULL;
455  while ((si = (KVHarpeeSi*)next())) {
456  if (si->Fired("Pany")) {
457  mult++;
458  fSiForPosition = si;
459  }
460  }
461  if (mult != 1) fSiForPosition = NULL;
462  SetBit(kPosIsOK);
463  }
464  return fSiForPosition == this;
465 }

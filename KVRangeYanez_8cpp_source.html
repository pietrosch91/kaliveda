145 KVIonRangeTableMaterial* KVRangeYanez::GetMaterialWithNameOrType(const Char_t* material)
146 {
147  // Returns pointer to material of given name or type.
148  // Note that any request for an element of the periodic table will cause
149  // the corresponding material to be created (if not already done).
150  // If given in the form of an isotope ("48Ca", "124Sn", etc.) this material will
151  // be isotopically pure. If not ("Ca", "natSn", etc.) the material generated will be
152  // a mixture of the most abundant naturally-occurring isotopes.
153 
154  CheckMaterialsList();
155  KVIonRangeTableMaterial* M = (KVIonRangeTableMaterial*)fMaterials->FindObject(material);
156  if (!M) {
157  M = (KVIonRangeTableMaterial*)fMaterials->FindObjectByType(material);
158  }
159  if (!M) {
160  // is the requested material an atomic element?
161  Int_t A = KVNucleus::IsMassGiven(material);
162  if (A > -1) {
163  KVNucleus n(material);
164  AddElementalMaterial(n.GetZ(), A);
165  M = (KVIonRangeTableMaterial*)fMaterials->FindObjectByType(material);
166  }
167  }
168  return M;
169 }

233 void KVRangeYanez::AddElementalMaterial(Int_t z, Int_t a)
234 {
235  // Adds a material composed of a single isotope of a chemical element.
236  // If the isotope (a) is not specified, we create a material containing the naturally
237  // occuring isotopes of the given element, weighted according to their abundance.
238  // If the mass is given, the material symbol will be "AX" where X is the symbol for the element
239  // e.g. "48Ca", "124Sn", etc.
240  // and the material name will be "Xxx-A" where Xxx is the name of the element
241  // e.g. "Calcium-48", "Tin-124", etc.
242  // Otherwise, we just use the element symbol and name for naturally-occurring
243  // mixtures of atomic elements ("Ca", "Calcium", etc.).
244 
245  KVIonRangeTableMaterial* mat;
246  if (!a) {
247  mat = MakeNaturallyOccuringElementMixture(z);
248  } else {
249  if (!gNDTManager) {
250  Error("AddElementalMaterial",
251  "Nuclear data tables have not been initialised");
252  return;
253  }
254  KVElementDensity* ed = (KVElementDensity*)gNDTManager->GetData(z, a, "ElementDensity");
255  if (!ed) {
256  Error("AddElementalMaterial",
257  "No element found in ElementDensity NDT-table with Z=%d", z);
258  return;
259  }
260  TString state = "solid";
261  if (ed->IsGas()) state = "gas";
262  mat = new KVRangeYanezMaterial(this, Form("%s-%d", ed->GetElementName(), a),
263  Form("%d%s", a, ed->GetElementSymbol()),
264  state, ed->GetValue(), z, a);
265  mat->Initialize();
266  }
267  CheckMaterialsList();
268  fMaterials->Add(mat);
269  mat->ls();
270 }

335 KVIonRangeTableMaterial* KVRangeYanez::MakeNaturallyOccuringElementMixture(Int_t z)
336 {
337  // create a material containing the naturally occuring isotopes of the given element,
338  // weighted according to their abundance.
339 
340  if (!gNDTManager) {
341  Error("MakeNaturallyOccuringElementMixture",
342  "Nuclear data tables have not been initialised");
343  return NULL;
344  }
345  KVElementDensity* ed = (KVElementDensity*)gNDTManager->GetData(z, z, "ElementDensity");
346  if (!ed) {
347  Error("AddElementalMaterial",
348  "No element found in ElementDensity NDT-table with Z=%d", z);
349  return 0x0;
350  }
351  TString state = "solid";
352  if (ed->IsGas()) state = "gas";
353  KVRangeYanezMaterial* mat =
354  new KVRangeYanezMaterial(this,
355  ed->GetElementName(),
356  ed->GetElementSymbol(),
357  state, ed->GetValue());
358  KVNucleus nuc(z);
359  KVNumberList isotopes = nuc.GetKnownARange();
360  isotopes.Begin();
361  while (!isotopes.End()) {
362 
363  nuc.SetA(isotopes.Next());
364  Double_t abundance = nuc.GetAbundance() / 100.;
365  if (abundance > 0.) mat->AddMixtureElement(z, nuc.GetA(), 1, abundance);
366 
367  }
368  mat->Initialize();
369  return (KVIonRangeTableMaterial*)mat;
370 }

377 void KVRangeYanez::ReadPredefinedMaterials(const Char_t* filename)
378 {
379  // Read materials from file whose name is given
380 
381  TString DataFilePath = filename;
382 
383  ifstream filestream;
384  if (!SearchAndOpenKVFile(DataFilePath, filestream, "data")) {
385  Error("ReadPredefinedMaterials", "Cannot open %s for reading", DataFilePath.Data());
386  return;
387  }
388  Info("ReadPredefinedMaterials", "Reading materials in file : %s", filename);
389 
390  Bool_t compound, mixture;
391  compound = mixture = kFALSE;
392 
393  KVString line;
394  while (filestream.good()) {
395  line.ReadLine(filestream);
396  if (filestream.good()) {
397  if (line.BeginsWith("//")) continue;
398  if (line.BeginsWith("COMPOUND")) {
399  compound = kTRUE;
400  mixture = kFALSE;
401  } else if (line.BeginsWith("MIXTURE")) {
402  compound = kFALSE;
403  mixture = kTRUE;
404  }
405  if (compound || mixture) {
406  // new compound or mixed material
407  KVString name, symbol, state;
408  Double_t density = -1;
409  KVString element[10];
410  Int_t natoms[10];
411  Int_t z[10], a[10];
412  Double_t proportion[10];
413  Int_t nelem = 0;
414  line.ReadLine(filestream);
415  while (filestream.good() && !line.IsWhitespace() && line != "\n") {
416  line.Begin("=");
417  KVString next = line.Next();
418  if (next == "name") name = line.Next();
419  else if (next == "symbol") symbol = line.Next();
420  else if (next == "state") state = line.Next();
421  else if (next == "density") density = line.Next().Atof();
422  else if (next == "nelem") {
423  nelem = line.Next().Atoi();
424  for (int i = 0; i < nelem; i++) {
425  line.ReadLine(filestream);
426  line.Begin(" ");
427  element[i] = line.Next();
428  a[i] = KVNucleus::IsMassGiven(element[i]);
429  KVNucleus n(element[i]);
430  z[i] = n.GetZ();
431  if (!a[i]) a[i] = TMath::Nint(n.GetNaturalA());
432  natoms[i] = line.Next().Atoi();
433  if (mixture) proportion[i] = line.Next().Atof();
434  }
435  }
436  line.ReadLine(filestream, kFALSE); //do not skip 'whitespace'
437  }
438  if (compound) AddCompoundMaterial(name, symbol, nelem, z, a, natoms, density);
439  else if (mixture) AddMixedMaterial(name, symbol, nelem, z, a, natoms, proportion, density);
440  compound = mixture = kFALSE;
441  }
442  }
443  }
444 }

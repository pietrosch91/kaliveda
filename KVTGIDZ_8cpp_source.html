76 KVTGIDZ::KVTGIDZ(const Char_t* name, Int_t npar, Int_t type, Int_t light, Int_t mass)
77  : KVTGID(name, "fede", 0.1, 150., npar, 4, 5)
78 {
79  // Create Z identification with given "name", using the generalised
80  // Tassan-Got functional KVTGIDFunctions::fede.
81  // npar = total number of parameters
82  // type = functional type (0: standard, 1:extended)
83  // light = with (1) or without (0) CsI light-energy dependence
84  // mass = mass formula used to calculate A for each Z (see KVNucleus::GetAFromZ)
85 
86  fType = type;
87  fLight = light;
88  fZorA = 1; //only Z identification
89  fMassFormula = mass;
90  SetParameter(0, type);
91  SetParameter(1, light);
92  SetParameter(2, fZorA);
93  SetParameter(3, mass);
94  Int_t np = 6;
95  fLambda = np++;
96  if (type) {
97  fAlpha = np++;
98  fBeta = np++;
99  fMu = np++;
100  fNu = np++;
101  fXi = np++;
102  } else {
103  fMu = np++;
104  }
105  fG = np++;
106  fPdx = np++;
107  fPdy = np++;
108  if (light) fEta = np;
109  SetLTGParameterNames();
110 }

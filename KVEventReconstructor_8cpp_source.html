232 Double_t KVEventReconstructor::GetTargetEnergyLossCorrection(KVReconstructedNucleus* ion)
233 {
234  // Calculate the energy loss in the current target of the multidetector
235  // for the reconstructed charged particle 'ion', assuming that the current
236  // energy and momentum of this particle correspond to its state on
237  // leaving the target.
238  //
239  // WARNING: for this correction to work, the target must be in the right 'state':
240  //
241  // gMultiDetArray->GetTarget()->SetIncoming(kFALSE);
242  // gMultiDetArray->GetTarget()->SetOutgoing(kTRUE);
243  //
244  // (see KVTarget::GetParticleEIncFromERes).
245  //
246  // The returned value is the energy lost in the target in MeV.
247  // The energy/momentum of 'ion' are not affected.
248 
249  if (!GetArray()->GetTarget() || !ion) return 0.0;
250  return (GetArray()->GetTarget()->GetParticleEIncFromERes(ion) - ion->GetEnergy());
251 }

143 Bool_t KVIDGridManager::ReadAsciiFile(const Char_t* filename)
144 {
145  //read file, create grids corresponding to information in file.
146  //note: any existing grids are not destroyed, use Clear() beforehand if you want to
147  //start afresh and anew (ais athat aOK?)
148  //
149  //the list of grids created by reading the file can be accessed with method
150  //GetLastReadGrids() after calling this method
151 
152  // clear list of read grids
153  fLastReadGrids.Clear();
154 
155  Bool_t is_it_ok = kFALSE;
156  ifstream gridfile(filename);
157  if (!gridfile.good()) {
158  Error("ReadAsciiFile", "File %s cannot be opened", filename);
159  return is_it_ok;
160  }
161 
162  KVString s;
163 
164  fGrids->Disconnect("Modified()", this, "Modified()");
165  while (gridfile.good()) {
166  //read a line
167  s.ReadLine(gridfile);
168  if (s.BeginsWith("++")) {
169  //New grid
170  //Get name of class by stripping off the '+' at the start of the line
171  s.Remove(0, 2);
172  /************ BACKWARDS COMPATIBILITY FIX *************
173  Old grid files may contain obsolete KVIDZGrid class
174  We replace by KVIDZAGrid with SetOnlyZId(kTRUE)
175  */
176  //Make new grid using this class
177  KVIDGraph* grid = 0;
178  Bool_t onlyz = kFALSE;
179  if (s == "KVIDZGrid") {
180  s = "KVIDZAGrid";
181  onlyz = kTRUE;
182  }
183  TClass* clas = TClass::GetClass(s.Data());
184  if (!clas) {
185  Fatal("ReadAsciiFile",
186  "Cannot load TClass information for %s", s.Data());
187  }
188  grid = (KVIDGraph*) clas->New();
189  fLastReadGrids.Add(grid);
190  //read grid
191  grid->ReadFromAsciiFile(gridfile);
192  if (onlyz) grid->SetOnlyZId(kTRUE);
193  }
194  }
195 
196  gridfile.close();
197  is_it_ok = kTRUE;
198  Modified(); // emit signal to say something changed
199  fGrids->Connect("Modified()", "KVIDGridManager", this, "Modified()");
200  return is_it_ok;
201 }

211 Int_t KVIDGridManager::WriteAsciiFile(const Char_t* filename, const TCollection* selection)
212 {
213  // Write grids in file 'filename'.
214  // If selection=0 (default), write all grids.
215  // If selection!=0, write only grids in list.
216  // Returns number of grids written in file.
217 
218  ofstream gridfile(filename);
219 
220  const TCollection* list_of_grids = (selection ? selection : fGrids);
221  TIter next(list_of_grids);
222  KVIDGraph* grid = 0;
223  Int_t n_saved = 0;
224  while ((grid = (KVIDGraph*) next())) {
225 
226  grid->WriteToAsciiFile(gridfile);
227  Info("WriteAsciiFile", "%s saved", grid->GetName());
228  n_saved++;
229 
230  }
231 
232  gridfile.close();
233  return n_saved;
234 }

269 void KVIDGridManager::GetListOfIDTelescopeLabels(KVString& list)
270 {
271  // Replace contents of KVString with a comma-separated list of all
272  // different labels of ID telescopes associated with current list of ID grids.
273 
274  list = "";
275  TIter next(fGrids);
276  KVIDGraph* grid = 0;
277  KVString lab;
278  while ((grid = (KVIDGraph*) next())) {
279  lab.Form("/%s/", grid->GetIDTelescopeLabel());
280  if (!list.Contains(lab)) list.Append(lab);
281  }
282  list.ReplaceAll("//", ",");
283  list.ReplaceAll("/", "");
284  if (list.EqualTo(",")) list = "";
285 }

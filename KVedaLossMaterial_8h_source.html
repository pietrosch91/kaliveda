70  static void SetNoLimits(Bool_t on = kTRUE)
71  {
72  // Normally all range, dE, Eres functions are limited to range 0<=E<=Emax,
73  // where Emax is nominal max energy for which range tables are valid
74  // (usually 400MeV/u for Z<3, 250MeV/u for Z>3)
75  // if higher energies are required, call this static method in order to recalculate the Emax limits
76  // in such way that:
77  // range function is always monotonically increasing function of Einc
78  // stopping power is concave (i.e. no minimum of stopping power followed by an increase)
79  // at the most, the new limit will be 1 GeV/nucleon.
80  // at the least, it will remain at the nominal (400 or 250 MeV/nucleon) level.
81  fNoLimits = on;
82  };

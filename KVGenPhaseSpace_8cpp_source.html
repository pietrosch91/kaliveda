214 Bool_t KVGenPhaseSpace::SetBreakUpChannel(const KVNucleus& CN, KVEvent* e)
215 {
216  // Define the break-up channel, i.e. the initial compound nucleus with its
217  // kinematics & excitation energy, and a list of nuclei to be produced
218  // by the decay. Nuclei produced by decay may be excited.
219  //
220  // Returns kTRUE if decay is energetically allowed, kFALSE if not.
221 
222  fEvent = e;
223  fCompound = CN;
224  fMult = e->GetMult();
225 
226  fOK = CheckBreakUpChannel();
227  if (!fOK) return fOK;
228 
229  if (fMasses) delete [] fMasses;
230  fMasses = new Double_t[fMult];
231  fEvent->GetMasses(fMasses);
232 
233  InitialiseMCSampler();
234 
235  return fOK;
236 }

251 Double_t KVGenPhaseSpace::Generate()
252 {
253  // Generate 1 event for this break-up channel.
254  // The weight associated to the event is returned.
255  // We also store the value in the list of parameters of the
256  // KVEvent object pointed to by fEvent:
257  // fEvent->GetParameters()->SetValue("PHASESPACE_WEIGHT", weight);
258  //
259  // NB: in order to get e.g. correct kinetic energy distributions for particles,
260  // this weight MUST be used as e.g. the filling weight for any histograms
261  // (see example in Class Description)
262 
263  if (!fOK) {
264  Warning("Generate", "Generator is not initialised correctly.");
265  return 0.;
266  }
267  Double_t weight = ((TGenPhaseSpace*)fMCSampler)->Generate();
268  fEvent->GetParameters()->SetValue("PHASESPACE_WEIGHT", weight);
269  for (int i = 0; i < fMult; i++) {
270  fEvent->GetParticle(i + 1)->Set4Mom(*(((TGenPhaseSpace*)fMCSampler)->GetDecay(i)));
271  }
272  return weight;
273 }

283 Bool_t KVGenPhaseSpace::CheckBreakUpChannel()
284 {
285  // We check that:
286  // - mass and charge are conserved
287  // - the excitation energy of the compound is greater than the channel Q-value
288  // (taking into account any excitation energy of nuclei in exit channel)
289 
290  Int_t ztot = (Int_t)fEvent->GetSum("GetZ");
291  Int_t atot = (Int_t)fEvent->GetSum("GetA");
292  if (ztot != fCompound.GetZ() || atot != fCompound.GetA()) {
293  Warning("CheckBreakUpChannel", "Compound has (Z,A)=(%d,%d), break-up channel has (Z,A)=(%d,%d)",
294  fCompound.GetZ(), fCompound.GetA(), ztot, atot);
295  return kFALSE;
296  }
297 
298  Double_t exmin = -fEvent->GetChannelQValue();
299  fEtot = fCompound.GetExcitEnergy() - exmin;
300  if (fEtot < 0.) {
301  Warning("CheckBreakUpChannel", "Excitation energy of compound must be > %.2lf MeV", exmin);
302  return kFALSE;
303  }
304 
305  return kTRUE;
306 }

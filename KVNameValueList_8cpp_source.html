181 void KVNameValueList::ClearSelection(TRegexp& sel)
182 {
183  // Remove from list all parameters whose name matches the regular expression
184  // Examples:
185  // remove all parameters starting with "toto": TRegexp sel("^toto")
186  // remove all parameters with "toto" in name: TRegexp sel("toto")
187 
188  TList toBeRemoved;
189  Int_t np1 = GetNpar();
190  for (Int_t ii = 0; ii < np1; ii += 1) {
191  TString name = GetParameter(ii)->GetName();
192  if (name.Contains(sel)) toBeRemoved.Add(new TNamed(name.Data(), ""));
193  }
194  if (toBeRemoved.GetEntries()) {
195  TIter next(&toBeRemoved);
196  TNamed* tbr;
197  while ((tbr = (TNamed*)next())) RemoveParameter(tbr->GetName());
198  toBeRemoved.Delete();
199  }
200 }

465 void KVNameValueList::ReadEnvFile(const Char_t* filename)
466 {
467  // Read all name-value pairs in the TEnv format file and store in list.
468  // Clears any previously stored values.
469  //
470  // values are read as strings from the TEnv and we use
471  // TString::IsDigit, TString::IsFloat to decide whether to store
472  // them as integers, floats, or strings.
473  // booleans are recognized as: TRUE, FALSE, ON, OFF, YES, NO, OK, NOT
474  // (to disable this feature and read such values as strings, call
475  // SetIgnoreBool(kTRUE))
476  //
477  // Special case:
478  // if the parameter name contains the string NumberList
479  // then we store the value string as is, as in this case
480  // it is assumed to be the string representation of a
481  // KVNumberList (easily confused with floating point numbers)
482 
483  Clear();
484  KVEnv env_file;
485  Int_t status = env_file.ReadFile(filename, kEnvAll);
486  if (status == -1) {
487  Error("ReadEnvFile", "The file %s does not exist", filename);
488  return;
489  }
490  THashList* name_value_list = env_file.GetTable();
491  TIter next_nv(name_value_list);
492  TEnvRec* nv_pair;
493  while ((nv_pair = (TEnvRec*)next_nv())) {
494  TString parname(nv_pair->GetName());
495  if (parname == "KVNameValueList.Name") SetName(nv_pair->GetValue());
496  else if (parname == "KVNameValueList.Title") SetTitle(nv_pair->GetValue());
497  else if (parname.Contains("NumberList")) SetValue(parname, nv_pair->GetValue());
498  else {
499  TString parval(nv_pair->GetValue());
500  if (parval.IsDigit()) SetValue(parname, parval.Atoi());
501  else if (parval.IsFloat()) SetValue(parname, parval.Atof());
502  else {
503  TString PARVAL(parval);
504  PARVAL.ToUpper();
505  if (!fIgnoreBool && (PARVAL == "TRUE" || PARVAL == "FALSE" || PARVAL == "ON" || PARVAL == "OFF"
506  || PARVAL == "YES" || PARVAL == "NO" || PARVAL == "OK" || PARVAL == "NOT"))
507  SetValue(parname, (Bool_t)env_file.GetValue(parname, 0));
508  else SetValue(parname, parval);
509  }
510  }
511  }
512 }

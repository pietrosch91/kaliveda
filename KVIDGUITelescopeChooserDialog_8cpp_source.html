258 void KVIDGUITelescopeChooserDialog::HighlightSelectedTelescopes()
259 {
260  // Modify widgets to reflect the ID telescopes selected in the list fSelection
261 
262  // Get 'type' of first telescope
263  KVIDTelescope* id = (KVIDTelescope*)fSelection->At(0);
264  if (!id) return;
265  // look for 'type' in list of types
266  TGLBEntry* lbe = fComboBox994->FindEntry(id->GetLabel());
267  if (!lbe) {
268  cout << "<KVIDGUITelescopeChooserDialog::HighlightSelectedTelescopes()> : Type "
269  << id->GetLabel() << " not found in combo box" << endl;
270  return;
271  }
272  // make 'type' be selected entry
273  fComboBox994->Select(lbe->EntryId());
274  FillTelescopeListWithType(id->GetLabel());
275  // now 'select' telescopes
276  TIter next_tel(fSelection);
277  int first_tel = -1;
278  while ((id = (KVIDTelescope*)next_tel())) {
279  lbe = fListBox980->FindEntry(id->GetName());
280  if (lbe) {
281  if (first_tel < 0) first_tel = lbe->EntryId();
282  fListBox980->Select(lbe->EntryId());
283  }
284  }
285  // scroll down to first selected telescope
286  fListBox980->SetTopEntry(first_tel);
287 }

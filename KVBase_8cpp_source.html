58 <h2>KVBase</h2>
59 <h4>Base class for KaliVeda framework</h4>
60 This is the base class for many classes in the KaliVeda framework. Each
61 KVBase object has:
62 
63  - a name - Get/SetName()
64  - a type - Get/SetType()
65  - a number - Get/SetNumber()
66  - a label - Get/SetLabel()
67 
68 When objects are accessed through a TObject/TNamed base pointer, it is possible
69 to test whether an object is derived from KVBase, using the bit
70 `KVBase::kIsKaliVedaObject`:
71 
72  TObject* ob = (address of some object)
73  if( ob->TestBit( KVBase::kIsKaliVedaObject ) ){
74 
75 This class also provides a number of general utilities, often as static
76 (stand-alone) methods.
77

79

81 
82  KVBase::GetKVSourceDir()
83  KVBase::GetKVBuildDate()
84  KVBase::GetKVBuildType()
85  KVBase::GetKVBuildUser()
86  KVBase::GetKVVersion()
87 
88 give info on the sources, when and where they were built, and by whom.
89

92 The entire KaliVeda framework is initialised by the static method KVBase::InitEnvironment()

96

98 KaliVeda installation tree are given:
99 
100  KVBase::SearchKVFile(...)
101  KVBase::SearchAndOpenKVFile(...)
102

405 void KVBase::InitEnvironment()
406 {
407  // STATIC method to Initialise KaliVeda environment
408  // Reads config files in <code>\$(pkgdatadir)/etc</code> and sets up environment
409  // (data repositories, datasets, etc. etc.).
410  // Adds directory where kaliveda shared libs are installed to the dynamic
411  // path - for finding and loading plugins.
412  // Resets the `gRandom` random number sequence using a clock-based seed
413  // (i.e. random sequences do not repeat).
414  //
415  // Normally, the first object created which inherits from KVBase will
416  // perform this initialisation; if you need to set up the environment before
417  // creating a KVBase object, or if you just want to be absolutely sure that
418  // the environment has been initialised, you can call this method.
419  //
420  // #### Note for GNU-style installations
421  // If KaliVeda is built with the cmake option `-Dgnuinstall=yes` then each
422  // user will have a working directory which will be used to store any files
423  // generated by KaliVeda at runtime.
424  //
425  // By default the location of user's working directory is set to
426  //~~~~~~~~~
427  // $(HOME)/.kaliveda
428  //~~~~~~~~~
429  // but can be changed with variable
430  //~~~~~~~~~
431  // KaliVeda.WorkingDirectory: [directory]
432  //~~~~~~~~~
433  // in configuration file `.kvrootrc`.
434  //`[directory]` must be an absolute pathname, but can use shell variables like <code>\$(HOME)</code>.
435 
436  if (!fEnvIsInit) {//test if environment already initialised
437 
438  // Add path to kaliveda libraries to dynamic loader path
439  // This is needed to find plugins
440  // and also to be able to compile with kaliveda in the interpreter
441  TString libdir = GetLIBDIRFilePath();
442  gSystem->AddDynamicPath(libdir);
443  // force re-reading of rootmap files in new dynamic path
444  gInterpreter->LoadLibraryMap();
445  // Add path to kaliveda header files
446  // This is needed to be able to compile with kaliveda in the interpreter
447  TString incdir = GetINCDIRFilePath();
448  incdir.Prepend("-I");
449  gSystem->AddIncludePath(incdir);
450 
451  //set up environment using kvrootrc file
452  if (!gEnv->Defined("DataSet.DatabaseFile")) {
453  ReadConfigFiles();
454  }
455 #ifdef WITH_GNU_INSTALL
456  // set working directory & create if needed
457  fWorkingDirectory = gEnv->GetValue("KaliVeda.WorkingDirectory", "$(HOME)/.kaliveda");
458  gSystem->ExpandPathName(fWorkingDirectory);
459  gSystem->mkdir(fWorkingDirectory, kTRUE);
460 #else
461  // set environment variable used in database makefiles
462  fWorkingDirectory = KV_ROOT;
463 #endif
464  // set environment variable used in database makefiles
465  gSystem->Setenv("KV_WORK_DIR", fWorkingDirectory);
466 
467  //generate new seed from system clock
468  gRandom->SetSeed(0);
469 
470  // initialisation has been performed
471  fEnvIsInit = kTRUE;
472 
473  }
474 }

486 void KVBase::ReadConfigFiles()
487 {
488  // Read all configuration files
489  // System config files are read first in the order they appear in file
490  // ${ETCDIR}/config.files
491  // Then we read any of the following files if they exist:
492  // ${HOME}/.kvrootrc
493  // ${PWD}/.kvrootrc
494 
495  TString tmp = GetETCDIRFilePath("config.files");
496  ifstream conflist;
497  conflist.open(tmp.Data());
498  if (!conflist.good()) {
499  ::Fatal("KVBase::ReadConfigFiles", "Cannot open %s", tmp.Data());
500  return;
501  }
502  KVString file;
503  file.ReadLine(conflist);
504  conflist.close();
505  file.Begin(";");
506  while (!file.End()) {
507  tmp = GetETCDIRFilePath(file.Next().Data());
508  //skip over any missing files - this is needed when installing from
509  //e.g. ubuntu packages if not all packages are installed
510  if (!gSystem->AccessPathName(tmp.Data())) gEnv->ReadFile(tmp.Data(), kEnvChange);
511  }
512 
513  AssignAndDelete(tmp, gSystem->ConcatFileName(gSystem->Getenv("HOME"), ".kvrootrc"));
514  gEnv->ReadFile(tmp.Data(), kEnvChange);
515 
516  tmp = "./.kvrootrc";
517  gEnv->ReadFile(tmp.Data(), kEnvChange);
518 
519  // load plugin handlers
520  gROOT->GetPluginManager()->LoadHandlersFromEnv(gEnv);
521 
522  // load mime types/icon definitions when not in batch (i.e. GUI-less) mode
523  if (!gROOT->IsBatch()) ReadGUIMimeTypes();
524 }

627 void KVBase::Streamer(TBuffer& R__b)
628 {
629  //Backwards compatible streamer for KVBase objects
630  //Needed to handle 'fLabel' char array in class version 1
631  //Objects written with version < 3 did not have kIsKaliVedaObject bit set,
632  //we set it here when reading object.
633 
634  if (R__b.IsReading()) {
635  UInt_t R__s, R__c;
636  Version_t R__v = R__b.ReadVersion(&R__s, &R__c);
637  if (R__v > 1) {
638  if (R__v < 4) {
639  TNamed::Streamer(R__b);
640  R__b >> fNumber;
641  R__b >> fLabel;
642  if (R__v < 3) SetBit(kIsKaliVedaObject);
643  R__b.CheckByteCount(R__s, R__c, KVBase::IsA());
644  } else {
645  //AUTOMATIC STREAMER EVOLUTION FOR CLASS VERSION > 3
646  KVBase::Class()->ReadBuffer(R__b, this, R__v, R__s, R__c);
647  }
648  return;
649  }
650  //OLD STREAMER FOR CLASS VERSION 1
651  TNamed::Streamer(R__b);
652  R__b >> fNumber;
653  UInt_t LabelLength;
654  R__b >> LabelLength;
655  if (LabelLength) {
656  Char_t* Label = new Char_t[LabelLength];
657  R__b.ReadFastArray(Label, LabelLength);
658  fLabel = Label;
659  delete[]Label;
660  }
661  SetBit(kIsKaliVedaObject);
662  R__b.CheckByteCount(R__s, R__c, KVBase::IsA());
663  } else {
664  KVBase::Class()->WriteBuffer(R__b, this);
665  }
666 }

683 Bool_t SearchFile(const Char_t* name, TString& fullpath, int ndirs, ...)
684 {
685  //Search for file in an arbitrary number of locations, return kTRUE if file found and put full path to file in 'fullpath"
686  //
687  //'name' is a filename (not an absolute pathname) i.e. "toto.dat"
688  //'fullpath" will contain the full path to the file if it is found (if file not found, fullpath="")
689  //'ndirs" is the number of directories to search in
690  //the remaining arguments are the names of 'ndirs' paths to search in, i.e.
691  //
692  // SearchFile("toto.dat", fullpath, 2, gSystem->pwd(), gSystem->HomeDirectory());
693  //
694  //means: search for a file 'toto.dat' in current working directory, then user's home directory.
695 
696  if (ndirs <= 0)
697  return kFALSE;
698 
699  va_list args;
700  va_start(args, ndirs);
701 
702  for (; ndirs; ndirs--) { //loop over directories
703 
704  AssignAndDelete(fullpath,
705  gSystem->ConcatFileName(va_arg(args, const char*),
706  name));
707  if (!gSystem->AccessPathName(fullpath.Data())) {
708  va_end(args);
709  return kTRUE;
710  }
711 
712  }
713 
714  va_end(args);
715  fullpath = ""; //clear fullpath string to avoid using it by mistake
716  return kFALSE;
717 }

732 Bool_t KVBase::SearchKVFile(const Char_t* name, TString& fullpath,
733  const Char_t* kvsubdir)
734 {
735  //search for files in the following order:
736  // if 'name' = absolute path the function returns kTRUE if the file exists
737  // if name != absolute path:
738  // 1. a. if 'kvsubdir'="" (default) look for file in $(pkgdatadir) directory
739  // 1. b. if 'kvsubdir'!="" look for file in $(pkgdatadir)/'kvsubdir'
740  // 2. look for file with this name in user's home directory
741  // 3. look for file with this name in working directory
742  //in all cases the function returns kTRUE if the file was found.
743  //'fullpath' then contains the absolute path to the file
744 
745  if (gSystem->IsAbsoluteFileName(name)) {
746  //absolute path
747  fullpath = name;
748  return !gSystem->AccessPathName(name);
749  }
750 
751  TString kvfile_dir;
752  if (strcmp(kvsubdir, "")) {
753  //subdirectory name given
754  kvfile_dir = GetDATADIRFilePath(kvsubdir);
755  } else
756  kvfile_dir = GetDATADIRFilePath();
757  return SearchFile(name, fullpath, 3, kvfile_dir.Data(),
758  gSystem->HomeDirectory(), gSystem->pwd());
759 }

781 Bool_t KVBase::SearchAndOpenKVFile(const Char_t* name, ifstream& file, const Char_t* kvsubdir, KVLockfile* locks)
782 {
783  //Search and open for READING a file:
784  //
785  //search for ascii file (and open it, if found) in the following order:
786  // if 'name' = absolute path the function returns kTRUE if the file exists
787  // if name != absolute path:
788  // 1. a. if 'kvsubdir'="" (default) look for file in $(pkdatadir) directory
789  // 1. b. if 'kvsubdir'!="" look for file in $(pkdatadir)/'kvsubdir'
790  // 2. look for file with this name in user's home directory
791  // 3. look for file with this name in working directory
792  //if the file is not found, kFALSE is returned.
793  //if file is found and can be opened, file' is then an ifstream connected to the open (ascii) file
794  //
795  //LOCKFILE:
796  //If a KVLockfile pointer is given, we use it to get a lock on the file before opening it.
797  //If this lock is not successful, the file is not opened and we return an error message.
798 
799  TString fullpath;
800  if (SearchKVFile(name, fullpath, kvsubdir)) {
801  //put lock on file if required
802  if (locks && !locks->Lock(fullpath.Data())) return kFALSE;
803  file.open(fullpath.Data());
804  if (file.good()) {
805  //cout << "Opened file : " << fullpath.Data() << endl;
806  return kTRUE;
807  }
808  //unlock file if not opened successfully
809  if (locks) locks->Release();
810  }
811  return kFALSE;
812 }

835 Bool_t KVBase::SearchAndOpenKVFile(const Char_t* name, ofstream& file, const Char_t* kvsubdir, KVLockfile* locks)
836 {
837  //Search and CREATE i.e. open for WRITING a file:
838  //
839  //open for writing an ascii file in the location determined in the following way:
840  // if 'name' = absolute path we use the full path
841  // if name != absolute path:
842  // 1. a. if 'kvsubdir'="" (default) file will be in $(pkdatadir) directory
843  // 1. b. if 'kvsubdir'!="":
844  // if 'kvsubdir' is an absolute pathname, file in 'kvsubdir'
845  // if 'kvsubdir' is not an absolute pathname,
846  // file will be in '$(pkdatadir)/kvsubdir'
847  //if an existing file is found, a warning is printed and the existing file 'toto' is renamed
848  //"toto.date". where 'date' is created with TDatime::AsSQLDate
849  // file' is then an ofstream connected to the opened file
850  //
851  //LOCKFILE:
852  //If a KVLockfile pointer is given, we use it to get a lock on the file before opening it.
853  //If this lock is not successful, the file is not opened and we return an error message.
854 
855  KVString fullpath;
856  if (gSystem->IsAbsoluteFileName(name)) {
857  fullpath = name;
858  } else if (gSystem->IsAbsoluteFileName(kvsubdir)) {
859  AssignAndDelete(fullpath,
860  gSystem->ConcatFileName(kvsubdir, name));
861  } else if (strcmp(kvsubdir, "")) {
862  KVString path = GetDATADIRFilePath(kvsubdir);
863  AssignAndDelete(fullpath,
864  gSystem->ConcatFileName(path.Data(), name));
865  } else {
866  fullpath = GetDATADIRFilePath(name);
867  }
868  //Backup file if necessary
869  BackupFileWithDate(fullpath.Data());
870  //put lock on file if required
871  if (locks && !locks->Lock(fullpath.Data())) return kFALSE;
872  file.open(fullpath.Data());
873  return kTRUE;
874 }

890 void KVBase::BackupFileWithDate(const Char_t* path)
891 {
892  //`path` gives the full path (can include environment variables, special symbols)
893  //to a file which will be renamed with an extension containing the current date and time
894  //(in SQL format).
895  //
896  //Example:
897  //
898  // KVBase::BackupFileWithDate("$(HOME)/toto.txt")
899  //
900  //The file `toto.txt` will be renamed `toto.txt.2007-05-02_16:22:37`
901 
902  KVString fullpath = path;
903  gSystem->ExpandPathName(fullpath);
904  if (!gSystem->AccessPathName(fullpath.Data())) {//does the file exist ?
905  //backup file
906  TDatime now;
907  KVString date(now.AsSQLString());
908  date.ReplaceAll(' ', '_');
909  TString backup = fullpath + "." + date;
910  //lock both files
911  KVLockfile lf1(fullpath.Data()), lf2(backup.Data());
912  if (lf1.Lock() && lf2.Lock()) {
913  gSystem->Rename(fullpath.Data(), backup.Data());
914  printf("Info in <KVBase::BackupFileWithDate(const Char_t *)> : Existing file %s renamed %s\n",
915  fullpath.Data(), backup.Data());
916  }
917  }
918 }

1193 Bool_t KVBase::FindExecutable(TString& exec, const Char_t* path)
1194 {
1195  //By default, `FindExecutable(exec)` will look for the executable named by `exec`
1196  //in the directories contained in the environment variable `PATH`. You can override
1197  //this by giving your own search `path` as second argument (remember to write
1198  //environment variables as <code>\$(PATH)</code>, for cross-platform compatibility).
1199  //
1200  //If `exec` is not found, and if it does not end with `.exe`, we look for `exec.exe`
1201  //This is for compatibility with Windows/cygwin environments.
1202  //
1203  //If the executable is found, returns `kTRUE` and `exec` then holds full path to executable.
1204  //Returns `kFALSE` if `exec` not found in paths.
1205  //
1206  //If `exec` is an absolute pathname, we return `kTRUE` if the file exists
1207  //(we do not use `path`).
1208 
1209  TString spath(path), backup(exec), backup2(exec), expandexec(exec);
1210  gSystem->ExpandPathName(expandexec);
1211  if (gSystem->IsAbsoluteFileName(expandexec.Data())) {
1212  //executable given as absolute path
1213  //we check if it exists
1214  if (!gSystem->AccessPathName(expandexec)) {
1215  exec = expandexec;
1216  return kTRUE;
1217  } else {
1218  //try with ".exe" in case of Windows system
1219  if (!expandexec.EndsWith(".exe")) {
1220  expandexec += ".exe";
1221  if (!gSystem->AccessPathName(expandexec)) {
1222  exec = expandexec;
1223  return kTRUE;
1224  }
1225  }
1226  }
1227  exec = backup;
1228  return kFALSE;
1229  }
1230  gSystem->ExpandPathName(spath);
1231  if (KVBase::FindFile(spath.Data(), exec))
1232  return kTRUE;
1233  if (!backup.EndsWith(".exe")) {
1234  backup += ".exe";
1235  if (KVBase::FindFile(spath.Data(), backup)) {
1236  exec = backup;
1237  return kTRUE;
1238  }
1239  }
1240  exec = backup2;
1241  return kFALSE;

1368 {

1370  //
1371  //Given a base class `base` and a derived class `derived`, we search `gEnv` to find the
1372  //URI corresponding to this plugin.
1373  //
1374  //Example: given a plugin such as
1375  //
1376  //~~~~~~~~
1377  //Plugin.KVIDTelescope: ^PHOS$ KVIDPhoswich KVIndra "KVIDPhoswich()"
1378  //~~~~~~~~
1379  //
1380  //then calling `KVBase::GetPluginURI("KVIDTelescope", "KVIDPhoswich")` will return `"PHOS"`.
1381  //
1382  //Most of the code is copied from `TPluginManager::LoadHandlersFromEnv`
1383 
1384  TIter next(gEnv->GetTable());
1385  TEnvRec* er;
1386  static TString tmp;
1387 
1388  while ((er = (TEnvRec*) next())) {
1389  const char* s;
1390  if ((s = strstr(er->GetName(), "Plugin."))) {
1391  // use s, i.e. skip possible OS and application prefix to Plugin.
1392  // so that GetValue() takes properly care of returning the value
1393  // for the specified OS and/or application
1394  const char* val = gEnv->GetValue(s, (const char*)0);
1395  if (val) {
1396  Int_t cnt = 0;
1397  s += 7;
1398  //is it the right base class ?
1399  if (strcmp(s, base)) continue; //skip to next env var if not right base
1400 
1401  char* v = StrDup(val);
1402  while (1) {
1403  TString regexp = strtok(!cnt ? v : 0, "; ");
1404  if (regexp.IsNull()) break;
1405  TString clss = strtok(0, "; ");
1406  if (clss.IsNull()) break;
1407  TString plugin = strtok(0, "; ");
1408  if (plugin.IsNull()) break;
1409  TString ctor = strtok(0, ";\"");
1410  if (!ctor.Contains("("))
1411  ctor = strtok(0, ";\"");
1412  if (clss == derived) {
1413  //found the required plugin
1414  //we remove the 'regexp' operator '^' from the beginning
1415  //and '$' from the end of the URI, if necessary
1416  if (regexp.MaybeRegexp()) {
1417  regexp.Remove(TString::kBoth, '^');
1418  regexp.Remove(TString::kBoth, '$');
1419  }
1420  tmp = regexp;
1421  delete [] v;
1422  return tmp.Data();
1423  }
1424  cnt++;
1425  }
1426  delete [] v;
1427  }
1428  }
1429  }
1430  tmp = "";
1431  return tmp;
1432 }

1581 Bool_t KVBase::AreEqual(Double_t A, Double_t B, Long64_t maxdif)
1582 {
1583  // \brief Comparison between two 64-bit floating-point values
1584  //
1585  // Returns `kTRUE` if the integer representations of the two values are within
1586  // `maxdif` of each other. By default `maxdif=1`, which means that we consider that `x==y` if the
1587  // difference between them is no greater than the precision of `Double_t`
1588  // variables, i.e. `4.94065645841246544e-324`
1589  //
1590  // Based on the function `AlmostEqual2sComplement(float, float, int)`
1591  // by Bruce Dawson http://www.cygnus-software.com/papers/comparingfloats/comparingfloats.htm
1592 
1593  union converter {
1594  Double_t f;
1595  Long64_t i;
1596  } zero, val1, val2;
1597 
1598  assert(maxdif > 0);
1599 
1600  if (A == B) return true;
1601 
1602  /* rustine to obtain the (64-bit) constant value 0x8000000000000000
1603  even on 32-bit machines (there is probably an easier way!) */
1604  zero.i = 1;
1605  zero.f = -zero.f;
1606  zero.i -= 1;
1607 
1608  val1.f = A;
1609  val2.f = B;
1610  Long64_t Aint, Bint;
1611  Aint = val1.i;
1612  Bint = val2.i;
1613  if (Aint < 0) Aint = zero.i - Aint;
1614  if (Bint < 0) Bint = zero.i - Bint;
1615 
1616  Long64_t intDiff = abs(val1.i - val2.i);
1617 
1618  if (intDiff <= maxdif) return true;
1619 
1620  return false;
1621 }

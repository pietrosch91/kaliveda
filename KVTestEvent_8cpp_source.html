32 //KVTestEvent
33 //
34 //A simple event generator for testing charged particle array response.
35 //Each event consists of N KVNucleus nuclei with randomly drawn atomic number
36 //Z, kinetic energy and direction.
37 //
38 //To use, first set the multiplicity:
39 // KVTestEvent::KVTestEvent(UInt_t n, Option_t * t)
40 //or KVTestEvent::SetMult(UInt_t mult) ... after default ctor
41 //
42 //Then set the range in Z and the range in energy for the generated nuclei:
43 //KVTestEvent::SetZRange(UInt_t zlo, UInt_t zhi)
44 //KVTestEvent::SetERange(Float_t elo, Float_t ehi) (in MeV)
45 //or KVTestEvent::SetERangeAMeV with limits in MeV/nucleon
46 //
47 //The masses of the nuclei are calculated from the Z assuming the valley-
48 //of-stability isotope.
49 //
50 //By default angular ranges cover the full 4pi range, but this can be changed
51 //with KVTestEvent::SetThetaRange and KVTestEvent::SetPhiRange.
52 //Also by default directions are drawn isotropically within the set angular
53 //ranges (Option_t *t="isotropic") but by choosing option "random" this can be
54 //replaced with a flat angular distribution (see KVPosition::GetRandomDirection)

240 void KVTestEvent::Generate()
241 {
242 //
243 //Generate an event based on given parameters.
244 //Before using, at least set multiplicity (using constructor with arguments or
245 //KVTestEvent::SetMult), range in Z (KVTestEvent::SetZRange) and range
246 //in energy (KVTestEvent::SetERange or KVTestEvent::SetERangeAMeV).
247 //
248 
249  for (UInt_t i = 1; i <= fGenMult; i++) {
250  UInt_t z =
251  gRandom->Integer(fUpper_Limit_Z - fLower_Limit_Z + 1) +
252  fLower_Limit_Z;
253  Float_t ke =
254  gRandom->Uniform(fUpper_Limit_E - fLower_Limit_E) +
255  fLower_Limit_E;
256  KVNucleus* kvn = AddParticle();
257  kvn->SetZ(z);
258  if (kAMeV)
259  ke *= kvn->GetA();
260  kvn->SetRandomMomentum(ke, fLower_Limit_theta, fUpper_Limit_theta,
261  fLower_Limit_phi, fUpper_Limit_phi, fOption);
262  }
263 }

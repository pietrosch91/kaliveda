13 // //
14 // KVFileDialog //
15 // //
16 // This class creates a file selection dialog. It contains a combo box //
17 // to select the desired directory. A listview with the different //
18 // files in the current directory and a combo box with which you can //
19 // select a filter (on file extensions). //
20 // When creating a file dialog one passes a pointer to a TGFileInfo //
21 // object. In this object you can set the fFileTypes and fIniDir to //
22 // specify the list of file types for the filter combo box and the //
23 // initial directory. When the KVFileDialog ctor returns the selected //
24 // file name can be found in the TGFileInfo::fFilename field and the //
25 // selected directory in TGFileInfo::fIniDir. The fFilename and //
26 // fIniDir are deleted by the TGFileInfo dtor. //
27 // //

366 Bool_t KVFileDialog::ProcessMessage(Long_t msg, Long_t parm1, Long_t)
367 {
368  // Process messages generated by the user input in the file dialog.
369 
370  if (!fFc->GetDisplayStat()) return kTRUE; // Cancel button was pressed
371 
372  TGTreeLBEntry* e;
373  TGTextLBEntry* te;
374  TGFileItem* f;
375  void* p = 0;
376  TString txt;
377  TString sdir = gSystem->WorkingDirectory();
378 
379  switch (GET_MSG(msg)) {
380  case kC_COMMAND:
381  switch (GET_SUBMSG(msg)) {
382  case kCM_BUTTON:
383  switch (parm1) {
384  case kIDF_OK:
385  // same code as under kTE_ENTER
386  if ((fDlgMode != kKVFDDirectory) && (fTbfname->GetTextLength() == 0)) {
387  txt = "Please provide file name or use \"Cancel\"";
388  new TGMsgBox(fClient->GetRoot(), GetMainFrame(),
389  "Missing File Name", txt, kMBIconExclamation,
390  kMBOk);
391  return kTRUE;
392  } else if (!gSystem->AccessPathName(fTbfname->GetString(), kFileExists) &&
393  !strcmp(fOk->GetTitle(), "Save") &&
394  (!(fCheckB->GetState() == kButtonDown))) {
395  Int_t ret;
396  txt = TString::Format("File name %s already exists, OK to overwrite it?",
397  fTbfname->GetString());
398  new TGMsgBox(fClient->GetRoot(), GetMainFrame(),
399  "File Name Exist", txt.Data(), kMBIconExclamation,
400  kMBYes | kMBNo, &ret);
401  if (ret == kMBNo)
402  return kTRUE;
403  }
404  if (fFileInfo->fMultipleSelection) {
405  if (fFileInfo->fFilename)
406  delete [] fFileInfo->fFilename;
407  fFileInfo->fFilename = 0;
408  } else {
409  if (fFileInfo->fFilename)
410  delete [] fFileInfo->fFilename;
411  if (gSystem->IsAbsoluteFileName(fTbfname->GetString()))
412  fFileInfo->fFilename = StrDup(fTbfname->GetString());
413  else
414  fFileInfo->fFilename = gSystem->ConcatFileName(fFc->GetDirectory(),
415  fTbfname->GetString());
416  fFileInfo->fFilename = StrDup(gSystem->UnixPathName(gSystem->ExpandPathName(fFileInfo->fFilename)));
417  }
418  if (fCheckB && (fCheckB->GetState() == kButtonDown))
419  fFileInfo->fOverwrite = kTRUE;
420  else
421  fFileInfo->fOverwrite = kFALSE;
422  DeleteWindow();
423  break;
424 
425  case kIDF_CANCEL:
426  if (fFileInfo->fFilename)
427  delete [] fFileInfo->fFilename;
428  fFileInfo->fFilename = 0;
429  if (fFc->GetDisplayStat())
430  fFc->SetDisplayStat(kFALSE);
431  if (fFileInfo->fFileNamesList != 0) {
432  fFileInfo->fFileNamesList->Delete();
433  fFileInfo->fFileNamesList = 0;
434  }
435  DeleteWindow();
436  return kTRUE; //no need to redraw fFc
437  break;
438 
439  case kIDF_CDUP:
440  fFc->ChangeDirectory("..");
441  fTreeLB->Update(fFc->GetDirectory());
442  if (fFileInfo->fIniDir) delete [] fFileInfo->fIniDir;
443  fFileInfo->fIniDir = StrDup(fFc->GetDirectory());
444  if (strcmp(gSystem->WorkingDirectory(), fFc->GetDirectory())) {
445  gSystem->cd(fFc->GetDirectory());
446  }
447  break;
448 
449  case kIDF_NEW_FOLDER: {
450  char answer[128];
451  strlcpy(answer, "(empty)", sizeof(answer));
452  new TGInputDialog(gClient->GetRoot(), GetMainFrame(),
453  "Enter directory name:",
454  answer/*"(empty)"*/, answer);
455 
456  while (strcmp(answer, "(empty)") == 0) {
457  new TGMsgBox(gClient->GetRoot(), GetMainFrame(), "Error",
458  "Please enter a valid directory name.",
459  kMBIconStop, kMBOk);
460  new TGInputDialog(gClient->GetRoot(), GetMainFrame(),
461  "Enter directory name:",
462  answer, answer);
463  }
464  if (strcmp(answer, "") == 0) // Cancel button was pressed
465  break;
466 
467  if (strcmp(gSystem->WorkingDirectory(), fFc->GetDirectory())) {
468  gSystem->cd(fFc->GetDirectory());
469  }
470  if (gSystem->MakeDirectory(answer) != 0)
471  new TGMsgBox(gClient->GetRoot(), GetMainFrame(), "Error",
472  TString::Format("Directory name \'%s\' already exists!", answer),
473  kMBIconStop, kMBOk);
474  else {
475  fFc->DisplayDirectory();
476  }
477  gSystem->ChangeDirectory(sdir.Data());
478  break;
479  }
480 
481  case kIDF_LIST:
482  fFv->SetViewMode(kLVList);
483  fDetails->SetState(kButtonUp);
484  break;
485 
486  case kIDF_DETAILS:
487  fFv->SetViewMode(kLVDetails);
488  fList->SetState(kButtonUp);
489  break;
490  }
491  break;
492 
493  case kCM_COMBOBOX:
494  switch (parm1) {
495  case kIDF_FSLB:
496  e = (TGTreeLBEntry*) fTreeLB->GetSelectedEntry();
497  if (e) {
498  fFc->ChangeDirectory(e->GetPath()->GetString());
499  fTreeLB->Update(fFc->GetDirectory());
500  if (fFileInfo->fIniDir) delete [] fFileInfo->fIniDir;
501  fFileInfo->fIniDir = StrDup(fFc->GetDirectory());
502  if (strcmp(gSystem->WorkingDirectory(), fFc->GetDirectory())) {
503  gSystem->cd(fFc->GetDirectory());
504  }
505  }
506  break;
507 
508  case kIDF_FTYPESLB:
509  te = (TGTextLBEntry*) fTypes->GetSelectedEntry();
510  if (te) {
511  //fTbfname->Clear();
512  //fTbfname->AddText(0, fFileInfo->fFileTypes[te->EntryId()+1]);
513  fFileInfo->fFileTypeIdx = te->EntryId();
514  fFc->SetFilter(fFileInfo->fFileTypes[fFileInfo->fFileTypeIdx + 1]);
515  fFc->DisplayDirectory();
516  fClient->NeedRedraw(fName);
517  }
518  break;
519  }
520  break;
521 
522  default:
523  break;
524  } // switch(GET_SUBMSG(msg))
525  break;
526 
527  case kC_CONTAINER:
528  switch (GET_SUBMSG(msg)) {
529  case kCT_ITEMCLICK:
530  if (parm1 == kButton1) {
531  if (fFc->NumSelected() > 0) {
532  if (fFileInfo->fMultipleSelection == kFALSE) {
533  TGLVEntry* e2 = (TGLVEntry*) fFc->GetNextSelected(&p);
534  if ((e2) && !R_ISDIR(((TGFileItem*)e2)->GetType())) {
535  fTbfname->Clear();
536  if (e2->GetItemName())
537  fTbfname->AddText(0, e2->GetItemName()->GetString());
538  fClient->NeedRedraw(fName);
539  }
540  } else {
541  TString tmpString;
542  TList* tmp = fFc->GetSelectedItems();
543  TObjString* el;
544  TIter next(tmp);
545  if (fFileInfo->fFileNamesList != 0) {
546  fFileInfo->fFileNamesList->Delete();
547  } else {
548  fFileInfo->fFileNamesList = new TList();
549  }
550  while ((el = (TObjString*) next())) {
551  char* s = gSystem->ConcatFileName(fFc->GetDirectory(),
552  el->GetString());
553  tmpString += "\"" + el->GetString() + "\" ";
554  fFileInfo->fFileNamesList->Add(new TObjString(s));
555  delete [] s;
556  }
557  fTbfname->Clear();
558  fTbfname->AddText(0, tmpString);
559  fClient->NeedRedraw(fName);
560  }
561  }
562  }
563  break;
564 
565  case kCT_ITEMDBLCLICK:
566 
567  if (parm1 == kButton1) {
568  if (fFc->NumSelected() == 1) {
569  f = (TGFileItem*) fFc->GetNextSelected(&p);
570  if (f && R_ISDIR(f->GetType())) {
571  fFc->ChangeDirectory(f->GetItemName()->GetString());
572  fTreeLB->Update(fFc->GetDirectory());
573  if (fFileInfo->fIniDir) delete [] fFileInfo->fIniDir;
574  fFileInfo->fIniDir = StrDup(fFc->GetDirectory());
575  if (strcmp(gSystem->WorkingDirectory(), fFc->GetDirectory())) {
576  gSystem->cd(fFc->GetDirectory());
577  }
578  } else {
579  if (!strcmp(fOk->GetTitle(), "Save") &&
580  (!(fCheckB->GetState() == kButtonDown))) {
581 
582  Int_t ret;
583  txt = TString::Format("File name %s already exists, OK to overwrite it?",
584  fTbfname->GetString());
585  new TGMsgBox(fClient->GetRoot(), GetMainFrame(),
586  "File Name Exist", txt.Data(), kMBIconExclamation,
587  kMBYes | kMBNo, &ret);
588  if (ret == kMBNo)
589  return kTRUE;
590  }
591  if (fFileInfo->fFilename)
592  delete [] fFileInfo->fFilename;
593  if (gSystem->IsAbsoluteFileName(fTbfname->GetString()))
594  fFileInfo->fFilename = StrDup(fTbfname->GetString());
595  else

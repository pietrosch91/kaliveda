201 void KVRelativeVelocity::Fill(KVNucleus* cc)
202 {
203 
204 //
205 // Filling method
206 //
207 //
208 // Use the FillVar(val) or the FillVar(val,w) method to increment the quantity
209 // of the global variable.
210 // The weight w is evaluated from the properties of the KVNucleus passed as an
211 // argument.
212 //
213 // For example, to evaluate the mean charge of all fragments,
214 // you may proceed as follows:
215 //
216 // FillVar(c->GetZ());
217 //
218 // Another example: to evaluate the mean parallel velocity weighted by the charge
219 // of the nucleus:
220 //
221 // FillVar(c->GetV().Z(),c->GetZ());
222 //
223  if (!heaviest) {
224  heaviest = new TList;
225  heaviest->SetOwner(kFALSE);
226  heaviest->Add(cc);
227  } else {
228  for (Int_t ii = 0; ii < heaviest->GetEntries(); ii += 1) {
229  TVector3 ww = cc->BoostVector() - ((KVNucleus*)heaviest->At(ii))->BoostVector();
230  FillVar(ww.Mag());
231  }
232  heaviest->Add(cc);
233  }

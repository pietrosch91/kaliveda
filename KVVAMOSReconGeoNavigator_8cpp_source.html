16 <h4>Propagate nuclei through the geometry of VAMOS for their reconstruction/calibrate/identification</h4>

19 // Given a valid ROOT geometry, we propagate the nuclei in the VAMOS focal-plane
20 // detection area and, every time a nucleus traverses a volume made of a TGeoMaterial
21 // with a name corresponding to a material known by this range table,
22 // we calculate the step through the material (STEP) and the energy losse of the nucleus,
23 // and store it in the nucleus's list KVParticle::fParameters in the form
24 //
25 // "STEP:[detector name]" = [step in material in cm]
26 // "DE:[detector name]" = [energy losse in material in MeV]
27 //
28 // In the case of multilayer detectors, "[detector name]" in the previous
29 // examples is replaced by "[detector name]/[layer name]"
30 //
31 //
32 // For the calibration of the times of flight, we also store the distance (DPATH in cm) between the intersection point at the focal plane
33 // and the point at the entrance of the volume if it is the first active volume of
34 // a detector
35 //
36 // "DPATH:[detector name]" = [distance FP point to active volume entrance point]
37 //
38 // DPATH has the sign + if the volume is behind the focal plane or - is it
39 // is at the front of it.
40 //
41 // The time of flight between the center of the target and the point at the entrance of the first active volume of each detector
42 //
43 // "TOF:[detector name]" = [time of flight from target to active volume in cm/ns]
44 //
45 // This class is used for the reconstruction in KVVAMOSReconNuc. It can be used
46 // also for the calibration (time and energy) of detectors.
47 // The propagation of the nucleus is first done from the focal-plane point
48 // toward the inverse direction of the reconstructed trajectory (KVVAMOSReconNuc::fRT.dirFP direction) to found the starting point.
49 // Then a second propagation is done from this starting point, in the in the inverse direction
50 // to perform the calculations.
51 // If the trajectory is not correctly reconstructed then a central trajectory
52 // is used (i.e. Theta = Phi = ThetaF = PhiF = 0 degree and Xf = Yf = 0 cm ).
53 // In this case a warning will be displayed.
54 //
55 // DPATH is used to calculate the correct flight distance of the nucleus.
56 // For example if the time TSED1_HF is used for a time-of-flight
57 // measurement then the real flight distance is the reconstructed path, from
58 // the target point to the focal plane, plus the value of DPATH:SED1/ACTIVE_SED2_Myl
59 //
60 //
61 // EXAMPLE
62 // =======
63 //
64 // In this case the VAMOS geometry is made up of 2 SeDs, 1 HARPEE ionization chamber and 1 silicon detector.
65 // The ionization chamber has 4 layers (window + gas + gas + gas), with an "ACTIVE" gas layer
66 //
67 // For a well reconstructed nucleus in VAMOS, this navigator is used and then
68 // you can access to the result of the propagation with:
69 //
70 // nuc->GetParameters()->Print()
71 //
72 // KVNameValueList::ParticleParameters : Parameters associated with a particle in an event (0xa163800)
73 // <STEP:SED1/ACTIVE_SED1_Myl=0.000126196>
74 // <DPATH:SED1/ACTIVE_SED1_Myl=-71.6663>
75 // <STEP:SED2/ACTIVE_SED2_Myl=0.000126196>
76 // <DPATH:SED2/ACTIVE_SED2_Myl=16.0801>
77 // <STEP:CHI/CHI_Myl=0.000150153>
78 // <STEP:CHI/CHI_C4H10=2.07011>
79 // <STEP:CHI/ACTIVE_CHI_C4H10=10.4677>
80 // <DPATH:CHI/ACTIVE_CHI_C4H10=56.8245>
81 // <STEP:CHI/CHI_C4H10_1=1.27931>
82 // <STEP:SI_17=0.0500511>
83 // <DPATH:SI_17=68.575>
84 //
85 // here 'nuc' is a KVVAMOSReconNuc object.

201 void KVVAMOSReconGeoNavigator::ParticleEntersNewVolume(KVNucleus* nuc)
202 {
203  // Overrides method in KVGeoNavigator base class.
204  // Every time a particle enters a new volume, we check the material to see
205  // if it is known (i.e. contained in the range table fRangeTable).
206  // If so, then we calculate the step through the material (STEP) of the nucleus
207  // and the distance (DPATH in cm) between the intersection point at the focal plane
208  // and the point at the entrance of the volume if it is the first active volume of a detector.
209  // DPATH has the sign + if the volume is behind the focal plane or - if it
210  // is at the front of it.
211  //
212 
213  KVVAMOSReconNuc* rnuc = (KVVAMOSReconNuc*)nuc;
214 
215  // stop the propagation if the current volume is the stopping detector
216  // of the nucleus but after the process of this volume
217  if (rnuc->GetStoppingDetector()) {
218  TGeoVolume* stopVol = (TGeoVolume*)((KVVAMOSDetector*)rnuc->GetStoppingDetector())->GetActiveVolumes()->Last();
219 
220  if (GetCurrentVolume() == stopVol) SetStopPropagation();
221  }
222 
223  if (fDoNothing) return;
224 
225 
226  TGeoMaterial* material = GetCurrentVolume()->GetMaterial();
227  KVIonRangeTableMaterial* irmat = 0;
228 
229  // skip the process if the current material is unkown
230  if ((irmat = fRangeTable->GetMaterial(material))) {
231 
232  KVString dname;
233  Bool_t multi;
234  TString absorber_name;
235  Bool_t is_active = kFALSE;
236  if (GetCurrentDetectorNameAndVolume(dname, multi)) {
237  is_active = kTRUE;
238  if (multi) {
239  absorber_name.Form("%s/%s", dname.Data(), GetCurrentNode()->GetName());
240  is_active = absorber_name.Contains("ACTIVE_");
241  } else absorber_name = dname;
242  } else
243  absorber_name = irmat->GetName();
244 
245  // Coordinates of the vector between the intersection point at the
246  // focal plane and the point at the entrance of the current detector
247  Double_t X = GetEntryPoint().X() - fOrigine.X();
248  Double_t Y = GetEntryPoint().Y() - fOrigine.Y();
249  Double_t Z = GetEntryPoint().Z() - fOrigine.Z();
250 
251  // Norm of this vector. The signe gives an infomation about the detector position
252  // (1: behind; -1: in front of) with respect to the focal plane.
253  Double_t Delta = TMath::Sign(1., Z) * TMath::Sqrt(X * X + Y * Y + Z * Z);
254 
255  if ((fCalib & kECalib) || (fCalib & kTCalib)) {
256  if (fE > 1e-3) {
257 
258  // velocity before material
259  Double_t Vi = nuc->GetVelocity().Mag();
260 
261  // energy lost in the material
262  Double_t DE = irmat->GetLinearDeltaEOfIon(
263  nuc->GetZ(), nuc->GetA(), fE, GetStepSize(), 0.,
264  material->GetTemperature(),
265  material->GetPressure());
266  fE -= DE;
267  nuc->SetEnergy(fE);
268 
269  //set flag to say that particle has been slowed down
270  nuc->SetIsDetected();
271 
272  // velocity after material
273  Double_t Vf = nuc->GetVelocity().Mag();
274 
275  if (fCalib & kTCalib) {
276  //from current start point to the entrance point
277  fTOF += (Delta - fStartPath) / Vi;
278  fStartPath = Delta;
279 
280  //nuc->GetParameters()->SetValue(Form("TOF:%s",absorber_name.Data()), fTOF);
281  if (is_active) nuc->GetParameters()->SetValue(Form("TOF:%s", dname.Data()), fTOF);
282  else if ((fCalib & kFullTCalib) == kFullTCalib) nuc->GetParameters()->SetValue(Form("TOF:%s", absorber_name.Data()), fTOF);
283 
284 
285  // from the entrance to the exit of the material
286  Double_t step = GetStepSize();
287  fTOF += CalculateLinearDeltaT(Vi, Vf, step);
288  fStartPath += step;
289  }
290 
291  if (fCalib & kECalib) {
292  if (is_active) nuc->GetParameters()->SetValue(Form("DE:%s", dname.Data()), DE);
293  else if ((fCalib & kFullECalib) == kFullECalib) nuc->GetParameters()->SetValue(Form("DE:%s", absorber_name.Data()), DE);
294  }
295  }
296  }
297 
298  if (is_active) nuc->GetParameters()->SetValue(Form("DPATH:%s", dname.Data()), Delta);
299  else if ((fCalib & kFullTCalib) == kFullTCalib) nuc->GetParameters()->SetValue(Form("DPATH:%s", absorber_name.Data()), Delta);
300  nuc->GetParameters()->SetValue(Form("STEP:%s", absorber_name.Data()), GetStepSize());
301 
302  }
303 }

320 void KVVAMOSReconGeoNavigator::PropagateNucleus(KVVAMOSReconNuc* nuc, ECalib cal)
321 {
322  // Propagates the VAMOS reconstructed nucleus along its focal plane
323  // direction in the focal-plane detection area.
324  // Calculate the calibration for a well know calibration
325  // nucleus (Z, A, E) i.e. energy losses, steps, in each crossed detector, in the
326  // stripping foil and in the target (if it exists) as well as
327  // times of flight are calculated. We assume that you have set Z, A
328  // and energy of the nucleus before calling this method (see SetZandA(...)
329  // and SetEnergy(...) ).
330  //
331  // The calculated quantities are stored in the nucleus's list KVParticle::fParameters ( see main class description ).
332 
333  //Set the calibration choice
334  fCalib = cal;
335 
336  //If this is the first absorber that the particle crosses, we set a "reminder" of its
337  //initial energy
338  nuc->SetE0();
339  fE = nuc->GetEnergy();
340 
341  // For gGeoManager the origin is the target point.
342  // Starting point has to be set from this origin.
343 
344  // intersection point between reconstructed trajectory and the Focal
345  // Plane in the focal plane frame of reference
346  Double_t XYZ[3];
347 
348  //propagation direction at the focal plane
349  Double_t XYZdir[3];
350 
351  // Take reconstructed path if the trajector is correctly reconstructed
352  // or take the central trajectory otherwise
353  if (!nuc->GetCodes().TestFPCode(kFPCode0) && (nuc->GetPath() > 0)) {
354  fStartPath = -nuc->GetPath();
355  XYZ[0] = nuc->GetXf();
356  XYZ[1] = nuc->GetYf();
357  XYZ[2] = 0.;
358  nuc->GetFocalPlaneDirection().GetXYZ(XYZdir);
359  gVamos->FocalToTargetVect(XYZdir, XYZdir);
360  } else {
361  fStartPath = -gVamos->GetFocalPlanePosition();
362  XYZ[0] = XYZ[1] = XYZ[2] = 0.;
363  XYZdir[0] = XYZdir[1] = 0.;
364  XYZdir[2] = 1.;
365  Warning("PropagateNucleus", "A central trajectory is used to propagate the nucleus (FPCode%d, path= %f cm)", nuc->GetCodes().GetFPCodeIndex(), nuc->GetPath());
366  }
367  // intersection point between reconstructed trajectory and the Focal
368  // Plane in the frame of reference centered at the target point
369  gVamos->FocalToTarget(XYZ, XYZ);
370  fOrigine.SetXYZ(XYZ[0], XYZ[1], XYZ[2]);
371 
372  // propagation direction = direction of the trajectory at the focal plane
373  gVamos->FocalToTargetVect(XYZdir, XYZdir);
374  TVector3 direction(XYZdir);
375 
376  // Calculate energy losses in target and in tripping foil if the option
377  // kECalib or kFullCalib is chosen
378  if ((fCalib & kECalib) || (fCalib & kTCalib)) {
379 
380  // nuc->SetEnergy( fE );
381  Double_t DE = 0.;
382  TVector3 norm(0., 0., 1.); //vector normal to the target and to the stripping foil
383  Double_t nz = nuc->GetMomentum().Unit().Z();
384  fTOF = 0.;
385 
386  //target, we assume that the nucleus is not slow down in the target
387  if ((fE > 1e-3) && gMultiDetArray->GetTarget()) {
388  gMultiDetArray->GetTarget()->SetIncoming(kFALSE);
389  gMultiDetArray->GetTarget()->SetOutgoing(kTRUE);
390  DE = gMultiDetArray->GetTarget()->GetELostByParticle(nuc, &norm);
391  fE -= DE;
392  nuc->SetEnergy(fE);
393 
394  if (fCalib & kECalib) nuc->GetParameters()->SetValue(Form("DE:TARGET_%s", gMultiDetArray->GetTarget()->GetName()), DE);
395 
396  }
397 
398  //stripping foil
399  if ((fE > 1e-3) && gVamos->GetStripFoil()) {
400 
401  // velocity before stripping foil
402  Double_t Vi = nuc->GetVelocity().Mag();
403 
404  DE = gVamos->GetStripFoil()->GetELostByParticle(nuc, &norm);
405  fE -= DE;
406  nuc->SetEnergy(fE);
407 
408  // velocity after stripping foil
409  Double_t Vf = nuc->GetVelocity().Mag();
410 
411  //target to stripping foil
412  if (fCalib & kTCalib) {
413  Double_t path = gVamos->GetStripFoilPosition() / TMath::Max(nz, 1.e-03);
414  fTOF += path / Vi;
415  fStartPath += path;
416 
417  if ((fCalib & kFullTCalib) == kFullTCalib) {
418  nuc->GetParameters()->SetValue(Form("TOF:STRIPFOIL_%s", gVamos->GetStripFoil()->GetName()), fTOF);
419  nuc->GetParameters()->SetValue(Form("DPATH:STRIPFOIL_%s", gVamos->GetStripFoil()->GetName()), fStartPath);
420  }
421 
422  // from the entrance to the exit of the stripping foil
423  Double_t step = gVamos->GetStripFoil()->GetThickness() / TMath::Max(nz, 1.e-03);
424  fTOF += CalculateLinearDeltaT(Vi, Vf, step);
425  fStartPath += step;
426  nuc->GetParameters()->SetValue(Form("STEP:%s", gVamos->GetStripFoil()->GetName()), step);
427  }
428 
429  if ((fCalib & kFullECalib) == kFullECalib) nuc->GetParameters()->SetValue(Form("DE:STRIPFOIL_%s", gVamos->GetStripFoil()->GetName()), DE);
430  }
431  }
432 
433  // stop the E and T calculation if the nucleus does not have energy anymore
434  // but continue the propagation
435  if (fE < 1e-3) {
436  fCalib = kNoCalib;
437  fE = 1; // set a non null energy to continue the propagation
438  }
439 
440  // Backward propagation to find the second origine point located
441  // just before the first detector on this direction
442  fDoNothing = kTRUE;
443  direction = -direction;
444  nuc->SetMomentum(direction);
445  PropagateParticle(nuc, &fOrigine);
446 
447  // Forward propagation
448  fDoNothing = kFALSE;
449  direction = -direction;
450 
451  nuc->SetMomentum(fE, direction);
452  const Double_t* posi = GetGeometry()->GetCurrentPoint();
453  XYZ[0] = (posi[0] + GetExitPoint().X()) / 2.;
454  XYZ[1] = (posi[1] + GetExitPoint().Y()) / 2.;
455  XYZ[2] = (posi[2] + GetExitPoint().Z()) / 2.;
456  fFOrigine.SetXYZ(XYZ[0], XYZ[1], XYZ[2]);
457  // Info("PropagateNucleus","FOrigine vol. name: %s",GetCurrentVolume()->GetName());
458  // cout<<"FORIGINE POINT: "; fFOrigine.Print();
459  PropagateParticle(nuc, &fFOrigine);
460 
461  //set the initial momentum/energy to the nucleus
462  nuc->ResetEnergy();
463 
464 }

479 Double_t KVVAMOSReconGeoNavigator::CalculateLinearDeltaT(Double_t vi, Double_t vf, Double_t step) const
480 {
481  // Calculates the time spent for crossing a material for a given step.
482  // We assume that the velocity evolves linearly from the initial velocity 'vi'
483  // to the final velocity 'vf' reached after a distance 'step' i.e.
484  // v(x) = a x + b = dx/dt for x between 0 and step
485  // where
486  // a = (vf-vi)/step and b = vi
487  //
488  // Finally, the returned value is DeltaT = [ln(a*step+b)-ln(b)]/a
489  //
490 
491  if ((step <= 0) || (vi <= 0)) return 0.;
492  if (vi == vf) return step / vi;
493 
494  Double_t a = (vf - vi) / step;
495  Double_t b = vi;
496 
497  return (TMath::Log(a * step + b) - TMath::Log(b)) / a;
498 }

186 void KVZAFinderDialog::ProcessIdentification()
187 {
188  fZmin = fZminEntry->GetIntNumber();
189  fZmax = fZmaxEntry->GetIntNumber();
190  fAList = fAEntry->GetText();
191 
192  Int_t zmin = ((KVIDentifier*)fGrid->GetIdentifiers()->First())->GetPID();
193  if (fZmin < zmin) fZmin = zmin;
194  Int_t zmax = ((KVIDentifier*)fGrid->GetIdentifiers()->Last())->GetPID();
195  if (fZmax > zmax) fZmax = zmax;
196 
197  KVZALineFinder toto((KVIDZAGrid*)fGrid, fHisto);
198 
199 // fLinearBar->SetRange(0, ((TH2F*)fHisto)->GetSum());
200 // fLinearBar->SetRange(0, fHisto->GetNbinsX()*fHisto->GetNbinsY());
201 // fLinearBar->Reset();
202 // toto.Connect("IncrementLinear(Float_t)","TGHProgressBar",
203 // fLinearBar, "SetPosition(Float_t)");
204  fProgressBar->SetRange(0, zmax - zmin);
205  fProgressBar->Reset();
206  toto.Connect("Increment(Float_t)", "TGHProgressBar",
207  fProgressBar, "SetPosition(Float_t)");
208 
209  toto.SetAList(fAList.Data());
210  toto.SetNbinsByZ(fBinsByZ);
211 
212  fTestButton->SetEnabled(kFALSE);
213  fCloseButton->SetEnabled(kFALSE);
214 
215  toto.ProcessIdentification(fZmin, fZmax);
216  fTestButton->SetEnabled(kTRUE);
217  fCloseButton->SetEnabled(kTRUE);
218 
219  toto.Disconnect("IncrementLinear(Float_t)",
220  fLinearBar, "SetPosition(Float_t)");
221  toto.Disconnect("Increment(Float_t)",
222  fProgressBar, "SetPosition(Float_t)");
223 
224 
225  if (!gIDGridEditor) {
226  new KVIDGridEditor;
227  gIDGridEditor->StartViewer();
228  }
229 
230  gIDGridEditor->SetHisto(toto.GetHisto());
231  gIDGridEditor->SetGrid(toto.GetGrid(), kFALSE);
232 
233  DoClose();
234 }

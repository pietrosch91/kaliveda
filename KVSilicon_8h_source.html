87 inline Bool_t KVSilicon::IsCalibrated() const
88 {
89  //Redefined.
90  //Only one of the channel-volts calibrators needs to be ready,
91  //plus the volts-energy calibrator
92 
93  if (fVoltE && fVoltE->GetStatus()) {
94  Bool_t ok_gg = (fChVoltGG != 0) ? fChVoltGG->GetStatus() : 0;
95  Bool_t ok_pg = (fChVoltPG != 0) ? fChVoltPG->GetStatus() : 0;
96  return (ok_gg || ok_pg);
97  }
98 
99  return kFALSE;
100 }

35 void KVDataPatchList::FillPatchList()
36 {
37  // Fill static list of patches from KVDataPatch plugin list
38 
39  KVString plugins = GetListOfPlugins("KVDataPatch");
40  if (plugins == "") {
41  Info("FillPatchList", "No data patch plugins defined");
42  return;
43  }
44  plugins.Begin(" ");
45  while (!plugins.End()) {
46  KVString plug = plugins.Next(kTRUE);
47  TPluginHandler* h = LoadPlugin("KVDataPatch", GetPluginURI("KVDataPatch", plug));
48  if (h) {
49  KVDataPatch* p = (KVDataPatch*)h->ExecPlugin(0);
50  fPatchList->Add(p);
51  } else
52  Warning("FillPatchList", "Cannot load plugin data patch %s", plug.Data());
53  }
54 }

108 void KVDataPatchList::InitializePatchList(TString dataset, TString datatype, Int_t runnumber,
109  TString dataseries, Int_t datareleasenumber, const TList* streamerinfolist)
110 {
111  // Fill lists of active patches for given data
112 
113  Info("InitializePatchList",
114  "Initializing patches for |%s|%s|%04d|%s|%02d|",
115  dataset.Data(), datatype.Data(), runnumber, dataseries.Data(), datareleasenumber);
116  fActiveEventPatches.Clear();
117  fActiveParticlePatches.Clear();
118  fEventPatches = kFALSE;

120  TIter nxtP(fPatchList);
121  KVDataPatch* patch;
122  while ((patch = (KVDataPatch*)nxtP())) {
123  if (patch->IsRequired(dataset, datatype, runnumber, dataseries, datareleasenumber, streamerinfolist)) {
124  if (patch->IsEventPatch()) {
125  fActiveEventPatches.Add(patch);
126  fEventPatches = kTRUE;
127  }
128  if (patch->IsParticlePatch()) {
129  fActiveParticlePatches.Add(patch);
130  fParticlePatches = kTRUE;
131  }
132  }
133  }
134 }

141 void KVDataPatchList::Apply(KVEvent* event)
142 {
143  // Apply all currently active patches to event
144 
145  if (fEventPatches) {
146  // Apply event-level patches
147  TIter nxt(&fActiveEventPatches);
148  KVDataPatch* patch;
149  while ((patch = (KVDataPatch*)nxt())) patch->ApplyToEvent(event);
150  }
151  if (fParticlePatches) {
152  // Apply particle-level patches
153  TIter nxt(&fActiveParticlePatches);
154  KVDataPatch* patch;
155  KVNucleus* particle;
156  while ((particle = event->GetNextParticle())) {
157  while ((patch = (KVDataPatch*)nxt())) patch->ApplyToParticle(particle);
158  nxt.Reset();
159  }
160  }
161 }

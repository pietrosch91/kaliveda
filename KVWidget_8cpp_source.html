74 KVWidget::KVWidget(TObject* obj, Long_t func)
75 {
76 // Create a new widget associating the detector object *obj with the function func
77  fWidID = fCurrentID++;
78  // as the display is updated, and widgets are deleted and then redrawn,
79  // fCurrentID will just keep increasing, eventually becoming too big for a Long_t (?)
80  // So after a while it just starts out at zero again, which should work (I think...)
81  if (fCurrentID > kMAX_WIDGET_ID)
82  fCurrentID = 0;
83  fFunction = func;
84  fObject = obj;

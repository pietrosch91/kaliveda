81 Int_t KVIDCsIRLLine::Compare(const TObject* obj) const
82 {
83  //Used to sort lists of KVIDCsIRLLines.
84  //Sorts the lines as a function of increasing Z, then increasing A
85  //but 8Be is between 6He and 6Li
86 
87  int zline1 = GetZ();
88  int zline2 =
89  const_cast < KVIDZALine* >(static_cast <
90  const KVIDZALine* >(obj))->GetZ();
91  int aline1 = GetA();
92  int aline2 =
93  const_cast < KVIDCsIRLLine* >(static_cast <
94  const KVIDCsIRLLine*
95  >(obj))->GetA();
96  //both same line ?
97  if (zline1 == zline2 && aline1 == aline2)
98  return 0;
99  //special case : 8Be. If one of the two lines is 8Be, special treatment
100  //all lines with Z<3 are below 8Be, all lines with Z>=3 are above 8Be
101  if (zline1 == 4 && aline1 == 8) {
102  if (zline2 < 3)
103  return 1;
104  else
105  return -1;
106  }
107  if (zline2 == 4 && aline2 == 8) {
108  if (zline1 < 3)
109  return -1;
110  else
111  return 1;
112  }
113  //standard sorting when neither line is 8be
114  return KVIDZALine::Compare(obj);
115 }

150 void KVIDCsIRLLine::SetAsymWidth(Double_t d_l, Double_t d_r)
151 {
152  //Set the natural width of the line depending on its asymptotic separation
153  //a gauche (d_l) and a droite (d_r) from a neighbouring line.
154  //This method is used by KVIDZGrid::CalculateLineWidths

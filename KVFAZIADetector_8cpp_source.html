116 KVList* KVFAZIADetector::PrepareIDTelescopeList()
117 {
118 
119  KVList* lsub = new KVList();
120  lsub->SetOwner(kFALSE);
121 
122  KVList* ltel = GetAlignedIDTelescopes();
123  TList* ldet = GetAlignedDetectors();
124  KVNumberList nl;
125  nl.SetMinMax(0, ltel->GetEntries() - 1);
126  for (Int_t ii = 0; ii < ldet->GetEntries(); ii += 1) {
127  KVDetector* det = (KVDetector*)ldet->At(ii);
128  for (Int_t jj = 0; jj < ltel->GetEntries(); jj += 1) {
129  KVIDTelescope* tel = (KVIDTelescope*)ltel->At(jj);
130  if (tel->GetDetector(1) == det || (tel->GetSize() == 2 && tel->GetDetector(2) == det)) {
131  if (nl.Contains(jj)) {
132  lsub->Add(tel);
133  nl.Remove(jj);
134  }
135  }
136  }
137  }
138  return lsub;

339 Bool_t KVFAZIADetector::SetProperties()
340 {
341  // detector name are assumed to be defined as
342  // label-xxx
343  // where xxx is computed as follow :
344  // 100*block number+10*quartet number+telescope number
345  // and label can be SI1, SI2 or CSI
346  // For example SI1-123 is the Silicon Si1 of the block 1, the quartet 2 and the telescope 3
347  //
348  //
349 
350  KVString tmp;
351  KVString sname(GetName());
352 
353  sname.Begin("-");
354  SetLabel(sname.Next());
355  if (!strcmp(GetLabel(), "SI1")) fIdentifier = kSI1;
356  else if (!strcmp(GetLabel(), "SI2")) fIdentifier = kSI2;
357  else if (!strcmp(GetLabel(), "CSI")) fIdentifier = kCSI;
358 
359  gFazia->AddDetectorLabel(GetLabel());
360 
361  tmp = sname.Next();
362  if (tmp == "RUTH") {
363  fIsRutherford = kTRUE;
364  fIndex = fTelescope = fQuartet = fBlock = 0;
365  } else if (tmp.IsDigit()) {
366  fIndex = tmp.Atoi();
367  fBlock = fIndex / 100;
368  fQuartet = (fIndex - fBlock * 100) / 10;
369  fTelescope = fIndex - fBlock * 100 - fQuartet * 10;
370  } else {
371  Info("SetProperties", "Unkown format for the detector %s", GetName());
372  }
373 
374  KVSignal* sig = 0;
375  //"QH1", "I1", "QL1", "Q2", "I2", "Q3
376  if (fSignals)
377  delete fSignals;
378  fSignals = new KVList(kTRUE);
379  KVString lsignals = "";
380  if (!strcmp(GetLabel(), "SI1")) {
381  lsignals = "QH1,I1,QL1";
382  } else if (!strcmp(GetLabel(), "SI2")) {
383  lsignals = "Q2,I2";
384  } else if (!strcmp(GetLabel(), "CSI")) {
385  lsignals = "Q3";
386  } else {
387  Warning("SetProperties", "Unknown label \"%s\" for this detector : %s\n", GetLabel(), GetName());
388  lsignals = "";
389  }
390  TClass* cl = 0;
391  lsignals.Begin(",");
392  while (!lsignals.End()) {
393 
394  KVString ssig = lsignals.Next();
395  cl = TClass::GetClass(Form("KV%s", ssig.Data()));
396  sig = (KVSignal*)cl->New();
397  sig->SetName(ssig.Data());
398  sig->SetType(ssig.Data());
399 
400  sig->LoadPSAParameters();
401  sig->SetDetectorName(GetName());
402 
403  fSignals->Add(sig);
404  }
405 
406  SetCalibrators();
407 
408  return kTRUE;
409 }

416 const Char_t* KVFAZIADetector::GetNewName(KVString oldname)
417 {
418  // Translate an old-style FAZIA detector name (e.g. "SI1-T1-Q2-B001")
419  // to the new format ("SI1-121")
420 
421  Int_t tt = 0, qq = 0, bb = 0;
422  KVString tmp = "";
423  KVString lab = "";
424  oldname.Begin("-");
425 
426  if (!oldname.End()) {
427  lab = oldname.Next();
428  }
429  if (!oldname.End()) {
430  tmp = oldname.Next();
431  tmp.ReplaceAll("T", "");
432  tt = tmp.Atoi();
433  }
434  if (!oldname.End()) {
435  tmp = oldname.Next();
436  tmp.ReplaceAll("Q", "");
437  qq = tmp.Atoi();
438  }
439  if (!oldname.End()) {
440  tmp = oldname.Next();
441  tmp.ReplaceAll("B", "");
442  bb = tmp.Atoi();
443  }
444  static KVString newname;
445  newname.Form("%s-%d", lab.Data(), bb * 100 + qq * 10 + tt);
446  return newname.Data();

464 Bool_t KVFAZIADetector::Fired(Option_t*)
465 {
466  // Returns kTRUE if detector was hit (fired) in an event
467  //
468  // The test is made on charge signals of the detectors
469  // if one of them return kTRUE to KVSignal::IsFired() method KVDetector::Fired() return kTRUE
470  // if not return kFALSE and the detector will not be considered in following analysis
471  // except if one detector after it has been fired
472  //
473  // If the detector is in "simulation mode", i.e. if SetSimMode(kTRUE) has been called,
474  // this method returns kTRUE if the calculated energy loss in the active layer is > 0.
475  //
476 
477  if (!IsDetecting()) return kFALSE; //detector not working, no answer at all
478  if (IsSimMode()) return (GetActiveLayer()->GetEnergyLoss() > 0.); // simulation mode: detector fired if energy lost in active layer
479  KVSignal* sig;
480  if (fSignals) {
481  TIter next(fSignals);
482  while ((sig = (KVSignal*)next())) {
483  if (sig->GetN() > 0) {
484  if (sig->IsCharge()) {
485 
486  //pre process to use the test method KVSignal::IsFired()
487  sig->ComputeEndLine();
488  sig->TreateSignal();
489 
490  if (sig->IsFired()) {
491  return kTRUE;
492  } else {
493 
494  }
495  }
496  } else {
497  //Warning("Fired","%s has empty signal %s",GetName(),sig->GetName());
498  }
499  }
500  } else {
501  Warning("Fired", "%s : No signal attached to this detector ...", GetName());
502  }
503 
504  return kFALSE;
505 }

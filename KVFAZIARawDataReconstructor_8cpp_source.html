80 void KVFAZIARawDataReconstructor::InitRun()
81 {
82  // Creates new ROOT file with TTree for reconstructed events.
83  // By default this file will be written in the same data repository as the raw data file we are reading.
84  // This can be changed by setting the environment variable(s):
85  //
86  // Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
87  // [name of dataset].Reconstruction.DataAnalysisTask.OutputRepository: [name of repository]
88  //
89  // If no value is set for the current dataset (second variable), the value of the
90  // first variable will be used. If neither is defined, the new file will be written in the same repository as
91  // the raw file (if possible, i.e. if repository is not remote).
92 
93  // Create new KVReconstructedEvent filled with KVFAZIAReconNuc object
94  // used to reconstruct & store events
95 
96  if (!recev) recev = new KVReconstructedEvent(50, "KVFAZIAReconNuc");
97 
98  // get dataset to which we must associate new run
99  KVDataSet* OutputDataset =
100  gDataRepositoryManager->GetDataSet(gDataSet->GetOutputRepository(taskname), gDataSet->GetName());
101 
102  file = OutputDataset->NewRunfile(datatype.Data(), GetCurrentRunNumber());
103 
104  std::cout << "Writing \"" << datatype.Data() << "\" events in ROOT file " << file->GetName() << std::endl;
105 
106  //tree for reconstructed events
107  tree = new TTree("ReconstructedEvents", Form("%s : %s : %s events created from raw data",
108  gFaziaDB->GetRun(GetCurrentRunNumber())->GetName(),
109  gFaziaDB->GetRun(GetCurrentRunNumber())->GetTitle(),
110  datatype.Data())
111  );
112 
113  //leaves for reconstructed events
114  KVEvent::MakeEventBranch(tree, "FAZIAReconEvent", "KVReconstructedEvent", &recev);
115 
116  Info("InitRun", "Created reconstructed data tree %s : %s", tree->GetName(), tree->GetTitle());
117  nb_recon = 0;
118 
119 }

132 Bool_t KVFAZIARawDataReconstructor::Analysis()
133 {
134  // Analysis of event.
135  // RawData TTree is filled with values of parameters for event.
136  // If event has INDRA trigger information (IsIndraEvent()==kTRUE)
137  // then
138  // *) event reconstruction is performed for 'Physics' events
139  // *) or the GeneTree is filled with pulser/laser data for 'Gene' events
140 
141 
142  recev->SetNumber(GetEvent()->GetNumber());
143  gFazia->ReconstructEvent(recev, GetDetectorEvent());
144 
145  ExtraProcessing();
146 
147  nb_recon++;
148  tree->Fill();
149 
150  recev->Clear();
151 
152  return kTRUE;
153 }

160 void KVFAZIARawDataReconstructor::ExtraProcessing()
161 {
162  KVString label = "";
163 
164  KVFAZIADetector* det = 0;
165  KVSignal* sig = 0;
166  KVReconstructedNucleus* recnuc = 0;
167  while ((recnuc = recev->GetNextParticle())) {
168  TIter next_d(recnuc->GetDetectorList());
169  while ((det = (KVFAZIADetector*)next_d())) {
170  TIter next_s(det->GetListOfSignals());
171  while ((sig = (KVSignal*)next_s())) {
172  if (sig->HasFPGA()) {
173  for (Int_t ii = 0; ii < sig->GetNFPGAValues(); ii += 1) {
174  //SI2-T3-Q1-B003.Q2.RawAmplitude=14
175  if (ii == 0) label = "FPGAEnergy";
176  if (ii == 1) label = "FPGAFastEnergy"; //only for CsI Q3
177  TString ene = GetEvent()->GetFPGAEnergy(
178  det->GetBlockNumber(),
179  det->GetQuartetNumber(),
180  det->GetTelescopeNumber(),
181  sig->GetType(),
182  ii
183  );
184 
185  recnuc->GetParameters()->SetValue(
186  Form("%s.%s.%s", det->GetName(), sig->GetName(), label.Data()),
187  ene.Data()
188  );
189  }
190  }
191  if (!sig->PSAHasBeenComputed()) {
192  sig->TreateSignal();
193  }
194 
195  KVNameValueList* psa = sig->GetPSAResult();
196  if (psa) {
197  *(recnuc->GetParameters()) += *psa;
198  delete psa;
199  }
200  }
201  }
202  }
203 }

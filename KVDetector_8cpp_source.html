364 void KVDetector::DetectParticle(KVNucleus* kvp, TVector3* norm)
365 {
366  //Calculate the energy loss of a charged particle traversing the detector,
367  //the particle is slowed down, it is added to the list of all particles hitting the
368  //detector. The apparent energy loss of the particle in the active layer of the
369  //detector is set.
370  //Do nothing if particle has zero (or -ve) energy.
371  //
372  //If the optional argument 'norm' is given, it is supposed to be a vector
373  //normal to the detector, oriented from the origin towards the detector.
374  //In this case the effective thicknesses of the detector's absorbers 'seen' by the particle
375  //depending on its direction of motion is used for the calculation.
376 
377  if (kvp->GetKE() <= 0.)
378  return;
379 
380  AddHit(kvp); //add nucleus to list of particles hitting detector in the event
381  //set flag to say that particle has been slowed down
382  kvp->SetIsDetected();
383  //If this is the first absorber that the particle crosses, we set a "reminder" of its
384  //initial energy
385  if (!kvp->GetPInitial())
386  kvp->SetE0();
387 
388  Double_t* thickness = 0;
389  if (norm) {
390  // modify thicknesses of all layers according to orientation,
391  // and store original thicknesses in array
392  TVector3 p = kvp->GetMomentum();
393  thickness = new Double_t[fAbsorbers->GetEntries()];
394  KVMaterial* abs;
395  int i = 0;
396  TIter next(fAbsorbers);
397  while ((abs = (KVMaterial*) next())) {
398  thickness[i++] = abs->GetThickness();
399  Double_t T = abs->GetEffectiveThickness((*norm), p);
400  abs->SetThickness(T);
401  }
402  }
403  Double_t eloss = GetTotalDeltaE(kvp->GetZ(), kvp->GetA(), kvp->GetEnergy());
404  Double_t dE = GetDeltaE(kvp->GetZ(), kvp->GetA(), kvp->GetEnergy());
405  if (norm) {
406  // reset thicknesses of absorbers
407  KVMaterial* abs;
408  int i = 0;
409  TIter next(fAbsorbers);
410  while ((abs = (KVMaterial*) next())) {
411  abs->SetThickness(thickness[i++]);
412  }
413  delete [] thickness;
414  }
415  Double_t epart = kvp->GetEnergy() - eloss;
416  if (epart < 1e-3) {
417  //printf("%s, pb d arrondi on met l energie de la particule a 0\n",GetName());
418  epart = 0.0;
419  }
420  kvp->SetEnergy(epart);
421  Double_t eloss_old = GetEnergyLoss();
422  SetEnergyLoss(eloss_old + dE);
423 }

491 Double_t KVDetector::GetParticleEIncFromERes(KVNucleus* kvp, TVector3* norm)
492 {
493  //Calculate the energy of particle 'kvn' before its passage through the detector,
494  //based on the current kinetic energy, Z & A of nucleus 'kvn', supposed to be
495  //after passing through the detector.
496  //
497  //If the optional argument 'norm' is given, it is supposed to be a vector
498  //normal to the detector, oriented from the origin towards the detector.
499  //In this case the effective thicknesses of the detector's absorbers 'seen' by the particle
500  //depending on its direction of motion is used for the calculation.
501 
502  Double_t Einc = 0.0;
503  //make 'clone' of particle
504  KVNucleus* clone_part = new KVNucleus(kvp->GetZ(), kvp->GetA());
505  clone_part->SetMomentum(kvp->GetMomentum());
506  //composite detector - calculate losses in all layers
507  KVMaterial* abs;
508  TIter next(fAbsorbers, kIterBackward); //work through list backwards
509  while ((abs = (KVMaterial*) next())) {
510 
511  //calculate energy of particle before current absorber
512  Einc = abs->GetParticleEIncFromERes(clone_part, norm);
513 
514  //set new energy of particle
515  clone_part->SetKE(Einc);
516  }
517  //delete clone
518  delete clone_part;
519  return Einc;
520 }

970 Double_t KVDetector::GetCorrectedEnergy(KVNucleus* nuc, Double_t e, Bool_t transmission)

973  // including inactive absorber layers.
974  // e = energy loss in active layer (if not given, we use current value)
975  // transmission = kTRUE (default): the particle is assumed to emerge with
976  // a non-zero residual energy Eres after the detector.
977  // = kFALSE: the particle is assumed to stop in the detector.
978  //
979  // WARNING: if transmission=kTRUE, and if the residual energy after the detector
980  // is known (i.e. measured in a detector placed after this one), you should
981  // first call
982  // SetEResAfterDetector(Eres);
983  // before calling this method. Otherwise, especially for heavy ions, the
984  // correction may be false for particles which are just above the punch-through energy.
985  //
986  // WARNING 2: if measured energy loss in detector active layer is greater than
987  // maximum possible theoretical value for given nucleus' Z & A, this may be because
988  // the A was not measured but calculated from Z and hence could be false, or perhaps
989  // there was an (undetected) pile-up of two or more particles in the detector.
990  // In this case we return the uncorrected energy measured in the active layer
991  // and we add the following parameters to the particle (in nuc->GetParameters()):
992  //
993  // GetCorrectedEnergy.Warning = 1
994  // GetCorrectedEnergy.Detector = [name]
995  // GetCorrectedEnergy.MeasuredDE = [value]
996  // GetCorrectedEnergy.MaxDE = [value]
997  // GetCorrectedEnergy.Transmission = 0 or 1
998  // GetCorrectedEnergy.ERES = [value]
999 
1000  Int_t z = nuc->GetZ();
1001  Int_t a = nuc->GetA();
1002 
1003  if (e < 0.) e = GetEnergy();
1004  if (e <= 0) {
1005  SetEResAfterDetector(-1.);
1006  return 0;
1007  }
1008 
1009  enum SolType solution = kEmax;
1010  if (!transmission) solution = kEmin;
1011

1013  Double_t maxDE = GetMaxDeltaE(z, a);
1014  Double_t EINC, ERES = GetEResAfterDetector();

1017  nuc->GetParameters()->SetValue("GetCorrectedEnergy.Detector", GetName());
1018  nuc->GetParameters()->SetValue("GetCorrectedEnergy.MeasuredDE", e);
1019  nuc->GetParameters()->SetValue("GetCorrectedEnergy.MaxDE", maxDE);
1020  nuc->GetParameters()->SetValue("GetCorrectedEnergy.Transmission", (Int_t)transmission);
1021  nuc->GetParameters()->SetValue("GetCorrectedEnergy.ERES", ERES);
1022  return e;
1023 
1024  }
1025  if (transmission && ERES > 0.) {
1026  // if residual energy is known we use it to calculate EINC.
1027  // if EINC < max of dE curve, we change solution
1028  EINC = GetIncidentEnergyFromERes(z, a, ERES);
1029  if (EINC < GetEIncOfMaxDeltaE(z, a)) solution = kEmin;
1030  // we could keep the EINC value calculated using ERES, but then
1031  // the corrected dE of this detector would not depend on the
1032  // measured dE !
1033  }
1034  EINC = GetIncidentEnergy(z, a, e, solution);

1846 Double_t KVDetector::GetIncidentEnergy(Int_t Z, Int_t A, Double_t delta_e, enum SolType type)
1847 {
1848  // Overrides KVMaterial::GetIncidentEnergy
1849  // Returns incident energy corresponding to energy loss delta_e in active layer of detector for a given nucleus.
1850  // If delta_e is not given, the current energy loss in the active layer is used.
1851  //
1852  // By default the solution corresponding to the highest incident energy is returned
1853  // This is the solution found for Einc greater than the maximum of the dE(Einc) curve.
1854  // If you want the low energy solution set SolType = KVIonRangeTable::kEmin.
1855  //
1856  // WARNING: calculating the incident energy of a particle using only the dE in a detector
1857  // is ambiguous, as in general (and especially for very heavy ions) the maximum of the dE
1858  // curve occurs for Einc greater than the punch-through energy, therefore it is not always
1859  // true to assume that if the particle does not stop in the detector the required solution
1860  // is that for type=KVIonRangeTable::kEmax. For a range of energies between punch-through
1861  // and dE_max, the required solution is still that for type=KVIonRangeTable::kEmin.
1862  // If the residual energy of the particle is unknown, there is no way to know which is the
1863  // correct solution.
1864  //
1865  // WARNING 2
1866  // If the given energy loss in the active layer is greater than the maximum theoretical dE
1867  // for given Z & A, (dE > GetMaxDeltaE(Z,A)) then we return a NEGATIVE incident energy
1868  // corresponding to the maximum, GetEIncOfMaxDeltaE(Z,A)
1869 
1870  if (Z < 1) return 0.;
1871 
1872  Double_t DE = (delta_e > 0 ? delta_e : GetEnergyLoss());
1873 
1874  // If the given energy loss in the active layer is greater than the maximum theoretical dE
1875  // for given Z & A, (dE > GetMaxDeltaE(Z,A)) then we return a NEGATIVE incident energy
1876  // corresponding to the maximum, GetEIncOfMaxDeltaE(Z,A)
1877  if (DE > GetMaxDeltaE(Z, A)) return -GetEIncOfMaxDeltaE(Z, A);
1878 
1879  TF1* dE = GetELossFunction(Z, A);
1880  Double_t e1, e2;
1881  dE->GetRange(e1, e2);
1882  switch (type) {
1883  case kEmin:
1884  e2 = GetEIncOfMaxDeltaE(Z, A);
1885  break;
1886  case kEmax:
1887  e1 = GetEIncOfMaxDeltaE(Z, A);
1888  break;
1889  }
1890  return dE->GetX(DE, e1, e2);
1891 }

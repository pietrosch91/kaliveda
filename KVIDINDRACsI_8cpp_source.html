92 const Char_t* KVIDINDRACsI::GetArrayName()
93 {
94  // Name of telescope given in the form CSI_R_L_Ring-numberTelescope-number
95  //where ring and telescope numbers are those of the smallest (in angular terms)
96  //detector of the telescope (if both are the same size, either one will do).
97  // The detectors are signified by their TYPE names i.e. KVDetector::GetType
98 
99  //in order to access angular dimensions of detectors, we need their KVTelescopes
100  KVINDRATelescope* de_det = dynamic_cast<KVINDRADetector*>(GetDetector(1))->GetTelescope();
101  UInt_t ring, mod;
102  ring = de_det->GetRingNumber();
103  mod = de_det->GetNumber();
104  TString dummy;
105  dummy.Form("%s_R_L", GetDetector(1)->GetType());
106  SetType(dummy.Data());
107  dummy.Form("%s_%02d%02d", GetType(), ring, mod);
108  SetName(dummy.Data());
109 
110  return fName.Data();
111 }

119 Bool_t KVIDINDRACsI::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)

122 
123  IDR->SetIDType(GetType());
124  IDR->IDattempted = kTRUE;
125 
126  //perform identification
127  Double_t csir = (y < 0. ? GetIDMapY() : y);
128  Double_t csil = (x < 0. ? GetIDMapX() : x);
129  CsIGrid->Identify(csil, csir, IDR);
130

133 
134  // general ID code for gammas

138  return kTRUE;

209 void KVIDINDRACsI::SetIdentificationStatus(KVReconstructedNucleus* n)
210 {
211  // For filtering simulations
212  // If n->GetEnergy() is above threshold for mass identification, we set
213  // n->IsAMeasured(kTRUE) (and n->IsZMeasured(kTRUE)).
214  // Otherwise, we just set n->IsZMeasured(kTRUE) and use the A given by
215  // the mass formula for the particle
216  //
217  // Z-dependence of A identification:
218  // all ok above threshold if Z<=16, decreasing probability for 17<=Z<=21
219  // no A identification for Z>21
220  n->SetZMeasured();
221 
222  if (n->GetA() > 5) {
223  if (fCsI->GetEnergy() > 40) n->SetAMeasured();
224  else n->SetZ(n->GetZ());
225  return;
226  }
227 
228  Bool_t okmass = gRandom->Uniform() < smootherstep(fThresMin[n->GetZ() - 1][n->GetA() - 1], fThresMax[n->GetZ() - 1][n->GetA() - 1], fCsI->GetEnergy());
229  if (okmass) {
230  n->SetAMeasured();
231  } else
232  n->SetZ(n->GetZ());
233 }

254 float KVIDINDRACsI::smootherstep(float edge0, float edge1, float x)
255 {
256  // Scale, and clamp x to 0..1 range
257  x = clamp((x - edge0) / (edge1 - edge0), 0.0, 1.0);
258  // Evaluate polynomial
259  return x * x * x * (x * (x * 6 - 15) + 10);
260 }

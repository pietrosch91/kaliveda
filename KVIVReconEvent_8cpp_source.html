25 // GetParticle(Int_t i) - returns ith reconstructed particle of event (i=1,...,GetINDRAMult())
26 // GetParticleWithName(const Char_t*) - returns the first particle with given name
27 // GetParticle(const Char_t*) - returns the first particle belonging to a given group
28 // UseMeanAngles() - particle's theta & phi are taken from mean theta & phi of detectors
29 // UseRandomAngles() - particle's theta & phi are randomized within theta & phi limits of detectors
30 // HasMeanAngles()/HasRandomAngles() - indicate in which of the two previous cases we find ourselves
31 //
32 // GetNextParticle(Option_t* opt) -
33 // Use this method to iterate over the list of particles in the INDRA event
34 // To make sure you begin at the start of the list, call ResetGetNextParticle()
35 // before beginning the iteration.
36 // After the last particle GetNextParticle() returns a null pointer and
37 // resets itself ready for a new iteration over the particle list:
38 //
39 // Example: (supposing "KVIVReconEvent* event" points to a valid event)
40 // KVINDRAReconNuc* indra_part; event->ResetGetNextParticle();
41 // while( (indra_part = event->GetNextParticle()) ){
42 // ...
43 // }
44 //
45 // If opt="ok" only particles with KVReconstructedNucleus::IsOK() = kTRUE

50 //
51 // GetNucleus(Int_t i) - returns ith reconstructed nucleuis of VAMOS event (i=1,...,GetVAMOSMult())
52 // GetNucleus(const Char_t*) - returns the first nucleus belonging to a given group
53 //
54 // GetNextNucleus(Option_t* opt) -
55 // Use this method to iterate over the list of nuclei in the VAMOS event
56 // To make sure you begin at the start of the list, call ResetGetNextNucleus()
57 // before beginning the iteration.
58 // After the last nucleus GetNextNucleus() returns a null pointer and
59 // resets itself ready for a new iteration over the nucleus list:
60 //
61 // Example: (supposing "KVIVReconEvent* event" points to a valid event)
62 // KVVAMOSReconNuc* vamos_nuc; event->ResetGetNextNucleus();
63 // while( (vamos_nuc = event->GetNextNucleus()) ){
64 // ...
65 // }
66 //
67 // If opt="ok" only nuclei with KVReconstructedNucleus::IsOK() = kTRUE
68 // are included in the iteration.
69 //
70 // Normally the VAMOS event is composed by only one nucleus.
71 //
72 //
73 // The "OK" status of particles/nuclei is defined by setting acceptable identification and calibration (reconstruction)

206 Int_t KVIVReconEvent::GetTotalMult(Option_t* opt)
207 {
208  //Returns total multiplicity (number of particles in INDRA + nucleus in VAMOS) of event.
209  //If opt = "" (default), returns number of particles in TClonesArray* fParticles (INDRA) + fVAMOSev->fParticles (VAMOS).
210  //If opt = "ok" only particles and nuclei with IsOK()==kTRUE are included.
211  //
212  //IN THE LATTER CASE, YOU MUST NOT USE THIS METHOD INSIDE A LOOP
213  //OVER THE INDRA'S EVENT USING GETNEXTPARTICLE() OR OVER THE VAMOS'S
214  //EVENT USING GETNEXTNUCLEUS!!!
215  //
216  //BE CAREFUL: the method GetMult(Option_t *) does not return the total
217  //multiplicity but only the multiplicity in INDRA.
218 
219  return GetINDRAMult(opt) + GetVAMOSMult(opt);
220 }

488 Bool_t KVVAMOS::ReadDetectorGroupFile(ifstream& ifile)
489 {
490  //Reads the groups of detectors to build for VAMOS from file loaded
491  //in ifile and stores them as a structure.
492  //Before reading, all old groups are deleted. A number is set
493  //to each new group.
494  //
495  //The comment lines begin with '#'.
496  //
497  //Each line correspond to a list with the name of each detector which
498  //can be punshed through by a same trajectory. Each name is
499  //separated by a space.
500  //A detector can only belong to one group.
501 
502  ClearStructures("GROUP"); // clear out (delete) old groups
503 
504  KVString sline, detname;
505  while (ifile.good()) { //reading the file
506  sline.ReadLine(ifile);
507 
508  if (sline.IsNull()) continue;
509  // Skip comment line
510  if (sline.BeginsWith("#")) continue;
511 
512  KVGroup* cur_grp = NULL;
513  KVGroup* det_grp = NULL;
514  KVSpectroDetector* det = NULL;
515  KVSpectroDetector* lastDet = NULL;
516 
517  sline.ReplaceAll("\t", " ");
518  sline.Begin(" ");
519  while (!sline.End()) {
520 
521  detname = sline.Next(kTRUE);
522  det = (KVSpectroDetector*)GetDetector(detname.Data());
523 
524  if (det) {
525  det_grp = det->GetGroup();
526 
527  if (!cur_grp) {
528  if (det_grp) cur_grp = det_grp;
529  else {
530  cur_grp = new KVGroup;
531  cur_grp->SetNumber(++fGr);
532  cur_grp->Add(det);
533  Add(cur_grp);
534  }
535  } else {
536  if (det_grp && det_grp != cur_grp)
537  Warning("ReadDetectorGroupFile",
538  "Detector %s : already belongs to %s, now seems to be in %s",
539  det->GetName(), det_grp->GetName(),
540  cur_grp->GetName());
541  else cur_grp->Add(det);
542  }
543 
544  det->GetNode()->SetName(det->GetName());
545 
546  if (lastDet && det != lastDet) {
547  lastDet->GetNode()->AddBehind(det);
548  det->GetNode()->AddInFront(lastDet);
549  }
550  lastDet = det;
551 
552  // Sort the detector list of each group
553  // from the the closest from the target to the furthest.
554  cur_grp->SortDetectors(kSortAscending);
555  } else Error("ReadDetectorGroupFile", "Detector %s not found", detname.Data());
556  }
557  }
558  return kTRUE;
559 }

1038 Bool_t KVVAMOS::Calibrate(KVReconstructedNucleus* nuc)
1039 {
1040  // Calculate and set the energy of a reconstructed nucleus from
1041  // the energy measured in the detectors listed in the detector
1042  // list of this nucleus. A calibration function is used:
1043  // Ecal = C_0 + C_1*E_CHI + C_2*E_SI + C_3*E_CSI;
1044  // where C_i are calibration parameters and E_i calibrated
1045  // energy of the detector i (KVVAMOSDetector::GetEnergy()).
1046  // To set these parameters use SetECalibParameters(...) method.
1047  //status code
1048 
1049  nuc->SetECode(kECode0);
1050  if (!GetECalibStatus()) return kFALSE;
1051 
1052  TIter next(nuc->GetDetectorList());
1053  KVVAMOSDetector* det = NULL;
1054  KVVAMOSDetector* csi, *si, *chi;
1055  csi = si = chi = NULL;
1056  while ((det = (KVVAMOSDetector*)next())) {
1057  if (det->IsType("CSI")) csi = det;
1058  else if (det->IsType("SI")) si = det;
1059  else if (det->IsType("CHI")) chi = det;
1060  }
1061 
1062  Double_t E = fECalibPar[0]
1063  + (chi ? fECalibPar[1] * chi->GetEnergy() : 0.)
1064  + (si ? fECalibPar[2] * si->GetEnergy() : 0.)
1065  + (csi ? fECalibPar[3] * csi->GetEnergy() : 0.);
1066 
1067  if (E <= 0) return kFALSE;
1068 
1069  nuc->SetEnergy(E);
1070  nuc->SetECode(kECode1);
1071  nuc->SetIsCalibrated();
1072 
1073  return kTRUE;
1074 }

1206  Int_t e_thick = TMath::Nint(e->GetThickness());

1229  set_up_telescope(de, e, idt, idtels);
1230  ++ntels;

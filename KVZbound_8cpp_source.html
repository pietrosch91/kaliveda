14 // Global variable returning the sum of the charges of particles and fragments.
15 //
16 // By default this sum includes all nuclei having a charge greater than or equal to 'zmin'.
17 //The value of zmin can be set calling the method SetZmin(z).
18 //The value of zmin can be obtained by calling the method GetZmin().
19 //By default, zmin is set to 3.
20 //N.B. in the "standard" definition of the Zbound variable used by the ALADIN
21 //collaboration, zmin=2 (i.e. Zbound=sum of all charged products except hydrogens)
22 //
23 //KVZbound can also be used to calculate the sum of Z for
24 //
25 // i) all nuclei with Z>=zmin
26 // ii) all nuclei with Z<=zmax
27 // iii) all nuclei with Z>=zmin and Z<=zmax
28 // iv) any of the above conditions + selection based on projection of particle's
29 //velocity along the 'z'-direction of whatever reference frame you choose (using SetFrame())
30 //
31 //EXAMPLE OF USE
32 //
33 //In order to add a KVZbound global variable to your analysis, place the following
34 //line in the InitAnalysis() method of your analysis class:
35 //
36 // KVVarGlob* vg = AddGV("KVZbound", "zbound");
37 //
38 //This gives the default behaviour (sum of Z for Z>=3).
39 //To change the selection conditions:
40 //
41 // i) sum of all nuclei with Z>=5:
42 // vg->SetParameter("Zmin", 5);
43 //
44 // ii) sum of all nuclei with Z<=12:
45 // vg->SetParameter("Zmax", 12);
46 //
47 // iii) sum of all nuclei with Z>=3 and Z<=20:
48 // vg->SetParameter("Zmin", 3);
49 // vg->SetParameter("Zmax", 20);
50 //
51 // iv) all nuclei with Z<=2 and V>=0 in the centre of mass frame:
52 // vg->SetParameter("Zmax", 2);
53 // vg->SetFrame("CM");
54 // vg->SetParameter("Vmin", 0);
55 //
56 // v) all nuclei with 3<=Z<=10 and -1.6<=V<=3.4 (cm/ns) in the "ELLIPSOID"
57 // frame (if you define a frame with such a name...)
58 // vg->SetParameter("Zmin", 3);
59 // vg->SetParameter("Zmax", 10);
60 // vg->SetFrame("ELLIPSOID");
61 // vg->SetParameter("Vmin", -1.6);
62 // vg->SetParameter("Vmax", 3.4);
63 //

298 void KVZbound::Fill(KVNucleus* c)
299 {
300  //Add the atomic number, Z, of the nucleus to the Zbound variable
301  //if it satisfies the conditions imposed by the user
302 
303  Int_t z = c->GetZ();
304  Bool_t ok = (TestBit(kZMin) ? z >= zmin : (TestBit(kZMax) ? z <= zmax : (z >= zmin && z <= zmax)));
305  if (ok && TestBit(kVTest)) {
306  Double_t v = c->GetFrame(fFrame.Data(), kFALSE)->GetVpar();
307  ok = (TestBit(kVMin) ? v >= vmin : (TestBit(kVMax) ? v <= vmax : (v >= vmin && v <= vmax)));
308  }
309  if (ok) FillVar(z);
310 }

342 void KVZbound::Init()
343 {
344  //Initialisation of variable based on current values of options and parameters
345 
346  ResetBit(kZMin);
347  ResetBit(kZMax);
348  ResetBit(kVMin);
349  ResetBit(kVMax);
350  ResetBit(kZBoth);
351  ResetBit(kVBoth);
352  //get type of Z limits - min, max, or both
353  if (IsParameterSet("Zmin") && IsParameterSet("Zmax")) {
354  SetBit(kZBoth); // Z>= Zmin && Z<=Zmax
355  zmin = (Int_t)GetParameter("Zmin");
356  zmax = (Int_t)GetParameter("Zmax");
357  } else if (IsParameterSet("Zmin")) {
358  SetBit(kZMin); // Z>= Zmin
359  zmin = (Int_t)GetParameter("Zmin");
360  } else if (IsParameterSet("Zmax")) {
361  SetBit(kZMax); // Z>= Zmax
362  zmax = (Int_t)GetParameter("Zmax");
363  } else {
364  //no Z parameter set
365  //default is to calculate Zbound for Z>= 3
366  SetBit(kZMin);
367  zmin = 3;
368  }
369  //get type of V limits - min, max, or both
370  if (IsParameterSet("Vmin") && IsParameterSet("Vmax")) {
371  SetBit(kVBoth); // V>= Vmin && V<=Vmax
372  vmin = GetParameter("Vmin");
373  vmax = GetParameter("Vmax");
374  } else if (IsParameterSet("Vmin")) {
375  SetBit(kVMin); // V>= Vmin
376  vmin = GetParameter("Vmin");
377  } else if (IsParameterSet("Vmax")) {
378  SetBit(kVMax); // V>= Vmax
379  vmax = GetParameter("Vmax");
380  }
381  //set kVTest flag if any velocity condition is imposed
382  SetBit(kVTest, (TestBit(kVBoth) || TestBit(kVMin) || TestBit(kVMax)));
383 }

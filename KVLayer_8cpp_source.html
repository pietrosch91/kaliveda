31 //<h3>KVLayer: collection of rings in a multidetector array</h3>
32 //End_Html
33 //
34 // A layer is composed of rings of telescopes (usually, though not necessarily, of the same type)
35 // all at the "same distance" from the target in a multidetector array.
36 // The geometry of different layers in the same array is not necessarily the same. They are used
37 // to define the hierarchy of detectors with respect to their distance from the target, i.e. the
38 // order in which a particle leaving the target will pass through them. The layer number
39 // (GetNumber()), defined by the order in which layers are added to the array, increases with
40 // increasing distance from the target i.e. particles always pass through the telescopes in layer
41 // 0 before those in layer 1 (unless there is a hole in layer 0 where particles can pass directly
42 // to the layer 1 telescopes!).

125 TGeoVolume* KVLayer::GetGeoVolume()
126 {
127  // Create and return TGeoVolume representing detectors in this layer
128 
129  TGeoVolume* mother_vol = gGeoManager->MakeVolumeAssembly(GetName());
130  //**** BUILD & ADD Rings ****
131  TIter next(GetStructures());
132  KVRing* det;
133  while ((det = (KVRing*)next())) {
134  TGeoVolume* det_vol = det->GetGeoVolume();
135  // position ring in layer
136  TGeoTranslation* tr = new TGeoTranslation(0, 0, det->GetDistance()); //distance set in KVRing::GetGeoVolume()
137  mother_vol->AddNode(det_vol, 1, tr);
138  }
139  return mother_vol;
140 }

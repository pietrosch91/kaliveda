113 Bool_t KVIDSiCsI_e475s::Identify(KVIdentificationResult* IDR, Double_t x, Double_t y)
114 {
115  //Particle identification and code setting using identification grid KVIDGChIoSi
116  IDR->SetIDType(GetType());
117  IDR->IDattempted = kTRUE;
118 
119  //identification
120  Double_t varX = (x < 0. ? GetIDMapX() : x);
121  Double_t varY = (y < 0. ? GetIDMapY() : y);
122 
123  if (fidgrid->IsIdentifiable(varX, varY))
124  fidgrid->Identify(varX, varY, IDR);
125 
126  Int_t quality = fidgrid->GetQualityCode();
127 
128  if (quality == KVIDZAGrid::kICODE8) {
129  // only if the final quality code is kICODE8 do we consider that it is
130  // worthwhile looking elsewhere. In all other cases, the particle has been
131  // "identified", even if we still don't know its Z and/or A (in this case
132  // we consider that we have established that they are unknowable).
133  return kFALSE;
134  }
135 
136  if (quality == KVIDZAGrid::kICODE7) {
137  // if the final quality code is kICODE7 (above last line in grid) then the estimated
138  // Z is only a minimum value (Zmin)
139  IDR->IDcode = kIDCode5;
140  return kTRUE;
141  }
142 
143  if (KVIDZAGrid::kICODE3 < quality && quality < KVIDZAGrid::kICODE7) {
144  // if the final quality code is kICODE4, kICODE5 or kICODE6 then this "nucleus"
145  // corresponds to a point which is inbetween the lines, i.e. noise
146  IDR->IDcode = kIDCode10;
147  return kTRUE;
148  }
149 
150  // set general ID code Si-CsI
151  IDR->IDcode = kIDCode3;
152  return kTRUE;
153 
154 
155 }

178 void KVIDSiCsI_e475s::CalculateParticleEnergy(KVReconstructedNucleus* nuc)
179 {
180 
181  // The energy of each particle is calculated as follows:
182  //
183  // E = dE_1 + dE_2 + ... + dE_N
184  //
185  // dE_1, dE_2, ... = energy losses measured in each detector through which
186  // the particle has passed (or stopped, in the case of dE_N).
187  // These energy losses are corrected for (Z,A)-dependent effects
188  // such as pulse-heigth defect in silicon detectors, losses in
189  // windows of gas detectors, etc.
190  //
191  // Whenever possible, the energy loss for fired detectors which are uncalibrated
192  // or not functioning is calculated. In this case the status returned by GetCalibStatus()
193  // will be KVIDTelescope::kCalibStatus_Calculated.
194  // If none of the detectors is calibrated, the particle's energy cannot be calculated &
195  // the status will be KVIDTelescope::kCalibStatus_NoCalibrations.
196  // Otherwise, the status code will be KVIDTelescope::kCalibStatus_OK.
197 
198  //status code
199  fCalibStatus = kCalibStatus_NoCalibrations;
200 
201  if (nuc->GetZ() == 0) return;
202 
203  KVDetector* det_dE = GetDetector(1);
204  KVDetector* det_Eres = GetDetector(2);
205 
206  if (!det_dE->IsCalibrated()) return;
207 
208  Double_t dE = det_dE->GetEnergy();
209  Double_t Eres = det_dE->GetEResFromDeltaE(nuc->GetZ(), nuc->GetA(), dE);
210  det_Eres->SetEnergyLoss(Eres);
211 
212  Double_t Einc = det_dE->GetCorrectedEnergy(nuc, dE, kTRUE) + Eres;
213 
214  fCalibStatus = kCalibStatus_Calculated;
215 
216  //Now we have to work our way up the list of detectors from which the particle was
217  //reconstructed. For each fired & calibrated detector which is only associated with
218  //one particle in the events, we add the corrected measured energy loss
219  //to the particle. For uncalibrated, unfired detectors and detectors through which
220  //more than one particle has passed, we calculate the corrected energy loss and add it
221  //to the particle.
222  int ndets = nuc->GetNumDet();
223  if (ndets > (int)GetSize()) { //particle passed through other detectors before this idtelesocpe
224  //look at detectors not in this id telescope
225  int idet = GetSize();//next detector after delta-e member of IDTelescope (stopping detector = 0)
226  while (idet < ndets) {
227 
228  KVDetector* det = nuc->GetDetector(idet);
229  if (det->Fired() && det->IsCalibrated() && det->GetNHits() == 1) {
230  Einc += det->GetCorrectedEnergy(nuc);
231  }
232  idet++;
233  }
234  }
235  //einc is now the energy of the particle before crossing the first detector
236  nuc->SetEnergy(Einc);
237  //set angles from the dimensions of the telescope in which particle detected
238  nuc->GetAnglesFromStoppingDetector();
239 
240 }

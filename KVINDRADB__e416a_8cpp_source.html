84 void KVINDRADB_e416a::ReadBICPressures()
85 {
86  //Read BIC pressures for different run ranges and enter into database.
87  //Format of file is:
88  //
89  //# some comments
90  //#which start with '#'
91 // Run Range : 289 385
92 // BIC_1 30.0
93 // BIC_2 30.0
94 // BIC_3 30.0
95  //
96  //Pressures (of CF4) are given in Torr.
97 
98  ifstream fin;
99  if (!OpenCalibFile("BICPressures", fin)) {
100  Error("ReadBICPressures()", "Could not open file %s",
101  GetCalibFileName("BICPressures"));
102  return;
103  }
104  Info("ReadBICPressures()", "Reading BIC pressures parameters...");
105 
106  TString sline;
107  UInt_t frun = 0, lrun = 0;
108  UInt_t run_ranges[MAX_NUM_RUN_RANGES][2];
109  UInt_t rr_number = 0;
110  Bool_t prev_rr = kFALSE; //was the previous line a run range indication ?
111  Bool_t read_pressure = kFALSE; // have we read any pressures recently ?
112 
113  KVDB_BIC_Pressures* parset = 0;
114  TList* par_list = new TList();
115 
116  //any ChIo not in list is assumed absent (pressure = 0)
117  Float_t pressure[3] = { 0, 0, 0 };
118 
119  while (fin.good()) { // parcours du fichier
120 
121  sline.ReadLine(fin);
122  if (sline.BeginsWith("Run Range :")) { // run range found
123  if (!prev_rr) { // New set of run ranges to read
124 
125  //have we just finished reading some pressures ?
126  if (read_pressure) {
127  parset = new KVDB_BIC_Pressures(pressure);
128  GetTable("BIC Pressures")->AddRecord(parset);
129  par_list->Add(parset);
130  LinkListToRunRanges(par_list, rr_number, run_ranges);
131  par_list->Clear();
132  for (int zz = 0; zz < 3; zz++) pressure[zz] = 0.;
133  read_pressure = kFALSE;
134  }
135  rr_number = 0;
136 
137  }
138  if (sscanf(sline.Data(), "Run Range : %u %u", &frun, &lrun) != 2) {
139  Warning("ReadBICPressures()",
140  "Bad format in line :\n%s\nUnable to read run range values",
141  sline.Data());
142  } else {
143  prev_rr = kTRUE;
144  run_ranges[rr_number][0] = frun;
145  run_ranges[rr_number][1] = lrun;
146  rr_number++;
147  if (rr_number == MAX_NUM_RUN_RANGES) {
148  Error("ReadBICPressures", "Too many run ranges (>%d)",
149  rr_number);
150  rr_number--;
151  }
152  }
153  } // Run Range found
154  if (fin.eof()) { //fin du fichier
155  //have we just finished reading some pressures ?
156  if (read_pressure) {
157  parset = new KVDB_BIC_Pressures(pressure);
158  GetTable("BIC Pressures")->AddRecord(parset);
159  par_list->Add(parset);
160  LinkListToRunRanges(par_list, rr_number, run_ranges);
161  par_list->Clear();
162  for (int zz = 0; zz < 3; zz++) pressure[zz] = 0.;
163  read_pressure = kFALSE;
164  }
165  }
166  if (sline.BeginsWith("BIC")) { //line with BIC pressure data
167 
168  prev_rr = kFALSE;
169 
170  //split up BIC name and pressure
171  TObjArray* toks = sline.Tokenize(' ');
172  TString chio = ((TObjString*)(*toks)[0])->String();
173  KVString press = ((TObjString*)(*toks)[1])->String();
174  delete toks;
175 
176  read_pressure = kTRUE;
177 
178  if (chio == "BIC_1") pressure[0] = press.Atof();
179  else if (chio == "BIC_2") pressure[1] = press.Atof();
180  else if (chio == "BIC_3") pressure[2] = press.Atof();
181  else read_pressure = kFALSE;
182 
183  } //line with BIC pressure data
184  } //parcours du fichier
185  delete par_list;
186  fin.close();
187 }

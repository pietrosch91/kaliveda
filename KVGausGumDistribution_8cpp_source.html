52  : TF1(name, this, &KVGausGumDistribution::GDk, xmin, xmax, 5,
53  "KVGausGumDistribution", "GDk"), fGaussComp(0), fGumbelComp(0)
54 {
55  // normalised sum of Gaussian distribution and Gumbel distribution of k-th rank
56  // f = eta*Gauss(mean,sigma) + (1-eta)*Gumbel(a,b)
57  // free parameters: gauss: (mean,sigma) gumbel: (a,b) eta
58  fRank = k;
59  fkFac = TMath::Power(k, k) / TMath::Factorial(k - 1);

179 Double_t KVGausGumDistribution::GDk(Double_t* x, Double_t* p)
180 {
181  // Evaluate normalised sum of Gaussian and Gumbel distribution of rank fRank for x
182  // with parameters
183  // par[0] = eta
184  // par[1] = mu
185  // par[2] = sigma
186  // par[3] = distance between gaussian and gumbel >=0
187  // par[4] = b
188 
189  if (p[4] == 0) return 0;
190 
191  Double_t am = p[1] - p[3];
192  Double_t s = (*x - am) / p[4];
193  Double_t gum = (fkFac / p[4]);
194  Double_t es = -fRank * (s + TMath::Exp(-s));
195  gum *= TMath::Exp(es);
196 
197  Double_t gau = fkGaussNor / p[2];
198  s = (*x - p[1]) / p[2];
199  s = TMath::Power(s, 2.) / 2.;
200  gau *= TMath::Exp(-s);
201 
202  Double_t gg = p[0] * gau + (1. - p[0]) * gum;
203  return gg;
204 }

117 void KVFAZIAUpDater::SetCalibrations(KVDBRun* dbrun)
118 {
119  //Loop on calibrations stores in the database
120  //and update parameters for each concerned calibrators
121  KVFAZIADetector* det = 0;
122  KVCalibrator* cal = 0;
123  KVDBParameterSet* par = 0;
124  TList* list = (TList*)dbrun->GetLinks("Calibrations");
125  TIter next(list);
126  while ((par = (KVDBParameterSet*)next())) {
127  TString sdet(par->GetName());
128 // Info("SetCalibrations", "%s", sdet.Data());
129 // par->Print();
130  det = (KVFAZIADetector*)gFazia->GetDetector(sdet.Data());
131  if (det && (cal = det->GetCalibrator(par->GetTitle()))) {
132  if (det->GetIdentifier() == KVFAZIADetector::kCSI) {
133  for (int nn = 0; nn < cal->GetNumberParams(); nn++) cal->SetParameter(nn, par->GetParameter(nn));
134  } else {
135  Double_t* params = new Double_t[par->GetParamNumber()];
136  for (int y = 0; y < par->GetParamNumber(); ++y) params[y] = par->GetParameter(y);
137  ((KVFAZIACalibrator*)cal)->ChangeParameters(params);
138  delete[] params;
139  }
140  cal->SetStatus(1);
141  }
142  }

386 void KVItvFinderDialog::LinearizeHisto(int nbins)
387 {
388  Double_t zmin = ((KVIDentifier*)fGrid->GetIdentifiers()->First())->GetPID() - 1.0;
389  Double_t zmax = 0;
390 
391  for (int iz = 1; iz < fGrid->GetIdentifiers()->GetSize() + 1; iz++) {
392  KVIDentifier* tmp = (KVIDentifier*)fGrid->GetIdentifiers()->At(iz);
393  if (tmp && tmp->GetPID() > zmax) zmax = tmp->GetPID();
394  }
395 
396  Int_t zbins = (Int_t)(zmax - zmin) * nbins;
397 
398  fLinearHisto = new TH1F("fLinearHisto", "fLinearHisto", zbins, zmin, zmax);
399 
400  KVIdentificationResult* idr = new KVIdentificationResult;
401  fGrid->SetOnlyZId(1);
402 
403  for (int i = 1; i <= fHisto->GetNbinsX(); i++) {
404  for (int j = 1; j <= fHisto->GetNbinsY(); j++) {
405  Stat_t poids = fHisto->GetBinContent(i, j);
406  if (poids == 0) continue;
407 
408  Axis_t x0 = fHisto->GetXaxis()->GetBinCenter(i);
409  Axis_t y0 = fHisto->GetYaxis()->GetBinCenter(j);
410  Axis_t wx = fHisto->GetXaxis()->GetBinWidth(i);
411  Axis_t wy = fHisto->GetYaxis()->GetBinWidth(j);
412 
413  if (x0 < 4) continue;
414 
415  Double_t x, y;
416  Int_t kmax = (Int_t) TMath::Min(20., poids);
417  Double_t weight = (kmax == 20 ? poids / 20. : 1.);
418  for (int k = 0; k < kmax; k++) {
419  x = gRandom->Uniform(x0 - .5 * wx, x0 + .5 * wx);
420  y = gRandom->Uniform(y0 - .5 * wy, y0 + .5 * wy);
421  if (fGrid->IsIdentifiable(x, y)) {
422  if (!fGrid->GetIdentifier("MassID") || (fGrid->GetIdentifier("MassID") && fGrid->GetIdentifier("MassID")->IsInside(x, y))) {
423  fGrid->KVIDZAGrid::Identify(x, y, idr);
424  Float_t PID = idr->PID;
425  fLinearHisto->Fill(PID, weight);
426  }
427  }
428  }
429  }
430  }
431  delete idr;
432 }

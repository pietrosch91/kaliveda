26 // PRINCIPE
27 // KVCalorimetry fait la somme des Z (Zsum), A (Asum), Ek (Eksum) et Q (Qsum) des noyaux considérés (méthode Fill(KVNucleus* ))
28 // Ces ingrédients permettent ensuite de calculer l'énergie d'excitation en utilisant
29 // le bilan énergétique suivant:
30 // Exci + Qini = Eksum + Qsum -> Exci = Eksum + Qsum - Qini
31 // A noter : Si l'utilisateur définit un repère via KVVarGlob::SetFrame(const Char_t* ) les énergies cinétiques
32 // des noyaux sont prises dans ce référentiel ( KVNucleus::GetFrame("nom_du_referentiel)"->GetKE() )
33 //------------------
34 // Exemple d utilisation :
35 //------------------
36 //
37 //KVNucleus alpha(2,4,10); //definition des noyaux
38 //KVNucleus triton(1,3);
39 //KVNucleus azote(7,16,40);
40 //
41 //KVCalorimetry ca;
42 //
43 //ca.Fill(&alpha); //Remplissage de la variable

46 //

76 //
77 // Deux modes de remplissage :
78 //------------------------
79 // 1. Mode normal (par défaut) comme explicité plus haut
80 // 2. Mode distinguant les noyaux suivant leur charge, ie séparation entre particules et fragments
81 // Actif avec l appel de la méthode void UseChargeDiff(Int_t FragmentMinimumCharge,Double_t ParticleFactor)
82 // le paramètre FragmentMinimumCharge est le critère discriminant :
83 // - KVNucleus::GetZ()<FragmentMinimumCharge -> particules

85 // le paramètre ParticleFactor correspond au facteur lorsque les grandeurs Zsum, Asum, Eksum, Qsum
86 // sont calculées.
87 // Dans la méthode SumUp on a : Eksum = \Sigma Ek(Z>=[FragmentMinimumCharge]) + [ParticleFactor]*\Sigma Ek(Z<[FragmentMinimumCharge])
88 // et dans la liste des ingrédients, sont ajoutés les deux contributions particules et fragments et les paramètres choisis
89 // sont également enregistrés

115 //----------------------------
116 //
117 // Deux modes de calcul :
118 //------------------------
119 // 1. Mode normal (par défaut) comme explicité plus haut
120 // 2. Mode incluant les neutrons libres

124 // Asum/[LevelDensityParameter] * T*T + Qi - \Sigma Ek - [NeutronMeanEnergyFactor]*Mn*T - \Sigma Q = 0 (methode Calculate)

455 void KVCalorimetry::UseChargeDiff(Int_t FragmentMinimumCharge, Double_t ParticleFactor)
456 {
457  //Make a difference between particle with a charge (GetZ) greater (fragments)
458  //or smaller (particles) than FragmentMinimumCharge.
459  //
460  //When sum on charge (Zsum), mass (Asum), energy (Eksum), are performed, two partial sums are done,
461  //respect to the previous distinction and particle ones will be multiply by ParticleFactor
462  //
463  //for example, for the kinetic energy, it gives at the end :
464  //Eksum = \Sigma Ek(Z>=[FragmentMinimumCharge]) + [ParticleFactor]*\Sigma Ek(Z<[FragmentMinimumCharge])
465  //this operation is done in the SumUp() method
466  //
467  //NOTE : when this method is called, Reset of the object are called also
468  //it has to be called before the first Fill
469  SetFragmentMinimumCharge(FragmentMinimumCharge);
470  SetParticleFactor(ParticleFactor);
471  kchargediff = kTRUE;
472  kIsModified = kTRUE;
473  Reset();
474 
475 }

561 void KVCalorimetry::Fill(KVNucleus* n)
562 {
563  // Remplissage des energies, masse, charge et defaut de masse
564  // Pour l'énergie cinétique, si l'utilisateur a utilisé en amont
565  // la méthode KVVarGlob::SetFrame(const Char_t*), c'est dans ce repère que les énergies sont sommées
566  // (à condition que chaque KVNucleus possede le repere avec un nom identique)
567  //
568  // Deux modes de remplissages :
569  //----------------------------
570  // - mode par défault, somme simple sur les A, Z, Ek, Q sans distinction du type de particules
571  //
572  // - mode avec distinction particules / fragments, actif si la méthode
573  // UseChargeDiff(Int_t FragmentMinimumCharge,Double_t ParticleFactor) a été appelée :
574  // ->Uune distinction entre produits avec une
575  // charge strictement inférieur à FragmentMinimumCharge (particules) et supérieur ou égale (fragments)
576  // est appliquée
577 
578  kIsModified = kTRUE;
579 
580  if (kchargediff) {
581 
582  if (n->GetZ() >= GetParValue("FragmentMinimumCharge")) {
583  AddIngValue("Zfrag", n->GetZ());
584  AddIngValue("Afrag", n->GetA());
585  AddIngValue("Ekfrag", n->GetFrame(fFrame.Data(), kFALSE)->GetKE());
586  AddIngValue("Qfrag", n->GetMassExcess());
587  AddIngValue("Mfrag", 1);
588  } else {
589  AddIngValue("Zpart", n->GetZ());
590  AddIngValue("Apart", n->GetA());
591  AddIngValue("Ekpart", n->GetFrame(fFrame.Data(), kFALSE)->GetKE());
592  AddIngValue("Qpart", n->GetMassExcess());
593  AddIngValue("Mpart", 1);
594  }
595 
596  return;
597 
598  }
599  KVCaloBase::Fill(n);
600

636 void KVCalorimetry::SumUp()
637 {
638  // protected method
639  // Appelé par Calculate pour mettre à jour les différents ingrédients
640  // de la calorimétrie :
641  //
642  // Trois modes de sommes:
643  //------------------
644  // - mode normal (par defaut)
645  // détermination de l excès de masse de la source recontruite, dernier ingrédient de l'équation :
646  // Exci + Qini = \Sigma Ek + \Sigma Q -> Exci = \Sigma Ek + \Sigma Q - Qini
647  //
648  // - mode avec distinction particules / fragments, actif si la méthode
649  // UseChargeDiff(Int_t FragmentMinimumCharge,Double_t ParticleFactor) a été appelée :
650  // -> une distinction entre produits avec une charge strictement inférieur à FragmentMinimumCharge (particules)
651  // et supérieur ou égale (fragments) est appliquée
652  // Ainsi dans la méthode SumUp() pour les énergies cinétiques, par exemple
653  // l'énergie cinétique de la source reconstruite sera
654  // Eksum = Ekfrag(Z>=[FragmentMinimumCharge]) + [ParticleFactor]*Ekpart(Z<[FragmentMinimumCharge])
655  // Détermination ensuite de l excès de masse de la source
656  //
657  // - mode avec prise en compte des neutrons libres, actif si la métode
658  // IncludeFreeNeutrons(Double_t AsurZ,Double_t NeutronMeanEnergyFactor,Double_t LevelDensityParameter)
659  // L'estimation du nombre neutrons, est fait en utilisant un AsurZ (paramètre de la calorimétrie)
660  // supposé de la source reconstruite :
661  // le nombre de neutrons libres est alors égal :
662  // Mn = [AsurZ]*Zsum - Asum
663  // Pour un Zsou reconstruit, on rajoute des neutrons pour que le Asou corresponde à un AsurZ prédéfini
664  // On en déduit ensuite l'exces de masse asscoié à ces neutrons
665  // Détermination ensuite de l excès de masse de la source
666 
667  // Les proprietes de la source sont calculees
668 
669  if (kchargediff) {
670  // somme des contributions fragments et particules
671  AddIngValue("Zsum", GetIngValue("Zfrag") + GetParValue("ParticleFactor")*GetIngValue("Zpart"));
672  AddIngValue("Asum", GetIngValue("Afrag") + GetParValue("ParticleFactor")*GetIngValue("Apart"));
673  AddIngValue("Eksum", GetIngValue("Ekfrag") + GetParValue("ParticleFactor")*GetIngValue("Ekpart"));
674  AddIngValue("Qsum", GetIngValue("Qfrag") + GetParValue("ParticleFactor")*GetIngValue("Qpart"));
675  AddIngValue("Msum", GetIngValue("Mfrag") + GetParValue("ParticleFactor")*GetIngValue("Mpart"));
676  }
677 
678  //printf("Eksum=%lf avant neutrons \n",GetIngValue("Eksum"));
679 
680  if (kfree_neutrons_included) {
681  // conservation du AsurZ du systeme --> multiplicite moyenne des neutrons
682  Double_t Mneutron = Double_t(TMath::Nint(GetParValue("AsurZ") * GetIngValue("Zsum") - GetIngValue("Asum")));
683  if (Mneutron < 0) {
684  //Warning("SumUp","Nombre de neutrons déduits négatif : %1.0lf -> on le met à zéro",Mneutron);
685  SetIngValue("Aexcess", TMath::Abs(Mneutron));
686  Mneutron = 0;
687  }
688  SetIngValue("Aneu", Mneutron);
689  SetIngValue("Qneu", Mneutron * nn.GetMassExcess(0, 1));
690  SetIngValue("Mneu", Mneutron);
691 
692  // prise en compte des neutrons dans la source
693  AddIngValue("Asum", GetIngValue("Mneu"));
694  AddIngValue("Qsum", GetIngValue("Qneu"));
695  AddIngValue("Msum", GetIngValue("Mneu"));
696 
697  }
698  //printf("Eksum=%lf apres neutrons \n",GetIngValue("Eksum"));
699  // defaut de masse de la source reconstruite
700  KVCaloBase::SumUp();
701 
702 }

732 Bool_t KVCalorimetry::Calculate(void)
733 {
734  //Réalisation de la calorimétrie
735  //Calcul de l'énergie d'excitation, température (optionnel), de l'énergie moyenne des neutrons (optionnel)
736  //appel de SumUp()
737  //Cette méthore retourne kTRUE si tout s'est bien passée, kFALSE si il y a un problème dans la résolution
738  //du polynome d'ordre 2
739  //
740  // Deux modes de calcul:
741  //------------------
742  // - mode normal (par defaut)
743  // Résolution de l'équation
744  // Exci + Qini = \Sigma Ek + \Sigma Q
745  // -> Exci = \Sigma Ek + \Sigma Q - Qini
746  //
747  // Optionnel :
748  // le calcul de la température peut être également fait si la méthode DeduceTemperature(Double_t LevelDensityParameter) a été appelée
749  // elle est obtenue via la formule : Exci = Asum/[LevelDensityParameter] * T*T
750  //
751  // - mode avec prise en compte des neutrons libres, actif si la métode
752  // IncludeFreeNeutrons(Double_t AsurZ,Double_t NeutronMeanEnergyFactor,Double_t LevelDensityParameter)
753  // Résolution de l'équation (polynome deuxième degrée en T (température) )
754  // Asum/[LevelDensityParameter] * T*T + Qi - \Sigma Ek - [NeutronMeanEnergyFactor]*Mn*T - \Sigma Q = 0
755  // on y obtient directement la température
756  //
757 
758  //Info("Calculate","Debut");
759 
760  if (!kIsModified) return kTRUE;
761  kIsModified = kFALSE;
762  // premier calcul depuis le dernier remplissage par Fill
763  SumUp();
764 
765  if (kfree_neutrons_included) {
766 
767  Double_t coefA = GetIngValue("Asum") / GetParValue("LevelDensityParameter");
768  Double_t coefB = -1.*GetParValue("NeutronMeanEnergyFactor") * GetIngValue("Mneu");
769  Double_t coefC = GetIngValue("Qini") - GetIngValue("Qsum") - GetIngValue("Eksum");
770 
771  // Resolution du polynome de degre 2
772  // Les champs ne sont remplis que si une solution reelle est trouvee
773  if (RootSquare(coefA, coefB, coefC)) {
774  // la solution max donne la temperature
775  SetIngValue("Temp", kracine_max);
776  SetIngValue("Exci", coefA * TMath::Power(GetIngValue("Temp"), 2.));
777 
778  // ajout de l'energie des neutrons a l energie totale de la source
779  SetIngValue("Ekneu", GetParValue("NeutronMeanEnergyFactor") * GetIngValue("Mneu")*GetIngValue("Temp"));
780  AddIngValue("Eksum", GetIngValue("Ekneu"));
781 
782  //parametre additionnel
783  //SetIngValue("Tmin",kracine_min); // la deuxieme solution de l'eq en T2
784  } else {
785  return kFALSE;
786  }
787 
788  } else {
789 
790  ComputeExcitationEnergy();
791  if (ktempdeduced) {
792  ComputeTemperature();
793  }
794 
795  }
796  return kTRUE;
797 }

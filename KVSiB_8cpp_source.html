181 Double_t KVSiB::GetEnergy()
182 {
183  //Redefinition of KVSilicon::GetEnergy().
184  //If energy lost in active layer is already set (e.g. by calculation of energy loss
185  //of charged particles), return its value.
186  //If not, we calculate it and set it using the values read from coders
187  //and the linear channel-energy calibration if initialised
188  //
189  //Returns -99 if (i) no calibration present (ii) calibration present but no data in acquisition parameters
190 
191  //fELoss already set, return its value
192  if (KVDetector::GetEnergy()) {
193  return KVDetector::GetEnergy();
194  }
195  Double_t ELoss = -99;
196  //is detector calibration available and ready ?
197  if (IsCalibrated()) {
198  ELoss = fLinCal->Compute(-1);
199  if (ELoss > -99.) {
200  SetEnergy(ELoss);
201  }
202  }
203  return ELoss;
204 }

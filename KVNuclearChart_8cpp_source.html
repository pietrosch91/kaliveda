271 void KVNuclearChart::ShowNucleusInfo(KVNucleus* nuc)
272 {
273  if ((fShownNucleus) && (fShownNucleus == nuc)) {
274  fSymbol->Clear();
275  fInfo->Clear();
276  fPad->GetListOfPrimitives()->Remove(fSymbol);

294  Int_t ymax = ax->FindBin(nuc->GetZ() + 0.5);
295  Int_t M = fHisto->Integral(xmin, xmax, ymin, ymax);
296  fInfo->AddText(Form("M = %d", M));
297  }
298  fPad->GetListOfPrimitives()->Remove(fInfo);
299  fInfo->Draw("same");
300

303 // symbText += Form("_{%d}", nuc->GetN());
304

307  fSymbol->Clear();

421 void KVNuclearChart::ShowMagicNumbers()

439  label->SetTextSize(0.025);

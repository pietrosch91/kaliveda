193 void KVFoxH2::Fill2(KVNucleus* n1, KVNucleus* n2)
194 {
195  // Calculate contribution to H(2) for the given pair of nuclei, if both of them
196  // have Z > Zmin (default Zmin=1; use SetParameter("Zmin", val) to change).
197  // By default, momenta in "CM" frame are used. (change with SetFrame()).
198  // NOTE: if one of the nuclei has zero momentum (strange, but could happen)
199  // it is excluded from the sum.
200 
201  Double_t p1, p2, cos_th_rel;
202 
203  // check Z of both particles
204  Int_t zmin = (Int_t)GetParameter("Zmin");
205  if (n1->GetZ() < zmin) return;
206  if (n2->GetZ() < zmin) return;
207 
208  if (n1 == n2) {
209  p1 = p2 = n1->GetFrame(fFrame.Data(), kFALSE)->GetMomentum().Mag();
210  cos_th_rel = 1.;
211  } else {
212  p1 = n1->GetFrame(fFrame.Data(), kFALSE)->GetMomentum().Mag();
213  p2 = n2->GetFrame(fFrame.Data(), kFALSE)->GetMomentum().Mag();
214  if ((p1 * p2) > 0.0) cos_th_rel = n1->GetFrame(fFrame.Data(), kFALSE)->GetMomentum().Dot(n2->GetFrame(fFrame.Data(), kFALSE)->GetMomentum()) / (p1 * p2);
215  else cos_th_rel = 1.;
216  }
217  if ((p1 * p2) > 0.) {
218  num += p1 * p2 * (3.*cos_th_rel * cos_th_rel - 1.);
219  den += 2.*p1 * p2;
220  }
221 }

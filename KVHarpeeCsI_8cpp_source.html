291 Bool_t KVHarpeeCsI::PositionIsOK()
292 {
293  // Returns true if all the conditions to access to the particle position
294  // are verified. In this case the position is given by the method
295  // GetPosition(...).
296  // The conditions are:
297  // -the multiplicity of fired ( "Pany" option of Fired() ) Harpee CsI
298  // detectors must be equal to one;
299  // -this detectector must be the fired detector.
300 
301  if (!TestBit(kPosIsOK)) {
302  Int_t mult = 0;
303  TIter next(fHarpeeCsIList);
304  KVHarpeeCsI* csi = NULL;
305  while ((csi = (KVHarpeeCsI*)next())) {
306  if (csi->Fired("Pany")) {
307  mult++;
308  fCsIForPosition = csi;
309  }
310  }
311  if (mult != 1) fCsIForPosition = NULL;
312  SetBit(kPosIsOK);
313  }
314  return fCsIForPosition == this;
315 }

355 Double_t KVHarpeeCsI::GetCorrectedEnergy(KVNucleus* nuc, Double_t light,
356  Bool_t transmission)
357 {
358  IGNORE_UNUSED(transmission);
359 
360  // Ensure the nucleus is not modified
361  const KVNucleus* const n(nuc);
362  assert(n);
363 
364  Int_t Z(n->GetZ());
365  Int_t A(n->GetA());
366 
367  assert((Z > 0) && (Z < 150));
368  assert((A > 0) && (A < 300));
369 
370  Double_t eloss(0.);
371  assert(fCal);
372 
373  fCal->SetZ(Z);
374  fCal->SetA(A);
375 
376  if (IsSimMode()) {
377  // Calculate energy loss from the predicted light output of the detected
378  // energy.
379  eloss = fCal->Compute(fCal->Invert(GetEnergy()));
380  } else if (light < 0.) {
381  // Light has not been set so use the ACQ data
382  eloss = fCal->Compute(GetACQData(GetEBaseName()) -
383  GetPedestal(GetEBaseName()));
384  } else {
385  // Light has been set manually
386  eloss = fCal->Compute(light);
387  }
388 
389  if (eloss < 0.) eloss = 0.;
390  SetEnergy(eloss);
391 
392  return eloss;
393 }

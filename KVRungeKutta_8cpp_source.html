167 void KVRungeKutta::Integrate(Double_t* ystart, Double_t x1, Double_t x2, Double_t h1)
168 {
169  // Runge-Kutta driver with adaptive stepsize control.
170  // Integrate nvar starting values ystart[0,...,nvar-1] from x1 to x2 with accuracy eps.
171  // h1 should be set as a guessed first stepsize.
172  // after call, GetNGoodSteps() and GetNBadSteps() give the number of good
173  // and bad (but retried and fixed) steps taken,
174  // and ystart values are replaced by values at the end of the integration interval.
175 
176  x = x1;
177  Double_t h = TMath::Sign(h1, x2 - x1);
178  nok = nbad = 0;
179  fOK = kTRUE;
180 
181  for (int i = 0; i < nvar; i++) y[i] = ystart[i];
182 
183  for (int nstp = 1; nstp <= MAXSTP; nstp++) { //Take at most MAXSTP steps.
184 
185  // calculate derivatives before performing step
186  fInitialDeriv = kTRUE;
187  CalcDerivs(x, y, dydx);
188  fInitialDeriv = kFALSE;
189 
190  for (Int_t i = 0; i < nvar; i++)
191  yscal[i] = TMath::Abs(y[i]) + TMath::Abs(dydx[i] * h) + TINY;
192 
193  if ((x + h - x2) * (x + h - x1) > 0.0) h = x2 - x;
194 
195  rkqs(h);
196  if (!fOK) {
197  Error("Integrate", "integration stopped at x=%g", x);
198  return;
199  }
200 
201  if (hdid == h) ++nok;
202  else ++nbad;
203  if ((x - x2) * (x2 - x1) >= 0.0) {
204  for (int i = 0; i < nvar; i++) ystart[i] = y[i];
205  return;
206  }
207 
208  if (TMath::Abs(hnext) <= hmin) {
209  Error("Integrate", "Step size %g too small", hnext);
210  Error("Integrate", "integration stopped at x=%g", x);
211  return;
212  }
213  h = hnext;
214  }
215  Error("Integrate", "Too many steps");
216  Error("Integrate", "integration stopped at x=%g", x);
217 }

230 void KVRungeKutta::rkqs(Double_t htry)
231 {
232  // Fifth-order Runge-Kutta step with monitoring of local truncation error to ensure accuracy and
233  // adjust stepsize. Input are the dependent variable vector y[1..n] and its derivative dydx[1..n]
234  // at the starting value of the independent variable x. Also input are the stepsize to be attempted
235  // htry, the required accuracy eps, and the vector yscal[1..n] against which the error is
236  // scaled. On output, y and x are replaced by their new values, hdid is the stepsize that was
237  // actually accomplished, and hnext is the estimated next stepsize. derivs is the user-supplied
238  // routine that computes the right-hand side derivatives.
239 
240  Double_t errmax, htemp, xnew;
241 
242  Double_t h = htry;
243  for (;;) {
244  rkck(h);
245  errmax = 0.0;
246  for (int i = 0; i < nvar; i++) errmax = TMath::Max(errmax, TMath::Abs(yerr[i] / yscal[i]));
247  errmax /= eps;
248  if (errmax <= 1.0) break;
249  htemp = SAFETY * h * TMath::Power(errmax, PSHRNK);
250  h = (h >= 0.0 ? TMath::Max(htemp, 0.1 * h) : TMath::Min(htemp, 0.1 * h));
251  xnew = x + h;
252  if (xnew == x) {
253  Error("rkqs", "stepsize underflow");
254  fOK = kFALSE;
255  return;
256  }
257  }
258  if (errmax > ERRCON) hnext = SAFETY * h * TMath::Power(errmax, PGROW);
259  else hnext = 5.0 * h;
260  x += (hdid = h);
261  for (int i = 0; i < nvar; i++) y[i] = yout[i];
262 }

273 void KVRungeKutta::rkck(Double_t h)
274 {
275  // Given values for n variables y[1..n] and their derivatives dydx[1..n] known at x, use
276  // the fifth-order Cash-Karp Runge-Kutta method to advance the solution over an interval h
277  // and return the incremented variables as yout[1..n]. Also return an estimate of the local
278  // truncation error in yout using the embedded fourth-order method. The user supplies the routine
279  // derivs(x,y,dydx) , which returns derivatives dydx at x.
280 
281  for (int i = 0; i < nvar; i++)
282  ytemp[i] = y[i] + b21 * h * dydx[i];
283  CalcDerivs(x + a2 * h, ytemp, ak2);
284  for (int i = 0; i < nvar; i++)
285  ytemp[i] = y[i] + h * (b31 * dydx[i] + b32 * ak2[i]);
286  CalcDerivs(x + a3 * h, ytemp, ak3);
287  for (int i = 0; i < nvar; i++)
288  ytemp[i] = y[i] + h * (b41 * dydx[i] + b42 * ak2[i] + b43 * ak3[i]);
289  CalcDerivs(x + a4 * h, ytemp, ak4);
290  for (int i = 0; i < nvar; i++)
291  ytemp[i] = y[i] + h * (b51 * dydx[i] + b52 * ak2[i] + b53 * ak3[i] + b54 * ak4[i]);
292  CalcDerivs(x + a5 * h, ytemp, ak5);
293  for (int i = 0; i < nvar; i++)
294  ytemp[i] = y[i] + h * (b61 * dydx[i] + b62 * ak2[i] + b63 * ak3[i] + b64 * ak4[i] + b65 * ak5[i]);
295  CalcDerivs(x + a6 * h, ytemp, ak6);
296  for (int i = 0; i < nvar; i++)
297  yout[i] = y[i] + h * (c1 * dydx[i] + c3 * ak3[i] + c4 * ak4[i] + c6 * ak6[i]);
298  for (int i = 0; i < nvar; i++)
299  yerr[i] = h * (dc1 * dydx[i] + dc3 * ak3[i] + dc4 * ak4[i] + dc5 * ak5[i] + dc6 * ak6[i]);
300 }
